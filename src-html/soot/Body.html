062/**
063 *   Abstract base class that models the body (code attribute) of a Java method.
064 *   Classes that implement an Intermediate Representation for a method body should subclass it.
065 *   In particular the classes GrimpBody, JimpleBody and BafBody all extend this
066 *   class. This class provides methods that are common to any IR, such as methods
067 *   to get the body's units (statements), traps, and locals.
068 *
069 *  @see soot.grimp.GrimpBody
070 *  @see soot.jimple.JimpleBody
071 *  @see soot.baf.BafBody
072 */
073@SuppressWarnings("serial")
074public abstract class Body extends AbstractHost implements Serializable
075{
076    /** The method associated with this Body. */
077    protected transient SootMethod method = null;
078
079    /** The chain of locals for this Body. */
080    protected Chain<Local> localChain = new HashChain<Local>();
081
082    /** The chain of traps for this Body. */
083    protected Chain<Trap> trapChain = new HashChain<Trap>();
084
085    /** The chain of units for this Body. */
086    protected PatchingChain<Unit> unitChain = new PatchingChain<Unit>(new HashChain<Unit>());
087

093        /**
094         * Returns an array containing some validators in order to validate the JimpleBody

097        private synchronized static BodyValidator[] getValidators() {
098                if (validators == null)
099                {
100                        validators = new BodyValidator[] {
101                                LocalsValidator.v(),
102                                TrapsValidator.v(),
103                                UnitBoxesValidator.v(),
104                                UsesValidator.v(),
105                                ValueBoxesValidator.v(),
106                                //CheckInitValidator.v(),
107                                CheckTypesValidator.v(),
108                                CheckVoidLocalesValidator.v(),
109                                CheckEscapingValidator.v()
110                        };
111                }
112                return validators;
113        };

156    /** Copies the contents of the given Body into this one. */
157    public Map<Object, Object> importBodyContentsFrom(Body b)
158    {
159        HashMap<Object, Object> bindings = new HashMap<Object, Object>();
160
161        {
162                // Clone units in body's statement list
163                for (Unit original : b.getUnits()) {
164                    Unit copy = (Unit) original.clone();
165        
166                    copy.addAllTagsOf(original);
167        
168                    // Add cloned unit to our unitChain.
169                    unitChain.addLast(copy);
170        
171                    // Build old <-> new map to be able to patch up references to other units
172                    // within the cloned units. (these are still refering to the original
173                    // unit objects).
174                    bindings.put(original, copy);
175                }
176        }
177
178        {
179                // Clone trap units.
180                for (Trap original : b.getTraps()) {
181                    Trap copy = (Trap) original.clone();
182        
183                    // Add cloned unit to our trap list.
184                    trapChain.addLast(copy);
185        
186                    // Store old <-> new mapping.
187                    bindings.put(original, copy);
188                }
189        }
190
191        {
192                // Clone local units.
193                for (Local original : b.getLocals()) {
194                    Local copy = (Local) original.clone();
195        
196                    // Add cloned unit to our trap list.
197                    localChain.addLast(copy);
198        
199                    // Build old <-> new mapping.
200                    bindings.put(original, copy);
201                }
202        }
203
204        {
205                // Patch up references within units using our (old <-> new) map.
206                for (UnitBox box : getAllUnitBoxes()) {
207                    Unit newObject, oldObject = box.getUnit();
208        
209                    // if we have a reference to an old object, replace it
210                    // it's clone.
211                    if( (newObject = (Unit)  bindings.get(oldObject)) != null )
212                        box.setUnit(newObject);
213        
214                }
215        }
216
217
218        {
219                // backpatching all local variables.
220                for (ValueBox vb : getUseBoxes()) {
221                    if(vb.getValue() instanceof Local)
222                        vb.setValue((Value) bindings.get(vb.getValue()));
223                }
224                for (ValueBox vb : getDefBoxes()) {
225                    if(vb.getValue() instanceof Local)
226                        vb.setValue((Value) bindings.get(vb.getValue()));
227                }
228        }
229        return bindings;
230    }

232    protected void runValidation(BodyValidator validator) {
233        final List<ValidationException> exceptionList = new ArrayList<ValidationException>();
234        validator.validate(this, exceptionList);
235        if (!exceptionList.isEmpty())
236                throw exceptionList.get(0);
237    }

240    /** Verifies a few sanity conditions on the contents on this body. */
241    public void validate()
242    {
243        List<ValidationException> exceptionList = new ArrayList<ValidationException>();
244        validate(exceptionList);
245        if (!exceptionList.isEmpty())
246                throw exceptionList.get(0);
247    }

249    /**
250     * Validates the jimple body and saves a list of all validation errors 
251     * @param exceptionList the list of validation errors
252     */
253    public void validate(List<ValidationException> exceptionList) {
254        final boolean runAllValidators = Options.v().debug() || Options.v().validate();
255        for (BodyValidator validator : getValidators()) {
256                if (!validator.isBasicValidator() && !runAllValidators)
257                        continue;
258                validator.validate(this, exceptionList);
259        }
260    }

299    /** Return LHS of the first identity stmt assigning from \@this. **/
300    public Local getThisLocal()
301    {
302        for (Unit s : getUnits())
303        {
304            if (s instanceof IdentityStmt &&
305                ((IdentityStmt)s).getRightOp() instanceof ThisRef)
306                return (Local)(((IdentityStmt)s).getLeftOp());
307        }
308
309        throw new RuntimeException("couldn't find identityref!"+" in "+getMethod());
310    }
311
312    /** Return LHS of the first identity stmt assigning from \@parameter i. **/
313    public Local getParameterLocal(int i)
314    {
315        for (Unit s : getUnits())
316        {
317            if (s instanceof IdentityStmt &&
318                ((IdentityStmt)s).getRightOp() instanceof ParameterRef)
319            {
320                IdentityStmt is = (IdentityStmt)s;
321                ParameterRef pr = (ParameterRef)is.getRightOp();
322                if (pr.getIndex() == i)
323                    return (Local)is.getLeftOp();
324            }
325        }
326
327        throw new RuntimeException("couldn't find parameterref" + i +"! in "+getMethod());
328    }
329
330    /**
331     * Get all the LHS of the identity statements assigning from parameter references.
332     *
333     * @return a list of size as per <code>getMethod().getParameterCount()</code> with all elements ordered as per the parameter index.
334     * @throws RuntimeException if a parameterref is missing
335     */
336    public List<Local> getParameterLocals(){
337        final int numParams = getMethod().getParameterCount();
338        final List<Local> retVal = new ArrayList<Local>(numParams);
339
340        //Parameters are zero-indexed, so the keeping of the index is safe
341        for (Unit u : getUnits()){
342            if (u instanceof IdentityStmt){
343                IdentityStmt is = ((IdentityStmt)u);
344                if (is.getRightOp() instanceof ParameterRef){
345                    ParameterRef pr = (ParameterRef) is.getRightOp();
346                    retVal.add(pr.getIndex(), (Local) is.getLeftOp());
347                }
348            }
349        }
350        if (retVal.size() != numParams)
351            throw new RuntimeException("couldn't find parameterref! in " + getMethod());
352        return retVal;
353    }

355    /**
356     * Returns the list of parameter references used in this body. The list is as long as
357     * the number of parameters declared in the associated method's signature.
358     * The list may have <code>null</code> entries for parameters not referenced in the body.
359     * The returned list is of fixed size.
360     */
361    public List<Value> getParameterRefs()
362    {
363        Value[] res = new Value[getMethod().getParameterCount()];
364        for (Unit s : getUnits()) {
365            if (s instanceof IdentityStmt) {
366                                Value rightOp = ((IdentityStmt)s).getRightOp();
367                                if (rightOp instanceof ParameterRef) {
368                                        ParameterRef parameterRef = (ParameterRef) rightOp;
369                                        res[parameterRef.getIndex()] = parameterRef;
370                                }
371                        }
372        }
373        return Arrays.asList(res);
374    }

394    /**
395     * Returns the result of iterating through all Units in this body
396     * and querying them for their UnitBoxes.  All UnitBoxes thus
397     * found are returned.  Branching Units and statements which use
398     * PhiExpr will have UnitBoxes; a UnitBox contains a Unit that is
399     * either a target of a branch or is being used as a pointer to
400     * the end of a CFG block.
401     *
402     * <p> This method is typically used for pointer patching, eg when
403     * the unit chain is cloned.
404     *
405     * @return A list of all the UnitBoxes held by this body's units.
406     * @see UnitBox
407     * @see #getUnitBoxes(boolean)
408     * @see Unit#getUnitBoxes()
409     * @see soot.shimple.PhiExpr#getUnitBoxes()
410     **/
411    public List<UnitBox> getAllUnitBoxes()
412    {
413        ArrayList<UnitBox> unitBoxList = new ArrayList<UnitBox>();
414        {
415            Iterator<Unit> it = unitChain.iterator();
416            while(it.hasNext()) {
417                Unit item = it.next();
418                unitBoxList.addAll(item.getUnitBoxes());
419            }
420        }
421
422        {
423            Iterator<Trap> it = trapChain.iterator();
424            while(it.hasNext()) {
425                Trap item = it.next();
426                unitBoxList.addAll(item.getUnitBoxes());
427            }
428        }
429
430        {
431            Iterator<Tag> it = getTags().iterator();
432            while(it.hasNext()) {
433                Tag t = it.next();
434                if( t instanceof CodeAttribute)
435                    unitBoxList.addAll(((CodeAttribute) t).getUnitBoxes());
436            }
437        }
438
439        return unitBoxList;
440    }
441
442    /**
443     * If branchTarget is true, returns the result of iterating
444     * through all branching Units in this body and querying them for
445     * their UnitBoxes. These UnitBoxes contain Units that are the
446     * target of a branch.  This is useful for, say, labeling blocks
447     * or updating the targets of branching statements.
448     *
449     * <p> If branchTarget is false, returns the result of iterating
450     * through the non-branching Units in this body and querying them
451     * for their UnitBoxes.  Any such UnitBoxes (typically from
452     * PhiExpr) contain a Unit that indicates the end of a CFG block.
453     *
454     * @return a list of all the UnitBoxes held by this body's
455     * branching units.
456     *
457     * @see UnitBox
458     * @see #getAllUnitBoxes()
459     * @see Unit#getUnitBoxes()
460     * @see soot.shimple.PhiExpr#getUnitBoxes()
461     **/
462    public List<UnitBox> getUnitBoxes(boolean branchTarget)
463    {
464        ArrayList<UnitBox> unitBoxList = new ArrayList<UnitBox>();
465        {
466            Iterator<Unit> it = unitChain.iterator();
467            while(it.hasNext()) {
468                Unit item = it.next();
469                if(branchTarget){
470                    if(item.branches())
471                        unitBoxList.addAll(item.getUnitBoxes());
472                }
473                else{
474                    if(!item.branches())
475                        unitBoxList.addAll(item.getUnitBoxes());
476                }
477            }
478        }
479
480        {
481            Iterator<Trap> it = trapChain.iterator();
482            while(it.hasNext()) {
483                Trap item = it.next();
484                unitBoxList.addAll(item.getUnitBoxes());
485            }
486        }
487
488        {
489            Iterator<Tag> it = getTags().iterator();
490            while(it.hasNext()) {
491                Tag t = it.next();
492                if( t instanceof CodeAttribute)
493                    unitBoxList.addAll(((CodeAttribute) t).getUnitBoxes());
494            }
495        }
496
497        return unitBoxList;
498    }

579    /**
580     * {@inheritDoc}
581     */
582    @Override
583    public String toString() {
584        ByteArrayOutputStream streamOut = new ByteArrayOutputStream();
585        PrintWriter writerOut = new PrintWriter(new EscapedWriter(new OutputStreamWriter(streamOut)));
586        try { 
587            Printer.v().printTo(this, writerOut);
588        } catch (RuntimeException e) {
589            e.printStackTrace(writerOut);
590        }
591        writerOut.flush();
592        writerOut.close();
593        return streamOut.toString();
594    }
