001package soot.rtlib.tamiflex;
002
003import java.lang.reflect.Constructor;
004import java.lang.reflect.Field;
005import java.lang.reflect.Method;
006
007public class UnexpectedReflectiveCall {
008        
009        private final static IUnexpectedReflectiveCallHandler handler;
010        
011        static {
012                String listenerClassName = System.getProperty("BOOSTER_LISTENER", "soot.rtlib.tamiflex.DefaultHandler");
013                try {
014                        handler = (IUnexpectedReflectiveCallHandler) Class.forName(listenerClassName).newInstance();
015                } catch (Exception e) {
016                        throw new Error("Error instantiating listener for Booster.",e);
017                }
018        }
019        
020        //delegate methods
021        
022        public static void classNewInstance(Class<?> c) {
023                handler.classNewInstance(c);
024        }
025
026        public static void classForName(String typeName) {
027                handler.classForName(typeName);
028        }
029
030        public static void constructorNewInstance(Constructor<?> c) {
031                handler.constructorNewInstance(c);
032        }
033
034        public static void methodInvoke(Object receiver, Method m) {
035                handler.methodInvoke(receiver, m);
036        }
037
038        public static void fieldSet(Object receiver, Field f) {
039                handler.fieldSet(receiver, f);
040        }
041
042        public static void fieldGet(Object receiver, Field f) {
043                handler.fieldGet(receiver, f);
044        }
045}
