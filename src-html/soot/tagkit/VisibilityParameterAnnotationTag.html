026package soot.tagkit;
027import java.util.*;
028
029/** Represents the visibility of an annotation attribute attatched 
030 * to a class, field, method or method param (only one of these each)
031 * has one or more annotations
032 * for Java 1.5.
033 */
034
035public class VisibilityParameterAnnotationTag implements  Tag
036{
037    
038    private int num_params;
039    private int kind;
040    private ArrayList<VisibilityAnnotationTag> visibilityAnnotations;
041    
042    public VisibilityParameterAnnotationTag(int num, int kind){
043        this.num_params = num;
044        this.kind = kind;
045    }
046    
047    // should also print here number of annotations and perhaps the annotations themselves
048    public String toString() {
049        StringBuffer sb = new StringBuffer("Visibility Param Annotation: num params: "+num_params+" kind: "+kind);
050        if (visibilityAnnotations != null){
051            for (VisibilityAnnotationTag tag : visibilityAnnotations) {
052                sb.append("\n");
053                if (tag != null)
054                        sb.append(tag.toString());
055            }
056        }
057        sb.append("\n");
058        return sb.toString();
059    }
060
061    /** Returns the tag name. */
062    public String getName() {
063        return "VisibilityParameterAnnotationTag";
064    }
065
066    public String getInfo(){
067        return "VisibilityParameterAnnotation";
068    }
069    
070    /** Returns the tag raw data. */
071    public byte[] getValue() {
072        throw new RuntimeException( "VisibilityParameterAnnotationTag has no value for bytecode" );
073    }
074
075    public void addVisibilityAnnotation(VisibilityAnnotationTag a){
076        if (visibilityAnnotations == null){
077            visibilityAnnotations = new ArrayList<VisibilityAnnotationTag>();
078        }
079        visibilityAnnotations.add(a);
080    }
081
082    public ArrayList<VisibilityAnnotationTag> getVisibilityAnnotations(){
083        return visibilityAnnotations;
084    }
085
086    public int getKind(){
087        return kind;
088    }
089}
