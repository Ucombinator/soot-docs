120        public boolean follows(E someObject, E someReferenceObject) {
121                Iterator<E> it = iterator(someObject);
122                while (it.hasNext()) {
123                        if (it.next() == someReferenceObject)
124                                return false;
125                }
126                return true;

142        public void insertAfter(E toInsert, E point) {
143                if (toInsert == null)
144                        throw new RuntimeException("Bad idea! You tried to insert "
145                                        + " a null object into a Chain!");
146
147                if (map.containsKey(toInsert))
148                        throw new RuntimeException("Chain already contains object.");
149
150                Link<E> temp = map.get(point);
151                if (temp == null) {
152                        throw new RuntimeException("Insertion point not found in chain!");
153                }
154
155                stateCount++;
156
157                Link<E> newLink = temp.insertAfter(toInsert);
158                map.put(toInsert, newLink);
159        }

161        public void insertAfter(Collection<? extends E> toInsert, E point) {
162                // if the list is null, treat it as an empty list
163                if (toInsert == null)
164                        throw new RuntimeException("Warning! You tried to insert "
165                                        + "a null list into a Chain!");
166
167                E previousPoint = point;
168                for (E o : toInsert) {
169                        insertAfter(o, previousPoint);
170                        previousPoint = o;
171                }
172        }

222         * that is defined by <code>point_source</code> and <code>point_target</code>.

231                o.add(toInsert);

236        /**
237         * Inserts instrumentation in a manner such that the resulting control flow
238         * graph (CFG) of the program will contain <code>toInsert</code> on an edge
239         * that is defined by <code>point_source</code> and <code>point_target</code>.
240         * 
241         * @param toInsert  instrumentation to be added in the Chain
242         * @param point_src the source point of an edge in CFG
243         * @param point_tgt the target point of an edge
244         */
245        public void insertOnEdge(Collection<? extends E> toInsert, E point_src, E point_tgt) {
246
247                if (toInsert == null)
248                        throw new RuntimeException("Bad idea! You tried to insert " + "a null object into a Chain!");
249
250                // Insert 'toInsert' before 'target' point in chain if the source point is null
251                if (point_src == null && point_tgt != null) {
252                        ((Unit) point_tgt).redirectJumpsToThisTo((Unit) toInsert.toArray()[0]);
253                        insertBefore(toInsert, point_tgt);
254                        return;
255                }
256
257                // Insert 'toInsert' after 'source' point in chain if the target point is null
258                if (point_src != null && point_tgt == null) {
259                        insertAfter(toInsert, point_src);
260                        return;
261                }
262
263                // Throw an exception if both source and target is null
264                if (point_src == null && point_tgt == null) {
265                        throw new RuntimeException("insertOnEdge failed! Both source and target points are null.");
266                }
267
268                // If target is right after the source in the Chain
269                // 1- Redirect all jumps (if any) from 'source' to 'target', to 'toInsert[0]'
270                //    (source->target) ==>  (source->toInsert[0])
271                // 2- Insert 'toInsert' after 'source' in Chain
272                if (getSuccOf(point_src) == point_tgt) {
273                        List<UnitBox> boxes = ((Unit) point_src).getUnitBoxes();
274                        for (UnitBox box : boxes) {
275                                if (box.getUnit() == point_tgt) {
276                                        box.setUnit((Unit) toInsert.toArray()[0]);
277                                }
278                        }
279                        insertAfter(toInsert, point_src);
280                        return;
281                }
282                
283                
284                // If the target is not right after the source in chain then,
285                // 1- Redirect all jumps (if any) from 'source' to 'target', to 'toInsert[0]'
286                //    (source->target) ==>  (source->toInsert[0])
287                //    1.1- if there are no jumps from source to target, then such an edge does not exist. Throw an exception.
288                // 2- Insert 'toInsert' before 'target' in Chain
289                // 3- If required, add a 'goto target' statement so that no other edge executes 'toInsert'
290                boolean validEdgeFound = false;
291                E originalPred = getPredOf(point_tgt);
292                
293                List<UnitBox> boxes = ((Unit) point_src).getUnitBoxes();
294                for (UnitBox box : boxes) {
295                        if (box.getUnit() == point_tgt) {
296
297                                if (point_src instanceof GotoStmt) {
298
299                                        box.setUnit((Unit) toInsert.toArray()[0]);
300                                        insertAfter(toInsert, point_src);
301
302                                        E goto_unit = (E) new JGotoStmt((Unit) point_tgt);
303                                        insertAfter(goto_unit, (E) toInsert.toArray()[toInsert.size() - 1]);
304                                        return;
305                                }
306
307                                box.setUnit((Unit) toInsert.toArray()[0]);
308
309                                validEdgeFound = true;
310                        }
311                }
312                if (validEdgeFound) {
313                        insertBefore(toInsert, point_tgt);
314
315                        if (originalPred != point_src) {
316                                if (originalPred instanceof GotoStmt)
317                                        return;
318
319                                E goto_unit = (E) new JGotoStmt((Unit) point_tgt);
320                                insertBefore(goto_unit, (E) toInsert.toArray()[0]);
321                        }
322                        return;
323                }
324
325                // In certain scenarios, the above code can add extra 'goto' units on a different edge
326                // So, an edge [src --> tgt] becomes [src -> goto tgt -> tgt].
327                // When this happens, the original edge [src -> tgt] ceases to exist.
328                // The following code handles such scenarios. 
329                if (getSuccOf(point_src) instanceof GotoStmt) {
330                        if (((Unit) getSuccOf(point_src)).getUnitBoxes().get(0).getUnit() == point_tgt) {
331
332                                ((Unit) getSuccOf(point_src)).redirectJumpsToThisTo((Unit) toInsert.toArray()[0]);
333                                insertBefore(toInsert, getSuccOf(point_src));
334
335                                return;
336                        }
337                }
338                
339                // If the control reaches this point, it means that an edge [stc -> tgt] as specified by user does not exist and is thus invalid
340                // Return an exception.
341                throw new RuntimeException(
342                                "insertOnEdge failed! No such edge found. The edge on which you want to insert an instrumentation is invalid.");
343        }

371        public static <T> HashChain<T> listToHashChain(List<T> list) {
372                HashChain<T> c = new HashChain<T>();
373                Iterator<T> it = list.iterator();
374                while (it.hasNext())
375                        c.addLast(it.next());
376                return c;
377        }

379        public boolean remove(Object item) {
380                if (item == null)
381                        throw new RuntimeException("Bad idea! You tried to remove "
382                                        + " a null object from a Chain!");
383
384                stateCount++;
385                /*
386                 * 4th April 2005 Nomair A Naeem map.get(obj) can return null only
387                 * return true if this is non null else return false
388                 */
389                if (map.get(item) != null) {
390                        Link<E> link = map.get(item);
391
392                        link.unlinkSelf();
393                        map.remove(item);
394                        return true;
395                }
396                return false;
397        }

399        public void addFirst(E item) {
400                if (item == null)
401                        throw new RuntimeException("Bad idea!  You tried to insert "
402                                        + "a null object into a Chain!");
403
404                stateCount++;
405                Link<E> newLink, temp;
406
407                if (map.containsKey(item))
408                        throw new RuntimeException("Chain already contains object.");
409
410                if (firstItem != null) {
411                        temp = map.get(firstItem);
412                        newLink = temp.insertBefore(item);
413                } else {
414                        newLink = new Link<E>(item);
415                        firstItem = lastItem = item;
416                }
417                map.put(item, newLink);
418        }
419
420        public void addLast(E item) {
421                if (item == null)
422                        throw new RuntimeException("Bad idea! You tried to insert "
423                                        + " a null object into a Chain!");
424
425                stateCount++;
426                Link<E> newLink, temp;
427                if (map.containsKey(item))
428                        throw new RuntimeException("Chain already contains object: " + item);
429
430                if (lastItem != null) {
431                        temp = map.get(lastItem);
432                        newLink = temp.insertAfter(item);
433                } else {
434                        newLink = new Link<E>(item);
435                        firstItem = lastItem = item;
436                }
437                map.put(item, newLink);
438        }

466        public E getSuccOf(E point) throws NoSuchElementException {
467                Link<E> link = map.get(point);
468                try {
469                        link = link.getNext();
470                } catch (NullPointerException e) {
471                        throw new NoSuchElementException();
472                }
473                if (link == null)
474                        return null;
475
476                return link.getItem();
477        }

479        public E getPredOf(E point) throws NoSuchElementException {
480                Link<E> link = map.get(point);
481                if (point == null)
482                        throw new RuntimeException("trying to hash null value.");
483
484                try {
485                        link = link.getPrevious();
486                } catch (NullPointerException e) {
487                        throw new NoSuchElementException();
488                }
489
490                if (link == null)
491                        return null;
492                else
493                        return link.getItem();
494        }

522        /**
523         * <p>
524         * Returns an iterator ranging from <code>head</code> to <code>tail</code>,
525         * inclusive.
526         * </p>
527         * 
528         * <p>
529         * If <code>tail</code> is the element immediately preceding
530         * <code>head</code> in this <code>HashChain</code>, the returned iterator
531         * will iterate 0 times (a special case to allow the specification of an
532         * empty range of elements). Otherwise if <code>tail</code> is not one of
533         * the elements following <code>head</code>, the returned iterator will
534         * iterate past the end of the <code>HashChain</code>, provoking a
535         * {@link NoSuchElementException}.
536         * </p>
537         * 
538         * @throws NoSuchElementException
539         *             if <code>head</code> is not an element of the chain.
540         */
541        public Iterator<E> iterator(E head, E tail) {
542                if (firstItem == null || isEmpty())
543                        return emptyIterator;
544                if (head != null && this.getPredOf(head) == tail) {
545                        return emptyIterator;
546                }
547                return new LinkIterator<E>(head, tail);
548        }

554        /** Returns a textual representation of the contents of this Chain. */
555        public String toString() {
556                StringBuilder strBuf = new StringBuilder();
557
558                Iterator<E> it = iterator();
559                boolean b = false;
560
561                strBuf.append("[");
562                while (it.hasNext()) {
563                        if (!b)
564                                b = true;
565                        else
566                                strBuf.append(", ");
567                        strBuf.append(it.next().toString());
568                }
569                strBuf.append("]");
570                return strBuf.toString();
571        }

573        @SuppressWarnings("serial")
574        class Link<X extends E> implements Serializable {
575                private Link<X> nextLink;
576                private Link<X> previousLink;
577                private X item;
578
579                public Link(X item) {
580                        this.item = item;
581                        nextLink = previousLink = null;
582                }
583
584                public Link<X> getNext() {
585                        return nextLink;
586                }
587
588                public Link<X> getPrevious() {
589                        return previousLink;
590                }
591
592                public void setNext(Link<X> link) {
593                        this.nextLink = link;
594                }
595
596                public void setPrevious(Link<X> link) {
597                        this.previousLink = link;
598                }
599
600                public void unlinkSelf() {
601                        bind(previousLink, nextLink);
602                }
603
604                public Link<X> insertAfter(X item) {
605                        Link<X> newLink = new Link<X>(item);
606
607                        bind(newLink, nextLink);
608                        bind(this, newLink);
609                        return newLink;
610                }
611
612                public Link<X> insertBefore(X item) {
613                        Link<X> newLink = new Link<X>(item);
614
615                        bind(previousLink, newLink);
616                        bind(newLink, this);
617                        return newLink;
618                }
619
620                private void bind(Link<X> a, Link<X> b) {
621                        if (a == null) {
622                                firstItem = (b == null) ? null : b.item;
623                        } else
624                                a.nextLink = b;
625
626                        if (b == null) {
627                                lastItem = (a == null) ? null : a.item;
628                        } else
629                                b.previousLink = a;
630                }
631
632                public X getItem() {
633                        return item;
634                }
635
636                public String toString() {
637                        if (item != null)
638                                return item.toString();
639                        else
640                                return "Link item is null" + super.toString();
641
642                }
643
644        }

646        class LinkIterator<X extends E> implements Iterator<E> {
647                private Link<E> currentLink;
648                boolean state; // only when this is true can remove() be called
649                // (in accordance w/ iterator semantics)
650
651                private X destination;
652                private long iteratorStateCount;
653
654                public LinkIterator(X item) {
655                        Link<E> nextLink = map.get(item);
656                        if (nextLink == null && item != null)
657                                throw new NoSuchElementException(
658                                                "HashChain.LinkIterator(obj) with obj that is not in the chain: "
659                                                                + item.toString());
660                        currentLink = new Link<E>(null);
661                        currentLink.setNext(nextLink);
662                        state = false;
663                        destination = null;
664                        iteratorStateCount = stateCount;
665                }
666
667                public LinkIterator(X from, X to) {
668                        this(from);
669                        destination = to;
670                }
671
672                public boolean hasNext() {
673                        if (stateCount != iteratorStateCount) {
674                                throw new ConcurrentModificationException();
675                        }
676
677                        if (destination == null)
678                                return (currentLink.getNext() != null);
679                        else
680                                // Ignore whether (currentLink.getNext() == null), so
681                                // next() will produce a NoSuchElementException if
682                                // destination is not in the chain.
683                                return (destination != currentLink.getItem());
684                }
685
686                public E next() throws NoSuchElementException {
687                        if (stateCount != iteratorStateCount)
688                                throw new ConcurrentModificationException();
689
690                        Link<E> temp = currentLink.getNext();
691                        if (temp == null) {
692                                String exceptionMsg;
693                                if (destination != null && destination != currentLink.getItem())
694                                        exceptionMsg = "HashChain.LinkIterator.next() reached end of chain without reaching specified tail unit";
695                                else
696                                        exceptionMsg = "HashChain.LinkIterator.next() called past the end of the Chain";
697                                throw new NoSuchElementException(exceptionMsg);
698                        }
699                        currentLink = temp;
700
701                        state = true;
702                        return currentLink.getItem();
703                }
704
705                public void remove() throws IllegalStateException {
706                        if (stateCount != iteratorStateCount)
707                                throw new ConcurrentModificationException();
708
709                        stateCount++;
710                        iteratorStateCount++;
711                        if (!state)
712                                throw new IllegalStateException();
713                        else {
714                                currentLink.unlinkSelf();
715                                map.remove(currentLink.getItem());
716                                state = false;
717                        }
718
719                }
720
721                public String toString() {
722                        if (currentLink == null)
723                                return "Current object under iterator is null"
724                                                + super.toString();
725                        else
726                                return currentLink.toString();
727                }
728
729        }
