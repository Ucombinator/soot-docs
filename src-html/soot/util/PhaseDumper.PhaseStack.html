048/**
049 * The <tt>PhaseDumper</tt> is a debugging aid.  It maintains two
050 * lists of phases to be debugged.  If a phase is on the
051 * <code>bodyDumpingPhases</code> list, then the intermediate
052 * representation of the bodies being manipulated by the phase is
053 * dumped before and after the phase is applied.  If a phase is on the
054 * <code>cfgDumpingPhases</code> list, then whenever a CFG is
055 * constructed during the phase, a dot file is dumped representing the
056 * CFG constructed.
057 */
058
059public class PhaseDumper {
060    // As a minor optimization, we leave these lists null in the
061    // case were no phases at all are to be dumped, which is the 
062    // most likely case.
063    private List bodyDumpingPhases = null;
064    private List cfgDumpingPhases = null;
065
066    private class PhaseStack extends ArrayList {
067        // We eschew java.util.Stack to avoid synchronization overhead.
068
069        private final static int initialCapacity = 4;
070        final static String EMPTY_STACK_PHASE_NAME = "NOPHASE";
071
072        PhaseStack() {
073            super(initialCapacity);
074        }
075
076        boolean empty() {
077            return (this.size() == 0);
078        }
079
080        String currentPhase() {
081            if (this.size() <= 0) {
082                return EMPTY_STACK_PHASE_NAME;
083            } else {
084                return (String) this.get(this.size() - 1);
085            }
086        }
087
088        String pop() {
089            return (String) this.remove(this.size() - 1);
090        }
091
092        String push(String phaseName) {
093            this.add(phaseName);
094            return phaseName;
095        }
096    }
097    private final PhaseStack phaseStack = new PhaseStack(); 
098    final static String allWildcard = "ALL";
099
100
101    public PhaseDumper(Singletons.Global g) {
102        if (! Options.v().dump_body().isEmpty()) {
103            bodyDumpingPhases = Options.v().dump_body();
104        }
105        if (! Options.v().dump_cfg().isEmpty()) {
106            cfgDumpingPhases = Options.v().dump_cfg();
107        }
108    }

121    private boolean isBodyDumpingPhase(String phaseName) {
122        return ((bodyDumpingPhases != null)
123                && (bodyDumpingPhases.contains(phaseName) || 
124                    bodyDumpingPhases.contains(allWildcard)));
125    }
126
127
128    private boolean isCFGDumpingPhase(String phaseName) {
129        if (cfgDumpingPhases == null) {
130            return false;
131        }
132        if (cfgDumpingPhases.contains(allWildcard)) {
133          return true;
134        } else {
135          while (true) { // loop exited by "return" or "break".
136            if (cfgDumpingPhases.contains(phaseName)) {
137              return true;
138            }
139            // Go on to check if phaseName is a subphase of a
140            // phase in cfgDumpingPhases.
141            int lastDot = phaseName.lastIndexOf('.');
142            if (lastDot < 0) {
143              break;
144            } else {
145              phaseName = phaseName.substring(0, lastDot);
146            }
147          }
148          return false;
149        }
150    }

153    private static java.io.File makeDirectoryIfMissing(Body b) 
154        throws java.io.IOException {
155        StringBuffer buf = 
156            new StringBuffer(soot.SourceLocator.v().getOutputDir());
157        buf.append(File.separatorChar);
158        String className = b.getMethod().getDeclaringClass().getName();
159        buf.append(className);
160        buf.append(File.separatorChar);
161        buf.append(b.getMethod().getSubSignature().replace('<', '[').replace('>', ']'));
162        java.io.File dir = new java.io.File(buf.toString());
163        if (dir.exists()) {
164            if (! dir.isDirectory()) {
165                throw new java.io.IOException(dir.getPath() + " exists but is not a directory.");
166            }
167        } else {
168            if (! dir.mkdirs())  {
169                throw new java.io.IOException("Unable to mkdirs " + dir.getPath());
170            }
171        }
172        return dir;
173    }
174
175
176    private static PrintWriter openBodyFile(Body b, String baseName) 
177        throws java.io.IOException {
178        File dir = makeDirectoryIfMissing(b);
179        String filePath = dir.toString() + File.separatorChar + baseName;
180        return 
181            new PrintWriter(new java.io.FileOutputStream(filePath));
182    }

185    /**
186     * Returns the next available name for a graph file.
187     */
188
189    private static String nextGraphFileName(Body b, String baseName) 
190        throws java.io.IOException {
191        // We number output files to allow multiple graphs per phase.
192        File dir = makeDirectoryIfMissing(b);
193        final String prefix = dir.toString() + File.separatorChar + baseName;
194        File file = null;
195        int fileNumber = 0;
196        do {
197            file = new File(prefix + fileNumber + DotGraph.DOT_EXTENSION);
198            fileNumber++;
199        } while (file.exists());
200        return file.toString();
201    }

203
204    private static void deleteOldGraphFiles(final Body b,
205                                            final String phaseName) {
206        try {
207            final File dir = makeDirectoryIfMissing(b);
208            final File[] toDelete = dir.listFiles(new java.io.FilenameFilter() {
209                    public boolean accept(File dir, String name) {
210                        return name.startsWith(phaseName) && 
211                            name.endsWith(DotGraph.DOT_EXTENSION);
212                    }
213                });
214            if (toDelete != null)
215                    for (File element : toDelete) {
216                        element.delete();
217                    }
218        } catch (java.io.IOException e) {
219            // Don't abort execution because of an I/O error, but report
220            // the error.
221            G.v().out.println("PhaseDumper.dumpBody() caught: " + e.toString());
222            e.printStackTrace(G.v().out);
223        }
224    }

227    // soot.Printer itself needs to create a BriefUnitGraph in order
228    // to format the text for a method's instructions, so this flag is
229    // a hack to avoid dumping graphs that we create in the course of
230    // dumping bodies or other graphs.  
231    //
232    // Note that this hack would not work if a PhaseDumper might be
233    // accessed by multiple threads.  So long as there is a single
234    // active PhaseDumper accessed through soot.G, it seems
235    // safe to assume it will be accessed by only a single thread.
236    private boolean alreadyDumping = false;
237    
238    public void dumpBody(Body b, String baseName) {
239        try {
240            alreadyDumping = true;
241            java.io.PrintWriter out = openBodyFile(b, baseName);
242            soot.Printer.v().setOption(Printer.USE_ABBREVIATIONS);
243            soot.Printer.v().printTo(b, out);
244            out.close();
245        } catch (java.io.IOException e) {
246            // Don't abort execution because of an I/O error, but let
247            // the user know.
248            G.v().out.println("PhaseDumper.dumpBody() caught: " + e.toString());
249            e.printStackTrace(G.v().out);
250        } finally {
251            alreadyDumping = false;
252        }
253    }
254
255    private void dumpAllBodies(String baseName, 
256                                      boolean deleteGraphFiles) {
257        List<SootClass> classes = Scene.v().getClasses(SootClass.BODIES);
258        for (SootClass cls : classes) {
259            for (Iterator m = cls.getMethods().iterator(); m.hasNext(); ) {
260                SootMethod method = (SootMethod) m.next();
261                if (method.hasActiveBody()) {
262                    Body body = method.getActiveBody();
263                    if (deleteGraphFiles) {
264                        deleteOldGraphFiles(body, baseName);
265                    }
266                    dumpBody(body, baseName);
267                }
268            }
269        }
270    }

273    /**
274     * Tells the <code>PhaseDumper</code> that a {@link Body}
275     * transforming phase has started, so that it can dump the
276     * phases's &ldquo;before&rdquo; file.  If the phase is to be
277     * dumped, <code>dumpBefore</code> deletes any old
278     * graph files dumped during previous runs of the phase.
279     *
280     * @param b the {@link Body} being transformed.
281     * @param phaseName the name of the phase that has just started.
282     */
283    public void dumpBefore(Body b, String phaseName) {
284        phaseStack.push(phaseName);
285        if (isBodyDumpingPhase(phaseName)) {
286                deleteOldGraphFiles(b, phaseName);
287                dumpBody(b, phaseName + ".in");
288        }
289    }

292    /**
293     * Tells the <code>PhaseDumper</code> that a {@link Body}
294     * transforming phase has ended, so that it can dump the
295     * phases's &ldquo;after&rdquo; file.
296     *
297     * @param b the {@link Body} being transformed.
298     *
299     * @param phaseName the name of the phase that has just ended.
300     *
301     * @throws IllegalArgumentException if <code>phaseName</code> does not
302     * match the <code>PhaseDumper</code>'s record of the current phase.
303     */
304    public void dumpAfter(Body b, String phaseName) {
305        String poppedPhaseName = phaseStack.pop();
306        if (poppedPhaseName != phaseName) {
307            throw new IllegalArgumentException("dumpAfter(" + phaseName + 
308                                               ") when poppedPhaseName == " +
309                                               poppedPhaseName);
310        }
311        if (isBodyDumpingPhase(phaseName)) {
312            dumpBody(b, phaseName + ".out");
313        }
314    }

317    /**
318     * Tells the <code>PhaseDumper</code> that a {@link Scene}
319     * transforming phase has started, so that it can dump the
320     * phases's &ldquo;before&rdquo; files.  If the phase is to be
321     * dumped, <code>dumpBefore</code> deletes any old
322     * graph files dumped during previous runs of the phase.
323     *
324     * @param phaseName the name of the phase that has just started.
325     */
326    public void dumpBefore(String phaseName) {
327        phaseStack.push(phaseName);
328        if (isBodyDumpingPhase(phaseName)) {
329            dumpAllBodies(phaseName + ".in", true);
330        }
331    }

334    /**
335     * Tells the <code>PhaseDumper</code> that a {@link Scene}
336     * transforming phase has ended, so that it can dump the
337     * phases's &ldquo;after&rdquo; files.
338     *
339     * @param phaseName the name of the phase that has just ended.
340     *
341     * @throws IllegalArgumentException if <code>phaseName</code> does not
342     * match the <code>PhaseDumper</code>'s record of the current phase.
343     */
344    public void dumpAfter(String phaseName) {
345        String poppedPhaseName = phaseStack.pop();
346        if (poppedPhaseName != phaseName) {
347            throw new IllegalArgumentException("dumpAfter(" + phaseName + 
348                                               ") when poppedPhaseName == " +
349                                               poppedPhaseName);
350        }
351        if (isBodyDumpingPhase(phaseName)) {
352            dumpAllBodies(phaseName + ".out", false);
353        }
354    }

357    /**
358     * Asks the <code>PhaseDumper</code> to dump the passed {@link
359     * DirectedGraph} if the current phase is being dumped.
360     *
361     * @param g the graph to dump.
362     *
363     * @param body the {@link Body} represented by <code>g</code>.
364     */
365    public void dumpGraph(DirectedGraph g, Body b) {
366        if (alreadyDumping) {
367            return;
368        }
369        try {
370            alreadyDumping = true;
371            String phaseName = phaseStack.currentPhase();
372            if (isCFGDumpingPhase(phaseName)) { 
373                try {
374                    String outputFile = nextGraphFileName(b, phaseName + "-" + 
375                                                          getClassIdent(g) + "-");
376                    DotGraph dotGraph = new CFGToDotGraph().drawCFG(g, b);
377                    dotGraph.plot(outputFile);
378
379                } catch (java.io.IOException e) {
380                    // Don't abort execution because of an I/O error, but 
381                    // report the error.
382                    G.v().out.println("PhaseDumper.dumpBody() caught: " + 
383                                      e.toString());
384                    e.printStackTrace(G.v().out);
385                }
386            }
387        } finally {
388            alreadyDumping = false;
389        }
390    }

393    /**
394     * Asks the <code>PhaseDumper</code> to dump the passed {@link
395     * ExceptionalGraph} if the current phase is being dumped.
396     *
397     * @param g the graph to dump.
398     */
399    public void dumpGraph(ExceptionalGraph g) {
400        if (alreadyDumping) {
401            return;
402        }
403        try {
404            alreadyDumping = true;
405            String phaseName = phaseStack.currentPhase();
406            if (isCFGDumpingPhase(phaseName)) {
407                try {
408                    String outputFile = nextGraphFileName(g.getBody(), 
409                                                          phaseName + "-" + 
410                                                          getClassIdent(g) + "-");
411                    CFGToDotGraph drawer = new CFGToDotGraph();
412                    drawer.setShowExceptions(Options.v().show_exception_dests());
413                    DotGraph dotGraph = drawer.drawCFG(g);
414                    dotGraph.plot(outputFile);
415
416                } catch (java.io.IOException e) {
417                    // Don't abort execution because of an I/O error, but 
418                    // report the error.
419                    G.v().out.println("PhaseDumper.dumpBody() caught: " + 
420                                      e.toString());
421                    e.printStackTrace(G.v().out);
422                }
423            }
424        } finally {
425            alreadyDumping = false;
426        }
427    }

429    /**
430     * A utility routine that returns the unqualified identifier
431     * naming the class of an object.
432     *
433     * @param obj The object whose class name is to be returned.
434     */
435    private String getClassIdent(Object obj) {
436        String qualifiedName = obj.getClass().getName();
437        int lastDotIndex = qualifiedName.lastIndexOf('.');
438        return qualifiedName.substring(lastDotIndex+1);
439    }
