027package soot.util;
028
029import java.io.Serializable;
030import java.util.Collection;
031import java.util.Iterator;
032import java.util.List;
033
034/** Augmented data type guaranteeing O(1) insertion and removal from a set
035 * of ordered, unique elements.
036 * @param <E> element type  */
037public interface Chain<E> extends Collection<E>, Serializable
038{
039    /** Inserts <code>toInsert</code> in the Chain before <code>point</code>. */
040    public void insertBefore(List<E> toInsert, E point);
041    /** Inserts <code>toInsert</code> in the Chain after <code>point</code>. */
042    public void insertAfter(List<E> toInsert, E point);
043    /** Inserts <code>toInsert</code> in the Chain after <code>point</code>. */
044    public void insertAfter(E toInsert, E point);
045    /** Inserts <code>toInsert</code> in the Chain before <code>point</code>. */
046    public void insertBefore(E toInsert, E point);
047    /** Inserts <code>toInsert</code> in the Chain before <code>point</code>.
048     * (It would probably be better to make Chain implement List)*/
049    public void insertBefore(Chain<E> toInsert, E point);
050    /** Inserts <code>toInsert</code> in the Chain after <code>point</code>. 
051     * (It would probably be better to make Chain implement List)*/
052    public void insertAfter(Chain<E> toInsert, E point);
053    
054    /**
055         * Inserts instrumentation in a manner such that the resulting control flow
056         * graph (CFG) of the program will contain <code>toInsert</code> on an edge
057         * that is defined by <code>point_source</code> and <code>point_target</code>.
058         * 
059         * @param toInsert  the instrumentation to be added in the Chain
060         * @param point_src the source point of an edge in CFG
061         * @param point_tgt the target point of an edge
062         */
063    public void insertOnEdge(E toInsert, E point_src, E point_tgt);
064    
065    /**
066         * Inserts instrumentation in a manner such that the resulting control flow
067         * graph (CFG) of the program will contain <code>toInsert</code> on an edge
068         * that is defined by <code>point_source</code> and <code>point_target</code>.
069         * 
070         * @param toInsert  instrumentation to be added in the Chain
071         * @param point_src the source point of an edge in CFG
072         * @param point_tgt the target point of an edge
073         */
074    public void insertOnEdge(List<E> toInsert, E point_src, E point_tgt);
075    
076    /**
077         * Inserts instrumentation in a manner such that the resulting control flow
078         * graph (CFG) of the program will contain <code>toInsert</code> on an edge
079         * that is defined by <code>point_source</code> and <code>point_target</code>.
080         * 
081         * @param toInsert  instrumentation to be added in the Chain
082         * @param point_src the source point of an edge in CFG
083         * @param point_tgt the target point of an edge
084         */
085    public void insertOnEdge(Chain<E> toInsert, E point_src, E point_tgt);
086    
087    /** Replaces <code>out</code> in the Chain by <code>in</code>. */
088    public void swapWith(E out, E in);
089
090    /** Removes the given object from this Chain.
091     *  Parameter has to be of type {@link Object} to be compatible
092     *  with the {@link Collection} interface. */
093    public boolean remove(Object u);
094
095    /** Adds the given object at the beginning of the Chain. */
096    public void addFirst(E u);
097
098    /** Adds the given object at the end of the Chain. */
099    public void addLast(E u);
100
101    /** Removes the first object contained in this Chain. */
102    public void removeFirst();
103    /** Removes the last object contained in this Chain. */
104    public void removeLast();
105
106    /** Returns true if object <code>someObject</code> follows object <code>someReferenceObject</code> in the Chain. */
107    public boolean follows(E someObject, E someReferenceObject);
108
109    /** Returns the first object in this Chain. */
110    public E getFirst();
111
112    /** Returns the last object in this Chain. */
113    public E getLast();
114    
115    /** Returns the object immediately following <code>point</code>. */
116    public E getSuccOf(E point);
117
118    /** Returns the object immediately preceding <code>point</code>. */
119    public E getPredOf(E point);
120
121    /** Returns an iterator over a copy of this chain. 
122     * This avoids ConcurrentModificationExceptions from being thrown
123     * if the underlying Chain is modified during iteration.
124     * Do not use this to remove elements which have not yet been
125     * iterated over! */
126    public Iterator<E> snapshotIterator();
127
128    /** Returns an iterator over this Chain. */
129    public Iterator<E> iterator();
130
131    /** Returns an iterator over this Chain, starting at the given object. */
132    public Iterator<E> iterator(E u);
133
134    /** Returns an iterator over this Chain, starting at head and reaching tail (inclusive). */
135    public Iterator<E> iterator(E head, E tail);
136
137    /** Returns the size of this Chain. */
138    public int size();
139    
140    /** Returns the number of times this chain has been modified. */
141        long getModificationCount();
142        
143        /**
144         * Gets all elements in the chain. There is no guarantee on sorting. On the other
145         * hand, the collection returned by this method is thread-safe. You can iterate
146         * over it even in the case of concurrent modifications to the underlying chain.
147         * @return All elements in the chain in an unsorted collection
148         */
149        public Collection<E> getElementsUnsorted();
150        
151}
