020package soot.util;
021
022public final class SharedBitSet {
023    BitVector value;
024    boolean own = true;
025    public SharedBitSet(int i) {
026        value = new BitVector( i );
027    }
028    public SharedBitSet() {
029        this(32);
030    }
031    private void acquire() {
032        if( own ) return;
033        own = true;
034        value = (BitVector) value.clone();
035    }
036    private void canonicalize() {
037        value = SharedBitSetCache.v().canonicalize( value );
038        own = false;
039    }
040    public boolean set(int bit) {
041        acquire();
042        return value.set(bit);
043    }
044    public void clear(int bit) {
045        acquire();
046        value.clear(bit);
047    }
048    public boolean get(int bit) {
049        return value.get(bit);
050    }
051    public void and( SharedBitSet other ) {
052        if( own ) {
053            value.and( other.value );
054        } else {
055            value = BitVector.and( value, other.value );
056            own = true;
057        }
058        canonicalize();
059    }
060    public void or( SharedBitSet other ) {
061        if( own ) {
062            value.or( other.value );
063        } else {
064            value = BitVector.or( value, other.value );
065            own = true;
066        }
067        canonicalize();
068    }
069    public boolean orAndAndNot( SharedBitSet orset, SharedBitSet andset, SharedBitSet andnotset ) {
070        acquire();
071        boolean ret = value.orAndAndNot( orset.value, andset.value, andnotset.value );
072        canonicalize();
073        return ret;
074    }
075    public boolean orAndAndNot( SharedBitSet orset, BitVector andset, SharedBitSet andnotset ) {
076        acquire();
077        boolean ret = value.orAndAndNot( orset.value, andset,
078                andnotset == null ? null : andnotset.value );
079        canonicalize();
080        return ret;
081    }
082    public BitSetIterator iterator() {
083        return value.iterator(); 
084    }
085    public String toString() {
086        StringBuffer b = new StringBuffer();
087        for( BitSetIterator it = iterator(); it.hasNext(); ) {
088            b.append( it.next() );
089            b.append( "," );
090        }
091        return b.toString();
092    }
093}
