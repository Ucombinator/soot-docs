028package soot.util.dot;
029
030import java.io.*;
031import java.util.*;
032
033/* Graph edges are the major elements of a graph
034 * @author Feng Qian 
035 */
036public class DotGraphEdge implements Renderable {
037  private boolean isDirected;
038  private DotGraphNode start, end;
039  private List<DotGraphAttribute> attributes;
040
041  /**
042   * Draws a directed edge.
043   * @param src, the source node
044   * @param dst, the end node
045   */
046  public DotGraphEdge(DotGraphNode src, DotGraphNode dst){
047    this.start = src;
048    this.end   = dst;
049    this.isDirected = true;
050  }
051
052  /**
053   * Draws a graph edge by specifying directed or undirected.
054   * @param src, the source node
055   * @param dst, the end node
056   * @param directed, the edge is directed or not
057   */
058  public DotGraphEdge(DotGraphNode src, DotGraphNode dst, boolean directed){
059    this.start = src;
060    this.end   = dst;
061    this.isDirected = directed;
062  }
063
064  /**
065   * Sets the label for the edge.
066   * @param label, a label string
067   */
068  public void setLabel(String label){
069    label = DotGraphUtility.replaceQuotes(label);
070    label = DotGraphUtility.replaceReturns(label);
071    this.setAttribute("label", "\""+label+"\"");
072  }
073
074  /**
075   * Sets the edge style.
076   * @param style, a style of edge
077   * @see DotGraphConstants
078   */
079  public void setStyle(String style){
080    this.setAttribute("style", style);
081  }
082
083  /**
084   * Sets an edge attribute.
085   * @param id, the attribute id to be set
086   * @param value, the attribute value
087   */
088  public void setAttribute(String id, String value) {
089    this.setAttribute(new DotGraphAttribute(id, value));    
090  }
091
092  /**
093   * Sets an edge attribute.
094   * @param attr, a {@link DotGraphAttribute} specifying the
095   * attribute name and value.
096   */
097  public void setAttribute(DotGraphAttribute attr) {
098    if (this.attributes == null) {
099      this.attributes = new LinkedList<DotGraphAttribute>();
100    }
101    
102    this.attributes.add(attr);    
103  }
104
105  public void render(OutputStream out, int indent) throws IOException {
106    StringBuffer line = new StringBuffer(start.getName());
107    line.append((this.isDirected)?"->":"--");
108    line.append(end.getName());
109
110    if (this.attributes != null) {
111      
112      line.append(" [");
113      Iterator<DotGraphAttribute> attrIt = this.attributes.iterator();
114      while (attrIt.hasNext()) {
115        DotGraphAttribute attr = attrIt.next();
116        line.append(attr.toString());
117        line.append(",");
118      }
119      line.append("]");
120    }
121
122    line.append(";");
123
124    DotGraphUtility.renderLine(out, new String(line), indent);
125  }
126}
