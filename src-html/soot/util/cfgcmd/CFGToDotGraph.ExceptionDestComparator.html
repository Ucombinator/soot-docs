055  /**
056   * <p>Returns a CFGToDotGraph converter which will draw the graph
057   * as a single arbitrarily-sized page, with full-length node labels.</p>
058   *
059   * <p> If asked to draw a <code>ExceptionalGraph</code>, the
060   * converter will identify the exceptions that will be thrown. By
061   * default, it will distinguish different edges by coloring regular
062   * control flow edges black, exceptional control flow edges red, and
063   * thrown exception edges light gray. Head and tail nodes are filled
064   * in, head nodes with gray, and tail nodes with light gray.</p>
065   */
066  public CFGToDotGraph() {
067    setOnePage(true);
068    setBriefLabels(false);
069    setShowExceptions(true);
070    setUnexceptionalControlFlowAttr("color", "black");
071    setExceptionalControlFlowAttr("color", "red");
072    setExceptionEdgeAttr("color", "lightgray");
073    setHeadAttr("fillcolor", "gray");
074    setTailAttr("fillcolor", "lightgray");
075  }

199  /**
200   * Returns an {@link Iterator} over a {@link Collection} which
201   * iterates over its elements in a specified order.  Used to order
202   * lists of destination nodes consistently before adding the
203   * corresponding edges to the graph. (Maintaining a consistent
204   * ordering of edges makes it easier to diff the dot files output
205   * for different graphs of a given method.)
206   *
207   * @param coll The collection to iterator over.
208   *
209   * @param comp The comparator for the ordering.
210   *
211   * @return An iterator which presents the elements of <code>coll</code> 
212   * in the order specified by <code>comp</code>.
213   */
214  private static <T> Iterator<T> sortedIterator(Collection<T> coll, Comparator<T> comp) {
215    if (coll.size() <= 1) {
216      return coll.iterator();
217    } else {
218      ArrayList<T> list = new ArrayList<T>(coll);
219      Collections.sort(list, comp);
220      return list.iterator();
221    }
222  }

224  /**
225   * Comparator used to order a list of nodes by the order in 
226   * which they were labeled.
227   */
228  private static class NodeComparator<T> implements Comparator<T> {
229    private DotNamer<T> namer;
230
231    NodeComparator(DotNamer<T> namer) {
232      this.namer = namer;
233    }
234
235    public int compare(T o1, T o2) {
236      return (namer.getNumber(o1) - namer.getNumber(o2));
237    }
238
239    public boolean equal(T o1, T o2) {
240      return (namer.getNumber(o1) == namer.getNumber(o2));
241    }
242  }
243
244
245  /**
246   * Comparator used to order a list of ExceptionDests by the order in
247   * which their handler nodes were labeled.
248   */
249  private static class ExceptionDestComparator<T> implements Comparator<ExceptionDest<T>> {
250    private DotNamer<T> namer;
251
252    ExceptionDestComparator(DotNamer<T> namer) {
253      this.namer = namer;
254    }
255
256    private int getValue(ExceptionDest<T> o) {
257      T handler = o.getHandlerNode();
258      if (handler == null) {
259        return Integer.MAX_VALUE;
260      } else {
261        return namer.getNumber(handler);
262      }
263    }
264
265    public int compare(ExceptionDest<T> o1, ExceptionDest<T> o2) {
266      return (getValue(o1) - getValue(o2));
267    }
268
269    public boolean equal(ExceptionDest<T> o1, ExceptionDest<T> o2) {
270      return (getValue(o1) == getValue(o2));
271    }
272  }

275  /**
276   * Create a <code>DotGraph</code> whose nodes and edges depict 
277   * a control flow graph without distinguished
278   * exceptional edges.
279   * 
280   * @param graph a <code>DirectedGraph</code> representing a CFG
281   *              (probably an instance of {@link UnitGraph}, {@link BlockGraph},
282   *              or one of their subclasses).
283   *
284   * @param body the <code>Body</code> represented by <code>graph</code> (used
285   * to format the text within nodes).  If no body is available, pass
286   * <code>null</code>.
287   *
288   * @return a visualization of <code>graph</code>.
289   */
290  public <N> DotGraph drawCFG(DirectedGraph<N> graph, Body body) {
291    DotGraph canvas = initDotGraph(body);
292    DotNamer<N> namer = new DotNamer<N>((int)(graph.size()/0.7f), 0.7f);
293    NodeComparator<N> comparator = new NodeComparator<N>(namer);
294
295    // To facilitate comparisons between different graphs of the same
296    // method, prelabel the nodes in the order they appear
297    // in the iterator, rather than the order that they appear in the
298    // graph traversal (so that corresponding nodes are more likely
299    // to have the same label in different graphs of a given method).
300    for (Iterator<N> nodesIt = graph.iterator(); nodesIt.hasNext(); ) {
301      namer.getName(nodesIt.next());
302    }
303
304    for (Iterator<N> nodesIt = graph.iterator(); nodesIt.hasNext(); ) {
305      N node = nodesIt.next();
306      canvas.drawNode(namer.getName(node));
307      for (Iterator<N> succsIt = sortedIterator(graph.getSuccsOf(node), comparator);
308           succsIt.hasNext(); ) {
309        N succ = succsIt.next();
310        canvas.drawEdge(namer.getName(node), namer.getName(succ));
311      }
312    }
313    setStyle(graph.getHeads(), canvas, namer,
314             DotGraphConstants.NODE_STYLE_FILLED, headAttr);
315    setStyle(graph.getTails(), canvas, namer, 
316             DotGraphConstants.NODE_STYLE_FILLED, tailAttr);
317    if (! isBrief) {
318      formatNodeText(body, canvas, namer);
319    }
320
321    return canvas;
322  }

325  /**
326   * Create a <code>DotGraph</code> whose nodes and edges depict the
327   * control flow in a <code>ExceptionalGraph</code>, with 
328   * distinguished edges for exceptional control flow.
329   * 
330   * @param graph the control flow graph
331   *
332   * @return a visualization of <code>graph</code>.
333   */
334  public <N> DotGraph drawCFG(ExceptionalGraph<N> graph) {
335    Body body = graph.getBody();
336    DotGraph canvas = initDotGraph(body);
337    DotNamer namer = new DotNamer((int)(graph.size()/0.7f), 0.7f);
338    NodeComparator nodeComparator = new NodeComparator(namer);
339
340    // Prelabel nodes in iterator order, to facilitate
341    // comparisons between graphs of a given method.
342    for (Iterator<N> nodesIt = graph.iterator(); nodesIt.hasNext(); ) {
343      namer.getName(nodesIt.next());
344    }
345
346    for (Iterator<N> nodesIt = graph.iterator(); nodesIt.hasNext(); ) {
347        N node = nodesIt.next();
348
349      canvas.drawNode(namer.getName(node));
350
351      for (Iterator<N> succsIt = sortedIterator(graph.getUnexceptionalSuccsOf(node),
352                                             nodeComparator); 
353           succsIt.hasNext(); ) {
354        N succ = succsIt.next();
355        DotGraphEdge edge = canvas.drawEdge(namer.getName(node), 
356                                            namer.getName(succ));
357        edge.setAttribute(unexceptionalControlFlowAttr);
358      }
359
360      for (Iterator<N> succsIt = sortedIterator(graph.getExceptionalSuccsOf(node),
361                                             nodeComparator); 
362           succsIt.hasNext(); ) {
363        N succ = succsIt.next();
364        DotGraphEdge edge = canvas.drawEdge(namer.getName(node),
365                                            namer.getName(succ));
366        edge.setAttribute(exceptionalControlFlowAttr);
367      }
368
369      if (showExceptions) {
370        for (Iterator<ExceptionDest<N>> destsIt =
371                sortedIterator(graph.getExceptionDests(node), new ExceptionDestComparator(namer));
372             destsIt.hasNext(); ) {
373          ExceptionDest<N> dest = destsIt.next();
374          Object handlerStart = dest.getHandlerNode();
375          if (handlerStart == null) {
376            // Giving each escaping exception its own, invisible
377            // exceptional exit node produces a less cluttered
378            // graph.
379            handlerStart = new Object() {
380                public String toString() {
381                  return "Esc";
382                }
383              };
384            DotGraphNode escapeNode = 
385              canvas.drawNode(namer.getName(handlerStart));
386            escapeNode.setStyle(DotGraphConstants.NODE_STYLE_INVISIBLE);
387          }
388          DotGraphEdge edge = canvas.drawEdge(namer.getName(node),
389                                              namer.getName(handlerStart));
390          edge.setAttribute(exceptionEdgeAttr);
391          edge.setLabel(formatThrowableSet(dest.getThrowables()));
392        }
393      }
394    }
395    setStyle(graph.getHeads(), canvas, namer,
396             DotGraphConstants.NODE_STYLE_FILLED, headAttr);
397    setStyle(graph.getTails(), canvas, namer, 
398             DotGraphConstants.NODE_STYLE_FILLED, tailAttr);
399    if (! isBrief) {
400      formatNodeText(graph.getBody(), canvas, namer);
401    }
402    return canvas;
403  }

406  /**
407   * A utility method that initializes a DotGraph object for use in any 
408   * variety of drawCFG().
409   *
410   * @param body The <code>Body</code> that the graph will represent
411   *             (used in the graph's title).  If no <code>Body</code>
412   *             is available, pass <code>null</code>.
413   *
414   * @return a <code>DotGraph</code> for visualizing <code>body</code>.
415   */
416  private DotGraph initDotGraph(Body body) {
417    String graphname = "cfg";
418    if (body != null) {
419      graphname = body.getMethod().getSubSignature();
420    } 
421    DotGraph canvas = new DotGraph(graphname);
422    canvas.setGraphLabel(graphname);
423    if (!onePage) {
424      canvas.setPageSize(8.5, 11.0);
425    }
426    if (isBrief) {
427      canvas.setNodeShape(DotGraphConstants.NODE_SHAPE_CIRCLE);
428    } else {
429      canvas.setNodeShape(DotGraphConstants.NODE_SHAPE_BOX);
430    }
431    return canvas;
432  }

435  /**
436   * A utility class for assigning unique names to DotGraph
437   * entities.  It maintains a mapping from CFG <code>Object</code>s
438   * to strings identifying the corresponding nodes in generated dot
439   * source.
440   */
441  @SuppressWarnings("serial")
442  private static class DotNamer<N> extends HashMap<N,Integer> {
443    private int nodecount = 0;
444
445    DotNamer(int initialCapacity, float loadFactor) {
446      super(initialCapacity, loadFactor);
447    }
448
449    String getName(N node) {
450      Integer index = this.get(node);
451      if (index == null) {
452        index = nodecount++;
453        this.put(node, index);
454      }
455      return index.toString();
456    }
457
458    int getNumber(N node) {
459      Integer index = this.get(node);
460      if (index == null) {
461        index = nodecount++;
462        this.put(node, index);
463      }
464      return index;
465    }
466  }

470  /**
471   * A utility method which formats the text for each node in 
472   * a <code>DotGraph</code> representing a CFG.
473   *
474   * @param body the <code>Body</code> whose control flow is visualized in
475   *             <code>canvas</code>.
476   *
477   * @param canvas the <code>DotGraph</code> for visualizing the CFG.
478   *
479   * @param namer provides a mapping from CFG objects to identifiers in
480   *              generated dot source.
481   */
482  private <N> void formatNodeText(Body body, DotGraph canvas, DotNamer<N> namer) {
483
484    LabeledUnitPrinter printer = null;
485    if (body != null) {
486      printer = new BriefUnitPrinter(body);
487      printer.noIndent();
488    }
489
490    for (Iterator<N> nodesIt = namer.keySet().iterator();
491         nodesIt.hasNext(); ) {
492      N node = nodesIt.next();
493      DotGraphNode dotnode = canvas.getNode(namer.getName(node));
494      String nodeLabel = null;
495      
496      if(node instanceof DominatorNode){
497          node = ((DominatorNode<N>)node).getGode();
498      }
499
500      if (printer == null) {
501        nodeLabel = node.toString();
502      } else {
503        if (node instanceof Unit) {
504          ((Unit) node).toString(printer);
505          String targetLabel = printer.labels().get(node);
506          if (targetLabel == null) {
507            nodeLabel = printer.toString();
508          } else {
509            nodeLabel = targetLabel + ": " + printer.toString();
510          }
511
512        } else if (node instanceof Block) {
513          Iterator<Unit> units = ((Block) node).iterator();
514          StringBuffer buffer = new StringBuffer();
515          while (units.hasNext()) {
516            Unit unit = units.next();
517            String targetLabel = (String) printer.labels().get(unit);
518            if (targetLabel != null) {
519              buffer.append(targetLabel)
520                .append(":\\n");
521            }
522            unit.toString(printer);
523            buffer.append(printer.toString())
524              .append("\\l");
525          }
526          nodeLabel = buffer.toString();
527        } else {
528          nodeLabel = node.toString();
529        }
530      }
531      dotnode.setLabel(nodeLabel);
532    }
533  }

536  /**
537   * Utility routine for setting some common formatting style for the
538   * {@link DotGraphNode}s corresponding to some collection of objects.
539   * 
540   * @param objects is the collection of {@link Object}s whose
541   *        nodes are to be styled.
542   * @param canvas the {@link DotGraph} containing nodes corresponding
543   *        to the collection.
544   * @param namer maps from {@link Object} to the strings used
545   *        to identify corresponding {@link DotGraphNode}s.
546   * @param style the style to set for each of the nodes.
547   * @param attrib if non-null, an additional attribute to associate
548   *        with each of the nodes.
549   */
550  private <N> void setStyle(Collection<? extends N> objects, DotGraph canvas, 
551                        DotNamer<N> namer, String style,
552                        DotGraphAttribute attrib) {
553    // Fill the entry and exit nodes.
554          for (N object : objects) {
555      DotGraphNode objectNode = canvas.getNode(namer.getName(object));
556      objectNode.setStyle(style);
557      objectNode.setAttribute(attrib);
558    }
559  }

561  /**
562   * Utility routine to format the list of names in 
563   * a ThrowableSet into a label for the edge showing where those
564   * Throwables are handled.
565   */
566  private String formatThrowableSet(ThrowableSet set) {
567    String input = set.toAbbreviatedString();
568
569    // Insert line breaks between individual Throwables (dot seems to
570    // orient these edges more or less vertically, most of the time).
571    int inputLength = input.length();
572    StringBuffer result = new StringBuffer(inputLength+5);
573    for (int i = 0; i < inputLength; i++) {
574      char c = input.charAt(i);
575      if (c == '+' || c == '-') {
576        result.append("\\l");
577      }
578      result.append(c);
579    }
580    return result.toString();
581  }
