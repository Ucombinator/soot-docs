110    public static int sizeOfType(Type t)
111    {
112        if(t instanceof DoubleWordType || t instanceof LongType || t instanceof DoubleType)
113            return 2;
114        else if(t instanceof VoidType)
115            return 0;
116        else
117            return 1;
118    }

120    public static int argCountOf(SootMethodRef m)
121    {
122        int argCount = 0;
123        Iterator<Type> typeIt = m.parameterTypes().iterator();
124
125        while(typeIt.hasNext())
126        {
127            Type t = (Type) typeIt.next();
128
129            argCount += sizeOfType(t);
130        }
131
132        return argCount;
133    }

135    public static String jasminDescriptorOf(Type type)
136    {
137        TypeSwitch sw;
138
139        type.apply(sw = new TypeSwitch()
140        {
141            public void caseBooleanType(BooleanType t)
142            {
143                setResult("Z");
144            }
145
146            public void caseByteType(ByteType t)
147            {
148                setResult("B");
149            }
150
151            public void caseCharType(CharType t)
152            {
153                setResult("C");
154            }
155
156            public void caseDoubleType(DoubleType t)
157            {
158                setResult("D");
159            }
160
161            public void caseFloatType(FloatType t)
162            {
163                setResult("F");
164            }
165
166            public void caseIntType(IntType t)
167            {
168                setResult("I");
169            }
170
171            public void caseLongType(LongType t)
172            {
173                setResult("J");
174            }
175
176            public void caseShortType(ShortType t)
177            {
178                setResult("S");
179            }
180
181            public void defaultCase(Type t)
182            {
183                throw new RuntimeException("Invalid type: " + t);
184            }
185
186            public void caseArrayType(ArrayType t)
187            {
188                StringBuffer buffer = new StringBuffer();
189
190                for(int i = 0; i < t.numDimensions; i++)
191                    buffer.append("[");
192
193                setResult(buffer.toString() + jasminDescriptorOf(t.baseType));
194            }
195
196            public void caseRefType(RefType t)
197            {
198                setResult("L" + t.getClassName().replace('.', '/') + ";");
199            }
200
201            public void caseVoidType(VoidType t)
202            {
203                setResult("V");
204            }
205        });
206
207        return (String) sw.getResult();
208

211    public static String jasminDescriptorOf(SootMethodRef m)
212    {
213        StringBuffer buffer = new StringBuffer();
214
215        buffer.append("(");
216
217        // Add methods parameters
218        {
219            Iterator<Type> typeIt = m.parameterTypes().iterator();
220
221            while(typeIt.hasNext())
222            {
223                Type t = (Type) typeIt.next();
224
225                buffer.append(jasminDescriptorOf(t));
226            }
227        }
228
229        buffer.append(")");
230
231        buffer.append(jasminDescriptorOf(m.returnType()));
232
233        return buffer.toString();
234    }

249    private String getVisibilityAnnotationAttr(VisibilityAnnotationTag tag){
250        StringBuffer sb = new StringBuffer();
251        if(tag == null){
252                return "";
253        }else 
254                if (tag.getVisibility() == AnnotationConstants.RUNTIME_VISIBLE){
255            sb.append(".runtime_visible_annotation\n");
256        }
257        else if(tag.getVisibility() == AnnotationConstants.RUNTIME_INVISIBLE){
258            sb.append(".runtime_invisible_annotation\n");
259        } else {
260                //source level annotation
261                return "";
262        }
263        if (tag.hasAnnotations()){
264            Iterator<AnnotationTag> it = tag.getAnnotations().iterator();
265            while (it.hasNext()){
266                AnnotationTag annot = it.next();
267                sb.append(".annotation ");
268                sb.append(soot.util.StringTools.getQuotedStringOf(annot.getType())+"\n");
269                for (AnnotationElem ae : annot.getElems()){
270                  sb.append(getElemAttr(ae));
271                }
272                sb.append(".end .annotation\n");
273            }
274        }
275        sb.append(".end .annotation_attr\n");
276        return sb.toString();    
277    }

280    private String getVisibilityParameterAnnotationAttr(VisibilityParameterAnnotationTag tag){
281        StringBuffer sb = new StringBuffer();
282        sb.append(".param ");
283        if (tag.getKind() == AnnotationConstants.RUNTIME_VISIBLE){
284            sb.append(".runtime_visible_annotation\n");
285        }
286        else {
287            sb.append(".runtime_invisible_annotation\n");
288        }
289        ArrayList<VisibilityAnnotationTag> vis_list = tag.getVisibilityAnnotations();
290        if (vis_list != null){
291            for (VisibilityAnnotationTag vat : vis_list) {
292                VisibilityAnnotationTag safeVat = vat == null
293                                ? getSafeVisibilityAnnotationTag(tag.getKind()) : vat;
294                sb.append(getVisibilityAnnotationAttr(safeVat));
295            }
296        }
297        sb.append(".end .param\n");
298        return sb.toString();    
299    }

311        private String getElemAttr(AnnotationElem elem){
312        StringBuffer result = new StringBuffer(".elem ");
313        switch (elem.getKind()){
314            case 'Z': {
315                       result.append(".bool_kind ");
316                       result.append("\"" + elem.getName() + "\" ");
317                       if (elem instanceof AnnotationIntElem) {
318                               result.append(((AnnotationIntElem) elem).getValue());
319                       } else {
320                               if (((AnnotationBooleanElem) elem).getValue()) {
321                                       result.append(1);
322                               } else {
323                                       result.append(0);
324                               }
325                       }
326                       result.append("\n");
327                       break;
328                      }
329            case 'S': {
330                        result.append(".short_kind ");
331                        result.append("\""+elem.getName()+"\" ");
332                        result.append(((AnnotationIntElem)elem).getValue());
333                        result.append("\n");
334                        break;
335                      }
336            case 'B': {
337                        result.append(".byte_kind ");
338                        result.append("\""+elem.getName()+"\" ");
339                        result.append(((AnnotationIntElem)elem).getValue());
340                        result.append("\n");
341                        break;
342                      }
343            case 'C': {
344                        result.append(".char_kind ");
345                        result.append("\""+elem.getName()+"\" ");
346                        result.append(((AnnotationIntElem)elem).getValue());
347                        result.append("\n");
348                        break;
349                      }
350            case 'I': {
351                        result.append(".int_kind ");
352                        result.append("\""+elem.getName()+"\" ");
353                        result.append(((AnnotationIntElem)elem).getValue());
354                        result.append("\n");
355                        break;
356                      }
357            case 'J': {
358                        result.append(".long_kind ");
359                        result.append("\""+elem.getName()+"\" ");
360                        result.append(((AnnotationLongElem)elem).getValue());
361                        result.append("\n");
362                        break; 
363                      }
364            case 'F': {
365                        result.append(".float_kind ");
366                        result.append("\""+elem.getName()+"\" ");
367                        result.append(((AnnotationFloatElem)elem).getValue());
368                        result.append("\n");
369                        break;
370                      }
371            case 'D': {
372                        result.append(".doub_kind ");
373                        result.append("\""+elem.getName()+"\" ");
374                        result.append(((AnnotationDoubleElem)elem).getValue());
375                        result.append("\n");
376                        break;
377                      }
378            case 's': {
379                        result.append(".str_kind ");
380                        result.append("\""+elem.getName()+"\" ");
381                        result.append(soot.util.StringTools.getQuotedStringOf(((AnnotationStringElem)elem).getValue()));
382                        result.append("\n");
383                        break;
384                      }
385            case 'e': {
386                        result.append(".enum_kind ");
387                        result.append("\""+elem.getName()+"\" ");
388                        result.append(soot.util.StringTools.getQuotedStringOf(((AnnotationEnumElem)elem).getTypeName())); 
389                        result.append(" ");
390                        result.append(soot.util.StringTools.getQuotedStringOf(((AnnotationEnumElem)elem).getConstantName())); 
391                        result.append("\n");
392                        break;
393                      }
394            case 'c': {
395                        result.append(".cls_kind ");
396                        result.append("\""+elem.getName()+"\" ");
397                        result.append(soot.util.StringTools.getQuotedStringOf(((AnnotationClassElem)elem).getDesc())); 
398                        result.append("\n");
399                        break;
400                      }
401            case '[': {
402                        result.append(".arr_kind ");
403                        result.append("\""+elem.getName()+"\" ");
404                        AnnotationArrayElem arrayElem = (AnnotationArrayElem)elem;
405                        result.append("\n");
406                        for (int i = 0; i < arrayElem.getNumValues(); i++){
407                            //result.append("\n");
408                            result.append(getElemAttr(arrayElem.getValueAt(i)));
409                        }
410                        result.append(".end .arr_elem\n");
411                        break;
412                      }
413            case '@': {
414                        result.append(".ann_kind ");
415                        result.append("\""+elem.getName()+"\"\n");
416                        AnnotationTag annot = ((AnnotationAnnotationElem)elem).getValue();
417                        result.append(".annotation ");
418                        result.append(soot.util.StringTools.getQuotedStringOf(annot.getType())+"\n");
419                        for (AnnotationElem ae : annot.getElems()){
420                          result.append(getElemAttr(ae));
421                        }
422                        result.append(".end .annotation\n");
423                        result.append(".end .annot_elem\n");
424                        break;
425                      }
426            default : {
427                throw new RuntimeException("Unknown Elem Attr Kind: "+elem.getKind());
428            }
429        }
430        return result.toString();
431    }

433    public AbstractJasminClass(SootClass sootClass)
434    {
435        if(Options.v().time())
436            Timers.v().buildJasminTimer.start();
437        
438        if(Options.v().verbose())
439            G.v().out.println("[" + sootClass.getName() + "] Constructing baf.JasminClass...");
440
441        code = new LinkedList<String>();
442
443        // Emit the header
444        {
445            int modifiers = sootClass.getModifiers();
446
447            
448            if ((sootClass.getTag("SourceFileTag") != null) && (!Options.v().no_output_source_file_attribute())){
449                String srcName = ((SourceFileTag)sootClass.getTag("SourceFileTag")).getSourceFile();
450                // Since Jasmin fails on backslashes and only Windows uses backslashes,
451                // but also accepts forward slashes, we transform it.
452                if (File.separatorChar == '\\')
453                   srcName = srcName.replace('\\', '/');
454                srcName = soot.util.StringTools.getEscapedStringOf(srcName);
455                
456                // if 'srcName' starts with a digit, Jasmin throws an 
457                // 'Badly formatted number' error. When analyzing an Android 
458                // applications (.apk) their name is stored in srcName and 
459                // can start with a digit.
460                if (!Options.v().android_jars().isEmpty()
461                                && !srcName.isEmpty()
462                                && Character.isDigit(srcName.charAt(0))) 
463                    srcName = "n_"+ srcName;
464                
465                // Jasmin does not support blanks and quotes, so get rid of them
466                srcName = srcName.replace(" ", "-");
467                srcName = srcName.replace("\"", "");
468                
469                if (!srcName.isEmpty())
470                        emit(".source "+srcName);
471            }
472            if(Modifier.isInterface(modifiers))
473            {
474                modifiers -= Modifier.INTERFACE;
475
476                emit(".interface " + Modifier.toString(modifiers) + " " + slashify(sootClass.getName()));
477            }
478            else
479                emit(".class " + Modifier.toString(modifiers) + " " + slashify(sootClass.getName()));
480
481            if(sootClass.hasSuperclass())
482                emit(".super " + slashify(sootClass.getSuperclass().getName()));
483            else
484                emit(".no_super");
485
486            emit("");
487        }
488
489        // Emit the interfaces
490        {
491            Iterator<SootClass> interfaceIt = sootClass.getInterfaces().iterator();
492
493            while(interfaceIt.hasNext())
494            {
495                SootClass inter = interfaceIt.next();
496
497                emit(".implements " + slashify(inter.getName()));
498            }
499
500            /* why do this????
501            if(sootClass.getInterfaceCount() != 0)
502                emit("");*/
503        }
504
505
506
507
508
509    
510        // emit class attributes.
511        Iterator<Tag> it =  sootClass.getTags().iterator(); 
512        while(it.hasNext()) {
513            Tag tag = it.next();
514            if(tag instanceof Attribute)
515                emit(".class_attribute "  + tag.getName() + " \"" + new String(Base64.encode(((Attribute)tag).getValue()))+"\"");
516        /*else {
517            emit("");
518        }*/
519        }
520
521
522    // emit synthetic attributes
523    if (sootClass.hasTag("SyntheticTag") || Modifier.isSynthetic(sootClass.getModifiers())){
524        emit(".synthetic\n");
525    }
526    // emit inner class attributes
527    InnerClassAttribute ica = (InnerClassAttribute) sootClass.getTag("InnerClassAttribute");
528    if (ica != null && ica.getSpecs().size() > 0){
529        if (!Options.v().no_output_inner_classes_attribute()){
530            emit(".inner_class_attr ");
531            for (InnerClassTag ict : ((InnerClassAttribute)sootClass.getTag(
532                        "InnerClassAttribute")).getSpecs()) {
533                //System.out.println("inner class tag: "+ict);
534                emit(".inner_class_spec_attr "+
535                    "\""+ict.getInnerClass()+"\" "+
536                
537                    "\""+ict.getOuterClass()+"\" "+
538                
539                    "\""+ict.getShortName()+"\" "+
540                    Modifier.toString(ict.getAccessFlags())+" "+
541
542                ".end .inner_class_spec_attr");
543            }
544            emit(".end .inner_class_attr\n");
545        }
546    }
547    if (sootClass.hasTag("EnclosingMethodTag")){
548        String encMeth = ".enclosing_method_attr ";
549        EnclosingMethodTag eMethTag = (EnclosingMethodTag)sootClass.getTag("EnclosingMethodTag");
550        encMeth += "\""+eMethTag.getEnclosingClass()+"\" ";
551        encMeth += "\""+eMethTag.getEnclosingMethod()+"\" ";
552        encMeth += "\""+eMethTag.getEnclosingMethodSig()+"\"\n";
553        emit(encMeth);
554    }
555    // emit deprecated attributes
556    if (sootClass.hasTag("DeprecatedTag")){
557        emit(".deprecated\n");
558    }
559    if (sootClass.hasTag("SignatureTag")){
560        String sigAttr = ".signature_attr ";
561        SignatureTag sigTag = (SignatureTag)sootClass.getTag("SignatureTag");
562        sigAttr += "\""+sigTag.getSignature()+"\"\n";
563        emit(sigAttr);
564    }
565    
566    Iterator<Tag> vit = sootClass.getTags().iterator();
567    while (vit.hasNext()){
568        Tag t = (Tag)vit.next();
569        if (t.getName().equals("VisibilityAnnotationTag")){
570            emit(getVisibilityAnnotationAttr((VisibilityAnnotationTag)t));
571        }
572    }
573
574        // Emit the fields
575        {
576            Iterator<SootField> fieldIt = sootClass.getFields().iterator();
577
578            while(fieldIt.hasNext())
579            {
580                SootField field = (SootField) fieldIt.next();
581
582                String fieldString = ".field " + Modifier.toString(field.getModifiers()) + " " +  "\"" + field.getName() + "\"" + " " + jasminDescriptorOf(field.getType());
583    
584                if (field.hasTag("StringConstantValueTag")){
585                    fieldString += " = ";
586                    fieldString += soot.util.StringTools.getQuotedStringOf(((StringConstantValueTag)field.getTag("StringConstantValueTag")).getStringValue());
587                }
588                else if (field.hasTag("IntegerConstantValueTag")){
589                    fieldString += " = ";
590                    fieldString += ((IntegerConstantValueTag)field.getTag("IntegerConstantValueTag")).getIntValue();
591                }
592                else if (field.hasTag("LongConstantValueTag")){
593                    fieldString += " = ";
594                    fieldString += ((LongConstantValueTag)field.getTag("LongConstantValueTag")).getLongValue();
595                }
596                else if (field.hasTag("FloatConstantValueTag")){
597                    fieldString += " = ";
598                    FloatConstantValueTag val = (FloatConstantValueTag) field.getTag("FloatConstantValueTag");
599                    fieldString += floatToString(val.getFloatValue());
600                }
601                else if (field.hasTag("DoubleConstantValueTag")){
602                    fieldString += " = ";
603                    DoubleConstantValueTag val = (DoubleConstantValueTag) field.getTag("DoubleConstantValueTag");
604                    fieldString += doubleToString(val.getDoubleValue());
605                }
606                if (field.hasTag("SyntheticTag") || Modifier.isSynthetic(field.getModifiers())){
607                    fieldString +=" .synthetic";
608                }
609
610                fieldString +="\n";
611                if (field.hasTag("DeprecatedTag")){
612                    fieldString +=".deprecated\n";
613                }
614                if (field.hasTag("SignatureTag")){
615                    fieldString += ".signature_attr ";
616                    SignatureTag sigTag = (SignatureTag)field.getTag("SignatureTag");
617                    fieldString += "\""+sigTag.getSignature()+"\"\n";
618                }
619                Iterator<Tag> vfit = field.getTags().iterator();
620                while (vfit.hasNext()){
621                    Tag t = (Tag)vfit.next();
622                    if (t.getName().equals("VisibilityAnnotationTag")){
623                        fieldString += getVisibilityAnnotationAttr((VisibilityAnnotationTag)t);
624                    }
625                }
626
627                emit(fieldString);
628
629                Iterator<Tag> attributeIt =  field.getTags().iterator(); 
630                while(attributeIt.hasNext()) {
631                    Tag tag = (Tag) attributeIt.next();
632                    if(tag instanceof Attribute)
633                        emit(".field_attribute " + tag.getName() + " \"" + new String(Base64.encode(((Attribute)tag).getValue())) +"\"");
634                }
635
636            }
637
638            if(sootClass.getFieldCount() != 0)
639                emit("");
640        }
641
642        // Emit the methods
643        {
644            Iterator<SootMethod> methodIt = sootClass.methodIterator();
645
646            while(methodIt.hasNext())
647            {
648                emitMethod(methodIt.next());
649                emit("");
650            }
651        }
652        
653        if(Options.v().time())
654            Timers.v().buildJasminTimer.end();
655    }

657    protected void assignColorsToLocals(Body body)
658    {
659        if(Options.v().verbose())
660            G.v().out.println("[" + body.getMethod().getName() +
661                "] Assigning colors to locals...");
662        
663        if(Options.v().time())
664            Timers.v().packTimer.start();
665
666        localToGroup = new HashMap<Local, Object>(body.getLocalCount() * 2 + 1, 0.7f);
667        groupToColorCount = new HashMap<Object, Integer>(body.getLocalCount() * 2 + 1, 0.7f);
668        localToColor = new HashMap<Local, Integer>(body.getLocalCount() * 2 + 1, 0.7f);
669        
670        // Assign each local to a group, and set that group's color count to 0.
671        {
672            Iterator<Local> localIt = body.getLocals().iterator();
673
674            while(localIt.hasNext())
675            {
676                Local l = localIt.next();
677                Object g;
678                
679                if(sizeOfType(l.getType()) == 1)
680                    g = IntType.v();
681                else
682                    g = LongType.v();
683                
684                localToGroup.put(l, g);
685                
686                if(!groupToColorCount.containsKey(g))
687                {
688                    groupToColorCount.put(g, new Integer(0));
689                }
690            }
691        }
692
693        // Assign colors to the parameter locals.
694        {
695            Iterator<Unit> codeIt = body.getUnits().iterator();
696
697            while(codeIt.hasNext())
698            {
699                Stmt s = (Stmt) codeIt.next();
700
701                if(s instanceof IdentityStmt &&
702                    ((IdentityStmt) s).getLeftOp() instanceof Local)
703                {
704                    Local l = (Local) ((IdentityStmt) s).getLeftOp();
705                    
706                    Object group = localToGroup.get(l);
707                    int count = groupToColorCount.get(group).intValue();
708                    
709                    localToColor.put(l, new Integer(count));
710                    
711                    count++;
712                    
713                    groupToColorCount.put(group, new Integer(count));
714                }
715            }
716        }
717        
718    }

720    protected void emitMethod(SootMethod method)
721    {
722       if (method.isPhantom())
723           return;
724
725       // Emit prologue
726            emit(".method " + Modifier.toString(method.getModifiers()) + " " +
727                 method.getName() + jasminDescriptorOf(method.makeRef()));
728
729            Iterator<SootClass> throwsIt = method.getExceptions().iterator();
730            while (throwsIt.hasNext()){
731                SootClass exceptClass = throwsIt.next();
732                emit(".throws "+exceptClass.getName());
733            }
734            if (method.hasTag("SyntheticTag") || Modifier.isSynthetic(method.getModifiers())){
735                emit(".synthetic");
736            }
737            if (method.hasTag("DeprecatedTag")){
738                emit(".deprecated");
739            }
740            if (method.hasTag("SignatureTag")){
741                String sigAttr = ".signature_attr ";
742                SignatureTag sigTag = (SignatureTag)method.getTag("SignatureTag");
743                sigAttr += "\""+sigTag.getSignature()+"\"";
744                emit(sigAttr);
745            }
746            if (method.hasTag("AnnotationDefaultTag")){
747                String annotDefAttr = ".annotation_default ";
748                AnnotationDefaultTag annotDefTag = (AnnotationDefaultTag)method.getTag("AnnotationDefaultTag");
749                annotDefAttr += getElemAttr(annotDefTag.getDefaultVal());
750                annotDefAttr += ".end .annotation_default";
751                emit(annotDefAttr);
752            }
753            Iterator<Tag> vit = method.getTags().iterator();
754            while (vit.hasNext()){
755                Tag t = (Tag)vit.next();
756                if (t.getName().equals("VisibilityAnnotationTag")){
757                    emit(getVisibilityAnnotationAttr((VisibilityAnnotationTag)t));
758                }
759                if (t.getName().equals("VisibilityParameterAnnotationTag")){
760                    emit(getVisibilityParameterAnnotationAttr((VisibilityParameterAnnotationTag)t));
761                }
762            }
763
764       if(method.isConcrete())
765       {
766            if(!method.hasActiveBody())
767                throw new RuntimeException("method: " + method.getName() + " has no active body!");
768            else
769                emitMethodBody(method);
770       }
771       
772       // Emit epilogue
773            emit(".end method");
774
775            Iterator<Tag> it =  method.getTags().iterator();
776            while(it.hasNext()) {
777                Tag tag = (Tag) it.next();
778                if(tag instanceof Attribute)
779                    emit(".method_attribute "  + tag.getName() + " \"" + new String(Base64.encode(tag.getValue())) +"\"");
780            }       
781    }
