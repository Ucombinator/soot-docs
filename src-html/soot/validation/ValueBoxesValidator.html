001package soot.validation;
002
003import java.util.IdentityHashMap;
004import java.util.List;
005import java.util.Set;
006
007import soot.Body;
008import soot.Unit;
009import soot.ValueBox;
010
011import static java.util.Collections.newSetFromMap;
012
013public enum ValueBoxesValidator implements BodyValidator {
014        INSTANCE;
015        
016        public static ValueBoxesValidator v() {
017                return INSTANCE;
018        }
019
020
021        @Override
022        /** Verifies that a ValueBox is not used in more than one place. */
023        public void validate(Body body, List<ValidationException> exception) {
024                Set<ValueBox> set = newSetFromMap(new IdentityHashMap<ValueBox, Boolean>());
025                
026                for (ValueBox vb: body.getUseAndDefBoxes()) {
027                        if (set.add(vb))
028                                continue;                       
029                        
030                        exception.add(new ValidationException(vb, "Aliased value box : "+vb+" in "+body.getMethod()));
031            
032                        for (Unit u : body.getUnits()) {
033                System.err.println(u);
034            }
035                }
036    }
037
038        @Override
039        public boolean isBasicValidator() {
040                return false;
041        }
042}
