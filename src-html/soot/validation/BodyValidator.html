001package soot.validation;
002
003import java.util.List;
004
005import soot.Body;
006
007/**
008 * Implement this interface if you want to provide your own body Validator
009 */
010public interface BodyValidator {
011        /**
012         * Validates the given body and saves all validation errors in the given list.
013         * @param body the body to check
014         * @param exceptions the list of exceptions
015         */
016        public void validate(Body body, List<ValidationException> exceptions);
017        
018        /**
019         * Basic validators run essential checks and are run always if validate is called.<br>
020         * If this method returns false and the caller of the validator respects this property,<br>
021         * the checks will only be run if the debug or validation option is activated. 
022         * @return whether this validator is a basic validator
023         */
024        public boolean isBasicValidator();
025}
