001package soot.validation;
002
003import java.util.List;
004
005import soot.Body;
006import soot.SootMethod;
007import soot.Unit;
008import soot.jimple.InvokeExpr;
009import soot.jimple.Stmt;
010
011public enum CheckEscapingValidator implements BodyValidator {
012        INSTANCE;
013        
014        public static CheckEscapingValidator v() {
015                return INSTANCE;
016        }
017        
018        @Override
019        public void validate(Body body, List<ValidationException> exception) {
020                for (Unit u : body.getUnits()) {
021                        if (u instanceof Stmt) {
022                                Stmt stmt = (Stmt) u;
023                                if (stmt.containsInvokeExpr()) {
024                                        InvokeExpr iexpr = stmt.getInvokeExpr();
025                                        SootMethod sm = iexpr.getMethod();
026                                        if (sm.getName().contains("'")
027                                                        || sm.getDeclaringClass().getName().contains("'"))
028                                                throw new ValidationException(stmt, "Escaped name in signature found");
029                                        for (int i = 0; i < sm.getParameterCount(); i++)
030                                                if (sm.getParameterType(i).toString().contains("'"))
031                                                        throw new ValidationException(stmt, "Escaped name in signature found");
032                                }
033                        }
034                }
035        }
036        
037        @Override
038        public boolean isBasicValidator() {
039                return false;
040        }
041
042}
