119    /*
120     *  Computes a map binding each unit in a method to the unique basic block    
121     *  that contains it.
122     */    
123    private  void buildUnitToBlockMap()
124    {
125        BlockGraph blockGraph = new ZonedBlockGraph(mBody); 
126        if(debug) {
127            G.v().out.println("Method " +  mBody.getMethod().getName()+ " Block Graph: ");
128            G.v().out.println(blockGraph);
129        }
130       
131        List<Block> blocks = blockGraph.getBlocks();
132        mUnitToBlockMap = new HashMap<Unit, Block>();
133        
134        Iterator<Block> blockIt = blocks.iterator();
135        while(blockIt.hasNext() ) {
136            Block block = blockIt.next();
137            
138            Iterator<Unit> unitIt = block.iterator();
139            while(unitIt.hasNext()) {
140                Unit unit = unitIt.next();                
141                mUnitToBlockMap.put(unit, block);                
142            }
143        }
144    }

366    /** 
367     *  Checks if the units occurring between [from, to] consume 
368     *. stack items not produced by these interval units. (ie if the
369     *  stack height ever goes negative between from and to, assuming the 
370     *  stack height is set to 0 upon executing the instruction following 'from'.
371     *  
372     */
373    private boolean isRequiredByFollowingUnits(Unit from, Unit to)
374    {
375        Iterator<Unit> it = mUnits.iterator(from, to);
376        int stackHeight = 0;
377        boolean res = false;
378        
379        if(from != to)  {
380            // advance past the 'from' unit
381            it.next();
382            while(it.hasNext()) {
383                Unit  currentInst = it.next();
384                if(currentInst == to)                     
385                    break;
386
387                stackHeight -= ((Inst)currentInst).getInCount();
388                if(stackHeight < 0 ) {
389                    res = true;
390                    break;
391                }
392                stackHeight += ((Inst)currentInst).getOutCount();
393            }
394        }
395        return res;
396    }

402    private int pushStoreToLoad(Unit from , Unit to, Block block)
403    {
404        Unit storePred =  block.getPredOf(from);
405        if(storePred != null) {
406            if(((Inst)storePred).getOutCount() == 1){
407                Set<Unit> unitsToMove = new HashSet<Unit>();
408                unitsToMove.add(storePred);
409                unitsToMove.add(from);
410                int h = ((Inst)storePred).getInCount();
411                Unit currentUnit = storePred;
412                if(h != 0) {           
413                    currentUnit = block.getPredOf(storePred);
414                    while(currentUnit != null) {
415                        
416                        h-= ((Inst)currentUnit).getOutCount();
417                        if(h<0){ // xxx could be more flexible here?
418                            if(debug) { G.v().out.println("xxx: negative");}
419                            return FAILURE;
420                        }
421                        h+= ((Inst)currentUnit).getInCount();
422                        unitsToMove.add(currentUnit);
423                        if(h == 0)
424                            break;
425                        currentUnit  = block.getPredOf(currentUnit);
426                    }
427                }
428                if(currentUnit == null) {
429                    if(debug) { G.v().out.println("xxx: null");}
430                    return FAILURE;        
431                }
432                
433                Unit uu = from;
434                for(;;) {
435                    uu = block.getSuccOf(uu);
436                    if(uu == to)
437                        break;
438                    Iterator<Unit> it2 = unitsToMove.iterator();
439                    while(it2.hasNext()) {
440                        Unit nu = it2.next();
441                        if(debug) { G.v().out.println("xxxspecial;success pushing forward stuff.");}
442                        
443                        
444                        if(!canMoveUnitOver(nu, uu)){
445                            if(debug) { G.v().out.println("xxx: cant move over faillure" + nu);}
446                            return FAILURE;
447                        }
448                        if(debug) { G.v().out.println("can move" + nu + " over " + uu);}
449                    }
450                }        
451                
452                // if we get here it means we can move all the units in the set pass the units in between [to, from]
453                Unit unitToMove = currentUnit; 
454                while(unitToMove != from) {        
455                    Unit succ = block.getSuccOf(unitToMove);
456                    if(debug) { G.v().out.println("moving " + unitToMove);}
457                    block.remove(unitToMove);
458                    block.insertBefore(unitToMove, to);                            
459                    unitToMove = succ;
460                }
461                block.remove(from);
462                block.insertBefore(from, to);
463                               
464                if(debug) { G.v().out.println("xxx1success pushing forward stuff.");}
465                return SPECIAL_SUCCESS;
466            }
467        }
468
469        return FAILURE;
470    }

475    /**
476     *
477     *
478     * @return FAILURE when store load elimination is not possible in any form.
479     * @return SUCCESS when a load in a store load pair can be moved IMMEDIATELY after it's defining store
480     * @return MAKE_DUP when a store/load/load trio can be replaced by a dup unit.
481     * @return MAKE_DUP_X1 when store/load/load trio can be replaced by a dup1_x1 unit
482     * @return SPECIAL_SUCCESS when a store/load pair can AND must be immediately annihilated. 
483     * @return HAS_CHANGED when store load elimination is not possible in any form, but some unit reordering has occured
484     */
485    
486    private  int stackIndependent(Unit from, Unit to, Block block, int aContext) 
487    {                
488        if(debug) { 
489            G.v().out.println("Trying: " + from + "/" + to +  " in block  " + block.getIndexInMethod()+":" );
490            G.v().out.println("context:" + (aContext == STORE_LOAD_ELIMINATION ? 
491                                             "STORE_LOAD_ELIMINATION" :
492                                             "STORE_LOAD_LOAD_ELIMINATION"));
493        }
494
495
496        int minStackHeightAttained = 0; // records the min stack height attained between [from, to]
497        int stackHeight = 0;           // records the stack height when similating the effects on the stack
498        Iterator<Unit> it = mUnits.iterator(mUnits.getSuccOf(from)); 
499        int res = FAILURE;
500        
501        Unit currentInst = it.next();   // get unit following the store
502        
503        if(aContext == STORE_LOAD_LOAD_ELIMINATION) {
504            currentInst = it.next(); // jump after first load 
505        } 
506        
507        // find minStackHeightAttained
508        while(currentInst != to && it.hasNext()) {
509            stackHeight -= ((Inst)currentInst).getInCount();
510            if(stackHeight < minStackHeightAttained)
511                minStackHeightAttained = stackHeight;
512            
513            
514            stackHeight += ((Inst)currentInst).getOutCount();                
515            
516            currentInst = it.next();
517        }
518                
519        // note: now stackHeight contains the delta height of the stack after executing the units contained in
520        // [from, to] non-inclusive.
521        
522        
523        if(debug) { 
524            G.v().out.println("nshv = " + stackHeight);
525            G.v().out.println("mshv = " + minStackHeightAttained);
526        }
527        
528        
529        boolean hasChanged = true;        
530        
531        // Iterate until an elimination clause is taken or no reordering of the code occurs         
532        while(hasChanged) {
533            hasChanged = false;
534
535            // check for possible sll elimination
536            if(aContext == STORE_LOAD_LOAD_ELIMINATION) {
537                                
538                if(stackHeight == 0 && minStackHeightAttained == 0){
539                    if(debug) { 
540                        G.v().out.println("xxx: succ: -1, makedup ");
541                    }
542                    return MAKE_DUP;
543                }
544                else if(stackHeight == -1 && minStackHeightAttained == -1){
545                    if(debug) { G.v().out.println("xxx: succ: -1, makedup , -1");}        
546                    return MAKE_DUP;
547                 }
548                else if(stackHeight == -2 && minStackHeightAttained == -2){if(debug) { G.v().out.println("xxx: succ -1 , make dupx1 ");}
549                return MAKE_DUP1_X1; }
550                
551                else  if (minStackHeightAttained < -2) {
552                    if(debug) { G.v().out.println("xxx: failled due: minStackHeightAttained < -2 ");}
553                    return FAILURE;
554                }                
555            }
556            
557            
558            // check for possible sl elimination
559            if(aContext == STORE_LOAD_ELIMINATION) {                
560                if(stackHeight == 0 && minStackHeightAttained == 0){
561                    if(debug) { G.v().out.println("xxx: success due: 0, SUCCESS ");}
562                    return SUCCESS;
563                }
564                /* xxx broken data depensie problem.
565                  else if (minStackHeightAttained == -1 && stackHeight == -1) { // try to make it more generic
566                    Unit u = (Unit) block.getPredOf(from);
567                    if(u instanceof FieldGetInst)
568                        if(block.getPredOf(u) instanceof Dup1Inst) {
569                            block.remove(u);
570                            block.insertBefore(u, to);
571                            block.remove(from);
572                            block.insertBefore(from, to);
573                            if(debug) { G.v().out.println("xxx: success due to 1, SPECIAL_SUCCESS2");}
574                            return SPECIAL_SUCCESS2;
575                        }                    
576                        }*/
577                else if (minStackHeightAttained < 0){
578                    return   pushStoreToLoad(from , to, block);
579                }                  
580            }        
581            
582
583            it = mUnits.iterator(mUnits.getSuccOf(from), to);
584            
585            Unit unitToMove = null; 
586            Unit u = it.next();
587
588            if(aContext == STORE_LOAD_LOAD_ELIMINATION) {
589                u = it.next();
590            } 
591            int currentH = 0;
592            
593            // find a candidate to move before the store/load/(load)  group 
594            while( u != to) {        
595                
596                if(((Inst)u).getNetCount() == 1) {
597                    // xxx remove this check
598                    if(u instanceof LoadInst || u instanceof PushInst || u instanceof NewInst || u instanceof StaticGetInst || u instanceof Dup1Inst) {
599                        
600                        // verify that unitToMove is not required by following units (until the 'to' unit)
601                        if(!isRequiredByFollowingUnits(u, to)) {
602                            unitToMove = u;
603                        }
604                        
605                    }
606                    else{
607                        if(debug) { G.v().out.println("1003:(LoadStoreOptimizer@stackIndependent): found unknown unit w/ getNetCount == 1" + u);}
608                    }
609                }
610                
611
612
613           
614                
615                if(unitToMove != null) {
616                    int flag = 0;
617                    //if(aContext == 0 || !(stackHeight > -2 && minStackHeightAttained == -2 ) ) {
618                        
619                    if(tryToMoveUnit(unitToMove, block, from, to, flag)) {
620                        if(stackHeight > -2 && minStackHeightAttained == -2){
621                            return HAS_CHANGED;
622                        }
623                        
624                        stackHeight--;
625                        if(stackHeight < minStackHeightAttained)
626                            minStackHeightAttained = stackHeight;
627                        hasChanged = true;
628                        break;
629                    }
630                }
631                
632                currentH += ((Inst)u).getNetCount();                
633                unitToMove = null;
634                u =  it.next();
635            }
636        }
637
638        if(isCommutativeBinOp(block.getSuccOf(to))) {
639            if(aContext == STORE_LOAD_ELIMINATION && stackHeight == 1 && minStackHeightAttained == 0) {
640                if(debug) { G.v().out.println("xxx: commutative ");}
641                return SPECIAL_SUCCESS;
642            }
643            else if( ((Inst) to).getOutCount()  == 1 &&
644                     ((Inst) to).getInCount() == 0  &&
645                     ((Inst) mUnits.getPredOf(to)).getOutCount() == 1 &&
646                     ((Inst) mUnits.getPredOf(to)).getInCount() == 0) {
647                
648                Object toPred =  mUnits.getPredOf(to);
649                block.remove((Unit) toPred);
650                block.insertAfter((Unit) toPred, to);
651                return HAS_CHANGED; // return has changed
652            } 
653            else return FAILURE;
654        }
655        if (aContext == STORE_LOAD_ELIMINATION)
656            return   pushStoreToLoad(from , to, block);
657        
658        return res;
659    }

769    {
770       
771        int h = 0;
772        Unit current = unitToMove;
773        boolean reachedStore = false;
774        boolean reorderingOccurred =false;
775              
776        if(debug) {G.v().out.println("[tryToMoveUnit]: trying to move:" + unitToMove);}
777        if(unitToMove == null) 
778            return false;                
779                
780        while( current != block.getHead()) {   // do not go past basic block limit
781            current = (Unit) mUnits.getPredOf(current);
782            
783            if(!canMoveUnitOver(current, unitToMove))
784                return false;
785
786            if(current == from)
787                reachedStore = true;
788                
789        
790            h -= ((Inst)current).getOutCount();                
791            if(h < 0 ){
792                if(debug) { G.v().out.println("1006:(LoadStoreOptimizer@stackIndependent): Stack went negative while trying to reorder code.");}
793
794
795
796                if(flag == 1) {
797                    if(current instanceof DupInst) {
798                        block.remove(unitToMove);
799                        block.insertAfter(unitToMove, current);        
800                        //                        block.insertAfter(  new BSwapInst(  )     ,unitToMove);
801                    }
802                    
803                }
804                return false;
805            }
806            h += ((Inst)current).getInCount();
807            
808            
809            if(h == 0 && reachedStore == true) {
810                if(!isRequiredByFollowingUnits(unitToMove, to)) {
811                    if(debug) { G.v().out.println("10077:(LoadStoreOptimizer@stackIndependent): reordering bytecode move: " + unitToMove + "before: " + current);}
812                    block.remove(unitToMove);
813                    block.insertBefore(unitToMove, current);
814                    
815                    reorderingOccurred = true;
816                    break;
817                }                    
818            }                                      
819        }
820            
821        if(reorderingOccurred) {
822            if(debug) { G.v().out.println("reordering occured");}
823            return true;
824        } else {
825            if(debug) { G.v().out.println("1008:(LoadStoreOptimizer@stackIndependent):failled to find a new slot for unit to move");}
826            return false;
827        }

835     *  replace should be in the same block. The map 'mUnitToBlockMap'   
836     *  is updated. The replacement unit is inserted in the position,
837     *  of the aToReplace2 if not null, otherwise in aToReplace1's slot.
838     *
839     *  @param aToReplace1 Unit to replace. (shouldn't be null)
840     *  @param aToReplace2 Second Unit to replace (can be  null)
841     *  @param aReplacement Unit that replaces the 2 previous units (shouldn't be null)

846                 
847        if(aToReplace2 != null) {

849            block.remove(aToReplace2);
850        } else {
851            block.insertAfter(aReplacement, aToReplace1);            
852        }
853
854        block.remove(aToReplace1);
855                                                        
856        // add the new unit the block map
857        mUnitToBlockMap.put(aReplacement, block);        
858    }
