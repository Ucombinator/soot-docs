101        @Override
102        protected void emitMethodBody(SootMethod method) {
103                if (Options.v().time())
104                        Timers.v().buildJasminTimer.end();
105
106                Body activeBody = method.getActiveBody();
107
108                if (!(activeBody instanceof BafBody)) {
109                        if (activeBody instanceof JimpleBody) {
110                                if (Options.v().verbose()) {
111                                        G.v().out
112                                                        .println("Was expecting Baf body for "
113                                                                        + method
114                                                                        + " but found a Jimple body. Will convert body to Baf on the fly.");
115                                }
116                                activeBody = PackManager.v().convertJimpleBodyToBaf(method);
117                        } else
118                                throw new RuntimeException("method: " + method.getName()
119                                                + " has an invalid active body!");
120                }
121
122                BafBody body = (BafBody) activeBody;
123
124                if (body == null)
125                        throw new RuntimeException("method: " + method.getName()
126                                        + " has no active body!");
127
128                if (Options.v().time())
129                        Timers.v().buildJasminTimer.start();
130
131                Chain<Unit> instList = body.getUnits();
132
133                int stackLimitIndex = -1;
134
135                subroutineToReturnAddressSlot = new HashMap<Unit, Integer>(10, 0.7f);
136
137                // Determine the unitToLabel map
138                {
139                        unitToLabel = new HashMap<Unit, String>(instList.size() * 2 + 1,
140                                        0.7f);
141                        labelCount = 0;
142
143                        for (UnitBox uBox : body.getUnitBoxes(true)) {
144                                // Assign a label for each statement reference
145                                {
146                                        InstBox box = (InstBox) uBox;
147
148                                        if (!unitToLabel.containsKey(box.getUnit()))
149                                                unitToLabel.put(box.getUnit(), "label" + labelCount++);
150                                }
151                        }
152                }
153
154                // Emit the exceptions, recording the Units at the beginning
155                // of handlers so that later on we can recognize blocks that
156                // begin with an exception on the stack.
157                Set<Unit> handlerUnits = new ArraySet<Unit>(body.getTraps().size());
158                {
159                        for (Trap trap : body.getTraps()) {
160                                handlerUnits.add(trap.getHandlerUnit());
161                                if (trap.getBeginUnit() != trap.getEndUnit()) {
162                                        emit(".catch " + slashify(trap.getException().getName())
163                                                        + " from " + unitToLabel.get(trap.getBeginUnit())
164                                                        + " to " + unitToLabel.get(trap.getEndUnit())
165                                                        + " using "
166                                                        + unitToLabel.get(trap.getHandlerUnit()));
167                                }
168                        }
169                }
170
171                // Determine where the locals go
172                {
173                        int localCount = 0;
174                        int[] paramSlots = new int[method.getParameterCount()];
175                        int thisSlot = 0;
176                        Set<Local> assignedLocals = new HashSet<Local>();
177
178                        localToSlot = new HashMap<Local, Integer>(
179                                        body.getLocalCount() * 2 + 1, 0.7f);
180
181                        // assignColorsToLocals(body);
182
183                        // Determine slots for 'this' and parameters
184                        {
185                                if (!method.isStatic()) {
186                                        thisSlot = 0;
187                                        localCount++;
188                                }
189
190                                for (int i = 0; i < method.getParameterCount(); i++) {
191                                        paramSlots[i] = localCount;
192                                        localCount += sizeOfType(method.getParameterType(i));
193                                }
194                        }
195
196                        // Handle identity statements
197                        {
198                                for (Unit u : instList) {
199                                        Inst s = (Inst) u;
200                                        if (s instanceof IdentityInst
201                                                        && ((IdentityInst) s).getLeftOp() instanceof Local) {
202                                                Local l = (Local) ((IdentityInst) s).getLeftOp();
203                                                IdentityRef identity = (IdentityRef) ((IdentityInst) s)
204                                                                .getRightOp();
205
206                                                int slot = 0;
207
208                                                if (identity instanceof ThisRef) {
209                                                        if (method.isStatic())
210                                                                throw new RuntimeException(
211                                                                                "Attempting to use 'this' in static method");
212
213                                                        slot = thisSlot;
214                                                } else if (identity instanceof ParameterRef)
215                                                        slot = paramSlots[((ParameterRef) identity)
216                                                                        .getIndex()];
217                                                else {
218                                                        // Exception ref. Skip over this
219                                                        continue;
220                                                }
221
222                                                localToSlot.put(l, new Integer(slot));
223                                                assignedLocals.add(l);
224
225                                        }
226                                }
227                        }
228
229                        // Assign the rest of the locals
230                        {
231                                for (Local local : body.getLocals()) {
232                                        if (assignedLocals.add(local)) {
233                                                localToSlot.put(local, new Integer(localCount));
234                                                localCount += sizeOfType(local.getType());
235                                        }
236                                }
237
238                                if (!Modifier.isNative(method.getModifiers())
239                                                && !Modifier.isAbstract(method.getModifiers())) {
240                                        emit("    .limit stack ?");
241                                        stackLimitIndex = code.size() - 1;
242
243                                        emit("    .limit locals " + localCount);
244                                }
245                        }
246                }
247
248                // Emit code in one pass
249                {
250                        isEmittingMethodCode = true;
251                        maxStackHeight = 0;
252                        isNextGotoAJsr = false;
253
254                        for (Unit u : instList) {
255                                Inst s = (Inst) u;
256
257                                if (unitToLabel.containsKey(s))
258                                        emit(unitToLabel.get(s) + ":");
259
260                                // emit this statement
261                                {
262                                        emitInst(s);
263                                }
264                        }
265
266                        isEmittingMethodCode = false;
267
268                        // calculate max stack height
269                        {
270                                maxStackHeight = 0;
271                                if (activeBody.getUnits().size() != 0) {
272                                        BlockGraph blockGraph = new BriefBlockGraph(activeBody);
273                                        List<Block> blocks = blockGraph.getBlocks();
274
275                                        if (blocks.size() != 0) {
276                                                // set the stack height of the entry points
277                                                List<Block> entryPoints = ((DirectedGraph<Block>) blockGraph)
278                                                                .getHeads();
279                                                for (Block entryBlock : entryPoints) {
280                                                        Integer initialHeight;
281                                                        if (handlerUnits.contains(entryBlock.getHead())) {
282                                                                initialHeight = new Integer(1);
283                                                        } else {
284                                                                initialHeight = new Integer(0);
285                                                        }
286                                                        if (blockToStackHeight == null) {
287                                                                blockToStackHeight = new HashMap<Block, Integer>();
288                                                        }
289                                                        blockToStackHeight.put(entryBlock, initialHeight);
290                                                        if (blockToLogicalStackHeight == null) {
291                                                                blockToLogicalStackHeight = new HashMap<Block, Integer>();
292                                                        }
293                                                        blockToLogicalStackHeight.put(entryBlock,
294                                                                        initialHeight);
295                                                }
296
297                                                // dfs the block graph using the blocks in the
298                                                // entryPoints list as roots
299                                                for (Block nextBlock : entryPoints) {
300                                                        calculateStackHeight(nextBlock);
301                                                        calculateLogicalStackHeightCheck(nextBlock);
302                                                }
303                                        }
304                                }
305                        }
306
307                        if (!Modifier.isNative(method.getModifiers())
308                                        && !Modifier.isAbstract(method.getModifiers()))
309                                code.set(stackLimitIndex, "    .limit stack " + maxStackHeight);
310                }
311
312                // emit code attributes
313                {
314                        for (Tag t : body.getTags()) {
315                                if (t instanceof JasminAttribute) {
316                                        emit(".code_attribute " + t.getName() + " \""
317                                                        + ((JasminAttribute) t).getJasminValue(unitToLabel)
318                                                        + "\"");
319                                }
320                        }
321                }
322        }

434                        @Override
435                        public void casePushInst(PushInst i) {
436                                if (i.getConstant() instanceof IntConstant) {
437                                        IntConstant v = (IntConstant) (i.getConstant());
438                                        if (v.value == -1)
439                                                emit("iconst_m1");
440                                        else if (v.value >= 0 && v.value <= 5)
441                                                emit("iconst_" + v.value);
442                                        else if (v.value >= Byte.MIN_VALUE
443                                                        && v.value <= Byte.MAX_VALUE)
444                                                emit("bipush " + v.value);
445                                        else if (v.value >= Short.MIN_VALUE
446                                                        && v.value <= Short.MAX_VALUE)
447                                                emit("sipush " + v.value);
448                                        else
449                                                emit("ldc " + v.toString());
450                                } else if (i.getConstant() instanceof StringConstant) {
451                                        emit("ldc " + i.getConstant().toString());
452                                } else if (i.getConstant() instanceof ClassConstant) {
453                                        emit("ldc_w "
454                                                        + ((ClassConstant) i.getConstant()).getValue());
455                                } else if (i.getConstant() instanceof DoubleConstant) {
456                                        DoubleConstant v = (DoubleConstant) (i.getConstant());
457
458                                        if ((v.value == 0) && ((1.0 / v.value) > 0.0))
459                                                emit("dconst_0");
460                                        else if (v.value == 1)
461                                                emit("dconst_1");
462                                        else {
463                                                String s = doubleToString(v);
464                                                emit("ldc2_w " + s);
465                                        }
466                                } else if (i.getConstant() instanceof FloatConstant) {
467                                        FloatConstant v = (FloatConstant) (i.getConstant());
468                                        if ((v.value == 0) && ((1.0f / v.value) > 1.0f))
469                                                emit("fconst_0");
470                                        else if (v.value == 1)
471                                                emit("fconst_1");
472                                        else if (v.value == 2)
473                                                emit("fconst_2");
474                                        else {
475                                                String s = floatToString(v);
476                                                emit("ldc " + s);
477                                        }
478                                } else if (i.getConstant() instanceof LongConstant) {
479                                        LongConstant v = (LongConstant) (i.getConstant());
480                                        if (v.value == 0)
481                                                emit("lconst_0");
482                                        else if (v.value == 1)
483                                                emit("lconst_1");
484                                        else
485                                                emit("ldc2_w " + v.toString());
486                                } else if (i.getConstant() instanceof NullConstant)
487                                        emit("aconst_null");
488                                else
489                                        throw new RuntimeException("unsupported opcode");
490                        }

1782        private void calculateStackHeight(Block aBlock) {
1783                int blockHeight = blockToStackHeight.get(aBlock).intValue();
1784                if (blockHeight > maxStackHeight) {
1785                        maxStackHeight = blockHeight;
1786                }
1787                
1788                for (Unit u : aBlock) {
1789                        Inst nInst = (Inst) u;
1790
1791                        blockHeight -= nInst.getInMachineCount();
1792
1793                        if (blockHeight < 0) {
1794                                throw new RuntimeException(
1795                                                "Negative Stack height has been attained in :"
1796                                                                + aBlock.getBody().getMethod().getSignature()
1797                                                                + " \n" + "StackHeight: " + blockHeight + "\n"
1798                                                                + "At instruction:" + nInst + "\n" + "Block:\n"
1799                                                                + aBlock + "\n\nMethod: "
1800                                                                + aBlock.getBody().getMethod().getName() + "\n"
1801                                                                + aBlock.getBody().getMethod());
1802                        }
1803
1804                        blockHeight += nInst.getOutMachineCount();
1805                        if (blockHeight > maxStackHeight) {
1806                                maxStackHeight = blockHeight;
1807                        }
1808                        // G.v().out.println(">>> " + nInst + " " + blockHeight);
1809                }
1810
1811                for (Block b : aBlock.getSuccs()) {
1812                        Integer i = blockToStackHeight.get(b);
1813                        if (i != null) {
1814                                if (i.intValue() != blockHeight) {
1815                                        throw new RuntimeException(aBlock.getBody().getMethod()
1816                                                        .getSignature()
1817                                                        + ": incoherent stack height at block merge point "
1818                                                        + b
1819                                                        + aBlock
1820                                                        + "\ncomputed blockHeight == "
1821                                                        + blockHeight
1822                                                        + " recorded blockHeight = "
1823                                                        + i.intValue());
1824                                }
1825
1826                        } else {
1827                                blockToStackHeight.put(b, new Integer(blockHeight));
1828                                calculateStackHeight(b);
1829                        }
1830                }
1831        }
