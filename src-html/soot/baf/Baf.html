487    public static String bafDescriptorOf(Type type)
488    {
489        TypeSwitch sw;
490
491        type.apply(sw = new TypeSwitch()
492        {
493            public void caseBooleanType(BooleanType t)
494            {
495                setResult("b");
496            }
497
498            public void caseByteType(ByteType t)
499            {
500                setResult("b");
501            }
502
503            public void caseCharType(CharType t)
504            {
505                setResult("c");
506            }
507
508            public void caseDoubleType(DoubleType t)
509            {
510                setResult("d");
511            }
512
513            public void caseFloatType(FloatType t)
514            {
515                setResult("f");
516            }
517
518            public void caseIntType(IntType t)
519            {
520                setResult("i");
521            }
522
523            public void caseLongType(LongType t)
524            {
525                setResult("l");
526            }
527
528            public void caseShortType(ShortType t)
529            {
530                setResult("s");
531            }
532
533            
534            public void defaultCase(Type t)
535            {
536                throw new RuntimeException("Invalid type: " + t);
537            }
538
539            public void caseRefType(RefType t)
540            {
541                setResult("r");
542            }
543
544
545        });
546
547        return (String) sw.getResult();
548    }
