093        /* (non-Javadoc)
094         * @see soot.AbstractASMBackend#getMinJavaVersion(soot.SootMethod)
095         */
096        @Override
097        protected int getMinJavaVersion(SootMethod method) {
098                final BafBody body = getBafBody(method);
099                int minVersion = Options.java_version_1_1;
100
101                for (Unit u : body.getUnits()) {
102                        if (u instanceof DynamicInvokeInst) {
103                                return Options.java_version_1_7;
104                        }
105                        if (u instanceof PushInst) {
106                                if (((PushInst) u).getConstant() instanceof ClassConstant) {
107                                        minVersion = Options.java_version_1_5;
108                                }
109                        }
110                }
111
112                return minVersion;
113        }

115        /* (non-Javadoc)
116         * @see soot.AbstractASMBackend#generateMethodBody(org.objectweb.asm.MethodVisitor, soot.SootMethod)
117         */
118        @Override
119        protected void generateMethodBody(MethodVisitor mv, SootMethod method) {
120                BafBody body = getBafBody(method);
121                Chain<Unit> instructions = body.getUnits();
122
123                /*
124                 * Create a label for each instruction that is the target of some branch
125                 */
126                for (UnitBox box : body.getUnitBoxes(true)) {
127                        Unit u = box.getUnit();
128                        if (!branchTargetLabels.containsKey(u)) {
129                                branchTargetLabels.put(u, new Label());
130                        }
131                }
132
133                Label startLabel = null;
134                if (Options.v().write_local_annotations()) {
135                        startLabel = new Label();
136                        mv.visitLabel(startLabel);
137                }
138                
139                /*
140                 * Handle all TRY-CATCH-blocks
141                 */
142                for (Trap trap : body.getTraps()) {
143                        // Check if the try-block contains any statement
144                        if (trap.getBeginUnit() != trap.getEndUnit()) {
145                                Label start = branchTargetLabels.get(trap.getBeginUnit());
146                                Label end = branchTargetLabels.get(trap.getEndUnit());
147                                Label handler = branchTargetLabels.get(trap.getHandlerUnit());
148                                String type = slashify(trap.getException().getName());
149                                mv.visitTryCatchBlock(start, end, handler, type);
150                        }
151                }
152
153                /*
154                 * Handle local variable slots for the "this"-local and the parameters
155                 */
156                int localCount = 0;
157                int[] paramSlots = new int[method.getParameterCount()];
158                Set<Local> assignedLocals = new HashSet<Local>();
159
160                /*
161                 * For non-static methods the first parameters and zero-slot is the
162                 * "this"-local
163                 */
164                if (!method.isStatic()) {
165                        ++localCount;
166                }
167
168                for (int i = 0; i < method.getParameterCount(); ++i) {
169                        paramSlots[i] = localCount;
170                        localCount += sizeOfType(method.getParameterType(i));
171                }
172
173                for (Unit u : instructions) {
174                        if (u instanceof IdentityInst
175                                        && ((IdentityInst) u).getLeftOp() instanceof Local) {
176                                Local l = (Local) ((IdentityInst) u).getLeftOp();
177                                IdentityRef identity = (IdentityRef) ((IdentityInst) u)
178                                                .getRightOp();
179
180                                int slot = 0;
181
182                                if (identity instanceof ThisRef) {
183                                        if (method.isStatic())
184                                                throw new RuntimeException(
185                                                                "Attempting to use 'this' in static method");
186                                } else if (identity instanceof ParameterRef)
187                                        slot = paramSlots[((ParameterRef) identity).getIndex()];
188                                else {
189                                        // Exception ref. Skip over this
190                                        continue;
191                                }
192                                localToSlot.put(l, slot);
193                                assignedLocals.add(l);
194                        }
195                }
196
197                for (Local local : body.getLocals()) {
198                        if (assignedLocals.add(local)) {
199                                localToSlot.put(local, localCount);
200                                localCount += sizeOfType(local.getType());
201                        }
202                }
203
204                
205                // Generate the code
206                for (Unit u : instructions) {
207                        if (branchTargetLabels.containsKey(u)) {
208                                mv.visitLabel(branchTargetLabels.get(u));
209                        }
210                        if (u.hasTag("LineNumberTag")) {
211                                LineNumberTag lnt = (LineNumberTag) u.getTag("LineNumberTag");
212                                Label l;
213                                if (branchTargetLabels.containsKey(u)) {
214                                        l = branchTargetLabels.get(u);
215                                } else {
216                                        l = new Label();
217                                        mv.visitLabel(l);
218                                }
219                                mv.visitLineNumber(lnt.getLineNumber(), l);
220                        }
221                        generateInstruction(mv, (Inst) u);
222                }
223                                
224                // Generate the local annotations
225                if (Options.v().write_local_annotations()) {
226                        Label endLabel = new Label();
227                        mv.visitLabel(endLabel);
228                        
229                        for (Local local : body.getLocals()) {
230                                Integer slot = localToSlot.get(local);
231                                if (slot != null) {
232                                        BafLocal l = (BafLocal) local;
233                                        if (l.getOriginalLocal() != null)
234                                        {
235                                                Local jimpleLocal = l.getOriginalLocal();
236                                                if (jimpleLocal != null)
237                                                        mv.visitLocalVariable(jimpleLocal.getName(), toTypeDesc(jimpleLocal.getType()),
238                                                                        null, startLabel, endLabel, slot);
239                                        }
240                                }
241                        }
242                }
243        }

337                        @Override
338                        public void casePushInst(PushInst i) {
339                                Constant c = i.getConstant();
340                                if (c instanceof IntConstant) {
341                                        int v = ((IntConstant) c).value;
342                                        switch (v) {
343                                        case -1:
344                                                mv.visitInsn(Opcodes.ICONST_M1);
345                                                break;
346                                        case 0:
347                                                mv.visitInsn(Opcodes.ICONST_0);
348                                                break;
349                                        case 1:
350                                                mv.visitInsn(Opcodes.ICONST_1);
351                                                break;
352                                        case 2:
353                                                mv.visitInsn(Opcodes.ICONST_2);
354                                                break;
355                                        case 3:
356                                                mv.visitInsn(Opcodes.ICONST_3);
357                                                break;
358                                        case 4:
359                                                mv.visitInsn(Opcodes.ICONST_4);
360                                                break;
361                                        case 5:
362                                                mv.visitInsn(Opcodes.ICONST_5);
363                                                break;
364                                        default:
365                                                if (v >= Byte.MIN_VALUE && v <= Byte.MAX_VALUE) {
366                                                        mv.visitIntInsn(Opcodes.BIPUSH, v);
367                                                } else if (v >= Short.MIN_VALUE && v <= Short.MAX_VALUE) {
368                                                        mv.visitIntInsn(Opcodes.SIPUSH, v);
369                                                } else {
370                                                        mv.visitLdcInsn(v);
371                                                }
372                                        }
373                                } else if (c instanceof StringConstant) {
374                                        mv.visitLdcInsn(((StringConstant) c).value);
375                                } else if (c instanceof ClassConstant) {
376                                        mv.visitLdcInsn(org.objectweb.asm.Type
377                                                        .getObjectType(((ClassConstant) c).getValue()));
378                                } else if (c instanceof DoubleConstant) {
379                                        double v = ((DoubleConstant) c).value;
380                                        /*
381                                         * Do not emit a DCONST_0 for negative zero, therefore we
382                                         * need the following check.
383                                         */
384                                        if (new Double(v).equals(0.0)) {
385                                                mv.visitInsn(Opcodes.DCONST_0);
386                                        } else if (v == 1) {
387                                                mv.visitInsn(Opcodes.DCONST_1);
388                                        } else {
389                                                mv.visitLdcInsn(v);
390                                        }
391                                } else if (c instanceof FloatConstant) {
392                                        float v = ((FloatConstant) c).value;
393                                        /*
394                                         * Do not emit a FCONST_0 for negative zero, therefore we
395                                         * need the following check.
396                                         */
397                                        if (new Float(v).equals(0.0f)) {
398                                                mv.visitInsn(Opcodes.FCONST_0);
399                                        } else if (v == 1) {
400                                                mv.visitInsn(Opcodes.FCONST_1);
401                                        } else if (v == 2) {
402                                                mv.visitInsn(Opcodes.FCONST_2);
403                                        } else {
404                                                mv.visitLdcInsn(v);
405                                        }
406                                } else if (c instanceof LongConstant) {
407                                        long v = ((LongConstant) c).value;
408                                        if (v == 0) {
409                                                mv.visitInsn(Opcodes.LCONST_0);
410                                        } else if (v == 1) {
411                                                mv.visitInsn(Opcodes.LCONST_1);
412                                        } else {
413                                                mv.visitLdcInsn(v);
414                                        }
415                                } else if (c instanceof NullConstant) {
416                                        mv.visitInsn(Opcodes.ACONST_NULL);
417                                } else {
418                                        throw new RuntimeException("unsupported opcode");
419                                }
420                        }
