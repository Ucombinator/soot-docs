086    public PAG( final SparkOptions opts ) {
087        this.opts = opts;
088        this.cgOpts = new CGOptions( PhaseOptions.v().getPhaseOptions("cg") );
089        if( opts.add_tags() ) {
090            nodeToTag = new HashMap<Node, Tag>();
091        }
092        if (opts.rta() && opts.on_fly_cg()) {
093                throw new RuntimeException("Incompatible options rta:true and on-fly-cg:true for cg.spark. Use -p cg-"
094                                + ".spark on-fly-cg:false when using RTA.");
095                }
096        typeManager = new TypeManager(this);
097        if( !opts.ignore_types() ) {
098            typeManager.setFastHierarchy( Scene.v().getOrMakeFastHierarchy() );
099        }
100        switch( opts.set_impl() ) {
101            case SparkOptions.set_impl_hash:
102                setFactory = HashPointsToSet.getFactory();
103                break;
104            case SparkOptions.set_impl_hybrid:
105                setFactory = HybridPointsToSet.getFactory();
106                break;
107            case SparkOptions.set_impl_heintze:
108                setFactory = SharedHybridSet.getFactory();
109                break;
110            case SparkOptions.set_impl_sharedlist:
111                setFactory = SharedListSet.getFactory();
112                break;
113            case SparkOptions.set_impl_array:
114                setFactory = SortedArraySet.getFactory();
115                break;
116            case SparkOptions.set_impl_bit:
117                setFactory = BitPointsToSet.getFactory();
118                break;
119            case SparkOptions.set_impl_double:
120                P2SetFactory oldF;
121                P2SetFactory newF;
122                switch( opts.double_set_old() ) {
123                    case SparkOptions.double_set_old_hash:
124                        oldF = HashPointsToSet.getFactory();
125                        break;
126                    case SparkOptions.double_set_old_hybrid:
127                        oldF = HybridPointsToSet.getFactory();
128                        break;
129                    case SparkOptions.double_set_old_heintze:
130                        oldF = SharedHybridSet.getFactory();
131                        break;
132                    case SparkOptions.double_set_old_sharedlist:
133                        oldF = SharedListSet.getFactory();
134                        break;
135                    case SparkOptions.double_set_old_array:
136                        oldF = SortedArraySet.getFactory();
137                        break;
138                    case SparkOptions.double_set_old_bit:
139                        oldF = BitPointsToSet.getFactory();
140                        break;
141                    default:
142                        throw new RuntimeException();
143                }
144                switch( opts.double_set_new() ) {
145                    case SparkOptions.double_set_new_hash:
146                        newF = HashPointsToSet.getFactory();
147                        break;
148                    case SparkOptions.double_set_new_hybrid:
149                        newF = HybridPointsToSet.getFactory();
150                        break;
151                    case SparkOptions.double_set_new_heintze:
152                        newF = SharedHybridSet.getFactory();
153                        break;
154                    case SparkOptions.double_set_new_sharedlist:
155                        newF = SharedListSet.getFactory();
156                        break;
157                    case SparkOptions.double_set_new_array:
158                        newF = SortedArraySet.getFactory();
159                        break;
160                    case SparkOptions.double_set_new_bit:
161                        newF = BitPointsToSet.getFactory();
162                        break;
163                    default:
164                        throw new RuntimeException();
165                }
166                setFactory = DoublePointsToSet.getFactory( newF, oldF );
167                break;
168            default:
169                throw new RuntimeException();
170        }
171        runGeomPTA = opts.geom_pta();
172    }

219    private PointsToSet reachingObjectsInternal( PointsToSet s, final SparkField f ) {
220        if( getOpts().field_based() || getOpts().vta() ) {
221            VarNode n = findGlobalVarNode( f );
222            if( n == null ) {
223                return EmptyPointsToSet.v();
224            }
225            return n.getP2Set();
226        }
227        if( (getOpts()).propagator() == SparkOptions.propagator_alias ) {
228            throw new RuntimeException( "The alias edge propagator does not compute points-to information for instance fields! Use a different propagator." );
229        }
230        PointsToSetInternal bases = (PointsToSetInternal) s;
231        final PointsToSetInternal ret = setFactory.newSet( 
232                (f instanceof SootField) ? ((SootField)f).getType() : null, this );
233        bases.forall( new P2SetVisitor() {
234        public final void visit( Node n ) {
235            Node nDotF = ((AllocNode) n).dot( f );
236            if(nDotF != null) ret.addAll( nDotF.getP2Set(), null );
237        }} );
238        return ret;
239    }

251    public void cleanUpMerges() {
252        if( opts.verbose() ) {
253            G.v().out.println( "Cleaning up graph for merged nodes" );
254        }
255        lookupInMap(simple);
256        lookupInMap(alloc);
257        lookupInMap(store);
258        lookupInMap(load);
259        lookupInMap(simpleInv);
260        lookupInMap(allocInv);
261        lookupInMap(storeInv);
262        lookupInMap(loadInv);
263        
264        somethingMerged = false;
265        if( opts.verbose() ) {
266            G.v().out.println( "Done cleaning up graph for merged nodes" );
267        }

285    /** Node uses this to notify PAG that n2 has been merged into n1. */
286    void mergedWith( Node n1, Node n2 ) {
287        if( n1.equals( n2 ) ) throw new RuntimeException( "oops" );
288
289        somethingMerged = true;
290        if( ofcg() != null ) ofcg().mergedWith( n1, n2 );
291
292        Map[] maps = { simple, alloc, store, load,
293            simpleInv, allocInv, storeInv, loadInv };
294        for (Map<Node, Object> m : maps) {
295            if( !m.keySet().contains( n2 ) ) continue;
296
297            Object[] os = { m.get( n1 ), m.get( n2 ) };
298            int size1 = getSize(os[0]); int size2 = getSize(os[1]);
299            if( size1 == 0 ) {
300                if( os[1] != null ) m.put( n1, os[1] );
301            } else if( size2 == 0 ) {
302                // nothing needed
303            } else if( os[0] instanceof HashSet ) {
304                if( os[1] instanceof HashSet ) {
305                    ((HashSet) os[0]).addAll( (HashSet) os[1] );
306                } else {
307                    Node[] ar = (Node[]) os[1];
308                    for (Node element0 : ar) {
309                        ( (HashSet<Node>) os[0] ).add( element0 );
310                    }
311                }
312            } else if( os[1] instanceof HashSet ) {
313                Node[] ar = (Node[]) os[0];
314                for (Node element0 : ar) {
315                    ((HashSet<Node>) os[1]).add( element0 );
316                }
317                m.put( n1, os[1] );
318            } else if( size1*size2 < 1000 ) {
319                Node[] a1 = (Node[]) os[0];
320                Node[] a2 = (Node[]) os[1];
321                Node[] ret = new Node[size1+size2];
322                System.arraycopy( a1, 0, ret, 0, a1.length ); 
323                int j = a1.length;
324                outer: for (Node rep : a2) {
325                    for( int k = 0; k < j; k++ )
326                        if( rep == ret[k] ) continue outer;
327                    ret[j++] = rep;
328                }
329                Node[] newArray = new Node[j];
330                System.arraycopy( ret, 0, newArray, 0, j );
331                m.put( n1, ret = newArray );
332            } else {
333                HashSet<Node> s = new HashSet<Node>( size1+size2 );
334                for (Object o : os) {
335                    if( o == null ) continue;
336                    if( o instanceof Set ) {
337                        s.addAll( (Set<Node>) o );
338                    } else {
339                        Node[] ar = (Node[]) o;
340                        for (Node element1 : ar) {
341                            s.add( element1 );
342                        }
343                    }
344                }
345                m.put( n1, s );
346            }
347            m.remove( n2 );
348        }
349    }
350    protected final static Node[] EMPTY_NODE_ARRAY = new Node[0];
351    protected <K extends Node> Node[] lookup( Map<K, Object> m, K key ) {
352        Object valueList = m.get( key );
353        if( valueList == null ) {
354            return EMPTY_NODE_ARRAY;
355        }
356        if( valueList instanceof Set ) {
357            try {
358            m.put( key, valueList = 
359                    ( (Set) valueList ).toArray( EMPTY_NODE_ARRAY ) );
360            } catch( Exception e ) {
361                for( Iterator it = ((Set)valueList).iterator(); it.hasNext(); ) {
362                    G.v().out.println( ""+it.next() );
363                }
364                throw new RuntimeException( ""+valueList+e );
365            }
366        }
367        Node[] ret = (Node[]) valueList;
368        if( somethingMerged ) {
369            for( int i = 0; i < ret.length; i++ ) {
370                Node reti = ret[i];
371                Node rep = reti.getReplacement();
372                if( rep != reti || rep == key ) {
373                    Set<Node> s;
374                    if( ret.length <= 75 ) {
375                        int j = i;
376                        outer: for( ; i < ret.length; i++ ) {
377                            reti = ret[i];
378                            rep = reti.getReplacement();
379                            if( rep == key ) continue;
380                            for( int k = 0; k < j; k++ )
381                                if( rep == ret[k] ) continue outer;
382                            ret[j++] = rep;
383                        }
384                        Node[] newArray = new Node[j];
385                        System.arraycopy( ret, 0, newArray, 0, j );
386                        m.put( key, ret = newArray );
387                    } else {
388                        s = new HashSet<Node>( ret.length * 2 );
389                        for( int j = 0; j < i; j++ ) s.add( ret[j] );
390                        for( int j = i; j < ret.length; j++ ) {
391                            rep = ret[j].getReplacement();
392                            if( rep != key ) {
393                                s.add( rep );
394                            }
395                        }
396                        m.put( key, ret = s.toArray( EMPTY_NODE_ARRAY ) );
397                    }
398                    break;
399                }
400            }
401        }
402        return ret;
403    }

461    private void addNodeTag( Node node, SootMethod m ) {
462        if( nodeToTag != null ) {
463            Tag tag;
464            if( m == null ) {
465                tag = new StringTag( node.toString() );
466            } else {
467                tag = new LinkTag( node.toString(), m, m.getDeclaringClass().getName() );
468            }
469            nodeToTag.put( node, tag );
470        }
471    }

473    public AllocNode makeAllocNode( Object newExpr, Type type, SootMethod m ) {
474        if( opts.types_for_sites() || opts.vta() )
475                newExpr = type;
476        AllocNode ret = valToAllocNode.get( newExpr );
477        if( ret == null ) {
478                valToAllocNode.put( newExpr, ret = new AllocNode( this, newExpr, type, m ) );
479            newAllocNodes.add( ret );
480            addNodeTag( ret, m );
481        } else if( !( ret.getType().equals( type ) ) ) {
482                throw new RuntimeException( "NewExpr "+newExpr+" of type "+type+
483                                " previously had type "+ret.getType() );
484        }
485        return ret;
486    }

532    /** Finds or creates the GlobalVarNode for the variable value, of type type. */
533    public GlobalVarNode makeGlobalVarNode( Object value, Type type) {
534        if( opts.rta() ) {
535            value = null;
536            type = RefType.v("java.lang.Object");
537        }
538        GlobalVarNode ret = valToGlobalVarNode.get( value );
539        if( ret == null ) {
540            valToGlobalVarNode.put( value, 
541                    ret = new GlobalVarNode( this, value, type ) );
542            
543            // if library mode is activated, add allocation of every possible type to accessible fields
544            if(cgOpts.library() != CGOptions.library_disabled) {
545                if (value instanceof SootField) {
546                        SootField sf = (SootField) value;
547                        
548                        if (accessibilityOracle.isAccessible(sf)){
549                                type.apply(new SparkLibraryHelper(this, ret, null));
550                        }
551                }
552            }
553            addNodeTag( ret, null );
554        } else if( !( ret.getType().equals( type ) ) ) {
555            throw new RuntimeException( "Value "+value+" of type "+type+
556                    " previously had type "+ret.getType() );
557        }
558        return ret;
559    }
560    /** Finds or creates the LocalVarNode for the variable value, of type type. */
561    public LocalVarNode makeLocalVarNode( Object value, Type type, SootMethod method ) {
562        if( opts.rta() ) {
563            value = null;
564            type = RefType.v("java.lang.Object");
565            method = null;
566        } else if( value instanceof Local ) {
567            Local val = (Local) value;
568            if( val.getNumber() == 0 ) Scene.v().getLocalNumberer().add(val);
569            LocalVarNode ret = localToNodeMap.get( val );
570            if( ret == null ) {
571                localToNodeMap.put( (Local) value,
572                    ret = new LocalVarNode( this, value, type, method ) );
573                addNodeTag( ret, method );
574            } else if( !( ret.getType().equals( type ) ) ) {
575                throw new RuntimeException( "Value "+value+" of type "+type+
576                        " previously had type "+ret.getType() );
577            }
578            return ret;
579        }
580        LocalVarNode ret = valToLocalVarNode.get( value );
581        if( ret == null ) {
582            valToLocalVarNode.put( value, 
583                    ret = new LocalVarNode( this, value, type, method ) );
584            addNodeTag( ret, method );
585        } else if( !( ret.getType().equals( type ) ) ) {
586            throw new RuntimeException( "Value "+value+" of type "+type+
587                    " previously had type "+ret.getType() );
588        }
589        return ret;
590    }

822    final public void addCallTarget( Edge e ) {
823        if( !e.passesParameters() ) return;
824        MethodPAG srcmpag = MethodPAG.v( this, e.src() );
825        MethodPAG tgtmpag = MethodPAG.v( this, e.tgt() );
826        Pair<Node, Node> pval;
827        
828        if( e.isExplicit() || e.kind() == Kind.THREAD
829                         || e.kind() == Kind.ASYNCTASK ) {
830            addCallTarget( srcmpag, tgtmpag, (Stmt) e.srcUnit(),
831                           e.srcCtxt(), e.tgtCtxt(), e );
832        }
833        else if( e.kind() == Kind.EXECUTOR ) {
834                InvokeExpr ie = e.srcStmt().getInvokeExpr();
835            boolean virtualCall = callAssigns.containsKey(ie);
836
837                Node parm = srcmpag.nodeFactory().getNode( ie.getArg(0) );
838                parm = srcmpag.parameterize( parm, e.srcCtxt() );
839                parm = parm.getReplacement();
840
841                Node thiz = tgtmpag.nodeFactory().caseThis();
842                thiz = tgtmpag.parameterize( thiz, e.tgtCtxt() );
843                thiz = thiz.getReplacement();
844
845                addEdge( parm, thiz );
846                pval = addInterproceduralAssignment(parm, thiz, e);
847                callAssigns.put(ie, pval);
848                callToMethod.put(ie, srcmpag.getMethod());
849
850                if (virtualCall && !virtualCallsToReceivers.containsKey(ie)) {
851                virtualCallsToReceivers.put(ie, parm);
852            }
853        }
854        else if( e.kind() == Kind.PRIVILEGED ) {
855                // Flow from first parameter of doPrivileged() invocation
856                // to this of target, and from return of target to the
857                // return of doPrivileged()
858                InvokeExpr ie = e.srcStmt().getInvokeExpr();
859
860                Node parm = srcmpag.nodeFactory().getNode( ie.getArg(0) );
861                parm = srcmpag.parameterize( parm, e.srcCtxt() );
862                parm = parm.getReplacement();
863
864                Node thiz = tgtmpag.nodeFactory().caseThis();
865                thiz = tgtmpag.parameterize( thiz, e.tgtCtxt() );
866                thiz = thiz.getReplacement();
867
868                addEdge( parm, thiz );
869                pval = addInterproceduralAssignment(parm, thiz, e);
870                callAssigns.put(ie, pval);
871                callToMethod.put(ie, srcmpag.getMethod());
872
873                if( e.srcUnit() instanceof AssignStmt ) {
874                        AssignStmt as = (AssignStmt) e.srcUnit();
875
876                        Node ret = tgtmpag.nodeFactory().caseRet();
877                        ret = tgtmpag.parameterize( ret, e.tgtCtxt() );
878                        ret = ret.getReplacement();
879
880                        Node lhs = srcmpag.nodeFactory().getNode(as.getLeftOp());
881                        lhs = srcmpag.parameterize( lhs, e.srcCtxt() );
882                        lhs = lhs.getReplacement();
883
884                        addEdge( ret, lhs );
885                        pval = addInterproceduralAssignment(ret, lhs, e);
886                        callAssigns.put(ie, pval);
887                        callToMethod.put(ie, srcmpag.getMethod());
888                }
889        }
890        else if( e.kind() == Kind.FINALIZE ) {
891                Node srcThis = srcmpag.nodeFactory().caseThis();
892                srcThis = srcmpag.parameterize( srcThis, e.srcCtxt() );
893                srcThis = srcThis.getReplacement();
894
895                Node tgtThis = tgtmpag.nodeFactory().caseThis();
896                tgtThis = tgtmpag.parameterize( tgtThis, e.tgtCtxt() );
897                tgtThis = tgtThis.getReplacement();
898
899                addEdge( srcThis, tgtThis );
900                pval = addInterproceduralAssignment(srcThis, tgtThis, e);
901        }
902        else if( e.kind() == Kind.NEWINSTANCE ) {
903                Stmt s = (Stmt) e.srcUnit();
904                InstanceInvokeExpr iie = (InstanceInvokeExpr) s.getInvokeExpr();
905
906                Node cls = srcmpag.nodeFactory().getNode( iie.getBase() );
907                cls = srcmpag.parameterize( cls, e.srcCtxt() );
908                cls = cls.getReplacement();
909                Node newObject = nodeFactory.caseNewInstance( (VarNode) cls );
910
911                Node initThis = tgtmpag.nodeFactory().caseThis();
912                initThis = tgtmpag.parameterize( initThis, e.tgtCtxt() );
913                initThis = initThis.getReplacement();
914
915                addEdge( newObject, initThis );
916                if (s instanceof AssignStmt) {
917                        AssignStmt as = (AssignStmt)s;
918                        Node asLHS = srcmpag.nodeFactory().getNode(as.getLeftOp());
919                        asLHS = srcmpag.parameterize( asLHS, e.srcCtxt());
920                        asLHS = asLHS.getReplacement();
921                        addEdge( newObject, asLHS);
922                }
923                
924                pval = addInterproceduralAssignment(newObject, initThis, e);
925                callAssigns.put(s.getInvokeExpr(), pval);
926                callToMethod.put(s.getInvokeExpr(), srcmpag.getMethod());
927        }
928        else if( e.kind() == Kind.REFL_INVOKE ) {
929                // Flow (1) from first parameter of invoke(..) invocation
930                // to this of target, (2) from the contents of the second (array) parameter
931                // to all parameters of the target, and (3)  from return of target to the
932                // return of invoke(..)
933                
934                //(1)
935                InvokeExpr ie = e.srcStmt().getInvokeExpr();
936                Value arg0 = ie.getArg(0);
937                //if "null" is passed in, omit the edge
938                if(arg0!=NullConstant.v()) {
939                        Node parm0 = srcmpag.nodeFactory().getNode( arg0 );
940                        parm0 = srcmpag.parameterize( parm0, e.srcCtxt() );
941                        parm0 = parm0.getReplacement();
942        
943                        Node thiz = tgtmpag.nodeFactory().caseThis();
944                        thiz = tgtmpag.parameterize( thiz, e.tgtCtxt() );
945                        thiz = thiz.getReplacement();
946        
947                        addEdge( parm0, thiz );
948                        pval = addInterproceduralAssignment(parm0, thiz, e);
949                        callAssigns.put(ie, pval);
950                        callToMethod.put(ie, srcmpag.getMethod());
951                }
952
953                //(2)
954                Value arg1 = ie.getArg(1);
955                SootMethod tgt = e.getTgt().method();
956                //if "null" is passed in, or target has no parameters, omit the edge
957                if(arg1!=NullConstant.v() && tgt.getParameterCount()>0) {
958                        Node parm1 = srcmpag.nodeFactory().getNode( arg1 );
959                        parm1 = srcmpag.parameterize( parm1, e.srcCtxt() );
960                        parm1 = parm1.getReplacement();
961                        FieldRefNode parm1contents = makeFieldRefNode( (VarNode) parm1, ArrayElement.v() );
962                        
963                        for(int i=0;i<tgt.getParameterCount(); i++) {
964                                //if no reference type, create no edge
965                                if(!(tgt.getParameterType(i) instanceof RefLikeType)) continue;
966                                
967                                Node tgtParmI = tgtmpag.nodeFactory().caseParm( i );
968                                tgtParmI = tgtmpag.parameterize( tgtParmI, e.tgtCtxt() );
969                                tgtParmI = tgtParmI.getReplacement();
970        
971                                addEdge( parm1contents, tgtParmI );
972                                pval = addInterproceduralAssignment(parm1contents, tgtParmI, e);
973                                callAssigns.put(ie, pval);
974                        }
975                }
976
977                //(3)
978                //only create return edge if we are actually assigning the return value and
979                //the return type of the callee is actually a reference type
980                if( e.srcUnit() instanceof AssignStmt && (tgt.getReturnType() instanceof RefLikeType)) {
981                        AssignStmt as = (AssignStmt) e.srcUnit();
982
983                        Node ret = tgtmpag.nodeFactory().caseRet();
984                        ret = tgtmpag.parameterize( ret, e.tgtCtxt() );
985                        ret = ret.getReplacement();
986
987                        Node lhs = srcmpag.nodeFactory().getNode(as.getLeftOp());
988                        lhs = srcmpag.parameterize( lhs, e.srcCtxt() );
989                        lhs = lhs.getReplacement();
990
991                        addEdge( ret, lhs );
992                        pval = addInterproceduralAssignment(ret, lhs, e);
993                        callAssigns.put(ie, pval);
994                }
995        }
996        else if( e.kind() == Kind.REFL_CLASS_NEWINSTANCE || e.kind() == Kind.REFL_CONSTR_NEWINSTANCE) {
997                // (1) create a fresh node for the new object
998                // (2) create edge from this object to "this" of the constructor
999                // (3) if this is a call to Constructor.newInstance and not Class.newInstance,
1000                //     create edges passing the contents of the arguments array of the call
1001                //     to all possible parameters of the target
1002                // (4) if we are inside an assign statement,
1003                //     assign the fresh object from (1) to the LHS of the assign statement
1004                
1005                Stmt s = (Stmt) e.srcUnit();
1006                InstanceInvokeExpr iie = (InstanceInvokeExpr) s.getInvokeExpr();
1007
1008                //(1)
1009                Node cls = srcmpag.nodeFactory().getNode( iie.getBase() );
1010                cls = srcmpag.parameterize( cls, e.srcCtxt() );
1011                cls = cls.getReplacement();
1012                if( cls instanceof ContextVarNode ) cls = findLocalVarNode( ((VarNode)cls).getVariable() );
1013                
1014                VarNode newObject = makeGlobalVarNode( cls, RefType.v( "java.lang.Object" ) );
1015                SootClass tgtClass = e.getTgt().method().getDeclaringClass();
1016                RefType tgtType = tgtClass.getType();                
1017                AllocNode site = makeAllocNode( new Pair<Node, SootClass>(cls, tgtClass), tgtType, null );
1018                addEdge( site, newObject );
1019
1020                //(2)
1021                Node initThis = tgtmpag.nodeFactory().caseThis();
1022                initThis = tgtmpag.parameterize( initThis, e.tgtCtxt() );
1023                initThis = initThis.getReplacement();
1024                addEdge( newObject, initThis );
1025                addInterproceduralAssignment(newObject, initThis, e);
1026                
1027                //(3)
1028                if(e.kind() == Kind.REFL_CONSTR_NEWINSTANCE) {
1029                        Value arg = iie.getArg(0);
1030                        SootMethod tgt = e.getTgt().method();
1031                        //if "null" is passed in, or target has no parameters, omit the edge
1032                        if(arg!=NullConstant.v() && tgt.getParameterCount()>0) {
1033                                Node parm0 = srcmpag.nodeFactory().getNode( arg );
1034                                parm0 = srcmpag.parameterize( parm0, e.srcCtxt() );
1035                                parm0 = parm0.getReplacement();
1036                                FieldRefNode parm1contents = makeFieldRefNode( (VarNode) parm0, ArrayElement.v() );
1037                                
1038                                for(int i=0;i<tgt.getParameterCount(); i++) {
1039                                        //if no reference type, create no edge
1040                                        if(!(tgt.getParameterType(i) instanceof RefLikeType)) continue;
1041                                        
1042                                        Node tgtParmI = tgtmpag.nodeFactory().caseParm( i );
1043                                        tgtParmI = tgtmpag.parameterize( tgtParmI, e.tgtCtxt() );
1044                                        tgtParmI = tgtParmI.getReplacement();
1045                
1046                                        addEdge( parm1contents, tgtParmI );
1047                                        pval = addInterproceduralAssignment(parm1contents, tgtParmI, e);
1048                                        callAssigns.put(iie, pval);
1049                                }
1050                        }
1051                }
1052                
1053                //(4)
1054                if (s instanceof AssignStmt) {
1055                        AssignStmt as = (AssignStmt)s;
1056                        Node asLHS = srcmpag.nodeFactory().getNode(as.getLeftOp());
1057                        asLHS = srcmpag.parameterize( asLHS, e.srcCtxt());
1058                        asLHS = asLHS.getReplacement();
1059                        addEdge( newObject, asLHS);
1060                }
1061                
1062                pval = addInterproceduralAssignment(newObject, initThis, e);
1063                callAssigns.put(s.getInvokeExpr(), pval);
1064                callToMethod.put(s.getInvokeExpr(), srcmpag.getMethod());
1065        }
1066        else {
1067                throw new RuntimeException( "Unhandled edge "+e );
1068        }
1069    }

1072    /** Adds method target as a possible target of the invoke expression in s.
1073     * If target is null, only creates the nodes for the call site,
1074     * without actually connecting them to any target method.
1075     **/
1076    final public void addCallTarget( MethodPAG srcmpag,
1077                                     MethodPAG tgtmpag,
1078                                     Stmt s,
1079                                     Context srcContext,
1080                                     Context tgtContext,
1081                                     Edge e ) {
1082        MethodNodeFactory srcnf = srcmpag.nodeFactory();
1083        MethodNodeFactory tgtnf = tgtmpag.nodeFactory();
1084        InvokeExpr ie = s.getInvokeExpr();
1085        boolean virtualCall = callAssigns.containsKey(ie);
1086        int numArgs = ie.getArgCount();
1087        for( int i = 0; i < numArgs; i++ ) {
1088            Value arg = ie.getArg( i );
1089            if( !( arg.getType() instanceof RefLikeType ) ) continue;
1090            if( arg instanceof NullConstant ) continue;
1091
1092            Node argNode = srcnf.getNode( arg );
1093            argNode = srcmpag.parameterize( argNode, srcContext );
1094            argNode = argNode.getReplacement();
1095
1096            Node parm = tgtnf.caseParm( i );
1097            parm = tgtmpag.parameterize( parm, tgtContext );
1098            parm = parm.getReplacement();
1099
1100            addEdge( argNode, parm );
1101            Pair<Node, Node> pval = addInterproceduralAssignment(argNode, parm, e);
1102                        callAssigns.put(ie, pval);
1103            callToMethod.put(ie, srcmpag.getMethod());
1104        }
1105        if( ie instanceof InstanceInvokeExpr ) {
1106            InstanceInvokeExpr iie = (InstanceInvokeExpr) ie;
1107
1108            Node baseNode = srcnf.getNode( iie.getBase() );
1109            baseNode = srcmpag.parameterize( baseNode, srcContext );
1110            baseNode = baseNode.getReplacement();
1111
1112            Node thisRef = tgtnf.caseThis();
1113            thisRef = tgtmpag.parameterize( thisRef, tgtContext );
1114            thisRef = thisRef.getReplacement();
1115            addEdge( baseNode, thisRef );
1116            Pair<Node, Node> pval = addInterproceduralAssignment(baseNode, thisRef, e);
1117                        callAssigns.put(ie, pval);
1118            callToMethod.put(ie, srcmpag.getMethod());
1119            if (virtualCall && !virtualCallsToReceivers.containsKey(ie)) {
1120                virtualCallsToReceivers.put(ie, baseNode);
1121            }
1122        }
1123        if( s instanceof AssignStmt ) {
1124            Value dest = ( (AssignStmt) s ).getLeftOp();
1125            if( dest.getType() instanceof RefLikeType && !(dest instanceof NullConstant) ) {
1126
1127                Node destNode = srcnf.getNode( dest );
1128                destNode = srcmpag.parameterize( destNode, srcContext );
1129                destNode = destNode.getReplacement();
1130
1131                Node retNode = tgtnf.caseRet();
1132                retNode = tgtmpag.parameterize( retNode, tgtContext );
1133                retNode = retNode.getReplacement();
1134
1135                addEdge( retNode, destNode );
1136                Pair<Node, Node> pval = addInterproceduralAssignment( retNode, destNode, e );
1137                                callAssigns.put(ie, pval);
1138                callToMethod.put(ie, srcmpag.getMethod());
1139            }
1140        }

1174    protected <K extends Node> boolean addToMap( Map<K, Object> m, K key, Node value ) {
1175        Object valueList = m.get( key );
1176
1177        if( valueList == null ) {
1178            m.put( key, valueList = new HashSet(4) );
1179        } else if( !(valueList instanceof Set) ) {
