060        private void buildVmatchEdges() {
061                // for each store and load pair
062                for (Iterator iter = pag.loadSourcesIterator(); iter.hasNext();) {
063                        final FieldRefNode frn1 = (FieldRefNode) iter.next();
064                        for (Iterator iter2 = pag.storeInvSourcesIterator(); iter2
065                                        .hasNext();) {
066                                final FieldRefNode frn2 = (FieldRefNode) iter2.next();
067                                VarNode base1 = frn1.getBase();
068                                VarNode base2 = frn2.getBase();
069
070
071//                              debug(frn1, frn2, base1, base2);
072
073                                // if they store & load the same field
074                                if (frn1.getField().equals(frn2.getField())) {
075
076                                        if (base1.getP2Set().hasNonEmptyIntersection(
077                                                        base2.getP2Set())) {
078
079                                                //                                              System.err.println("srcs:");
080                                                Node[] src = pag.loadLookup(frn1);
081                                                Node[] dst = pag.storeInvLookup(frn2);
082
083                                                for (int i = 0; i < src.length; i++) {
084                                                        //                                                                                                              System.err.println(src[i]);
085                                                        vmatches.put(src[i], dst);
086                                                }
087                                                //                                              System.err.println("dst:");
088                                                for (int i = 0; i < dst.length; i++) {
089                                                        //                                                                                                              System.err.println(dst[i]);
090                                                        invVmatches.put(dst[i], src);
091                                                }
092
093                                        }
094                                }
095
096                        }
097                }
098        }

155        private final static Predicate<Node> emptyP2SetPred = new Predicate<Node>() {
156        public boolean test(Node n) {
157            return !(n instanceof AllocNode) && n.getP2Set().isEmpty();
158        }
159    };

161        /**
162     * Generate a node declaration for a dot file.
163         * @param node the node
164     * @param p a predicate over nodes, which, if true, will
165     * cause the node to appear red
166         * @return the appropriate {@link String} for the dot file
167         */
168        public static String makeDotNodeLabel(Node n, Predicate<Node> p) {
169                String color = "";
170                String label;
171
172                if (p.test(n))
173                        color = ", color=red";
174                if (n instanceof LocalVarNode) {
175                        label = makeLabel((LocalVarNode) n);
176                } else if (n instanceof AllocNode) {
177                        label = makeLabel((AllocNode) n);
178                } else if (n instanceof FieldRefNode) {
179                        label = makeLabel((FieldRefNode) n);
180                } else {
181                        label = n.toString();
182                }
183                return makeNodeName(n) + "[label=\"" + label + "\"" + color + "];";
184
185        }
186
187        private static String translateLabel(Node n) { 
188            return makeDotNodeLabel(n, emptyP2SetPred);
189        }

262        private void dumpLocalP2Set(String mName, final PrintStream ps) {
263
264                for (Iterator iter = pag.getVarNodeNumberer().iterator(); iter
265                                .hasNext();) {
266                        VarNode vNode = (VarNode) iter.next();
267
268                        if (vNode instanceof LocalVarNode) {
269                                final LocalVarNode lvNode = (LocalVarNode) vNode;
270                                if (lvNode.getMethod() != null
271                                                && lvNode.getMethod().getName().equals(mName)) {
272
273                                        ps.println("\t" + makeNodeName(lvNode) + " [label=\""
274                                                        + makeLabel(lvNode) + "\"];");
275                                        lvNode.getP2Set().forall(new P2SetToDotPrinter(lvNode, ps));
276
277                                }
278                        }
279                }
280        }

338        /**
339         * @param lvNode
340         * @param ps
341         */
342        private void dumpForwardReachableNodesFrom(final LocalVarNode lvNode,
343                        final PrintStream ps) {
344                ps.println("\t" + translateLabel(lvNode));
345
346                Node[] succs = pag.simpleInvLookup(lvNode);
347                for (int i = 0; i < succs.length; i++) {
348                        ps.println("\t" + translateLabel(succs[i]));
349                        // print edge
350                        ps.println("\t" + translateEdge(lvNode, succs[i], "assign"));
351                }
352
353                succs = pag.allocInvLookup(lvNode);
354                for (int i = 0; i < succs.length; i++) {
355                        ps.println("\t" + translateLabel(succs[i]));
356                        // print edge
357                        ps.println("\t" + translateEdge(lvNode, succs[i], "new"));
358                }
359
360                succs = pag.loadInvLookup(lvNode);
361                for (int i = 0; i < succs.length; i++) {
362                        final FieldRefNode frNode = (FieldRefNode) succs[i];
363                        ps.println("\t" + translateLabel(frNode));
364                        ps.println("\t" + translateLabel(frNode.getBase()));
365
366                        // print edge
367                        ps.println("\t" + translateEdge(lvNode, frNode, "load"));
368                        ps.println("\t"
369                                        + translateEdge(frNode, frNode.getBase(), "getBase"));
370
371                }
372
373                succs = pag.storeLookup(lvNode);
374
375                for (int i = 0; i < succs.length; i++) {
376                        final FieldRefNode frNode = (FieldRefNode) succs[i];
377                        ps.println("\t" + translateLabel(frNode));
378                        ps.println("\t" + translateLabel(frNode.getBase()));
379                        // print edge
380                        ps.println("\t" + translateEdge(frNode, lvNode, "store"));
381                        ps.println("\t"
382                                        + translateEdge(frNode, frNode.getBase(), "getBase"));
383                }
384        }

386        public void traceNode(int id) {
387                buildVmatchEdges();
388                String fName = "trace." + id + ".dot";
389                try {
390                        FileOutputStream fos = new FileOutputStream(new File(fName));
391                        PrintStream ps = new PrintStream(fos);
392                        ps.println("digraph G {");
393
394                        // iterate over all variable nodes
395                        for (Iterator iter = pag.getVarNodeNumberer().iterator(); iter
396                                        .hasNext();) {
397                                final VarNode n = (VarNode) iter.next();
398
399                                if (n.getNumber() == id) {
400                                        LocalVarNode lvn = (LocalVarNode) n;
401
402                                        printOneNode(lvn);
403
404                                        trace(lvn, ps, new HashSet<Node>(), TRACE_MAX_LVL);
405                                }
406                        }
407                        
408                        ps.print("}");
409                        ps.close();
410                        fos.close();
411                } catch (IOException e) {
412                        e.printStackTrace();
413                }
414        }

495                                continue;
496                        ps.println("\t" + translateLabel(succs[i]));
497                        // print edge
498                        ps.println("\t" + translateEdge(node, succs[i], "assign"));

508                        // print edge
509                        ps.println("\t" + translateEdge(node, succs[i], "new"));
510                }
511
512                succs = vmatches.get(node);
513                if (succs != null) {
514                        //                      System.err.println(succs.length);

516                                //                              System.err.println(succs[i]);

521                                ps.println("\t" + translateEdge(node, succs[i], "vmatch"));
522                                trace((VarNode) succs[i], ps, visitedNodes, level-1);
523                        }
524                }
525        }

535        public static String makeLabel(LocalVarNode n) {
536                SootMethod sm = n.getMethod();
537                return "LV " + n.getVariable().toString() + " " + n.getNumber() + "\\n"
538                                + sm.getDeclaringClass() + "\\n" + sm.getName();

542         * @param node
543         * @return
544         */
545        public static String makeLabel(FieldRefNode node) {
546                if (node.getField() instanceof SootField) {
547                        final SootField sf = (SootField) node.getField();
548                        return "FNR " + makeLabel(node.getBase()) + "." + sf.getName();
549                } else
550                        return "FNR " + makeLabel(node.getBase()) + "." + node.getField();
551        }

560                else
561                        return base.toString();

566        class P2SetToDotPrinter extends P2SetVisitor {
567
568      private final Node curNode;
569      
570      private final PrintStream ps;
571      P2SetToDotPrinter(Node curNode, PrintStream ps) {
572        this.curNode = curNode;
573        this.ps = ps;
574      }
575      
576                public void visit(Node n) {
577                        ps.println("\t" + makeNodeName(n) + " [label=\""
578                                        + makeLabel((AllocNode) n) + "\"];");
579
580                        ps.print("\t" + makeNodeName(curNode) + " -> ");
581                        ps.println(makeNodeName(n) + ";");
582                }
583
584        }
