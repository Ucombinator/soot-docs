085  public static FieldAccessMap buildStoreMap(PAG pag) {
086    FieldAccessMap ret = new FieldAccessMap();
087    Iterator frNodeIter = pag.storeInvSourcesIterator();
088    while (frNodeIter.hasNext()) {
089      FieldRefNode frNode = (FieldRefNode) frNodeIter.next();
090      SparkField field = frNode.getField();
091      Node[] targets = pag.storeInvLookup(frNode);
092      for (int i = 0; i < targets.length; i++) {
093        VarNode target = (VarNode) targets[i];
094        if (target instanceof GlobalVarNode)
095          continue;
096        ret.put(field, new Pair<FieldRefNode, LocalVarNode>(frNode, (LocalVarNode) target));
097      }
098    }
099    return ret;
100  }

219    FieldToEdgesMap storesOnField = new FieldToEdgesMap();
220    Iterator frNodeIter = pag.storeInvSourcesIterator();

222      FieldRefNode frNode = (FieldRefNode) frNodeIter.next();
223      VarNode source = frNode.getBase();
224      SparkField field = frNode.getField();
225      Node[] targets = pag.storeInvLookup(frNode);
226      for (int i = 0; i < targets.length; i++) {
227        VarNode target = (VarNode) targets[i];
228        storesOnField.put(field, new Pair<VarNode, VarNode>(target, source));
229      }
230    }
231    return storesOnField;
232  }

308  private static HybridPointsToSet convertToHybrid(final PointsToSetInternal set) {
309    HybridPointsToSet ret = null;
310    if (set instanceof HybridPointsToSet) {
311      ret = (HybridPointsToSet) set;
312    } else if (set instanceof DoublePointsToSet) {
313      assert ((DoublePointsToSet) set).getNewSet().isEmpty();
314      ret = (HybridPointsToSet) ((DoublePointsToSet) set).getOldSet();
315    }
316    return ret;
317  }

335  public static boolean hasRecursiveField(SootClass sootClass) {
336    Chain fields = sootClass.getFields();
337    for (Iterator iter = fields.iterator(); iter.hasNext();) {
338      SootField sootField = (SootField) iter.next();
339      Type type = sootField.getType();
340      if (type instanceof RefType) {
341        RefType refType = (RefType) type;
342        SootClass sootClass2 = refType.getSootClass();
343        if (sootClass == sootClass2) {
344          return true;
345        }
346      }
347    }
348    return false;
349  }

369  public static boolean noRefTypeParameters(SootMethod method) {
370    if (!method.isStatic())
371      return false;
372    Predicate<Type> notRefTypePred = new Predicate<Type>() {
373
374      @Override
375      public boolean test(Type obj_) {
376        return !(obj_ instanceof RefType) && !(obj_ instanceof ArrayType);
377      }
378
379    };
380    return notRefTypePred.test(method.getReturnType()) && soot.jimple.spark.ondemand.genericutil.Util.forAll(method.getParameterTypes(), notRefTypePred);
381  }

396  public static Collection<? extends SootMethod> getCallTargets(Type type, SootMethod invokedMethod) {
397    if (isConstructor(invokedMethod)) {
398      return Collections.singleton(invokedMethod);
399    }
400    Type receiverType = invokedMethod.getDeclaringClass().getType();
401    ChunkedQueue chunkedQueue = new ChunkedQueue();
402    Iterator iter = chunkedQueue.reader();
403    VirtualCalls.v().resolve(type, receiverType, invokedMethod.getNumberedSubSignature(), null, chunkedQueue);
404    Set<SootMethod> ret = new ArraySet<SootMethod>();
405    for (; iter.hasNext();) {
406      SootMethod target = (SootMethod) iter.next();
407      ret.add(target);
408    }
409    return ret;
410  }

417    File dir = new File(dirName);
418
419    if (!dir.exists()) {
420      try {
421        if (!Options.v().output_jar()) {
422          dir.mkdirs();
423        }
424      } catch (SecurityException se) {
425        G.v().out.println("Unable to create " + dirName);
426        throw new CompilationDeathException(CompilationDeathException.COMPILATION_ABORTED);
427      }
428    }
429    return dirName;
