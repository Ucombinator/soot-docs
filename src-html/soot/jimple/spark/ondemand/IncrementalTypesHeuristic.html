019package soot.jimple.spark.ondemand;
020
021import java.util.HashSet;
022import java.util.Set;
023
024import soot.RefType;
025import soot.SootField;
026import soot.jimple.spark.internal.TypeManager;
027import soot.jimple.spark.ondemand.genericutil.Util;
028import soot.jimple.spark.ondemand.pautil.SootUtil;
029import soot.jimple.spark.ondemand.pautil.SootUtil.CallSiteAndContext;
030import soot.jimple.spark.pag.ArrayElement;
031import soot.jimple.spark.pag.SparkField;
032
033public class IncrementalTypesHeuristic implements FieldCheckHeuristic {
034
035    private final TypeManager manager;
036
037    private static final boolean EXCLUDE_TYPES = false;
038
039    private static final String[] EXCLUDED_NAMES = new String[] { "ca.mcgill.sable.soot.SootMethod" };
040
041    private Set<RefType> typesToCheck = new HashSet<RefType>();
042
043    private Set<RefType> notBothEndsTypes = new HashSet<RefType>();
044
045    private RefType newTypeOnQuery = null;
046
047
048    /*
049     * (non-Javadoc)
050     * 
051     * @see AAA.algs.Heuristic#newQuery()
052     */
053    public boolean runNewPass() {
054//        if (!aggressive && reachedAggressive) {
055//            aggressive = true;
056//            return true;
057//        }
058        if (newTypeOnQuery != null) {
059            boolean added = typesToCheck.add(newTypeOnQuery);
060            if (SootUtil.hasRecursiveField(newTypeOnQuery.getSootClass())) {
061                notBothEndsTypes.add(newTypeOnQuery);
062            }
063            newTypeOnQuery = null;
064            return added;
065        }
066        return false;
067    }
068
069    /*
070     * (non-Javadoc)
071     * 
072     * @see AAA.algs.Heuristic#validateMatchesForField(soot.jimple.spark.pag.SparkField)
073     */
074    public boolean validateMatchesForField(SparkField field) {
075        // if (true) return true;
076        if (field instanceof ArrayElement) {
077            return true;
078        }
079        SootField sootField = (SootField) field;
080        RefType declaringType = sootField.getDeclaringClass().getType();
081        if (EXCLUDE_TYPES) {
082            for (String typeName : EXCLUDED_NAMES) {
083                if (Util.stringContains(declaringType.toString(), typeName)) {
084                    return false;
085                }
086            }
087        }
088        for (RefType typeToCheck : typesToCheck) {
089            if (manager.castNeverFails(declaringType, typeToCheck)) {
090                return true;
091            }
092        }
093        if (newTypeOnQuery == null) {
094            newTypeOnQuery = declaringType;
095            // System.err.println("adding type " + declaringType);
096        }
097        // System.err.println("false for " + field);
098        return false;
099    }
100
101    public IncrementalTypesHeuristic(TypeManager manager) {
102        super();
103        this.manager = manager;
104    }
105
106    public String toString() {
107        StringBuffer ret = new StringBuffer();
108        ret.append("types ");
109        ret.append(typesToCheck.toString());
110        if (!notBothEndsTypes.isEmpty()) {
111            ret.append(" not both ");
112            ret.append(notBothEndsTypes.toString());
113        }
114        return ret.toString();
115    }
116
117    public boolean validFromBothEnds(SparkField field) {
118        if (field instanceof SootField) {
119            SootField sootField = (SootField) field;
120            RefType declaringType = sootField.getDeclaringClass().getType();
121            for (RefType type : notBothEndsTypes) {
122                if (manager.castNeverFails(declaringType, type)) {
123                    return false;
124                }
125            }
126        }
127        return true;
128    }
129

131        // TODO make real heuristic
132        return true;
133    }

135}
