019package soot.jimple.spark.ondemand.genericutil;
020
021import java.util.Collection;
022
023/**
024 * @author manu_s
025 *
026 */
027public final class Stack<T> implements Cloneable {
028
029    private T[] elems;
030
031    private int size = 0;
032
033    @SuppressWarnings("unchecked")
034    public Stack(int numElems_) {
035        elems = (T[])new Object[numElems_];
036    }
037
038    public Stack() {
039        this(4);
040    }
041
042    @SuppressWarnings("unchecked")
043    public void push(T obj_) {
044        assert obj_ != null;
045        if (size == elems.length) {
046            // lengthen array
047            Object[] tmp = elems;
048            elems = (T[])new Object[tmp.length * 2];
049            System.arraycopy(tmp, 0, elems, 0, tmp.length);
050        }
051        elems[size] = obj_;
052        size++;
053    }
054    
055    public void pushAll(Collection<T> c) {
056        for (T t : c) {
057                        push(t);
058                }
059    }
060    public T pop() {
061        if (size == 0) return null;
062        size--;
063        T ret =  elems[size];
064        elems[size] = null;
065        return ret;
066    }
067
068    public T peek() {
069        if (size == 0) return null;
070        return elems[size - 1];
071    }
072    
073    public int size() {
074        return size;
075    }
076    
077    public boolean isEmpty() {
078        return size == 0;
079    }
080    
081    public void clear() {
082        size = 0;
083    }
084
085    @SuppressWarnings("unchecked")
086    public Stack<T> clone() {
087        Stack<T> ret = null;
088                try {
089                        ret = (Stack<T>)super.clone();
090                        ret.elems = (T[])new Object[elems.length];
091                        System.arraycopy(elems, 0, ret.elems, 0, size);
092                        return ret;
093                } catch (CloneNotSupportedException e) {
094                        // should not happen
095                        throw new InternalError();
096                }
097    }
098    
099    public Object get(int i) {
100        return elems[i];
101    }
102    
103    public boolean contains(Object o) {
104        return Util.arrayContains(elems, o, size);
105    }
106    
107    /**
108     * returns first index
109     * @param o
110     * @return
111     */
112    public int indexOf(T o) {
113        for (int i = 0; i < size && elems[i] != null; i++) {
114            if (elems[i].equals(o)) return i;
115        }
116        return -1;
117    }
118    public String toString() {
119        StringBuffer s = new StringBuffer();
120                s.append("[");
121                for (int i=0; i<size && elems[i] != null; i++) {
122                    if (i>0) s.append(", ");
123                    s.append(elems[i].toString());
124                }
125                s.append("]");
126                return s.toString();
127    }
128}
