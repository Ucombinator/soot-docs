052  /** Factorial */
053  public static BigInteger fact(BigInteger n_) {
054    BigInteger result = BigInteger.ONE;
055    for (BigInteger i = BigInteger.ONE; i.compareTo(n_) <= 0; i = i.add(BigInteger.ONE))
056      result = result.multiply(i);
057    return result;
058  }

060  /**
061   * Factorial on doubles; avoids overflow problems present when using integers.
062   * 
063   * @param n_
064   *          arg on which to compute factorial
065   * @return (<code>double</code> approximation to) factorial of largest
066   *         positive integer <= (n_ + epsilon)
067   */
068  public static double fact(double n_) {
069    n_ += 1e-6;
070    double result = 1.0;
071    for (double i = 1; i <= n_; i += 1.0)
072      result *= i;
073    return result;
074  }

085  public static int binaryLogUp(int n_) {
086    int k = 0;

141  /**
142   * Test whether <em>some</em> element of the given {@link Collection}
143   * satisfies the given {@link Predicate}.
144   */
145  public static <T> boolean forSome(Collection<T> c_, Predicate<T> p_) {
146    for (T t : c_) {
147      if (p_.test(t)) {
148        return true;
149      }
150    }
151    return false;
152  }

154  /**
155   * Test whether <em>some</em> element of the given {@link Collection}
156   * satisfies the given {@link Predicate}.
157   * 
158   * @return The first element satisfying the predicate; otherwise null.
159   */
160  public static <T> T find(Collection<T> c_, Predicate<T> p_) {
161    for (Iterator<T> iter = c_.iterator(); iter.hasNext();) {
162      T obj = iter.next();
163      if (p_.test(obj))
164        return obj;
165    }
166
167    return null;
168  }

172   * satisfies the given {@link Predicate}.
173   *

175   */
176  public static <T> Collection<T> findAll(Collection<T> c_, Predicate<T> p_) {
177    Collection<T> result = new LinkedList<T>();
178
179    for (Iterator<T> iter = c_.iterator(); iter.hasNext();) {
180      T obj = iter.next();
181      if (p_.test(obj))
182        result.add(obj);
183    }
184
185    return result;

213  /**
214   * Map a list: generate a new list with each element mapped. The new list is
215   * always an {@link ArrayList}; it would have been more precise to use
216   * {@link java.lang.reflect reflection} to create a list of the same type as
217   * 'srcList', but reflection works really slowly in some implementations, so

220  public static <T, U> List<U> map(List<T> srcList, Mapper<T, U> mapper_) {
221    ArrayList<U> result = new ArrayList<U>();
222    for (Iterator<T> srcIter = srcList.iterator(); srcIter.hasNext();)
223      result.add(mapper_.map(srcIter.next()));
224    return result;

227  /**
228   * Filter a collection: generate a new list from an existing collection,
229   * consisting of the elements satisfying some predicate. The new list is
230   * always an {@link ArrayList}; it would have been more precise to use
231   * {@link java.lang.reflect reflection} to create a list of the same type as
232   * 'srcList', but reflection works really slowly in some implementations, so
233   * it's best to avoid it.
234   */
235  public static <T> List<T> filter(Collection<T> src_, Predicate<T> pred_) {
236    ArrayList<T> result = new ArrayList<T>();
237    for (Iterator<T> srcIter = src_.iterator(); srcIter.hasNext();) {
238      T curElem = srcIter.next();
239      if (pred_.test(curElem))
240        result.add(curElem);
241    }
242    return result;
243  }

296  /** Replace all occurrences of a given substring in a given {@link String}. */
297  public static String replaceAll(String str_, String sub_, String newSub_) {
298    if (str_.indexOf(sub_) == -1)
299      return str_;
300    int subLen = sub_.length();
301    int idx;
302    StringBuffer result = new StringBuffer(str_);
303    while ((idx = result.toString().indexOf(sub_)) >= 0)
304      result.replace(idx, idx + subLen, newSub_);
305    return result.toString();
306  }

313  /** Generate strings with fully qualified names or not */
314  public static final boolean FULLY_QUALIFIED_NAMES = false;
315
316  /** Write object fields to string */
317  public static String objectFieldsToString(Object obj) {
318    // Temporarily disable the security manager
319    SecurityManager oldsecurity = System.getSecurityManager();
320    System.setSecurityManager(new SecurityManager() {
321      public void checkPermission(Permission perm) {
322      }
323    });
324
325    Class c = obj.getClass();
326    StringBuffer buf = new StringBuffer(FULLY_QUALIFIED_NAMES ? c.getName() : removePackageName(c.getName()));
327    while (c != Object.class) {
328      Field[] fields = c.getDeclaredFields();
329
330      if (fields.length > 0)
331        buf = buf.append(" (");
332
333      for (int i = 0; i < fields.length; i++) {
334        // Make this field accessible
335        fields[i].setAccessible(true);
336
337        try {
338          Class type = fields[i].getType();
339          String name = fields[i].getName();
340          Object value = fields[i].get(obj);
341
342          // name=value : type
343          buf = buf.append(name);
344          buf = buf.append("=");
345          buf = buf.append(value == null ? "null" : value.toString());
346          buf = buf.append(" : ");
347          buf = buf.append(FULLY_QUALIFIED_NAMES ? type.getName() : removePackageName(type.getName()));
348        } catch (IllegalAccessException e) {
349          e.printStackTrace();
350        }
351
352        buf = buf.append(i + 1 >= fields.length ? ")" : ",");
353      }
354      c = c.getSuperclass();
355    }
356    // Reinstate the security manager
357    System.setSecurityManager(oldsecurity);
358
359    return buf.toString();
360  }

362  /** Remove the package name from a fully qualified class name */
363  public static String removePackageName(String fully_qualified_name_) {
364    if (fully_qualified_name_ == null)
365      return null;
366
367    int lastdot = fully_qualified_name_.lastIndexOf('.');
368
369    if (lastdot < 0) {
370      return "";
371    } else {
372      return fully_qualified_name_.substring(lastdot + 1);
373    }
374  }

410  public static <T> boolean intersecting(final Set<T> s1, final Set<T> s2) {
411    return forSome(s1, new Predicate<T>() {

445  public static <T> List<T> pickNAtRandom(List<T> vals, int n, long seed) {
446    if (vals.size() <= n) {
447      return vals;
448    }
449    HashSet<T> elems = new HashSet<T>();
450    Random rand = new Random(seed);
451    for (int i = 0; i < n; i++) {
452      boolean added = true;
453      do {
454      int randIndex = rand.nextInt(n);
455      added = elems.add(vals.get(randIndex));
456      } while (!added);
457      
458    }
459    return new ArrayList<T>(elems);    
460  }
