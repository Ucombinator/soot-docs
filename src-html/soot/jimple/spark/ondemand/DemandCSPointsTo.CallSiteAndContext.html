306        public PointsToSet reachingObjects(Local l) {
307                if(lazy)
308                        /*
309                         * create a lazy points-to set; this will not actually compute context information until we ask whether this points-to set
310                         * has a non-empty intersection with another points-to set and this intersection appears to be non-empty; when this is the case
311                         * then the points-to set will call doReachingObjects(..) to refine itself
312                         */                     
313                        return new LazyContextSensitivePointsToSet(l,new WrappedPointsToSet((PointsToSetInternal) pag.reachingObjects(l)),this);
314                else
315                        return doReachingObjects(l);
316        }

318        public PointsToSet doReachingObjects(Local l) {
319                //lazy initialization
320                if(fieldToStores==null) {
321                init();
322                }
323                PointsToSet result;
324        Map<Local, PointsToSet> cache;
325            if(refineCallGraph) {  //we use different caches for different settings  
326            cache = reachingObjectsCache;
327            } else {
328            cache = reachingObjectsCacheNoCGRefinement;
329            }
330        result = cache.get(l);           
331            if(result==null) {
332                result = computeReachingObjects(l);
333                if(useCache) {
334                    cache.put(l, result);
335                }
336            }       
337            assert consistentResult(l,result);
338            return result;
339        }

341    /**
342     * Returns <code>false</code> if an inconsistent computation occurred, i.e. if result
343     * differs from the result computed by {@link #computeReachingObjects(Local)} on l.
344     */
345    private boolean consistentResult(Local l, PointsToSet result) {
346        PointsToSet result2 = computeReachingObjects(l);
347        if(!(result instanceof EqualsSupportingPointsToSet) || !(result2 instanceof EqualsSupportingPointsToSet)) {
348            //cannot compare, assume everything is fine
349            return true;
350        }
351        EqualsSupportingPointsToSet eq1 = (EqualsSupportingPointsToSet) result;
352        EqualsSupportingPointsToSet eq2 = (EqualsSupportingPointsToSet) result2;
353        return new PointsToSetEqualsWrapper(eq1).equals(new PointsToSetEqualsWrapper(eq2)); 
354    }

359    protected PointsToSet computeReachingObjects(Local l) {
360        VarNode v = pag.findLocalVarNode(l);
361                if (v == null) {

363                  return EmptyPointsToSet.v();
364                }
365        PointsToSet contextSensitiveResult = computeRefinedReachingObjects(v);
366        if(contextSensitiveResult == null ) {
367            //had to abort; return Spark's points-to set in a wrapper
368            return new WrappedPointsToSet(v.getP2Set());
369        } else {
370            return contextSensitiveResult;

374    /**
375     * Computes the refined set of reaching objects for l.
376     * Returns <code>null</code> if refinement failed.
377     */
378    protected PointsToSet computeRefinedReachingObjects(VarNode v) {
379        // must reset the refinement heuristic for each query
380        this.fieldCheckHeuristic = HeuristicType.getHeuristic(
381            heuristicType, pag.getTypeManager(), getMaxPasses());
382        doPointsTo = true;
383        numPasses = 0;
384        PointsToSet contextSensitiveResult = null;
385        while (true) {
386                numPasses++;
387                if (DEBUG_PASS != -1 && numPasses > DEBUG_PASS) {
388                        break;
389                }
390                if (numPasses > maxPasses) {
391                        break;
392                }
393                if (DEBUG) {
394                        G.v().out.println("PASS " + numPasses);
395                        G.v().out.println(fieldCheckHeuristic);
396                }
397                clearState();
398                pointsTo = new AllocAndContextSet();
399                try {
400                        refineP2Set(new VarAndContext(v, EMPTY_CALLSTACK), null);
401                        contextSensitiveResult = pointsTo;
402                } catch (TerminateEarlyException e) {
403                }
404                if (!fieldCheckHeuristic.runNewPass()) {
405                        break;
406                }
407        }
408        return contextSensitiveResult;
409    }

412                boolean sameSCCAlready = false;

414                // assert assignEdge.getSrc() instanceof LocalVarNode :

424                }

443        protected PointsToSetInternal checkContextsForAllocsCache(
444                        VarAndContext varAndContext, AllocAndContextSet ret,
445                        PointsToSetInternal locs) {
446                PointsToSetInternal retSet = null;
447                if (contextsForAllocsCache.containsKey(varAndContext)) {
448                        for (AllocAndContext allocAndContext : contextsForAllocsCache.get(
449                                        varAndContext).getO2()) {
450                                if (locs.contains(allocAndContext.alloc)) {
451                                        ret.add(allocAndContext);
452                                }
453                        }
454                        final PointsToSetInternal oldLocs = contextsForAllocsCache.get(
455                                        varAndContext).getO1();
456                        final PointsToSetInternal tmpSet = new HybridPointsToSet(locs
457                                        .getType(), pag);
458                        locs.forall(new P2SetVisitor() {
459
460                                @Override
461                                public void visit(Node n) {
462                                        if (!oldLocs.contains(n)) {
463                                                tmpSet.add(n);
464                                        }
465                                }
466                        });
467                        retSet = tmpSet;
468                        oldLocs.addAll(tmpSet, null);
469                } else {
470                        PointsToSetInternal storedSet = new HybridPointsToSet(locs
471                                        .getType(), pag);
472                        storedSet.addAll(locs, null);
473                        contextsForAllocsCache.put(varAndContext,
474                                        new Pair<PointsToSetInternal, AllocAndContextSet>(
475                                                        storedSet, new AllocAndContextSet()));
476                        retSet = locs;
477                }
478                return retSet;
479        }

649        /*
650         * (non-Javadoc)
651         * 
652         * @see AAA.summary.Refiner#dumpPathForBadLoc(soot.jimple.spark.pag.VarNode,
653         *      soot.jimple.spark.pag.AllocNode)
654         */
655        protected void dumpPathForLoc(VarNode v, final AllocNode badLoc,
656                        String filePrefix) {
657                final HashSet<VarNode> visited = new HashSet<VarNode>();
658                final DotPointerGraph dotGraph = new DotPointerGraph();
659                final class Helper {
660                        boolean handle(VarNode curNode) {
661                                assert curNode.getP2Set().contains(badLoc);
662                                visited.add(curNode);
663                                Node[] newEdges = pag.allocInvLookup(curNode);
664                                for (int i = 0; i < newEdges.length; i++) {
665                                        AllocNode alloc = (AllocNode) newEdges[i];
666                                        if (alloc.equals(badLoc)) {
667                                                dotGraph.addNew(alloc, curNode);
668                                                return true;
669                                        }
670                                }
671                                for (AssignEdge assignEdge : csInfo.getAssignEdges(curNode)) {
672                                        VarNode other = assignEdge.getSrc();
673                                        if (other.getP2Set().contains(badLoc)
674                                                        && !visited.contains(other) && handle(other)) {
675                                                if (assignEdge.isCallEdge()) {
676                                                        dotGraph.addCall(other, curNode, assignEdge
677                                                                        .getCallSite());
678                                                } else {
679                                                        dotGraph.addAssign(other, curNode);
680                                                }
681                                                return true;
682                                        }
683                                }
684                                Node[] loadEdges = pag.loadInvLookup(curNode);
685                                for (int i = 0; i < loadEdges.length; i++) {
686                                        FieldRefNode frNode = (FieldRefNode) loadEdges[i];
687                                        SparkField field = frNode.getField();
688                                        VarNode base = frNode.getBase();
