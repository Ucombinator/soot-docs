102        @Override
103        public void do_before_propagation()
104        {
105//              if ( complex_cons == null )
106                        do_pts_interval_merge();
107                
108//              if ( !(me instanceof LocalVarNode) )
109                        do_flow_edge_interval_merge();
110                        
111                // This pointer filter, please read the comments at this line in file FullSensitiveNode.java
112                Node wrappedNode = getWrappedNode();
113                if (wrappedNode instanceof LocalVarNode
114                                && ((LocalVarNode) wrappedNode).isThisPtr()) {
115                        SootMethod func = ((LocalVarNode) wrappedNode).getMethod();
116                        if (!func.isConstructor()) {
117                                // We don't process the specialinvoke call edge
118                                SootClass defClass = func.getDeclaringClass();
119                                Hierarchy typeHierarchy = Scene.v().getActiveHierarchy();
120
121                                for (Iterator<AllocNode> it = new_pts.keySet().iterator(); it
122                                                .hasNext();) {
123                                        AllocNode obj = it.next();
124                                        if (obj.getType() instanceof RefType) {
125                                                SootClass sc = ((RefType) obj.getType()).getSootClass();
126                                                if (defClass != sc) {
127                                                        try {
128                                                                SootMethod rt_func = typeHierarchy
129                                                                                .resolveConcreteDispatch(sc, func);
130                                                                if (rt_func != func) {
131                                                                        it.remove();
132                                                                        // Also preclude it from propagation again
133                                                                        pt_objs.put(
134                                                                                        obj,
135                                                                                        (PtInsIntervalManager) deadManager);
136                                                                }
137                                                        } catch (RuntimeException e) {
138                                                                // If the input program has a wrong type cast, resolveConcreteDispatch fails and it goes here
139                                                                // We simply ignore this error
140                                                        }
141                                                }
142                                        }
143                                }
144                        }
145                }
146        }

256         * An efficient implementation of differential propagation.
257         */
258        @Override
259        public void propagate(GeomPointsTo ptAnalyzer, IWorklist worklist) 
260        {
261                int i, j;
262                AllocNode obj;
263                SegmentNode pts, pe, int_entry1[], int_entry2[];
264                PtInsIntervalManager pim1, pim2;
265                PtInsNode qn, objn;
266                boolean added, has_new_edges;
267                
268                // We first build the new flow edges via the field dereferences
269                if ( complex_cons != null ) {
270                        for ( Map.Entry<AllocNode, PtInsIntervalManager> entry : new_pts.entrySet() ) {
271                                obj = entry.getKey();
272                                int_entry1 = entry.getValue().getFigures();
273                                
274                                for (PlainConstraint pcons : complex_cons) {
275                                        // Construct the two variables in assignment
276                                        objn = (PtInsNode)ptAnalyzer.findAndInsertInstanceField(obj, pcons.f);
277                                        if ( objn == null ) {
278                                                // This combination of allocdotfield must be invalid
279                                                // This expression p.f also renders that p cannot point to obj, so we remove it
280                                                // We label this event and sweep the garbage later
281                                                pt_objs.put(obj, (PtInsIntervalManager)deadManager);
282                                                entry.setValue( (PtInsIntervalManager)deadManager );
283                                                break;
284                                        }
285                                        
286                                        if ( objn.willUpdate == false ) {
287                                                // This must be a store constraint
288                                                // This object field is not need for computing 
289                                                // the points-to information of the seed pointers
290                                                continue;
291                                        }
292                                        
293                                        qn = (PtInsNode) pcons.otherSide;
294                                        
295                                        for ( i = 0; i < HeapInsIntervalManager.Divisions; ++i ) {
296                                                pts = int_entry1[i];
297                                                while ( pts != null && pts.is_new ) {
298                                                        switch ( pcons.type ) {
299                                                        case Constants.STORE_CONS:
300                                                                // Store, qv -> pv.field
301                                                                // pts.I2 may be zero, pts.L may be less than zero
302                                                                if ( qn.add_simple_constraint_3( objn,
303                                                                                pcons.code == GeometricManager.ONE_TO_ONE ? pts.I1 : 0,
304                                                                                pts.I2,
305                                                                                pts.L
306                                                                ) )
307                                                                        worklist.push( qn );
308                                                                break;
309                                                                
310                                                        case Constants.LOAD_CONS:
311                                                                // Load, pv.field -> qv
312                                                                if ( objn.add_simple_constraint_3( qn, 
313                                                                                pts.I2, 
314                                                                                pcons.code == GeometricManager.ONE_TO_ONE ? pts.I1 : 0,
315                                                                                pts.L
316                                                                ) )
317                                                                        worklist.push( objn );
318                                                                break;
319                                                        }
320                                                        
321                                                        pts = pts.next;
322                                                }
323                                        }
324                                }
325                        }
326                }
327                
328                for ( Map.Entry<PtInsNode, PtInsIntervalManager> entry1 : flowto.entrySet() ) {
329                        // First, we get the flow-to intervals
330                        added = false;
331                        qn = entry1.getKey();
332                        pim1 = entry1.getValue();
333                        int_entry1 = pim1.getFigures();
334                        has_new_edges = pim1.isThereUnprocessedFigures();
335                        Map<AllocNode, PtInsIntervalManager> objs = ( has_new_edges ? pt_objs : new_pts );
336                        
337                        for ( Map.Entry<AllocNode, PtInsIntervalManager> entry2 : objs.entrySet() ) {
338                                // Second, we get the points-to intervals
339                                obj = entry2.getKey();
340                                pim2 = entry2.getValue();
341                                
342                                if ( pim2 == deadManager ) continue;
343                                if (!ptAnalyzer.castNeverFails(obj.getType(), qn.getWrappedNode().getType()))
344                                        continue;
345                                
346                                int_entry2 = pim2.getFigures();
347                                
348                                // We pair up all the interval points-to tuples and interval flow edges
349                                for (i = 0; i < PtInsIntervalManager.Divisions; ++i) {
350                                        pts = int_entry2[i];
351                                        while ( pts != null ) {
352                                                if (  !has_new_edges && !pts.is_new )
353                                                        break;
354                                                
355                                                for ( j = 0; j < PtInsIntervalManager.Divisions; ++j) {
356                                                        pe = int_entry1[j];
357                                                        while ( pe != null ) {
358                                                                if ( pts.is_new || pe.is_new ) {
359                                                                        // Propagate this object
360                                                                        if ( add_new_points_to_tuple(pts, pe, obj, qn) )
361                                                                                added = true;
362                                                                }
363                                                                else
364                                                                        break;
365                                                                
366                                                                pe = pe.next;
367                                                        }
368                                                }
369                                                
370                                                pts = pts.next;
371                                        }
372                                }
373                        }
374                        
375                        if ( added )
376                                worklist.push( qn );
377                        
378                        // Now, we clean the new edges if necessary
379                        if ( has_new_edges ) {
380                                pim1.flush();
381                        }
382                }
383        }

420         * Query if this pointer and qv could point to the same object under any contexts
421         */
422        @Override
423        public boolean heap_sensitive_intersection(IVarAbstraction qv) 
424        {
425                int i, j;
426                PtInsNode qn;
427                SegmentNode p, q, pt[], qt[];
428                
429                qn = (PtInsNode)qv;
430                
431                for (Iterator<AllocNode> it = pt_objs.keySet().iterator(); it.hasNext();) {
432                        AllocNode an = it.next();
433                        if ( an instanceof StringConstantNode ) continue;
434                        qt = qn.find_points_to(an);
435                        if (qt == null) continue;
436                        pt = find_points_to(an);
437
438                        for (i = 0; i < PtInsIntervalManager.Divisions; ++i) {
439                                p = pt[i];
440                                while (p != null) {
441                                        for (j = 0; j < PtInsIntervalManager.Divisions; ++j) {
442                                                q = qt[j];
443                                                while (q != null) {
444                                                        if (quick_intersecting_test(p, q))
445                                                                return true;
446                                                        q = q.next;
447                                                }
448                                        }
449                                        p = p.next;
450                                }
451                        }
452                }
453
454                return false;
455        }

486        @Override
487        public boolean pointer_interval_points_to(long l, long r, AllocNode obj) 
488        {
489                SegmentNode[] int_entry = find_points_to(obj);
490                
491                // Check all-to-many figures
492                if ( int_entry[PtInsIntervalManager.ALL_TO_MANY] != null ) return true;
493                
494                for ( int i = 1; i < HeapInsIntervalManager.Divisions; ++i ) {
495                        SegmentNode p = int_entry[i];
496                        while ( p != null ) {
497                                long R = p.I1 + p.L;
498                                if ( (l <= p.I1 && p.I1 < r) || ( p.I1 <= l && l < R) )
499                                        return true;
500                                p = p.next;
501                        }
502                }
503                
504                return false;
505        }

540        @Override
541        public void get_all_context_sensitive_objects(long l, long r, PtSensVisitor visitor) 
542        {
543                if ( parent != this ) {
544                        getRepresentative().get_all_context_sensitive_objects(l, r, visitor);
545                        return;
546                }
547                
548                GeomPointsTo geomPTA = (GeomPointsTo)Scene.v().getPointsToAnalysis();
549                
550                for ( Map.Entry<AllocNode, PtInsIntervalManager> entry : pt_objs.entrySet() ) {
551                        AllocNode obj = entry.getKey();
552                        PtInsIntervalManager im = entry.getValue();
553                        SegmentNode[] int_entry = im.getFigures();
554                        
555                        // We first get the 1-CFA contexts for the object
556                        SootMethod sm = obj.getMethod();
557                        int sm_int = 0;
558                        long n_contexts = 1;
559                        if ( sm != null ) {
560                                sm_int = geomPTA.getIDFromSootMethod(sm);
561                                n_contexts = geomPTA.context_size[sm_int];
562                        }
563                        
564                        // We search for all the pointers falling in the range [1, r) that may point to this object
565                        for ( int i = 0; i < PtInsIntervalManager.Divisions; ++i ) {
566                                SegmentNode p = int_entry[i];
567                                while ( p != null ) {
568                                        long R = p.I1 + p.L;
569                                        long objL = -1, objR = -1;
570                                        
571                                        // Now we compute which context sensitive objects are pointed to by this pointer
572                                        if ( i == PtInsIntervalManager.ALL_TO_MANY ) {
573                                                // all-to-many figures
574                                                objL = p.I2;
575                                                objR = p.I2 + p.L;
576                                        }
577                                        else {
578                                                // We compute the intersection
579                                                if ( l <= p.I1 && p.I1 < r ) {    
580                                                        if ( i != PtInsIntervalManager.MANY_TO_ALL ) {
581                                                                long d = r - p.I1;
582                                                                if ( d > p.L ) d = p.L;
583                                                                objL = p.I2;
584                                                                objR = objL + d;
585                                                        }
586                                                        else {
587                                                                objL = 1;
588                                                                objR = 1 + n_contexts;
589                                                        }
590                                                }
591                                                else if (p.I1 <= l && l < R) {
592                                                        if ( i != PtInsIntervalManager.MANY_TO_ALL ) {
593                                                                long d = R - l;
594                                                                if ( R > r ) d = r - l;
595                                                                objL = p.I2 + l - p.I1;
596                                                                objR = objL + d;
597                                                        }
598                                                        else {
599                                                                objL = 1;
600                                                                objR = 1 + n_contexts;
601                                                        }
602                                                }
603                                        }
604                                        
605                                        // Now we test which context versions should this interval [objL, objR) maps to
606                                        if ( objL != -1 && objR != -1 )
607                                                visitor.visit(obj, objL, objR, sm_int);
608                                        
609                                        p = p.next; 
610                                }
611                        }
612                }
613        }

669        private boolean addPointsTo( int code, AllocNode obj )
670        {
671                PtInsIntervalManager im = pt_objs.get(obj);
672                
673                if ( im == null ) {
674                        im = new PtInsIntervalManager();
675                        pt_objs.put(obj, im);
676                }
677                else if ( im == deadManager ) {
678                        // We preclude the propagation of this object
679                        return false;
680                }
681                
682                // pres has been filled properly before calling this method
683                if ( im.addNewFigure(code, pres) != null ) {
684                        new_pts.put(obj, im);
685                        return true;
686                }
687                
688                return false;
689        }

691        private boolean addFlowsTo( int code, PtInsNode qv )
692        {
693                PtInsIntervalManager im = flowto.get(qv);
694                
695                if ( im == null ) {
696                        im = new PtInsIntervalManager();
697                        flowto.put(qv, im);
698                }
699                
700                // pres has been filled properly before calling this method
701                return im.addNewFigure(code, pres) != null;
702        }

704        // Implement the pointer assignment inference rules
705        private static boolean add_new_points_to_tuple( SegmentNode pts, SegmentNode pe, 
706                        AllocNode obj, PtInsNode qn )
707        {
708                long interI, interJ;
709                int code = 0;
710
711                // Special Cases
