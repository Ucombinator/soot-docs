061/**
062 * We provide a set of methods to evaluate the quality of geometric points-to
063 * analysis. The evaluation methods are:
064 * 
065 * 1. Count the basic points-to information, such as average points-to set size,
066 * constraints evaluation graph size, etc; 
067 * 2. Virtual function resolution
068 * comparison; 
069 * 3. Static casts checking; 
070 * 4. All pairs alias analysis; 
071 * 5. Building heap graph (not used yet).
072 * 
073 * @author xiao
074 * 
075 */
076public class GeomEvaluator {
077
078        private GeomPointsTo ptsProvider;
079        private PrintStream outputer;
080        private EvalResults evalRes;
081        private boolean solved; // Used in the anonymous class visitor
082
083        public GeomEvaluator(GeomPointsTo gpts, PrintStream ps) 
084        {
085                ptsProvider = gpts;
086                outputer = ps;
087                evalRes = new EvalResults();
088        }

090        /**
091         * Collecting basic statistical information for SPARK.
092         */
093        public void profileSparkBasicMetrics()
094        {
095                int n_legal_var = 0;
096                
097                int[] limits = new int[] { 1, 5, 10, 25, 50, 75, 100 };
098                evalRes.pts_size_bar_spark = new Histogram(limits);
099
100                for (IVarAbstraction pn : ptsProvider.pointers) {
101                        // We don't consider exception pointers
102                        Node var = pn.getWrappedNode();
103                        if (ptsProvider.isExceptionPointer(var))
104                                continue;
105                        
106                        ++n_legal_var;
107
108                        int size = var.getP2Set().size();
109                        evalRes.pts_size_bar_spark.addNumber(size);
110                        evalRes.total_spark_pts += size;
111                        if (size > evalRes.max_pts_spark)
112                                evalRes.max_pts_spark = size;
113                }
114                
115                evalRes.avg_spark_pts = (double) evalRes.total_spark_pts / n_legal_var;
116        }

118        /**
119         * Summarize the geometric points-to analysis and report the basic metrics.
120         */
121        public void profileGeomBasicMetrics(boolean testSpark) 
122        {
123                int n_legal_var = 0, n_alloc_dot_fields = 0;
124                
125                int[] limits = new int[] { 1, 5, 10, 25, 50, 75, 100 };
126                evalRes.pts_size_bar_geom = new Histogram(limits);
127                
128                if ( testSpark ) {
129                        evalRes.total_spark_pts = 0;
130                        evalRes.max_pts_spark = 0;
131                        evalRes.pts_size_bar_spark = new Histogram(limits);
132                }
133                
134                // We first count the LOC
135                for (SootMethod sm : ptsProvider.getAllReachableMethods()) {
136                        if (!sm.isConcrete())
137                                continue;
138                        if (!sm.hasActiveBody()) {
139                                sm.retrieveActiveBody();
140                        }
141
142                        evalRes.loc += sm.getActiveBody().getUnits().size();
143                }
144                                
145                for (IVarAbstraction pn : ptsProvider.pointers) {
146                        // We don't consider those un-processed pointers because their
147                        // points-to information is equivalent to SPARK
148                        if (!pn.hasPTResult())
149                                continue;
150                        
151                        pn = pn.getRepresentative();
152                        Node var = pn.getWrappedNode();
153                        
154                        if (ptsProvider.isExceptionPointer(var))
155                                continue;
156                        
157                        if (var instanceof AllocDotField) {
158                                ++n_alloc_dot_fields;
159                        }
160                        ++n_legal_var;
161                        
162                        // ...spark
163                        int size;
164                        
165                        if ( testSpark ) {
166                                size = var.getP2Set().size();
167                                evalRes.pts_size_bar_spark.addNumber(size);
168                                evalRes.total_spark_pts += size;
169                                if (size > evalRes.max_pts_spark)
170                                        evalRes.max_pts_spark = size;
171                        }
172                        
173                        // ...geom
174                        size = pn.num_of_diff_objs();
175                        evalRes.pts_size_bar_geom.addNumber(size);
176                        evalRes.total_geom_ins_pts += size;
177                        if (size > evalRes.max_pts_geom)
178                                evalRes.max_pts_geom = size;
179                }
180
181                evalRes.avg_geom_ins_pts = (double) evalRes.total_geom_ins_pts / n_legal_var;
182                if ( testSpark )
183                        evalRes.avg_spark_pts = (double) evalRes.total_spark_pts / n_legal_var;
184                
185                outputer.println("");
186                outputer.println("----------Statistical Result of geomPTA <Data Format: geomPTA (SPARK)>----------");
187                outputer.printf("Lines of code (jimple): %.1fK\n", (double) evalRes.loc / 1000);
188                outputer.printf("Reachable Methods: %d (%d)\n",
189                                ptsProvider.getNumberOfMethods(),
190                                ptsProvider.getNumberOfSparkMethods());
191                outputer.printf("Reachable User Methods: %d (%d)\n",
192                                ptsProvider.n_reach_user_methods,
193                                ptsProvider.n_reach_spark_user_methods);
194                outputer.println("#All Pointers: "
195                                + ptsProvider.getNumberOfPointers());
196                outputer.println("#Core Pointers: " + n_legal_var
197                                + ", in which #AllocDot Fields: " + n_alloc_dot_fields);
198                outputer.printf(
199                                "Total/Average Projected Points-to Tuples [core pointers]: %d (%d) / %.3f (%.3f) \n",
200                                evalRes.total_geom_ins_pts, evalRes.total_spark_pts,
201                                evalRes.avg_geom_ins_pts,
202                                evalRes.avg_spark_pts);
203                outputer.println("The largest points-to set size [core pointers]: "
204                                + evalRes.max_pts_geom + " (" + evalRes.max_pts_spark + ")");
205
206                outputer.println();
207                evalRes.pts_size_bar_geom.printResult(outputer,
208                                "Points-to Set Sizes Distribution [core pointers]:",
209                                evalRes.pts_size_bar_spark);
210        }

212        /**
213         * We assess the quality of building the 1-cfa call graph with the geometric
214         * points-to result.
215         */
216        private void test_1cfa_call_graph(LocalVarNode vn, SootMethod caller,
217                        SootMethod callee_signature, Histogram ce_range) 
218        {
219                long l, r;
220                IVarAbstraction pn = ptsProvider.findInternalNode(vn);
221                if ( pn == null ) return;
222                pn = pn.getRepresentative();
223                Set<SootMethod> tgts = new HashSet<SootMethod>();
224                Set<AllocNode> set = pn.get_all_points_to_objects();
225
226                LinkedList<CgEdge> list = ptsProvider.getCallEdgesInto(ptsProvider
227                                .getIDFromSootMethod(caller));
228
229                FastHierarchy hierarchy = Scene.v().getOrMakeFastHierarchy();
230                
231                for (Iterator<CgEdge> it = list.iterator(); it.hasNext();) {
232                        CgEdge p = it.next();
233
234                        l = p.map_offset;
235                        r = l + ptsProvider.max_context_size_block[p.s];
236                        tgts.clear();
237
238                        for (AllocNode obj : set) {
239                                if (!pn.pointer_interval_points_to(l, r, obj))
240                                        continue;
241
242                                Type t = obj.getType();
243
244                                if (t == null)
245                                        continue;
246                                else if (t instanceof AnySubType)
247                                        t = ((AnySubType) t).getBase();
248                                else if (t instanceof ArrayType)
249                                        t = RefType.v("java.lang.Object");
250
251                                try {
252                                        tgts.add(hierarchy.resolveConcreteDispatch(
253                                                        ((RefType) t).getSootClass(), callee_signature));
254                                } catch (Exception e) {
255
256                                }
257                        }
258
259                        tgts.remove(null);
260                        ce_range.addNumber(tgts.size());
261                }
262        }

264        /**
265         * Report the virtual callsites resolution result for the user's code.
266         */
267        public void checkCallGraph() 
268        {
269                int[] limits = new int[] { 1, 2, 4, 8 };
270                evalRes.total_call_edges = new Histogram(limits);
271
272                CallGraph cg = Scene.v().getCallGraph();
273
274                for ( Stmt callsite : ptsProvider.multiCallsites ) {
275                        Iterator<Edge> edges = cg.edgesOutOf(callsite);
276                        if ( !edges.hasNext() ) continue;
277                        evalRes.n_callsites++;
278                        
279                        // get an edge
280                        Edge anyEdge = edges.next();
281                        SootMethod src = anyEdge.src();
282                        
283                        if ( !ptsProvider.isReachableMethod(src) ||
284                                        !ptsProvider.isValidMethod(src) )
285                                continue;
286                        
287                        // get the base pointer
288                        CgEdge p = ptsProvider.getInternalEdgeFromSootEdge(anyEdge);
289                        LocalVarNode vn = (LocalVarNode)p.base_var;
290                        
291                        // test the call graph
292                        int edge_cnt = 1;
293                        while ( edges.hasNext() ) {
294                                ++edge_cnt;
295                                edges.next();
296                        }
297                        evalRes.n_geom_call_edges += edge_cnt;
298                        if ( edge_cnt == 1 ) ++evalRes.n_geom_solved_all;
299                        
300                        // test app method
301                        if ( !src.isJavaLibraryMethod() ) {
302                                InvokeExpr ie = callsite.getInvokeExpr();
303                                
304                                if ( edge_cnt == 1 ) {
305                                        ++evalRes.n_geom_solved_app;
306                                        
307                                        if ( ptsProvider.getOpts().verbose() ) {
308                                                outputer.println();
309                                                outputer.println("<<<<<<<<<   Additional Solved Call   >>>>>>>>>>");
310                                                outputer.println(src.toString());
311                                                outputer.println(ie.toString());
312                                        }
313                                } else {
314                                        // We try to test if this callsite is solvable
315                                        // under some contexts
316                                        Histogram call_edges = new Histogram(limits);
317                                        test_1cfa_call_graph(vn, src, ie.getMethod(), call_edges);
318                                        evalRes.total_call_edges.merge(call_edges);
319                                        call_edges = null;
320                                }
321                                
322                                evalRes.n_geom_user_edges += edge_cnt;
323                                evalRes.n_user_callsites++;
324                        }
325                }
326
327                ptsProvider.ps.println();
328                ptsProvider.ps.println("--------> Virtual Callsites Evaluation <---------");
329                ptsProvider.ps.printf("Total virtual callsites (app code): %d (%d)\n", 
330                                evalRes.n_callsites, evalRes.n_user_callsites);
331                ptsProvider.ps.printf("Total virtual call edges (app code): %d (%d)\n", 
332                                evalRes.n_geom_call_edges, evalRes.n_geom_user_edges);
333                ptsProvider.ps.printf("Virtual callsites additionally solved by geomPTA compared to SPARK (app code) = %d (%d)\n", 
334                                evalRes.n_geom_solved_all, evalRes.n_geom_solved_app);
335                evalRes.total_call_edges.printResult(ptsProvider.ps,
336                                "Testing of unsolved callsites on 1-CFA call graph: ");
337
338                if (ptsProvider.getOpts().verbose())
339                        ptsProvider.outputNotEvaluatedMethods();
340        }

342        /**
343         * Count how many aliased base pointers appeared in all user's functions.
344         */
345        public void checkAliasAnalysis() 
346        {
347                Set<IVarAbstraction> access_expr = new HashSet<IVarAbstraction>();
348                ArrayList<IVarAbstraction> al = new ArrayList<IVarAbstraction>();
349                Value[] values = new Value[2];
350
351                for (SootMethod sm : ptsProvider.getAllReachableMethods()) {
352                        if (sm.isJavaLibraryMethod())
353                                continue;
354                        if (!sm.isConcrete())
355                                continue;
356                        if (!sm.hasActiveBody()) {
357                                sm.retrieveActiveBody();
358                        }
359                        if (!ptsProvider.isValidMethod(sm))
360                                continue;
361
362                        // We first gather all the pointers
363                        // access_expr.clear();
364                        for (Iterator<Unit> stmts = sm.getActiveBody().getUnits().iterator(); stmts
365                                        .hasNext();) {
366                                Stmt st = (Stmt) stmts.next();
367
368                                if (st instanceof AssignStmt) {
369                                        AssignStmt a = (AssignStmt) st;
370                                        values[0] = a.getLeftOp();
371                                        values[1] = a.getRightOp();
372
373                                        for (Value v : values) {
374                                                // We only care those pointers p involving in the
375                                                // expression: p.f
376                                                if (v instanceof InstanceFieldRef) {
377                                                        InstanceFieldRef ifr = (InstanceFieldRef) v;
378                                                        final SootField field = ifr.getField();
379                                                        if (!(field.getType() instanceof RefType))
380                                                                continue;
381
382                                                        LocalVarNode vn = ptsProvider
383                                                                        .findLocalVarNode((Local) ifr.getBase());
384                                                        if (vn == null)
385                                                                continue;
386                                                        
387                                                        if ( ptsProvider.isExceptionPointer(vn) )
388                                                                continue;
389                                                        
390                                                        IVarAbstraction pn = ptsProvider.findInternalNode(vn);
391                                                        if ( pn == null ) continue;
392                                                        pn = pn.getRepresentative();
393                                                        if (pn.hasPTResult()) access_expr.add(pn);
394                                                }
395                                        }
396                                }
397                        }
398                }
399
400                access_expr.remove(null);
401                al.addAll(access_expr);
402                access_expr = null;
403                
404                // Next, we pair up all the pointers
405                Date begin = new Date();
406                int size = al.size();
407
408                for (int i = 0; i < size; ++i) {
409                        IVarAbstraction pn = al.get(i);
410                        VarNode n1 = (VarNode) pn.getWrappedNode();
411
412                        for (int j = i + 1; j < size; ++j) {
413                                IVarAbstraction qn = al.get(j);
414                                VarNode n2 = (VarNode) qn.getWrappedNode();
415
416                                if (pn.heap_sensitive_intersection(qn))
417                                        evalRes.n_hs_alias++;
418
419                                // We directly use the SPARK points-to sets
420                                if (n1.getP2Set().hasNonEmptyIntersection(n2.getP2Set()))
421                                        evalRes.n_hi_alias++;
422                        }
423                }
424                
425                evalRes.n_alias_pairs = size * (size - 1) / 2;
426                Date end = new Date();
427
428                ptsProvider.ps.println();
429                ptsProvider.ps.println("--------> Alias Pairs Evaluation <---------");
430                ptsProvider.ps.println("Number of pointer pairs in app code: " + evalRes.n_alias_pairs);
431                ptsProvider.ps
432                                .printf("Heap sensitive alias pairs (by Geom): %d, Percentage = %.3f%%\n",
433                                                evalRes.n_hs_alias, (double) evalRes.n_hs_alias / evalRes.n_alias_pairs * 100);
434                ptsProvider.ps
435                                .printf("Heap insensitive alias pairs (by SPARK): %d, Percentage = %.3f%%\n",
436                                                evalRes.n_hi_alias, (double) evalRes.n_hi_alias / evalRes.n_alias_pairs * 100);
437                ptsProvider.ps.printf("Using time: %dms \n",
438                                end.getTime() - begin.getTime());
439                ptsProvider.ps.println();
440        }

442        /**
443         * Count how many static casts can be determined safe.
444         */
445        public void checkCastsSafety() 
446        {
447                
448
449                for (SootMethod sm : ptsProvider.getAllReachableMethods()) {
450                        if (sm.isJavaLibraryMethod())
451                                continue;
452                        if (!sm.isConcrete())
453                                continue;
454                        if (!sm.hasActiveBody()) {
455                                sm.retrieveActiveBody();
456                        }
457                        if (!ptsProvider.isValidMethod(sm))
458                                continue;
459
460                        // All the statements in the method
461                        for (Iterator<Unit> stmts = sm.getActiveBody().getUnits().iterator(); stmts
462                                        .hasNext();) {
463                                Stmt st = (Stmt) stmts.next();
464
465                                if (st instanceof AssignStmt) {
466                                        Value rhs = ((AssignStmt) st).getRightOp();
467                                        Value lhs = ((AssignStmt) st).getLeftOp();
468                                        if (rhs instanceof CastExpr
469                                                        && lhs.getType() instanceof RefLikeType) {
470
471                                                Value v = ((CastExpr) rhs).getOp();
472                                                VarNode node = ptsProvider.findLocalVarNode(v);
473                                                if (node == null)
474                                                        continue;
475                                                IVarAbstraction pn = ptsProvider.findInternalNode(node);
476                                                if (pn == null)
477                                                        continue;
478
479                                                pn = pn.getRepresentative();
480                                                if ( !pn.hasPTResult() ) continue;
481                                                
482                                                evalRes.total_casts++;
483                                                final Type targetType = 
484                                                                (RefLikeType) ((CastExpr) rhs).getCastType();
485
486                                                // We first use the geometric points-to result to
487                                                // evaluate
488                                                solved = true;
489                                                Set<AllocNode> set = pn.get_all_points_to_objects();
490                                                for (AllocNode obj : set) {
491                                                        solved = ptsProvider.castNeverFails(obj.getType(),
492                                                                        targetType);
493                                                        if (solved == false)
494                                                                break;
495                                                }
496
497                                                if (solved)
498                                                        evalRes.geom_solved_casts++;
499
500                                                // Second is the SPARK result
501                                                solved = true;
502                                                node.getP2Set().forall(new P2SetVisitor() {
503                                                        public void visit(Node arg0) {
504                                                                if (solved == false)
505                                                                        return;
506                                                                solved = ptsProvider.castNeverFails(
507                                                                                arg0.getType(), targetType);
508                                                        }
509                                                });
510
511                                                if (solved)
512                                                        evalRes.spark_solved_casts++;
513                                        }
514                                }
515                        }
516                }
517
518                ptsProvider.ps.println();
519                ptsProvider.ps
520                                .println("-----------> Static Casts Safety Evaluation <------------");
521                ptsProvider.ps.println("Total casts (app code): " + evalRes.total_casts);
522                ptsProvider.ps.println("Safe casts: Geom = " + evalRes.geom_solved_casts
523                                + ", SPARK = " + evalRes.spark_solved_casts);
524        }

526        /**
527         * Estimate the size of the def-use graph for the heap memory. The heap
528         * graph is estimated without context information.
529         */
530        public void estimateHeapDefuseGraph() 
531        {
532                final Map<IVarAbstraction, int[]> defUseCounterForGeom = new HashMap<IVarAbstraction, int[]>();
533                final Map<AllocDotField, int[]> defUseCounterForSpark = new HashMap<AllocDotField, int[]>();
534
535                Date begin = new Date();
536
537                for (SootMethod sm : ptsProvider.getAllReachableMethods()) {
538                        if (sm.isJavaLibraryMethod())
539                                continue;
540                        if (!sm.isConcrete())
541                                continue;
542                        if (!sm.hasActiveBody()) {
543                                sm.retrieveActiveBody();
544                        }
545                        if (!ptsProvider.isValidMethod(sm))
546                                continue;
547
548                        // We first gather all the memory access expressions
549                        for (Iterator<Unit> stmts = sm.getActiveBody().getUnits().iterator(); stmts
550                                        .hasNext();) {
551                                Stmt st = (Stmt) stmts.next();
552
553                                if (!(st instanceof AssignStmt))
554                                        continue;
555
556                                AssignStmt a = (AssignStmt) st;
557                                final Value lValue = a.getLeftOp();
558                                final Value rValue = a.getRightOp();
559
560                                InstanceFieldRef ifr = null;
561
562                                if (lValue instanceof InstanceFieldRef) {
563                                        // Def statement
564                                        ifr = (InstanceFieldRef) lValue;
565                                } else if (rValue instanceof InstanceFieldRef) {
566                                        // Use statement
567                                        ifr = (InstanceFieldRef) rValue;
568                                }
569
570                                if (ifr != null) {
571                                        final SootField field = ifr.getField();
572
573                                        LocalVarNode vn = ptsProvider.findLocalVarNode((Local) ifr
574                                                        .getBase());
575                                        if (vn == null)
576                                                continue;
577                                        IVarAbstraction pn = ptsProvider.findInternalNode(vn);
578                                        if (pn == null)
579                                                continue;
580                                        pn = pn.getRepresentative();
581                                        if (!pn.hasPTResult())
582                                                continue;
583
584                                        // Spark
585                                        vn.getP2Set().forall(new P2SetVisitor() {
586
587                                                @Override
588                                                public void visit(Node n) {
589                                                        IVarAbstraction padf = ptsProvider
590                                                                        .findAndInsertInstanceField((AllocNode) n,
591                                                                                        field);
592                                                        AllocDotField adf = (AllocDotField) padf
593                                                                        .getWrappedNode();
594                                                        int[] defUseUnit = defUseCounterForSpark.get(adf);
595                                                        if (defUseUnit == null) {
596                                                                defUseUnit = new int[2];
597                                                                defUseCounterForSpark.put(adf, defUseUnit);
598                                                        }
599
600                                                        if (lValue instanceof InstanceFieldRef) {
601                                                                defUseUnit[0]++;
602                                                        } else {
603                                                                defUseUnit[1]++;
604                                                        }
605                                                }
606                                        });
607
608                                        // Geom
609
610                                        Set<AllocNode> objsSet = pn.get_all_points_to_objects();
611                                        for (AllocNode obj : objsSet) {
612                                                /*
613                                                 * We will create a lot of instance fields. Because in
614                                                 * points-to analysis, we concern only the reference
615                                                 * type fields. But here, we concern all the fields read
616                                                 * write including the primitive type fields.
617                                                 */
618                                                IVarAbstraction padf = ptsProvider
619                                                                .findAndInsertInstanceField(obj, field);
620                                                int[] defUseUnit = defUseCounterForGeom.get(padf);
621                                                if (defUseUnit == null) {
622                                                        defUseUnit = new int[2];
623                                                        defUseCounterForGeom.put(padf, defUseUnit);
624                                                }
625
626                                                if (lValue instanceof InstanceFieldRef) {
627                                                        defUseUnit[0]++;
628                                                } else {
629                                                        defUseUnit[1]++;
630                                                }
631                                        }
632                                }
633                        }
634                }
635
636                for (int[] defUseUnit : defUseCounterForSpark.values()) {
637                        evalRes.n_spark_du_pairs += ((long) defUseUnit[0]) * defUseUnit[1];
638                }
639
640                for (int[] defUseUnit : defUseCounterForGeom.values()) {
641                        evalRes.n_geom_du_pairs += ((long) defUseUnit[0]) * defUseUnit[1];
642                }
643
644                Date end = new Date();
645
646                ptsProvider.ps.println();
647                ptsProvider.ps
648                                .println("-----------> Heap Def Use Graph Evaluation <------------");
649                ptsProvider.ps
650                                .println("The edges in the heap def-use graph is (by Geom): "
651                                                + evalRes.n_geom_du_pairs);
652                ptsProvider.ps
653                                .println("The edges in the heap def-use graph is (by Spark): "
654                                                + evalRes.n_spark_du_pairs);
655                ptsProvider.ps.printf("Using time: %dms \n",

658        }
