019package soot.jimple.spark.geom.heapinsE;

021import soot.jimple.spark.geom.dataRep.RectangleNode;
022import soot.jimple.spark.geom.dataRep.SegmentNode;
023import soot.jimple.spark.geom.geomPA.Constants;
024import soot.jimple.spark.geom.geomPA.IFigureManager;

026/**
027 * An abstraction for the management of all the heap insensitive encoding figures.
028 * We employ the naive management strategy, which is a linked list based manager.
029 * 
030 * For HeapIns analysis, we have four types of figures:
031 * Type                 |               Index
032 * all-to-many  |               0
033 * many-to-all  |               1
034 * one-to-one   |               2
035 * all-to-all
036 * all-to-all is special because whenever it was presented, all others can be deleted (its semantics is context insensitive).
037 * The corresponding index means header[0] stores all the all-to-one figures, and so on.
038 * 
039 * @author xiao
040 *
041 */
042public class HeapInsIntervalManager extends IFigureManager {
043        public static int Divisions = 3;
044        
045        // Type IDs for the figures
046        public static final int ALL_TO_ALL = -1;                // A special case
047        public static final int ALL_TO_MANY = 0;
048        public static final int MANY_TO_ALL = 1;
049        public static final int ONE_TO_ONE = 2;
050        
051        // Recording the size of each type of figure
052        private int size[] = {0, 0, 0};
053        // Recording the geometric figures, categorized by the type IDs.
054        private SegmentNode header[] = {null, null, null};
055        private boolean hasNewFigure = false;
056        
057        
058        public SegmentNode[] getFigures()
059        {
060                return header;
061        }
062        
063        public int[] getSizes()
064        {
065                return size;
066        }
067        
068        public boolean isThereUnprocessedFigures()
069        {
070                return hasNewFigure;
071        }
072        
073        public void flush()
074        {
075                hasNewFigure = false;
076                
077                for ( int i = 0; i < Divisions; ++i ) {
078                        SegmentNode p = header[i];
079                        while ( p != null && p.is_new == true ) {
080                                p.is_new = false;
081                                p = p.next;
082                        }
083                }
084        }
085        
086        /**
087         * Delete all the shapes recorded.
088         */
089        public void clear()
090        {
091                for ( int i = 0; i < Divisions; ++i ) {
092                        size[i] = 0;
093                        header[i] = null;
094                }
095                
096                hasNewFigure = false;
097        }
098        
099        /*
100         * pnew.L < 0 is a special case we used to indicate a square: L = L_prime
101         * This case is specially handled because it is very common in the program.
102         * And, treating it as a MANY-TO-ALL is loss of precision.
103         */
104        public SegmentNode addNewFigure( int code, RectangleNode pnew )
105        {
106                SegmentNode p;
107                
108                if ( code == ALL_TO_ALL ) {
109                        // The input figure is a all-to-all figure
110                        // Directly clean all the existing intervals unless the all-to-all figure is existing.
111                        if ( header[ALL_TO_MANY] != null && 
112                                        header[ALL_TO_MANY].I2 == 0 )
113                                return null;
114                        
115                        p = new SegmentNode();
116                        
117                        code = ALL_TO_MANY;
118                        p.I1 = p.I2 = 0;
119                        p.L = Constants.MAX_CONTEXTS;
120                        for ( int i = 0; i < Divisions; ++i ) {
121                                size[i] = 0;
122                                header[i] = null;
123                        }
124                }
125                else {
126                        // Before inserting into the figure list, we do duplicate testing
127                        
128                        // This is a all-to-many or one-to-one figure
129                        if ( code == ALL_TO_MANY || code == ONE_TO_ONE ) {
130                                p = header[ALL_TO_MANY];
131                                while ( p != null ) {
132                                        if ( (p.I2 <= pnew.I2) && (p.I2 + p.L >= pnew.I2 + pnew.L) )
133                                                return null;
134                                        p = p.next;
135                                }
136                        }
137                        
138                        // This is a many-to-all or one-to-one figure
139                        if ( code == MANY_TO_ALL || code == ONE_TO_ONE ) {
140                                p = header[MANY_TO_ALL];
141                                while ( p != null ) {
142                                        if ( (p.I1 <= pnew.I1) && (p.I1 + p.L >= pnew.I1 + pnew.L) )
143                                                return null;
144                                        p = p.next;
145                                }
146                        }
147                        
148                        // This is a one-to-one figure
149                        if ( code == ONE_TO_ONE ) {
150                                p = header[ONE_TO_ONE];
151                                while ( p != null ) {
152                                        // We don't process the case: the input figure is a square but the tested figure is a segment
153                                        if ( p.I1 - p.I2 == pnew.I1 - pnew.I2 ) {
154                                                // On the same line
155                                                if ( p.I1 <= pnew.I1 && p.I1 + p.L >= pnew.I1 + pnew.L )
156                                                        return null;
157                                        }
158                                        
159                                        p = p.next;
160                                }
161                        }
162        
163                        // Insert the new interval immediately, and we delay the merging until necessary
164                        p = new SegmentNode( pnew );
165                        
166                        if ( code == ALL_TO_MANY )
167                                clean_garbage_all_to_many(p);
168                        else if ( code == MANY_TO_ALL )
169                                clean_garbage_many_to_all(p);
170                        else
171                                clean_garbage_one_to_one(p);
172                }
173                
174                hasNewFigure = true;
175                size[code]++;
176                p.next = header[code];
177                header[code] = p;
178                return p;
179        }
180        
181        // This function tries to do the geometric merging
182        public void mergeFigures(int upperSize)
183        {
184                if ( !hasNewFigure ) return;
185                
186                /*
187                 * We start the merging from ONE_TO_ONE, because the generated figure may be merged with those figures in MANY_TO_ALL
188                 */
189                if ( size[ONE_TO_ONE] > upperSize && 
190                                header[ONE_TO_ONE].is_new == true ) {
191                        
192                        // We prefer to generate a heap insensitive figure
193                        SegmentNode p = generate_many_to_all( header[ONE_TO_ONE] );
194                        clean_garbage_many_to_all(p);
195                        p.next = header[MANY_TO_ALL];
196                        header[MANY_TO_ALL] = p;
197                        header[ONE_TO_ONE] = null;
198                        size[MANY_TO_ALL]++;
199                        size[ONE_TO_ONE] = 0;
200                }
201
202                if ( size[MANY_TO_ALL] > upperSize &&
203                                header[MANY_TO_ALL].is_new == true ) {
204                        
205                        header[MANY_TO_ALL] = generate_many_to_all( header[MANY_TO_ALL] );
206                        size[MANY_TO_ALL] = 1;
207                }
208                
209                if ( size[ALL_TO_MANY] > upperSize &&
210                                header[ALL_TO_MANY].is_new == true ) {
211                        
212                        header[ALL_TO_MANY] = generate_all_to_many( header[ALL_TO_MANY] );
213                        size[ALL_TO_MANY] = 1;
214                }
215        }
216        
217        public void removeUselessSegments()
218        {
219                int i;
220                SegmentNode p, q, temp;
221                
222                p = header[ONE_TO_ONE];
223                size[ONE_TO_ONE] = 0;
224                q = null;
225                
226                while ( p != null ) {
227                        boolean contained = false;
228                        long L = p.L;
229
230                        for ( i = 0; i < 2; ++i ) {
231                                temp = header[i];
232                                while ( temp != null ) {
233                                        if ( temp.I1 == 0 || ((temp.I1 <= p.I1) && (temp.I1 + temp.L >= p.I1 + L)) ) {
234                                                if (temp.I2 == 0
235                                                                || ((temp.I2 <= p.I2) && (temp.I2 + temp.L >= p.I2 + L))) {
236                                                        contained = true;
237                                                        break;
238                                                }
239                                        }
240                                        
241                                        temp = temp.next;
242                                }
243                        }
244                        
245                        temp = p.next;
246                        if ( contained == false ) {
247                                p.next = q;
248                                q = p;
249                                ++size[ONE_TO_ONE];
250                        }
251                        p = temp;
252                }
253                
254                header[ONE_TO_ONE] = q;
255        }

257        /**
258         * Merge all the ONE_TO_ONE figures pointed to by mp.
259         * The result is in the form (p, q, 0, I, L).
260         */
261        private SegmentNode generate_all_to_many( SegmentNode mp ) 
262        {
263                long left, right;
264                SegmentNode p;
265
266                left = mp.I2;
267                right = left + mp.L;
268                p = mp.next;
269
270                while (p != null) {
271                        if (p.I2 < left) left = p.I2;
272                        long t = p.I2 + p.L;
273                        if ( t > right ) right = t;
274                        p = p.next;
275                }
276
277                // We reuse the first element in the list mp
278                mp.I1 = 0;
279                mp.I2 = left;
280                mp.L = right - left;
281                mp.next = null;
282
283                return mp;
284        }
285        
286        /** 
287         * The result is in the form: (p, q, I, 0, L)
288         */
289        private SegmentNode generate_many_to_all( SegmentNode mp ) 
290        {
291                long left, right;
292                SegmentNode p;
293
294                left = mp.I1;
295                right = left + mp.L;
296                p = mp.next;
297
298                while (p != null) {
299                        if (p.I1 < left) left = p.I1;
300                        long t = p.I1 + p.L;
301                        if ( t > right ) right = t;
302                        p = p.next;
303                }
304                
305                mp.I1 = left;
306                mp.I2 = 0;
307                mp.L = right - left;
308                mp.next = null;
309                
310                return mp;
311        }

313        // Clean garbages in the MANY_TO_ALL list that the information is already covered by mp
314        // BTW, we also do simple adjacent figures concatenation 
315        private void clean_garbage_many_to_all(SegmentNode predator) 
316        {
317                SegmentNode p, q, list;
318                int num;
319                long right, left;
320
321                list = header[MANY_TO_ALL];
322                p = q = null;
323                num = 0;
324                left = predator.I1;
325                right = left + predator.L;
326
327                while (list != null) {
328                        
329                        // We first process the overlapped cases
330                        if (list.I1 >= left) {
331                                if (list.I1 <= right) {
332                                        if ( list.I1 + list.L > right ) {
333                                                // We extend predator to the right
334                                                right = list.I1 + list.L;
335                                        }
336                                        // else, this figure is completely contained in predator, we swallow it
337                                        
338                                        list = list.next;
339                                        continue;
340                                }
341                                //else, this figure has no overlap with the predator
342                        } else if (list.I1 + list.L >= left) {
343                                // We extend predator to the left
344                                left = list.I1;
345                                list = list.next;
346                                continue;
347                        }
348                        
349                        // No intersection, no overlap
350                        // Notice that, we have to preserve the order of the list
351                        // Because the newly inserted figures are headed at the list
352                        if ( q == null ) {
353                                p = q = list;
354                        }
355                        else {
356                                q.next = list;
357                                q = list;
358                        }
359                        
360                        ++num;
361                        list = list.next;
362                }
363
364                predator.I1 = left;
365                predator.L = right - left;
366                if ( q != null ) q.next = null;
367                header[MANY_TO_ALL] = p;
368                size[MANY_TO_ALL] = num;
369        }
370        
371        // Clean the ALL_TO_MANY list
372        private void clean_garbage_all_to_many(SegmentNode predator) 
373        {
374                SegmentNode p, q, list;
375                int num;
376                long right, left;
377
378                list = header[ALL_TO_MANY];
379                p = q = null;
380                num = 0;
381                left = predator.I2;
382                right = predator.I2 + predator.L;
383
384                while (list != null) {
385                        if (list.I2 >= left) {
386                                if (list.I2 <= right) {
387                                        if ( list.I2 + list.L > right ) {
388                                                // We extend predator to the right
389                                                right = list.I2 + list.L;
390                                        }
391
392                                        list = list.next;
393                                        continue;
394                                }
395                        } else if (list.I2 + list.L >= left) {
396                                // We extend predator to the left
397                                left = list.I2;
398                                list = list.next;
399                                continue;
400                        }
401
402                        // No intersection, no overlap
403                        // Notice that, we have to preserve the order of the list
404                        // Because the unprocessed points-to tuples are headed at the list
405                        if ( q == null ) {
406                                p = q = list;
407                        }
408                        else {
409                                q.next = list;
410                                q = list;
411                        }
412                        
413                        ++num;
414                        list = list.next;
415                }
416
417                predator.I2 = left;
418                predator.L = right - left;
419                if ( q != null ) q.next = null;
420                header[ALL_TO_MANY] = p;
421                size[ALL_TO_MANY] = num;
422        }
423        
424        /*
425         * Eliminate the redundant ONE_TO_ONE figures
426         */
427        private void clean_garbage_one_to_one(SegmentNode predator) 
428        {
429                SegmentNode p, q, list;
430                int num;
431                
432                list = header[ONE_TO_ONE];
433                p = q = null;
434                num = 0;
435
436                while (list != null) {
437                        long L = list.L;
438                        if ( (predator.I2 - predator.I1 == list.I2 - list.I1) && 
439                                        predator.I1 <= list.I1 && 
440                                        (predator.I1 + predator.L >= list.I2 + L) )
441                                // The checked figure is completely contained in the predator
442                                // So we ignore it
443                                ;
444                        else { 
445                                if ( q == null ) {
446                                        p = q = list;
447                                }
448                                else {
449                                        q.next = list;
450                                        q = list;
451                                }
452                                
453                                ++num;
454                        }
455                        
456                        list = list.next;
457                }
458
459                if ( q != null ) q.next = null;
460                header[ONE_TO_ONE] = p;
461                size[ONE_TO_ONE] = num;
462        }
