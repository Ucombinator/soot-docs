112        @Override
113        public void do_before_propagation() 
114        {
115                // We first perform the geometric merging
116                do_pts_interval_merge();
117                do_flow_edge_interval_merge();
118                
119                /*
120                 * The following code eliminates the spurious points-to relation for THIS pointer.
121                 * For example we have two classes A and B, B is a child class of A.
122                 * We have a virtual function foo defined in both A and B.
123                 * We have a pointer p in type A.
124                 * pts(p) = { o1, o2 }, where o1 is in type A and o2 is in type B.
125                 * Therefore, the call p.foo() will be resolved to call both A::foo and B::foo.
126                 * Then, in the points-to analysis, we have two assignments: p -> A::foo.THIS, p -> B::foo.THIS
127                 * At this time, obviously, although with the type filter, A::foo.THIS will receive the object o2, which is definitely a fake.
128                 * Thus, we need a new filter to guarantee that A::foo.THIS only points to o1.
129                 * We call this filter "this pointer filter".
130                 */
131                Node wrappedNode = getWrappedNode();
132                if ( wrappedNode instanceof LocalVarNode &&
133                                ((LocalVarNode)wrappedNode).isThisPtr() ) {
134                        SootMethod func = ((LocalVarNode)wrappedNode).getMethod();
135                        if ( !func.isConstructor() ) {
136                                // We don't process the specialinvoke call edge
137                                SootClass defClass = func.getDeclaringClass();
138                                Hierarchy typeHierarchy = Scene.v().getActiveHierarchy();
139                                
140                                for ( Iterator<AllocNode> it = new_pts.keySet().iterator(); it.hasNext(); ) {
141                                        AllocNode obj = it.next();
142                                        if ( obj.getType() instanceof RefType ) {
143                                                SootClass sc = ((RefType)obj.getType()).getSootClass();
144                                                if ( defClass != sc ) {
145                                                        try {
146                                                                SootMethod rt_func = typeHierarchy.resolveConcreteDispatch(sc, func);
147                                                                if ( rt_func != func ) {
148                                                                        it.remove();
149                                                                        // Also preclude it from propagation again
150                                                                        pt_objs.put(obj, (GeometricManager)deadManager);
151                                                                }
152                                                        }
153                                                        catch ( RuntimeException e ) {
154                                                                // If the input program has a wrong type cast, resolveConcreteDispatch fails and it goes here
155                                                                // We simply ignore this error
156                                                        }
157                                                }
158                                        }
159                                }
160                        }
161                }
162        }

265        @Override
266        public void propagate(GeomPointsTo ptAnalyzer, IWorklist worklist) 
267        {
268                int i, j;
269                AllocNode obj;
270                SegmentNode pts, pe, entry_pts[], entry_pe[];
271                GeometricManager gm1, gm2;
272                FullSensitiveNode qn, objn;
273                boolean added, hasNewPointsTo;
274
275                if ( pt_objs.size() == 0 ) return;
276//              System.err.println("+++ Process assignment for: " + toString());
277                
278                // We first build the flow edges that flow in to/out of object fields
279                if ( complex_cons != null ) {
280                        
281                        for ( Map.Entry<AllocNode, GeometricManager> entry : new_pts.entrySet() ) {
282                                obj = entry.getKey();
283                                entry_pts = entry.getValue().getFigures();
284                                
285                                for ( PlainConstraint pcons : complex_cons ) {
286                                        // For each newly points-to object, construct its instance field
287                                        objn = (FullSensitiveNode)ptAnalyzer.findInstanceField(obj, pcons.f);
288                                        if ( objn == null ) {
289                                                // This combination of allocdotfield must be invalid
290                                                // This expression p.f also renders that p cannot point to obj, so we remove it
291                                                // We label this event and sweep the garbage later
292                                                pt_objs.put(obj, (GeometricManager)deadManager);
293                                                entry.setValue( (GeometricManager)deadManager );
294                                                break;
295                                        }
296                                        
297                                        if ( objn.willUpdate == false ) {
298                                                // This must be a store constraint
299                                                // This object field is not need for computing 
300                                                // the points-to information of the seed pointers
301                                                continue;
302                                        }
303                                        
304                                        qn = (FullSensitiveNode) pcons.otherSide;
305//                                      System.err.println("--load/store-->: " + qn.toString());
306                                        
307                                        for ( i = 0; i < GeometricManager.Divisions; ++i ) {
308                                                pts = entry_pts[i];
309                                                
310                                                while ( pts != null && pts.is_new == true ) {
311                                                        switch ( pcons.type ) {
312                                                        case Constants.STORE_CONS:
313                                                                // Store, qv -> pv.field
314                                                                if ( instantiateStoreConstraint( qn, objn, pts, (pcons.code<<8) | i ) )
315                                                                        worklist.push( qn );
316                                                                break;
317                                                                
318                                                        case Constants.LOAD_CONS:
319                                                                // Load, pv.field -> qv
320                                                                if ( instantiateLoadConstraint( objn, qn, pts, (pcons.code<<8) | i ) )
321                                                                        worklist.push( objn );
322                                                                break;
323                                                        }
324                                                        
325                                                        pts = pts.next;
326                                                }
327                                        }
328                                }
329                        }
330                }
331                
332                if ( flowto.size() == 0 ) return;
333                
334                // Next, we process the assignments (e.g. p = q)
335                for (Map.Entry<FullSensitiveNode, GeometricManager> entry1 : flowto.entrySet()) { 
336                        added = false;
337                        qn = entry1.getKey();
338                        gm1 = entry1.getValue();
339                        entry_pe = gm1.getFigures();
340                        
341//                      System.err.println("-assign->: " + qn.toString());
342                        
343                        // We specialize the two cases that we hope it running faster
344                        // We have new flow-to edges
345                        if ( gm1.isThereUnprocessedFigures() ) {
346                                
347                                // Second, we get the points-to shapes
348                                for ( Map.Entry<AllocNode, GeometricManager> entry2 : pt_objs.entrySet() ) {
349                                        obj = entry2.getKey();
350                                        gm2 = entry2.getValue();
351                                        
352                                        // Avoid the garbage
353                                        if ( gm2 == deadManager ) continue;
354                                        
355                                        // Type filtering and flow-to-this filtering, a simple approach
356                                        if (!ptAnalyzer.castNeverFails(obj.getType(), qn.getType())) continue;
357                                        
358                                        entry_pts = gm2.getFigures();
359                                        hasNewPointsTo = gm2.isThereUnprocessedFigures();
360                                        
361                                        // We pair up all the geometric points-to tuples and flow edges
362                                        for ( j = 0; j < GeometricManager.Divisions; ++j ) {
363                                                pe = entry_pe[j];
364                                                                                                
365                                                while ( pe != null ) {
366                                                        if ( pe.is_new == false && hasNewPointsTo == false )
367                                                                break;
368                                                        
369                                                        for (i = 0; i < GeometricManager.Divisions ; ++i) {
370                                                                pts = entry_pts[i];
371                                                                
372                                                                while ( pts != null &&
373                                                                                ( pts.is_new || pe.is_new ) ) {
374                                                                        // Propagate this object
375                                                                        if ( reasonAndPropagate( qn, obj, pts, pe, (i<<8)|j) )
376                                                                                added = true;
377                                                                        pts = pts.next;
378                                                                }
379                                                        }
380                                                        
381                                                        pe = pe.next;
382                                                }
383                                        }
384                                }
385                                
386                                gm1.flush();
387                        }
388                        else {
389                                // We don't have new edges, thereby we can do the pairing up faster
390                                
391                                for ( Map.Entry<AllocNode, GeometricManager> entry2 : new_pts.entrySet() ) {
392                                        obj = entry2.getKey();
393                                        gm2 = entry2.getValue();
394                                        
395                                        // Avoid the garbage
396                                        if ( gm2 == deadManager ) continue;
397                                        
398                                        // Type filtering and flow-to-this filtering, a simple approach
399                                        if (!ptAnalyzer.castNeverFails(obj.getType(), qn.getType())) continue;
400                                        
401                                        entry_pts = gm2.getFigures();
402                                        
403                                        // We pair up all the geometric points-to tuples and flow edges
404                                        for ( i = 0; i < GeometricManager.Divisions; ++i ) {
405                                                pts = entry_pts[i];
406                                                
407                                                while ( pts != null && pts.is_new == true ) {
408                                                        for (j = 0; j < GeometricManager.Divisions ; ++j) {
409                                                                pe = entry_pe[j];
410                                                                
411                                                                while ( pe != null ) {
412                                                                        // Propagate this object
413                                                                        if ( reasonAndPropagate( qn, obj, pts, pe, (i<<8)|j) )
414                                                                                added = true;
415                                                                        pe = pe.next;
416                                                                }
417                                                        }
418                                                        
419                                                        pts = pts.next;
420                                                }
421                                        }
422                                }
423                        }
424                        
425                        if ( added )
426                                worklist.push( qn );
427                }
428                
429//              System.err.println();
430        }

473        @Override
474        public boolean heap_sensitive_intersection(IVarAbstraction qv) 
475        {
476                int i, j;
477                FullSensitiveNode qn;
478                SegmentNode p, q, pt[], qt[];
479                boolean localToSameMethod;
480                
481                qn = (FullSensitiveNode)qv;
482                localToSameMethod = (enclosingMethod() == qv.enclosingMethod());
483                
484                for (Iterator<AllocNode> it = pt_objs.keySet().iterator(); it.hasNext();) {
485                        AllocNode an = it.next();
486                        if ( an instanceof ClassConstantNode ) continue;
487                        if ( an instanceof StringConstantNode ) continue;
488                        qt = qn.find_points_to(an);
489                        if (qt == null) continue;
490                        pt = find_points_to(an);
491
492                        for (i = 0; i < GeometricManager.Divisions; ++i) {
493                                p = pt[i];
494                                while (p != null) {
495                                        for (j = 0; j < GeometricManager.Divisions; ++j) {
496                                                q = qt[j];
497                                                while (q != null) {
498                                                        if ( localToSameMethod ) {
499                                                                // We can use a more precise alias testing
500                                                                if ( p.intersect(q) ) return true;
501                                                        }
502                                                        else {
503                                                                if ( p.projYIntersect(q) ) return true;
504                                                        }
505                                                        q = q.next;
506                                                }
507                                        }
508                                        p = p.next;
509                                }
510                        }
511                }
512
513                return false;
514        }

553        public void injectPts()
554        {
555                final GeomPointsTo geomPTA = (GeomPointsTo)Scene.v().getPointsToAnalysis();
556                pt_objs = new HashMap<AllocNode, GeometricManager>();
557                
558                me.getP2Set().forall( new P2SetVisitor() {
559                        @Override
560                        public void visit(Node n) {
561                                if ( geomPTA.isValidGeometricNode(n) )
562                                        pt_objs.put((AllocNode)n, (GeometricManager)stubManager);
563                        }
564                });
565                
566                new_pts = null;

569        @Override
570        public boolean pointer_interval_points_to(long l, long r, AllocNode obj) 
571        {       
572                SegmentNode[] int_entry = find_points_to(obj);
573                
574                for ( int i = 0; i < GeometricManager.Divisions; ++i ) {
575                        SegmentNode p = int_entry[i];
576                        while ( p != null ) {
577                                long R = p.I1 + p.L;
578                                if ( (l <= p.I1 && p.I1 < r) || ( p.I1 <= l && l < R) )
579                                        return true;
580                                p = p.next;
581                        }
582                }
583                
584                return false;
585        }

593        @Override
594        public void get_all_context_sensitive_objects( long l, long r, PtSensVisitor visitor ) 
595        {       
596                if ( parent != this ) {
597                        getRepresentative().get_all_context_sensitive_objects(l, r, visitor);
598                        return;
599                }
600                
601                GeomPointsTo geomPTA = (GeomPointsTo)Scene.v().getPointsToAnalysis();
602                
603                for ( Map.Entry<AllocNode, GeometricManager> entry : pt_objs.entrySet() ) {
604                        AllocNode obj = entry.getKey();
605                        SootMethod sm = obj.getMethod();
606                        int sm_int = geomPTA.getIDFromSootMethod(sm);
607                        if ( sm_int == -1 ) continue;
608                        
609                        GeometricManager gm = entry.getValue();
610                        SegmentNode[] int_entry = gm.getFigures();
611                        
612                        for ( int i = 0; i < GeometricManager.Divisions; ++i ) {
613                                // We iterate all the figures
614                                SegmentNode p = int_entry[i];
615                                
616                                while ( p != null ) {
617                                        long L = p.I1;
618                                        long R = L + p.L;
619                                        long objL = -1, objR = -1;
620                                        
621                                        // Now we compute which context sensitive objects are pointed to by this pointer
622                                        if ( l <= L && L < r ) {
623                                                // l----------r
624                                                //    L----R            or
625                                                //    L------------R
626                                                if ( i == GeometricManager.ONE_TO_ONE ) {
627                                                        long d = r - L;
628                                                        if ( R < r ) d = p.L;
629                                                        objL = p.I2;
630                                                        objR = objL + d;
631                                                }
632                                                else {
633                                                        objL = p.I2;
634                                                        objR = p.I2 + ((RectangleNode)p).L_prime;
635                                                }
636                                        }
637                                        else if (L <= l && l < R) {
638                                                //     l---------r
639                                                // L-------R                or
640                                                // L--------------------R
641                                                if ( i == GeometricManager.ONE_TO_ONE ) {
642                                                        long d = R - l;
643                                                        if ( R > r ) d = r - l;
644                                                        objL = p.I2 + l - L;
645                                                        objR = objL + d;
646                                                }
647                                                else {
648                                                        objL = p.I2;
649                                                        objR = p.I2 + ((RectangleNode)p).L_prime;
650                                                }
651                                        }
652                                        
653                                        // Now we test which context versions this interval [objL, objR) maps to
654                                        if ( objL != -1 && objR != -1 )
655                                                visitor.visit(obj, objL, objR, sm_int);
656                                        
657                                        p = p.next; 
658                                }
659                        }
660                }
661        }

684        /**
685         * A non-interface public function.
686         * It adds the points-to tuple to the geometric manager.
687         */
688        private boolean addPointsTo(int code, AllocNode obj) 
689        {
690                GeometricManager gm = pt_objs.get(obj);
691                
692                if ( gm == null ) {
693                        gm = new GeometricManager();
694                        pt_objs.put(obj, gm);
695                }
696                else if ( gm == deadManager ) {
697                        // We preclude the propagation of this object
698                        return false;
699                }
700                
701                SegmentNode p = gm.addNewFigure( code, pres );
702                if ( p != null ) {
703                        new_pts.put(obj, gm);
704                        return true;
705                }
706                
707                return false;
708        }

754        /**
755         * Implement the inference rules when the input points-to figure is a one-to-one mapping.
756         */
757        private static int infer_pts_is_one_to_one( SegmentNode pts, SegmentNode pe, int code )
758        {
759                long interI, interJ;
760                
761                // The left-end is the larger one
762                interI = pe.I1 < pts.I1 ? pts.I1 : pe.I1;
763                // The right-end is the smaller one
764                interJ = (pe.I1 + pe.L < pts.I1 + pts.L ? pe.I1 + pe.L : pts.I1 + pts.L);
765                
766                if ( interI < interJ ) {
767                        switch ( code ) {
768                        case GeometricManager.ONE_TO_ONE:
769                                // assignment is a 1-1 mapping
770                                pres.I1 = interI - pe.I1 + pe.I2;
771                                pres.I2 = interI - pts.I1 + pts.I2;
772                                pres.L = interJ - interI;
773                                return GeometricManager.ONE_TO_ONE;
774                                
775                        case GeometricManager.MANY_TO_MANY:
776                                // assignment is a many-many mapping
777                                pres.I1 = pe.I2;
778                                pres.I2 = interI - pts.I1 + pts.I2;
779                                pres.L = ((RectangleNode)pe).L_prime;
780                                pres.L_prime = interJ - interI;
781                                return GeometricManager.MANY_TO_MANY;
782                        }
783                }
784                
785                return GeometricManager.Undefined_Mapping;
786        }

788        /**
789         * Implement the inference rules when the input points-to figure is a many-to-many mapping.
790         */
791        private static int infer_pts_is_many_to_many( RectangleNode pts, SegmentNode pe, int code )
792        {
793                long interI, interJ;
794                
795                // The left-end is the larger one
796                interI = pe.I1 < pts.I1 ? pts.I1 : pe.I1;
797                // The right-end is the smaller one
798                interJ = (pe.I1 + pe.L < pts.I1 + pts.L ? pe.I1 + pe.L : pts.I1 + pts.L);
799                
800                if ( interI < interJ ) {
801                        switch ( code ) {
802                        case GeometricManager.ONE_TO_ONE:
803                                // assignment is a 1-1 mapping
804                                pres.I1 = interI - pe.I1 + pe.I2;
805                                pres.I2 = pts.I2;
806                                pres.L = interJ - interI;
807                                pres.L_prime = pts.L_prime;
808                                break;
809                                
810                        case GeometricManager.MANY_TO_MANY:
811                                // assignment is a many-many mapping
812                                pres.I1 = pe.I2;
813                                pres.I2 = pts.I2;
814                                pres.L = ((RectangleNode)pe).L_prime;
815                                pres.L_prime = pts.L_prime;
816                                break;
817                        }
818                        
819                        return GeometricManager.MANY_TO_MANY;
820                }
821                
822                return GeometricManager.Undefined_Mapping;
823        }

825        /**
826         * Implements the pointer assignment inference rules.
827         * The pts and pe are the points-to tuple and flow edge
828         * pres is the computed result
829         * code indicates the types of the pts and pe
830         * 
831         * Return value is used to indicate the type of the result
832         */
833        private static boolean reasonAndPropagate( FullSensitiveNode qn, AllocNode obj, 
834                        SegmentNode pts, SegmentNode pe, int code )
835        {
836                int ret_type = GeometricManager.Undefined_Mapping;
837                
838                switch ( code >> 8 ) {
839                case GeometricManager.ONE_TO_ONE:
840                        // points-to is a 1-1 mapping
841                        ret_type = infer_pts_is_one_to_one(pts, pe, code & 255 );
842                        break;
843                        
844                case GeometricManager.MANY_TO_MANY:
845                        // points-to is a mangy-many mapping
846                        ret_type = infer_pts_is_many_to_many((RectangleNode)pts, pe, code & 255 );
847                        break;
848                }
849                
850                if (ret_type != GeometricManager.Undefined_Mapping)
851                        return qn.addPointsTo( ret_type, obj );
852                
853                return false;
854        }

856        /**
857         * The last parameter code can only be 1-1 and many-1
858         */
859        private static boolean instantiateLoadConstraint(FullSensitiveNode objn,
860                        FullSensitiveNode qn, SegmentNode pts, int code ) 
861        {
862                int ret_type = GeometricManager.Undefined_Mapping;
863                
864                if ( (code>>8) == GeometricManager.ONE_TO_ONE ) {
865                        // assignment is a 1-1 mapping
866                        
867                        pres.I1 = pts.I2;
868                        pres.I2 = pts.I1;
869                        
870                        switch ( code & 255 ) {
871                        case GeometricManager.ONE_TO_ONE:
872                                // points-to is a 1-1 mapping
873                                pres.L = pts.L;
874                                ret_type = GeometricManager.ONE_TO_ONE;
875                                break;
876                                
877                        case GeometricManager.MANY_TO_MANY:
878                                // points-to is a many-many mapping
879                                pres.L = ((RectangleNode)pts).L_prime;
880                                pres.L_prime = pts.L;
881                                ret_type = GeometricManager.MANY_TO_MANY;
882                                break;
883                        }
884                }
885                else {
886                        // The target pointer must be a global, in JIMPLE's case
887                        pres.I1 = pts.I2;
888                        pres.I2 = 1;
889                        pres.L_prime = 1;
890                        
891                        switch ( code & 255 ) {
892                        case GeometricManager.ONE_TO_ONE:
893                                // points-to is a 1-1 mapping or 1-many mapping 
894                                pres.L = pts.L;
895                                ret_type = GeometricManager.MANY_TO_MANY;
896                                break;
897                                
898                        case GeometricManager.MANY_TO_MANY:
899                                // points-to is a many-many mapping
900                                pres.L = ((RectangleNode)pts).L_prime;
901                                ret_type = GeometricManager.MANY_TO_MANY;
902                                break;
903                        }
904                }
905                
906                return objn.addFlowsTo(ret_type, qn);
907        }
