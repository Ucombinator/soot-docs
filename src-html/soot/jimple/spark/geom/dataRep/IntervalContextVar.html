019package soot.jimple.spark.geom.dataRep;
020
021import soot.jimple.spark.pag.Node;
022
023/**
024 * A particular class to encode contexts in interval manner.
025 * 
026 * @author xiao
027 *
028 */
029public class IntervalContextVar extends ContextVar 
030                        implements Comparable<IntervalContextVar> {
031
032        // The interval is [L, R), which stands for a set of consecutive contexts
033        public long L = 0, R = 0;
034        
035        public IntervalContextVar() {}
036        
037        public IntervalContextVar( long l, long r, Node v )
038        {
039                assert l < r;
040                L = l;
041                R = r;
042                var = v;
043        }
044        
045        public IntervalContextVar( IntervalContextVar o )
046        {
047                L = o.L;
048                R = o.R;
049                var = o.var;
050        }
051        
052        @Override
053        public String toString()
054        {
055                return "<" + var.toString() + ", " + L + ", " + R + ">";
056        }
057        
058        @Override
059        public boolean equals( Object o )
060        {
061                IntervalContextVar other = (IntervalContextVar)o;
062                return ( other.L == L ) && (other.R == R) && (other.var == var);
063        }
064
065        @Override
066        public int hashCode()
067        {
068                int ch = (int) ((L+R) % Integer.MAX_VALUE);
069                int ans = var.hashCode() + ch;
070                if ( ans < 0 ) ans = var.hashCode();
071                return ans;
072        }
073
074        @Override
075        public int compareTo(IntervalContextVar o) 
076        {
077                if ( L == o.L )
078                        return R < o.R ? -1 : 1;
079                
080                return L < o.L ? -1 : 1;
081        }
082
083        @Override
084        public boolean contains(ContextVar cv) 
085        {
086                IntervalContextVar icv = (IntervalContextVar)cv;
087                if ( L <= icv.L && R >= icv.R ) return true;
088                return false;
089        }
090
091        @Override
092        public boolean merge(ContextVar cv) 
093        {
094                IntervalContextVar icv = (IntervalContextVar)cv;
095                
096                if ( icv.L < L ) {
097                        if ( L <= icv.R ) {
098                                L = icv.L;
099                                if ( R < icv.R ) R = icv.R;
100                                return true;
101                        }
102                }
103                else {
104                        if ( icv.L <= R ) {
105                                if ( R < icv.R ) R = icv.R;
106                                return true;
107                        }
108                }
109                
110                return false;
111        }
112
113        @Override
114        public boolean intersect(ContextVar cv) 
115        {
116                IntervalContextVar icv = (IntervalContextVar)cv;
117                
118                if ( L <= icv.L && icv.L < R ) return true;
119                if ( icv.L <= L && L < icv.R ) return true;
120                return false;
121        }
122}
