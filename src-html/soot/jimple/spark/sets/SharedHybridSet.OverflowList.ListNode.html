011/*
012 * Possible sources of inefficiency:
013 * -It seems like there must be a way to immediately know which subset bitvector
014 * will exist, since sets are shared when a node with a smaller points-to set is
015 * pointing to another node.  Why not just take that node's bitvector as a base?
016 * -addAll could probably use many improvements.
017 * -Cast masking - calling typeManager.get
018 * -An interesting problem is that when merging a bitvector into an overflow list, if 
019 * the one being merged
020 * in has a bitvector, the mask or exclude might mask it down to a bitvector with very
021 * few ones.  (In fact, this might even result in one with 0 ones!)
022 * Should that new bitvector stay as a bitvector, or be converted to an
023 * overflow list?  And how can we tell when it will have few ones?  (Modify BitVector?)
024 * 
025 */
026/**
027 * A shared representation of a points-to set which uses a bit vector + a list
028 * of extra elements, an "overflow list", to make adding single elements fast in
029 * most cases.
030 * 
031 * The bit vector may be shared by multiple points-to sets, while the overflow
032 * list is specific to each points-to set.
033 * 
034 * To facilitate sharing of the bitvectors, there is a "hash table" of all
035 * existing bitvectors kept, called BitVectorLookupMap, where the ith element
036 * contains a list of all existing bitvectors of cardinality i (i.e. has i one
037 * bits).
038 * 
039 * @author Adam Richard
040 * 
041 */
042
043public class SharedHybridSet extends PointsToSetInternal {

068        public boolean contains(Node n) {
069                // Which should be checked first, bitVector or overflow? (for
070                // performance)
071                // I think the bit vector, since it only takes O(1) to check many
072                // elements
073
074                // Check the bit vector
075                if (bitVector != null && bitVector.contains(n))
076                        return true;
077
078                // Check overflow
079                if (overflow.contains(n))
080                        return true;
081
082                return false;
083        }

089        /**
090         * @return an overflow list of all elements in a that aren't in b (b is
091         *         assumed to be a subset of a)
092         */
093        private OverflowList remainder(PointsToBitVector a, PointsToBitVector b) {
094                // Since a contains everything b contains, doing an XOR will give
095                // everything
096                // in a not in b
097                PointsToBitVector xorResult = new PointsToBitVector(a);
098                xorResult.xor(b);
099                // xorResult must now contain <= 20 elements, assuming
100                // OVERFLOW_THRESHOLD <= OVERFLOW_SIZE
101                return new OverflowList(xorResult);
102        }
103
104        // Look for an existing bitvector in the lookupMap which is a subset of the
105        // newBitVector (the bitVector to set as the new points-to set), with only a
106        // few
107        // elements missing. If we find one, make that set the new `bitVector', and
108        // the leftovers the new `overflow'
109        //szBitVector is the size of the ORIGINAL bit vector, NOT the size of newBitVector
110        private void findAppropriateBitVector(PointsToBitVector newBitVector, PointsToBitVector otherBitVector, int otherSize, int szBitvector) {
111                //First check "other" and "this"'s bitvector, to maximize sharing and
112                //minimize searching for a new bitvector
113                if (otherBitVector != null && 
114                                otherSize <= numElements &&
115                                otherSize + OVERFLOW_THRESHOLD >= numElements &&
116                                otherBitVector.isSubsetOf(newBitVector))
117                {
118                        setNewBitVector(szBitvector, otherBitVector);
119                        overflow = remainder(newBitVector, otherBitVector);
120                }
121                else if (bitVector != null && 
122                                szBitvector <= numElements &&
123                                szBitvector + OVERFLOW_THRESHOLD >= numElements &&
124                                bitVector.isSubsetOf(newBitVector))
125                {
126                        overflow = remainder(newBitVector, bitVector);
127                }
128                else
129                {
130                        for (int overFlowSize = 0; overFlowSize < OVERFLOW_THRESHOLD; ++overFlowSize) 
131                        {
132                                int bitVectorCardinality = numElements - overFlowSize;
133                                if (bitVectorCardinality < 0) break;   //We might be trying to add a bitvector
134                                        //with <OVERFLOW_THRESHOLD ones (in fact, there might be bitvectors with 0
135                                        //ones).  This results from merging bitvectors and masking out certain values.
136                                if (bitVectorCardinality < AllSharedHybridNodes.v().lookupMap.map.length
137                                                && AllSharedHybridNodes.v().lookupMap.map[bitVectorCardinality] != null) 
138                                {
139                                        List<PointsToBitVector> lst = AllSharedHybridNodes.v()
140                                                        .lookupMap.map[bitVectorCardinality];
141                                        for (PointsToBitVector candidate : lst) {
142                                                // for each existing bit vector with bitVectorCardinality
143                                                // ones
144                                                if (candidate.isSubsetOf(newBitVector)) {
145                                                        setNewBitVector(szBitvector, candidate);
146                                                        overflow = remainder(newBitVector, candidate);
147                                                        return;
148                                                }
149                                        }
150                                }
151                        }
152                        // Didn't find an appropriate bit vector to use as a base; add the new
153                        // bit vector to the map of all bit vectors and set it as the new base
154                        // bit vector
155                        setNewBitVector(szBitvector, newBitVector);
156                        overflow.removeAll();
157                        AllSharedHybridNodes.v().lookupMap.add(numElements, newBitVector);
158                }
159        }

161        //Allows for reference counting and deleting the old bit vector if it
162        //isn't being shared
163        private void setNewBitVector(int size, PointsToBitVector newBitVector)
164        {
165                newBitVector.incRefCount();
166                if (bitVector != null) 
167                {
168                        bitVector.decRefCount();
169                
170                        if (bitVector.unused())
171                        {
172                                //delete bitVector from lookupMap
173                                AllSharedHybridNodes.v().lookupMap.remove(size, bitVector);
174                        }
175                }
176                bitVector = newBitVector;
177        }

179        public boolean add(Node n) 
180        {
181                /*
182                 * This algorithm is described in the paper "IBM Research Report: Fast
183                 * Pointer Analysis" by Hirzel, Dincklage, Diwan, and Hind, pg. 11
184                 */
185                if (contains(n))
186                        return false;
187                ++numElements;
188
189                if (!overflow.full()) {
190                        overflow.add(n);
191                } else {
192
193                        // Put everything in the bitvector
194                        PointsToBitVector newBitVector;
195                        if (bitVector == null)
196                                newBitVector = new PointsToBitVector(pag.getAllocNodeNumberer()
197                                                .size());
198                        else
199                                newBitVector = new PointsToBitVector(bitVector);
200                        newBitVector.add(n); // add n to it
201                        add(newBitVector, overflow);
202                        
203                        // Now everything is in newBitVector, and it must have numElements
204                        // ones
205
206                        // The algorithm would still work without this step, but wouldn't be
207                        // a
208                        // shared implmentation at all.
209                        findAppropriateBitVector(newBitVector, null, 0, numElements - overflow.size() - 1);
210
211                }
212                return true;
213        }

215        private boolean nativeAddAll(SharedHybridSet other, SharedHybridSet exclude) {
216                /*
217                 * If one of the shared hybrid sets has a bitvector but the other
218                 * doesn't, set that bitvector as the base bitvector and add the stuff
219                 * from the other overflow list. If they both have a bitvector, AND them
220                 * together, then add it to the lookupMap. If neither of them has a
221                 * bitvector, just combine the overflow lists.
222                 */
223
224        BitVector mask = getBitMask(other, pag);
225
226                if (exclude != null)
227                {
228                        if (exclude.overflow.size() > 0) 
229                        {
230                                // Make exclude only a bitvector, for simplicity
231                                PointsToBitVector newBitVector;
232                                if (exclude.bitVector == null) {
233                                        newBitVector = new PointsToBitVector(pag.getAllocNodeNumberer()
234                                                        .size());
235                                } else {
236                                        newBitVector = new PointsToBitVector(exclude.bitVector);
237                                }
238                                add(newBitVector, exclude.overflow);
239                                exclude = new SharedHybridSet(type, pag);
240                                exclude.bitVector = newBitVector;
241                        }
242
243                        //It's possible at this point that exclude could have been passed in non-null,
244                        //but with no elements.  Simplify the rest of the algorithm by setting it to null
245                        //in that case.
246                        else if (exclude.bitVector == null) exclude = null;
247                }
248                
249                int originalSize = size(), 
250                    originalOnes = originalSize - overflow.size(),
251                    otherBitVectorSize = other.size() - other.overflow.size();
252
253                // Decide on the base bitvector
254                if (bitVector == null) {
255                        bitVector = other.bitVector;
256                        if (bitVector != null) { // Maybe both bitvectors were null; in
257                                                     // that case, no need to do this
258                                bitVector.incRefCount();
259
260                                // Since merging in new bits might add elements that
261                                // were
262                                // already in the overflow list, we have to remove and re-add
263                                // them all.
264                                // TODO: Can this be avoided somehow?
265                                // Maybe by allowing an element to be both in the overflow set
266                                // and
267                                // the bitvector?
268                                // Or could it be better done by checking just the bitvector and
269                                // removing elements that are there?
270                                OverflowList toReAdd = overflow;
271                                overflow = new OverflowList();
272                                
273                                boolean newBitVectorCreated = false;  //whether a new bit vector
274                                        //was created, which is used to decide whether to re-add the
275                                        //overflow list as an overflow list again or merge it into the
276                                        //new bit vector.
277                                
278                                numElements = otherBitVectorSize;
279                                if (exclude != null || mask != null) 
280                                {
281                                        PointsToBitVector result = new PointsToBitVector(bitVector);
282                                        if (exclude != null) result.andNot(exclude.bitVector);
283                                        if (mask != null) result.and(mask);
284                                        if (!result.equals(bitVector))
285                                        {
286                                                add(result, toReAdd);
287                                                int newBitVectorSize = result.cardinality(); 
288                                                numElements = newBitVectorSize;
289                                                findAppropriateBitVector(result, other.bitVector, otherBitVectorSize, otherBitVectorSize);
290                                                newBitVectorCreated = true;
291                                        }
292                                }
293
294                                if (!newBitVectorCreated)  //if it was, then toReAdd has
295                                        //already been re-added
296                                {
297                                        for (OverflowList.ListNode i = toReAdd.overflow; i != null; i = i.next) {
298                                                add(i.elem);
299                                        }                                       
300                                }
301                        }
302                } else if (other.bitVector != null) {
303                        // Now both bitvectors are non-null; merge them
304                        PointsToBitVector newBitVector = new PointsToBitVector(other.bitVector);
305                        if (exclude != null)
306                                newBitVector.andNot(exclude.bitVector);
307                        if (mask != null) newBitVector.and(mask);
308                        
309                        newBitVector.or(bitVector);
310
311                        if (!newBitVector.equals(bitVector)) // if some elements were
312                                                                                                        // actually added
313                        {
314
315                                //At this point newBitVector is bitVector + some new bits
316                                
317                                // Have to make a tough choice - is it better at this point to
318                                // put both overflow lists into this bitvector (which involves
319                                // recalculating bitVector.cardinality() again since there might 
320                                // have been overlap), or is it better to re-add both the
321                                // overflow lists to the set?
322                                // I suspect the former, so I'll do that.
323
324                                // Basically we now want to merge both overflow lists into this
325                                // new
326                                // bitvector (if it is indeed a new bitvector), then add that
327                                // resulting
328                                // huge bitvector to the lookupMap, unless a subset of it is
329                                // already there.
330
331                                if (other.overflow.size() != 0) {
332                                        PointsToBitVector toAdd = 
333                                                new PointsToBitVector(newBitVector.size());
334                                        add(toAdd, other.overflow);
335                                        if (mask != null) toAdd.and(mask);
336                                        if (exclude != null) toAdd.andNot(exclude.bitVector);
337                                        newBitVector.or(toAdd);
338                                }
339                                //At this point newBitVector is still bitVector + some new bits
340
341                                int numOnes = newBitVector.cardinality();  //# of bits in the 
342                                        //new bitvector
343                                int numAdded = add(newBitVector, overflow);
344                                numElements += numOnes - originalOnes   //number of new bits
345                                        + numAdded - overflow.size();   //might be negative due to 
346                                                //elements in overflow already being in the new bits
347
348                                if (size() > originalSize)
349                                {
350                                        findAppropriateBitVector(newBitVector, other.bitVector, otherBitVectorSize, originalOnes);
351                                        //checkSize();
352                                        return true;
353                                }
354                                else 
355                                {
356                                        //checkSize();
357                                        return false;   //It might happen that the bitvector being merged in adds some bits
358                                        //to the existing bitvector, but that those new bits are all elements that were already
359                                        //in the overflow list.  In that case, the set might not change, and if not we return false.
360                                        //We also leave the set the way it was by not calling findAppropriateBitvector,
361                                        //which maximizes sharing and is fastest in the short term.  I'm not sure whether it
362                                        //would be faster overall to keep the already calculated bitvector anyway.
363                                }
364                        }
365                }
366                // Add all the elements in the overflow list of other, unless they're in
367                // exclude
368                OverflowList overflow = other.overflow;
369                for (OverflowList.ListNode i = overflow.overflow; i != null; i = i.next) {
370//              for (int i = 0; i < overflow.size(); ++i) {
371                        Node nodeToMaybeAdd = i.elem;
372                        if ((exclude == null) || !exclude.contains(nodeToMaybeAdd)) {
373                                if (mask == null || mask.get(nodeToMaybeAdd.getNumber()))
374                                {
375                                        add(nodeToMaybeAdd);
376                                }
377                        }
378                }
379
380                //checkSize();
381                return size() > originalSize;
382        }

417        public boolean addAll(PointsToSetInternal other,
418                        final PointsToSetInternal exclude) {
419                // Look at the sort of craziness we have to do just because of a lack of
420                // multimethods
421                if (other == null)
422                        return false;
423                if ((!(other instanceof SharedHybridSet))
424                                || (exclude != null && !(exclude instanceof SharedHybridSet))) {
425                        return super.addAll(other, exclude);
426                } else {
427                        return nativeAddAll((SharedHybridSet) other,
428                                        (SharedHybridSet) exclude);
429                }
430        }

432        public boolean forall(P2SetVisitor v) {
433                // Iterate through the bit vector. Ripped from BitPointsToSet again.
434                // It seems there should be a way to share code between BitPointsToSet
435                // and
436                // SharedHybridSet, but I don't know how at the moment.
437                if (bitVector != null) {
438                        for (BitSetIterator it = bitVector.iterator(); it.hasNext();) {
439                                v.visit((Node) pag.getAllocNodeNumberer().get(it.next()));
440                        }
441                }
442                // Iterate through the overflow list
443                for (OverflowList.ListNode i = overflow.overflow; i != null; i = i.next) {
444                        v.visit(i.elem);
445                }
446                return v.getReturnValue();
447        }

473        private class OverflowList {
474                public class ListNode {
475                        public Node elem;
476                        public ListNode next;
477                        public ListNode(Node elem, ListNode next)
478                        {
479                                this.elem = elem;
480                                this.next = next;
481                        }
482                }
483                public OverflowList() {
484                }
485
486                public OverflowList(PointsToBitVector bv) {
487                        BitSetIterator it = bv.iterator(); // Iterates over only the 1 bits
488                        while (it.hasNext()) {
489                                // Get the next node in the bitset by looking it up in the
490                                // pointer assignment graph.
491                                // Ripped from BitPointsToSet.
492                                Node n = (Node) (pag.getAllocNodeNumberer().get(it.next()));
493                                add(n);
494                        }
495
496                }
497
498                public void add(Node n) {
499                        if (full())
500                                throw new RuntimeException(
501                                                "Can't add an element to a full overflow list.");
502                        overflow = new ListNode(n, overflow);
503                        ++overflowElements;
504                }
505
506                public int size() {
507                        return overflowElements;
508                }
509
510                public boolean full() {
511                        return overflowElements == OVERFLOW_SIZE;
512                }
513
514                public boolean contains(Node n) {
515                        for (ListNode l = overflow; l != null; l = l.next) {
516                                if (n == l.elem)
517                                        return true;
518                        }
519                        return false;
520                }
521
522                public void removeAll() {
523                        overflow = null;
524                        overflowElements = 0;
525                }
