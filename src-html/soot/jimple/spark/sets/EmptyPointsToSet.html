020package soot.jimple.spark.sets;
021import soot.*;
022import soot.jimple.ClassConstant;
023import soot.jimple.spark.pag.Node;
024
025import java.util.*;
026
027/** Implementation of an empty, immutable points-to set.
028 * @author Ondrej Lhotak
029 */
030public class EmptyPointsToSet extends PointsToSetInternal {
031    public EmptyPointsToSet( Singletons.Global g ) { super(null); }
032    public static EmptyPointsToSet v() { return G.v().soot_jimple_spark_sets_EmptyPointsToSet(); }
033    
034    public EmptyPointsToSet(Singletons.Global g, Type type) {
035        super(type);
036    }
037
038    /** Returns true if this set contains no run-time objects. */
039    public boolean isEmpty() { return true; }
040    /** Returns true if this set shares some objects with other. */
041    public boolean hasNonEmptyIntersection( PointsToSet other ) {
042        return false;
043    }
044    /** Set of all possible run-time types of objects in the set. */
045    public Set<Type> possibleTypes() { return Collections.emptySet(); }
046    /** Adds contents of other into this set, returns true if this set 
047     * changed. */
048    public boolean addAll( PointsToSetInternal other,
049            PointsToSetInternal exclude ) {
050        throw new RuntimeException( "can't add into empty immutable set" );
051    }
052    /** Calls v's visit method on all nodes in this set. */
053    public boolean forall( P2SetVisitor v ) {
054        return false;
055    }
056    /** Adds n to this set, returns true if n was not already in this set. */
057    public boolean add( Node n ) {
058        throw new RuntimeException( "can't add into empty immutable set" );
059    }
060    /** Returns true iff the set contains n. */
061    public boolean contains( Node n ) {
062        return false;
063    }
064
065    public Set<String> possibleStringConstants() { return Collections.emptySet(); }
066    public Set<ClassConstant> possibleClassConstants() { return Collections.emptySet(); }
067    
068    /**
069     * {@inheritDoc}
070     */
071    public String toString() {
072        return "{}";
073    }
074
075}
