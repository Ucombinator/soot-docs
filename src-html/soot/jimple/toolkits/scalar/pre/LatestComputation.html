026package soot.jimple.toolkits.scalar.pre;
027
028import soot.*;
029import soot.toolkits.scalar.*;
030import soot.toolkits.graph.*;
031import java.util.*;
032
033/**
034 * Performs a Latest-Computation on the given graph. a computation is latest,
035 * when we can't delay it anymore. This uses the Delayability-analysis. More
036 * precise: The delayability-analysis says us already until which point we can
037 * delay a computation from the earliest computation-point. We just have to
038 * search for points, where there's a computation, or, where we can't delay the
039 * computation to one of the successors.
040 */
041public class LatestComputation {
042        private Map<Unit, FlowSet<EquivalentValue>> unitToLatest;
043
044        /**
045         * given a DelayabilityAnalysis and the computations of each unit,
046         * calculates the latest computation-point for each expression. the
047         * <code>equivRhsMap</code> could be calculated on the fly, but it is
048         * <b>very</b> likely that it already exists (as similar maps are used for
049         * calculating Earliestness, Delayed,...
050         *
051         * @param dg
052         *            a ExceptionalUnitGraph
053         * @param delayed
054         *            the delayability-analysis of the same graph.
055         * @param equivRhsMap
056         *            all computations of the graph
057         */
058        public LatestComputation(UnitGraph unitGraph, DelayabilityAnalysis delayed,
059                        Map<Unit, EquivalentValue> equivRhsMap) {
060                this(unitGraph, delayed, equivRhsMap, new ArrayPackedSet<EquivalentValue>(
061                                new CollectionFlowUniverse<EquivalentValue>(equivRhsMap.values())));
062        }
063
064        /**
065         * given a DelayabilityAnalysis and the computations of each unit,
066         * calculates the latest computation-point for each expression.<br>
067         * the <code>equivRhsMap</code> could be calculated on the fly, but it is
068         * <b>very</b> likely that it already exists (as similar maps are used for
069         * calculating Earliestness, Delayed,...<br>
070         * the shared set allows more efficient set-operations, when they the
071         * computation is merged with other analyses/computations.
072         *
073         * @param dg
074         *            a ExceptionalUnitGraph
075         * @param delayed
076         *            the delayability-analysis of the same graph.
077         * @param equivRhsMap
078         *            all computations of the graph
079         * @param set
080         *            the shared flowSet
081         */
082        public LatestComputation(UnitGraph unitGraph, DelayabilityAnalysis delayed,
083                        Map<Unit, EquivalentValue> equivRhsMap, BoundedFlowSet<EquivalentValue> set) {
084                unitToLatest = new HashMap<Unit, FlowSet<EquivalentValue>>(unitGraph.size() + 1, 0.7f);
085
086                for (Unit currentUnit : unitGraph) {
087                        /* create a new Earliest-list for each unit */
088                        /*
089                         * basically the latest-set is: (delayed) INTERSECT (comp UNION
090                         * (UNION_successors ~Delayed)) = (delayed) MINUS
091                         * ((INTERSECTION_successors Delayed) MINUS comp).
092                         */
093
094                        FlowSet<EquivalentValue> delaySet = delayed.getFlowBefore(currentUnit);
095
096                        /* Calculate (INTERSECTION_successors Delayed) */
097                        FlowSet<EquivalentValue> succCompSet =  set.topSet();
098                        for (Unit successor : unitGraph.getSuccsOf(currentUnit)) {
099                                succCompSet.intersection(delayed.getFlowBefore(successor), succCompSet);
100                        }
101                        /*
102                         * remove the computation of this set: succCompSet is then:
103                         * ((INTERSECTION_successors Delayed) MINUS comp)
104                         */
105                        if (equivRhsMap.get(currentUnit) != null)
106                                succCompSet.remove(equivRhsMap.get(currentUnit));
107
108                        /* make the difference: */
109                        FlowSet<EquivalentValue> latest = delaySet.emptySet();
110                        delaySet.difference(succCompSet, latest);
111
112                        unitToLatest.put(currentUnit, latest);
113                }
114        }
115
116        /**
117         * returns the set of expressions, that have their latest computation just
118         * before <code>node</code>.
119         *
120         * @param node
121         *            an Object of the flow-graph (in our case always a unit).
122         * @return a FlowSet containing the expressions.
123         */
124        public FlowSet<EquivalentValue> getFlowBefore(Object node) {
125                return unitToLatest.get(node);
126        }
127}
