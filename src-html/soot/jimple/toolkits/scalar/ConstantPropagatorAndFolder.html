028package soot.jimple.toolkits.scalar;
029import soot.options.*;
030import soot.*;
031import soot.toolkits.scalar.*;
032import soot.jimple.*;
033
034import java.util.*;
035
036import soot.toolkits.graph.*;

038/** Does constant propagation and folding. 
039 * Constant folding is the compile-time evaluation of constant
040 * expressions (i.e. 2 * 3). */
041public class ConstantPropagatorAndFolder extends BodyTransformer
042{
043    public ConstantPropagatorAndFolder( Singletons.Global g ) {}
044    public static ConstantPropagatorAndFolder v() { return G.v().soot_jimple_toolkits_scalar_ConstantPropagatorAndFolder(); }
045
046    protected void internalTransform(Body b, String phaseName, Map<String,String> options)
047    {
048        int numFolded = 0;
049        int numPropagated = 0;
050
051        if (Options.v().verbose())
052            G.v().out.println("[" + b.getMethod().getName() +
053                               "] Propagating and folding constants...");
054
055        UnitGraph g = new ExceptionalUnitGraph(b);
056        LocalDefs localDefs = LocalDefs.Factory.newLocalDefs(g);
057
058        // Perform a constant/local propagation pass.
059        Orderer<Unit> orderer = new PseudoTopologicalOrderer<Unit>();
060        
061        // go through each use box in each statement
062        for (Unit u : orderer.newList(g, false)) {
063
064            // propagation pass
065            for (ValueBox useBox : u.getUseBoxes()) {
066                Value value = useBox.getValue();
067                if (value instanceof Local) {
068                    Local local = (Local) value;
069                    List<Unit> defsOfUse = localDefs.getDefsOfAt(local, u);
070                    if (defsOfUse.size() == 1) {
071                        DefinitionStmt defStmt = (DefinitionStmt) defsOfUse.get(0);
072                        Value rhs = defStmt.getRightOp();
073                        if (rhs instanceof NumericConstant
074                                        || rhs instanceof StringConstant
075                                        || rhs instanceof NullConstant) {
076                            if (useBox.canContainValue(rhs)) {
077                                useBox.setValue(rhs);
078                                numPropagated++;
079                            }
080                        }
081                        else if (rhs instanceof CastExpr) {
082                                CastExpr ce = (CastExpr) rhs;
083                                if (ce.getCastType() instanceof RefType
084                                                && ce.getOp() instanceof NullConstant) {
085                                defStmt.getRightOpBox().setValue(NullConstant.v());
086                                numPropagated++;
087                                }
088                        }
089                    }
090                }
091            }
092                
093            // folding pass
094            for (ValueBox useBox : u.getUseBoxes()) {
095                Value value = useBox.getValue();
096                if (!(value instanceof Constant)) {
097                    if (Evaluator.isValueConstantValued(value)) {
098                        Value constValue =
099                            Evaluator.getConstantValueOf(value);
100                        if (useBox.canContainValue(constValue)) {
101                            useBox.setValue(constValue);
102                            numFolded++;
103                        }
104                    }
105                }
106            }
107        }
108
109       if (Options.v().verbose())
110            G.v().out.println("[" + b.getMethod().getName() +
111                "]     Propagated: " + numPropagated + ", Folded:  " + numFolded);
112
113    } // optimizeConstants
114
115}
