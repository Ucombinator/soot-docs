001package soot.jimple.toolkits.ide.icfg;
002
003import heros.InterproceduralCFG;
004
005import java.util.Collection;
006import java.util.List;
007import java.util.Set;
008
009import soot.Value;
010import soot.toolkits.graph.DirectedGraph;
011
012/**
013 * An {@link InterproceduralCFG} which supports the computation of predecessors.
014 */
015public interface BiDiInterproceduralCFG<N, M> extends InterproceduralCFG<N, M> {
016
017        public List<N> getPredsOf(N u);
018        
019        public Collection<N> getEndPointsOf(M m);
020
021        public List<N> getPredsOfCallAt(N u);
022
023        public Set<N> allNonCallEndNodes();
024                
025        //also exposed to some clients who need it
026        public DirectedGraph<N> getOrCreateUnitGraph(M body);
027
028        public List<Value> getParameterRefs(M m);
029        
030        /**
031         * Gets whether the given statement is a return site of at least one call
032         * @param n The statement to check
033         * @return True if the given statement is a return site, otherwise false
034         */
035        public boolean isReturnSite(N n);
036        
037        /**
038         * Checks whether the given statement is rachable from the entry point
039         * @param u The statement to check
040         * @return True if there is a control flow path from the entry point of the
041         * program to the given statement, otherwise false
042         */
043        public boolean isReachable(N u);
044
045}
