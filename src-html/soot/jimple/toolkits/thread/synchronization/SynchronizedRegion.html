001package soot.jimple.toolkits.thread.synchronization;
002
003import java.util.*;
004
005import soot.jimple.Stmt;
006import soot.toolkits.scalar.Pair;
007
008public class SynchronizedRegion
009{
010        public Stmt prepStmt;
011        public Stmt entermonitor;
012        public Stmt beginning; // first stmt of body
013        public List<Pair<Stmt, Stmt>> earlyEnds; // list of <return/branch stmt, exitmonitor> pairs
014        public Pair<Stmt, Stmt> exceptionalEnd; // <throw stmt, exitmonitor> pair
015        public Pair<Stmt, Stmt> end; // <goto stmt, exitmonitor> pair
016        public Stmt last; // the last stmt before exception handling (usually a goto, return, or branch stmt from one of the ends)
017        public Stmt after;
018
019        public SynchronizedRegion()
020        {
021                this.prepStmt = null;
022                this.entermonitor = null;
023                this.beginning = null;
024                this.earlyEnds = new ArrayList<Pair<Stmt, Stmt>>();
025                this.exceptionalEnd = null;
026                this.end = null;
027                this.last = null;
028                this.after = null;
029        }
030        
031        public SynchronizedRegion(SynchronizedRegion sr)
032        {
033                this.prepStmt = sr.prepStmt;
034                this.entermonitor = sr.entermonitor;
035                this.beginning = sr.beginning;
036                this.earlyEnds = new ArrayList<Pair<Stmt, Stmt>>();
037                this.earlyEnds.addAll(sr.earlyEnds);
038                this.exceptionalEnd = null;
039                this.end = sr.end;
040                this.last = sr.last;
041                this.after = sr.after;
042        }
043
044        protected Object clone()
045        {
046                return new SynchronizedRegion(this);
047        }
048}
