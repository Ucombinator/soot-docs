002package soot.jimple.toolkits.thread.synchronization;

004import java.util.*;
005
006import soot.*;
007import soot.util.*;
008
009// Written by Richard L. Halpert on August 11, 2007
010// Acts as a dummy value that gets put in a transaction's lockset,
011// indicating that a new static object needs to be inserted into the
012// program for use as a lock.
013
014public class NewStaticLock implements Value
015{
016        SootClass sc; // The class to which to add a static lock.
017        static int nextidnum = 1;
018        int idnum;
019        
020        public NewStaticLock(SootClass sc)
021        {
022                this.sc = sc;
023                this.idnum = nextidnum++;
024        }
025        
026        public SootClass getLockClass()
027        {
028                return sc;
029        }
030
031    @Override
032    public List<ValueBox> getUseBoxes()
033    {
034        return Collections.emptyList();
035    }
036
037    /** Clones the object.  Not implemented here. */
038    public Object clone() 
039    {
040        return new NewStaticLock(sc);
041    }
042
043    /** Returns true if this object is structurally equivalent to c. 
044     * AbstractDataSources are equal and equivalent if their sourcename is the same */
045    public boolean equivTo(Object c)
046    {
047        return equals(c);
048    }
049    
050    public boolean equals(Object c)
051    {
052        if(c instanceof NewStaticLock)
053        {
054                return ((NewStaticLock) c).idnum == idnum;
055        }
056        return false;
057    }
058    
059    /** Returns a hash code consistent with structural equality for this object. */
060    public int equivHashCode()
061    {
062        return hashCode();
063    }
064    
065    public int hashCode()
066    {
067        return idnum;
068    }
069    
070    public void toString( UnitPrinter up ) {}
071    
072    public Type getType()
073    {
074        return NullType.v();
075    }
076    
077    public void apply(Switch sw)
078    {
079        throw new RuntimeException("Not Implemented");
080    }
081    
082    public String toString()
083    {
084        return "<new static lock in " + sc.toString() + ">";
085    }
086}
