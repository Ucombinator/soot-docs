405                                                }

409                                                        throw new RuntimeException("Lock Region has no ends!  Where should we put the exception handling???");
410
411                                                // Add throwable
412                                                Local throwableLocal = Jimple.v().newLocal("throwableLocal" + (throwableNum++), RefType.v("java.lang.Throwable"));
413                                                b.getLocals().add(throwableLocal);
414                                                // Add stmts

417                                                        throw new RuntimeException("WHY IS clr.last NULL???");
418                                                if(newCatch == null)
419                                                        throw new RuntimeException("WHY IS newCatch NULL???");
420                                                units.insertAfter(newCatch, csr.last);

422                                                Stmt newThrow = Jimple.v().newThrowStmt(throwableLocal);
423                                                units.insertAfter(newThrow, newExitmonitor);
424                                                // Add traps
425                                                SootClass throwableClass = Scene.v().loadClassAndSupport("java.lang.Throwable");

427                                                b.getTraps().addFirst(Jimple.v().newTrap(throwableClass, csr.beginning, lastEnd, newCatch));
428                                                csr.exceptionalEnd = new Pair<Stmt, Stmt>(newThrow, newExitmonitor);
429                                        }
430                                }
431                                lockNum++;
432                        }
433                        
434                        // deal with waits and notifys
435                        {
436                                for(Unit uNotify : tn.notifys)
437                                {
438                                        Stmt sNotify = (Stmt) uNotify;
439                                        Stmt newNotify =

442                                                        clo,

444                                                        Collections.EMPTY_LIST));
445                                        if(newPrep != null)
446                                        {
447                                                Stmt tmp = (Stmt) newPrep.clone();
448                                        units.insertBefore(tmp, sNotify);
449                                    units.insertBefore(newNotify, tmp);
450                                }
451                                else
452                                        units.insertBefore(newNotify, sNotify);
453                                        
454                                        redirectTraps(b, sNotify, newNotify);
455                                        units.remove(sNotify);
456                                }
457
458                                // Replace base object of calls to wait with appropriate lockobj
459                                for(Unit uWait : tn.waits)
460                                {
461                                        Stmt sWait = (Stmt) uWait;
462                                        ((InstanceInvokeExpr) sWait.getInvokeExpr()).setBase(clo); // WHAT IF THIS IS THE WRONG LOCK IN A PAIR OF NESTED LOCKS???
463                                        if(newPrep != null)
464                                                units.insertBefore((Stmt) newPrep.clone(), sWait);
465                                }
466                        }
467                }
468        }

472        public InstanceFieldRef reconstruct(Body b, PatchingChain<Unit> units, InstanceFieldRef lock, Stmt insertBefore, boolean redirect)
473        {
474                G.v().out.println("Reconstructing " + lock);
475
476                if(!(lock.getBase() instanceof FakeJimpleLocal))
477                {
478                        G.v().out.println("  base is not a FakeJimpleLocal");
479                        return lock;
480                }
481                FakeJimpleLocal fakeBase = (FakeJimpleLocal) lock.getBase();
482                
483                if(!(fakeBase.getInfo() instanceof LockableReferenceAnalysis))
484                        throw new RuntimeException("InstanceFieldRef cannot be reconstructed due to missing LocksetAnalysis info: " + lock);
485                LockableReferenceAnalysis la = (LockableReferenceAnalysis) fakeBase.getInfo();
486                
487                EquivalentValue baseEqVal = la.baseFor(lock);
488                if(baseEqVal == null)
489                        throw new RuntimeException("InstanceFieldRef cannot be reconstructed due to lost base from Lockset");
490                Value base = baseEqVal.getValue();
491                Local baseLocal;
492                if(base instanceof InstanceFieldRef)
493                {
494                        Value newBase = reconstruct(b, units, (InstanceFieldRef) base, insertBefore, redirect);
495                        baseLocal = Jimple.v().newLocal("baseLocal" + (baseLocalNum++), newBase.getType());
496                        b.getLocals().add(baseLocal);
497                                        
498                        // make it equal to the right value
499                        Stmt baseAssign = Jimple.v().newAssignStmt(baseLocal, newBase);
500                        if(redirect == true)
501                                units.insertBefore(baseAssign, insertBefore);
502                        else
503                                units.insertBeforeNoRedirect(baseAssign, insertBefore);
504                }
505                else if(base instanceof Local)
506                        baseLocal = (Local) base;
507                else
508                        throw new RuntimeException("InstanceFieldRef cannot be reconstructed because it's base is of an unsupported type" + base.getType() + ": " + base);
509                
510                InstanceFieldRef newLock = Jimple.v().newInstanceFieldRef(baseLocal, lock.getField().makeRef());
511                G.v().out.println("  as " + newLock);
512                return newLock;
513        }

516        static Map<EquivalentValue, StaticFieldRef> lockEqValToLock = new HashMap<EquivalentValue, StaticFieldRef>();
517        static public Value getLockFor(EquivalentValue lockEqVal)
518        {
519                Value lock = lockEqVal.getValue();
520                
521                if( lock instanceof InstanceFieldRef )
522                        return lock;
523                        
524                if( lock instanceof ArrayRef ) // it would be better to lock the array ref for each value of the index!
525                        return ((ArrayRef) lock).getBase();
526                
527                if( lock instanceof Local )
528                        return lock;
529                        
530                if( lock instanceof StaticFieldRef || lock instanceof NewStaticLock)
531                {
532                        if( lockEqValToLock.containsKey(lockEqVal) )
533                                return lockEqValToLock.get(lockEqVal);
534                        
535                        SootClass lockClass = null;
536                        if( lock instanceof StaticFieldRef )
537                        {
538                                StaticFieldRef sfrLock = (StaticFieldRef) lock;
539                                lockClass = sfrLock.getField().getDeclaringClass();
540                        }
541                        else if( lock instanceof NewStaticLock )
542                        {
543                                DeadlockAvoidanceEdge dae = (DeadlockAvoidanceEdge) lock;
544                                lockClass = dae.getLockClass();
545                        }
546                SootMethod clinitMethod = null;
547                JimpleBody clinitBody = null;
548                Stmt firstStmt = null;
549                        boolean addingNewClinit = !lockClass.declaresMethod("void <clinit>()");
550                        if(addingNewClinit)
551                {
552                        clinitMethod = new SootMethod("<clinit>", new ArrayList(), VoidType.v(), Modifier.PUBLIC | Modifier.STATIC);
553                        clinitBody = Jimple.v().newBody(clinitMethod);
554                        clinitMethod.setActiveBody(clinitBody);
555                        lockClass.addMethod(clinitMethod);
556                }
557                else
558                {
559                                clinitMethod = lockClass.getMethod("void <clinit>()");
560                                clinitBody = (JimpleBody) clinitMethod.getActiveBody();
561                                firstStmt = clinitBody.getFirstNonIdentityStmt();
562                }
563                PatchingChain<Unit> clinitUnits = clinitBody.getUnits();
564                
565                        Local lockLocal = Jimple.v().newLocal("objectLockLocal" + lockNumber, RefType.v("java.lang.Object"));
566                        // lockNumber is increased below
567                        clinitBody.getLocals().add(lockLocal); // TODO: add name conflict avoidance code
568                            
569            // assign new object to lock obj
570            Stmt newStmt = Jimple.v().newAssignStmt(lockLocal, Jimple.v().newNewExpr(RefType.v("java.lang.Object")));
571                        if(addingNewClinit)
572                                clinitUnits.add(newStmt);
573                        else
574                                clinitUnits.insertBeforeNoRedirect(newStmt, firstStmt);
575
576                        // initialize new object
577            SootClass objectClass = Scene.v().loadClassAndSupport("java.lang.Object");
578            RefType type = RefType.v(objectClass);

610        public void redirectTraps(Body b, Unit oldUnit, Unit newUnit)
611        {
612                Chain<Trap> traps = b.getTraps();
613                for(Trap trap : traps)
614                {
615                        if(trap.getHandlerUnit() == oldUnit)
616                                trap.setHandlerUnit(newUnit);
617                        if(trap.getBeginUnit() == oldUnit)
618                                trap.setBeginUnit(newUnit);
619                        if(trap.getEndUnit() == oldUnit)
620                                trap.setEndUnit(newUnit);
621                }
622        }
