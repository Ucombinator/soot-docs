015/**
016 * @author Richard L. Halpert
017 * Finds Synchronized Regions and creates a set of CriticalSection objects from them.
018 */
019public class SynchronizedRegionFinder extends ForwardFlowAnalysis<Unit, FlowSet<SynchronizedRegionFlowPair>>
020{
021    FlowSet<SynchronizedRegionFlowPair> emptySet = new ArraySparseSet<SynchronizedRegionFlowPair>();
022
023    Map unitToGenerateSet;
024
025    Body body;
026    Chain<Unit> units;
027        SootMethod method;
028        ExceptionalUnitGraph egraph;
029        LocalUses slu;
030        CriticalSectionAwareSideEffectAnalysis tasea;
031//      SideEffectAnalysis sea;
032        
033        List<Object> prepUnits;
034
035    CriticalSection methodTn;
036        
037        public boolean optionPrintDebug = false;
038        public boolean optionOpenNesting = true;
039        
040    SynchronizedRegionFinder(UnitGraph graph, Body b, boolean optionPrintDebug, boolean optionOpenNesting, ThreadLocalObjectsAnalysis tlo)
041        {
042                super(graph);
043
044                this.optionPrintDebug = optionPrintDebug;
045                this.optionOpenNesting = optionOpenNesting;
046
047                body = b;
048                units = b.getUnits();
049                method = body.getMethod();
050
051                if(graph instanceof ExceptionalUnitGraph)
052                        egraph = (ExceptionalUnitGraph) graph;
053                else
054                        egraph = new ExceptionalUnitGraph(b);
055                                
056                
057                slu = LocalUses.Factory.newLocalUses(egraph);
058                
059                
060                if( G.v().Union_factory == null ) {
061                    G.v().Union_factory = new UnionFactory() {
062                        public Union newUnion() { return FullObjectSet.v(); }
063                    };
064                }
065                
066        tasea = new CriticalSectionAwareSideEffectAnalysis(Scene.v().getPointsToAnalysis(), 
067                                Scene.v().getCallGraph(), null, tlo);
068                                        
069        prepUnits = new ArrayList<Object>();
070        
071                methodTn = null;
072                if(method.isSynchronized())
073                {
074                        // Entire method is transactional
075                        methodTn = new CriticalSection(true, method, 1);
076                        methodTn.beginning = ((JimpleBody) body).getFirstNonIdentityStmt();
077                }
078        doAnalysis();
079                if(method.isSynchronized() && methodTn != null)
080                {
081                        for(Iterator<Unit> tailIt = graph.getTails().iterator(); tailIt.hasNext(); )
082                        {
083                                Stmt tail = (Stmt) tailIt.next();
084                                methodTn.earlyEnds.add(new Pair(tail, null)); // has no exitmonitor stmt yet
085                        }
086                }
087        }
088        
089    /**
090     * All INs are initialized to the empty set.
091     **/
092    protected FlowSet<SynchronizedRegionFlowPair> newInitialFlow()
093    {
094                FlowSet<SynchronizedRegionFlowPair> ret = emptySet.clone();
095                if(method.isSynchronized() && methodTn != null)
096                {
097                        ret.add(new SynchronizedRegionFlowPair(methodTn, true));
098                }
099        return ret;
100    }
101
102    /**
103     * OUT is the same as (IN minus killSet) plus the genSet.
104     **/
105    protected void flowThrough(FlowSet<SynchronizedRegionFlowPair> in, Unit unit, FlowSet<SynchronizedRegionFlowPair> out)
106    {
107                Stmt stmt = (Stmt) unit;
108
109        copy(in, out);
110        
111        // Determine if this statement is a preparatory statement for an
112        // upcoming transactional region. Such a statement would be a definition 
113        // which contains no invoke statement, and which corresponds only to 
114        // EnterMonitorStmt and ExitMonitorStmt uses.  In this case, the read
115        // set of this statement should not be considered part of the read set
116        // of any containing transaction
117        if(unit instanceof AssignStmt)
118        {
119                boolean isPrep = true;
120                Iterator<UnitValueBoxPair> uses = slu.getUsesOf((Unit) unit).iterator();
121                if(!uses.hasNext())
122                        isPrep = false;
123                while(uses.hasNext())
124                {
125                        UnitValueBoxPair use = uses.next();
126                        Unit useStmt = use.getUnit();
127                        if( !(useStmt instanceof EnterMonitorStmt) && !(useStmt instanceof ExitMonitorStmt) )
128                        {
129                                isPrep = false;
130                                break;
131                        }
132                }
133                if(isPrep)
134                {
135                        prepUnits.add(unit);
136                        if(optionPrintDebug)
137                        {
138                                G.v().out.println("prep: " + unit.toString());
139                        }
140                        return;
141                }
142        }
143                
144        // Determine if this statement is the start of a transaction
145        boolean addSelf = (unit instanceof EnterMonitorStmt);
146        
147                // Determine the level of transaction nesting of this statement
148                int nestLevel = 0;
149        Iterator<SynchronizedRegionFlowPair> outIt0 = out.iterator();
150        while(outIt0.hasNext())
151        {
152            SynchronizedRegionFlowPair srfp = outIt0.next();
153            if(srfp.tn.nestLevel > nestLevel && srfp.inside == true)
154                nestLevel = srfp.tn.nestLevel;
155        }
156
157                // Process this unit's effect on each txn
158                RWSet stmtRead = null;
159                RWSet stmtWrite = null;
160        Iterator outIt = out.iterator();
161        boolean printed = false;
162        while(outIt.hasNext())
163        {
164            SynchronizedRegionFlowPair srfp = (SynchronizedRegionFlowPair) outIt.next();
165            CriticalSection tn = srfp.tn;
166            
167            // Check if we are revisting the start of this existing transaction
168            if(tn.entermonitor == stmt)
169            {
170                srfp.inside = true;
171                addSelf = false; // this transaction already exists...
172            }
173            
174            // if this is the immediately enclosing transaction
175                if(srfp.inside == true && (tn.nestLevel == nestLevel || optionOpenNesting == false))
176                {
177                        printed = true; // for debugging purposes, indicated that we'll print a debug output for this statement
178                        
179                // Add this unit to the current transactional region
180                if(!tn.units.contains(unit))
181                        tn.units.add(unit);
182                        
183                        // Check what kind of statement this is
184                        // If it contains an invoke, save it for later processing as part of this transaction
185                        // If it is a monitorexit, mark that it's the end of the transaction
186                        // Otherwise, add it's read/write sets to the transaction's read/write sets
187                if(stmt.containsInvokeExpr())
188                {
189                        // Note if this unit is a call to wait() or notify()/notifyAll()
190                        String InvokeSig = stmt.getInvokeExpr().getMethod().getSubSignature();
191                        if((InvokeSig.equals("void notify()") || InvokeSig.equals("void notifyAll()")) && tn.nestLevel == nestLevel) // only applies to outermost txn
192                        {
193                                        if(!tn.notifys.contains(unit))
194                                        tn.notifys.add(unit);
195                                if(optionPrintDebug)
196                                        G.v().out.print("{x,x} ");
197                        }
198                        else if((InvokeSig.equals("void wait()") || InvokeSig.equals("void wait(long)") || InvokeSig.equals("void wait(long,int)")) && tn.nestLevel == nestLevel) // only applies to outermost txn
199                        {
200                                        if(!tn.waits.contains(unit))
201                                        tn.waits.add(unit);
202                                if(optionPrintDebug)
203                                        G.v().out.print("{x,x} ");
204                        }
205                        
206                        if(!tn.invokes.contains(unit))
207                        {
208                                // Mark this unit for later read/write set calculation (must be deferred until all tns have been found)
209                                tn.invokes.add(unit);
210                                
211                                // Debug Output
212                                if(optionPrintDebug)
213                                {
214                                                        stmtRead = tasea.readSet(tn.method, stmt, tn, new HashSet());
215                                                        stmtWrite = tasea.writeSet(tn.method, stmt, tn, new HashSet());
216
217                                                G.v().out.print("{");
218                                                if(stmtRead != null)
219                                                {
220                                                        G.v().out.print( ( (stmtRead.getGlobals()  != null ? stmtRead.getGlobals().size()  : 0)   + 
221                                                                                           (stmtRead.getFields()   != null ? stmtRead.getFields().size()   : 0) ) );
222                                                }
223                                                else
224                                                        G.v().out.print( "0" );
225                                                G.v().out.print(",");
226                                                if(stmtWrite != null)
227                                                {
228                                                        G.v().out.print( ( (stmtWrite.getGlobals() != null ? stmtWrite.getGlobals().size() : 0)   + 
229                                                                                           (stmtWrite.getFields()  != null ? stmtWrite.getFields().size()  : 0) ) );
230                                                }
231                                                else
232                                                        G.v().out.print( "0" );
233                                                G.v().out.print("} ");
234                                                }
235                            }
236                }
237                else if(unit instanceof ExitMonitorStmt && tn.nestLevel == nestLevel) // only applies to outermost txn
238                {
239                        // Mark this as end of this tn
240                        srfp.inside = false;
241                        
242                        // Check if this is an early end or fallthrough end
243                        Stmt nextUnit = stmt;
244                        do
245                        {
246                                nextUnit = (Stmt) units.getSuccOf(nextUnit);
247                        } while (nextUnit instanceof JNopStmt);
248                        if( nextUnit instanceof ReturnStmt ||
249                                nextUnit instanceof ReturnVoidStmt ||
250                                nextUnit instanceof ExitMonitorStmt )
251                        {
252                                tn.earlyEnds.add(new Pair(nextUnit, stmt)); // <early end stmt, exitmonitor stmt>
253                        }
254                        else if( nextUnit instanceof GotoStmt )
255                        {
256                                tn.end = new Pair(nextUnit, stmt); // <end stmt, exitmonitor stmt>
257                                tn.after = (Stmt) ((GotoStmt) nextUnit).getTarget();
258                        }
259                        else if( nextUnit instanceof ThrowStmt )
260                        {
261                                tn.exceptionalEnd = new Pair(nextUnit, stmt);
262                        }
263                        else
264                                throw new RuntimeException("Unknown bytecode pattern: exitmonitor not followed by return, exitmonitor, goto, or throw");
265
266                        if(optionPrintDebug)
267                                G.v().out.print("[0,0] ");
268                }
269                                else
270                {
271                        // Add this unit's read and write sets to this transactional region
272                        HashSet uses = new HashSet();
273                        stmtRead = tasea.readSet( method, stmt, tn, uses );
274                                stmtWrite = tasea.writeSet( method, stmt, tn, uses );
275
276                                tn.read.union(stmtRead);
277                                tn.write.union(stmtWrite);
278                                                                
279                                // Debug Output
280                        if(optionPrintDebug)
281                                {
282                                        G.v().out.print("[");
283                                        if(stmtRead != null)
284                                        {
285                                                G.v().out.print( ( (stmtRead.getGlobals()  != null ? stmtRead.getGlobals().size()  : 0)   + 
286                                                                                   (stmtRead.getFields()   != null ? stmtRead.getFields().size()   : 0) ) );
287                                        }
288                                        else
289                                                G.v().out.print( "0" );
290                                        G.v().out.print(",");
291                                        if(stmtWrite != null)
292                                        {
293                                                G.v().out.print( ( (stmtWrite.getGlobals() != null ? stmtWrite.getGlobals().size() : 0)   + 
294                                                                                   (stmtWrite.getFields()  != null ? stmtWrite.getFields().size()  : 0) ) );
295                                        }
296                                        else
297                                                G.v().out.print( "0" );
298                                        G.v().out.print("] ");
299                                        }
300                        }
301                        }
302        }
303        
304                // DEBUG output
305            if(optionPrintDebug)
306                {
307                        if(!printed)
308                        {
309                        G.v().out.print("[0,0] ");
310                        }
311                        G.v().out.println(unit.toString());
312                        
313                        // If this unit is an invoke statement calling a library function and the R/W sets are huge, print out the targets
314                        if(stmt.containsInvokeExpr() && 
315                                stmt.getInvokeExpr().getMethod().getDeclaringClass().toString().startsWith("java.") &&
316                                stmtRead != null && stmtWrite != null)
317                                {
318                                        if(stmtRead.size() < 25 && stmtWrite.size() < 25)
319                                        {
320                                                G.v().out.println("        Read/Write Set for LibInvoke:");
321                                                G.v().out.println("Read Set:(" + stmtRead.size() + ")" + stmtRead.toString().replaceAll("\n", "\n        "));
322                                                G.v().out.println("Write Set:(" + stmtWrite.size() + ")" + stmtWrite.toString().replaceAll("\n", "\n        "));
323                                        }
324                                }
325                }
326                
327                // If this statement was a monitorenter, and no transaction object yet exists for it,
328                // create one.
329        if(addSelf)
330        {
331                CriticalSection newTn = new CriticalSection(false, method, nestLevel + 1);
332                        newTn.entermonitor = stmt;
333                        newTn.beginning = (Stmt) units.getSuccOf(stmt);
334                        if(stmt instanceof EnterMonitorStmt)
335                                newTn.origLock = ((EnterMonitorStmt) stmt).getOp();
336                                
337                if(optionPrintDebug)
338                        G.v().out.println("Transaction found in method: " + newTn.method.toString());
339                        out.add(new SynchronizedRegionFlowPair(newTn, true));
340                        
341                        // This is a really stupid way to find out which prep applies to this txn.
342                        Iterator<Object> prepUnitsIt = prepUnits.iterator();
343                        while(prepUnitsIt.hasNext())
344                        {
345                                Unit prepUnit = (Unit) prepUnitsIt.next();
346                                
347                                Iterator<UnitValueBoxPair> uses = slu.getUsesOf(prepUnit).iterator();
348                        while(uses.hasNext())
349                        {
350                                UnitValueBoxPair use = (UnitValueBoxPair) uses.next();
351                                if(use.getUnit() == (Unit) unit)
352                                {// if this transaction's monitorenter statement is one of the uses of this preparatory unit
353                                        newTn.prepStmt = (Stmt) prepUnit;
354                                }
355                        }
356
357                        }
358                }
359    }

361    /**
362     * union
363     **/
364    protected void merge(FlowSet<SynchronizedRegionFlowPair> inSet1, FlowSet<SynchronizedRegionFlowPair> inSet2, FlowSet<SynchronizedRegionFlowPair> outSet)
365    {
366                inSet1.union(inSet2, outSet);
367    }
368
369    protected void copy(FlowSet<SynchronizedRegionFlowPair> sourceSet, FlowSet<SynchronizedRegionFlowPair> destSet)
370    {
371                destSet.clear();
372
373                Iterator<SynchronizedRegionFlowPair> it = sourceSet.iterator();
374                while(it.hasNext())
375                {
376                        SynchronizedRegionFlowPair tfp = it.next();
377                        destSet.add((SynchronizedRegionFlowPair)tfp.clone());
378                }
379    }
