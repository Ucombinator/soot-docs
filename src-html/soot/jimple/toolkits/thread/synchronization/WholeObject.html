031/** Generates side-effect information from a PointsToAnalysis. 
032 *  Uses various heuristic rules to filter out side-effects that 
033 *  are not visible to other threads in a Transactional program.
034 */
035class WholeObject
036{
037        Type type;
038        
039        public WholeObject(Type type)
040        {
041                this.type = type;
042        }
043        
044        public WholeObject()
045        {
046                this.type = null;
047        }
048
049        public String toString()
050        {
051                return "All Fields" + (type == null ? "" : " (" + type + ")");
052        }
053        
054        public int hashCode()
055        {
056                if(type == null)
057                        return 1;
058                return type.hashCode();
059        }
060        
061        public boolean equals(Object o)
062        {
063                if(type == null)
064                        return true;
065                if(o instanceof WholeObject)
066                {
067                        WholeObject other = (WholeObject) o;
068                        if(other.type == null)
069                                return true;
070                        else
071                                return (type == other.type);
072                }
073                else if(o instanceof FieldRef)
074                {
075                        return type == ((FieldRef)o).getType();
076                }
077                else if(o instanceof SootFieldRef)
078                {
079                        return type == ((SootFieldRef)o).type();
080                }
081                else if(o instanceof SootField)
082                {
083                        return type == ((SootField)o).getType();
084                }
085                else
086                {
087                        return true;
088                }
089        }
090}

111        public void findNTRWSets( SootMethod method ) {
112                if( methodToNTReadSet.containsKey( method )
113                                && methodToNTWriteSet.containsKey( method ) ) return;
114                
115                CodeBlockRWSet read = null;
116                CodeBlockRWSet write = null;
117                for( Iterator sIt = method.retrieveActiveBody().getUnits().iterator(); sIt.hasNext(); ) {
118                        final Stmt s = (Stmt) sIt.next();
119                        
120                        boolean ignore = false;
121
122                        // Ignore Reads/Writes inside another transaction
123                        if(criticalSections != null)
124                        {
125                                Iterator<CriticalSection> tnIt = criticalSections.iterator();
126                                while(tnIt.hasNext())
127                                {
128                                        CriticalSection tn = tnIt.next();
129                                        if(tn.units.contains(s) || tn.prepStmt == s)
130                                        {
131                                                ignore = true;
132                                                break;
133                                        }
134                                }
135                        }
136                        
137                        if(!ignore)
138                        {
139                                RWSet ntr = ntReadSet( method, s );
140                                if( ntr != null ) {
141                                        if( read == null ) read = new CodeBlockRWSet();
142                                        read.union( ntr );
143                                }
144                                RWSet ntw = ntWriteSet( method, s );
145                                if( ntw != null ) {
146                                        if( write == null ) write = new CodeBlockRWSet();
147                                        write.union( ntw );
148                                }
149                                if( s.containsInvokeExpr() )
150                                {
151                                        InvokeExpr ie = s.getInvokeExpr();
152                                        SootMethod calledMethod = ie.getMethod();
153                                        
154                                        // if it's an invoke on certain lib methods
155                                        if( calledMethod.getDeclaringClass().toString().startsWith("java.util") ||
156                                                calledMethod.getDeclaringClass().toString().startsWith("java.lang") )
157                                        {
158                                                // then it gets approximated
159                                                Local base = null;
160                                                if(ie instanceof InstanceInvokeExpr)
161                                                        base = (Local)((InstanceInvokeExpr) ie).getBase();
162
163                                                if(tlo == null || base == null || !tlo.isObjectThreadLocal(base, method) )
164                                                {
165                                                        // add its approximated read set to read
166                                        RWSet r;
167        //                              String InvokeSig = calledMethod.getSubSignature();
168        //                              if( InvokeSig.equals("void notify()") || InvokeSig.equals("void notifyAll()") || 
169        //                                      InvokeSig.equals("void wait()") || InvokeSig.equals("void wait(long)") || InvokeSig.equals("void wait(long,int)"))
170        //                                                      r = approximatedReadSet(method, s, base, true);
171        //                                              else
172                                                                r = approximatedReadSet(method, s, base, true);
173                                                        if( read == null ) read = new CodeBlockRWSet();
174                                                        if( r != null )
175                                                                read.union( r );
176
177                                                        // add its approximated write set to write
178                                                        RWSet w;
179        //                              if( InvokeSig.equals("void notify()") || InvokeSig.equals("void notifyAll()") || 
180        //                                      InvokeSig.equals("void wait()") || InvokeSig.equals("void wait(long)") || InvokeSig.equals("void wait(long,int)"))
181        //                                                      w = approximatedWriteSet(method, s, base, true);
182        //                                              else
183                                                                w = approximatedWriteSet(method, s, base, true);
184                                                        if( write == null ) write = new CodeBlockRWSet();
185                                                        if( w != null )
186                                                                write.union( w );
187                                                }
188                                        }
189                                }
190                        }
191                }
192                methodToNTReadSet.put( method, read );
193                methodToNTWriteSet.put( method, write );
194        }

261        private RWSet ntReadSet( SootMethod method, Stmt stmt )
262        {
263                if( stmt instanceof AssignStmt ) {
264                        AssignStmt a = (AssignStmt) stmt;
265                        Value r = a.getRightOp();
266                        if(r instanceof NewExpr) // IGNORE NEW STATEMENTS
267                                return null;
268                        return addValue( r, method, stmt );
269                }
270                return null;
271        }

359                        for(int i = 0; i < argCount; i++)

363                        AssignStmt a = (AssignStmt) stmt;
364                        Value r = a.getRightOp();

372                boolean ignore = false;
373                if(stmt.containsInvokeExpr())
374                {
375                        InvokeExpr ie = stmt.getInvokeExpr();
376                        SootMethod calledMethod = ie.getMethod();
377                        if(ie instanceof StaticInvokeExpr)
378                        {
379                                // ignore = false;
380                        }
381                        else if(ie instanceof InstanceInvokeExpr)
382                        {
383                                if(calledMethod.getSubSignature().startsWith("void <init>") && eoa.isInitMethodPureOnObject(calledMethod))
384                                {
385                                        ignore = true;
386                                }
387                                else if(tlo != null && !tlo.hasNonThreadLocalEffects(method, ie))
388                                {
389                                        ignore = true;
390                                }
391                        }
392                }
393
394                boolean inaccessibleUses = false;
395                RWSet ret = new CodeBlockRWSet();
396                tve.setExemptTransaction(tn);

398                while( !ignore && targets.hasNext() )
399                {

407

411                                if( target.getDeclaringClass().toString().startsWith("java.util") ||
412                                        target.getDeclaringClass().toString().startsWith("java.lang") )
413                                {
414/*                                      RWSet ntr;

454//                                              uses.clear();
455//                                              inaccessibleUses = true;
456                                                ret.union( ntr );
457                                        }
458                                }
459                        }
460                }
461                RWSet ntr = ntReadSet( method, stmt );
462                if( inaccessibleUses == false && ntr != null && stmt instanceof AssignStmt ) {
463                        AssignStmt a = (AssignStmt) stmt;
464                        Value r = a.getRightOp();

466                        {
467                                uses.add( ((InstanceFieldRef)r).getBase() );
468                        }
469                        else if(r instanceof StaticFieldRef)
470                        {
471                                uses.add( r );
472                        }
473                        else if(r instanceof ArrayRef)
474                        {
475                                uses.add( ((ArrayRef)r).getBase() );
476                        }
477                }
478                ret.union( ntr );
479
480                if( stmt.containsInvokeExpr() )
481                {
482                        InvokeExpr ie = stmt.getInvokeExpr();
483                        SootMethod calledMethod = ie.getMethod();

489                                // then it gets approximated as a NTReadSet
490                                Local base = null;
491                                if(ie instanceof InstanceInvokeExpr)
492                                        base = (Local)((InstanceInvokeExpr) ie).getBase();
493
494                                if(tlo == null || base == null || !tlo.isObjectThreadLocal(base, method) )
495                                {
496                                        // add its approximated read set to read
497                                        RWSet r;
498        //                  String InvokeSig = calledMethod.getSubSignature();
499        //                      if( InvokeSig.equals("void notify()") || InvokeSig.equals("void notifyAll()") ||

754        public RWSet valueRWSet( Value v, SootMethod m, Stmt s, CriticalSection tn )
755        {
756                RWSet ret = null;
757                
758                if(tlo != null)
759                {
760                        // fields/elements of local objects may be read/written w/o visible
761                        // side effects if the base object is local, or if the base is "this"
762                        // and the field itself is local (since "this" is always assumed shared)
763                        if( v instanceof InstanceFieldRef )
764                        {
765                                InstanceFieldRef ifr = (InstanceFieldRef) v;
766                                if( m.isConcrete() && !m.isStatic() && 
767                                        m.retrieveActiveBody().getThisLocal().equivTo(ifr.getBase()) && 
768                                        tlo.isObjectThreadLocal(ifr, m) )
769                                        return null;
770                                else if( tlo.isObjectThreadLocal(ifr.getBase(), m) )
771                                        return null;
772                        }
773                        else if( v instanceof ArrayRef && tlo.isObjectThreadLocal(((ArrayRef)v).getBase(), m) )
774                                return null;
775                }
776                
777                if( v instanceof InstanceFieldRef ) {
778                        InstanceFieldRef ifr = (InstanceFieldRef) v;
779                        PointsToSet base = pa.reachingObjects( (Local) ifr.getBase() );
780                        ret = new StmtRWSet();
781                        ret.addFieldRef( base, ifr.getField() );
782                }
783                else if( v instanceof StaticFieldRef )
784                {
785                        StaticFieldRef sfr = (StaticFieldRef) v;
786                        ret = new StmtRWSet();
787                        ret.addGlobal( sfr.getField() );
788                } 
789                else if( v instanceof ArrayRef )
790                {
791                        ArrayRef ar = (ArrayRef) v;
792                        PointsToSet base = pa.reachingObjects( (Local) ar.getBase() );
793                        ret = new StmtRWSet();
794                        ret.addFieldRef( base, PointsToAnalysis.ARRAY_ELEMENTS_NODE );
795                }
796                else if( v instanceof Local )
797                {
798                        Local vLocal = (Local) v;
799                        PointsToSet base = pa.reachingObjects( vLocal );
800                        ret = new CodeBlockRWSet();
801                        CodeBlockRWSet stmtRW = new CodeBlockRWSet();
802                        RWSet rSet = readSet(m, s, tn, new HashSet());
803                        if(rSet != null)
804                                stmtRW.union(rSet);
805                        RWSet wSet = writeSet(m, s, tn, new HashSet());
806                        if(wSet != null)
807                                stmtRW.union(wSet);
808                        // Should actually get a list of fields of this object that are read/written
809                        // make fake RW set of <base, all fields> (use a special class)
810                        // intersect with the REAL RW set of this stmt
811                        for(Iterator fieldsIt = stmtRW.getFields().iterator(); fieldsIt.hasNext(); )
812                        {
813                                Object field = fieldsIt.next();
814                                PointsToSet fieldBase = stmtRW.getBaseForField(field);
815                                if(base.hasNonEmptyIntersection(fieldBase))
816                                        ret.addFieldRef(base, field); // should use intersection of bases!!!
817                        }
818                }
819                else
820                {
821                        return null;
822                }
823                return ret;
824        }
