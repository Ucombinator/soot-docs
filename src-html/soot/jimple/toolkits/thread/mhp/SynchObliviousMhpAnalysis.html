020/** UnsynchronizedMhpAnalysis written by Richard L. Halpert 2006-12-09
021 *  Calculates May-Happen-in-Parallel (MHP) information as if in the absence
022 *  of synchronization. Any synchronization statements (synchronized, wait, 
023 *  notify, etc.) are ignored. If the program has no synchronization, then this 
024 *  actually generates correct MHP. This is useful if you are trying to generate
025 *  (replacement) synchronization. It is also useful if an approximation is
026 *  acceptable, because it runs much faster than a synch-aware MHP analysis.
027 *
028 *  This analysis uses may-alias information to determine the types of threads
029 *  launched and the call graph to determine which methods they may call.
030 *  This analysis uses a run-once/run-one-at-a-time/run-many classification to
031 *  determine if a thread may be run in parallel with itself.
032 */
033
034public class SynchObliviousMhpAnalysis implements MhpTester, Runnable
035{
036        List<AbstractRuntimeThread> threadList;
037        boolean optionPrintDebug;
038        boolean optionThreaded = false; // DOESN'T WORK if set to true... ForwardFlowAnalysis uses a static field in a thread-unsafe way
039        
040        Thread self;
041        
042        public SynchObliviousMhpAnalysis()
043        {
044                threadList = new ArrayList<AbstractRuntimeThread>();
045                optionPrintDebug = false;
046
047                self = null;
048
049                buildThreadList();
050        }
051
052        protected void buildThreadList() // can only be run once if optionThreaded is true
053        {
054                if(optionThreaded)
055                {
056                        if(self != null)
057                                return; // already running... do nothing
058
059                        self = new Thread(new SootThreadGroup(), this);
060                        self.start();
061                }
062                else
063                        run();
064        }
065
066        public void run()
067        {
068                SootMethod mainMethod = Scene.v().getMainClass().getMethodByName("main");
069
070                PointsToAnalysis pta = Scene.v().getPointsToAnalysis();
071                if (pta instanceof DemandCSPointsTo) {
072                        DemandCSPointsTo demandCSPointsTo = (DemandCSPointsTo) pta;
073                        pta = demandCSPointsTo.getPAG();
074                }               
075                if (!(pta instanceof PAG))
076                {
077                   throw new RuntimeException("You must use Spark for points-to analysis when computing MHP information!");
078                }
079                PAG pag = (PAG) pta;
080                SparkOptions so = pag.getOpts();
081                if(so.rta())
082                   throw new RuntimeException("MHP cannot be calculated using RTA due to incomplete call graph");
083
084                CallGraph callGraph = Scene.v().getCallGraph();
085
086                // Get a call graph trimmed to contain only the relevant methods (non-lib, non-native)
087//              G.v().out.println("    MHP: PegCallGraph");
088                PegCallGraph pecg = new PegCallGraph(callGraph);
089            
090            // Find allocation nodes that are run more than once
091            // Also find methods that are run more than once
092//              G.v().out.println("    MHP: AllocNodesFinder");
093                AllocNodesFinder anf = new AllocNodesFinder(pecg, callGraph, (PAG) pta);
094                Set<AllocNode> multiRunAllocNodes = anf.getMultiRunAllocNodes();
095                Set<SootMethod> multiCalledMethods = anf.getMultiCalledMethods();
096
097                // Find Thread.start() and Thread.join() statements (in live code)
098//              G.v().out.println("    MHP: StartJoinFinder");
099                StartJoinFinder sjf = new StartJoinFinder(callGraph, (PAG) pta); // does analysis
100                Map<Stmt, List<AllocNode>> startToAllocNodes = sjf.getStartToAllocNodes();
101                Map<Stmt, List<SootMethod>> startToRunMethods = sjf.getStartToRunMethods();
102                Map<Stmt, SootMethod> startToContainingMethod = sjf.getStartToContainingMethod();
103                Map<Stmt, Stmt> startToJoin = sjf.getStartToJoin();
104                
105                // Build MHP Lists
106//              G.v().out.println("    MHP: Building MHP Lists");
107                List<AbstractRuntimeThread> runAtOnceCandidates = new ArrayList<AbstractRuntimeThread>();
108                Iterator threadIt = startToRunMethods.entrySet().iterator();
109                int threadNum = 0;
110                while(threadIt.hasNext())
111                {
112                        // Get list of possible Runnable.run methods (actually, a list of peg chains)
113                        // and a list of allocation sites for this thread start statement
114                        // and the thread start statement itself
115                        Map.Entry e = (Map.Entry) threadIt.next();
116                        Stmt startStmt = (Stmt) e.getKey();
117                        List runMethods = (List) e.getValue();
118                        List threadAllocNodes = startToAllocNodes.get(e.getKey());
119
120                        // Get a list of all possible unique Runnable.run methods for this thread start statement
121                        AbstractRuntimeThread thread = new AbstractRuntimeThread(); // provides a list interface to the methods in a thread's sub-call-graph
122                        thread.setStartStmt(startStmt);
123//                      List threadMethods = new ArrayList();
124                        Iterator runMethodsIt = runMethods.iterator();
125                        while(runMethodsIt.hasNext())
126                        {
127                                SootMethod method = (SootMethod) runMethodsIt.next();
128                                if(!thread.containsMethod(method))
129                                {
130                                        thread.addMethod(method);
131                                        thread.addRunMethod(method);
132                                }
133                        }
134                        
135                        // Get a list containing all methods in the call graph(s) rooted at the possible run methods for this thread start statement
136                        // AKA a list of all methods that might be called by the thread started here
137                        int methodNum = 0;
138                        while(methodNum < thread.methodCount()) // iterate over all methods in threadMethods, even as new methods are being added to it
139                        {
140                                Iterator succMethodsIt = pecg.getSuccsOf(thread.getMethod(methodNum)).iterator();
141                                while(succMethodsIt.hasNext())
142                                {
143                                        SootMethod method = (SootMethod) succMethodsIt.next();
144                                        // if all edges into this method are of Kind THREAD, ignore it 
145                                        // (because it's a run method that won't be called as part of THIS thread) THIS IS NOT OPTIMAL
146                                        boolean ignoremethod = true;
147                                        Iterator edgeInIt = callGraph.edgesInto(method);
148                                        while(edgeInIt.hasNext())
149                                        {
150                                                Edge edge = (Edge) edgeInIt.next();
151                                                if( edge.kind() != Kind.THREAD
152                                                                && edge.kind() != Kind.EXECUTOR
153                                                                && edge.kind() != Kind.ASYNCTASK
154                                                                && thread.containsMethod(edge.src())) // called directly by any of the thread methods?
155                                                        ignoremethod = false;
156                                        }
157                                        if(!ignoremethod && !thread.containsMethod(method))
158                                                thread.addMethod(method);
159                                }
160                                methodNum++;
161                        }
162                        
163                        // Add this list of methods to MHPLists
164                        threadList.add(thread);
165                        if(optionPrintDebug)
166                                System.out.println(thread.toString());
167                        
168                        // Find out if the "thread" in "thread.start()" could be more than one object
169                        boolean mayStartMultipleThreadObjects = (threadAllocNodes.size() > 1) || so.types_for_sites();
170                        if(!mayStartMultipleThreadObjects) // if there's only one alloc node
171                        {
172                                if(multiRunAllocNodes.contains(threadAllocNodes.iterator().next())) // but it gets run more than once
173                                {
174                                        mayStartMultipleThreadObjects = true; // then "thread" in "thread.start()" could be more than one object
175                                }
176                        }
177                        
178                        if(mayStartMultipleThreadObjects)
179                                thread.setStartStmtHasMultipleReachingObjects();
180                        
181                        // Find out if the "thread.start()" statement may be run more than once
182                        SootMethod startStmtMethod = startToContainingMethod.get(startStmt);
183                        thread.setStartStmtMethod(startStmtMethod);
184                        boolean mayBeRunMultipleTimes = multiCalledMethods.contains(startStmtMethod); // if method is called more than once...
185                        if(!mayBeRunMultipleTimes)
186                        {
187                                UnitGraph graph = new CompleteUnitGraph(startStmtMethod.getActiveBody());
188                                MultiRunStatementsFinder finder = new MultiRunStatementsFinder(
189                                        graph, startStmtMethod, multiCalledMethods, callGraph);
190                                FlowSet multiRunStatements = finder.getMultiRunStatements(); // list of all units that may be run more than once in this method
191                                if(multiRunStatements.contains(startStmt))
192                                        mayBeRunMultipleTimes = true;
193                        }
194                        
195                        if(mayBeRunMultipleTimes)
196                        {
197                                thread.setStartStmtMayBeRunMultipleTimes();
198                        }
199
200                        // If a run-many thread.start() statement is (always) associated with a join statement in the same method,
201                        // then it may be possible to treat it as run-once, if this method is non-reentrant and called only
202                        // by one thread (sounds strict, but actually this is the most common case)
203                        if(mayBeRunMultipleTimes && startToJoin.containsKey(startStmt))
204                        {
205                                thread.setJoinStmt(startToJoin.get(startStmt));
206                                mayBeRunMultipleTimes = false; // well, actually, we don't know yet
207                                methodNum = 0;
208                                List<SootMethod> containingMethodCalls = new ArrayList<SootMethod>();
209                                containingMethodCalls.add(startStmtMethod);
210                                while(methodNum < containingMethodCalls.size()) // iterate over all methods in threadMethods, even as new methods are being added to it
211                                {
212                                        Iterator succMethodsIt = pecg.getSuccsOf(containingMethodCalls.get(methodNum)).iterator();
213                                        while(succMethodsIt.hasNext())
214                                        {
215                                                SootMethod method = (SootMethod) succMethodsIt.next();
216                                                if(method == startStmtMethod)
217                                                {// this method is reentrant
218                                                        mayBeRunMultipleTimes = true; // this time it's for sure
219                                                        thread.setStartMethodIsReentrant();
220                                                        thread.setRunsMany();
221                                                        break;
222                                                }
223                                                if(!containingMethodCalls.contains(method))
224                                                        containingMethodCalls.add(method);
225                                        }
226                                        methodNum++;
227                                }
228                                if(!mayBeRunMultipleTimes)
229                                {// There's still one thing that might cause this to be run multiple times: if it can be run in parallel with itself
230                                 // but we can't find that out 'till we're done
231                                        runAtOnceCandidates.add(thread);
232                                }
233                        }
234
235                        // If more than one thread might be started at this start statement,
236                        // and this start statement may be run more than once,
237                        // then add this list of methods to MHPLists *AGAIN*
238                        if(optionPrintDebug)
239                                System.out.println("Start Stmt " + startStmt.toString() + 
240                                        " mayStartMultipleThreadObjects=" + mayStartMultipleThreadObjects + " mayBeRunMultipleTimes=" + mayBeRunMultipleTimes);
241                        if(mayStartMultipleThreadObjects && mayBeRunMultipleTimes)
242                        {
243                                threadList.add(thread); // add another copy
244                                thread.setRunsMany();
245                                if(optionPrintDebug)
246                                        System.out.println(thread.toString());
247                        }
248                        else
249                                thread.setRunsOnce();
250                        threadNum++;
251                }
252
253                // do same for main method
254                AbstractRuntimeThread mainThread = new AbstractRuntimeThread();
255//              List mainMethods = new ArrayList();
256                threadList.add(mainThread);
257                mainThread.setRunsOnce();
258                mainThread.addMethod(mainMethod);
259                mainThread.addRunMethod(mainMethod);
260                mainThread.setIsMainThread();
261                // get all the successors, add to threadMethods
262                int methodNum = 0;
263                while(methodNum < mainThread.methodCount())
264                {
265                        Iterator succMethodsIt = pecg.getSuccsOf(mainThread.getMethod(methodNum)).iterator();
266                        while(succMethodsIt.hasNext())
267                        {
268                                SootMethod method = (SootMethod) succMethodsIt.next();
269                                // if all edges into this are of Kind THREAD, ignore it
270                                boolean ignoremethod = true;
271                                Iterator edgeInIt = callGraph.edgesInto(method);
272                                while(edgeInIt.hasNext())
273                                {
274                                        if( ((Edge) edgeInIt.next()).kind() != Kind.THREAD )
275                                                ignoremethod = false;
276                                }
277                                if(!ignoremethod && !mainThread.containsMethod(method))
278                                        mainThread.addMethod(method);
279                        }
280                        methodNum++;
281                }
282                if(optionPrintDebug)
283                        G.v().out.println(mainThread.toString());
284                        
285                // Revisit the containing methods of start-join pairs that are non-reentrant but might be called in parallel
286                boolean addedNew = true;
287                while(addedNew)
288                {
289                        addedNew = false;
290                        ListIterator<AbstractRuntimeThread> it = runAtOnceCandidates.listIterator();
291                        while(it.hasNext())
292                        {
293                                AbstractRuntimeThread someThread = it.next();
294                                SootMethod someStartMethod = someThread.getStartStmtMethod();
295                                if(mayHappenInParallelInternal(someStartMethod, someStartMethod))
296                                {
297                                        threadList.add(someThread); // add a second copy of it
298                                        someThread.setStartMethodMayHappenInParallel();
299                                        someThread.setRunsMany();
300                                        it.remove();
301                                        if(optionPrintDebug)
302                                                G.v().out.println(someThread.toString());
303                                        addedNew = true;
304                                }
305                        }
306                }
307                
308                // mark the remaining threads here as run-one-at-a-time
309                Iterator<AbstractRuntimeThread> it = runAtOnceCandidates.iterator();
310                while(it.hasNext())
311                {
312                        AbstractRuntimeThread someThread = it.next();
313                        someThread.setRunsOneAtATime();
314                }
315        }

317    public boolean mayHappenInParallel(SootMethod m1, Unit u1, SootMethod m2, Unit u2)
318    {
319                if(optionThreaded)
320                {
321                        if(self == null)
322                                return true; // not started...
323
324                        // Wait until finished
325                        G.v().out.println("[mhp] waiting for analysis thread to finish");
326                        try
327                        {
328                                self.join();
329                        }
330                        catch(InterruptedException ie)
331                        {
332                                return true;
333                        }
334                }
335
336                return mayHappenInParallelInternal(m1, m2);
337        }

360
361    private boolean mayHappenInParallelInternal(SootMethod m1, SootMethod m2)
362    {
363        if(threadList == null) // not run
364        {
365                return true;
366                }
367
368                int size = threadList.size();
369                for(int i = 0; i < size; i++)
370                {
371                        if(threadList.get(i).containsMethod(m1))
372                        {
373                                for(int j = 0; j < size; j++)
374                                {
375                                        if(threadList.get(j).containsMethod(m2) && i != j)
376                                        {
377                                                return true;
378                                        }
379                                }
380                        }
381                }
382                return false;
383        }

385        public void printMhpSummary()
386        {
387                if(optionThreaded)
388                {
389                        if(self == null)
390                                return; // not run... do nothing
391
392                        // Wait until finished
393                        G.v().out.println("[mhp] waiting for thread to finish");
394                        try
395                        {
396                                self.join();
397                        }
398                        catch(InterruptedException ie)
399                        {
400                                return;
401                        }
402                }
403
404                List<AbstractRuntimeThread> threads = new ArrayList<AbstractRuntimeThread>();
405                int size = threadList.size();
406                G.v().out.println("[mhp]");
407                for(int i = 0; i < size; i++)
408                {
409                        if( !threads.contains(threadList.get(i)) )
410                        {
411                                G.v().out.println("[mhp] " + 
412                                        threadList.get(i).toString().replaceAll(
413                                                "\n", "\n[mhp] ").replaceAll(
414                                                ">,",">\n[mhp]  "));
415                                G.v().out.println("[mhp]");
416                        }
417                        threads.add(threadList.get(i));
418                }
419        }

421        public List<SootClass> getThreadClassList()
422        {
423                if(optionThreaded)
424                {
425                        if(self == null)
426                                return null; // not run... do nothing
427
428                        // Wait until finished
429                        G.v().out.println("[mhp] waiting for thread to finish");
430                        try
431                        {
432                                self.join();
433                        }
434                        catch(InterruptedException ie)
435                        {
436                                return null;
437                        }
438                }
439                
440                if(threadList == null)
441                        return null;
442                
443                List<SootClass> threadClasses = new ArrayList<SootClass>();
444                int size = threadList.size();
445                for(int i = 0; i < size; i++)
446                {
447                        AbstractRuntimeThread thread = threadList.get(i);
448                        Iterator<Object> threadRunMethodIt = thread.getRunMethods().iterator();
449                        while(threadRunMethodIt.hasNext())
450                        {
451                                SootClass threadClass = ((SootMethod) threadRunMethodIt.next()).getDeclaringClass(); // what about subclasses???
452                                if( !threadClasses.contains(threadClass) && threadClass.isApplicationClass() ) // only include application threads
453                                        threadClasses.add(threadClass);
454                        }
455                }
456                return threadClasses;
457        }

459        public List<AbstractRuntimeThread> getThreads()
460        {
461                if(optionThreaded)
462                {
463                        if(self == null)
464                                return null; // not run... do nothing
465
466                        // Wait until finished
467                        G.v().out.println("[mhp] waiting for thread to finish");
468                        try
469                        {
470                                self.join();
471                        }
472                        catch(InterruptedException ie)
473                        {
474                                return null;
475                        }
476                }
477                
478                if(threadList == null)
479                        return null;
480
481                List<AbstractRuntimeThread> threads = new ArrayList<AbstractRuntimeThread>();
482                int size = threadList.size();
483                for(int i = 0; i < size; i++)
484                {
485                        if( !threads.contains(threadList.get(i)) )
486                        {
487                                threads.add(threadList.get(i));
488                        }
489                }
490                return threads;
491        }
