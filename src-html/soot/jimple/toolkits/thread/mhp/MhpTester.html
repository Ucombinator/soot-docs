001package soot.jimple.toolkits.thread.mhp;
002
003import java.util.*;
004
005import soot.*;
006import soot.jimple.toolkits.thread.AbstractRuntimeThread;
007
008/** MhpTester written by Richard L. Halpert 2007-03-15
009 *  An interface for any object that can provide May-Happen-in-Parallel info and
010 *  a list of the program's threads (List of AbstractRuntimeThreads)
011 */
012
013public interface MhpTester
014{
015    public boolean mayHappenInParallel(SootMethod m1, SootMethod m2); // method level MHP
016        
017    public boolean mayHappenInParallel(SootMethod m1, Unit u1, SootMethod m2, Unit u2); // stmt level MHP
018
019        public void printMhpSummary();
020        
021        public List<AbstractRuntimeThread> getThreads();
022}
