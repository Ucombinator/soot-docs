014public class MonitorDepth{
015        private String objName;
016        private int depth;
017        
018        MonitorDepth(String s, int d){
019                objName = s;
020                depth = d;
021        }
022        protected String getObjName(){
023                return objName;
024        }
025        protected void SetObjName(String s){
026                objName = s;
027        }
028        protected int getDepth(){
029                return depth;
030        }
031        protected void setDepth(int d){
032                depth = d;
033        }
034        protected void decreaseDepth(){
035                if (depth>0)    depth=depth-1;
036                else
037                        throw new RuntimeException("Error! You can not decrease a monitor depth of "+depth);
038        }
039        protected void increaseDepth(){
040                depth++;
041        }
042        
043}
