037/**
038 * Intra-procedural purity-graph analysis.
039 *
040 * You must pass an AbstractInterproceduralAnalysis object so that the
041 * intraprocedural part can resolve the effect of method calls.
042 * This manipulates PurityGraphBox.
043 */
044public class PurityIntraproceduralAnalysis extends ForwardFlowAnalysis
045{
046
047    AbstractInterproceduralAnalysis inter;
048
049    protected Object newInitialFlow()
050    { return new PurityGraphBox(); }
051
052    protected Object entryInitialFlow()
053    { return new PurityGraphBox(); }
054
055    protected void merge(Object in1, Object in2, Object out)
056    {
057        PurityGraphBox i1  = (PurityGraphBox)in1;
058        PurityGraphBox i2  = (PurityGraphBox)in2;
059        PurityGraphBox o   = (PurityGraphBox)out;
060        if (out!=i1) o.g = new PurityGraph(i1.g);
061        o.g.union(i2.g);
062    }
063
064    protected void copy(Object source, Object dest)
065    {
066        PurityGraphBox src  = (PurityGraphBox)source;
067        PurityGraphBox dst  = (PurityGraphBox)dest;
068        dst.g = new PurityGraph(src.g);
069    }

071
072    protected void flowThrough(Object inValue, Object unit, Object outValue)
073    {
074        PurityGraphBox i  = (PurityGraphBox)inValue;
075        PurityGraphBox o  = (PurityGraphBox)outValue;
076        Stmt         stmt = (Stmt)unit;
077
078        o.g = new PurityGraph(i.g);
079        
080        // ********************
081        // BIG PATTERN MATCHING
082        // ********************
083
084        // I throw much "match failure" Errors to ease debugging...
085        // => we could optimize the pattern matching a little bit
086
087        //G.v().out.println(" | |- exec "+stmt);
088
089
090        ///////////
091        // Calls //
092        ///////////
093        if (stmt.containsInvokeExpr()) {
094            inter.analyseCall(inValue, stmt, outValue);
095        }
096
097        /////////////
098        // AssignStmt
099        /////////////
100
101        else if (stmt instanceof AssignStmt) {
102            Value leftOp = ((AssignStmt)stmt).getLeftOp();
103            Value rightOp = ((AssignStmt)stmt).getRightOp();
104
105            // v = ...
106            if (leftOp instanceof Local) {
107                Local left = (Local)leftOp;
108
109                // remove optional cast
110                if (rightOp instanceof CastExpr) 
111                    rightOp = ((CastExpr)rightOp).getOp();
112
113                // ignore primitive types
114                if (!(left.getType() instanceof RefLikeType)) {
115                }
116
117                // v = v
118                else if (rightOp instanceof Local) {
119                    Local right = (Local) rightOp;
120                    o.g.assignLocalToLocal(right, left);
121                }
122
123                // v = v[i]
124                else if (rightOp instanceof ArrayRef) {
125                    Local right = (Local) ((ArrayRef)rightOp).getBase();
126                    o.g.assignFieldToLocal(stmt, right, "[]", left);
127                }
128
129                // v = v.f
130                else if (rightOp instanceof InstanceFieldRef) {
131                    Local  right  = 
132                        (Local) ((InstanceFieldRef)rightOp).getBase();
133                    String field = 
134                        ((InstanceFieldRef)rightOp).getField().getName();
135                    o.g.assignFieldToLocal(stmt, right, field, left);
136                }
137
138                // v = C.f
139                else if (rightOp instanceof StaticFieldRef) {
140                    o.g.localIsUnknown(left);
141                }
142
143                // v = cst
144                else if (rightOp instanceof Constant) {
145                    // do nothing...
146                }
147
148
149                // v = new / newarray / newmultiarray
150                else if (rightOp instanceof AnyNewExpr)  {
151                    o.g.assignNewToLocal(stmt, left);
152                }
153
154                // v = binary or unary operator
155                else if (rightOp instanceof BinopExpr ||
156                         rightOp instanceof UnopExpr ||
157                         rightOp instanceof InstanceOfExpr) {
158                    // do nothing...
159                }
160
161                else throw new Error("AssignStmt match failure (rightOp)"+stmt);
162            }
163
164            // v[i] = ...
165            else if (leftOp instanceof ArrayRef) {
166                Local left = (Local) ((ArrayRef)leftOp).getBase();
167
168                // v[i] = v
169                if (rightOp instanceof Local) {
170                    Local right = (Local)rightOp;
171                    if (right.getType() instanceof RefLikeType)
172                        o.g.assignLocalToField(right, left, "[]");
173                    else
174                        o.g.mutateField(left, "[]");
175                }
176
177                // v[i] = cst
178                else if (rightOp instanceof Constant)
179                    o.g.mutateField(left, "[]");
180                
181                else throw new Error("AssignStmt match failure (rightOp)"+stmt);
182            }
183
184            // v.f = ...
185            else if (leftOp instanceof InstanceFieldRef) {
186                Local  left  = (Local) ((InstanceFieldRef)leftOp).getBase();
187                String field = ((InstanceFieldRef)leftOp).getField().getName();
188
189                // v.f = v
190                if (rightOp instanceof Local) {
191                    Local right = (Local)rightOp;
192                    // ignore primitive types
193                    if (right.getType() instanceof RefLikeType)
194                        o.g.assignLocalToField(right, left, field);
195                    else
196                        o.g.mutateField(left, field);
197                }
198
199                // v.f = cst
200                else if (rightOp instanceof Constant) 
201                    o.g.mutateField(left, field);
202                
203                else throw new Error("AssignStmt match failure (rightOp) "+stmt);
204            }
205
206            // C.f = ...
207            else if (leftOp instanceof StaticFieldRef) {
208                String field = ((StaticFieldRef)leftOp).getField().getName();
209                
210                // C.f = v
211                if (rightOp instanceof Local) {
212                    Local right = (Local)rightOp;
213                    if (right.getType() instanceof RefLikeType)
214                        o.g.assignLocalToStaticField(right, field);
215                    else
216                        o.g.mutateStaticField(field);
217                }
218
219                // C.f = cst
220                else if (rightOp instanceof Constant)
221                    o.g.mutateStaticField(field);
222
223                else throw new Error("AssignStmt match failure (rightOp) "+stmt);
224            }
225
226            else throw new Error("AssignStmt match failure (leftOp) "+stmt);
227
228        }
229
230
231        ///////////////
232        // IdentityStmt
233        ///////////////
234
235        else if (stmt instanceof IdentityStmt) {
236            Local left    = (Local)((IdentityStmt)stmt).getLeftOp();
237            Value rightOp = ((IdentityStmt)stmt).getRightOp();
238            
239            if (rightOp instanceof ThisRef) {
240                o.g.assignThisToLocal(left);
241            }
242
243            else if (rightOp instanceof ParameterRef) {
244                ParameterRef p = (ParameterRef)rightOp;
245                // ignore primitive types
246                if (p.getType() instanceof RefLikeType)
247                    o.g.assignParamToLocal(p.getIndex(),left);
248            }
249
250            else if (rightOp instanceof CaughtExceptionRef) {
251                // local =  exception
252                o.g.localIsUnknown(left);
253            }
254
255            else throw new Error("IdentityStmt match failure (rightOp) "+stmt);
256
257        }
258
259
260        ////////////
261        // ThrowStmt
262        ////////////
263
264        else if (stmt instanceof ThrowStmt) {
265            Value op = ((ThrowStmt)stmt).getOp();
266
267            if (op instanceof Local) {
268                Local v = (Local)op;
269                o.g.localEscapes(v);
270            }
271
272            else if (op instanceof Constant) {
273                // do nothing...
274            }
275
276            else throw new Error("ThrowStmt match failure "+stmt);
277        }
278
279
280        /////////////
281        // ReturnStmt
282        /////////////
283
284        else if (stmt instanceof ReturnVoidStmt) {
285            // do nothing...
286        }
287
288        else if (stmt instanceof ReturnStmt) {
289            Value v = ((ReturnStmt)stmt).getOp();
290
291            if (v instanceof Local) {
292                // ignore primitive types
293                if (v.getType() instanceof RefLikeType)
294                    o.g.returnLocal((Local)v);
295            }
296
297            else if (v instanceof Constant) {
298                // do nothing...
299            }
300
301            else throw new Error("ReturnStmt match failure "+stmt);
302
303        }
304
305
306        //////////
307        // ignored
308        //////////
309        
310        else if (stmt instanceof IfStmt ||
311                 stmt instanceof GotoStmt ||
312                 stmt instanceof LookupSwitchStmt ||
313                 stmt instanceof TableSwitchStmt ||
314                 stmt instanceof MonitorStmt ||
315                 stmt instanceof BreakpointStmt ||
316                 stmt instanceof NopStmt) {
317            // do nothing...
318        }
319                 
320
321        else throw new Error("Stmt match faliure "+stmt);
322
323        //o.g.updateStat();
324    }

327    /**
328     * Draw the result of the intra-procedural analysis as one big dot file,
329     * named className.methodName.dot, containing one purity graph for each
330     * statement in the method.
331     */
332    public void drawAsOneDot(String prefix, String name)
333    {
334        DotGraph dot = new DotGraph(name);
335        dot.setGraphLabel(name);
336        dot.setGraphAttribute("compound","true");
337        dot.setGraphAttribute("rankdir","LR");
338        Map<Unit, Integer> node = new HashMap<Unit, Integer>();
339        int id = 0;
340        Iterator it = graph.iterator();
341        while (it.hasNext()) {
342            Unit stmt = (Unit)it.next();
343            PurityGraphBox ref = (PurityGraphBox) getFlowAfter(stmt);
344            DotGraph       sub = dot.createSubGraph("cluster"+id);
345            DotGraphNode label = sub.drawNode("head"+id);
346            String lbl = stmt.toString();
347            if (lbl.startsWith("lookupswitch")) lbl = "lookupswitch...";
348            if (lbl.startsWith("tableswitch"))  lbl = "tableswitch...";
349            sub.setGraphLabel(" ");
350            label.setLabel(lbl);
351            label.setAttribute("fontsize","18");
352            label.setShape("box");
353            ref.g.fillDotGraph("X"+id,sub);
354            node.put(stmt,new Integer(id));
355            id++;
356        }
357        it = graph.iterator();
358        while (it.hasNext()) {
359            Object src = it.next();
360            Iterator itt = graph.getSuccsOf(src).iterator();
361            while (itt.hasNext()) {
362                Object dst = itt.next();
363                DotGraphEdge edge =
364                    dot.drawEdge("head"+node.get(src),"head"+node.get(dst));
365                edge.setAttribute("ltail", "cluster"+node.get(src));
366                edge.setAttribute("lhead", "cluster"+node.get(dst));
367            }
368        }
369
370        File f = new File (SourceLocator.v().getOutputDir(),
371                           prefix+name+DotGraph.DOT_EXTENSION);
372        dot.plot(f.getPath());
373    }

376    /**
377     * Put into dst the purity graph obtained by merging all purity graphs at
378     * the method return.
379     * It is a valid summary that can be used in methodCall if you do
380     * interprocedural analysis.
381     * 
382     */
383    public void copyResult(Object dst)
384    {
385        PurityGraph r = new PurityGraph();
386        Iterator it = graph.getTails().iterator();
387        while (it.hasNext()) {
388            Stmt stmt = (Stmt)it.next();
389            PurityGraphBox ref = (PurityGraphBox) getFlowAfter(stmt);
390            r.union(ref.g);
391        }
392        r.removeLocals();
393        //r.simplifyLoad();
394        //r.simplifyInside();
395        //r.updateStat();
396        ((PurityGraphBox)dst).g = r;
397    }

399    /**
400     * Perform purity analysis on the Jimple unit graph g, as part of
401     * a larger interprocedural analysis.
402     * Once constructed, you may call copyResult and drawAsOneDot to query
403     * the analysis result.
404     */
405    PurityIntraproceduralAnalysis(UnitGraph g,
406                                  AbstractInterproceduralAnalysis inter)
407    {
408        super(g);
409        this.inter = inter;
410        doAnalysis();
411    }
