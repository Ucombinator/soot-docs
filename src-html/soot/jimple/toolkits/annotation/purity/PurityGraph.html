035/**
036 * Purity graphs are mutable structures that are updated in-place.
037 * You can safely hash graphs. Equality comparison means isomorphism
038 * (equal nodes, equal edges).
039 */
040
041/**
042 * Modifications with respect to the article:
043 *
044 * - "unanalizable call" are treated by first constructing a conservative
045 * calee graph where all parameters escape globally and return points to
046 * the global node, and then applying the standard analysable call construction
047 *
048 * - unanalysable calls add a mutation on the global node; the "field" is named
049 * "outside-world" and models the mutation of any static field, but also
050 * side-effects by native methods, such as I/O, that make methods impure
051 * (see below).
052 *
053 * - Whenever a method mutates the global node, it is marked as "impure"
054 * (this can be due to a side-effect or static field mutation), even if the
055 * global node is not rechable from parameter nodes through outside edges.
056 * It seems to me it was a defect from the article ?
057 * TODO: see if we must take the global node  into account also when stating
058 * whether a parameter is read-only or safe.
059 *
060 * - "simplifyXXX" functions are experimental... they may be unsound, and
061 * thus, not used now.
062 *
063 *
064 *
065 *
066 * NOTE:
067 * A lot of precision degradation comes from sequences of the form
068 *   this.field = y; z = this.field
069 * in initialisers: the second statement creates a load node because, as a
070 * parameter, this may have escaped and this.field may be externally modified
071 * in-between the two instructions. I am not sure this can actually happend
072 * in an initialiser... in a a function called directly and only by
073 * initialisers.
074 *
075 * For the moment, summary of unanalised methods are either pure, completely
076 * impure (modify args & side-effects) or partially impure (modify args but
077 * not the gloal node). We should really be able to specify more precisely
078 * which arguments are r/o or safe within this methods.
079 * E.g., the analysis java.lang.String: void getChars(int,int,char [],int)
080 * imprecisely finds that this is not safe (because of the internal call to
081 * System.arraycopy that, in general, may introduce aliases) => it pollutes 
082 * many things (e.g., StringBuffer append(String), and thus, exception 
083 * constructors, etc.)
084 *
085 */
086public class PurityGraph
087{
088    public static final boolean doCheck = false;

149    public boolean equals(Object o)
150    {
151        if (!(o instanceof PurityGraph)) return false;
152        PurityGraph g = (PurityGraph)o;
153        return nodes.equals(g.nodes)
154            //&& paramNodes.equals(g.paramNodes)  // redundant info
155            && edges.equals(g.edges)
156            && locals.equals(g.locals) 
157            && ret.equals(g.ret)
158            && globEscape.equals(g.globEscape)
159            //&& backEdges.equals(g.backEdges)    // redundant info
160            //&& backLocals.equals(g.backLocals)  // redundant info
161            && mutated.equals(g.mutated)
162            ;
163    }

165    /** 
166     * Caching: this semm to actually improve both speed and memory 
167     * consumption!
168     */
169    private static final Map<PurityNode, PurityNode> nodeCache =  new HashMap<PurityNode, PurityNode>();
170    private static final Map<PurityEdge, PurityEdge> edgeCache =  new HashMap<PurityEdge, PurityEdge>();
171    private static PurityNode cacheNode(PurityNode p)
172    {
173        if (!nodeCache.containsKey(p)) nodeCache.put(p,p);
174        return nodeCache.get(p);
175    }
176    private static PurityEdge cacheEdge(PurityEdge e)
177    {
178        if (!edgeCache.containsKey(e)) edgeCache.put(e,e);
179        return edgeCache.get(e);
180    }

182    /**
183     * Conservative constructor for unanalysable calls.
184     *
185     * <p>Note: this gives a valid summary for all native methods, including
186     * Thread.start().
187     *
188     * @param withEffect add a mutated abstract field for the global node to
189     * account for side-effects in the environment (I/O, etc.).
190     */
191    public static PurityGraph conservativeGraph(SootMethod m,
192                                                boolean withEffect)
193    {
194        PurityGraph g = new PurityGraph();
195        PurityNode glob = PurityGlobalNode.node;
196        g.nodes.add(glob);
197
198        // parameters & this escape globally
199        Iterator it = m.getParameterTypes().iterator();
200        int i = 0;
201        while (it.hasNext()) {
202            if (it.next() instanceof RefLikeType) {
203                PurityNode n = cacheNode(new PurityParamNode(i));
204                g.globEscape.add(n);
205                g.nodes.add(n);
206                g.paramNodes.add(n);
207            }
208            i++;
209        }
210
211        // return value escapes globally
212        if (m.getReturnType() instanceof RefLikeType) g.ret.add(glob);
213
214        // add a side-effect on the environment
215        // added by [AM]
216        if (withEffect) g.mutated.put(glob,"outside-world");
217
218        if (doCheck) g.sanityCheck();
219        return g;
220    }

223    /**
224     * Special constructor for "pure" methods returning a fresh object.
225     * (or simply pure if returns void or primitive).
226     */
227    public static PurityGraph freshGraph(SootMethod m)
228    {
229        PurityGraph g = new PurityGraph();
230        if (m.getReturnType() instanceof RefLikeType) {
231            PurityNode n = cacheNode(new PurityMethodNode(m));
232            g.ret.add(n);
233            g.nodes.add(n);
234        }
235        if (doCheck) g.sanityCheck();
236        return g;
237    }

259    /**
260     * Sanity check. Used internally for debugging!
261     */
262    protected void sanityCheck()
263    {
264        boolean err = false;
265        Iterator it = edges.keySet().iterator();
266        while (it.hasNext()) {
267            PurityNode src = (PurityNode)it.next();
268            Iterator itt = edges.get(src).iterator();
269            while (itt.hasNext()) {
270                PurityEdge e = (PurityEdge)itt.next();
271                if (!src.equals(e.getSource()))
272                    {G.v().out.println("invalid edge source "+e+", should be "+src);err=true;}
273                if (!nodes.contains(e.getSource()))
274                    {G.v().out.println("nodes does not contain edge source "+e);err=true;}
275                if (!nodes.contains(e.getTarget()))
276                    {G.v().out.println("nodes does not contain edge target "+e);err=true;}
277                if (!backEdges.get(e.getTarget()).contains(e))
278                    {G.v().out.println("backEdges does not contain edge "+e);err=true;}
279                if (!e.isInside() && !e.getTarget().isLoad())
280                    {G.v().out.println("target of outside edge is not a load node "+e);err=true;}
281            }
282        }
283        it = backEdges.keySet().iterator();
284        while (it.hasNext()) {
285            PurityNode dst = (PurityNode)it.next();
286            Iterator itt = backEdges.get(dst).iterator();
287            while (itt.hasNext()) {
288                PurityEdge e = (PurityEdge)itt.next();
289                if (!dst.equals(e.getTarget()))
290                    {G.v().out.println("invalid backEdge dest "+e+", should be "+dst);err=true;}
291                if (!edges.get(e.getSource()).contains(e))
292                    {G.v().out.println("backEdge not in edges "+e);err=true;}
293            }
294        }
295        it = nodes.iterator();
296        while (it.hasNext()) {
297            PurityNode n = (PurityNode)it.next();
298            if (n.isParam() && !paramNodes.contains(n))
299                {G.v().out.println("paramNode not in paramNodes "+n);err=true;}
300        }
301        it = paramNodes.iterator();
302        while (it.hasNext()) {
303            PurityNode n = (PurityNode)it.next();
304            if (!n.isParam())
305                {G.v().out.println("paramNode contains a non-param node "+n);err=true;}
306            if (!nodes.contains(n))
307                {G.v().out.println("paramNode not in nodes "+n);err=true;}
308        }
309        it = globEscape.iterator();
310        while (it.hasNext()) {
311            PurityNode n = (PurityNode)it.next();
312            if (!nodes.contains(n))
313                {G.v().out.println("globEscape not in nodes "+n);err=true;}
314        }
315        it = locals.keySet().iterator();
316        while (it.hasNext()) {
317            Local l = (Local)it.next();
318            Iterator itt = locals.get(l).iterator();
319            while (itt.hasNext()) {
320                PurityNode n = (PurityNode)itt.next();
321                if (!nodes.contains(n))
322                    {G.v().out.println("target of local node in nodes "+l+" / "+n);err=true;}
323                if (!backLocals.get(n).contains(l))
324                    {G.v().out.println("backLocals does contain local "+l+" / "+n);err=true;}
325            }
326        }
327        it = backLocals.keySet().iterator();
328        while (it.hasNext()) {
329            PurityNode n = (PurityNode)it.next();
330            Iterator itt = backLocals.get(n).iterator();
331            while (itt.hasNext()) {
332                Local l = (Local)itt.next();
333                if (!nodes.contains(n))
334                    {G.v().out.println("backLocal node not in in nodes "+l+" / "+n);err=true;}
335                if (!locals.get(l).contains(n))
336                    {G.v().out.println("locals does contain backLocal "+l+" / "+n);err=true;}
337            }
338        }
339        it = ret.iterator();
340        while (it.hasNext()) {
341            PurityNode n = (PurityNode)it.next();
342            if (!nodes.contains(n))
343                {G.v().out.println("target of ret not in nodes "+n);err=true;}
344        }
345        it = mutated.keySet().iterator();
346        while (it.hasNext()) {
347            PurityNode n = (PurityNode)it.next();
348            if (!nodes.contains(n))
349                {G.v().out.println("mutated node not in nodes "+n);err=true;}
350        }
351        if (err) {
352            dump();
353            DotGraph dot = new DotGraph("sanityCheckFailure");
354            fillDotGraph("chk",dot);
355            dot.plot("sanityCheckFailure.dot");
356            throw new Error("PurityGraph sanity check failed!!!");
357        }
358    }

364    protected void internalPassEdges(Set toColor, Set<PurityNode> dest, 
365                                     boolean consider_inside)
366    {
367        Iterator it = toColor.iterator();
368        while (it.hasNext()) {
369            PurityEdge edge = (PurityEdge) it.next();
370            if (consider_inside || !edge.isInside()) {
371                PurityNode node = edge.getTarget();
372                if (!dest.contains(node)) {
373                    dest.add(node);
374                    internalPassEdges(edges.get(node),dest,consider_inside);
375                }
376            }
377        }       
378    }
379
380    protected void internalPassNode(PurityNode node, Set<PurityNode> dest,
381                                    boolean consider_inside)
382    {
383        if (!dest.contains(node)) {
384            dest.add(node);
385            internalPassEdges(edges.get(node),dest,consider_inside);
386        }       
387    }
388
389    protected void internalPassNodes(Set toColor, Set<PurityNode> dest, 
390                                     boolean consider_inside)
391    {
392        Iterator it = toColor.iterator();
393        while (it.hasNext()) 
394            internalPassNode((PurityNode)it.next(),
395                             dest, consider_inside);
396    }

398    protected Set<PurityNode> getEscaping()
399    {
400        Set<PurityNode> escaping = new HashSet<PurityNode>();
401        internalPassNodes(ret,escaping,true);
402        internalPassNodes(globEscape,escaping,true);
403        internalPassNode(PurityGlobalNode.node,escaping,true);
404        internalPassNodes(paramNodes,escaping,true);
405        return escaping;
406    }

409    /**
410     * Call this on the merge of graphs at all return points of a method to
411     * know whether the method is pure.
412     */
413    public boolean isPure()
414    {
415        if (!mutated.get(PurityGlobalNode.node).isEmpty()) return false;
416        Set<PurityNode> A = new HashSet<PurityNode>();
417        Set<PurityNode> B = new HashSet<PurityNode>();
418        internalPassNodes(paramNodes, A, false);
419        internalPassNodes(globEscape, B, true);
420        internalPassNode(PurityGlobalNode.node,B,true);
421        Iterator<PurityNode> it = A.iterator();
422        while (it.hasNext()) {
423            PurityNode n = it.next();
424            if (B.contains(n) || !mutated.get(n).isEmpty()) return false;
425        }
426        return true;
427    }

429   /**
430    * We use a less restrictive notion of purity for constructors: pure 
431    * constructors can mutate fields of this. 
432    *
433    * @see isPure
434    */
435    public boolean isPureConstructor()
436    {
437        if (!mutated.get(PurityGlobalNode.node).isEmpty()) return false;
438        Set<PurityNode> A = new HashSet<PurityNode>();
439        Set<PurityNode> B = new HashSet<PurityNode>();
440        internalPassNodes(paramNodes, A, false);
441        internalPassNodes(globEscape, B, true);
442        internalPassNode(PurityGlobalNode.node,B,true);
443        PurityNode th = PurityThisNode.node;
444        Iterator<PurityNode> it = A.iterator();
445        while (it.hasNext()) {
446            PurityNode n = it.next();
447            if (B.contains(n) || 
448                (!n.equals(th) && !mutated.get(n).isEmpty())) return false;
449        }
450        return true;
451    }

453    /**
454     * A parameter (or this) can be:
455     * - read and write
456     * - read only
457     * - safe (read only & no externally visible alias is created)
458     */
459    static final int PARAM_RW   = 0;
460    static final int PARAM_RO   = 1;
461    static final int PARAM_SAFE = 2;
462
463    protected int internalParamStatus(PurityNode p)
464    {
465        if (!paramNodes.contains(p)) return PARAM_RW;
466
467        Set<PurityNode> S1 = new HashSet<PurityNode>();
468        internalPassNode(p, S1, false);
469        Iterator<PurityNode> it = S1.iterator();
470        while (it.hasNext()) {
471            PurityNode n = it.next();
472            if (n.isLoad() || n.equals(p)) {
473                if (!mutated.get(n).isEmpty() ||
474                    globEscape.contains(n)) return PARAM_RW;
475            }
476        }
477
478        Set<PurityNode> S2 = new HashSet<PurityNode>();
479        internalPassNodes(ret,S2,true);
480        internalPassNodes(paramNodes,S2,true);
481        it = S2.iterator();
482        while (it.hasNext()) {
483            Iterator itt = edges.get(it.next()).iterator();
484            while (itt.hasNext()) {
485                PurityEdge e = (PurityEdge)itt.next();
486                if (e.isInside() && S1.contains(e.getTarget()))
487                    return PARAM_RO;
488            }
489        }
490
491        return PARAM_SAFE;
492    }

497     * (PARAM_RO), read write (PARAM_RW), or safe (PARAM_SAFE).
498     * Returns PARAM_RW for primitive-type parameters.

546    /** Utility function to remove a node & all adjacent edges */
547    protected final void removeNode(PurityNode n)
548    {
549        Iterator it = edges.get(n).iterator();
550        while (it.hasNext()) {
551            PurityEdge e = (PurityEdge)it.next();
552            backEdges.remove(e.getTarget(),e);
553        }
554        it = backEdges.get(n).iterator();
555        while (it.hasNext()) {
556            PurityEdge e = (PurityEdge)it.next();
557            edges.remove(e.getSource(),e);
558        }
559        it = backLocals.get(n).iterator();
560        while (it.hasNext()) {
561            Local l = (Local)it.next();
562            locals.remove(l,n);
563        }
564        ret.remove(n);
565        edges.remove(n);
566        backEdges.remove(n);
567        backLocals.remove(n);
568        nodes.remove(n);
569        paramNodes.remove(n);
570        globEscape.remove(n);
571        mutated.remove(n);
572    }

575    /** Utility function to merge node src into dst; src is removed */
576    protected final void mergeNodes(PurityNode src, PurityNode dst)
577    {
578        Iterator it = (new LinkedList(edges.get(src))).iterator();
579        while (it.hasNext()) {
580            PurityEdge e = (PurityEdge)it.next();
581            PurityNode n = e.getTarget();
582            if (n.equals(src)) n = dst;     
583            PurityEdge ee = 
584                cacheEdge(new PurityEdge(dst, e.getField(), n, e.isInside()));
585            edges.remove(src, e);
586            edges.put(dst, ee);
587            backEdges.remove(n, e);
588            backEdges.put(n, ee);
589        }
590        it = (new LinkedList(backEdges.get(src))).iterator();
591        while (it.hasNext()) {
592            PurityEdge e = (PurityEdge)it.next();
593            PurityNode n = e.getSource();
594            if (n.equals(src)) n = dst;
595            PurityEdge ee = 
596                cacheEdge(new PurityEdge(n, e.getField(), dst, e.isInside()));
597            edges.remove(n, e);
598            edges.put(n, ee);
599            backEdges.remove(src, e);
600            backEdges.put(dst, ee);
601        }
602        it = (new LinkedList(backLocals.get(src))).iterator();
603        while (it.hasNext()) {
604            Local l = (Local)it.next();
605            locals.remove(l, src);
606            backLocals.remove(src, l);
607            locals.put(l,dst);
608            backLocals.put(dst, l);
609        }
610        {
611            Set m = mutated.get(src);
612            mutated.remove(src);
613            mutated.putAll(dst,m);
614        }
615        if (ret.contains(src)) {
616            ret.remove(src);
617            ret.add(dst);
618        }
619        if (globEscape.contains(src)) {
620            globEscape.remove(src);
621            globEscape.add(dst);
622        }
623        nodes.remove(src);
624        nodes.add(dst);
625        paramNodes.remove(src);
626        if (dst.isParam()) paramNodes.add(dst);
627    }

629    /** Experimental simplification: merge redundant load nodes. */
630    void simplifyLoad()
631    {
632        Iterator it = (new LinkedList(nodes)).iterator();
633        while (it.hasNext()) {
634            PurityNode p = (PurityNode)it.next();
635            Map<String, PurityNode> fmap = new HashMap<String, PurityNode>();
636            Iterator itt = (new LinkedList(edges.get(p))).iterator();
637            while (itt.hasNext()) {
638                PurityEdge e   = (PurityEdge)itt.next();
639                PurityNode tgt = e.getTarget();
640                if (!e.isInside() && !tgt.equals(p)) {
641                    String f = e.getField();
642                    if (fmap.containsKey(f) && nodes.contains(fmap.get(f))) 
643                        mergeNodes(tgt, fmap.get(f));
644                    else fmap.put(f,tgt);
645                }
646            }
647        }
648        if (doCheck) sanityCheck();
649    }

651    /** Experimental simplification: remove inside nodes not reachable
652        from escaping nodes (params, ret, globEscape) or load nodes. */
653    void simplifyInside()
654    {
655        Set<PurityNode> r = new HashSet<PurityNode>();
656        internalPassNodes(paramNodes,r,true);
657        internalPassNodes(ret,r,true);
658        internalPassNodes(globEscape,r,true);
659        internalPassNode(PurityGlobalNode.node,r,true);
660        Iterator it = nodes.iterator();
661        while (it.hasNext()) {
662            PurityNode n = (PurityNode) it.next();
663            if (n.isLoad()) internalPassNode(n,r,true);
664        } 
665        it = (new LinkedList(nodes)).iterator();
666        while (it.hasNext()) {
667            PurityNode n = (PurityNode) it.next();
668            if (n.isInside() && !r.contains(n)) removeNode(n);
669        }

729    /** 
730     * Load non-static: left = right.field, or left = right[?] if field is [].
731     */
732    void assignFieldToLocal(Stmt stmt, Local right, String field, Local left)
733    {
734        Set<PurityNode> esc = new HashSet<PurityNode>();
735        Set<PurityNode> escaping = getEscaping();
736
737        // strong update on local
738        localsRemove(left);
739        Iterator itRight = locals.get(right).iterator();
740        while (itRight.hasNext()) {
741            PurityNode nodeRight = (PurityNode) itRight.next();
742
743            Iterator itEdges = edges.get(nodeRight).iterator();
744            while (itEdges.hasNext()) {
745                PurityEdge edge = (PurityEdge) itEdges.next();
746                if (edge.isInside() && edge.getField().equals(field))
747                    localsPut(left, edge.getTarget());
748            }
749
750            if (escaping.contains(nodeRight)) esc.add(nodeRight);
751        }
752    
753        if (!esc.isEmpty()) {
754            // right can escape
755            
756            // we add a label load node & outside edges
757            PurityNode loadNode = cacheNode(new PurityStmtNode(stmt,false));
758            nodes.add(loadNode);
759            
760            Iterator<PurityNode> itEsc = esc.iterator();
761            while (itEsc.hasNext()) {
762                PurityNode node = itEsc.next();
763                PurityEdge edge = 
764                    cacheEdge(new PurityEdge(node, field, loadNode, false));
765                if (edges.put(node, edge))
766                    backEdges.put(loadNode, edge);
767            }
768            localsPut(left, loadNode);
769        }
770        if (doCheck) sanityCheck();
771    }

786                    cacheEdge(new PurityEdge(nodeLeft, field, nodeRight, true));

788                    backEdges.put(nodeRight, edge);
789            }
790            if (!nodeLeft.isInside())
791                mutated.put(nodeLeft, field);
792        }
793        if (doCheck) sanityCheck();

864    /** 
865     * Method call left = right.method(args).
866     *
867     * @param g is method's summary PurityGraph
868     * @param left can be null (no return value)
869     * @param right can be null (static call)
870     * @param args is a list of Value
871     */
872    void methodCall(PurityGraph g, Local right, List args, Local left)
873    {
874        MultiMap mu = new HashMultiMap();
875
876        // compute mapping relation g -> this
877        /////////////////////////////////////
878
879        Iterator it = args.iterator(); // (1) rule
880        int nb = 0;
881        while (it.hasNext()) {
882            Value arg = (Value)it.next();
883            if (arg instanceof Local && 
884                ((Local)arg).getType() instanceof RefLikeType) {
885                mu.putAll(cacheNode(new PurityParamNode(nb)),locals.get(arg));
886            }
887            nb++;
888        }
889        if (right!=null) // (1) rule for "this" argument
890            mu.putAll(PurityThisNode.node,locals.get(right));
891
892        // COULD BE OPTIMIZED!
893        // many times, we need to copy sets cause we mutate them within iterators
894        boolean hasChanged = true;
895        while (hasChanged) { // (2) & (3) rules fixpoint
896            hasChanged = false;
897
898            // (2)
899            it = (new LinkedList(mu.keySet())).iterator();
900            while (it.hasNext()) {
901                PurityNode n1 = (PurityNode)it.next();
902                Iterator it3 = (new LinkedList(mu.get(n1))).iterator();
903                while (it3.hasNext()) {
904                    PurityNode n3 = (PurityNode)it3.next();
905                    Iterator it12 = g.edges.get(n1).iterator();
906                    while (it12.hasNext()) {
907                        PurityEdge e12 = (PurityEdge)it12.next();
908                        if (!e12.isInside()) {
909                            Iterator it34 = edges.get(n3).iterator();
910                            while (it34.hasNext()) {
911                                PurityEdge e34 = (PurityEdge)it34.next();
912                                if (e34.isInside() &&
913                                    e12.getField().equals(e34.getField()))
914                                    if (mu.put(e12.getTarget(),e34.getTarget()))
915                                        hasChanged = true;
916                            }
917                        }
918                    }
919                }
920            }
921            
922            // (3)
923            it = g.edges.keySet().iterator();
924            while (it.hasNext()) {
925                PurityNode n1 = (PurityNode)it.next();
926                Iterator it3 = g.edges.keySet().iterator();
927                while (it3.hasNext()) {
928                    PurityNode n3 = (PurityNode)it3.next();
929
930                    // ((mu(n1) U {n1}) inter (mu(n3) U {n3})) not empty
931                    Set mu1 = new HashSet(mu.get(n1));
932                    Set mu3 = new HashSet(mu.get(n3));
933                    boolean cond = n1.equals(n3) || 
934                        mu1.contains(n3) || mu3.contains(n1);
935                    Iterator itt = mu1.iterator();
936                    while (!cond && itt.hasNext()) {
937                        cond = cond || mu3.contains(itt.next());
938                    }
939                    
940                    // add (mu(n4) U ({n4} inter PNodes)) to mu(n2)
941                    if (cond && (!n1.equals(n3) || n1.isLoad())) {
942                        Iterator it12 = g.edges.get(n1).iterator();
943                        while (it12.hasNext()) {
944                            PurityEdge e12 = (PurityEdge)it12.next();
945                            if (!e12.isInside()) {
946                                Iterator it34 = g.edges.get(n3).iterator();
947                                while (it34.hasNext()) {
948                                    PurityEdge e34 = (PurityEdge)it34.next();
949                                    if (e34.isInside()) {
950                                        if (e12.getField().equals(e34.getField())) {
951                                            PurityNode n2 = e12.getTarget();
952                                            PurityNode n4 = e34.getTarget();
953                                            
954                                            // add n4 (if not param node) to mu(n2)
955                                            if (!n4.isParam() && mu.put(n2,n4))
956                                                hasChanged = true;
957                                            
958                                            // add mu(n4) to mu(n2)
959                                            if (mu.putAll(n2,mu.get(n4)))
960                                                hasChanged = true;
961                                        }
962                                    }
963                                }
964                            }
965                        }
966                    }
967                }
968            }
969        }
970        
971        // extend mu into mu'
972        it = g.nodes.iterator();
973        while (it.hasNext()) {
974            PurityNode n = (PurityNode)it.next();
975            if (!n.isParam()) { 
976                mu.put(n,n);
977                nodes.add(n);
978            }
979        }
980
981
982        // combine g into this
983        //////////////////////
984
985        // project edges
986        it = g.edges.keySet().iterator();
987        while (it.hasNext()) {
988            PurityNode n1 = (PurityNode)it.next();
989            Iterator it12 = g.edges.get(n1).iterator();
990            while (it12.hasNext()) {
991                PurityEdge e12 = (PurityEdge)it12.next();
992                String     f   = e12.getField();
993                PurityNode n2  = e12.getTarget();
994                Iterator  itm1 = mu.get(n1).iterator();
995                while (itm1.hasNext()) {
996                    PurityNode mu1 = (PurityNode)itm1.next();
997
998                    if (e12.isInside()) {
999                        Iterator itm2 = mu.get(n2).iterator();
1000                        while (itm2.hasNext()) {
1001                            PurityNode mu2  = (PurityNode)itm2.next();
1002                            PurityEdge edge = 
1003                                cacheEdge(new PurityEdge(mu1,f,mu2,true));
1004                            edges.put(mu1,edge);
1005                            backEdges.put(mu2,edge);
1006                        }
1007                    }
1008                    else {
1009                        PurityEdge edge = 
1010                            cacheEdge(new PurityEdge(mu1,f,n2,false));
1011                        edges.put(mu1,edge);
1012                        backEdges.put(n2,edge);
1013                    }
1014                }
1015            }
1016        }
1017
1018        // return value
1019        if (left!=null) {
1020            // strong update on locals
1021            localsRemove(left);
1022            it = g.ret.iterator();
1023            while (it.hasNext())
1024                localsPutAll(left, mu.get(it.next()));
1025        }
1026        
1027        // global escape
1028        it = g.globEscape.iterator();
1029        while (it.hasNext())
1030            globEscape.addAll(mu.get(it.next()));
1031
1032        if (doCheck) sanityCheck();
1033
1034
1035        // simplification
1036        /////////////////       
1037                
1038        Set<PurityNode> escaping = getEscaping();
1039        it = (new LinkedList(nodes)).iterator();
1040        while (it.hasNext()) {
1041            PurityNode n = (PurityNode)it.next();
1042            if (!escaping.contains(n)) 
1043                if (n.isLoad()) 
1044                    // remove captured load nodes
1045                    removeNode(n);
1046                else {
1047                    // ... and outside edges from captured nodes
1048                    Iterator itt = (new LinkedList(edges.get(n))).iterator();
1049                    while (itt.hasNext()) {
1050                        PurityEdge e = (PurityEdge)itt.next();
1051                        if (!e.isInside()) {
1052                            edges.remove(n,e);
1053                            backEdges.remove(e.getTarget(),e);
1054                        }
1055                    }
1056                }  
1057        }
1058
1059        // update mutated
1060        /////////////////
1061
1062        it = g.mutated.keySet().iterator();
1063        while (it.hasNext()) {
1064            PurityNode n = (PurityNode)it.next();
1065            Iterator itt = mu.get(n).iterator();
1066            while (itt.hasNext()) {
1067                PurityNode nn = (PurityNode)itt.next();
1068                if (nodes.contains(nn) && !nn.isInside()) {
1069                    Iterator ittt = g.mutated.get(n).iterator();
1070                    while (ittt.hasNext()) {
1071                        String f = (String)ittt.next();
1072                        mutated.put(nn,f);
1073                    }
1074                }
1075            }
1076        }
1077
1078        if (doCheck) sanityCheck();
1079    }

1086    /** 
1087     * Fills a dot graph or subgraph with the graphical representation
1088     * of the purity graph.
1089     *
1090     * @param prefix is used to prefix all dot node and edge names. Use it
1091     * to avoid collision when several subgraphs are laid in the same dot
1092     * file!
1093     * 
1094     * @param out is a newly created dot graph or subgraph where to put the
1095     * result.
1096     *
1097     * <p>Note: outside edges, param and load nodes are gray dashed, while
1098     * inside edges and nodes are solid black.
1099     * Globally escaping nodes have a red label.
1100     */
1101    void fillDotGraph(String prefix, DotGraph out)
1102    {
1103        Map<PurityNode, String> nodeId = new HashMap<PurityNode, String>();
1104        int id = 0;
1105
1106        // add nodes 
1107        Iterator it = nodes.iterator();
1108        while (it.hasNext()) {
1109            PurityNode n = (PurityNode) it.next();
1110            String label = "N"+prefix+"_"+id;
1111            DotGraphNode node = out.drawNode(label);
1112            node.setLabel(n.toString());
1113            if (!n.isInside()) {
1114                node.setStyle("dashed");
1115                node.setAttribute("color","gray50");
1116            }
1117            if (globEscape.contains(n)) node.setAttribute("fontcolor","red");
1118            nodeId.put(n,label);
1119            id++;
1120        }
1121        
1122        // add edges
1123        it = edges.keySet().iterator();
1124        while (it.hasNext()) {
1125            PurityNode src = (PurityNode) it.next();
1126            Iterator itt = edges.get(src).iterator();
1127            while (itt.hasNext()) {
1128                PurityEdge e = (PurityEdge) itt.next();
1129                DotGraphEdge edge = 
1130                    out.drawEdge(nodeId.get(e.getSource()),
1131                                 nodeId.get(e.getTarget()));
1132                edge.setLabel(e.getField());
1133                if (!e.isInside()) {
1134                    edge.setStyle("dashed");
1135                    edge.setAttribute("color","gray50");
1136                    edge.setAttribute("fontcolor","gray40");
1137                }
1138            }
1139        }
1140
1141        // add locals
1142        it = locals.keySet().iterator();
1143        while (it.hasNext()) {
1144            Local local = (Local) it.next();
1145            if (!locals.get(local).isEmpty()) {
1146                String label = "L"+prefix+"_"+id;
1147                DotGraphNode node = out.drawNode(label);
1148                node.setLabel(local.toString());
1149                node.setShape("plaintext");         
1150                Iterator itt = locals.get(local).iterator();
1151                while (itt.hasNext()) {
1152                    PurityNode dst = (PurityNode) itt.next();
1153                    out.drawEdge(label,nodeId.get(dst));
1154                }
1155                id++;
1156            }
1157        }
1158
1159        // ret
1160        if (!ret.isEmpty()) {
1161            DotGraphNode node = out.drawNode("ret_"+prefix);
1162            node.setLabel("ret");
1163            node.setShape("plaintext");     
1164            Iterator itt = ret.iterator();
1165            while (itt.hasNext()) {
1166                PurityNode dst = (PurityNode) itt.next();
1167                out.drawEdge("ret_"+prefix,nodeId.get(dst));
1168            }       
1169        }
1170
1171        // add mutated
1172        it = mutated.keySet().iterator();
1173        while (it.hasNext()) {
1174            PurityNode n = (PurityNode)it.next();
1175            Iterator itt = mutated.get(n).iterator();
1176            while (itt.hasNext()) {
1177                String f = (String)itt.next();
1178                String label = "M"+prefix+"_"+id;
1179                DotGraphNode node = out.drawNode(label);
1180                node.setLabel("");
1181                node.setShape("plaintext");
1182                DotGraphEdge edge = out.drawEdge(nodeId.get(n),label);
1183                edge.setLabel(f);
1184                id++;
1185            }
1186        }
1187    }

1226    static private int maxInsideNodes = 0;
1227    static private int maxLoadNodes = 0;
1228    static private int maxInsideEdges = 0;
1229    static private int maxOutsideEdges = 0;

1232    void dumpStat()
1233    {
1234        G.v().out.println("Stat: "+
1235                          maxInsideNodes+" inNodes, "+
1236                          maxLoadNodes+" loadNodes, "+
1237                          maxInsideEdges+" inEdges, "+
1238                          maxOutsideEdges+" outEdges, "+
1239                          maxMutated+" mutated.");
1240    }

1242    void updateStat()
1243    {
1244        Iterator it = nodes.iterator();
1245        int insideNodes = 0;
1246        int loadNodes = 0;
1247        while (it.hasNext()) {
1248            PurityNode n = (PurityNode)it.next();
1249            if (n.isInside()) insideNodes++;
1250            else if (n.isLoad()) loadNodes++;
1251        }
1252        int insideEdges = 0;
1253        int outsideEdges = 0;
1254        it = edges.keySet().iterator();
1255        while (it.hasNext()) {
1256            Iterator itt = edges.get(it.next()).iterator();
1257            while (itt.hasNext()) {
1258                PurityEdge e = (PurityEdge)itt.next();
1259                if (e.isInside()) insideEdges++;
1260                else outsideEdges++;
1261            }
1262        }
1263        int mutatedFields = 0;
1264        it = mutated.keySet().iterator();
1265        while (it.hasNext()) mutatedFields += mutated.get(it.next()).size();
1266
1267        boolean changed = false;
1268        if (insideNodes>maxInsideNodes) 
1269            { maxInsideNodes=insideNodes; changed=true; }
1270        if (loadNodes>maxLoadNodes)
1271            { maxLoadNodes=loadNodes; changed=true; }
1272        if (insideEdges>maxInsideEdges)
1273            { maxInsideEdges=insideEdges; changed=true; }
1274        if ( outsideEdges>maxOutsideEdges)
1275            { maxOutsideEdges=outsideEdges; changed=true; }
1276        if (mutatedFields>maxMutated)
1277            { maxMutated=mutatedFields; changed=true; }
1278        if (changed) dumpStat();
1279    }
