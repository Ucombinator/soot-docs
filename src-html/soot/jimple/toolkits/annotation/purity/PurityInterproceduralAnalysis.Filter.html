208    /** Filter out some method. */
209    static private class Filter implements SootMethodFilter {
210        public boolean want(SootMethod method) { 
211            // could be optimized with HashSet....
212            String c = method.getDeclaringClass().toString();
213            String m = method.getName();
214            String[][] o = PurityInterproceduralAnalysis.pureMethods;
215            for (String[] element : o)
216                        if (m.equals(element[1]) && c.startsWith(element[0])) return false;
217            o = PurityInterproceduralAnalysis.impureMethods;
218            for (String[] element : o)
219                        if (m.equals(element[1]) && c.startsWith(element[0])) return false;
220            o = PurityInterproceduralAnalysis.alterMethods;
221            for (String[] element : o)
222                        if (m.equals(element[1]) && c.startsWith(element[0])) return false;
223            return true;
224        }
225    }

227    /** The constructor does it all! */
228    PurityInterproceduralAnalysis(CallGraph        cg,
229                                  Iterator<SootMethod>         heads,
230                                  PurityOptions    opts)
231    {
232        super(cg,new Filter(),heads, opts.dump_cg());
233        
234        if (opts.dump_cg()) {
235            G.v().out.println("[AM] Dumping empty .dot call-graph");
236            drawAsOneDot("EmptyCallGraph");
237        }
238
239        Date start = new Date();
240        G.v().out.println("[AM] Analysis began");
241        doAnalysis(opts.verbose());
242        G.v().out.println("[AM] Analysis finished");
243        Date finish = new Date();
244        long runtime = finish.getTime() - start.getTime();
245        G.v().out.println("[AM] run time: "+runtime/1000.+" s");
246
247        if (opts.dump_cg()) {
248            G.v().out.println("[AM] Dumping annotated .dot call-graph");
249            drawAsOneDot("CallGraph");
250        }
251
252        if (opts.dump_summaries()) {
253            G.v().out.println("[AM] Dumping .dot summaries of analysed methods");
254            drawAsManyDot("Summary_",false);
255        }
256
257        if (opts.dump_intra()) {
258            G.v().out.println("[AM] Dumping .dot full intra-procedural method analyses");
259            // relaunch the interprocedural analysis once on each method
260            // to get a purity graph at each statement, not only summaries
261            Iterator it = getAnalysedMethods();
262            while (it.hasNext()) {
263                SootMethod method = (SootMethod)it.next();
264                Body body = method.retrieveActiveBody();
265                ExceptionalUnitGraph graph = new ExceptionalUnitGraph(body);
266                if (opts.verbose()) G.v().out.println("  |- "+method);
267                PurityIntraproceduralAnalysis r = 
268                    new PurityIntraproceduralAnalysis(graph, this);
269                r.drawAsOneDot("Intra_",method.toString());
270                PurityGraphBox b = new PurityGraphBox();
271                r.copyResult(b);
272            }
273        }
274
275
276        {
277            G.v().out.println("[AM] Annotate methods. ");
278            Iterator it = getAnalysedMethods();
279            while (it.hasNext()) {
280                SootMethod m = (SootMethod)it.next();
281                PurityGraphBox b = (PurityGraphBox)getSummaryFor(m);
282
283                // purity
284                boolean isPure;
285                if (m.toString().indexOf("<init>")!=-1)
286                    isPure = b.g.isPureConstructor() ;
287                else 
288                    isPure = b.g.isPure();
289                /* m.addTag(new GenericAttribute("isPure",
290                   (new String(isPure?"yes":"no")).getBytes()));
291                */
292                m.addTag(new StringTag("purity: "+(isPure?"pure":"impure")));
293                if(isPure && opts.annotate())
294                        m.addTag(new GenericAttribute("Pure", new byte[0]));
295                if (opts.print()) 
296                    G.v().out.println("  |- method "+m.toString()+" is "+(isPure?"pure":"impure"));
297
298                // param & this ro / safety
299                if (!m.isStatic()) {
300                    int status = b.g.thisStatus();
301                    String s;
302                    switch (status) {
303                    case PurityGraph.PARAM_RW: s = "read/write";break;
304                    case PurityGraph.PARAM_RO: s = "read-only";break;
305                    case PurityGraph.PARAM_SAFE: s = "Safe";break;
306                    default: s = "unknown";
307                    }
308                    /*  m.addTag(new GenericAttribute("thisStatus",s.getBytes()));
309                     */
310                    m.addTag(new StringTag("this: "+s));
311                    if (opts.print()) G.v().out.println("  |   |- this is "+s);
312                }
313                
314                Iterator itt = m.getParameterTypes().iterator();
315                int i = 0;
316                while (itt.hasNext()) {
317                    if (itt.next() instanceof RefLikeType) {
318                        int status = b.g.paramStatus(i);
319                        String s;
320                        switch (status) {
321                        case PurityGraph.PARAM_RW: s = "read/write";break;
322                        case PurityGraph.PARAM_RO: s = "read-only";break;
323                        case PurityGraph.PARAM_SAFE: s = "safe";break;
324                        default: s = "unknown";
325                        }
326                        /*
327                          m.addTag(new GenericAttribute("param"+i+"Status",
328                          s.getBytes()));
329                        */
330                        m.addTag(new StringTag("param"+i+": "+s));
331                        if (opts.print())
332                            G.v().out.println("  |   |- param "+i+" is "+s);
333                    }
334                    i++;
335                }
336            }
337        }
338
339    }

344    protected void merge(Object in1, Object in2, Object out)
345    {
346        PurityGraphBox i1  = (PurityGraphBox)in1;
347        PurityGraphBox i2  = (PurityGraphBox)in2;
348        PurityGraphBox o   = (PurityGraphBox)out;
349        if (out!=i1) o.g = new PurityGraph(i1.g);
350        o.g.union(i2.g);
351    }

353    protected void copy(Object source, Object dest)
354    {
355        PurityGraphBox src  = (PurityGraphBox)source;
356        PurityGraphBox dst  = (PurityGraphBox)dest;
357        dst.g = new PurityGraph(src.g);
358    }

360    protected void analyseMethod(SootMethod method,
361                                 Object     dst)
362    {
363        Body body = method.retrieveActiveBody();
364        ExceptionalUnitGraph graph = new ExceptionalUnitGraph(body);
365        PurityIntraproceduralAnalysis r = 
366            new PurityIntraproceduralAnalysis(graph, this);
367        r.copyResult(dst);
368    }

370    /**
371     * @see PurityGraph.conservativeGraph
372     * @see PurityGraph.freshGraph
373     */
374    protected Object summaryOfUnanalysedMethod(SootMethod method)
375    {
376        PurityGraphBox b = new PurityGraphBox();
377        String c = method.getDeclaringClass().toString();
378        String m = method.getName();
379
380        // impure with side-effect, unless otherwise specified
381        b.g = PurityGraph.conservativeGraph(method,true);
382
383        String[][] o = PurityInterproceduralAnalysis.pureMethods;
384        for (String[] element : o)
385                if (m.equals(element[1]) && c.startsWith(element[0]))
386                b.g = PurityGraph.freshGraph(method);
387
388        o = PurityInterproceduralAnalysis.alterMethods;
389        for (String[] element : o)
390                if (m.equals(element[1]) && c.startsWith(element[0]))
391                b.g = PurityGraph.conservativeGraph(method,false);
392
393        return b;
394    }

396    /**
397     * @param stmt any statement containing an InvokeExpr
398     * @see PurityGraph.methodCall
399     */
400    protected void applySummary(Object src,
401                                Stmt   stmt,
402                                Object summary,
403                                Object dst)
404    {
405        // extract call info
406        InvokeExpr e = stmt.getInvokeExpr();
407        Local ret = null;
408        if (stmt instanceof AssignStmt) {
409            Local v = (Local)((AssignStmt)stmt).getLeftOp();
410            if (v.getType() instanceof RefLikeType) ret = v;
411        }
412        Local obj = null;
413        if (!(e instanceof StaticInvokeExpr)) 
414            obj = (Local)((InstanceInvokeExpr)e).getBase();
415        List args = e.getArgs();
416        
417        // call methoCall on the PurityGraph
418        PurityGraphBox s = (PurityGraphBox)src;
419        PurityGraphBox d = (PurityGraphBox)dst;
420        PurityGraph g = new PurityGraph(s.g);
421        g.methodCall(((PurityGraphBox)summary).g, obj, args, ret);
422        d.g = g;
423    }
