021/*
022    RefIntPair class
023
024    Immutable pair of an EquivalentValue representing a reference
025    and an interger (representing a constant).
026    With pretty printing for BranchedRefVarsAnalysis.
027*/
028
029package soot.jimple.toolkits.annotation.nullcheck;
030import soot.*;
031
032/**
033 * @deprecated only used by deprecated type {@link BranchedRefVarsAnalysis}; flagged
034 * for future deletion
035 */
036@Deprecated
037public class RefIntPair
038{
039    private EquivalentValue _ref;
040    private int _val;
041    // constructor is not public so that people go throught the ref pair constants factory on the analysis
042    RefIntPair(EquivalentValue r, int v, BranchedRefVarsAnalysis brva)
043    {
044        this._ref = r;
045        this._val = v;
046    }
047
048    public EquivalentValue ref ()
049    { return this._ref; }
050
051    public int val ()
052    { return this._val; }
053
054    public String toString()
055    {
056        String prefix = "("+_ref+", ";
057        if (_val == BranchedRefVarsAnalysis.kNull)
058            return prefix+"null)";
059        else if (_val == BranchedRefVarsAnalysis.kNonNull)
060            return prefix+"non-null)";
061        else if (_val == BranchedRefVarsAnalysis.kTop)
062            return prefix+"top)";
063        else if (_val == BranchedRefVarsAnalysis.kBottom)
064            return prefix+"bottom)";
065        else
066            return prefix+_val+")";
067    }
068} // end class RefIntPair
