030/*
031     README FIRST - IMPORTANT IMPLEMENTATION NOTE
032
033     As per the analysis presented in the report, there are four possible 
034     pairs for given reference r:
035     (r, kBottom)
036     (r, kNonNull)
037     (r, kNull)
038     (r, kTop)
039
040     To save space an simplify operations, we implemented those 4 values 
041     with two bits rather than four, namely:
042
043     (r, kTop) in a set is represented by having (r, kNull) and (r, kNonNull) 
044     in the set. Ditto, (r, kBottom) is represented by having neither in the set.
045    
046     Keep this in mind as you read the code, it helps. Honnest :)
047
048     -- Janus
049
050 */

163    private final  HashMap<EquivalentValue, RefIntPair> kRefBotttomPairs = new HashMap<EquivalentValue, RefIntPair>(2293, 0.7f);
164    private final  HashMap<EquivalentValue, RefIntPair> kRefNonNullPairs = new HashMap<EquivalentValue, RefIntPair>(2293, 0.7f);
165    private final  HashMap<EquivalentValue, RefIntPair> kRefNullPairs = new HashMap<EquivalentValue, RefIntPair>(2293, 0.7f);
166    private final  HashMap<EquivalentValue, RefIntPair> kRefTopPairs = new HashMap<EquivalentValue, RefIntPair>(2293, 0.7f);

168    // make that (r, v) pairs are constants
169    // i.e. the same r and v values always generate the same (r, v) object    
170    public  RefIntPair getKRefIntPair(EquivalentValue r, int v)
171    {
172        HashMap<EquivalentValue, RefIntPair> pairsMap = null;
173
174        if (v == kNonNull)
175            pairsMap =  kRefNonNullPairs;
176        else if (v == kNull)
177            pairsMap = kRefNullPairs;
178        else if (v == kTop)
179            pairsMap = kRefTopPairs;
180        else if (v == kBottom)
181            pairsMap = kRefBotttomPairs;
182        else
183            throw new RuntimeException("invalid constant ("+v+")");
184        
185        if (pairsMap.containsKey(r))
186            return pairsMap.get(r);
187        else {
188            RefIntPair pair = new RefIntPair(r, v, this);
189            pairsMap.put(r, pair);
190            return pair;
191        }
192    } // end getKRefIntPair

203    // isAlwaysNull returns true if the reference r is known to be always null
204    private  final boolean isAlwaysNull(Value r)
205    {
206        return ((r instanceof NullConstant) || 
207                (r.getType() instanceof NullType));
208    } // end isAlwaysNull

211    // isAlwaysTop returns true if the reference r is known to be always top for this analysis
212    // i.e. its value is undecidable by this analysis
213    private  final boolean isAlwaysTop(Value r)
214    {
215        if (isNotConservative)
216            return false;
217        else
218            return r instanceof InstanceFieldRef || r instanceof StaticFieldRef;
219    } // end isAlwaysTop
220
221    protected boolean isAlwaysNonNull(Value ro) {
222        if( ro instanceof NewExpr ) return true;
223        if( ro instanceof NewArrayExpr ) return true;
224        if( ro instanceof NewMultiArrayExpr ) return true;
225        if( ro instanceof ThisRef ) return true;
226        if( ro instanceof CaughtExceptionRef ) return true;
227        if( ro instanceof StringConstant ) return true;
228        return false;
229    }

233    // isAnalyzedRef returns true if the reference r is to be analyzed by this analysis
234    // i.e. its value is not always known (or undecidable)
235    private  final boolean isAnalyzedRef(Value r)
236    {
237        if (isAlwaysNull(r) || isAlwaysTop(r)) 
238            return false;
239        else if (r instanceof Local ||
240                 r instanceof InstanceFieldRef ||
241                 r instanceof StaticFieldRef) {
242            Type rType = r.getType();
243            
244            return (rType instanceof RefType || rType instanceof ArrayType);
245        } else
246            return false;
247    } // end isAnalyzedRef

249    // refInfo is a helper method to tranform our two bit representation back to the four constants
250    // For a given reference and a flow set, tell us if r is bottom, top, null or non-null
251    // Note: this method will fail if r is not in the flow set
252    protected  final int refInfo(EquivalentValue r, FlowSet fs)
253    {
254        boolean isNull = fs.contains(getKRefIntPair(r, kNull));
255        boolean isNonNull = fs.contains(getKRefIntPair(r, kNonNull));
256        
257        if (isNull && isNonNull)
258            return kTop;
259        else if (isNull)
260            return kNull;
261        else if (isNonNull)
262            return kNonNull;
263        else
264            return kBottom;
265    } // end refInfo
266
267    protected  final int refInfo(Value r, FlowSet fs)
268    {
269        return refInfo(getEquivalentValue(r), fs);
270    } // end refInfo
271    
272    // Like refInfo, but the reference doesn't have to be in the flow set
273    // note: it still need to be a reference, i.e. ArrayType or RefType
274    public int anyRefInfo(Value r, FlowSet f)
275    {
276        if (isAlwaysNull(r))
277            return kNull;
278        else if (isAlwaysTop(r))
279            return kTop;
280        else if( isAlwaysNonNull(r) )
281            return kNonNull;
282        else
283            return refInfo(r, f);
284    } // end anyRefInfo

289                uAddInfoToFlowSet

291
292       Adding a pair (r, v) to a set is always a two steps process:
293       a) remove all pairs (r, *)
294       b) add the pair (r, v)

296       The methods above handle that.

303    // method to add (r, kTop) to the gen set (and remove it from the pre set)
304    private  final void uAddTopToFlowSet(EquivalentValue r, FlowSet genFS, FlowSet preFS)
305    {
306        RefIntPair nullPair = getKRefIntPair(r, kNull);
307        RefIntPair nullNonPair = getKRefIntPair(r, kNonNull);
308        
309        if (genFS != preFS) {
310            preFS.remove(nullPair, preFS);
311            preFS.remove(nullNonPair, preFS);
312        }
313        
314        genFS.add(nullPair, genFS);
315        genFS.add(nullNonPair, genFS);
316    } // end uAddTopToFlowSet
317    
318    private  final void uAddTopToFlowSet(Value r, FlowSet genFS, FlowSet preFS)
319    {
320        uAddTopToFlowSet(getEquivalentValue(r), genFS, preFS);
321    } // end uAddTopToFlowSet

335    // method to add (r, kNonNull) or (r, kNull) to the gen set (and remove it from the pre set)
336    private  final void uAddInfoToFlowSet(EquivalentValue r, int v, FlowSet genFS, FlowSet preFS)
337    {
338        int kill;
339        if (v == kNull)
340            kill = kNonNull;
341        else if (v == kNonNull)
342            kill = kNull;
343        else
344            throw new RuntimeException("invalid info");
345        
346        if (genFS != preFS) {
347            preFS.remove(getKRefIntPair(r, kill), preFS);
348        }
349        
350        genFS.remove(getKRefIntPair(r, kill), genFS);
351        genFS.add(getKRefIntPair(r, v), genFS);
352    } // end uAddInfoToFlowSet

410    // method to initialize refTypeLocals, refTypeInstFields, refTypeInstFieldBases
411    // refTypeStaticFields, and refTypeValues
412    // those lists contains fields that can/need to be analyzed
413    private void initRefTypeLists()
414    {
415        refTypeLocals = new ArrayList<EquivalentValue>();
416        refTypeInstFields = new ArrayList<EquivalentValue>();
417        refTypeInstFieldBases = new ArrayList<EquivalentValue>();
418        refTypeStaticFields = new ArrayList<EquivalentValue>();
419        refTypeValues = new ArrayList<EquivalentValue>();
420
421        // build list of locals
422        Iterator it = ((UnitGraph)graph).getBody().getLocals().iterator();
423        
424        while (it.hasNext()) {
425            Local l = (Local) (it.next());
426            
427            if (l.getType() instanceof RefType ||
428                l.getType() instanceof ArrayType) {
429                refTypeLocals.add(getEquivalentValue(l));
430            }
431        }
432        
433        if (isNotConservative) {
434            // build list of fields
435            // if the analysis is not conservative (if it is then we will only work on locals)
436            
437            Iterator unitIt = graph.iterator();
438            
439            while(unitIt.hasNext()) {
440                
441                Unit s = (Unit) unitIt.next();
442                
443                Iterator boxIt;
444                boxIt = s.getUseBoxes().iterator();
445                while(boxIt.hasNext()) initRefTypeLists( (ValueBox) boxIt.next() );
446                boxIt = s.getDefBoxes().iterator();
447                while(boxIt.hasNext()) initRefTypeLists( (ValueBox) boxIt.next() );
448                
449            }
450            
451        } // end build list of fields
452
453        refTypeValues.addAll(refTypeLocals);
454        refTypeValues.addAll(refTypeInstFields);
455        refTypeValues.addAll(refTypeStaticFields);
456        
457        // G.v().out.println("Analyzed references: " + refTypeValues);
458    } // end initRefTypeLists

460    private void initRefTypeLists( ValueBox box ) {
461        Value val = box.getValue();
462        Type opType = null;
463        
464        if (val instanceof InstanceFieldRef) {
465            
466            InstanceFieldRef ir = (InstanceFieldRef) val;
467            
468            opType = ir.getType(); 
469            if (opType instanceof RefType ||
470                opType instanceof ArrayType) {
471                
472                EquivalentValue eir = getEquivalentValue(ir); 
473                
474                if (!refTypeInstFields.contains(eir)) {
475                    refTypeInstFields.add(eir);
476
477                    EquivalentValue eirbase = getEquivalentValue(ir.getBase());
478                    if (!refTypeInstFieldBases.contains(eirbase))
479                        refTypeInstFieldBases.add(eirbase);
480                }
481
482            }
483        } else if (val instanceof StaticFieldRef) {
484            
485            StaticFieldRef sr = (StaticFieldRef) val;
486            opType = sr.getType();
487            
488            if (opType instanceof RefType ||
489                opType instanceof ArrayType) {
490                
491                EquivalentValue esr = getEquivalentValue(sr);
492
493                if (!refTypeStaticFields.contains(esr)) {
494                    refTypeStaticFields.add(esr);
495                }
496            }
497        }
498    }
499    // method to initialize the emptySet, fullSet and tempFlowSet
500    // from the refTypeValues
501    private void initUniverseSets()
502    {
503        FlowUniverse localUniverse;
504        
505        Object[] refTypeValuesArray = refTypeValues.toArray();
506        int len = refTypeValuesArray.length;
507        Object[] universeArray = new Object[2*len];
508        int i;
509        
510        // kRefIntPairs = new HashMap(len*2 + 1, 0.7f);
511        // ideally we would like to be able to do the above to avoid that Map growth
512        // but that would screw concurent execution of this analysis
513        // and making that field non- would require changing our  utility methods to non-
514        
515        for (i = 0; i < len; i++) {
516            int j = i*2;
517            EquivalentValue  r = (EquivalentValue) refTypeValuesArray[i];
518            universeArray[j] = getKRefIntPair(r, kNull);
519            universeArray[j+1] = getKRefIntPair(r,  kNonNull);
520        }
521        
522        localUniverse = new ArrayFlowUniverse(universeArray);
523        
524        emptySet = new ArrayPackedSet(localUniverse);   
525        fullSet = emptySet.clone();
526        ((ArrayPackedSet) emptySet).complement(fullSet);
527        
528        tempFlowSet = (FlowSet) newInitialFlow();
529    } // end initUniverseSets

533    private void initUnitSets()
534    {
535        int cap = graph.size() * 2 + 1;
536        float load = 0.7f;
537        
538        unitToGenerateSet = new HashMap<Unit, FlowSet>(cap, load);
539        unitToPreserveSet = new HashMap<Unit, FlowSet>(cap, load);
540        
541        unitToAnalyzedChecksSet = new HashMap<Unit, HashSet<Value>>(cap, load);
542        unitToArrayRefChecksSet = new HashMap<Unit, HashSet<Value>>(cap, load);
543        unitToInstanceFieldRefChecksSet = new HashMap<Unit, HashSet<Value>>(cap, load);
544        unitToInstanceInvokeExprChecksSet = new HashMap<Unit, HashSet<Value>>(cap, load);
545        unitToLengthExprChecksSet = new HashMap<Unit, HashSet<Value>>(cap, load);
546        
547        
548        Iterator unitIt = graph.iterator();
549        
550        while(unitIt.hasNext()) {
551            
552            Unit s = (Unit) unitIt.next();
553            
554            FlowSet genSet = emptySet.clone();
555            FlowSet preSet = fullSet.clone();
556            
557            HashSet<Value> analyzedChecksSet = new HashSet<Value>(5,  load);
558            HashSet<Value> arrayRefChecksSet = new HashSet<Value>(5,  load);
559            HashSet<Value> instanceFieldRefChecksSet = new HashSet<Value>(5,  load);
560            HashSet<Value> instanceInvokeExprChecksSet = new HashSet<Value>(5,  load);
561            HashSet<Value> lengthExprChecksSet = new HashSet<Value>(5,  load);
562            
563            
564            // *** KILL PHASE ***
565            
566            // naivity here.  kill all the fields after an invoke, i.e. promote them to top
567            if (careForMethodCalls && ((Stmt)s).containsInvokeExpr()) {
568                uListAddTopToFlowSet(refTypeInstFields, genSet, preSet);
569                uListAddTopToFlowSet(refTypeStaticFields, genSet, preSet);
570            }
571            
572            if (careForAliases && (s instanceof AssignStmt)) {
573                AssignStmt as = (AssignStmt) s;
574                Value lhs = as.getLeftOp();
575                
576                if (refTypeInstFieldBases.contains(lhs)) {
577                    // we have a write to a local 'f' and 
578                    // there is a reference to f.x.
579                    // The LHS will certainly be a local.
580                    // here, we kill "f.*".
581
582                    Iterator<EquivalentValue> refTypeInstFieldsIt=refTypeInstFields.iterator();
583                    
584                    while (refTypeInstFieldsIt.hasNext()) {
585                        EquivalentValue eifr = refTypeInstFieldsIt.next();
586                        InstanceFieldRef ifr = (InstanceFieldRef) eifr.getValue();
587                        
588                        if (ifr.getBase() == lhs) {
589                            uAddTopToFlowSet(eifr, genSet, preSet);
590                        }
591                    }
592                }
593                
594                if (lhs instanceof InstanceFieldRef) {
595                    
596                    // we have a write to 'f.x', 
597                    // so we'd better kill 'g.x' for all g.
598                    
599                    String lhsName = 
600                        ((InstanceFieldRef) lhs).getField().getName();
601                    
602                    Iterator<EquivalentValue> refTypeInstFieldsIt = refTypeInstFields.iterator();
603                    
604                    while (refTypeInstFieldsIt.hasNext()) {
605                        EquivalentValue eifr = refTypeInstFieldsIt.next();
606                        InstanceFieldRef ifr = (InstanceFieldRef) eifr.getValue();
607                        
608                        String name = ifr.getField().getName();
609                        
610                        if (name.equals(lhsName)) {
611                            uAddTopToFlowSet(eifr, genSet, preSet);
612                        }
613                    }
614                }
615            } // end if (s instanceof AssignStmt)
616            
617            // kill rhs of defs
618            {
619                Iterator boxIt = s.getDefBoxes().iterator();
620                
621                while(boxIt.hasNext()) {
622                    
623                    ValueBox box = (ValueBox) boxIt.next();
624                    Value boxValue = box.getValue();
625                    
626                    if (isAnalyzedRef(boxValue)) {
627                        uAddTopToFlowSet(boxValue, genSet, preSet);
628                    }
629                }
630            } // done killing rhs of defs
631            
632            // GENERATION PHASE
633            
634            if (s instanceof DefinitionStmt) {
635                DefinitionStmt as = (DefinitionStmt) s;
636                Value ro = as.getRightOp();
637                Value lo = as.getLeftOp();
638                
639                // take out the cast from "x = (type) y;"
640                if (ro instanceof CastExpr)
641                    ro = ((CastExpr) ro).getOp();
642                
643                if (isAnalyzedRef(lo)) {
644                    if (isAlwaysNonNull(ro)) {
645                        uAddInfoToFlowSet(lo, kNonNull, genSet, preSet);
646                    } else if (isAlwaysNull(ro)) {
647                        uAddInfoToFlowSet(lo, kNull, genSet, preSet);
648                    } else if (isAlwaysTop(ro)) {
649                        uAddTopToFlowSet(lo, genSet, preSet);
650                    }
651                }
652            } // end DefinitionStmt gen case
653
654            // check use and def boxes for dereferencing operations
655            // since those operations cause a null pointer check
656            // after the statement we know the involved references are non-null
657            {
658                Iterator boxIt;
659                boxIt = s.getUseBoxes().iterator();
660                while(boxIt.hasNext()) {
661                    Value boxValue = ((ValueBox) boxIt.next()).getValue();
662                    Value base = null;
663                    
664                    if(boxValue instanceof InstanceFieldRef) {
665                        base = ((InstanceFieldRef) (boxValue)).getBase();
666                        instanceFieldRefChecksSet.add(base);
667                    } else if (boxValue instanceof ArrayRef) {
668                        base = ((ArrayRef) (boxValue)).getBase();
669                        arrayRefChecksSet.add(base);
670                    } else if (boxValue instanceof InstanceInvokeExpr) {
671                        base = ((InstanceInvokeExpr) boxValue).getBase();
672                        instanceInvokeExprChecksSet.add(base);                        
673                    } else if (boxValue instanceof LengthExpr) {
674                        base = ((LengthExpr) boxValue).getOp();
675                        lengthExprChecksSet.add(base);
676                    } else if (s instanceof ThrowStmt) {
677                        base = ((ThrowStmt)s).getOp();
678                    } else if (s instanceof MonitorStmt) {
679                        base = ((MonitorStmt)s).getOp();
680                    }
681                    
682                    if (base != null && isAnalyzedRef(base)) { 
683                        uAddInfoToFlowSet(base, kNonNull, genSet, preSet);
684                        analyzedChecksSet.add(base);
685                    }
686                }
687                boxIt = s.getDefBoxes().iterator();
688                while(boxIt.hasNext()) {
689                    
690                    Value boxValue = ((ValueBox) boxIt.next()).getValue();
691                    Value base = null;
692                    
693                    if(boxValue instanceof InstanceFieldRef) {
694                        base = ((InstanceFieldRef) (boxValue)).getBase();
695                        instanceFieldRefChecksSet.add(base);
696                    } else if (boxValue instanceof ArrayRef) {
697                        base = ((ArrayRef) (boxValue)).getBase();
698                        arrayRefChecksSet.add(base);
699                    } else if (boxValue instanceof InstanceInvokeExpr) {
700                        base = ((InstanceInvokeExpr) boxValue).getBase();
701                        instanceInvokeExprChecksSet.add(base);                        
702                    } else if (boxValue instanceof LengthExpr) {
703                        base = ((LengthExpr) boxValue).getOp();
704                        lengthExprChecksSet.add(base);
705                    } else if (s instanceof ThrowStmt) {
706                        base = ((ThrowStmt)s).getOp();
707                    } else if (s instanceof MonitorStmt) {
708                        base = ((MonitorStmt)s).getOp();
709                    }
710                    
711                    if (base != null && isAnalyzedRef(base)) { 
712                        uAddInfoToFlowSet(base, kNonNull, genSet, preSet);
713                        analyzedChecksSet.add(base);
714                    }
715                }
716            } // done check use and def boxes
717
718            unitToGenerateSet.put(s, genSet);
719            unitToPreserveSet.put(s, preSet);
720            
721            unitToAnalyzedChecksSet.put(s, analyzedChecksSet);
722            unitToArrayRefChecksSet.put(s, arrayRefChecksSet);
723            unitToInstanceFieldRefChecksSet.put(s, instanceFieldRefChecksSet);
724            unitToInstanceInvokeExprChecksSet.put(s, instanceInvokeExprChecksSet);
725            unitToLengthExprChecksSet.put(s, lengthExprChecksSet);
726        }
727    } // initUnitSets

729    protected void flowThrough(Object inValue, Unit stmt, List outFallValue, List outBranchValues)
730    {
731        FlowSet in = (FlowSet) inValue;
732        FlowSet out = tempFlowSet;
733        FlowSet pre = unitToPreserveSet.get(stmt);
734        FlowSet gen = unitToGenerateSet.get(stmt);
735
736        // Perform perservation
737        in.intersection(pre, out);
738        
739        // Perform generation
740        out.union(gen, out);
741        
742        // Manually add any x = y; when x and y are both analyzed references
743        // these are not  sets.
744        if (stmt instanceof AssignStmt) {
745            AssignStmt as = (AssignStmt) stmt;
746            Value rightOp = as.getRightOp();
747            Value leftOp = as.getLeftOp();
748            
749            // take out the cast from "x = (type) y;"
750            if (rightOp instanceof CastExpr)
751                rightOp = ((CastExpr) rightOp).getOp();
752            
753            if (isAnalyzedRef(leftOp) && isAnalyzedRef(rightOp)) {
754                int roInfo = refInfo(rightOp, in);
755                
756                if (roInfo == kTop)
757                    uAddTopToFlowSet(leftOp, out);
758                else if (roInfo != kBottom)
759                    uAddInfoToFlowSet(leftOp, roInfo, out);
760            }
761        }
762        
763        // Copy the out value to all branch boxes.
764        {
765            Iterator it = outBranchValues.iterator();
766            while (it.hasNext()) {
767                FlowSet fs = (FlowSet) (it.next());
768                
769                copy(out, fs);
770            }
771        }
772        
773        // Copy the out value to the fallthrough box (don't need iterator)
774        {
775            Iterator it = outFallValue.iterator();
776            while (it.hasNext()) {
777                FlowSet fs = (FlowSet) (it.next());
778                
779                copy(out, fs);
780            }
781        }
782        
783        if (isBranched && (stmt instanceof IfStmt)) {
784            Value cond = ((IfStmt) stmt).getCondition();
785            Value op1 = ((BinopExpr) cond).getOp1();
786            Value op2 = ((BinopExpr) cond).getOp2();
787
788            // make sure at least one of the op is a reference being analyzed
789            // and that none is a reference that is always Top
790            if ((!(isAlwaysTop(op1) || isAlwaysTop(op2))) && (isAnalyzedRef(op1) || isAnalyzedRef(op2))) {
791                Value toGen = null;
792                int toGenInfo = kBottom;
793
794                int op1Info = anyRefInfo(op1, in);
795                int op2Info = anyRefInfo(op2, in);
796                boolean op1isKnown = (op1Info == kNull || op1Info == kNonNull);
797                boolean op2isKnown = (op2Info == kNull || op2Info == kNonNull);
798                
799                if (op1isKnown) {
800                    if (!op2isKnown) {
801                        toGen = op2;
802                        toGenInfo = op1Info;
803                    }
804                } else if (op2isKnown) {
805                    toGen =  op1;
806                    toGenInfo = op2Info;
807                }
808                
809                // only generate info for analyzed references that are top or bottom
810                if ((toGen != null) && isAnalyzedRef(toGen)) {
811                    int fInfo = kBottom;
812                    int bInfo = kBottom;
813                    
814                    if (cond instanceof EqExpr) {
815                        // branching mean op1 == op2
816                        bInfo = toGenInfo;
817                        if (toGenInfo == kNull) {
818                            // falling through mean toGen != null
819                            fInfo = kNonNull;
820                        }
821                        
822                    } else if (cond instanceof NeExpr) {
823                        // if we don't branch that mean op1 == op2
824                        fInfo = toGenInfo;
825                        if (toGenInfo == kNull) {
826                            // branching through mean toGen != null
827                            bInfo = kNonNull;
828                        }
829                    } else
830                        throw new RuntimeException("invalid condition");
831
832                    if (fInfo != kBottom) {
833                        Iterator it = outFallValue.iterator();
834
835                        while(it.hasNext()) {
836                            FlowSet fs = (FlowSet) (it.next());
837
838                            copy(out, fs);
839                            uAddInfoToFlowSet(toGen, fInfo, fs);
840                        }
841                    }
842
843                    if (bInfo !=  kBottom) {
844                        Iterator it = outBranchValues.iterator();
845
846                        while (it.hasNext()) {
847                            FlowSet fs = (FlowSet) (it.next());
848
849                            copy(out, fs);
850                            uAddInfoToFlowSet(toGen, bInfo, fs);
851                        }
852                    } 
853                }
854            }
855        }
856    } // end flowThrough

859    protected void merge(Object in1, Object in2, Object out)
860    {
861        FlowSet inSet1 = (FlowSet) in1;
862        FlowSet inSet2 = (FlowSet) in2;
863        FlowSet inSet1Copy = inSet1.clone();
864        FlowSet inSet2Copy = inSet2.clone();
865        // we do that in case out is in1 or in2
866
867        FlowSet outSet = (FlowSet) out;
868
869        inSet1.intersection(inSet2, outSet);
870        // first step, set out to the intersection of in1 & in2
871        // but we are not over, the intersection doesn't handle the top & bottom cases
872        Iterator<EquivalentValue> it = refTypeValues.iterator();
873        while (it.hasNext()) {
874            EquivalentValue r = it.next();
875            int refInfoIn1 = refInfo(r, inSet1Copy);
876            int refInfoIn2 = refInfo(r, inSet2Copy);
877            if (refInfoIn1 != refInfoIn2) {
878                // only process if they are not equal, otherwise the intersection has done its job
879                if ((refInfoIn1 == kTop) || (refInfoIn2 == kTop)) {
880                    // ok, r is top in one of the sets but not the other, make it top in the outSet
881                    uAddTopToFlowSet(r, outSet);
882                } else if (refInfoIn1 == kBottom) {
883                    // r is bottom in set1 but not set2, promote to the value in set2
884                    uAddInfoToFlowSet(r, refInfoIn2, outSet);
885                } else if (refInfoIn2 == kBottom) {
886                    // r is bottom in set2 but not set1, promote to the value in set1
887                    uAddInfoToFlowSet(r, refInfoIn1, outSet);
888                } else {
889                    // r is known in both set, but it's a different value in each set, make it top
890                    uAddTopToFlowSet(r, outSet);
891                }
892            }
893        }
894    } // end merge
