020package soot.jimple.toolkits.annotation;
021
022import soot.*;
023
024import java.util.*;
025import soot.jimple.*;
026import soot.tagkit.*;
027
028public class LineNumberAdder extends SceneTransformer {
029
030    public LineNumberAdder( Singletons.Global g) {}
031    public static LineNumberAdder v() { return G.v().soot_jimple_toolkits_annotation_LineNumberAdder();}
032
033    public void internalTransform(String phaseName, Map opts){
034
035        Iterator it = Scene.v().getApplicationClasses().iterator();
036        while (it.hasNext()){
037            SootClass sc = (SootClass)it.next();
038            // make map of first line to each method
039            HashMap<Integer, SootMethod> lineToMeth = new HashMap<Integer, SootMethod>();
040            Iterator methIt = sc.getMethods().iterator();
041            while (methIt.hasNext()){
042                SootMethod meth = (SootMethod)methIt.next();
043                if (!meth.isConcrete()) continue;
044                Body body = meth.retrieveActiveBody();
045                Stmt s = (Stmt)body.getUnits().getFirst();
046                while (s instanceof IdentityStmt){
047                    s = (Stmt)body.getUnits().getSuccOf(s);
048                }
049                if (s.hasTag("LineNumberTag")){
050                    LineNumberTag tag = (LineNumberTag)s.getTag("LineNumberTag");
051                    lineToMeth.put(new Integer(tag.getLineNumber()), meth); 
052                }
053            } 
054            Iterator methIt2 = sc.getMethods().iterator();
055            while (methIt2.hasNext()){
056                SootMethod meth = (SootMethod)methIt2.next();
057                if (!meth.isConcrete()) continue;
058                Body body = meth.retrieveActiveBody();
059                Stmt s = (Stmt)body.getUnits().getFirst();
060                while (s instanceof IdentityStmt){
061                    s = (Stmt)body.getUnits().getSuccOf(s);
062                }
063                if (s.hasTag("LineNumberTag")){
064                    LineNumberTag tag = (LineNumberTag)s.getTag("LineNumberTag");
065                    int line_num = tag.getLineNumber() - 1;
066                    // already taken
067                    if (lineToMeth.containsKey(new Integer(line_num))){
068                        meth.addTag(new LineNumberTag(line_num + 1));
069                    }
070                    // still available - so use it for this meth
071                    else {
072                        meth.addTag(new LineNumberTag(line_num));
073                    }
074                }
075            }
076            
077        }
078    }                
079}
