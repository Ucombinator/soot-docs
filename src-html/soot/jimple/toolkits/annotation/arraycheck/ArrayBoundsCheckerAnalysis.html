068    /* A little bit different from ForwardFlowAnalysis */
069    public ArrayBoundsCheckerAnalysis(Body body, 
070                                      boolean takeClassField,
071                                      boolean takeFieldRef,
072                                      boolean takeArrayRef,
073                                      boolean takeCSE,
074                                      boolean takeRectArray)
075    {
076        classfieldin = takeClassField;
077        fieldin = takeFieldRef;
078        arrayin = takeArrayRef;
079        csin = takeCSE;
080        rectarray = takeRectArray;
081        
082        SootMethod thismethod = body.getMethod();
083
084        if (Options.v().debug()) 
085            G.v().out.println("ArrayBoundsCheckerAnalysis started on  "+thismethod.getName());
086
087        ailanalysis = new ArrayIndexLivenessAnalysis(new ExceptionalUnitGraph(body), fieldin, arrayin, csin, rectarray);
088        
089        if (fieldin)
090        {
091            this.localToFieldRef = ailanalysis.getLocalToFieldRef();
092            this.fieldToFieldRef = ailanalysis.getFieldToFieldRef();
093        }
094        
095        if (arrayin)
096        {
097            if (rectarray)
098            {
099                this.multiarraylocals = ailanalysis.getMultiArrayLocals();
100                this.rectarrayset = new HashSet<Local>();
101
102                RectangularArrayFinder pgbuilder = RectangularArrayFinder.v();
103
104                Iterator<Local> localIt = multiarraylocals.iterator();
105                while (localIt.hasNext())
106                {
107                    Local local = localIt.next();
108                    
109                    MethodLocal mlocal = new MethodLocal(thismethod, local);
110                    
111                    if (pgbuilder.isRectangular(mlocal))
112                        this.rectarrayset.add(local);
113                }
114            }
115        }
116        
117        if (csin)
118        {
119            this.localToExpr = ailanalysis.getLocalToExpr();
120        }
121
122        if (classfieldin)
123        {
124            this.cfield = ClassFieldAnalysis.v();    
125        }
126
127        this.graph = new ArrayRefBlockGraph(body);
128
129        blockToBeforeFlow = new HashMap<Block, WeightedDirectedSparseGraph>(graph.size()*2+1, 0.7f);
130        
131        edgeMap = new HashMap<FlowGraphEdge, WeightedDirectedSparseGraph>(graph.size()*2+1, 0.7f);
132        
133        edgeSet = buildEdgeSet(graph);
134        
135        doAnalysis();
136        
137        convertToUnitEntry();
138        
139        if (Options.v().debug()) 
140            G.v().out.println("ArrayBoundsCheckerAnalysis finished.");
141    }

143    private void convertToUnitEntry()
144    {
145        unitToBeforeFlow = new HashMap<Unit, WeightedDirectedSparseGraph>();
146        Iterator<Block> blockIt = blockToBeforeFlow.keySet().iterator();
147        while (blockIt.hasNext())
148        {
149            Block block = blockIt.next();
150            Unit first = block.getHead();
151            unitToBeforeFlow.put(first, blockToBeforeFlow.get(block));
152        }
153    }

155    /** buildEdgeSet creates a set of edges from directed graph.
156     */
157    public Set<FlowGraphEdge> buildEdgeSet(DirectedGraph<Block> dg)
158    {
159        HashSet<FlowGraphEdge> edges = new HashSet<FlowGraphEdge>();
160
161        Iterator<Block> blockIt = dg.iterator();
162        while (blockIt.hasNext())
163        {
164            Block s = blockIt.next();
165
166            List<Block> preds = graph.getPredsOf(s);
167            List<Block> succs = graph.getSuccsOf(s);
168
169            /* Head units has in edge from itself to itself.*/
170            if (preds.size() == 0)
171            {
172                edges.add(new FlowGraphEdge(s,s));
173            }
174            
175            /* End units has out edge from itself to itself.*/
176            if (succs.size() == 0)
177            {
178                edges.add(new FlowGraphEdge(s,s));
179            }
180            else
181            {
182                Iterator succIt = succs.iterator();
183                while (succIt.hasNext())
184                {
185                    edges.add(new FlowGraphEdge(s, succIt.next()));
186                }
187            }
188        }
189        
190        return edges;
191    }

251    /* override ForwardFlowAnalysis
252     */
253    private void doAnalysis()
254    {
255        Date start = new Date();
256        if (Options.v().debug())
257            G.v().out.println("Building PseudoTopological order list on "+start);
258
259        LinkedList allUnits = (LinkedList)SlowPseudoTopologicalOrderer.v().newList(this.graph,false);
260                        
261        BoundedPriorityList changedUnits = 
262            new BoundedPriorityList(allUnits);            
263
264        //               LinkedList changedUnits = new LinkedList(allUnits);        
265
266        Date finish = new Date();
267        if (Options.v().debug())
268        {
269            long runtime = finish.getTime()-start.getTime();
270            long mins = runtime/60000;
271            long secs = (runtime%60000)/1000;
272            G.v().out.println("Building PseudoTopological order finished. "
273                               +"It took "+mins+" mins and "+secs+" secs.");
274        }
275
276        start = new Date();
277
278        HashSet changedUnitsSet = new HashSet(allUnits);
279
280        List<Object> changedSuccs;
281
282        /* An temporary object. */
283        FlowGraphEdge tmpEdge = new FlowGraphEdge();
284
285        /* If any output flow set has unknow value, it will be put in this set
286         */
287        HashSet<Block> unvisitedNodes = new HashSet<Block>(graph.size()*2+1, 0.7f);
288
289        /* adjust livelocals set */
290        {
291            Iterator blockIt = graph.iterator();
292            while (blockIt.hasNext())
293            {
294                Block block = (Block)blockIt.next();
295                HashSet<IntContainer> livelocals = (HashSet<IntContainer>)ailanalysis.getFlowBefore(block.getHead());
296                livelocals.add(zero);
297            }
298        }
299
300        /* Set initial values and nodes to visit. */
301        {
302            stableRoundOfUnits = new HashMap<Block, Integer>();
303            
304            Iterator it = graph.iterator();
305
306            while(it.hasNext())
307            {
308                Block block = (Block)it.next();
309
310                unvisitedNodes.add(block);
311                stableRoundOfUnits.put(block, new Integer(0));
312
313                /* only take out the necessary node set. */
314                HashSet livelocals = (HashSet)ailanalysis.getFlowBefore(block.getHead());
315                
316                blockToBeforeFlow.put(block, new WeightedDirectedSparseGraph(livelocals, false));                
317            }
318
319            Iterator<FlowGraphEdge> edgeIt = edgeSet.iterator();
320            while (edgeIt.hasNext())
321            {
322                FlowGraphEdge edge = edgeIt.next();
323                
324                Block target = (Block)edge.to;
325                HashSet livelocals = (HashSet)ailanalysis.getFlowBefore(target.getHead());
326                
327                edgeMap.put(edge, new WeightedDirectedSparseGraph(livelocals, false));
328            }
329        }
330
331        /* perform customized initialization. */
332        {
333            List headlist = graph.getHeads();
334            Iterator headIt = headlist.iterator();
335
336            while (headIt.hasNext())
337            {
338                Object head = headIt.next() ;
339                FlowGraphEdge edge = new FlowGraphEdge(head, head);
340
341                WeightedDirectedSparseGraph initgraph = 
342                    edgeMap.get(edge) ;
343
344                initgraph.setTop();
345            }        
346        }
347
348        /* Perform fixed point flow analysis.
349         */
350        {
351            WeightedDirectedSparseGraph beforeFlow = 
352                new WeightedDirectedSparseGraph(null, false);
353
354            //            DebugMsg.counter1 += allUnits.size();
355
356            while(!changedUnits.isEmpty())
357            {
358                Block s = (Block) changedUnits.removeFirst();
359                changedUnitsSet.remove(s);
360
361                // DebugMsg.counter2++;
362
363                /* previousAfterFlow, old-out, it is null initially */
364                WeightedDirectedSparseGraph previousBeforeFlow = 
365                    blockToBeforeFlow.get(s);
366                
367                beforeFlow.setVertexes(previousBeforeFlow.getVertexes());
368
369                // Compute and store beforeFlow
370                {
371                    List preds = graph.getPredsOf(s);
372
373                    /* the init node */
374                    if (preds.size() == 0)
375                    {
376                        tmpEdge.changeEndUnits(s,s);
377                        copy(edgeMap.get(tmpEdge), beforeFlow);
378                    }
379                    else if (preds.size() == 1)
380                    {
381                        tmpEdge.changeEndUnits(preds.get(0),s);
382                        copy(edgeMap.get(tmpEdge), beforeFlow);
383                        
384                        //                             widenGraphs(beforeFlow, previousBeforeFlow);
385                    }
386                    else 
387                    {                    
388                    /* has 2 or more preds, Updated by Feng. */
389                        Object predFlows[] = new Object[preds.size()] ;
390 
391                        boolean allUnvisited = true;
392                        Iterator predIt = preds.iterator();
393                        
394                        int index = 0 ;
395
396                        int lastVisited = 0;
397
398                        while(predIt.hasNext())
399                        {
400                            Object pred = predIt.next();
401
402                                  tmpEdge.changeEndUnits(pred,s);
403            
404                            if (!unvisitedNodes.contains(pred))
405                            {
406                                allUnvisited = false;
407                                lastVisited = index;
408                            }
409
410                            predFlows[index++] = edgeMap.get(tmpEdge);
411                              }
412                        // put the visited node as the first one
413
414                        if (allUnvisited)
415                        {
416                            G.v().out.println("Warning : see all unvisited node");
417                        }
418                        else
419                        {
420                            Object tmp = predFlows[0];
421                            predFlows[0] = predFlows[lastVisited];
422                            predFlows[lastVisited] = tmp;
423                        }
424
425                        mergebunch(predFlows, s, previousBeforeFlow, beforeFlow); 
426                    }
427                    
428                    copy(beforeFlow, previousBeforeFlow);
429                }
430
431                /* Compute afterFlow and store it.
432                 * Now we do not have afterFlow, we only have the out edges.
433                 * Different out edges may have different flow set.
434                 * It returns back a list of succ units that edge has been
435                 * changed.
436                 */
437                {
438                    changedSuccs = flowThrough(beforeFlow, s);
439                }
440
441                {
442                    for (int i = 0; i < changedSuccs.size(); i++)
443                    {
444                        Object succ = changedSuccs.get(i);
445                        if (!changedUnitsSet.contains(succ))
446                        {
447                            changedUnits.add(succ);
448                            changedUnitsSet.add(succ);
449                        }
450                    }
451                }
452                
453                /* Decide to remove or add unit from/to unvisitedNodes set */
454                {
455                    unvisitedNodes.remove(s);
456                }
457            }
458        }
459
460        finish = new Date();
461        if (Options.v().debug())
462        {
463            long runtime = finish.getTime()-start.getTime();
464            long mins = runtime/60000;
465            long secs = (runtime/60000)/1000;
466            G.v().out.println("Doing analysis finished."
467                               + " It took "+mins+" mins and "+secs+ "secs.");
468        }
469    }

471    /* Flow go through a node, the output will be put into edgeMap, and also
472     * the changed succ will be in a list to return back.
473     */
474    private List<Object> flowThrough(Object inValue, Object unit)
475    {        
476        ArrayList<Object> changedSuccs = new ArrayList<Object>();
477      
478        WeightedDirectedSparseGraph ingraph = 
479            (WeightedDirectedSparseGraph)inValue;
480
481        Block block = (Block)unit ;
482
483        List succs = block.getSuccs();
484
485        // leave out the last element.
486        Unit s = block.getHead();
487        Unit nexts = block.getSuccOf(s);
488        while (nexts != null)
489        {
490            /* deal with array references */
491            assertArrayRef(ingraph, s);
492
493            /* deal with normal expressions. */
494            assertNormalExpr(ingraph, s);
495
496            s = nexts; 
497            nexts = block.getSuccOf(nexts);
498        }
499
500        // at the end of block, it should update the out edges.
501        if (s instanceof IfStmt)
502        {
503          if (!assertBranchStmt(ingraph, s, block, succs, changedSuccs))
504            updateOutEdges(ingraph, block, succs, changedSuccs);
505        }
506        else
507        {
508            assertArrayRef(ingraph, s);
509            assertNormalExpr(ingraph, s);
510            updateOutEdges(ingraph, block, succs, changedSuccs);
511        }
512
513        return changedSuccs;
514    }

516    private void assertArrayRef(Object in, Unit unit)
517    {
518        if (!(unit instanceof AssignStmt))
519            return;
520
521        Stmt s = (Stmt)unit;
522
523        WeightedDirectedSparseGraph ingraph = (WeightedDirectedSparseGraph)in;
524
525        /*
526        ArrayRef op = null;
527       
528        Value leftOp = ((AssignStmt)s).getLeftOp();
529        Value rightOp = ((AssignStmt)s).getRightOp();
530
531        if (leftOp instanceof ArrayRef)
532            op = (ArrayRef)leftOp;
533
534        if (rightOp instanceof ArrayRef)
535            op = (ArrayRef)rightOp;
536
537        if (op == null)
538            return;
539        */
540        
541        if (!s.containsArrayRef())
542            return;
543
544        ArrayRef op = s.getArrayRef();
545        
546
547        Value base = (op).getBase();
548        Value index = (op).getIndex();
549
550        HashSet livelocals = (HashSet)ailanalysis.getFlowAfter(s);
551        if (!livelocals.contains(base) && !livelocals.contains(index))
552            return;
553        
554        if (index instanceof IntConstant)
555        {
556            int weight = ((IntConstant)index).value;
557            weight = -1-weight;
558
559            ingraph.addEdge(base, zero, weight);
560        }
561        else
562        {
563            // add two edges.
564            // index <= a.length -1;
565            ingraph.addEdge(base, index, -1);
566
567            // index >= 0
568            ingraph.addEdge(index, zero, 0);
569        }
570    }

572    private void assertNormalExpr(Object in, Unit s)
573    {
574
575        WeightedDirectedSparseGraph ingraph = (WeightedDirectedSparseGraph)in;
576
577        /* If it is a contains invoke expr, the parameter should be analyzed. */
578        if (fieldin)
579        {
580            Stmt stmt = (Stmt)s;
581            if (stmt.containsInvokeExpr())
582            {
583                HashSet tokills = new HashSet();
584                Value expr = stmt.getInvokeExpr();
585                List parameters = ((InvokeExpr)expr).getArgs();
586                
587                /* kill only the locals in hierarchy. */
588                if (strictness == 0)
589                {
590                    Hierarchy hierarchy = Scene.v().getActiveHierarchy();
591
592                    for (int i=0; i<parameters.size(); i++)
593                    {
594                        Value para = (Value)parameters.get(i);
595                        Type type = para.getType();
596                        if (type instanceof RefType)
597                        {
598                            SootClass pclass = ((RefType)type).getSootClass();
599
600                            /* then we are looking for the possible types. */
601                            Iterator<Object> keyIt = localToFieldRef.keySet().iterator();
602                            while (keyIt.hasNext())
603                            {
604                                Value local = (Value)keyIt.next();
605
606                                Type ltype = local.getType();
607                                
608                                SootClass lclass = ((RefType)ltype).getSootClass();
609
610                                if (hierarchy.isClassSuperclassOfIncluding(pclass, lclass) ||
611                                    hierarchy.isClassSuperclassOfIncluding(lclass, pclass))
612                                {
613                                    HashSet toadd = localToFieldRef.get(local);
614                                    tokills.addAll(toadd);
615                                }                                    
616                            }
617                        }
618                    }
619                    
620                    if (expr instanceof InstanceInvokeExpr)
621                    {
622                        Value base = ((InstanceInvokeExpr)expr).getBase();
623                        Type type = base.getType();
624                        if (type instanceof RefType)
625                        {
626                            SootClass pclass = ((RefType)type).getSootClass();
627
628                            /* then we are looking for the possible types. */
629                            Iterator<Object> keyIt = localToFieldRef.keySet().iterator();
630                            while (keyIt.hasNext())
631                            {
632                                Value local = (Value)keyIt.next();
633
634                                Type ltype = local.getType();
635                                
636                                SootClass lclass = ((RefType)ltype).getSootClass();
637
638                                if (hierarchy.isClassSuperclassOfIncluding(pclass, lclass) ||
639                                    hierarchy.isClassSuperclassOfIncluding(lclass, pclass))
640                                {
641                                    HashSet toadd = localToFieldRef.get(local);
642                                    tokills.addAll(toadd);
643                                }                                    
644                            }
645                        }
646                    }
647                }
648                else
649                    /* kill all instance field reference. */ 
650                if (strictness == 1)
651                {
652                    boolean killall = false;
653                    if (expr instanceof InstanceInvokeExpr)
654                        killall = true;
655                    else
656                    {
657                        for (int i=0; i<parameters.size(); i++)
658                        {
659                            Value para = (Value)parameters.get(i);
660                            if (para.getType() instanceof RefType)
661                            {
662                                killall = true;
663                                break;
664                            }
665                        }
666                    }
667
668                    if (killall)
669                    {
670                        Iterator<Object> keyIt = localToFieldRef.keySet().iterator();
671                        while (keyIt.hasNext())
672                        {
673                            HashSet toadd = localToFieldRef.get(keyIt.next());
674                            tokills.addAll(toadd);
675                        }                            
676                    }
677                }
678                else
679                if (strictness == 2)
680                {
681                    //                    tokills.addAll(allFieldRefs);
682                    HashSet vertexes = ingraph.getVertexes();
683                    Iterator nodeIt = vertexes.iterator();
684                    while (nodeIt.hasNext())
685                    {
686                        Object node = nodeIt.next();
687                        if (node instanceof FieldRef)
688                            ingraph.killNode(node);
689                    }
690                }
691
692                /*
693                Iterator killIt = tokills.iterator();
694                while (killIt.hasNext())
695                    ingraph.killNode(killIt.next());
696                */
697            }
698        }
699
700        if (arrayin)
701        {
702            Stmt stmt = (Stmt)s;
703
704            if (stmt.containsInvokeExpr())
705            {
706                if (strictness == 2)
707                {
708                    /*
709                    Iterator killIt = allArrayRefs.iterator();
710                    while (killIt.hasNext())
711                    {
712                        ingraph.killNode(killIt.next());
713                    }
714                    */
715                    HashSet vertexes = ingraph.getVertexes();
716                    Iterator nodeIt = vertexes.iterator();
717                    while (nodeIt.hasNext())
718                    {
719                        Object node = nodeIt.next();
720                        if (node instanceof ArrayRef)
721                            ingraph.killNode(node);
722
723                        /*
724                        if (rectarray)
725                            if (node instanceof Array2ndDimensionSymbol)
726                                ingraph.killNode(node);
727                        */
728                    }
729                }
730            }
731        }
732
733        if (!(s instanceof AssignStmt))
734            return;
735
736        Value leftOp = ((AssignStmt)s).getLeftOp();
737        Value rightOp = ((AssignStmt)s).getRightOp();
738
739
740        HashSet livelocals = (HashSet)ailanalysis.getFlowAfter(s);
741
742        if (fieldin)
743        {
744            if (leftOp instanceof Local)
745            {
746                HashSet fieldrefs = localToFieldRef.get(leftOp);
747                
748                if (fieldrefs != null)
749                {
750                    Iterator refsIt = fieldrefs.iterator();
751                    while (refsIt.hasNext())
752                    {
753                        Object ref = refsIt.next();
754                        if (livelocals.contains(ref))
755                            ingraph.killNode(ref);
756                    }
757                }
758            }
759            else
760            if (leftOp instanceof InstanceFieldRef)
761            {
762                SootField field = ((InstanceFieldRef)leftOp).getField();
763                
764                HashSet fieldrefs = fieldToFieldRef.get(field);
765                
766                if (fieldrefs != null)
767                {
768                    Iterator refsIt = fieldrefs.iterator();
769                    while (refsIt.hasNext())
770                    {
771                        Object ref = refsIt.next();
772                        if (livelocals.contains(ref))
773                            ingraph.killNode(ref);
774                    }
775                }                
776            }
777        }
778
779        if (arrayin)
780        {
781            /* a = ..;  kill all references of a;
782               i = ..;  kill all references with index i;
783            */
784            if (leftOp instanceof Local)
785            {
786                /*
787                HashSet arrayrefs = (HashSet)localToArrayRef.get(leftOp);
788                
789                if (arrayrefs != null)
790                {
791                    Iterator refsIt = arrayrefs.iterator();
792                    while (refsIt.hasNext())
793                    {
794                        Object ref = refsIt.next();
795                        ingraph.killNode(ref);
796                    }
797                }
798                */
799                HashSet vertexes = ingraph.getVertexes();
800                Iterator nodeIt = vertexes.iterator();
801                while (nodeIt.hasNext())
802                {
803                    Object node = nodeIt.next();
804                    if (node instanceof ArrayRef)
805                    {
806                        Value base = ((ArrayRef)node).getBase();
807                        Value index = ((ArrayRef)node).getIndex();
808                        
809                        if (base.equals(leftOp) || index.equals(leftOp))
810                            ingraph.killNode(node);                        
811                    }
812
813                    if (rectarray)
814                    {
815                        if (node instanceof Array2ndDimensionSymbol)
816                        {
817                            Object base = ((Array2ndDimensionSymbol)node).getVar();
818                            if (base.equals(leftOp))
819                                ingraph.killNode(node);
820                        }
821                    }
822                }
823            }
824            else
825                /* kill all array references */
826            if (leftOp instanceof ArrayRef)
827            {
828                /*
829                Iterator allrefsIt = allArrayRefs.iterator();
830                while (allrefsIt.hasNext())
831                {
832                    Object ref = allrefsIt.next();
833                    ingraph.killNode(ref);
834                }
835                */
836                HashSet vertexes = ingraph.getVertexes();
837                {
838                    Iterator nodeIt = vertexes.iterator();
839                    while (nodeIt.hasNext())
840                    {
841                        Object node = nodeIt.next();
842                        if (node instanceof ArrayRef)
843                            ingraph.killNode(node);
844                    }
845                }
846
847                /* only when multiarray was given a new value to its sub dimension, we kill all second dimensions of arrays */
848                /*
849                if (rectarray)
850                {
851                    Value base = ((ArrayRef)leftOp).getBase();
852
853                    if (multiarraylocals.contains(base))
854                    {
855                        Iterator nodeIt = vertexes.iterator();
856                        while (nodeIt.hasNext())
857                        {
858                            Object node = nodeIt.next();
859                            if (node instanceof Array2ndDimensionSymbol)
860                                ingraph.killNode(node);
861                        }
862                    }
863                }
864                */
865            }
866        }
867
868        if ( !livelocals.contains(leftOp) && !livelocals.contains(rightOp))
869            return;
870
871        // i = i;
872        if (rightOp.equals(leftOp))
873            return;
874
875        if (csin)
876        {
877            HashSet exprs = localToExpr.get(leftOp);
878            if (exprs != null)
879            {
880                Iterator exprIt = exprs.iterator();
881                while (exprIt.hasNext())
882                {
883                    ingraph.killNode(exprIt.next());
884                }
885            }
886        }
887
888
889        // i = i + c; is special
890        if (rightOp instanceof AddExpr)
891        {
892            Value op1 = ((AddExpr)rightOp).getOp1();
893            Value op2 = ((AddExpr)rightOp).getOp2();
894
895            if (op1 == leftOp && op2 instanceof IntConstant)
896            {
897                int inc_w = ((IntConstant)op2).value;
898                ingraph.updateWeight(leftOp, inc_w);
899                return;
900            }
901            else
902            if (op2 == leftOp && op1 instanceof IntConstant)
903            {
904                int inc_w = ((IntConstant)op1).value;
905                ingraph.updateWeight(leftOp, inc_w);
906                return;
907            }
908        }
909        
910        // i = i - c; is also special
911        if (rightOp instanceof SubExpr)
912        {
913            Value op1 = ((SubExpr)rightOp).getOp1();
914            Value op2 = ((SubExpr)rightOp).getOp2();
915
916            if ((op1 == leftOp) && (op2 instanceof IntConstant))
917            {
918                int inc_w = - ((IntConstant)op2).value;
919                ingraph.updateWeight(leftOp, inc_w);
920                return;
921            }
922        }
923
924        // i = j; i = j + c; i = c + j; need two operations, kill node and add new relationship.
925        // kill left hand side,
926
927        ingraph.killNode(leftOp);
928
929        // add new relationship.
930
931        // i = c;
932        if (rightOp instanceof IntConstant)
933        {
934            int inc_w = ((IntConstant)rightOp).value;
935            ingraph.addMutualEdges(zero, leftOp, inc_w);
936            return;
937        }
938
939        // i = j;
940        if (rightOp instanceof Local)
941        {
942            ingraph.addMutualEdges(rightOp, leftOp, 0);
943            return;
944        }
945
946        if (rightOp instanceof FieldRef)
947        {
948            if (fieldin)
949            {
950                ingraph.addMutualEdges(rightOp, leftOp, 0);
951            }
952
953            if (classfieldin)
954            {            
955                SootField field = ((FieldRef)rightOp).getField();
956                IntValueContainer flength = (IntValueContainer)cfield.getFieldInfo(field);
957
958                if (flength != null)
959                {
960                    if (flength.isInteger())
961                    {
962                        ingraph.addMutualEdges(zero, leftOp, flength.getValue());
963                    }
964                }        
965            }
966
967            return;
968        }
969
970        /*
971        if (rectarray)
972        {
973            Type leftType = leftOp.getType();
974
975            if ((leftType instanceof ArrayType) && (rightOp instanceof ArrayRef))
976            {
977                Local base = (Local)((ArrayRef)rightOp).getBase();
978                
979                SymbolArrayLength sv = (SymbolArrayLength)lra_analysis.getSymbolLengthAt(base, s);
980                if (sv != null)
981                {
982                    ingraph.addMutualEdges(leftOp, sv.next(), 0);
983                }
984            }
985        }
986        */
987
988        if (arrayin)
989        {
990            if (rightOp instanceof ArrayRef)
991            {
992                ingraph.addMutualEdges(rightOp, leftOp, 0);
993
994                if (rectarray)
995                {
996                    Value base = ((ArrayRef)rightOp).getBase();
997
998                    if (rectarrayset.contains(base))
999                    {
1000                        ingraph.addMutualEdges(leftOp, Array2ndDimensionSymbol.v(base), 0);
1001                    }
1002                }
1003
1004                return;
1005            }
1006        }
1007
1008        if (csin)
1009        {
1010            Value rhs = rightOp;
1011
1012            if (rhs instanceof BinopExpr)
1013            {
1014                Value op1 = ((BinopExpr)rhs).getOp1();
1015                Value op2 = ((BinopExpr)rhs).getOp2();
1016
1017                if (rhs instanceof AddExpr)
1018                {
1019                    if ((op1 instanceof Local) &&
1020                        (op2 instanceof Local))
1021                    {
1022                        ingraph.addMutualEdges(rhs, leftOp, 0);
1023                        return;
1024                    }
1025                }
1026                else
1027                if (rhs instanceof MulExpr)
1028                {
1029                    if ((op1 instanceof Local) ||
1030                        (op2 instanceof Local))
1031                    {
1032                        ingraph.addMutualEdges(rhs, leftOp, 0);
1033                        return;
1034                    }                    
1035                }
1036                else
1037                if (rhs instanceof SubExpr)
1038                {
1039                    if (op2 instanceof Local)
1040                    {
1041                        ingraph.addMutualEdges(rhs, leftOp, 0);
1042                        return;
1043                    }
1044                }
1045            }
1046        }
1047
1048
1049        // i = j + c; or i = c + j;
1050        if (rightOp instanceof AddExpr)
1051        {
1052            Value op1 = ((AddExpr)rightOp).getOp1();
1053            Value op2 = ((AddExpr)rightOp).getOp2();
1054
1055            if ( (op1 instanceof Local) && (op2 instanceof IntConstant) )
1056            {
1057                int inc_w = ((IntConstant)op2).value;
1058                ingraph.addMutualEdges(op1, leftOp, inc_w);
1059                return;
1060            }
1061
1062            if ( (op2 instanceof Local) && (op1 instanceof IntConstant) )
1063            {
1064                int inc_w = ((IntConstant)op1).value;
1065                ingraph.addMutualEdges(op2, leftOp, inc_w);
1066                return;
1067            }
1068        }
1069
1070        // only i = j - c was considered. 
1071        if (rightOp instanceof SubExpr)
1072        {
1073            Value op1 = ((SubExpr)rightOp).getOp1();
1074            Value op2 = ((SubExpr)rightOp).getOp2();
1075
1076            if ((op1 instanceof Local) && (op2 instanceof IntConstant))
1077            {
1078                int inc_w = - ((IntConstant)op2).value;
1079                ingraph.addMutualEdges(op1, leftOp, inc_w);
1080                return;
1081            }
1082        }
1083        
1084        // new experessions can also generate relationship
1085        // a = new A[i]; a = new A[c];
1086        if (rightOp instanceof NewArrayExpr)
1087        {
1088            Value size = ((NewArrayExpr)rightOp).getSize();
1089            if (size instanceof Local)
1090            {
1091                ingraph.addMutualEdges(size, leftOp, 0);
1092                return;
1093            }
1094
1095            if (size instanceof IntConstant)
1096            {
1097                int inc_w = ((IntConstant)size).value;
1098                ingraph.addMutualEdges(zero, leftOp, inc_w);
1099                return;
1100            }
1101        }
1102
1103        // a = new A[i][].. ; a = new A[c]...;
1104        if (rightOp instanceof NewMultiArrayExpr)
1105        {
1106            Value size = ((NewMultiArrayExpr)rightOp).getSize(0);
1107            if (size instanceof Local)
1108            {
1109                ingraph.addMutualEdges(size, leftOp, 0);
1110            }
1111            else
1112            if (size instanceof IntConstant)
1113            {
1114                int inc_w = ((IntConstant)size).value;
1115                ingraph.addMutualEdges(zero, leftOp, inc_w);
1116            }
1117
1118            if (arrayin && rectarray)
1119            {
1120                if (((NewMultiArrayExpr)rightOp).getSizeCount() > 1)
1121                {
1122                    size = ((NewMultiArrayExpr)rightOp).getSize(1);
1123                    if (size instanceof Local)
1124                    {
1125                        ingraph.addMutualEdges(size, Array2ndDimensionSymbol.v(leftOp), 0);
1126                    }
1127                    else
1128                    if (size instanceof IntConstant)
1129                    {
1130                        int inc_w = ((IntConstant)size).value;
1131                        ingraph.addMutualEdges(zero, Array2ndDimensionSymbol.v(leftOp), inc_w);
1132                    }
1133                }
1134            }
1135
1136            return;
1137        }
1138
1139        // i = a.length
1140        if (rightOp instanceof LengthExpr)
1141        {
1142            Value base = ((LengthExpr)rightOp).getOp();
1143            ingraph.addMutualEdges(base, leftOp, 0);
1144            return;
1145        }
1146    }

1148  /* assert the branch statement
1149   * return true, if the out condition changed,
1150   *        false, otherwise
1151   */
1152    private boolean assertBranchStmt(Object in,
1153                                     Unit s, Block current,
1154                                     List succs,
1155                                     List<Object> changedSuccs)
1156    {
1157        IfStmt ifstmt = (IfStmt)s;
1158
1159        // take out the condition.
1160        Value cmpcond = ifstmt.getCondition();
1161
1162        if (!(cmpcond instanceof ConditionExpr))
1163            return false;
1164
1165        // how may succs?
1166        if (succs.size() != 2) {
1167          return false;
1168        }
1169
1170        Stmt targetUnit = ifstmt.getTarget();
1171        Block targetBlock = (Block)succs.get(0);
1172        Block nextBlock = (Block)succs.get(1);
1173
1174        if (!targetUnit.equals(targetBlock.getHead()))
1175        {
1176            Block swap = targetBlock;
1177            targetBlock = nextBlock;
1178            nextBlock = swap;
1179        }
1180
1181        Value op1 = ((ConditionExpr)cmpcond).getOp1();
1182        Value op2 = ((ConditionExpr)cmpcond).getOp2();
1183
1184        HashSet livelocals = (HashSet)ailanalysis.getFlowAfter(s);
1185
1186        if (!livelocals.contains(op1) && !livelocals.contains(op2))
1187            return false;
1188
1189        WeightedDirectedSparseGraph ingraph = (WeightedDirectedSparseGraph)in;
1190
1191        WeightedDirectedSparseGraph targetgraph = ingraph.dup();
1192
1193        // EqExpr, GeExpr, GtExpr, LeExpr, LtExpr, NeExpr 
1194        if ((cmpcond instanceof EqExpr) ||
1195            (cmpcond instanceof NeExpr) )
1196        {
1197            Object node1 = op1, node2 = op2;
1198            int weight = 0;
1199            if (node1 instanceof IntConstant)
1200            {
1201                      weight = ((IntConstant)node1).value;
1202                node1 = zero;
1203            }
1204            if (node2 instanceof IntConstant)
1205            {
1206                weight = ((IntConstant)node2).value;
1207                node2 = zero;
1208            }
1209
1210            if (node1 == node2)
1211                return false;
1212            
1213            if (cmpcond instanceof EqExpr)
1214                targetgraph.addMutualEdges(node1, node2, weight);
1215            else
1216                ingraph.addMutualEdges(node1, node2, weight);
1217        }
1218        else
1219            // i > j
1220        if ((cmpcond instanceof GtExpr) ||
1221            // i >= j
1222            (cmpcond instanceof GeExpr) )
1223        {
1224            Object node1 = op1, node2 = op2;
1225            int weight = 0;
1226            
1227            if (node1 instanceof IntConstant)
1228            {
1229                weight += ((IntConstant)node1).value;
1230                node1 = zero;
1231            }
1232
1233            if (node2 instanceof IntConstant)
1234            {
1235                weight -= ((IntConstant)node2).value;
1236                node2 = zero;
1237            }
1238            
1239            if (node1 == node2)
1240                return false;
1241
1242            if (cmpcond instanceof GtExpr)
1243            {
1244                targetgraph.addEdge(node1, node2, weight-1);
1245                ingraph.addEdge(node2, node1, -weight);
1246            }
1247            else
1248            {
1249                targetgraph.addEdge(node1, node2, weight);
1250                ingraph.addEdge(node2, node1, -weight-1);                                
1251            }
1252        }
1253        else
1254            // i < j 
1255        if ((cmpcond instanceof LtExpr) ||
1256            (cmpcond instanceof LeExpr) )
1257        {
1258            Object node1 = op1, node2 = op2;
1259            int weight = 0;
1260            
1261            if (node1 instanceof IntConstant)
1262            {
1263                weight -= ((IntConstant)node1).value;
1264                node1 = zero;
1265            }
1266
1267            if (node2 instanceof IntConstant)
1268            {
1269                weight += ((IntConstant)node2).value;
1270                node2 = zero;
1271            }
1272
1273            if (node1 == node2)
1274                return false;
1275            
1276            if (cmpcond instanceof LtExpr)
1277            {
1278                targetgraph.addEdge(node2, node1, weight-1);
1279                ingraph.addEdge(node1, node2, -weight);
1280            }
1281            else
1282            {
1283                targetgraph.addEdge(node2, node1, weight);
1284                ingraph.addEdge(node1, node2, -weight-1);
1285            }
1286        }
1287        else
1288            return false;
1289
1290        // update out edges and changed succs.
1291        // targetgraph -> targetBlock
1292
1293        FlowGraphEdge targetEdge = new FlowGraphEdge(current, targetBlock);
1294        WeightedDirectedSparseGraph prevtarget = edgeMap.get(targetEdge);
1295        boolean changed = false;
1296
1297               targetgraph.makeShortestPathGraph();
1298
1299        WeightedDirectedSparseGraph tmpgraph =
1300            new WeightedDirectedSparseGraph(prevtarget.getVertexes(), true);
1301
1302        copy(targetgraph, tmpgraph);
1303
1304        if (!tmpgraph.equals(prevtarget))
1305        {
1306            prevtarget.replaceAllEdges(tmpgraph);
1307            changed = true;
1308        }
1309
1310        if (changed)
1311            changedSuccs.add(targetBlock);
1312
1313        
1314        // ingraph -> nextBlock
1315        FlowGraphEdge nextEdge = new FlowGraphEdge(current, nextBlock);
1316        WeightedDirectedSparseGraph prevnext = edgeMap.get(nextEdge);
1317        changed = false;
1318
1319               ingraph.makeShortestPathGraph();
1320
1321        tmpgraph = new WeightedDirectedSparseGraph(prevnext.getVertexes(), true);
1322            
1323        copy(ingraph, tmpgraph);
1324
1325        if (!tmpgraph.equals(prevnext))
1326        {
1327            prevnext.replaceAllEdges(tmpgraph);
1328            changed = true;
1329        }
1330
1331        if (changed)
1332            changedSuccs.add(nextBlock);                
1333
1334        return true;
1335
1336    }
