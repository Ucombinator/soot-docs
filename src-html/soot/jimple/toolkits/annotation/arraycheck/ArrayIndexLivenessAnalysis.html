103    public ArrayIndexLivenessAnalysis(DirectedGraph dg, 
104                                      boolean takeFieldRef, 
105                                      boolean takeArrayRef,
106                                      boolean takeCSE,
107                                      boolean takeRectArray)
108    {
109        super(dg);
110        
111        fieldin = takeFieldRef;
112        arrayin = takeArrayRef;
113        csin = takeCSE;
114        rectarray = takeRectArray;
115        
116        if (Options.v().debug()) 
117            G.v().out.println("Enter ArrayIndexLivenessAnalysis");
118        
119        eug = (ExceptionalUnitGraph)dg;
120        retrieveAllArrayLocals(eug.getBody(), fullSet);
121        
122        /* compute gen set, kill set, and condition set */
123        genOfUnit = new HashMap<Stmt, HashSet<Object>>(eug.size()*2+1);
124        absGenOfUnit = new HashMap<Stmt, HashSet<Value>>(eug.size()*2+1);
125        killOfUnit = new HashMap<Stmt, HashSet<Value>>(eug.size()*2+1);
126        conditionOfGen = new HashMap<Stmt, HashSet<Value>>(eug.size()*2+1);
127        
128        if (fieldin)
129        {
130            localToFieldRef = new HashMap<Object, HashSet<Value>>();
131            fieldToFieldRef = new HashMap<Object, HashSet<Value>>();
132            allFieldRefs = new HashSet<Value>();
133        }
134        
135        if (arrayin)
136        {
137            localToArrayRef = new HashMap();
138            allArrayRefs = new HashSet();
139            
140            killArrayRelated = new HashMap<DefinitionStmt, Value>();
141            killAllArrayRef = new HashMap<DefinitionStmt, Boolean>();
142            
143            if (rectarray)
144            {
145                multiarraylocals = new HashSet<Local>();
146                retrieveMultiArrayLocals(eug.getBody(), multiarraylocals);
147            }
148        }
149        
150        if (csin)
151        {
152            localToExpr = new HashMap<Value, HashSet<Value>>();
153        }
154
155        getAllRelatedMaps(eug.getBody());
156
157        getGenAndKillSet(eug.getBody(), absGenOfUnit, genOfUnit, killOfUnit, conditionOfGen);
158
159        doAnalysis();
160
161        if (Options.v().debug()) 
162            G.v().out.println("Leave ArrayIndexLivenessAnalysis");
163    }

200    private void getAllRelatedMaps(Body body)
201    {
202        Iterator unitIt = body.getUnits().iterator();
203        while (unitIt.hasNext())
204        {
205            Stmt stmt = (Stmt)unitIt.next();
206            
207            if (csin)
208            {
209                if (stmt instanceof DefinitionStmt)
210                {
211                    Value rhs = ((DefinitionStmt)stmt).getRightOp();
212
213                    if (rhs instanceof BinopExpr) 
214                    {
215                        Value op1 = ((BinopExpr)rhs).getOp1();
216                        Value op2 = ((BinopExpr)rhs).getOp2();
217                        
218                        if (rhs instanceof AddExpr) 
219                        {
220                            // op1 + op2 --> a + b
221                            if ((op1 instanceof Local) &&
222                                (op2 instanceof Local)) 
223                            {
224                                HashSet<Value> refs = localToExpr.get(op1);
225                                if (refs == null)
226                                {
227                                    refs = new HashSet<Value>();
228                                    localToExpr.put(op1, refs);
229                                }
230                                refs.add(rhs);
231                                
232                                refs = localToExpr.get(op2);
233                                if (refs == null)
234                                {
235                                    refs = new HashSet<Value>();
236                                    localToExpr.put(op2, refs);
237                                }
238                                refs.add(rhs);
239                            }
240                        }
241                        // a * b, a * c, c * a
242                        else if (rhs instanceof MulExpr)
243                        {
244                            HashSet<Value> refs = localToExpr.get(op1);
245                            if (refs == null)
246                            {
247                                refs = new HashSet<Value>();
248                                localToExpr.put(op1, refs);
249                            }
250                            refs.add(rhs);
251                
252                            refs = localToExpr.get(op2);
253                            if (refs == null)
254                            {
255                                refs = new HashSet<Value>();
256                                localToExpr.put(op2, refs);
257                            }
258                            refs.add(rhs);
259                        }
260                        else if (rhs instanceof SubExpr)
261                        {
262                            if (op2 instanceof Local)
263                            {
264                                HashSet<Value> refs = localToExpr.get(op2);
265                                if (refs == null)
266                                {
267                                    refs = new HashSet<Value>();
268                                    localToExpr.put(op2, refs);
269                                }
270                                refs.add(rhs);
271                                
272                                if (op1 instanceof Local)
273                                {
274                                    refs = localToExpr.get(op1);
275                                    if (refs == null)
276                                    {
277                                        refs = new HashSet<Value>();
278                                        localToExpr.put(op1, refs);
279                                    }
280                                    refs.add(rhs);
281                                }
282                            }
283                        }
284                    }
285                }
286            }
287            
288            for (ValueBox vbox : stmt.getUseAndDefBoxes())
289            {
290                Value v = vbox.getValue();
291                
292                if (fieldin)
293                {
294                    if (v instanceof InstanceFieldRef)
295                    {
296                        Value base = ((InstanceFieldRef)v).getBase();
297                        SootField field = ((InstanceFieldRef)v).getField();
298                        
299                        HashSet<Value> baseset = localToFieldRef.get(base);
300                        if (baseset == null)
301                        {
302                            baseset = new HashSet<Value>();
303                            localToFieldRef.put(base, baseset);
304                        }
305                        
306                        baseset.add(v);
307                        
308                        HashSet<Value> fieldset = fieldToFieldRef.get(field);
309                        if (fieldset == null)
310                        {
311                            fieldset = new HashSet<Value>();
312                            fieldToFieldRef.put(field, fieldset);
313                        }
314
315                        fieldset.add(v);
316                    }
317
318                    if (v instanceof FieldRef)
319                        allFieldRefs.add(v);
320                }
321
322                if (arrayin)
323                {
324                    // a = ...   --> kill all a[x] nodes.
325                    // a[i] = .. --> kill all array references.
326                    // m(a)      --> kill all array references 
327                    // i = ...   --> kill all array reference with index as i
328                /*
329                if (v instanceof ArrayRef)
330                {
331                    Value base = ((ArrayRef)v).getBase();
332                    Value index = ((ArrayRef)v).getIndex();
333
334                    HashSet refset = (HashSet)localToArrayRef.get(base);
335                    if (refset == null)
336                    {
337                        refset = new HashSet();
338                        localToArrayRef.put(base, refset);
339                    }
340                    refset.add(v);
341
342                    if (index instanceof Local)
343                    {
344                        refset = (HashSet)localToArrayRef.get(index);
345                        if (refset == null)
346                        {
347                            refset = new HashSet();
348                            localToArrayRef.put(index, refset);
349                        }
350
351                        refset.add(v);
352                    }           
353                    allArrayRefs.add(v);
354                }
355                */
356                }
357            }
358        }
359    }

361    private void retrieveAllArrayLocals(Body body, Set<Local> container)
362    {
363        for (Local local : body.getLocals())
364        {
365            Type type = local.getType();
366            
367            if (type instanceof IntType
368                || type instanceof ArrayType)
369                container.add(local);
370        }
371    }

373    private void retrieveMultiArrayLocals(Body body, Set<Local> container)
374    {
375        for (Local local : body.getLocals())
376        {
377            Type type = local.getType();
378            
379            if (type instanceof ArrayType)
380            {
381                if (((ArrayType)type).numDimensions > 1)
382                    this.multiarraylocals.add(local);
383            }
384        }
385    }

387    private void getGenAndKillSetForDefnStmt(DefinitionStmt asstmt, 
388                                             HashMap<Stmt, HashSet<Value>> absgen,
389                                             HashSet<Object> genset,
390                                             HashSet<Value> absgenset,
391                                             HashSet<Value> killset,
392                                             HashSet<Value> condset)
393    {
394        /* kill left hand side */
395        Value lhs = asstmt.getLeftOp();
396        Value rhs = asstmt.getRightOp();
397        
398        boolean killarrayrelated = false;
399        boolean killallarrayref = false;
400        
401        if (fieldin)
402        {
403            if (lhs instanceof Local)
404            {
405                HashSet<Value> related = localToFieldRef.get(lhs);
406                if (related != null)
407                    killset.addAll(related);
408            }
409            else
410                if (lhs instanceof StaticFieldRef)
411                {
412                    killset.add(lhs);
413                    condset.add(lhs);
414                }
415                else
416                    if (lhs instanceof InstanceFieldRef)
417                    {
418                        SootField field = ((InstanceFieldRef)lhs).getField();
419                        HashSet<Value> related = fieldToFieldRef.get(field);
420                        if (related != null)
421                            killset.addAll(related);
422                        condset.add(lhs);
423                    }
424            
425            if (asstmt.containsInvokeExpr())
426            {
427                /*
428                Value expr = asstmt.getInvokeExpr();
429                List parameters = ((InvokeExpr)expr).getArgs();
430
431                // add the method invocation
432                boolean killall = false;
433                if (expr instanceof InstanceInvokeExpr)
434                    killall = true;
435                else
436                {
437                    for (int i=0; i<parameters.size(); i++)
438                    {
439                    Value para = (Value)parameters.get(i);
440                    if (para.getType() instanceof RefType)
441                    {
442                        killall = true;
443                        break;
444                    }
445                    }
446                }
447    
448                if (killall)
449                {
450                    killset.addAll(allInstFieldRefs);
451                }   
452                */
453
454                killset.addAll(allFieldRefs);
455            } 
456        }
457        
458        if (arrayin)
459        {
460            // a = ... or i = ...
461            if (lhs instanceof Local)
462            {
463                killarrayrelated = true;
464            }
465            else
466                // a[i] = ...
467                if (lhs instanceof ArrayRef)
468                {
469                    killallarrayref = true;
470                    condset.add(lhs);
471                }
472            
473            // invokeexpr kills all array references.
474            if (asstmt.containsInvokeExpr())
475            {
476                killallarrayref = true;
477            }
478        }
479        
480        if (csin)
481        {
482            HashSet<Value> exprs = localToExpr.get(lhs);
483            if (exprs != null)
484                killset.addAll(exprs);
485
486            if (rhs instanceof BinopExpr)
487            {
488                Value op1 = ((BinopExpr)rhs).getOp1();
489                Value op2 = ((BinopExpr)rhs).getOp2();
490
491                if (rhs instanceof AddExpr)
492                {
493                    if ((op1 instanceof Local) &&
494                        (op2 instanceof Local))
495                        genset.add(rhs);
496                }
497                else
498                    if (rhs instanceof MulExpr)
499                    {
500                        if ((op1 instanceof Local) ||
501                            (op2 instanceof Local))
502                            genset.add(rhs);
503                    }
504                    else
505                        if (rhs instanceof SubExpr)
506                        {
507                            if (op2 instanceof Local)
508                                genset.add(rhs);
509                        }
510            }
511        }
512        
513        if ((lhs instanceof Local)&&(fullSet.contains(lhs)))
514        {
515            killset.add(lhs);
516            /* speculatively add lhs as live condition. */
517            condset.add(lhs);
518        }
519        else if (lhs instanceof ArrayRef)
520        {
521            /* a[i] generate a and i. */
522
523            Value base = ((ArrayRef)lhs).getBase();
524            Value index = ((ArrayRef)lhs).getIndex();
525
526            absgenset.add(base);
527            
528            if (index instanceof Local)
529            {
530                absgenset.add(index);
531            }
532        }
533        
534        if (rhs instanceof Local)
535        {
536            /* only lhs=rhs is valid. */
537            /*
538              if (lhs instanceof Local && fullSet.contains(rhs))
539              genset.add(rhs);
540            */
541            if (fullSet.contains(rhs))
542                genset.add(rhs);
543            
544            /*
545              if (fieldin && (lhs instanceof FieldRef))
546              genset.add(rhs);
547            */
548        }
549        else if (rhs instanceof FieldRef)
550        {
551            if (fieldin)
552                genset.add(rhs);
553        }
554        else if (rhs instanceof ArrayRef)
555        {
556            /* lhs=a[i]. */
557
558            Value base = ((ArrayRef)rhs).getBase();
559            Value index = ((ArrayRef)rhs).getIndex();
560            
561            absgenset.add(base);
562            if (index instanceof Local)
563            {
564                absgenset.add(index);
565            }
566            
567            if (arrayin)
568            {
569                genset.add(rhs);
570
571                if (rectarray)
572                    genset.add(Array2ndDimensionSymbol.v(base));
573            }
574        }
575        else if (rhs instanceof NewArrayExpr)
576        {
577            /* a = new A[i]; */
578            Value size = ((NewArrayExpr)rhs).getSize();
579            if (size instanceof Local)
580                genset.add(size);
581        }
582        else if (rhs instanceof NewMultiArrayExpr)
583        {
584            /* a = new A[i][]...;*/
585            /* More precisely, we should track other dimensions. */
586                        
587            List sizes = ((NewMultiArrayExpr)rhs).getSizes();
588            Iterator sizeIt = sizes.iterator();
589            while (sizeIt.hasNext())
590            {
591                Value size = (Value)sizeIt.next();
592                
593                if (size instanceof Local)
594                    genset.add(size);
595            }
596        }
597        else if (rhs instanceof LengthExpr)
598        {
599            /* lhs = lengthof rhs */
600            Value op = ((LengthExpr)rhs).getOp();
601            genset.add(op);
602        }
603        else if (rhs instanceof JAddExpr)                        
604        {
605            /* lhs = rhs+c, lhs=c+rhs */
606            Value op1 = ((JAddExpr)rhs).getOp1();
607            Value op2 = ((JAddExpr)rhs).getOp2();
608            
609            if ((op1 instanceof IntConstant) &&
610                (op2 instanceof Local))
611            {
612                genset.add(op2);
613            }
614            else if ((op2 instanceof IntConstant) &&
615                     (op1 instanceof Local))
616            {
617                genset.add(op1);
618            }
619        }
620        else if (rhs instanceof JSubExpr)
621        {
622            Value op1 = ((JSubExpr)rhs).getOp1();
623            Value op2 = ((JSubExpr)rhs).getOp2();
624            
625            if ((op1 instanceof Local) &&
626                (op2 instanceof IntConstant))
627            {
628                genset.add(op1);
629            }
630        }
631        
632        if (arrayin)
633        {
634            if (killarrayrelated)
635                killArrayRelated.put(asstmt, lhs);
636            
637            if (killallarrayref)
638                killAllArrayRef.put(asstmt, new Boolean(true));
639        }
640    }

642    private void getGenAndKillSet(Body body, HashMap<Stmt, HashSet<Value>> absgen, HashMap<Stmt, HashSet<Object>> gen, HashMap<Stmt, HashSet<Value>> kill, HashMap<Stmt, HashSet<Value>> condition)
643    {
644        for (Unit u : body.getUnits())
645        {
646            Stmt stmt = (Stmt)u;
647
648            HashSet<Object> genset = new HashSet<Object>();
649            HashSet<Value> absgenset = new HashSet<Value>();
650            HashSet<Value> killset = new HashSet<Value>();
651            HashSet<Value> condset = new HashSet<Value>();
652            
653            if (stmt instanceof DefinitionStmt)
654            {
655                getGenAndKillSetForDefnStmt((DefinitionStmt)stmt, absgen,
656                                            genset, absgenset, 
657                                            killset, condset);
658
659            }
660            else if (stmt instanceof IfStmt)
661            {
662                /* if one of condition is living, than other one is live. */
663                Value cmpcond = ((IfStmt)stmt).getCondition();
664                
665                if (cmpcond instanceof ConditionExpr)
666                {
667                    Value op1 = ((ConditionExpr)cmpcond).getOp1();
668                    Value op2 = ((ConditionExpr)cmpcond).getOp2();
669                    
670                    if (fullSet.contains(op1) && fullSet.contains(op2))
671                    {
672                        condset.add(op1);
673                        condset.add(op2);
674                        
675                        genset.add(op1);
676                        genset.add(op2);
677                    }
678                }
679            }
680            
681            if (genset.size() != 0)
682                gen.put(stmt, genset);
683            if (absgenset.size() != 0)
684                absgen.put(stmt, absgenset);
685            if (killset.size() != 0)
686                kill.put(stmt, killset);
687            if (condset.size() != 0)
688                condition.put(stmt, condset);
689        }
690    }

706    protected void flowThrough(Object inValue, Object unit, Object outValue)
707    {
708        HashSet inset = (HashSet)inValue;
709        HashSet outset = (HashSet)outValue;
710        Stmt stmt = (Stmt)unit;
711        
712        /* copy in set to out set. */
713        outset.clear();
714        outset.addAll(inset);
715        
716        HashSet<Object> genset = genOfUnit.get(unit);
717        HashSet<Value> absgenset = absGenOfUnit.get(unit);
718        HashSet<Value> killset = killOfUnit.get(unit);
719        HashSet<Value> condset = conditionOfGen.get(unit);
720        
721        if (killset != null)
722            outset.removeAll(killset);
723        
724        if (arrayin)
725        {
726            Boolean killall = killAllArrayRef.get(stmt);
727            
728            if ((killall != null) && killall.booleanValue())
729            {
730                List keylist = new ArrayList(outset);
731                Iterator keyIt = keylist.iterator();
732                while (keyIt.hasNext())
733                {
734                    Object key = keyIt.next();
735                    if (key instanceof ArrayRef)
736                        outset.remove(key);
737                }
738            }
739            else
740            {
741                Object local = killArrayRelated.get(stmt);
742                if (local != null)
743                {
744                    List keylist = new ArrayList(outset);
745                    Iterator keyIt = keylist.iterator();
746                    while (keyIt.hasNext())
747                    {
748                        Object key = keyIt.next();
749                        if (key instanceof ArrayRef)
750                        {
751                            Value base = ((ArrayRef)key).getBase();
752                            Value index = ((ArrayRef)key).getIndex();
753                            
754                            if (base.equals(local) || index.equals(local))
755                                outset.remove(key);
756                        }
757                        
758                        if (rectarray)
759                        {
760                            if (key instanceof Array2ndDimensionSymbol)
761                            {
762                                Object base = ((Array2ndDimensionSymbol)key).getVar();
763                                if (base.equals(local))
764                                    outset.remove(key);
765                            }
766                        }
767                    }
768                }
769            }
770        }
771        
772        if (genset != null)
773        {
774            if (condset == null || (condset.size()==0))
775                outset.addAll(genset);
776            else
777            {
778                Iterator condIt = condset.iterator();
779                while (condIt.hasNext())
780                {
781                    if (inset.contains(condIt.next()))
782                    {
783                        outset.addAll(genset);
784                        break;
785                    }
786                }
787            }
788        }
789        
790        if (absgenset != null)
791            outset.addAll(absgenset);
792    }

794    protected void merge(Object in1, Object in2, Object out)
795    {
796        HashSet inset1 = (HashSet)in1;
797        HashSet inset2 = (HashSet)in2;
798        HashSet outset = (HashSet)out;
799        
800        HashSet src = inset1;
801        
802        if (outset == inset1)
803            src = inset2;
804        else if (outset == inset2)
805            src = inset1;
806        else
807        {
808            outset.clear();
809            outset.addAll(inset2);
810        }
811        
812        outset.addAll(src);
813    }

815    protected void copy(Object source, Object dest)
816    {
817        if (source == dest)
818            return;
819        
820        HashSet sourceSet = (HashSet)source;
821        HashSet destSet = (HashSet)dest;
822        
823        destSet.clear();
824        destSet.addAll(sourceSet);
825    }
