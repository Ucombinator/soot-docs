034class ConstraintChecker extends AbstractStmtSwitch {
035        private final ClassHierarchy hierarchy;
036        private final boolean fix; // if true, fix constraint violations

038        private JimpleBody stmtBody;

040        public ConstraintChecker(TypeResolver resolver, boolean fix) {
041                this.fix = fix;
042
043                hierarchy = resolver.hierarchy();
044        }

046        public void check(Stmt stmt, JimpleBody stmtBody) throws TypeException {
047                try {
048                        this.stmtBody = stmtBody;
049                        stmt.apply(this);
050                } catch (RuntimeTypeException e) {
051                        StringWriter st = new StringWriter();
052                        PrintWriter pw = new PrintWriter(st);
053                        e.printStackTrace(pw);
054                        pw.close();
055                        throw new TypeException(st.toString());
056                }
057        }

059        @SuppressWarnings("serial")
060        private static class RuntimeTypeException extends RuntimeException {
061                RuntimeTypeException(String message) {
062                        super(message);
063                }
064        }
065
066        static void error(String message) {
067                throw new RuntimeTypeException(message);
068        }

070        private void handleInvokeExpr(InvokeExpr ie, Stmt invokestmt) {
071                // Handle the parameters
072                SootMethodRef method = ie.getMethodRef();
073                for (int i = 0; i < ie.getArgCount(); i++) {
074                        if (ie.getArg(i) instanceof Local) {
075                                Local local = (Local) ie.getArg(i);
076                                if (!hierarchy.typeNode(local.getType()).hasAncestorOrSelf(
077                                                hierarchy.typeNode(method.parameterType(i)))) {
078                                        if (fix) {
079                                                ie.setArg(i, insertCast(local, method.parameterType(i), invokestmt));
080                                        } else {
081                                                error("Type Error");
082                                        }
083                                }
084                        }
085                }
086
087                if (ie instanceof InterfaceInvokeExpr) {
088                        InterfaceInvokeExpr invoke = (InterfaceInvokeExpr) ie;
089                        Value base = invoke.getBase();
090                        if (base instanceof Local) {
091                                Local local = (Local) base;
092                                if (!hierarchy.typeNode(local.getType()).hasAncestorOrSelf(
093                                                hierarchy.typeNode(method.declaringClass().getType()))) {
094                                        if (fix) {
095                                                invoke.setBase(insertCast(local, method.declaringClass().getType(), invokestmt));
096                                        } else {
097                                                error("Type Error(7): local " + local + " is of incompatible type " + local.getType());
098                                        }
099                                }
100                        }
101                } else if (ie instanceof SpecialInvokeExpr) {
102                        SpecialInvokeExpr invoke = (SpecialInvokeExpr) ie;
103                        Value base = invoke.getBase();
104                        if (base instanceof Local) {
105                                Local local = (Local) base;
106                                if (!hierarchy.typeNode(local.getType()).hasAncestorOrSelf(
107                                                hierarchy.typeNode(method.declaringClass().getType()))) {
108                                        if (fix) {
109                                                invoke.setBase(insertCast(local, method.declaringClass().getType(), invokestmt));
110                                        } else {
111                                                error("Type Error(9)");
112                                        }
113                                }
114                        }
115                } else if (ie instanceof VirtualInvokeExpr) {
116                        VirtualInvokeExpr invoke = (VirtualInvokeExpr) ie;
117                        Value base = invoke.getBase();
118                        if (base instanceof Local) {
119                                Local local = (Local) base;
120                                if (!hierarchy.typeNode(local.getType()).hasAncestorOrSelf(
121                                                hierarchy.typeNode(method.declaringClass().getType()))) {
122                                        if (fix) {
123                                                invoke.setBase(insertCast(local, method.declaringClass().getType(), invokestmt));
124                                        } else {
125                                                error("Type Error(13)");
126                                        }
127                                }
128                        }
129                } else if (ie instanceof StaticInvokeExpr) {
130                        // No base to handle
131                } else if (ie instanceof DynamicInvokeExpr) {
132                        DynamicInvokeExpr die = (DynamicInvokeExpr) ie;
133                        SootMethodRef bootstrapMethod = die.getMethodRef();
134                        for (int i = 0; i < die.getBootstrapArgCount(); i++) {
135                                if (die.getBootstrapArg(i) instanceof Local) {
136                                        Local local = (Local) die.getBootstrapArg(i);
137                                        if (!hierarchy.typeNode(local.getType()).hasAncestorOrSelf(
138                                                        hierarchy.typeNode(bootstrapMethod.parameterType(i)))) {
139                                                if (fix) {
140                                                        ie.setArg(i, insertCast(local, bootstrapMethod.parameterType(i), invokestmt));
141                                                } else {
142                                                        error("Type Error");
143                                                }
144                                        }
145                                }
146                        }
147                } else {
148                        throw new RuntimeException("Unhandled invoke expression type: " + ie.getClass());
149                }
150        }

160        public void caseAssignStmt(AssignStmt stmt) {
161                Value l = stmt.getLeftOp();
162                Value r = stmt.getRightOp();
163
164                TypeNode left = null;
165
166                // ******** LEFT ********
167
168                if (l instanceof ArrayRef) {
169                        ArrayRef ref = (ArrayRef) l;
170                        TypeNode base = hierarchy.typeNode(((Local) ref.getBase()).getType());
171
172                        if (!base.isArray()) {
173                                error("Type Error(16)");
174                        }
175
176                        left = base.element();
177
178                        Value index = ref.getIndex();
179
180                        if (index instanceof Local) {
181                                if (!hierarchy.typeNode(((Local) index).getType()).hasAncestorOrSelf(hierarchy.typeNode(IntType.v()))) {
182                                        error("Type Error(17)");
183                                }
184                        }
185                } else if (l instanceof Local) {
186                        try {
187                                left = hierarchy.typeNode(((Local) l).getType());
188                        } catch (InternalTypingException e) {
189                                G.v().out.println("untyped local: " + l);
190                                throw e;
191                        }
192                } else if (l instanceof InstanceFieldRef) {
193                        InstanceFieldRef ref = (InstanceFieldRef) l;
194
195                        TypeNode base = hierarchy.typeNode(((Local) ref.getBase()).getType());
196
197                        if (!base.hasAncestorOrSelf(hierarchy.typeNode(ref.getField().getDeclaringClass().getType()))) {
198                                if (fix) {
199                                        ref.setBase(insertCast((Local) ref.getBase(), ref.getField().getDeclaringClass().getType(), stmt));
200                                } else {
201                                        error("Type Error(18)");
202                                }
203                        }
204
205                        left = hierarchy.typeNode(ref.getField().getType());
206                } else if (l instanceof StaticFieldRef) {
207                        StaticFieldRef ref = (StaticFieldRef) l;
208                        left = hierarchy.typeNode(ref.getField().getType());
209                } else {
210                        throw new RuntimeException("Unhandled assignment left hand side type: " + l.getClass());
211                }
212
213                // ******** RIGHT ********
214
215                if (r instanceof ArrayRef) {
216                        ArrayRef ref = (ArrayRef) r;
217                        TypeNode base = hierarchy.typeNode(((Local) ref.getBase()).getType());
218
219                        if (!base.isArray()) {
220                                error("Type Error(19): " + base + " is not an array type");
221                        }
222
223                        if (base == hierarchy.NULL) {
224                                return;
225                        }
226
227                        if (!left.hasDescendantOrSelf(base.element())) {
228                                if (fix) {
229                                        Type lefttype = left.type();
230                                        if (lefttype instanceof ArrayType) {
231                                                ArrayType atype = (ArrayType) lefttype;
232                                                ref.setBase(insertCast((Local) ref.getBase(),
233                                                                ArrayType.v(atype.baseType, atype.numDimensions + 1), stmt));
234                                        } else {
235                                                ref.setBase(insertCast((Local) ref.getBase(), ArrayType.v(lefttype, 1), stmt));
236                                        }
237                                } else {
238                                        error("Type Error(20)");
239                                }
240                        }
241
242                        Value index = ref.getIndex();
243
244                        if (index instanceof Local) {
245                                if (!hierarchy.typeNode(((Local) index).getType()).hasAncestorOrSelf(hierarchy.typeNode(IntType.v()))) {
246                                        error("Type Error(21)");
247                                }
248                        }
249                } else if (r instanceof DoubleConstant) {
250                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(DoubleType.v()))) {
251                                error("Type Error(22)");
252                        }
253                } else if (r instanceof FloatConstant) {
254                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(FloatType.v()))) {
255                                error("Type Error(45)");
256                        }
257                } else if (r instanceof IntConstant) {
258                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(IntType.v()))) {
259                                error("Type Error(23)");
260                        }
261                } else if (r instanceof LongConstant) {
262                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(LongType.v()))) {
263                                error("Type Error(24)");
264                        }
265                } else if (r instanceof NullConstant) {
266                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(NullType.v()))) {
267                                error("Type Error(25)");
268                        }
269                } else if (r instanceof StringConstant) {
270                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(RefType.v("java.lang.String")))) {
271                                error("Type Error(26)");
272                        }
273                } else if (r instanceof ClassConstant) {
274                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(RefType.v("java.lang.Class")))) {
275                                error("Type Error(27)");
276                        }
277                } else if (r instanceof BinopExpr) {
278                        // ******** BINOP EXPR ********
279
280                        BinopExpr be = (BinopExpr) r;
281
282                        Value lv = be.getOp1();
283                        Value rv = be.getOp2();
284
285                        TypeNode lop;
286                        TypeNode rop;
287
288                        // ******** LEFT ********
289                        if (lv instanceof Local) {
290                                lop = hierarchy.typeNode(((Local) lv).getType());
291                        } else if (lv instanceof DoubleConstant) {
292                                lop = hierarchy.typeNode(DoubleType.v());
293                        } else if (lv instanceof FloatConstant) {
294                                lop = hierarchy.typeNode(FloatType.v());
295                        } else if (lv instanceof IntConstant) {
296                                lop = hierarchy.typeNode(IntType.v());
297                        } else if (lv instanceof LongConstant) {
298                                lop = hierarchy.typeNode(LongType.v());
299                        } else if (lv instanceof NullConstant) {
300                                lop = hierarchy.typeNode(NullType.v());
301                        } else if (lv instanceof StringConstant) {
302                                lop = hierarchy.typeNode(RefType.v("java.lang.String"));
303                        } else if (lv instanceof ClassConstant) {
304                                lop = hierarchy.typeNode(RefType.v("java.lang.Class"));
305                        } else {
306                                throw new RuntimeException("Unhandled binary expression left operand type: " + lv.getClass());
307                        }
308
309                        // ******** RIGHT ********
310                        if (rv instanceof Local) {
311                                rop = hierarchy.typeNode(((Local) rv).getType());
312                        } else if (rv instanceof DoubleConstant) {
313                                rop = hierarchy.typeNode(DoubleType.v());
314                        } else if (rv instanceof FloatConstant) {
315                                rop = hierarchy.typeNode(FloatType.v());
316                        } else if (rv instanceof IntConstant) {
317                                rop = hierarchy.typeNode(IntType.v());
318                        } else if (rv instanceof LongConstant) {
319                                rop = hierarchy.typeNode(LongType.v());
320                        } else if (rv instanceof NullConstant) {
321                                rop = hierarchy.typeNode(NullType.v());
322                        } else if (rv instanceof StringConstant) {
323                                rop = hierarchy.typeNode(RefType.v("java.lang.String"));
324                        } else if (rv instanceof ClassConstant) {
325                                rop = hierarchy.typeNode(RefType.v("java.lang.Class"));
326                        } else {
327                                throw new RuntimeException("Unhandled binary expression right operand type: " + rv.getClass());
328                        }
329
330                        if ((be instanceof AddExpr) || (be instanceof SubExpr) || (be instanceof MulExpr)
331                                        || (be instanceof DivExpr) || (be instanceof RemExpr) || (be instanceof AndExpr)
332                                        || (be instanceof OrExpr) || (be instanceof XorExpr)) {
333                                if (!(left.hasDescendantOrSelf(lop) && left.hasDescendantOrSelf(rop))) {
334                                        error("Type Error(27)");
335                                }
336                        } else if ((be instanceof ShlExpr) || (be instanceof ShrExpr) || (be instanceof UshrExpr)) {
337                                if (!(left.hasDescendantOrSelf(lop) && hierarchy.typeNode(IntType.v()).hasAncestorOrSelf(rop))) {
338                                        error("Type Error(28)");
339                                }
340                        } else if ((be instanceof CmpExpr) || (be instanceof CmpgExpr) || (be instanceof CmplExpr)
341                                        || (be instanceof EqExpr) || (be instanceof GeExpr) || (be instanceof GtExpr)
342                                        || (be instanceof LeExpr) || (be instanceof LtExpr) || (be instanceof NeExpr)) {
343                                try {
344                                        lop.lca(rop);
345                                } catch (TypeException e) {
346                                        error(e.getMessage());
347                                }
348
349                                if (!left.hasDescendantOrSelf(hierarchy.typeNode(IntType.v()))) {
350                                        error("Type Error(29)");
351                                }
352                        } else {
353                                throw new RuntimeException("Unhandled binary expression type: " + be.getClass());
354                        }
355                } else if (r instanceof CastExpr) {
356                        CastExpr ce = (CastExpr) r;
357                        TypeNode cast = hierarchy.typeNode(ce.getCastType());
358                        if (ce.getOp() instanceof Local) {
359                                TypeNode op = hierarchy.typeNode(((Local) ce.getOp()).getType());
360
361                                try {
362                                        // we must be careful not to reject primitive type casts
363                                        // (e.g. int to long)
364                                        if (cast.isClassOrInterface() || op.isClassOrInterface()) {
365                                                cast.lca(op);
366                                        }
367                                } catch (TypeException e) {
368                                        G.v().out.println(r + "[" + op + "<->" + cast + "]");
369                                        error(e.getMessage());
370                                }
371                        }
372
373                        if (!left.hasDescendantOrSelf(cast)) {
374                                error("Type Error(30)");
375                        }
376                } else if (r instanceof InstanceOfExpr) {
377                        InstanceOfExpr ioe = (InstanceOfExpr) r;
378                        TypeNode type = hierarchy.typeNode(ioe.getCheckType());
379                        TypeNode op = hierarchy.typeNode(ioe.getOp().getType());
380
381                        try {
382                                op.lca(type);
383                        } catch (TypeException e) {
384                                G.v().out.println(r + "[" + op + "<->" + type + "]");
385                                error(e.getMessage());
386                        }
387
388                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(IntType.v()))) {
389                                error("Type Error(31)");
390                        }
391                } else if (r instanceof InvokeExpr) {
392                        InvokeExpr ie = (InvokeExpr) r;
393
394                        handleInvokeExpr(ie, stmt);
395
396                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(ie.getMethodRef().returnType()))) {
397                                error("Type Error(32)");
398                        }
399                } else if (r instanceof NewArrayExpr) {
400                        NewArrayExpr nae = (NewArrayExpr) r;
401
402                        Type baseType = nae.getBaseType();
403                        TypeNode right;
404
405                        if (baseType instanceof ArrayType) {
406                                right = hierarchy.typeNode(ArrayType.v(((ArrayType) baseType).baseType,
407                                                ((ArrayType) baseType).numDimensions + 1));
408                        } else {
409                                right = hierarchy.typeNode(ArrayType.v(baseType, 1));
410                        }
411
412                        if (!left.hasDescendantOrSelf(right)) {
413                                error("Type Error(33)");
414                        }
415
416                        Value size = nae.getSize();
417                        if (size instanceof Local) {
418                                TypeNode var = hierarchy.typeNode(((Local) size).getType());
419
420                                if (!var.hasAncestorOrSelf(hierarchy.typeNode(IntType.v()))) {
421                                        error("Type Error(34)");
422                                }
423                        }
424                } else if (r instanceof NewExpr) {
425                        NewExpr ne = (NewExpr) r;
426
427                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(ne.getBaseType()))) {
428                                error("Type Error(35)");
429                        }
430                } else if (r instanceof NewMultiArrayExpr) {
431                        NewMultiArrayExpr nmae = (NewMultiArrayExpr) r;
432
433                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(nmae.getBaseType()))) {
434                                error("Type Error(36)");
435                        }
436
437                        for (int i = 0; i < nmae.getSizeCount(); i++) {
438                                Value size = nmae.getSize(i);
439                                if (size instanceof Local) {
440                                        TypeNode var = hierarchy.typeNode(((Local) size).getType());
441
442                                        if (!var.hasAncestorOrSelf(hierarchy.typeNode(IntType.v()))) {
443                                                error("Type Error(37)");
444                                        }
445                                }
446                        }
447                } else if (r instanceof LengthExpr) {
448                        LengthExpr le = (LengthExpr) r;
449
450                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(IntType.v()))) {
451                                error("Type Error(38)");
452                        }
453
454                        if (le.getOp() instanceof Local) {
455                                if (!hierarchy.typeNode(((Local) le.getOp()).getType()).isArray()) {
456                                        error("Type Error(39)");
457                                }
458                        }
459                } else if (r instanceof NegExpr) {
460                        NegExpr ne = (NegExpr) r;
461                        TypeNode right;
462
463                        if (ne.getOp() instanceof Local) {
464                                right = hierarchy.typeNode(((Local) ne.getOp()).getType());
465                        } else if (ne.getOp() instanceof DoubleConstant) {
466                                right = hierarchy.typeNode(DoubleType.v());
467                        } else if (ne.getOp() instanceof FloatConstant) {
468                                right = hierarchy.typeNode(FloatType.v());
469                        } else if (ne.getOp() instanceof IntConstant) {
470                                right = hierarchy.typeNode(IntType.v());
471                        } else if (ne.getOp() instanceof LongConstant) {
472                                right = hierarchy.typeNode(LongType.v());
473                        } else {
474                                throw new RuntimeException("Unhandled neg expression operand type: " + ne.getOp().getClass());
475                        }
476
477                        if (!left.hasDescendantOrSelf(right)) {
478                                error("Type Error(40)");
479                        }
480                } else if (r instanceof Local) {
481                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(((Local) r).getType()))) {
482                                if (fix) {
483                                        stmt.setRightOp(insertCast((Local) r, left.type(), stmt));
484                                } else {
485                                        error("Type Error(41)");
486                                }
487                        }
488                } else if (r instanceof InstanceFieldRef) {
489                        InstanceFieldRef ref = (InstanceFieldRef) r;
490
491                        TypeNode baseType = hierarchy.typeNode(((Local) ref.getBase()).getType());
492                        if (!baseType.hasAncestorOrSelf(hierarchy.typeNode(ref.getField().getDeclaringClass().getType()))) {
493                                if (fix) {
494                                        ref.setBase(insertCast((Local) ref.getBase(), ref.getField().getDeclaringClass().getType(), stmt));
495                                } else {
496                                        error("Type Error(42)");
497                                }
498                        }
499
500                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(ref.getField().getType()))) {
501                                error("Type Error(43)");
502                        }
503                } else if (r instanceof StaticFieldRef) {
504                        StaticFieldRef ref = (StaticFieldRef) r;
505
506                        if (!left.hasDescendantOrSelf(hierarchy.typeNode(ref.getField().getType()))) {
507                                error("Type Error(44)");
508                        }
509                } else {
510                        throw new RuntimeException("Unhandled assignment right hand side type: " + r.getClass());
511                }
512        }

514        public void caseIdentityStmt(IdentityStmt stmt) {
515                TypeNode left = hierarchy.typeNode(((Local) stmt.getLeftOp()).getType());
516
517                Value r = stmt.getRightOp();
518
519                if (!(r instanceof CaughtExceptionRef)) {
520                        TypeNode right = hierarchy.typeNode(r.getType());
521                        if (!left.hasDescendantOrSelf(right)) {
522                                error("Type Error(46) [" + left + " <- " + right + "]");
523                        }
524                } else {
525                        List<RefType> exceptionTypes = TrapManager.getExceptionTypesOf(stmt, stmtBody);
526                        Iterator<RefType> typeIt = exceptionTypes.iterator();
527
528                        while (typeIt.hasNext()) {
529                                Type t = typeIt.next();
530
531                                if (!left.hasDescendantOrSelf(hierarchy.typeNode(t))) {
532                                        error("Type Error(47)");
533                                }
534                        }
535
536                        if (!left.hasAncestorOrSelf(hierarchy.typeNode(RefType.v("java.lang.Throwable")))) {
537                                error("Type Error(48)");
538                        }
539                }
540        }

565        public void caseIfStmt(IfStmt stmt) {
566                ConditionExpr cond = (ConditionExpr) stmt.getCondition();
567
568                BinopExpr expr = cond;
569                Value lv = expr.getOp1();
570                Value rv = expr.getOp2();
571
572                TypeNode lop;
573                TypeNode rop;
574
575                // ******** LEFT ********
576                if (lv instanceof Local) {
577                        lop = hierarchy.typeNode(((Local) lv).getType());
578                } else if (lv instanceof DoubleConstant) {
579                        lop = hierarchy.typeNode(DoubleType.v());
580                } else if (lv instanceof FloatConstant) {
581                        lop = hierarchy.typeNode(FloatType.v());
582                } else if (lv instanceof IntConstant) {
583                        lop = hierarchy.typeNode(IntType.v());
584                } else if (lv instanceof LongConstant) {
585                        lop = hierarchy.typeNode(LongType.v());
586                } else if (lv instanceof NullConstant) {
587                        lop = hierarchy.typeNode(NullType.v());
588                } else if (lv instanceof StringConstant) {
589                        lop = hierarchy.typeNode(RefType.v("java.lang.String"));
590                } else if (lv instanceof ClassConstant) {
591                        lop = hierarchy.typeNode(RefType.v("java.lang.Class"));
592                } else {
593                        throw new RuntimeException("Unhandled binary expression left operand type: " + lv.getClass());
594                }
595
596                // ******** RIGHT ********
597                if (rv instanceof Local) {
598                        rop = hierarchy.typeNode(((Local) rv).getType());
599                } else if (rv instanceof DoubleConstant) {
600                        rop = hierarchy.typeNode(DoubleType.v());
601                } else if (rv instanceof FloatConstant) {
602                        rop = hierarchy.typeNode(FloatType.v());
603                } else if (rv instanceof IntConstant) {
604                        rop = hierarchy.typeNode(IntType.v());
605                } else if (rv instanceof LongConstant) {
606                        rop = hierarchy.typeNode(LongType.v());
607                } else if (rv instanceof NullConstant) {
608                        rop = hierarchy.typeNode(NullType.v());
609                } else if (rv instanceof StringConstant) {
610                        rop = hierarchy.typeNode(RefType.v("java.lang.String"));
611                } else if (rv instanceof ClassConstant) {
612                        rop = hierarchy.typeNode(RefType.v("java.lang.Class"));
613                } else {
614                        throw new RuntimeException("Unhandled binary expression right operand type: " + rv.getClass());
615                }
616
617                try {
618                        lop.lca(rop);
619                } catch (TypeException e) {
620                        error(e.getMessage());
621                }
622        }

624        public void caseLookupSwitchStmt(LookupSwitchStmt stmt) {
625                Value key = stmt.getKey();
626
627                if (key instanceof Local) {
628                        if (!hierarchy.typeNode(((Local) key).getType()).hasAncestorOrSelf(hierarchy.typeNode(IntType.v()))) {
629                                error("Type Error(50)");
630                        }
631                }
632        }

663        public void caseThrowStmt(ThrowStmt stmt) {
664                if (stmt.getOp() instanceof Local) {

681        private Local insertCast(Local oldlocal, Type type, Stmt stmt) {
682                Local newlocal = Jimple.v().newLocal("tmp", type);
683                stmtBody.getLocals().add(newlocal);
684
685                Unit u = Util.findFirstNonIdentityUnit(stmtBody, stmt);
686                stmtBody.getUnits().insertBefore(Jimple.v().newAssignStmt(newlocal, Jimple.v().newCastExpr(oldlocal, type)), u);
687                return newlocal;
688        }
