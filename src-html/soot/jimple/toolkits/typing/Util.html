001package soot.jimple.toolkits.typing;
002
003import soot.Body;
004import soot.Unit;
005import soot.jimple.IdentityStmt;
006import soot.jimple.Stmt;
007
008public class Util {
009    
010    /**
011     * A new "normal" statement cannot be inserted in the middle of special
012     * "identity statements" (a = @parameter or b = @this in Jimple).
013     * 
014     * This method returns the last "identity statement" of the method.
015     * @param b
016     * @param s
017     * @return
018     */
019    public static Unit findLastIdentityUnit(Body b, Stmt s) {
020        Unit u2 = s;
021        Unit u1 = s;
022        while (u1 instanceof IdentityStmt) {
023            u2 = u1;
024            u1 = b.getUnits().getSuccOf(u1);
025        }
026        return u2;
027    }
028    
029    /**
030     * Returns the first statement after all the "identity statements".
031     * @param b
032     * @param s
033     * @return
034     */
035    public static Unit findFirstNonIdentityUnit(Body b, Stmt s) {
036        Unit u1 = s;
037        while (u1 instanceof IdentityStmt)
038            u1 = b.getUnits().getSuccOf(u1);
039        return u1;
040    }
041
042}
