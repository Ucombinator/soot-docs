090/**
091 * This checks all uses against the rules in Jimple, except some uses are not
092 * checked where the bytecode verifier guarantees use validity.
093 * @author Ben Bellamy
094 */
095public class UseChecker extends AbstractStmtSwitch
096{
097        private JimpleBody jb;
098
099        private Typing tg;
100        private IUseVisitor uv;
101        
102        private LocalDefs defs = null;
103        private LocalUses uses = null;
104        
105        public UseChecker(JimpleBody jb)
106        {
107                this.jb = jb;
108        }
109
110        public void check(Typing tg, IUseVisitor uv)
111        {
112                this.tg = tg;
113                this.uv = uv;
114                if (this.tg == null)
115                        throw new RuntimeException("null typing passed to useChecker");
116
117                for ( Iterator<Unit> i = this.jb.getUnits().snapshotIterator();
118                        i.hasNext(); )
119                {
120                        if ( uv.finish() )
121                                return;
122                        i.next().apply(this);
123                }
124        }
125
126        private void handleInvokeExpr(InvokeExpr ie, Stmt stmt)
127        {
128                SootMethodRef m = ie.getMethodRef();
129
130                if ( ie instanceof InstanceInvokeExpr )
131                {
132                        InstanceInvokeExpr iie = (InstanceInvokeExpr)ie;
133                        iie.setBase(this.uv.visit(
134                                iie.getBase(),m.declaringClass().getType(), stmt));
135                }
136
137                for ( int i = 0; i < ie.getArgCount(); i++ )
138                        ie.setArg(i, this.uv.visit(
139                                ie.getArg(i), m.parameterType(i), stmt));
140        }
141
142        private void handleBinopExpr(BinopExpr be, Stmt stmt, Type tlhs)
143        {
144                Value opl = be.getOp1(), opr = be.getOp2();
145                Type tl = AugEvalFunction.eval_(this.tg, opl, stmt, this.jb),
146                        tr = AugEvalFunction.eval_(this.tg, opr, stmt, this.jb);
147
148                if ( be instanceof AddExpr
149                        || be instanceof SubExpr
150                        || be instanceof MulExpr
151                        || be instanceof DivExpr
152                        || be instanceof RemExpr
153                        || be instanceof GeExpr
154                        || be instanceof GtExpr
155                        || be instanceof LeExpr
156                        || be instanceof LtExpr
157                        || be instanceof ShlExpr
158                        || be instanceof ShrExpr
159                        || be instanceof UshrExpr )
160                {
161                        if ( tlhs instanceof IntegerType )
162                        {
163                                be.setOp1(this.uv.visit(opl, IntType.v(), stmt));
164                                be.setOp2(this.uv.visit(opr, IntType.v(), stmt));
165                        }
166                }
167                else if ( be instanceof CmpExpr
168                        || be instanceof CmpgExpr
169                        || be instanceof CmplExpr )
170                {
171                        // No checks in the original assigner
172                }
173                else if ( be instanceof AndExpr
174                        || be instanceof OrExpr
175                        || be instanceof XorExpr )
176                {
177                        be.setOp1(this.uv.visit(opl, tlhs, stmt));
178                        be.setOp2(this.uv.visit(opr, tlhs, stmt));
179                }
180                else if ( be instanceof EqExpr
181                        || be instanceof NeExpr )
182                {
183                        if ( tl instanceof BooleanType && tr instanceof BooleanType )
184                        { }
185                        else if ( tl instanceof Integer1Type || tr instanceof Integer1Type )
186                        { }
187                        else if ( tl instanceof IntegerType )
188                        {
189                                be.setOp1(this.uv.visit(opl, IntType.v(), stmt));
190                                be.setOp2(this.uv.visit(opr, IntType.v(), stmt));
191                        }
192                }
193        }
194
195        private void handleArrayRef(ArrayRef ar, Stmt stmt)
196        {
197                ar.setIndex(this.uv.visit(ar.getIndex(), IntType.v(), stmt));
198        }
199
200        private void handleInstanceFieldRef(InstanceFieldRef ifr, Stmt stmt)
201        {
202                ifr.setBase(this.uv.visit(ifr.getBase(),
203                        ifr.getFieldRef().declaringClass().getType(), stmt));
204        }

209        {

213        public void caseAssignStmt(AssignStmt stmt)
214        {
215                Value lhs = stmt.getLeftOp();
216                Value rhs = stmt.getRightOp();
217                Type tlhs = null;
218
219                if ( lhs instanceof Local )
220                        tlhs = this.tg.get((Local)lhs);
221                else if ( lhs instanceof ArrayRef )
222                {
223                        ArrayRef aref = (ArrayRef) lhs;
224                        Local base = (Local) aref.getBase();
225
226                        // Try to force Type integrity. The left side must agree on the
227                        // element type of the right side array reference.
228                        ArrayType at;
229                        if (this.tg.get(base) instanceof ArrayType)
230                                at = (ArrayType)this.tg.get(base);
231                        else
232                                at = this.tg.get(base).makeArrayType();
233                        tlhs = ((ArrayType)at).getElementType();
234
235                        this.handleArrayRef(aref, stmt);
236
237                        aref.setBase((Local) this.uv.visit(aref.getBase(), at, stmt));
238                        stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
239                        stmt.setLeftOp(this.uv.visit(lhs, tlhs, stmt));
240                }
241                else if ( lhs instanceof FieldRef )
242                {
243                        tlhs = ((FieldRef)lhs).getFieldRef().type();
244                        if ( lhs instanceof InstanceFieldRef )
245                                this.handleInstanceFieldRef((InstanceFieldRef)lhs, stmt);
246                }
247
248                // They may have been changed above
249                lhs = stmt.getLeftOp();
250                rhs = stmt.getRightOp();
251
252                if ( rhs instanceof Local )
253                        stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
254                else if ( rhs instanceof ArrayRef )
255                {
256                        ArrayRef aref = (ArrayRef) rhs;
257                        Local base = (Local) aref.getBase();
258
259                        //try to force Type integrity
260                        ArrayType at;
261                        Type et = null;
262                        if (this.tg.get(base) instanceof ArrayType)
263                                at = (ArrayType)this.tg.get(base);
264                        else {
265                                Type bt = this.tg.get(base);
266                                at = bt.makeArrayType();
267                                
268                                // If we have a type of java.lang.Object and access it like an object,
269                                // this could lead to any kind of object, so we have to look at the uses.
270                                // For some fixed type T, we assume that we can fix the array to T[].
271                                if (bt instanceof RefType) {
272                                        RefType rt = (RefType) bt;
273                                        if (rt.getSootClass().getName().equals("java.lang.Object")
274                                                        || rt.getSootClass().getName().equals("java.io.Serializable")
275                                                        || rt.getSootClass().getName().equals("java.lang.Cloneable")) {
276                                                if (defs == null) {
277                                                defs = LocalDefs.Factory.newLocalDefs(jb);
278                                                        uses = LocalUses.Factory.newLocalUses(jb, defs);
279                                                }
280                                                
281                                                outer: for (UnitValueBoxPair usePair : uses.getUsesOf(stmt)) {
282                                                        Stmt useStmt = (Stmt) usePair.getUnit();
283                                                        if (useStmt.containsInvokeExpr())
284                                                                for (int i = 0; i < useStmt.getInvokeExpr().getArgCount(); i++)
285                                                                        if (useStmt.getInvokeExpr().getArg(i) == usePair.getValueBox().getValue()) {
286                                                                                et = useStmt.getInvokeExpr().getMethod().getParameterType(i);
287                                                                                at = et.makeArrayType();
288                                                                                break outer;
289                                                                        }
290                                                }
291                                        }
292                                }
293                        }
294                        Type trhs = ((ArrayType)at).getElementType();
295
296                        this.handleArrayRef(aref, stmt);
297
298                        aref.setBase((Local) this.uv.visit(aref.getBase(), at, stmt));
299                        stmt.setRightOp(this.uv.visit(rhs, trhs, stmt));
300                }
301                else if ( rhs instanceof InstanceFieldRef )
302                {
303                        this.handleInstanceFieldRef((InstanceFieldRef)rhs, stmt);
304                        stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
305                }
306                else if ( rhs instanceof BinopExpr )
307                        this.handleBinopExpr((BinopExpr)rhs, stmt, tlhs);
308                else if ( rhs instanceof InvokeExpr )
309                {
310                        this.handleInvokeExpr((InvokeExpr)rhs, stmt);
311                        stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
312                }
313                else if ( rhs instanceof CastExpr )
314                        stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
315                else if ( rhs instanceof InstanceOfExpr )
316                {
317                        InstanceOfExpr ioe = (InstanceOfExpr)rhs;
318                        ioe.setOp(this.uv.visit(
319                                ioe.getOp(), RefType.v("java.lang.Object"), stmt));
320                        stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
321                }
322                else if ( rhs instanceof NewArrayExpr )
323                {
324                        NewArrayExpr nae = (NewArrayExpr)rhs;
325                        nae.setSize(this.uv.visit(nae.getSize(), IntType.v(), stmt));
326                        stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
327                }
328                else if ( rhs instanceof NewMultiArrayExpr )
329                {
330                        NewMultiArrayExpr nmae = (NewMultiArrayExpr)rhs;
331                        for ( int i = 0; i < nmae.getSizeCount(); i++ )
332                                nmae.setSize(i, this.uv.visit(
333                                        nmae.getSize(i), IntType.v(), stmt));
334                        stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
335                }
336                else if ( rhs instanceof LengthExpr )
337                {
338                        stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
339                }
340                else if ( rhs instanceof NegExpr )
341                {
342                        ((NegExpr)rhs).setOp(this.uv.visit(
343                                ((NegExpr)rhs).getOp(), tlhs, stmt));
344                }
345                else if ( rhs instanceof Constant )
346                        if (!(rhs instanceof NullConstant))
347                                stmt.setRightOp(this.uv.visit(rhs, tlhs, stmt));
348        }
