044public class BytecodeHierarchy implements IHierarchy

046        private static class AncestryTreeNode
047        {
048                public final AncestryTreeNode next;
049                public final RefType type;
050                
051                public AncestryTreeNode(AncestryTreeNode next, RefType type)
052                {
053                        this.next = next;
054                        this.type = type;
055                }
056        }
057        
058        /* Returns a collection of nodes, each with type Object, each at the leaf
059        end of a different path from root to Object. */
060        private static Collection<AncestryTreeNode> buildAncestryTree(RefType root)
061        {
062                LinkedList<AncestryTreeNode> leafs = new LinkedList<AncestryTreeNode>();
063                leafs.add(new AncestryTreeNode(null, root));
064                
065                LinkedList<AncestryTreeNode> r = new LinkedList<AncestryTreeNode>();
066                while ( !leafs.isEmpty() )
067                {
068                        AncestryTreeNode node = leafs.remove();
069                        if ( TypeResolver.typesEqual(
070                                node.type, RefType.v("java.lang.Object")) )
071                                r.add(node);
072                        else
073                        {
074                                SootClass sc = node.type.getSootClass();
075                                
076                                for ( Iterator<SootClass> i = sc.getInterfaces().iterator(); i.hasNext(); )
077                                        leafs.add(new AncestryTreeNode(
078                                                node, (i.next()).getType()));
079                                
080                                // The superclass of all interfaces is Object
081                                // -- try to discard phantom interfaces.
082                                if ( ( !sc.isInterface() || sc.getInterfaceCount() == 0 ) && !sc.isPhantom())
083                                        leafs.add(new AncestryTreeNode(
084                                                node, sc.getSuperclass().getType()));
085                                
086                        }
087                }
088                return r;
089        }
090        
091        private static RefType leastCommonNode(
092                AncestryTreeNode a, AncestryTreeNode b)
093        {
094                RefType r = null;
095                while ( a != null && b != null
096                        && TypeResolver.typesEqual(a.type, b.type) )
097                {
098                        r = a.type;
099                        a = a.next;
100                        b = b.next;
101                }
102                return r;
103        }
104        
105        public Collection<Type> lcas(Type a, Type b)
106        {
107                return lcas_(a, b);
108        }
109        
110        public static Collection<Type> lcas_(Type a, Type b)
111        {
112                if ( TypeResolver.typesEqual(a, b) )
113                        return Collections.<Type>singletonList(a);
114                else if ( a instanceof BottomType )
115                        return Collections.<Type>singletonList(b);
116                else if ( b instanceof BottomType )
117                        return Collections.<Type>singletonList(a);
118                else if ( a instanceof IntegerType && b instanceof IntegerType )
119                        return Collections.<Type>singletonList(IntType.v());
120                
121                // Implicit type widening: Integer+Float -> Float
122                else if ( a instanceof IntegerType && b instanceof FloatType )
123                        return Collections.<Type>singletonList(FloatType.v());
124                else if ( b instanceof IntegerType && a instanceof FloatType )
125                        return Collections.<Type>singletonList(FloatType.v());
126                
127                // Disallow type sharing for primitives in general 
128                else if ( a instanceof PrimType || b instanceof PrimType )
129                        return Collections.<Type>emptyList();
130                
131                // Null reference handling
132                else if ( a instanceof NullType )
133                        return Collections.<Type>singletonList(b);
134                else if ( b instanceof NullType )
135                        return Collections.<Type>singletonList(a);
136                
137                // a and b are both ArrayType or RefType
138                else if ( a instanceof ArrayType && b instanceof ArrayType )
139                {
140                        Type eta = ((ArrayType)a).getElementType(),
141                                etb = ((ArrayType)b).getElementType();
142                        Collection<Type> ts;
143                        
144                        // Primitive arrays are not covariant but all other arrays are
145                        if ( eta instanceof PrimType || etb instanceof PrimType )
146                                ts = Collections.<Type>emptyList();
147                        else
148                                ts = lcas_(eta, etb);
149                        
150                        LinkedList<Type> r = new LinkedList<Type>();
151                        if ( ts.isEmpty() )
152                        {
153                                // From Java Language Spec 2nd ed., Chapter 10, Arrays
154                                r.add(RefType.v("java.lang.Object"));
155                                r.add(RefType.v("java.io.Serializable"));
156                                r.add(RefType.v("java.lang.Cloneable"));
157                        }
158                        else
159                                for ( Type t : ts )
160                                        r.add(t.makeArrayType());
161                        return r;
162                }
163                else if ( a instanceof ArrayType || b instanceof ArrayType )
164                {
165                        Type rt;
166                        if ( a instanceof ArrayType )
167                                rt = b;
168                        else
169                                rt = a;
170                        
171                        /* If the reference type implements Serializable or Cloneable then 
172                        these are the least common supertypes, otherwise the only one is 
173                        Object. */
174                        
175                        LinkedList<Type> r = new LinkedList<Type>();
176                        /* Do not consider Object to be a subtype of Serializable or Cloneable
177                        (it can appear this way if phantom-refs is enabled and rt.jar is not
178                        available) otherwise an infinite loop can result. */
179                        if (!TypeResolver.typesEqual(RefType.v("java.lang.Object"), rt)) {
180                            if ( ancestor_(RefType.v("java.io.Serializable"), rt) )
181                                r.add(RefType.v("java.io.Serializable"));
182                            if ( ancestor_(RefType.v("java.lang.Cloneable"), rt) )
183                                r.add(RefType.v("java.lang.Cloneable"));
184                        }
185                        
186                        if ( r.isEmpty() )
187                                r.add(RefType.v("java.lang.Object"));
188                        return r;
189                }
190                // a and b are both RefType
191                else
192                {
193                        Collection<AncestryTreeNode> treea = buildAncestryTree((RefType)a),
194                                treeb = buildAncestryTree((RefType)b);
195                        
196                        LinkedList<Type> r = new LinkedList<Type>();
197                        for ( AncestryTreeNode nodea : treea )
198                                for ( AncestryTreeNode nodeb : treeb )
199                                {
200                                        RefType t = leastCommonNode(nodea, nodeb);
201                                        
202                                        boolean least = true;
203                                        for ( ListIterator<Type> i = r.listIterator(); i.hasNext(); )
204                                        {
205                                                Type t_ = i.next();
206                                                
207                                                if ( ancestor_(t, t_) )
208                                                {
209                                                        least = false;
210                                                        break;
211                                                }
212                                                
213                                                if ( ancestor_(t_, t) )
214                                                        i.remove();
215                                        }
216                                        
217                                        if ( least )
218                                                r.add(t);
219                                }
220                        
221                        //in case of phantom classes that screw up type resolution here,
222                        //default to only possible common reftype, java.lang.Object
223                        //kludge on a kludge on a kludge...
224                        //syed - 05/06/2009
225                        if ( r.isEmpty() )
226                                r.add(RefType.v("java.lang.Object"));
227                        return r;
228                }
229        }
230        
231        public boolean ancestor(Type ancestor, Type child)
232        {
233                return ancestor_(ancestor, child);
234        }
235        
236        public static boolean ancestor_(Type ancestor, Type child)
237        {
238                if ( TypeResolver.typesEqual(ancestor, child) )
239                        return true;
240                else if ( child instanceof BottomType )
241                        return true;
242                else if ( ancestor instanceof BottomType )
243                        return false;
244                else if ( ancestor instanceof IntegerType
245                        && child instanceof IntegerType )
246                        return true;
247                else if ( ancestor instanceof PrimType || child instanceof PrimType )
248                        return false;
249                else if ( child instanceof NullType )
250                        return true;
251                else if ( ancestor instanceof NullType )
252                        return false;
253                else return Scene.v().getOrMakeFastHierarchy().canStoreType(
254                        child, ancestor);
255        }
256        
257        private static Deque<RefType> superclassPath(RefType t)
258        {
259                Deque<RefType> r = new LinkedList<RefType>();
260                r.addFirst(t);
261                
262                SootClass sc = t.getSootClass();
263                while ( sc.hasSuperclass() )
264                {
265                        sc = sc.getSuperclass();
266                        r.addFirst(sc.getType());
267                }
268                
269                return r;
270        }
271        
272        public static RefType lcsc(RefType a, RefType b)
273        {
274                if (a == b)
275                        return a;
276                
277                Deque<RefType> pathA = superclassPath(a);
278                Deque<RefType> pathB = superclassPath(b);
279                RefType r = null;
280                while ( !(pathA.isEmpty() || pathB.isEmpty()) 
281                        && TypeResolver.typesEqual(pathA.getFirst(), pathB.getFirst()) )
282                {
283                        r = pathA.removeFirst();
284                        pathB.removeFirst();
285                }
286                return r;
287        }
