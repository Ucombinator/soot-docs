021package soot.jimple.toolkits.typing.fast;
022
023import java.util.*;
024import soot.*;
025
026/**
027 * @author Ben Bellamy
028 */
029public class Typing
030{
031        private HashMap<Local, Type> map;
032        
033        public Typing(Collection<Local> vs)
034        {
035                this.map = new HashMap<Local, Type>();
036                for ( Local v : vs )
037                        this.map.put(v, BottomType.v());
038        }
039        
040        public Typing(Typing tg)
041        {
042                this.map = new HashMap<Local, Type>(tg.map);
043        }
044        
045        public Type get(Local v) { return this.map.get(v); }
046        
047        public Type set(Local v, Type t) { return this.map.put(v, t); }
048        
049        public String toString()
050        {
051                StringBuffer sb = new StringBuffer();
052                sb.append('{');
053                for ( Local v : this.map.keySet() )
054                {
055                        sb.append(v);
056                        sb.append(':');
057                        sb.append(this.get(v));
058                        sb.append(',');
059                }
060                sb.append('}');
061                return sb.toString();
062        }
063        
064        public static void minimize(List<Typing> tgs, IHierarchy h)
065        {
066                outer: for ( ListIterator<Typing> i = tgs.listIterator(); i.hasNext(); )
067                {
068                        Typing tgi = i.next();
069
070                        // Throw out duplicate typings
071                        for ( Typing tgj : tgs ) {
072                                // if compare = 1, then tgi is the more general typing
073                                // We shouldn't pick that one as we would then end up
074                                // with lots of locals typed to Serializable etc.
075                                if ( tgi != tgj && compare(tgi, tgj, h) == 1 ) {
076                                        i.remove();
077                                        continue outer;
078                                }
079                        }
080                }
081                
082        }
083        
084        public static int compare(Typing a, Typing b, IHierarchy h)
085        {
086                int r = 0;
087                for ( Local v : a.map.keySet() )
088                {
089                        Type ta = a.get(v), tb = b.get(v);
090                        
091                        int cmp;
092                        if ( TypeResolver.typesEqual(ta, tb) )
093                                cmp = 0;
094                        else if ( h.ancestor(ta, tb) )
095                                cmp = 1;
096                        else if ( h.ancestor(tb, ta) )
097                                cmp = -1;
098                        else
099                                return -2;
100                        
101                        if ( (cmp == 1 && r == -1) || (cmp == -1 && r == 1) )
102                                return 2;
103                        if ( r == 0 )
104                                r = cmp;
105                }
106                return r;
107        }
108}
