031/** Represents a type variable. **/

038  private int rank = 0;

061  public int hashCode()
062  {
063    if(rep != this)
064      {
065        return ecr().hashCode();
066      }
067   
068    return id;
069  }
070
071  public boolean equals(Object obj)
072  {
073    if(rep != this)
074      {
075        return ecr().equals(obj);
076      }
077
078    if(obj == null)
079      {
080        return false;
081      }
082
083    if(!obj.getClass().equals(getClass()))
084      {
085        return false;
086      }
087    
088    TypeVariable ecr = ((TypeVariable) obj).ecr();
089
090    if(ecr != this)
091      {
092        return false;
093      }
094    
095    return true;
096  }
097
098  public int compareTo(Object o)
099  {
100    if(rep != this)
101      {
102        return ecr().compareTo(o);
103      }
104
105    return id - ((TypeVariable) o).ecr().id;
106  }

108  private TypeVariable ecr()
109  {
110    if(rep != this)
111      {
112        rep = rep.ecr();
113      }
114
115    return rep;
116  }

118  public TypeVariable union(TypeVariable var) throws TypeException
119  {
120    if(rep != this)
121      {
122        return ecr().union(var);
123      }
124
125    TypeVariable y = var.ecr();
126
127    if(this == y)
128      {
129        return this;
130      }
131    
132    if(rank > y.rank)
133      {
134        y.rep = this;
135
136        merge(y);
137        y.clear();
138
139        return this;
140      }
141
142    rep = y;
143    if(rank == y.rank)
144      {
145        y.rank++;
146      }
147
148    y.merge(this);
149    clear();
150
151    return y;
152  }
153
154  private void clear()
155  {
156    inv_approx = null;
157    approx = null;
158    type = null;
159    parents = null;
160    children = null;
161  }
162
163  private void merge(TypeVariable var) throws TypeException
164  {
165    // Merge types
166    if(type == null)
167      {
168        type = var.type;
169      }
170    else if(var.type != null)
171      {
172        error("Type Error(22): Attempt to merge two types.");
173      }
174
175    // Merge parents
176    {
177      Set<TypeVariable> set = new TreeSet<TypeVariable>(parents);
178      set.addAll(var.parents);
179      set.remove(this);
180      parents = Collections.unmodifiableList(new LinkedList<TypeVariable>(set));
181    }
182
183    // Merge children
184    {
185      Set<TypeVariable> set = new TreeSet<TypeVariable>(children);
186      set.addAll(var.children);
187      set.remove(this);
188      children = Collections.unmodifiableList(new LinkedList<TypeVariable>(set));
189    }
190  }

202  public void addParent(TypeVariable variable)
203  {
204    if(rep != this)
205      {
206        ecr().addParent(variable);
207        return;
208      }
209
210    TypeVariable var = variable.ecr();
211 
212    if(var == this)
213      {
214        return;
215      }
216
217    {
218      Set<TypeVariable> set = new TreeSet<TypeVariable>(parents);
219      set.add(var);
220      parents = Collections.unmodifiableList(new LinkedList<TypeVariable>(set));
221    }
222    
223    {
224      Set<TypeVariable> set = new TreeSet<TypeVariable>(var.children);
225      set.add(this);
226      var.children = Collections.unmodifiableList(new LinkedList<TypeVariable>(set));
227    }
228  }

370  /** Computes approximative types.  The work list must be 
371   *  initialized with all constant type variables. */
372  public static void computeApprox(TreeSet<TypeVariable> workList) throws TypeException
373  {
374    while(workList.size() > 0)
375      {
376        TypeVariable var = workList.first();
377        workList.remove(var);
378
379        var.fixApprox(workList);
380      }
381  }
382
383  public static void computeInvApprox(TreeSet<TypeVariable> workList) throws TypeException
384  {
385    while(workList.size() > 0)
386      {
387        TypeVariable var = workList.first();
388        workList.remove(var);
389
390        var.fixInvApprox(workList);
391      }
392  }
393
394  private void fixApprox(TreeSet<TypeVariable> workList) throws TypeException
395  {
396    if(rep != this)
397      {
398        ecr().fixApprox(workList);
399        return;
400      }
401
402    for (TypeVariable typeVariable : parents) {
403        TypeVariable parent = typeVariable.ecr();
404
405        if(parent.approx == null)
406          {
407            parent.approx = approx;
408            workList.add(parent);
409          }
410        else
411          {
412            TypeNode type = parent.approx.lca_2(approx);
413
414            if(type != parent.approx)
415              {
416                parent.approx = type;
417                workList.add(parent);
418              }
419          }
420      }
421
422    if(type != null)
423      {
424        approx = type;
425      }
426  }
427
428  private void fixInvApprox(TreeSet<TypeVariable> workList) throws TypeException
429  {
430    if(rep != this)
431      {
432        ecr().fixInvApprox(workList);
433        return;
434      }
435
436    for (TypeVariable typeVariable : children) {
437        TypeVariable child = typeVariable.ecr();
438
439        if(child.inv_approx == null)
440          {
441            child.inv_approx = inv_approx;
442            workList.add(child);
443          }
444        else
445          {
446            TypeNode type = child.inv_approx.gcd_2(inv_approx);
447
448            if(type != child.inv_approx)
449              {
450                child.inv_approx = type;
451                workList.add(child);
452              }
453          }
454      }
455
456    if(type != null)
457      {
458        inv_approx = type;
459      }
460  }

462  public String toString()
463  {
464    if(rep != this)
465      {
466        return ecr().toString();
467      }
468    
469    StringBuffer s = new StringBuffer();
470    s.append(",[parents:");
471
472    {
473      boolean comma = false;
474      
475      for (TypeVariable typeVariable : parents) {
476          if(comma)
477            {
478              s.append(",");
479            }
480          else
481            {
482              comma = true;
483            }
484          s.append(typeVariable.id());
485        }
486    }
487    
488    s.append("],[children:");
489
490    {
491      boolean comma = false;
492      
493      for (TypeVariable typeVariable : children) {
494          if(comma)
495            {
496              s.append(",");
497            }
498          else
499            {
500              comma = true;
501            }
502          s.append(typeVariable.id());
503        }
504    }
505    
506    s.append("]");
507    return "[id:" + id + ((type != null) ? (",type:" + type) : "") + ",approx:" + approx + ",inv_approx:" + inv_approx + s + "]";
508  }

510  public void fixParents()
511  {
512    if(rep != this)
513      {
514        ecr().fixParents();
515        return;
516      }
517
518    {
519      Set<TypeVariable> set = new TreeSet<TypeVariable>(parents);
520      parents = Collections.unmodifiableList(new LinkedList<TypeVariable>(set));
521    }
522  }

529        return;
530      }
531
532    {

534      children = Collections.unmodifiableList(new LinkedList<TypeVariable>(set));
535    }
