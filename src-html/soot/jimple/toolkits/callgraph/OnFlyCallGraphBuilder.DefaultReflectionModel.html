091                public void classForName(SootMethod source, Stmt s) {
092                List<Local> stringConstants = methodToStringConstants.get(source);
093                if( stringConstants == null )
094                    methodToStringConstants.put(source, stringConstants = new ArrayList<Local>());
095                        InvokeExpr ie = s.getInvokeExpr();
096                Value className = ie.getArg(0);
097                if( className instanceof StringConstant ) {
098                    String cls = ((StringConstant) className ).value;
099                    constantForName( cls, source, s );
100                } else {
101                    Local constant = (Local) className;
102                    if( options.safe_forname() ) {
103                        for (SootMethod tgt : EntryPoints.v().clinits()) {
104                            addEdge( source, s, tgt, Kind.CLINIT );
105                        }
106                    } else {
107                        for (SootClass cls : Scene.v().dynamicClasses()) {
108                            for (SootMethod clinit : EntryPoints.v().clinitsOf(cls)) {
109                                addEdge( source, s, clinit, Kind.CLINIT);
110                            }
111                        }
112                        VirtualCallSite site = new VirtualCallSite( s, source, null, null, Kind.CLINIT );
113                        List<VirtualCallSite> sites = stringConstToSites.get(constant);
114                        if (sites == null) {
115                            stringConstToSites.put(constant, sites = new ArrayList<VirtualCallSite>());
116                            stringConstants.add(constant);
117                        }
118                        sites.add(site);
119                    }
120                }        
121                }

123                public void classNewInstance(SootMethod source, Stmt s) {
124                        if( options.safe_newinstance() ) {
125                                for (SootMethod tgt : EntryPoints.v().inits()) {
126                                        addEdge( source, s, tgt, Kind.NEWINSTANCE );
127                                }
128                        } else {
129                                for (SootClass cls : Scene.v().dynamicClasses()) {
130                                        SootMethod sm = cls.getMethodUnsafe(sigInit);
131                                        if( sm != null ) {
132                                                addEdge( source, s, sm, Kind.NEWINSTANCE );
133                                        }
134                                }
135
136                                if( options.verbose() ) {
137                                        G.v().out.println( "Warning: Method "+source+
138                                                        " is reachable, and calls Class.newInstance;"+
139                                                        " graph will be incomplete!"+
140                                        " Use safe-newinstance option for a conservative result." );
141                                }
142                        } 
143                }

145                public void contructorNewInstance(SootMethod source, Stmt s) {
146                        if( options.safe_newinstance() ) {
147                                for (SootMethod tgt : EntryPoints.v().allInits()) {
148                                        addEdge( source, s, tgt, Kind.NEWINSTANCE );
149                                }
150                        } else {
151                                for (SootClass cls : Scene.v().dynamicClasses()) {
152                                        for(SootMethod m: cls.getMethods()) {
153                                                if(m.getName().equals("<init>")) {
154                                                        addEdge( source, s, m, Kind.NEWINSTANCE );
155                                                }
156                                        }
157                                }
158                                if( options.verbose() ) {
159                                        G.v().out.println( "Warning: Method "+source+
160                                                        " is reachable, and calls Constructor.newInstance;"+
161                                                        " graph will be incomplete!"+
162                                        " Use safe-newinstance option for a conservative result." );
163                                }
164                        } 
165                }

167                public void methodInvoke(SootMethod container, Stmt invokeStmt) {
168                        if( !warnedAlready(container) ) {
169                                if( options.verbose() ) {
170                                        G.v().out.println( "Warning: call to "+
171                                                        "java.lang.reflect.Method: invoke() from "+container+
172                                        "; graph will be incomplete!" );
173                                }
174                                markWarned(container);
175                        }
176                }

178                private void markWarned(SootMethod m) {
179                        warnedAlready.add(m);
180                }

182                private boolean warnedAlready(SootMethod m) {
183                        return warnedAlready.contains(m);
184                }

219                /**
220                 * Adds an edge to all class initializers of all possible receivers
221                 * of Class.forName() calls within source.
222                 */
223                public void classForName(SootMethod container, Stmt forNameInvokeStmt) {
224                        Set<String> classNames = reflectionInfo.classForNameClassNames(container);
225                        if(classNames==null || classNames.isEmpty()) {
226                                registerGuard(container, forNameInvokeStmt, "Class.forName() call site; Soot did not expect this site to be reached");
227                        } else {
228                                for (String clsName : classNames) {
229                            constantForName( clsName, container, forNameInvokeStmt );
230                                }
231                        }
232                }
233
234                /**
235                 * Adds an edge to the constructor of the target class from this call to
236                 * {@link Class#newInstance()}.
237                 */
238                public void classNewInstance(SootMethod container, Stmt newInstanceInvokeStmt) {
239                        Set<String> classNames = reflectionInfo.classNewInstanceClassNames(container);
240                        if(classNames==null || classNames.isEmpty()) {
241                                registerGuard(container, newInstanceInvokeStmt, "Class.newInstance() call site; Soot did not expect this site to be reached");
242                        } else {
243                                for (String clsName : classNames) {
244                                        SootClass cls = Scene.v().getSootClass(clsName);
245                                        SootMethod constructor = cls.getMethodUnsafe(sigInit);
246                                        if( constructor != null ) {
247                                                addEdge( container, newInstanceInvokeStmt, constructor, Kind.REFL_CLASS_NEWINSTANCE );
248                                        }
249                                }
250                        }
251                }
252
253                /** 
254                 * Adds a special edge of kind {@link Kind#REFL_CONSTR_NEWINSTANCE} to all possible target constructors
255                 * of this call to {@link Constructor#newInstance(Object...)}.
256                 * Those kinds of edges are treated specially in terms of how parameters are assigned,
257                 * as parameters to the reflective call are passed into the argument array of
258                 * {@link Constructor#newInstance(Object...)}.
259                 * @see PAG#addCallTarget(Edge) 
260                 */
261                public void contructorNewInstance(SootMethod container, Stmt newInstanceInvokeStmt) {
262                        Set<String> constructorSignatures = reflectionInfo.constructorNewInstanceSignatures(container);
263                        if(constructorSignatures==null || constructorSignatures.isEmpty()) {
264                                registerGuard(container, newInstanceInvokeStmt, "Constructor.newInstance(..) call site; Soot did not expect this site to be reached");
265                        } else {
266                                for (String constructorSignature : constructorSignatures) {
267                                        SootMethod constructor = Scene.v().getMethod(constructorSignature);
268                                        addEdge( container, newInstanceInvokeStmt, constructor, Kind.REFL_CONSTR_NEWINSTANCE );
269                                }
270                        }
271                }
272
273                /** 
274                 * Adds a special edge of kind {@link Kind#REFL_INVOKE} to all possible target methods
275                 * of this call to {@link Method#invoke(Object, Object...)}.
276                 * Those kinds of edges are treated specially in terms of how parameters are assigned,
277                 * as parameters to the reflective call are passed into the argument array of
278                 * {@link Method#invoke(Object, Object...)}.
279                 * @see PAG#addCallTarget(Edge) 
280                 */
281                public void methodInvoke(SootMethod container, Stmt invokeStmt) {
282                        Set<String> methodSignatures = reflectionInfo.methodInvokeSignatures(container);
283                        if (methodSignatures == null || methodSignatures.isEmpty()) {
284                                registerGuard(container, invokeStmt, "Method.invoke(..) call site; Soot did not expect this site to be reached");
285                        } else {
286                                for (String methodSignature : methodSignatures) {
287                                        SootMethod method = Scene.v().getMethod(methodSignature);
288                                        addEdge( container, invokeStmt, method, Kind.REFL_INVOKE );
289                                }
290                        }
291                }
292
293                private void registerGuard(SootMethod container, Stmt stmt, String string) {
294                        guards.add(new Guard(container,stmt,string));
295
296                        if(options.verbose()) {
297                                G.v().out.println("Incomplete trace file: Class.forName() is called in method '" +
298                                                container+"' but trace contains no information about the receiver class of this call.");
299                                if(options.guards().equals("ignore")) {
300                                        G.v().out.println("Guarding strategy is set to 'ignore'. Will ignore this problem.");
301                                } else if(options.guards().equals("print")) {
302                                        G.v().out.println("Guarding strategy is set to 'print'. " +
303                                                        "Program will print a stack trace if this location is reached during execution.");
304                                } else if(options.guards().equals("throw")) {
305                                        G.v().out.println("Guarding strategy is set to 'throw'. Program will throw an " +
306                                                        "Error if this location is reached during execution.");
307                                } else {
308                                        throw new RuntimeException("Invalid value for phase option (guarding): "+options.guards());
309                                }
310                        }
311                        
312                        if(!registeredTransformation) {
313                                registeredTransformation=true;
314                                PackManager.v().getPack("wjap").add(new Transform("wjap.guards",new SceneTransformer() {
315                                        
316                                        @Override
317                                        protected void internalTransform(String phaseName, Map<String, String> options) {
318                                                for (Guard g : guards) {
319                                                        insertGuard(g);
320                                                }
321                                        }
322                                }));
323                                PhaseOptions.v().setPhaseOption("wjap.guards", "enabled");
324                        }
325                }
326                
327                private void insertGuard(Guard guard) {
328                        if(options.guards().equals("ignore")) return;
329                        
330                        SootMethod container = guard.container;
331                        Stmt insertionPoint = guard.stmt;
332                        if(!container.hasActiveBody()) {
333                                G.v().out.println("WARNING: Tried to insert guard into "+container+" but couldn't because method has no body.");
334                        } else {
335                                Body body = container.getActiveBody();
336                                
337                                //exc = new Error
338                                RefType runtimeExceptionType = RefType.v("java.lang.Error");
339                                NewExpr newExpr = Jimple.v().newNewExpr(runtimeExceptionType);
340                                LocalGenerator lg = new LocalGenerator(body);
341                                Local exceptionLocal = lg.generateLocal(runtimeExceptionType);
342                                AssignStmt assignStmt = Jimple.v().newAssignStmt(exceptionLocal, newExpr);
343                                body.getUnits().insertBefore(assignStmt, insertionPoint);
344                                
345                                //exc.<init>(message)
346                                SootMethodRef cref = runtimeExceptionType.getSootClass().getMethod("<init>", Collections.<Type>singletonList(RefType.v("java.lang.String"))).makeRef();
347                                SpecialInvokeExpr constructorInvokeExpr = Jimple.v().newSpecialInvokeExpr(exceptionLocal, cref, StringConstant.v(guard.message));
348                                InvokeStmt initStmt = Jimple.v().newInvokeStmt(constructorInvokeExpr);
349                                body.getUnits().insertAfter(initStmt, assignStmt);
350                                
351                                if(options.guards().equals("print")) {
352                                        //exc.printStackTrace();
353                                        VirtualInvokeExpr printStackTraceExpr = Jimple.v().newVirtualInvokeExpr(exceptionLocal, Scene.v().getSootClass("java.lang.Throwable").getMethod("printStackTrace", Collections.<Type>emptyList()).makeRef());
354                                        InvokeStmt printStackTraceStmt = Jimple.v().newInvokeStmt(printStackTraceExpr);
355                                        body.getUnits().insertAfter(printStackTraceStmt, initStmt);
356                                } else if(options.guards().equals("throw")) {
357                                        body.getUnits().insertAfter(Jimple.v().newThrowStmt(exceptionLocal), initStmt);
358                                } else {
359                                        throw new RuntimeException("Invalid value for phase option (guarding): "+options.guards());
360                                }
361                        }
362                }

392    public OnFlyCallGraphBuilder( ContextManager cm, ReachableMethods rm ) {
393        this.cm = cm;
394        this.rm = rm;
395        worklist = rm.listener();
396        options = new CGOptions( PhaseOptions.v().getPhaseOptions("cg") );
397        if( !options.verbose() ) {
398            G.v().out.println( "[Call Graph] For information on where the call graph may be incomplete, use the verbose option to the cg phase." );
399        }
400        
401        if(options.reflection_log()==null || options.reflection_log().length()==0) {
402                reflectionModel = new DefaultReflectionModel();
403        } else {
404                reflectionModel = new TraceBasedReflectionModel();
405        }

407    public OnFlyCallGraphBuilder( ContextManager cm, ReachableMethods rm, boolean appOnly ) {
408        this( cm, rm );
409        this.appOnly = appOnly;

411    public void processReachables() {
412        while(true) {
413            if( !worklist.hasNext() ) {
414                rm.update();
415                if( !worklist.hasNext() ) break;
416            }
417            MethodOrMethodContext momc = (MethodOrMethodContext) worklist.next();
418            SootMethod m = momc.method();
419            if( appOnly && !m.getDeclaringClass().isApplicationClass() ) continue;
420            if( analyzedMethods.add( m ) ) processNewMethod( m );
421            processNewMethodContext( momc );
422        }
423    }
424    public boolean wantTypes( Local receiver ) {
425        return receiverToSites.get(receiver) != null;
426    }
427    public void addType( Local receiver, Context srcContext, Type type, Context typeContext ) {
428        FastHierarchy fh = Scene.v().getOrMakeFastHierarchy();
429        for( Iterator<VirtualCallSite> siteIt = receiverToSites.get( receiver ).iterator(); siteIt.hasNext(); ) {
430            final VirtualCallSite site = siteIt.next();
431            if( site.kind() == Kind.THREAD && !fh.canStoreType( type, clRunnable))
432                continue;
433            if( site.kind() == Kind.EXECUTOR && !fh.canStoreType( type, clRunnable))
434                continue;
435            if( site.kind() == Kind.ASYNCTASK && !fh.canStoreType( type, clAsyncTask ))
436                continue;
437
438            if( site.iie() instanceof SpecialInvokeExpr && site.kind != Kind.THREAD
439                        && site.kind != Kind.EXECUTOR
440                        && site.kind != Kind.ASYNCTASK ) {
441                SootMethod target = VirtualCalls.v().resolveSpecial( 
442                            (SpecialInvokeExpr) site.iie(),
443                            site.subSig(),
444                            site.container(),
445                            appOnly );
446                //if the call target resides in a phantom class then "target" will be null;
447                //simply do not add the target in that case
448                if(target!=null) {
449                        targetsQueue.add( target );
450                }
451            } else {
452                VirtualCalls.v().resolve( type,
453                        receiver.getType(),
454                        site.subSig(),
455                        site.container(), 
456                        targetsQueue,
457                        appOnly);
458            }
459            while(targets.hasNext()) {
460                SootMethod target = (SootMethod) targets.next();
461                cm.addVirtualEdge(
462                        MethodContext.v( site.container(), srcContext ),
463                        site.stmt(),
464                        target,
465                        site.kind(),
466                        typeContext );
467            }
468        }
469    }

473    public void addStringConstant( Local l, Context srcContext, String constant ) {
474        for( Iterator<VirtualCallSite> siteIt = (stringConstToSites.get( l )).iterator(); siteIt.hasNext(); ) {
475            final VirtualCallSite site = siteIt.next();
476            if( constant == null ) {
477                if( options.verbose() ) {
478                    G.v().out.println( "Warning: Method "+site.container()+
479                        " is reachable, and calls Class.forName on a"+
480                        " non-constant String; graph will be incomplete!"+
481                        " Use safe-forname option for a conservative result." );
482                }
483            } else {
484                if( constant.length() > 0 && constant.charAt(0) == '[' ) {
485                    if( constant.length() > 1 && constant.charAt(1) == 'L' 
486                    && constant.charAt(constant.length()-1) == ';' ) {
487                        constant = constant.substring(2,constant.length()-1);
488                    } else continue;
489                }
490                if( !Scene.v().containsClass( constant ) ) {
491                    if( options.verbose() ) {
492                        G.v().out.println( "Warning: Class "+constant+" is"+
493                            " a dynamic class, and you did not specify"+
494                            " it as such; graph will be incomplete!" );
495                    }
496                } else {
497                    SootClass sootcls = Scene.v().getSootClass( constant );
498                    if( !sootcls.isApplicationClass() && !sootcls.isPhantom() ) {
499                        sootcls.setLibraryClass();
500                    }
501                    for (SootMethod clinit : EntryPoints.v().clinitsOf(sootcls)) {
502                        cm.addStaticEdge(
503                                MethodContext.v( site.container(), srcContext ),
504                                site.stmt(),
505                                clinit,
506                                Kind.CLINIT );
507                    }
508                }
509            }
510        }

515    private void addVirtualCallSite( Stmt s, SootMethod m, Local receiver,
516            InstanceInvokeExpr iie, NumberedString subSig, Kind kind ) {
517        List<VirtualCallSite> sites = (List<VirtualCallSite>) receiverToSites.get(receiver);
518        if (sites == null) {
519            receiverToSites.put(receiver, sites = new ArrayList<VirtualCallSite>());
520            List<Local> receivers = (List<Local>) methodToReceivers.get(m);
521            if( receivers == null )
522                methodToReceivers.put(m, receivers = new ArrayList<Local>());
523            receivers.add(receiver);
524        }
525        sites.add(new VirtualCallSite(s, m, iie, subSig, kind));
526    }
527    private void processNewMethod( SootMethod m ) {
528        if( m.isNative() || m.isPhantom() ) {
529            return;
530        }
531        Body b = m.retrieveActiveBody();
532        getImplicitTargets( m );
533        findReceivers(m, b);
534    }
535    private void findReceivers(SootMethod m, Body b) {
536        for( final Unit u : b.getUnits() ) {
537            final Stmt s = (Stmt) u;
538            if (s.containsInvokeExpr()) {
539                InvokeExpr ie = s.getInvokeExpr();
540
541                if (ie instanceof InstanceInvokeExpr) {
542                    InstanceInvokeExpr iie = (InstanceInvokeExpr) ie;
543                    Local receiver = (Local) iie.getBase();
544                    NumberedString subSig = 
545                        iie.getMethodRef().getSubSignature();
546                    addVirtualCallSite( s, m, receiver, iie, subSig,
547                            Edge.ieToKind(iie) );
548                    if( subSig == sigStart ) {
549                        addVirtualCallSite( s, m, receiver, iie, sigRun,
550                                Kind.THREAD );
551                    }
552                    else if( subSig == sigExecutorExecute 
553                                || subSig == sigHandlerPost
554                                || subSig == sigHandlerPostAtFrontOfQueue
555                                || subSig == sigHandlerPostAtTime
556                                || subSig == sigHandlerPostAtTimeWithToken
557                                || subSig == sigHandlerPostDelayed ) {
558                        if (iie.getArgCount() > 0) {
559                                Value runnable = iie.getArg(0);
560                                if (runnable instanceof Local)
561                                        addVirtualCallSite( s, m, (Local) runnable, iie, sigRun,
562                                                Kind.EXECUTOR );
563                        }
564                    }
565                    else if( subSig == sigExecute  ) {
566                        addVirtualCallSite( s, m, receiver, iie, sigDoInBackground,
567                                Kind.ASYNCTASK );
568                    }
569                } else if (ie instanceof DynamicInvokeExpr) {
570                        if(options.verbose())
571                                G.v().out.println("WARNING: InvokeDynamic to "+ie+" not resolved during call-graph construction.");
572                } else {
573                        SootMethod tgt = ie.getMethod();
574                        if(tgt!=null) {
575                                addEdge(m, s, tgt);
576                                String signature = tgt.getSignature();
577                                if( signature.equals( "<java.security.AccessController: java.lang.Object doPrivileged(java.security.PrivilegedAction)>" )
578                                                ||  signature.equals( "<java.security.AccessController: java.lang.Object doPrivileged(java.security.PrivilegedExceptionAction)>" )
579                                                ||  signature.equals( "<java.security.AccessController: java.lang.Object doPrivileged(java.security.PrivilegedAction,java.security.AccessControlContext)>" )
580                                                ||  signature.equals( "<java.security.AccessController: java.lang.Object doPrivileged(java.security.PrivilegedExceptionAction,java.security.AccessControlContext)>" ) ) {
581                                        
582                                        Local receiver = (Local) ie.getArg(0);
583                                        addVirtualCallSite( s, m, receiver, null, sigObjRun,
584                                                        Kind.PRIVILEGED );
585                                }
586                        } else {
587                                if(!Options.v().ignore_resolution_errors()) {
588                                        throw new InternalError("Unresolved target "+ie.getMethod()+". Resolution error should have occured earlier.");
589                                }
590                        }
591                }
592            }
593        }

598    private void getImplicitTargets( SootMethod source ) {
599        final SootClass scl = source.getDeclaringClass();
600        if( source.isNative() || source.isPhantom() ) return;
601        if( source.getSubSignature().indexOf( "<init>" ) >= 0 ) {
602            handleInit(source, scl);
603        }
604        Body b = source.retrieveActiveBody();
605        for (Unit u : b.getUnits()) {
606            final Stmt s = (Stmt) u;
607            if( s.containsInvokeExpr() ) {
608                InvokeExpr ie = s.getInvokeExpr();
609                final String methRefSig = ie.getMethodRef().getSignature();
610                if( methRefSig.equals( "<java.lang.reflect.Method: java.lang.Object invoke(java.lang.Object,java.lang.Object[])>" ) ) {
611                        reflectionModel.methodInvoke(source,s);
612                }
613                else if( methRefSig.equals( "<java.lang.Class: java.lang.Object newInstance()>" ) ) {
614                        reflectionModel.classNewInstance(source,s);
615                }
616                else if( methRefSig.equals( "<java.lang.reflect.Constructor: java.lang.Object newInstance(java.lang.Object[])>" ) ) {
617                        reflectionModel.contructorNewInstance(source, s);
618                }
619                if( ie.getMethodRef().getSubSignature() == sigForName ) {
620                        reflectionModel.classForName(source,s);
621                }
622                if( ie instanceof StaticInvokeExpr ) {
623                        SootClass cl = ie.getMethodRef().declaringClass();
624                        for (SootMethod clinit : EntryPoints.v().clinitsOf(cl)) {
625                                addEdge( source, s, clinit, Kind.CLINIT );
626                        }
627                }
628            }
629            if( s.containsFieldRef() ) {
630                FieldRef fr = s.getFieldRef();
631                if( fr instanceof StaticFieldRef ) {
632                    SootClass cl = fr.getFieldRef().declaringClass();
633                    for (SootMethod clinit : EntryPoints.v().clinitsOf(cl)) {
634                        addEdge( source, s, clinit, Kind.CLINIT );
635                    }
636                }
637            }
638            if( s instanceof AssignStmt ) {
639                Value rhs = ((AssignStmt)s).getRightOp();
640                if( rhs instanceof NewExpr ) {
641                    NewExpr r = (NewExpr) rhs;
642                    SootClass cl = r.getBaseType().getSootClass();
643                    for (SootMethod clinit : EntryPoints.v().clinitsOf(cl)) {
644                        addEdge( source, s, clinit, Kind.CLINIT );
645                    }
646                } else if( rhs instanceof NewArrayExpr || rhs instanceof NewMultiArrayExpr ) {
647                    Type t = rhs.getType();
648                    if( t instanceof ArrayType ) t = ((ArrayType)t).baseType;
649                    if( t instanceof RefType ) {
650                        SootClass cl = ((RefType) t).getSootClass();
651                        for (SootMethod clinit : EntryPoints.v().clinitsOf(cl)) {
652                            addEdge( source, s, clinit, Kind.CLINIT );
653                        }
654                    }
655                }
656            }
657        }
658    }

660    private void processNewMethodContext( MethodOrMethodContext momc ) {
661        SootMethod m = momc.method();
662        Iterator<Edge> it = cicg.edgesOutOf(m);
663        while( it.hasNext() ) {
664            Edge e = (Edge) it.next();
665            cm.addStaticEdge( momc, e.srcUnit(), e.tgt(), e.kind() );
666        }
667    }

669    private void handleInit(SootMethod source, final SootClass scl) {
670        addEdge( source, null, scl, sigFinalize, Kind.FINALIZE );
671    }
672    private void constantForName( String cls, SootMethod src, Stmt srcUnit ) {
673        if( cls.length() > 0 && cls.charAt(0) == '[' ) {
674            if( cls.length() > 1 && cls.charAt(1) == 'L' && cls.charAt(cls.length()-1) == ';' ) {
675                cls = cls.substring(2,cls.length()-1);
676                constantForName( cls, src, srcUnit );
677            }
678        } else {
679            if( !Scene.v().containsClass( cls ) ) {
680                if( options.verbose() ) {
681                    G.v().out.println( "Warning: Class "+cls+" is"+
682                        " a dynamic class, and you did not specify"+
683                        " it as such; graph will be incomplete!" );
684                }
685            } else {
686                SootClass sootcls = Scene.v().getSootClass( cls );
687                if (!sootcls.isPhantomClass()) {
688                        if( !sootcls.isApplicationClass() ) {
689                            sootcls.setLibraryClass();
690                        }
691                        for (SootMethod clinit : EntryPoints.v().clinitsOf(sootcls)) {
692                            addEdge( src, srcUnit, clinit, Kind.CLINIT );
693                        }
694                }
695
696            }
697        }
698    }

705    private void addEdge(  SootMethod src, Stmt stmt, SootClass cls, NumberedString methodSubSig, Kind kind ) {
706        SootMethod sm = cls.getMethodUnsafe( methodSubSig );
707        if( sm != null ) {
708            addEdge( src, stmt, sm, kind );
709        }
710    }
711    private void addEdge( SootMethod src, Stmt stmt, SootMethod tgt ) {
712        InvokeExpr ie = stmt.getInvokeExpr();
713        addEdge( src, stmt, tgt, Edge.ieToKind(ie) );
714    }
