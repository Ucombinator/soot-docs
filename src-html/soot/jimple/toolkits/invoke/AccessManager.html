039    /** Returns true iff target is legally accessible from container.
040     *  Illegal access occurs when any of the following cases holds:
041     *    1. target is private, but container.declaringClass() != 
042     *              target.declaringClass(); or,
043     *    2. target is package-visible, and its package differs from
044     *              that of container; or,
045     *    3. target is protected, and either:
046     *          a. container doesn't belong to target.declaringClass,
047     *                 or any subclass of ; 
048     */
049    public static boolean isAccessLegal(SootMethod container, ClassMember target)
050    {
051        SootClass targetClass = target.getDeclaringClass();
052        SootClass containerClass = container.getDeclaringClass();
053
054        if (!isAccessLegal(container, targetClass))
055            return false;
056
057        // Condition 1 above.
058        if (target.isPrivate() && 
059                !targetClass.getName().equals(containerClass.getName()))
060            return false;
061
062        // Condition 2. Check the package names.
063        if (!target.isPrivate() && !target.isProtected() 
064            && !target.isPublic())
065        {
066            if (!targetClass.getPackageName().equals
067                     (containerClass.getPackageName()))
068                return false;
069        }
070
071        // Condition 3.  
072        if (target.isProtected())
073        {
074            Hierarchy h = Scene.v().getActiveHierarchy();
075
076            // protected means that you can be accessed by your children.
077            // i.e. container must be in a child of target.
078            if (h.isClassSuperclassOfIncluding(targetClass, containerClass))
079                return true;
080
081            return false;
082        }
083
084        return true;        
085    }

087    /** Returns true if an access to <code>target</code> is legal from code in <code>container</code>. */
088    public static boolean isAccessLegal(SootMethod container, SootClass target)
089    {
090        return target.isPublic() || 
091            container.getDeclaringClass().getPackageName().equals(target.getPackageName());
092    }

094    /**
095     * Returns true if the statement <code>stmt</code> contains an illegal access to a field or method,
096     * assuming the statement is in method <code>container</code>
097     * @param container
098     * @param stmt
099     * @return
100     */
101    public static boolean isAccessLegal(SootMethod container, Stmt stmt) {
102                if (stmt.containsInvokeExpr()) {
103                        return AccessManager.isAccessLegal(container, stmt.getInvokeExpr().getMethod());
104                } else if (stmt instanceof AssignStmt) {
105                        AssignStmt as=(AssignStmt)stmt;
106                        if (as.getRightOp() instanceof FieldRef) {
107                                FieldRef r=(FieldRef)as.getRightOp();
108                                return AccessManager.isAccessLegal(container, r.getField());                            
109                        }
110                        if (as.getLeftOp() instanceof FieldRef) {
111                                FieldRef r=(FieldRef)as.getLeftOp();
112                                return AccessManager.isAccessLegal(container, r.getField());                                            
113                        }
114                }       
115                return true;
116    }

118    /**
119     * Resolves illegal accesses in the interval ]before,after[
120     * by creating accessor methods.
121     * <code>before</code> and <code>after</code> can be null to indicate beginning/end respectively.
122     * @param body
123     * @param before
124     * @param after
125     */
126    public static void createAccessorMethods(Body body, Stmt before, Stmt after) {
127        soot.util.Chain units=body.getUnits();
128        
129        if (before!=null && !units.contains(before))
130                throw new RuntimeException();
131        if (after!=null && !units.contains(after))
132                throw new RuntimeException();
133        
134        ArrayList<Unit> unitList = new ArrayList<Unit>(); unitList.addAll(units);
135        
136        boolean bInside=before==null;
137        for (Unit unit : unitList) {
138                Stmt s=(Stmt)unit;
139                
140                if (bInside) {
141                        if (s==after)
142                                return;
143                        
144                        if (!isAccessLegal(body.getMethod(), s))
145                                createAccessorMethod(body.getMethod(), s);
146                        
147                } else {
148                        if (s==before)
149                                bInside=true;
150                }
151        }
152        
153    }

156    /**
157     * Creates a name for an accessor method.
158     * @param member
159     * @param setter
160     * @return
161     */
162    public static String createAccessorName(ClassMember member, boolean setter) {
163        SootClass target=member.getDeclaringClass();
164        
165        String name="access$";
166        if (member instanceof SootField) {
167                SootField f=(SootField)member;
168                if (setter) {
169                        name +="set$";
170                } else {
171                        name +="get$";
172                }   
173                name +=f.getName();
174        } else {
175                SootMethod m=(SootMethod)member;
176                name += m.getName() + "$";
177                
178                for (Iterator it=m.getParameterTypes().iterator(); it.hasNext();) {
179                        Type type=(Type)it.next();
180                        name += type.toString().replaceAll("\\.", "\\$\\$") + "$";
181                }
182        }
183                return name;
184    }

186    /**
187     * Turns a field access or method call into a call to an accessor method.
188     * Reuses existing accessors based on name mangling (see createAccessorName) 
189     * @param container
190     * @param stmt
191     */
192    public static void createAccessorMethod(SootMethod container, Stmt stmt) 
193        {
194        //System.out.println("Creating accessor method: \n" + 
195        //              " method: " + container + " \n" +
196                //              " stmt: " + stmt);
197                
198        Body containerBody=container.getActiveBody();
199        soot.util.Chain containerStmts=containerBody.getUnits();
200        if (!containerStmts.contains(stmt))
201                throw new RuntimeException();
202        
203        if (stmt.containsInvokeExpr()) {
204                createInvokeAccessor(container, stmt);          
205        } else if (stmt instanceof AssignStmt) {
206                AssignStmt as=(AssignStmt)stmt;
207                FieldRef ref;
208                if (as.getLeftOp() instanceof FieldRef) {
209//                       set
210                        ref=(FieldRef)as.getLeftOp();
211                        createSetAccessor(container, as, ref);
212                
213                } else if (as.getRightOp() instanceof FieldRef) {
214                        // get
215                         ref=(FieldRef)as.getRightOp();
216                         createGetAccessor(container, as, ref);                         
217                } else {
218                        throw new RuntimeException("Expected class member access");
219                }               
220        } else
221                throw new RuntimeException("Expected class member access");
222    }

224        private static void createGetAccessor(SootMethod container, AssignStmt as, FieldRef ref) {
225        java.util.List parameterTypes=new LinkedList();
226        java.util.List<SootClass> thrownExceptions=new LinkedList<SootClass>();
227        
228        
229        Body accessorBody = Jimple.v().newBody();
230        soot.util.Chain accStmts=accessorBody.getUnits();
231        LocalGenerator lg=new LocalGenerator(accessorBody);
232        
233        Body containerBody=container.getActiveBody();
234        soot.util.Chain containerStmts=containerBody.getUnits();        
235        
236                SootClass target=ref.getField().getDeclaringClass();
237                
238                SootMethod accessor;
239                
240                String name=createAccessorName(ref.getField(), false);
241                accessor = target.getMethodByNameUnsafe(name);
242                if (accessor == null) {                 
243                         Type returnType=ref.getField().getType();
244                         Local thisLocal=lg.generateLocal(target.getType());
245                         if (ref instanceof InstanceFieldRef) {
246                                parameterTypes.add(target.getType());
247                                accStmts.addFirst(                                              
248                                                Jimple.v().newIdentityStmt(thisLocal, 
249                                Jimple.v().newParameterRef(target.getType(),0)));
250                         }
251                        Local l=lg.generateLocal(ref.getField().getType());
252                        Value v;
253                        if (ref instanceof InstanceFieldRef) {
254                                v=Jimple.v().newInstanceFieldRef(thisLocal,ref.getFieldRef());
255                        } else {
256                                v=Jimple.v().newStaticFieldRef(ref.getFieldRef());
257                        }
258                        accStmts.add(Jimple.v().newAssignStmt(l, v));
259                        accStmts.add(Jimple.v().newReturnStmt(l));
260                        
261                        accessor=new SootMethod(name, parameterTypes, returnType, 
262                                        Modifier.PUBLIC | Modifier.STATIC,
263                                        thrownExceptions);      
264                        accessorBody.setMethod(accessor);       
265                        accessor.setActiveBody(accessorBody);
266                        target.addMethod(accessor);
267                }
268                java.util.List args=new LinkedList();
269                if (ref instanceof InstanceFieldRef) {
270                        args.add(((InstanceFieldRef)ref).getBase());
271                }
272                InvokeExpr newExpr= 
273                        Jimple.v().newStaticInvokeExpr(accessor.makeRef(), args);
274                
275                as.setRightOp(newExpr);
276        }

278        private static void createSetAccessor(SootMethod container, AssignStmt as, FieldRef ref) {
279        java.util.List parameterTypes=new LinkedList();
280        java.util.List<SootClass> thrownExceptions=new LinkedList<SootClass>();
281        
282        Body accessorBody = Jimple.v().newBody();
283        soot.util.Chain accStmts=accessorBody.getUnits();
284        LocalGenerator lg=new LocalGenerator(accessorBody);
285        
286        Body containerBody=container.getActiveBody();
287        soot.util.Chain containerStmts=containerBody.getUnits();
288        
289                SootClass target=ref.getField().getDeclaringClass();
290                SootMethod accessor;
291                
292                String name=createAccessorName(ref.getField(), true);
293                accessor = target.getMethodByNameUnsafe(name);
294                if (accessor == null) {                 
295                        Local thisLocal=lg.generateLocal(target.getType());
296                        int paramID=0;
297                        if (ref instanceof InstanceFieldRef) {
298                                accStmts.add(
299                                                Jimple.v().newIdentityStmt(thisLocal, 
300                                                                Jimple.v().newParameterRef(target.getType(),paramID)));                         
301                                parameterTypes.add(target.getType());
302                                paramID++;
303                        }
304                        parameterTypes.add(ref.getField().getType());
305                        Local l=lg.generateLocal(ref.getField().getType());
306                        accStmts.add(
307                                        Jimple.v().newIdentityStmt(l, 
308                                        Jimple.v().newParameterRef(ref.getField().getType(),paramID)));
309                        paramID++;
310                        if (ref instanceof InstanceFieldRef) {
311                                accStmts.add(Jimple.v().newAssignStmt(
312                                        Jimple.v().newInstanceFieldRef(thisLocal,ref.getFieldRef()), l));
313                        } else {
314                                accStmts.add(Jimple.v().newAssignStmt(
315                                                Jimple.v().newStaticFieldRef(ref.getFieldRef()), l));
316                        }
317                        accStmts.addLast(Jimple.v().newReturnVoidStmt());
318                        Type returnType=VoidType.v();
319                        
320                        accessor=new SootMethod(name, parameterTypes, returnType, 
321                                        Modifier.PUBLIC | Modifier.STATIC,
322                                        thrownExceptions);      
323                        accessorBody.setMethod(accessor);       
324                        accessor.setActiveBody(accessorBody);
325                        target.addMethod(accessor);
326                }
327                
328                java.util.List args=new LinkedList();
329                if (ref instanceof InstanceFieldRef) {
330                        args.add(((InstanceFieldRef)ref).getBase());
331                }
332                args.add(as.getRightOp());
333                InvokeExpr newExpr= 
334                        Jimple.v().newStaticInvokeExpr(accessor.makeRef(), args);
335                
336                Stmt newStmt=Jimple.v().newInvokeStmt(newExpr);
337                
338            containerStmts.insertAfter(newStmt, as);
339                containerStmts.remove(as);
340        }

342        private static void createInvokeAccessor(SootMethod container, Stmt stmt) {     
343        java.util.List parameterTypes=new LinkedList();
344        java.util.List<SootClass> thrownExceptions=new LinkedList<SootClass>();
345        Type returnType;
346        
347        Body accessorBody = Jimple.v().newBody();
348        soot.util.Chain accStmts=accessorBody.getUnits();
349        LocalGenerator lg=new LocalGenerator(accessorBody);
350        
351        Body containerBody=container.getActiveBody();
352        soot.util.Chain containerStmts=containerBody.getUnits();
353        
354                InvokeExpr expr=stmt.getInvokeExpr();
355                SootMethod method=expr.getMethod();
356                //System.out.println("method: " + method);
357        
358                SootClass target=method.getDeclaringClass();
359                //System.out.println("target: " + target);
360                
361                //System.out.println("method ref: " + expr.getMethodRef());
362                
363                SootMethod accessor;
364                
365                String name=createAccessorName(method, true);
366                accessor = target.getMethodByNameUnsafe(name);
367                if (accessor == null) {                 
368                        java.util.List arguments=new LinkedList();
369                        
370                        if (expr instanceof InstanceInvokeExpr) {
371                                parameterTypes.add(target.getType());
372                        }
373                        
374                        parameterTypes.addAll(method.getParameterTypes());
375                        returnType=method.getReturnType();
376                        thrownExceptions.addAll(method.getExceptions());
377                        
378                        int paramID=0;
379                        for (java.util.Iterator it=parameterTypes.iterator(); it.hasNext();) {
380                                Type type=(Type)it.next();
381                                Local l=lg.generateLocal(type);
382                                //System.out.println("local type: " + type);
383                                accStmts.add(
384                                                Jimple.v().newIdentityStmt(l, 
385                                                Jimple.v().newParameterRef(type, paramID)));
386                                arguments.add(l);
387                                paramID++;
388                        }
389                        
390                        InvokeExpr accExpr;
391                        
392                        if (expr instanceof StaticInvokeExpr) {
393                                accExpr=Jimple.v().newStaticInvokeExpr(method.makeRef(), arguments);
394                        } else if (expr instanceof VirtualInvokeExpr) {
395                                Local thisLocal=(Local)arguments.get(0);
396                                arguments.remove(0);
397                                accExpr=Jimple.v().newVirtualInvokeExpr(thisLocal, method.makeRef(), arguments);
398                        } else if (expr instanceof SpecialInvokeExpr) {
399                                Local thisLocal=(Local)arguments.get(0);
400                                arguments.remove(0);
401                                accExpr=Jimple.v().newSpecialInvokeExpr(thisLocal, method.makeRef(), arguments);
402                        } else
403                                throw new RuntimeException("");
404                        
405                        Stmt s;
406                        if (returnType instanceof VoidType) {
407                                s=Jimple.v().newInvokeStmt(accExpr);
408                                accStmts.add(s);
409                                accStmts.add(Jimple.v().newReturnVoidStmt());
410                        } else {
411                                Local resultLocal=lg.generateLocal(returnType);
412                                s=Jimple.v().newAssignStmt(resultLocal,accExpr);
413                                accStmts.add(s);
414                                accStmts.add(Jimple.v().newReturnStmt(resultLocal));
415                        }
416                        
417                        accessor=new SootMethod(name, parameterTypes, returnType, 
418                                        Modifier.PUBLIC | Modifier.STATIC,
419                                        thrownExceptions);      
420                        accessorBody.setMethod(accessor);       
421                        accessor.setActiveBody(accessorBody);
422                        target.addMethod(accessor);
423                }
424                
425                java.util.List args=new LinkedList();
426                if (expr instanceof InstanceInvokeExpr) {
427                        args.add(((InstanceInvokeExpr)expr).getBase());
428                }
429                args.addAll(expr.getArgs());
430                InvokeExpr newExpr= 
431                                Jimple.v().newStaticInvokeExpr(accessor.makeRef(), args);
432                                
433                stmt.getInvokeExprBox().setValue(newExpr);
434        }

436        /** Modifies code so that an access to <code>target</code> is legal from 
437         * code in <code>container</code>. 
438         * 
439         * The "accessors" option assumes suitable accessor methods will be created after checking. 
440         * 
441         * */
442    public static boolean ensureAccess(SootMethod container, ClassMember target, String options)
443    {
444        boolean accessors=options.equals("accessors");
445        boolean allowChanges = !(options.equals("none"));
446        boolean safeChangesOnly = !(options.equals("unsafe"));
447        
448        
449        SootClass targetClass = target.getDeclaringClass();
450        if (!ensureAccess(container, targetClass, options))
451            return false;
452
453        if (isAccessLegal(container, target))
454            return true;
455
456        if (!allowChanges && !accessors)
457            return false;
458
459        
460          //throw new RuntimeException("Not implemented yet!");
461        
462        if (target.getDeclaringClass().isApplicationClass())
463        {
464                if (accessors)
465                        return true;
466                
467                if (safeChangesOnly)
468                        throw new RuntimeException("Not implemented yet!");
469                
470                target.setModifiers(target.getModifiers() | Modifier.PUBLIC);
471            return true;
472        }
473        else
474            return false;
475    }

477    /** Modifies code so that an access to <code>target</code> is legal from code in <code>container</code>. */
478    public static boolean ensureAccess(SootMethod container, SootClass target, String options)
479    {
480        boolean accessors=options.equals("accessors");
481        boolean allowChanges = !(options.equals("none"));
482        boolean safeChangesOnly = !(options.equals("unsafe"));
483
484        if (isAccessLegal(container, target))
485            return true;
486
487        if (!allowChanges && !accessors)
488            return false;
489
490        if (safeChangesOnly && !accessors)
491            throw new RuntimeException("Not implemented yet!");
492        
493        if (accessors)
494                return false;
495        
496        if (target.isApplicationClass())
497        {
498                
499                target.setModifiers(target.getModifiers() | Modifier.PUBLIC);
500            return true;
501        }
502        else
503            return false;
504    }
