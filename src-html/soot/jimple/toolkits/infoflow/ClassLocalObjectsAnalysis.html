103        private void prepare()
104        {
105                // Get list of all methods
106                allMethods = getAllReachableMethods(sootClass);
107                
108                // Get list of external methods
109                externalMethods = uf.getExtMethods(sootClass);
110                SootClass superclass = sootClass;
111                if(superclass.hasSuperclass())
112                        superclass = superclass.getSuperclass();
113                while(superclass.hasSuperclass())
114                {
115                        if(superclass.isApplicationClass())
116                        externalMethods.addAll(uf.getExtMethods(superclass));
117                        superclass = superclass.getSuperclass();
118                }
119
120                // Get list of internal methods
121                internalMethods = new ArrayList<SootMethod>();
122                for (SootMethod method : allMethods) {
123                        if(!externalMethods.contains(method))
124                                internalMethods.add(method);
125                }
126                
127                // Get list of all fields
128                allFields = getAllFields(sootClass);
129                
130                // Get list of external fields
131                externalFields = uf.getExtFields(sootClass);
132                superclass = sootClass;
133                if(superclass.hasSuperclass())
134                        superclass = superclass.getSuperclass();
135                while(superclass.hasSuperclass())
136                {
137                        if(superclass.isApplicationClass())
138                        externalFields.addAll(uf.getExtFields(superclass));
139                        superclass = superclass.getSuperclass();
140                }
141
142                // Get list of internal fields
143                internalFields = new ArrayList<SootField>();
144                for (SootField field : allFields) {
145                        if(!externalFields.contains(field))
146                                internalFields.add(field);
147                }
148                
149        }

151        // Returns a list of reachable methods in class sc and its superclasses
152        public static List<SootMethod> getAllReachableMethods(SootClass sc)
153        {
154                ReachableMethods rm = Scene.v().getReachableMethods();
155                
156                // Get list of reachable methods declared in this class
157                List<SootMethod> allMethods = new ArrayList<SootMethod>();
158                Iterator methodsIt = sc.methodIterator();
159                while(methodsIt.hasNext())
160                {
161                        SootMethod method = (SootMethod) methodsIt.next();
162                        if(rm.contains(method))
163                                allMethods.add(method);
164                }
165                
166                // Add reachable methods declared in superclasses
167                SootClass superclass = sc;
168                if(superclass.hasSuperclass())
169                        superclass = superclass.getSuperclass();
170                while(superclass.hasSuperclass()) // we don't want to process Object
171                {
172                Iterator scMethodsIt = superclass.methodIterator();
173                while(scMethodsIt.hasNext())
174                {
175                                SootMethod scMethod = (SootMethod) scMethodsIt.next();
176                                if(rm.contains(scMethod))
177                                        allMethods.add(scMethod);
178                }
179                        superclass = superclass.getSuperclass();
180                }
181                return allMethods;
182        }
183        
184        // Returns a list of fields in class sc and its superclasses
185        public static List<SootField> getAllFields(SootClass sc)
186        {
187                // Get list of reachable methods declared in this class
188                // Also get list of fields declared in this class
189                List<SootField> allFields = new ArrayList<SootField>();
190                for (SootField field : sc.getFields())
191                {
192                        allFields.add(field);
193                }
194                
195                // Add reachable methods and fields declared in superclasses
196                SootClass superclass = sc;
197                if(superclass.hasSuperclass())
198                        superclass = superclass.getSuperclass();
199                while(superclass.hasSuperclass()) // we don't want to process Object
200                {
201                        for (SootField scField : superclass.getFields())
202                {
203                                allFields.add(scField);
204                }
205                        superclass = superclass.getSuperclass();
206                }
207                return allFields;
208        }

210        private void doAnalysis()
211        {
212                // Combine the DFA results for each of this class's methods, using safe
213                // approximations for which parameters, fields, and globals are shared
214                // or local.
215                
216                // Separate fields into shared and local.  Initially fields are known to be
217                // shared if they have any external accesses, or if they're static.
218                // Methods are iterated over, moving fields to shared if shared data flows to them.
219                // This is repeated until no fields move for a complete iteration.
220                
221                // Populate localFields and sharedFields with fields of this class
222                localFields = new ArrayList<SootField>();
223                sharedFields = new ArrayList<SootField>();
224                Iterator<SootField> fieldsIt = allFields.iterator();
225                while(fieldsIt.hasNext())
226                {
227                        SootField field = fieldsIt.next();
228                        if( fieldIsInitiallyLocal(field) )
229                                localFields.add(field);
230                        else
231                                sharedFields.add(field);
232                }
233                
234                // Add inner fields to localFields and sharedFields, if present
235                localInnerFields = new ArrayList<SootField>();
236                sharedInnerFields = new ArrayList<SootField>();
237                Iterator<SootMethod> methodsIt = allMethods.iterator();
238                while(methodsIt.hasNext())
239                {
240                        SootMethod method = methodsIt.next();
241                        
242                        // Get data flow summary
243                        MutableDirectedGraph dataFlowSummary;
244                        if(primitiveDfa != null)
245                        {
246                                dataFlowSummary = primitiveDfa.getMethodInfoFlowSummary(method);
247                                
248                                if(printdfgs && method.getDeclaringClass().isApplicationClass())
249                                {
250                                        G.v().out.println("Attempting to print graphs (will succeed only if ./dfg/ is a valid path)");
251                                        DirectedGraph primitiveGraph = primitiveDfa.getMethodInfoFlowAnalysis(method).getMethodAbbreviatedInfoFlowGraph();
252                                        InfoFlowAnalysis.printGraphToDotFile("dfg/" + method.getDeclaringClass().getShortName() + "_" + method.getName() + "_primitive", 
253                                                primitiveGraph, method.getName() + "_primitive", false);
254
255                                        DirectedGraph nonPrimitiveGraph = dfa.getMethodInfoFlowAnalysis(method).getMethodAbbreviatedInfoFlowGraph();
256                                        InfoFlowAnalysis.printGraphToDotFile("dfg/" + method.getDeclaringClass().getShortName() + "_" + method.getName(),
257                                                nonPrimitiveGraph, method.getName(), false);
258                                }
259                        }
260                        else
261                        {
262                                dataFlowSummary = dfa.getMethodInfoFlowSummary(method);
263                                
264                                if(printdfgs && method.getDeclaringClass().isApplicationClass())
265                                {
266                                        G.v().out.println("Attempting to print graph (will succeed only if ./dfg/ is a valid path)");
267                                        DirectedGraph nonPrimitiveGraph = dfa.getMethodInfoFlowAnalysis(method).getMethodAbbreviatedInfoFlowGraph();
268                                        InfoFlowAnalysis.printGraphToDotFile("dfg/" + method.getDeclaringClass().getShortName() + "_" + method.getName(),
269                                                nonPrimitiveGraph, method.getName(), false);
270                                }
271                        }
272                                
273                        // Iterate through nodes
274                        Iterator<Object> nodesIt = dataFlowSummary.getNodes().iterator();
275                        while(nodesIt.hasNext())
276                        {
277                                EquivalentValue node = (EquivalentValue) nodesIt.next();
278                                if(node.getValue() instanceof InstanceFieldRef)
279                                {
280                                        InstanceFieldRef ifr = (InstanceFieldRef) node.getValue();
281                                        if( !localFields.contains(ifr.getField()) && !sharedFields.contains(ifr.getField()) &&
282                                                !localInnerFields.contains(ifr.getField()) ) // && !sharedInnerFields.contains(ifr.getField()))
283                                        {
284                                                // this field is read or written, but is not in the lists of fields!
285                                                localInnerFields.add(ifr.getField());
286                                        }
287                                }
288                        }
289                }
290                
291                // Propagate (aka iterate iterate iterate iterate! hope it's not too slow)
292                boolean changed = true;
293                while(changed)
294                {
295                        changed = false;
296//                      G.v().out.println("Starting iteration:");
297                        methodsIt = allMethods.iterator();
298                        while(methodsIt.hasNext())
299                        {
300                                SootMethod method = methodsIt.next();
301                                // we can't learn anything from non-concrete methods, and statics can't write non-static fields
302                                if(method.isStatic() || !method.isConcrete())
303                                        continue;
304                                
305                                ListIterator<SootField> localFieldsIt = ((List<SootField>) localFields).listIterator();
306                                while(localFieldsIt.hasNext())
307                                {
308                                        SootField localField = localFieldsIt.next();
309                                        List sourcesAndSinks = new ArrayList();
310
311                                        MutableDirectedGraph dataFlowSummary;
312                                        if(primitiveDfa != null)
313                                                dataFlowSummary = primitiveDfa.getMethodInfoFlowSummary(method);
314                                        else
315                                                dataFlowSummary = dfa.getMethodInfoFlowSummary(method);
316                                        
317                                        EquivalentValue node = InfoFlowAnalysis.getNodeForFieldRef(method, localField);
318                                        if(dataFlowSummary.containsNode(node))
319                                        {
320                                                sourcesAndSinks.addAll(dataFlowSummary.getSuccsOf(node));
321                                                sourcesAndSinks.addAll(dataFlowSummary.getPredsOf(node));
322                                        }
323
324                                        Iterator sourcesAndSinksIt = sourcesAndSinks.iterator();
325                                        if(localField.getDeclaringClass().isApplicationClass() &&
326                                           sourcesAndSinksIt.hasNext())
327                                        {
328//                                              if(!printedMethodHeading)
329//                                              {
330//                                                      G.v().out.println("    Method: " + method.toString());
331//                                                      printedMethodHeading = true;
332//                                              }
333//                                              G.v().out.println("        Field: " + localField.toString());
334                                        }
335                                        while(sourcesAndSinksIt.hasNext())
336                                        {
337                                                EquivalentValue sourceOrSink = (EquivalentValue) sourcesAndSinksIt.next();
338                                                Ref sourceOrSinkRef = (Ref) sourceOrSink.getValue();
339                                                boolean fieldBecomesShared = false;
340                                                if(sourceOrSinkRef instanceof ParameterRef) // or return ref
341                                                {
342                                                        fieldBecomesShared = !parameterIsLocal(method, sourceOrSink, true);
343                                                }
344                                                else if(sourceOrSinkRef instanceof ThisRef) // or return ref
345                                                {
346                                                        fieldBecomesShared = !thisIsLocal(method, sourceOrSink);
347                                                }
348                                                else if(sourceOrSinkRef instanceof InstanceFieldRef)
349                                                {
350                                                        fieldBecomesShared = sharedFields.contains( ((FieldRef)sourceOrSinkRef).getField() ) || sharedInnerFields.contains( ((FieldRef)sourceOrSinkRef).getField() );
351                                                }
352                                                else if(sourceOrSinkRef instanceof StaticFieldRef)
353                                                {
354                                                        fieldBecomesShared = true;
355                                                }
356                                                else
357                                                {
358                                                        throw new RuntimeException("Unknown type of Ref in Data Flow Graph:");
359                                                }
360                                                
361                                                if(fieldBecomesShared)
362                                                {
363//                                                      if(localField.getDeclaringClass().isApplicationClass())
364//                                                              G.v().out.println("            Source/Sink: " + sourceOrSinkRef.toString() + " is SHARED");
365                                                        localFieldsIt.remove();
366                                                        sharedFields.add(localField);
367                                                        changed = true;
368                                                        break; // other sources don't matter now... it only takes one to taint the field
369                                                }
370                                                else
371                                                {
372//                                                      if(localField.getDeclaringClass().isApplicationClass())
373//                                                              G.v().out.println("            Source: " + sourceRef.toString() + " is local");
374                                                }
375                                        }
376                                }
377
378
379                                ListIterator<SootField> localInnerFieldsIt = ((List<SootField>) localInnerFields).listIterator();
380//                              boolean printedMethodHeading = false;
381                                while(!changed && localInnerFieldsIt.hasNext())
382                                {
383                                        SootField localInnerField = localInnerFieldsIt.next();
384                                        List sourcesAndSinks = new ArrayList();
385
386                                        MutableDirectedGraph dataFlowSummary;
387                                        if(primitiveDfa != null)
388                                                dataFlowSummary = primitiveDfa.getMethodInfoFlowSummary(method);
389                                        else
390                                                dataFlowSummary = dfa.getMethodInfoFlowSummary(method);
391                                        
392                                        EquivalentValue node = InfoFlowAnalysis.getNodeForFieldRef(method, localInnerField);
393                                        if(dataFlowSummary.containsNode(node))
394                                        {
395                                                sourcesAndSinks.addAll(dataFlowSummary.getSuccsOf(node));
396                                                sourcesAndSinks.addAll(dataFlowSummary.getPredsOf(node));
397                                        }
398
399                                        Iterator sourcesAndSinksIt = sourcesAndSinks.iterator();
400                                        if(localInnerField.getDeclaringClass().isApplicationClass() &&
401                                           sourcesAndSinksIt.hasNext())
402                                        {
403//                                              if(!printedMethodHeading)
404//                                              {
405//                                                      G.v().out.println("    Method: " + method.toString());
406//                                                      printedMethodHeading = true;
407//                                              }
408//                                              G.v().out.println("        Field: " + localField.toString());
409                                        }
410                                        while(sourcesAndSinksIt.hasNext())
411                                        {
412                                                EquivalentValue sourceOrSink = (EquivalentValue) sourcesAndSinksIt.next();
413                                                Ref sourceOrSinkRef = (Ref) sourceOrSink.getValue();
414                                                boolean fieldBecomesShared = false;
415                                                if(sourceOrSinkRef instanceof ParameterRef) // or return ref
416                                                {
417                                                        fieldBecomesShared = !parameterIsLocal(method, sourceOrSink, true);
418                                                }
419                                                else if(sourceOrSinkRef instanceof ThisRef) // or return ref
420                                                {
421                                                        fieldBecomesShared = !thisIsLocal(method, sourceOrSink);
422                                                }
423                                                else if(sourceOrSinkRef instanceof InstanceFieldRef)
424                                                {
425                                                        fieldBecomesShared = sharedFields.contains( ((FieldRef)sourceOrSinkRef).getField() ) || sharedInnerFields.contains( ((FieldRef)sourceOrSinkRef).getField() );
426                                                }
427                                                else if(sourceOrSinkRef instanceof StaticFieldRef)
428                                                {
429                                                        fieldBecomesShared = true;
430                                                }
431                                                else
432                                                {
433                                                        throw new RuntimeException("Unknown type of Ref in Data Flow Graph:");
434                                                }
435                                                
436                                                if(fieldBecomesShared)
437                                                {
438//                                                      if(localField.getDeclaringClass().isApplicationClass())
439//                                                              G.v().out.println("            Source/Sink: " + sourceOrSinkRef.toString() + " is SHARED");
440                                                        localInnerFieldsIt.remove();
441                                                        sharedInnerFields.add(localInnerField);
442                                                        changed = true;
443                                                        break; // other sources don't matter now... it only takes one to taint the field
444                                                }
445                                                else
446                                                {
447//                                                      if(localField.getDeclaringClass().isApplicationClass())
448//                                                              G.v().out.println("            Source: " + sourceRef.toString() + " is local");
449                                                }
450                                        }
451                                }
452                        }
453                }
454                
455                // Print debug output
456                if(dfa.printDebug())
457                {
458                        G.v().out.println("        Found local/shared fields for " + sootClass.toString());
459                        G.v().out.println("          Local fields: ");
460                        Iterator<SootField> localsToPrintIt = localFields.iterator();
461                        while(localsToPrintIt.hasNext())
462                        {
463                                SootField localToPrint = localsToPrintIt.next();
464                                if(localToPrint.getDeclaringClass().isApplicationClass())
465                                        G.v().out.println("                  " + localToPrint);
466                        }
467                        G.v().out.println("          Shared fields: ");
468                        Iterator<SootField> sharedsToPrintIt = sharedFields.iterator();
469                        while(sharedsToPrintIt.hasNext())
470                        {
471                                SootField sharedToPrint = sharedsToPrintIt.next();
472                                if(sharedToPrint.getDeclaringClass().isApplicationClass())
473                                        G.v().out.println("                  " + sharedToPrint);
474                        }
475                        G.v().out.println("          Local inner fields: ");
476                        localsToPrintIt = localInnerFields.iterator();
477                        while(localsToPrintIt.hasNext())
478                        {
479                                SootField localToPrint = localsToPrintIt.next();
480                                if(localToPrint.getDeclaringClass().isApplicationClass())
481                                        G.v().out.println("                  " + localToPrint);
482                        }
483                        G.v().out.println("          Shared inner fields: ");
484                        sharedsToPrintIt = sharedInnerFields.iterator();
485                        while(sharedsToPrintIt.hasNext())
486                        {
487                                SootField sharedToPrint = sharedsToPrintIt.next();
488                                if(sharedToPrint.getDeclaringClass().isApplicationClass())
489                                        G.v().out.println("                  " + sharedToPrint);
490                        }
491                }
492        }

494        private void propagate()
495        {
496                // Initialize worklist
497                ArrayList<SootMethod> worklist = new ArrayList<SootMethod>();
498                worklist.addAll(entryMethods);
499                
500                // Initialize set of contexts
501                methodToContext = new HashMap<SootMethod, CallLocalityContext>(); // TODO: add the ability to share a map with another CLOA to save memory (be careful of context-sensitive call graph)
502                for (SootMethod method : worklist) {
503                        methodToContext.put(method, getContextFor(method));
504                }
505                
506                // Propagate
507                Date start = new Date();
508                if(dfa.printDebug())
509                        G.v().out.println("CLOA: Starting Propagation at " + start);
510                while(worklist.size() > 0)
511                {
512                        ArrayList<SootMethod> newWorklist = new ArrayList<SootMethod>();
513                        for (SootMethod containingMethod : worklist) {
514                                CallLocalityContext containingContext = methodToContext.get(containingMethod);
515
516                                if(dfa.printDebug())
517                                        G.v().out.println("      " + containingMethod.getName() + " " + containingContext.toShortString());
518                                
519                                // Calculate the context for each invoke stmt in the containingMethod
520                                Map<Stmt, CallLocalityContext> invokeToContext = new HashMap<Stmt, CallLocalityContext>();
521                                for(Iterator edgesIt = Scene.v().getCallGraph().edgesOutOf(containingMethod); edgesIt.hasNext(); )
522                                {
523                                        Edge e = (Edge) edgesIt.next();
524                                        if( !e.src().getDeclaringClass().isApplicationClass() || e.srcStmt() == null )
525                                                continue;
526                                        CallLocalityContext invokeContext;
527                                        if( !invokeToContext.containsKey(e.srcStmt()) )
528                                        {
529                                                invokeContext = getContextFor(e, containingMethod, containingContext);
530                                                invokeToContext.put(e.srcStmt(), invokeContext);
531                                        }
532                                        else
533                                        {
534                                                invokeContext = invokeToContext.get(e.srcStmt());
535                                        }
536                                        
537                                        if( !methodToContext.containsKey(e.tgt()) )
538                                        {
539                                                methodToContext.put(e.tgt(), invokeContext);
540                                                newWorklist.add(e.tgt());
541                                        }
542                                        else
543                                        {
544//                                              G.v().out.println("        Merging Contexts for " + e.tgt());
545                                                boolean causedChange = methodToContext.get(e.tgt()).merge(invokeContext); // The contexts being merged could be from different DFAs.  If so, primitive version might be bigger.
546                                                if( causedChange )
547                                                        newWorklist.add(e.tgt());
548                                        }
549                                }
550                        }
551                        worklist = newWorklist;
552                }
553        long longTime = ((new Date()).getTime() - start.getTime()) / 100;
554        float time = (longTime) / 10.0f;
555                if(dfa.printDebug())
556                        G.v().out.println("CLOA: Ending Propagation after " + time + "s");
557        }
558        
559        public CallLocalityContext getMergedContext(SootMethod method)
560        {
561                if(methodToContext.containsKey(method))
562                        return methodToContext.get(method);
563                
564                return null;
565        }

567        private CallLocalityContext getContextFor(Edge e, SootMethod containingMethod, CallLocalityContext containingContext)
568        {
569                // get new called method and calling context
570                InvokeExpr ie;
571                if(e.srcStmt().containsInvokeExpr())
572                        ie = e.srcStmt().getInvokeExpr();
573                else
574                        ie = null;
575                        
576                SootMethod callingMethod = e.tgt();
577                CallLocalityContext callingContext = new CallLocalityContext(dfa.getMethodInfoFlowSummary(callingMethod).getNodes()); // just keeps a map from NODE to SHARED/LOCAL
578                
579                // We will use the containing context that we have to determine if base/args are local
580                if(callingMethod.isConcrete())
581                {
582                        Body b = containingMethod.retrieveActiveBody();
583                
584                        // check base
585                        if(ie != null && ie instanceof InstanceInvokeExpr)
586                        {
587                                InstanceInvokeExpr iie = (InstanceInvokeExpr) ie;
588                                if( !containingMethod.isStatic() && iie.getBase().equivTo(b.getThisLocal()) )
589                                {
590                                        // calling another method on same object... basically copy the previous context
591                                        Iterator<Object> localRefsIt = containingContext.getLocalRefs().iterator();
592                                        while(localRefsIt.hasNext())
593                                        {
594                                                EquivalentValue rEqVal = (EquivalentValue) localRefsIt.next();
595                                                Ref r = (Ref) rEqVal.getValue();
596                                                if(r instanceof InstanceFieldRef)
597                                                {
598                                                        EquivalentValue newRefEqVal = InfoFlowAnalysis.getNodeForFieldRef(callingMethod, ((FieldRef) r).getFieldRef().resolve());
599                                                        if(callingContext.containsField(newRefEqVal)) // if not, then we're probably calling a parent class's method, so some fields are missing
600                                                                callingContext.setFieldLocal(newRefEqVal); // must make a new eqval for the method getting called
601                                                }
602                                                else if(r instanceof ThisRef)
603                                                        callingContext.setThisLocal();
604                                        }
605                                }
606                                else if( SmartMethodLocalObjectsAnalysis.isObjectLocal(dfa, containingMethod, containingContext, iie.getBase()) )
607                                {
608                                        // calling a method on a local object
609                                        callingContext.setAllFieldsLocal();
610                                        callingContext.setThisLocal();
611                                }
612                                else
613                                {
614                                        // calling a method on a shared object
615                                        callingContext.setAllFieldsShared();
616                                        callingContext.setThisShared();
617                                }
618                        }
619                        else
620                        {
621                                callingContext.setAllFieldsShared();
622                                callingContext.setThisShared();
623                        }
624                        
625                        // check args
626                        if(ie == null)
627                                callingContext.setAllParamsShared();
628                        else
629                        {
630                                for(int param = 0; param < ie.getArgCount(); param++)
631                                {
632                                        if( SmartMethodLocalObjectsAnalysis.isObjectLocal(dfa, containingMethod, containingContext, ie.getArg(param)) )
633                                                callingContext.setParamLocal(param);
634                                        else
635                                                callingContext.setParamShared(param);
636                                }
637                        }
638                }
639                else
640                {
641                        // The only conservative solution for a bodyless method is to assume everything is shared
642                        callingContext.setAllFieldsShared();
643                        callingContext.setThisShared();                         
644                        callingContext.setAllParamsShared();
645                }
646                return callingContext;
647        }

649        public CallLocalityContext getContextFor(SootMethod sm) { return getContextFor(sm, false); }
650        private CallLocalityContext getContextFor(SootMethod sm, boolean includePrimitiveDataFlowIfAvailable)
651        {
652                CallLocalityContext context;
653                if(includePrimitiveDataFlowIfAvailable)
654                        context = new CallLocalityContext(primitiveDfa.getMethodInfoFlowSummary(sm).getNodes());
655                else
656                        context = new CallLocalityContext(dfa.getMethodInfoFlowSummary(sm).getNodes());
657                
658                // Set context for every parameter that is shared
659                for(int i = 0; i < sm.getParameterCount(); i++) // no need to worry about return value... 
660                {
661                        EquivalentValue paramEqVal = InfoFlowAnalysis.getNodeForParameterRef(sm, i);
662                        if(parameterIsLocal(sm, paramEqVal, includePrimitiveDataFlowIfAvailable))
663                        {
664                                context.setParamLocal(i);
665                        }
666                        else
667                        {
668                                context.setParamShared(i);
669                        }
670                }
671                
672                for (SootField sf : getLocalFields()) {
673                        EquivalentValue fieldRefEqVal = InfoFlowAnalysis.getNodeForFieldRef(sm, sf);
674                        context.setFieldLocal(fieldRefEqVal);
675                }
676                
677                for (SootField sf : getSharedFields()) {
678                        EquivalentValue fieldRefEqVal = InfoFlowAnalysis.getNodeForFieldRef(sm, sf);
679                        context.setFieldShared(fieldRefEqVal);
680                }
681                return context;
682        }

684        public boolean isObjectLocal(Value localOrRef, SootMethod sm) { return isObjectLocal(localOrRef, sm, false); }  
685        private boolean isObjectLocal(Value localOrRef, SootMethod sm, boolean includePrimitiveDataFlowIfAvailable)
686        {
687                if(localOrRef instanceof StaticFieldRef)
688                {
689                        return false;
690                }
691                
692                if(dfa.printDebug())
693                        G.v().out.println("      CLOA testing if " + localOrRef + " is local in " + sm);
694
695                SmartMethodLocalObjectsAnalysis smloa = getMethodLocalObjectsAnalysis(sm, includePrimitiveDataFlowIfAvailable);
696                if(localOrRef instanceof InstanceFieldRef)
697                {
698                        InstanceFieldRef ifr = (InstanceFieldRef) localOrRef;
699                        if( ifr.getBase().equivTo(smloa.getThisLocal()) )
700                                return isFieldLocal(ifr.getFieldRef().resolve());
701                        else
702                        {
703                                // if referred object is local, then find out if field is local in that object
704                                if(isObjectLocal(ifr.getBase(), sm, includePrimitiveDataFlowIfAvailable))
705                                {
706                                        boolean retval = loa.isFieldLocalToParent(ifr.getFieldRef().resolve());
707                                        if(dfa.printDebug())
708                                                G.v().out.println("      " + (retval ? "local" : "shared"));
709                                        return retval;
710                                }
711                                else
712                                {
713                                        if(dfa.printDebug())
714                                                G.v().out.println("      shared");
715                                        return false;
716                                }
717                        }
718                }
719                // TODO Prepare a CallLocalityContext!
720                CallLocalityContext context = getContextFor(sm);
721                
722                boolean retval = smloa.isObjectLocal(localOrRef, context);
723                if(dfa.printDebug())
724                        G.v().out.println("      " + (retval ? "local" : "shared"));
725                return retval;
726        }

728        public SmartMethodLocalObjectsAnalysis getMethodLocalObjectsAnalysis(SootMethod sm) { return getMethodLocalObjectsAnalysis(sm, false); }
729        private SmartMethodLocalObjectsAnalysis getMethodLocalObjectsAnalysis(SootMethod sm, boolean includePrimitiveDataFlowIfAvailable)
730        {
731                if(includePrimitiveDataFlowIfAvailable && primitiveDfa != null)
732                {
733                        Body b = sm.retrieveActiveBody();
734                        UnitGraph g = new ExceptionalUnitGraph(b);
735                        return new SmartMethodLocalObjectsAnalysis(g, primitiveDfa);
736                }
737                else if(!methodToMethodLocalObjectsAnalysis.containsKey(sm))
738                {
739                        // Analyze this method
740                        Body b = sm.retrieveActiveBody();
741                        UnitGraph g = new ExceptionalUnitGraph(b);
742                        SmartMethodLocalObjectsAnalysis smloa = new SmartMethodLocalObjectsAnalysis(g, dfa);
743                        methodToMethodLocalObjectsAnalysis.put(sm, smloa);
744                }
745                return methodToMethodLocalObjectsAnalysis.get(sm);
746        }

791        public boolean parameterIsLocal(SootMethod method, EquivalentValue parameterRef) { return parameterIsLocal(method, parameterRef, false); }
792        protected boolean parameterIsLocal(SootMethod method, EquivalentValue parameterRef, boolean includePrimitiveDataFlowIfAvailable)
793        {
794                if(dfa.printDebug() && method.getDeclaringClass().isApplicationClass())
795                        G.v().out.println("        Checking PARAM " + parameterRef + " for " + method);
796                        
797                // Check if param is primitive or ref type
798                ParameterRef param = (ParameterRef) parameterRef.getValue();
799                if( !(param.getType() instanceof RefLikeType) && (!dfa.includesPrimitiveInfoFlow() || method.getName().equals("<init>")) ) // TODO fix
800                {
801                        if(dfa.printDebug() && method.getDeclaringClass().isApplicationClass())
802                                G.v().out.println("          PARAM is local (primitive)");
803                        return true; // primitive params are always considered local
804                }
805                
806                // Check if method is externally called
807                List extClassCalls = uf.getExtCalls(sootClass);
808                Iterator extClassCallsIt = extClassCalls.iterator();
809                while(extClassCallsIt.hasNext())
810                {
811                        Pair extCall = (Pair) extClassCallsIt.next();
812                        Stmt s = (Stmt) extCall.getO2();
813                        if(s.getInvokeExpr().getMethodRef().resolve() == method)
814                        {
815                                if(dfa.printDebug() && method.getDeclaringClass().isApplicationClass())
816                                        G.v().out.println("          PARAM is shared (external access)");
817                                return false; // If so, assume it's params are shared
818                        }
819                }
820                
821                // For each internal call, check if arg is local or shared
822                List intClassCalls = uf.getIntCalls(sootClass);
823                Iterator intClassCallsIt = intClassCalls.iterator(); // returns all internal accesses
824                while(intClassCallsIt.hasNext())
825                {
826                        Pair intCall = (Pair) intClassCallsIt.next();
827                        SootMethod containingMethod = (SootMethod) intCall.getO1();
828                        Stmt s = (Stmt) intCall.getO2();
829                        InvokeExpr ie = s.getInvokeExpr();
830                        if(ie.getMethodRef().resolve() == method)
831                        {
832                                if(((ParameterRef) parameterRef.getValue()).getIndex() >= 0)
833                                {
834                                        if(!isObjectLocal( ie.getArg( ((ParameterRef) parameterRef.getValue()).getIndex() ), containingMethod, includePrimitiveDataFlowIfAvailable)) // WORST CASE SCENARIO HERE IS INFINITE RECURSION!
835                                        {
836                                                if(dfa.printDebug() && method.getDeclaringClass().isApplicationClass())
837                                                        G.v().out.println("          PARAM is shared (internal propagation)");
838                                                return false; // if arg is shared for any internal call, then param is shared
839                                        }
840                                }
841                                else
842                                {
843                                        if(s instanceof DefinitionStmt)
844                                        {
845                                                Value obj = ((DefinitionStmt) s).getLeftOp();
846                                                if(!isObjectLocal( obj, containingMethod, includePrimitiveDataFlowIfAvailable)) // WORST CASE SCENARIO HERE IS INFINITE RECURSION!
847                                                {
848                                                        if(dfa.printDebug() && method.getDeclaringClass().isApplicationClass())
849                                                                G.v().out.println("          PARAM is shared (internal propagation)");
850                                                        return false; // if arg is shared for any internal call, then param is shared
851                                                }
852                                        }
853                                }
854                        }
855                }
856                if(dfa.printDebug() && method.getDeclaringClass().isApplicationClass())
857                        G.v().out.println("          PARAM is local SO FAR (internal propagation)");
858                return true; // if argument is always local, then parameter is local
859        }
