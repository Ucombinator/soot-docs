011// LocalObjectsAnalysis written by Richard L. Halpert, 2007-02-24
012// Constructs data flow tables for each method of every application class.  Ignores indirect flow.
013// These tables conservatively approximate how data flows from parameters,
014// fields, and globals to parameters, fields, globals, and the return value.
015// Note that a ref-type parameter (or field or global) might allow access to a
016// large data structure, but that entire structure will be represented only by
017// the parameter's one node in the data flow graph.
018// Provides a high level interface to access the data flow information.
019

042        public ClassLocalObjectsAnalysis getClassLocalObjectsAnalysis(SootClass sc)
043        {
044                if(!classToClassLocalObjectsAnalysis.containsKey(sc))
045                {
046                        ClassLocalObjectsAnalysis cloa = newClassLocalObjectsAnalysis(this, dfa, uf, sc);
047                        classToClassLocalObjectsAnalysis.put(sc, cloa);
048                }
049                return classToClassLocalObjectsAnalysis.get(sc);
050        }

058        public boolean isObjectLocalToParent(Value localOrRef, SootMethod sm)
059        {
060                // Handle obvious case
061                if( localOrRef instanceof StaticFieldRef )
062                        return false;
063
064                ClassLocalObjectsAnalysis cloa = getClassLocalObjectsAnalysis(sm.getDeclaringClass());
065                return cloa.isObjectLocal(localOrRef, sm);
066        }
067        
068        public boolean isFieldLocalToParent(SootField sf) // To parent class!
069        {
070                // Handle obvious case
071                if( sf.isStatic() )
072                        return false;
073
074                ClassLocalObjectsAnalysis cloa = getClassLocalObjectsAnalysis(sf.getDeclaringClass());
075                return cloa.isFieldLocal(sf);
076        }

536        // returns a list of all methods that can be invoked on an object of type sc
537        public List<SootMethod> getAllMethodsForClass(SootClass sootClass)
538        {
539                // Determine which methods are reachable in this program
540                ReachableMethods rm = Scene.v().getReachableMethods();
541
542                // Get list of reachable methods declared in this class
543                // Also get list of fields declared in this class
544                List<SootMethod> scopeMethods = new ArrayList<SootMethod>();
545                Iterator scopeMethodsIt = sootClass.methodIterator();
546                while(scopeMethodsIt.hasNext())
547                {
548                        SootMethod scopeMethod = (SootMethod) scopeMethodsIt.next();
549                        if(rm.contains(scopeMethod))
550                                scopeMethods.add(scopeMethod);
551                }
552                
553                // Add reachable methods and fields declared in superclasses
554                SootClass superclass = sootClass;
555                if(superclass.hasSuperclass())
556                        superclass = sootClass.getSuperclass();
557                while(superclass.hasSuperclass()) // we don't want to process Object
558                {
559                Iterator scMethodsIt = superclass.methodIterator();
560                while(scMethodsIt.hasNext())
561                {
562                                SootMethod scMethod = (SootMethod) scMethodsIt.next();
563                                if(rm.contains(scMethod))
564                                        scopeMethods.add(scMethod);
565                }
566                        superclass = superclass.getSuperclass();
567                }
568                return scopeMethods;
569        }

571        public boolean hasNonLocalEffects(SootMethod containingMethod, InvokeExpr ie, SootMethod context)
572        {
573                SootMethod target = ie.getMethodRef().resolve();
574                MutableDirectedGraph dataFlowGraph = dfa.getMethodInfoFlowSummary(target); // TODO actually we want a graph that is sensitive to scalar data, too
575                
576                // For a static invoke, check if any fields or any shared params are read/written
577                if(ie instanceof StaticInvokeExpr)
578                {
579                        Iterator graphIt = dataFlowGraph.iterator();
580                        while(graphIt.hasNext())
581                        {
582                                EquivalentValue nodeEqVal = (EquivalentValue) graphIt.next();
583                                Ref node = (Ref) nodeEqVal.getValue();
584                                if(node instanceof FieldRef)
585                                {
586                                        if( dataFlowGraph.getPredsOf(nodeEqVal).size() > 0 ||
587                                                dataFlowGraph.getSuccsOf(nodeEqVal).size() > 0 )
588                                        {
589                                                return true;
590                                        }
591                                }
592                                else if(node instanceof ParameterRef)
593                                {
594                                        if( dataFlowGraph.getPredsOf(nodeEqVal).size() > 0 ||
595                                                dataFlowGraph.getSuccsOf(nodeEqVal).size() > 0 )
596                                        {
597                                                ParameterRef pr = (ParameterRef) node;
598                                                if(pr.getIndex() != -1)
599                                                {
600                                                        if( !isObjectLocalToContext(ie.getArg(pr.getIndex()), containingMethod, context) )
601                                                                return true;
602                                                }
603                                        }
604                                }
605                        }
606                }
607                else if(ie instanceof InstanceInvokeExpr)
608                {
609                        // For a instance invoke on local object, check if any static fields or any shared params are read/written
610                        InstanceInvokeExpr iie = (InstanceInvokeExpr) ie;
611                        if( isObjectLocalToContext(iie.getBase(), containingMethod, context) )
612                        {
613                                Iterator graphIt = dataFlowGraph.iterator();
614                                while(graphIt.hasNext())
615                                {
616                                        EquivalentValue nodeEqVal = (EquivalentValue) graphIt.next();
617                                        Ref node = (Ref) nodeEqVal.getValue();
618                                        if(node instanceof StaticFieldRef)
619                                        {
620                                                if( dataFlowGraph.getPredsOf(nodeEqVal).size() > 0 ||
621                                                        dataFlowGraph.getSuccsOf(nodeEqVal).size() > 0 )
622                                                {
623                                                        return true;
624                                                }
625                                        }
626                                        else if(node instanceof ParameterRef)
627                                        {
628                                                if( dataFlowGraph.getPredsOf(nodeEqVal).size() > 0 ||
629                                                        dataFlowGraph.getSuccsOf(nodeEqVal).size() > 0 )
630                                                {
631                                                        ParameterRef pr = (ParameterRef) node;
632                                                        if(pr.getIndex() != -1)
633                                                        {
634                                                                if( !isObjectLocalToContext(ie.getArg(pr.getIndex()), containingMethod, context) )
635                                                                        return true;
636                                                        }
637                                                }
638                                        }
639                                }
640                        }
641                        // For a instance invoke on shared object, check if any fields or any shared params are read/written
642                        else
643                        {
644                                Iterator graphIt = dataFlowGraph.iterator();
645                                while(graphIt.hasNext())
646                                {
647                                        EquivalentValue nodeEqVal = (EquivalentValue) graphIt.next();
648                                        Ref node = (Ref) nodeEqVal.getValue();
649                                        if(node instanceof FieldRef)
650                                        {
651                                                if( dataFlowGraph.getPredsOf(nodeEqVal).size() > 0 ||
652                                                        dataFlowGraph.getSuccsOf(nodeEqVal).size() > 0 )
653                                                {
654                                                        return true;
655                                                }
656                                        }
657                                        else if(node instanceof ParameterRef)
658                                        {
659                                                if( dataFlowGraph.getPredsOf(nodeEqVal).size() > 0 ||
660                                                        dataFlowGraph.getSuccsOf(nodeEqVal).size() > 0 )
661                                                {
662                                                        ParameterRef pr = (ParameterRef) node;
663                                                        if(pr.getIndex() != -1)
664                                                        {
665                                                                if( !isObjectLocalToContext(ie.getArg(pr.getIndex()), containingMethod, context) )
666                                                                        return true;
667                                                        }
668                                                }
669                                        }
670                                }
671                        }
672                }
673                return false;
674        }
675}
