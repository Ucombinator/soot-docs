009// ClassInfoFlowAnalysis written by Richard L. Halpert, 2007-02-22
010// Constructs data flow tables for each method of a class.  Ignores indirect flow.
011// These tables conservatively approximate how data flows from parameters,
012// fields, and globals to parameters, fields, globals, and the return value.
013// Note that a ref-type parameter (or field or global) might allow access to a
014// large data structure, but that entire structure will be represented only by
015// the parameter's one node in the data flow graph.
016
017public class ClassInfoFlowAnalysis
018{
019        SootClass sootClass;
020        InfoFlowAnalysis dfa; // used to access the data flow analyses of other classes
021        
022        Map<SootMethod, SmartMethodInfoFlowAnalysis> methodToInfoFlowAnalysis;
023        Map<SootMethod, HashMutableDirectedGraph<EquivalentValue>> methodToInfoFlowSummary;

037        public SmartMethodInfoFlowAnalysis getMethodInfoFlowAnalysis(SootMethod method)
038        {
039                if(!methodToInfoFlowAnalysis.containsKey(method))
040                {
041                        methodCount++;
042
043                        // First do simple version that doesn't follow invoke expressions
044                        // The "smart" version will be computed later, but since it may
045                        // request its own DataFlowGraph, we need this simple version first.
046                        if(!methodToInfoFlowSummary.containsKey(method))
047                        {
048                                HashMutableDirectedGraph<EquivalentValue> dataFlowGraph =
049                                                simpleConservativeInfoFlowAnalysis(method);
050                                methodToInfoFlowSummary.put(method, dataFlowGraph);
051                        }
052                        
053                        // Then do smart version that does follow invoke expressions, if possible
054                        if(method.isConcrete())
055                        {
056                                Body b = method.retrieveActiveBody();
057                                UnitGraph g = new ExceptionalUnitGraph(b);
058                                SmartMethodInfoFlowAnalysis smdfa = new SmartMethodInfoFlowAnalysis(g, dfa);
059
060                                methodToInfoFlowAnalysis.put(method, smdfa);
061                                methodToInfoFlowSummary.remove(method);
062                                methodToInfoFlowSummary.put(method, smdfa.getMethodInfoFlowSummary());
063                                return smdfa;
064//                              G.v().out.println(method + " has SMART infoFlowGraph: ");
065//                              printDataFlowGraph(mdfa.getMethodDataFlowGraph());
066                        }
067                }
068
069                return methodToInfoFlowAnalysis.get(method);
070        }

074        {

077                        methodCount++;

082                        HashMutableDirectedGraph<EquivalentValue> dataFlowGraph =
083                                        simpleConservativeInfoFlowAnalysis(method);
084                        methodToInfoFlowSummary.put(method, dataFlowGraph);
085                        
086                        // Then do smart version that does follow invoke expressions, if possible
087                        if(method.isConcrete() && doFullAnalysis)// && method.getDeclaringClass().isApplicationClass())
088                        {
089                                Body b = method.retrieveActiveBody();
090                                UnitGraph g = new ExceptionalUnitGraph(b);
091                                SmartMethodInfoFlowAnalysis smdfa = new SmartMethodInfoFlowAnalysis(g, dfa);
092
093                                methodToInfoFlowAnalysis.put(method, smdfa);
094                                methodToInfoFlowSummary.remove(method);
095                                methodToInfoFlowSummary.put(method, smdfa.getMethodInfoFlowSummary());
096
097//                              G.v().out.println(method + " has SMART infoFlowGraph: ");
098//                              printDataFlowGraph(mdfa.getMethodDataFlowGraph());
099                        }
100                }
101
102                return methodToInfoFlowSummary.get(method);
103        }

159        /** Does not require any fixed point calculation */
160        private HashMutableDirectedGraph<EquivalentValue> simpleConservativeInfoFlowAnalysis(SootMethod sm)
161        {
162                // Constructs a graph representing the data flow between fields, parameters, and the
163                // return value of this method.  The graph nodes are EquivalentValue wrapped Refs.
164                // This version is rather stupid... it just assumes all values flow to all others,
165                // except for the return value, which is flowed to by all, but flows to none.
166                
167                // This version is also broken... it can't handle the ThisRef without
168                // flow sensitivity.
169                
170                // If this method cannot have a body, then we can't analyze it... 
171                if(!sm.isConcrete())
172                        return triviallyConservativeInfoFlowAnalysis(sm);
173                        
174                Body b = sm.retrieveActiveBody();
175                UnitGraph g = new ExceptionalUnitGraph(b);
176                HashSet<EquivalentValue> fieldsStaticsParamsAccessed = new HashSet<EquivalentValue>();              
177
178                // Get list of fields, globals, and parameters that are accessed
179                for (Unit u : g)
180                {
181                        Stmt s = (Stmt) u;
182                        if( s instanceof IdentityStmt )
183                        {
184                                IdentityStmt is = (IdentityStmt) s;
185                                IdentityRef ir = (IdentityRef) is.getRightOp();
186                                if( ir instanceof ParameterRef )
187                                {
188                                        ParameterRef pr = (ParameterRef) ir;
189                                        fieldsStaticsParamsAccessed.add(InfoFlowAnalysis.getNodeForParameterRef(sm, pr.getIndex()));
190                                }
191                        }
192                        if(s.containsFieldRef())
193                        {
194                                FieldRef ref = s.getFieldRef();
195                                if( ref instanceof StaticFieldRef )
196                                {
197                                        // This should be added to the list of fields accessed
198                                        // static fields "belong to everyone"
199                                        StaticFieldRef sfr = (StaticFieldRef) ref;
200                                        fieldsStaticsParamsAccessed.add(InfoFlowAnalysis.getNodeForFieldRef(sm, sfr.getField()));
201                                }
202                                else if( ref instanceof InstanceFieldRef )
203                                {
204                                        // If this field is a field of this class,
205                                        // then this should be added to the list of fields accessed
206                                        InstanceFieldRef ifr = (InstanceFieldRef) ref;
207                                        Value base = ifr.getBase();
208                                        if(base instanceof Local)
209                                        {
210                                                if( dfa.includesInnerFields() || ((!sm.isStatic()) && base.equivTo(b.getThisLocal())) )
211                                                        fieldsStaticsParamsAccessed.add(InfoFlowAnalysis.getNodeForFieldRef(sm, ifr.getField()));
212                            }
213                                }
214                        }
215                }
216                
217                // Each accessed field, global, and parameter becomes a node in the graph
218                HashMutableDirectedGraph<EquivalentValue> dataFlowGraph = new MemoryEfficientGraph<EquivalentValue>();
219                Iterator<EquivalentValue> accessedIt1 = fieldsStaticsParamsAccessed.iterator();
220                while(accessedIt1.hasNext())
221                {
222                        EquivalentValue o = accessedIt1.next();
223                        dataFlowGraph.addNode(o);
224                }
225                
226                // Add all of the nodes necessary to ensure that this is a complete data flow graph
227                // Add every parameter of this method
228                for(int i = 0; i < sm.getParameterCount(); i++)
229                {
230                        EquivalentValue parameterRefEqVal = InfoFlowAnalysis.getNodeForParameterRef(sm, i);
231                        if(!dataFlowGraph.containsNode(parameterRefEqVal))
232                                dataFlowGraph.addNode(parameterRefEqVal);
233                }
234                
235                // Add every relevant field of this class (static methods don't get non-static fields)
236                for (SootField sf : sm.getDeclaringClass().getFields())
237                {
238                        if(sf.isStatic() || !sm.isStatic())
239                        {
240                                EquivalentValue fieldRefEqVal = InfoFlowAnalysis.getNodeForFieldRef(sm, sf);
241                                if(!dataFlowGraph.containsNode(fieldRefEqVal))
242                                        dataFlowGraph.addNode(fieldRefEqVal);
243                        }
244                }
245                
246                // Add every field of this class's superclasses
247                SootClass superclass = sm.getDeclaringClass();
248                if(superclass.hasSuperclass())
249                        superclass = sm.getDeclaringClass().getSuperclass();
250                while(superclass.hasSuperclass()) // we don't want to process Object
251                {
252                        for (SootField scField : superclass.getFields())
253                {
254                                if(scField.isStatic() || !sm.isStatic())
255                                {
256                                        EquivalentValue fieldRefEqVal = InfoFlowAnalysis.getNodeForFieldRef(sm, scField);
257                                        if(!dataFlowGraph.containsNode(fieldRefEqVal))
258                                                dataFlowGraph.addNode(fieldRefEqVal);
259                                }
260                }
261                        superclass = superclass.getSuperclass();
262                }
263                
264                // The return value also becomes a node in the graph
265                ParameterRef returnValueRef = null;
266                if(sm.getReturnType() != VoidType.v())
267                {
268                        returnValueRef = new ParameterRef(sm.getReturnType(), -1);
269                        dataFlowGraph.addNode(InfoFlowAnalysis.getNodeForReturnRef(sm));
270                }
271                
272//              ThisRef thisRef = null;
273                if(!sm.isStatic())
274                {
275//                      thisRef = new ThisRef(sootClass.getType());
276                        dataFlowGraph.addNode(InfoFlowAnalysis.getNodeForThisRef(sm));
277                        fieldsStaticsParamsAccessed.add(InfoFlowAnalysis.getNodeForThisRef(sm));
278                }
279                
280                // Create an edge from each node (except the return value) to every other node (including the return value)
281                // non-Ref-type nodes are ignored
282                accessedIt1 = fieldsStaticsParamsAccessed.iterator();
283                while(accessedIt1.hasNext())
284                {
285                        EquivalentValue r = accessedIt1.next();
286                        Ref rRef = (Ref) r.getValue();
287                        if( !(rRef.getType() instanceof RefLikeType) && !dfa.includesPrimitiveInfoFlow())
288                                continue;
289                        Iterator<EquivalentValue> accessedIt2 = fieldsStaticsParamsAccessed.iterator();
290                        while(accessedIt2.hasNext())
291                        {
292                                EquivalentValue s = accessedIt2.next();
293                                Ref sRef = (Ref) s.getValue();
294                                if( rRef instanceof ThisRef && sRef instanceof InstanceFieldRef )
295                                        ; // don't add this edge
296                                else if( sRef instanceof ThisRef && rRef instanceof InstanceFieldRef )
297                                        ; // don't add this edge
298                                else if( sRef instanceof ParameterRef && dfa.includesInnerFields())
299                                        ; // don't add edges to parameters if we are including inner fields
300                                else if( sRef.getType() instanceof RefLikeType )
301                                        dataFlowGraph.addEdge(r, s);
302                        }
303                        if( returnValueRef != null && (returnValueRef.getType() instanceof RefLikeType || dfa.includesPrimitiveInfoFlow()))
304                                dataFlowGraph.addEdge(r, InfoFlowAnalysis.getNodeForReturnRef(sm));
305                }
306                
307                return dataFlowGraph;
308        }

310        /** Does not require the method to have a body */
311        public HashMutableDirectedGraph<EquivalentValue> triviallyConservativeInfoFlowAnalysis(SootMethod sm)
312        {
313                HashSet<EquivalentValue> fieldsStaticsParamsAccessed = new HashSet<EquivalentValue>();
314                
315                // Add all of the nodes necessary to ensure that this is a complete data flow graph
316                // Add every parameter of this method
317                for(int i = 0; i < sm.getParameterCount(); i++)
318                {
319                        EquivalentValue parameterRefEqVal = InfoFlowAnalysis.getNodeForParameterRef(sm, i);
320                        fieldsStaticsParamsAccessed.add(parameterRefEqVal);
321                }
322                
323                // Add every relevant field of this class (static methods don't get non-static fields)
324                for(Iterator<SootField> it = sm.getDeclaringClass().getFields().iterator(); it.hasNext(); )
325                {
326                        SootField sf = it.next();
327                        if(sf.isStatic() || !sm.isStatic())
328                        {
329                                EquivalentValue fieldRefEqVal = InfoFlowAnalysis.getNodeForFieldRef(sm, sf);
330                                fieldsStaticsParamsAccessed.add(fieldRefEqVal);
331                        }
332                }
333                
334                // Add every field of this class's superclasses
335                SootClass superclass = sm.getDeclaringClass();
336                if(superclass.hasSuperclass())
337                        superclass = sm.getDeclaringClass().getSuperclass();
338                while(superclass.hasSuperclass()) // we don't want to process Object
339                {
340                Iterator<SootField> scFieldsIt = superclass.getFields().iterator();
341                while(scFieldsIt.hasNext())
342                {
343                                SootField scField = scFieldsIt.next();
344                                if(scField.isStatic() || !sm.isStatic())
345                                {
346                                        EquivalentValue fieldRefEqVal = InfoFlowAnalysis.getNodeForFieldRef(sm, scField);
347                                        fieldsStaticsParamsAccessed.add(fieldRefEqVal);
348                                }
349                }
350                        superclass = superclass.getSuperclass();
351                }
352                
353                // Don't add any static fields outside of the class... unsafe???
354
355                // Each field, global, and parameter becomes a node in the graph
356                HashMutableDirectedGraph<EquivalentValue> dataFlowGraph = new MemoryEfficientGraph<EquivalentValue>();
357                Iterator<EquivalentValue> accessedIt1 = fieldsStaticsParamsAccessed.iterator();
358                while(accessedIt1.hasNext())
359                {
360                        EquivalentValue o = accessedIt1.next();
361                        dataFlowGraph.addNode(o);
362                }
363                
364                // The return value also becomes a node in the graph
365                ParameterRef returnValueRef = null;
366                if(sm.getReturnType() != VoidType.v())
367                {
368                        returnValueRef = new ParameterRef(sm.getReturnType(), -1);
369                        dataFlowGraph.addNode(InfoFlowAnalysis.getNodeForReturnRef(sm));
370                }
371                
372                if(!sm.isStatic())
373                {
374                        dataFlowGraph.addNode(InfoFlowAnalysis.getNodeForThisRef(sm));
375                        fieldsStaticsParamsAccessed.add(InfoFlowAnalysis.getNodeForThisRef(sm));
376                }
377                
378                // Create an edge from each node (except the return value) to every other node (including the return value)
379                // non-Ref-type nodes are ignored
380                accessedIt1 = fieldsStaticsParamsAccessed.iterator();
381                while(accessedIt1.hasNext())
382                {
383                        EquivalentValue r = accessedIt1.next();
384                        Ref rRef = (Ref) r.getValue();
385                        if( !(rRef.getType() instanceof RefLikeType) && !dfa.includesPrimitiveInfoFlow() )
386                                continue;
387                        Iterator<EquivalentValue> accessedIt2 = fieldsStaticsParamsAccessed.iterator();
388                        while(accessedIt2.hasNext())
389                        {
390                                EquivalentValue s = accessedIt2.next();
391                                Ref sRef = (Ref) s.getValue();
392                                if( rRef instanceof ThisRef && sRef instanceof InstanceFieldRef )
393                                        ; // don't add this edge
394                                else if( sRef instanceof ThisRef && rRef instanceof InstanceFieldRef )
395                                        ; // don't add this edge
396                                else if( sRef.getType() instanceof RefLikeType )
397                                        dataFlowGraph.addEdge(r, s);
398                        }
399                        if( returnValueRef != null && (returnValueRef.getType() instanceof RefLikeType || dfa.includesPrimitiveInfoFlow()) )
400                                dataFlowGraph.addEdge(r, InfoFlowAnalysis.getNodeForReturnRef(sm));
401                }
402                
403                return dataFlowGraph;
404        }
