020package soot.jimple.toolkits.pointer;
021import java.util.*;
022import soot.toolkits.scalar.*;
023import soot.toolkits.graph.*;
024import soot.*;
025import soot.jimple.*;

027/** A flow analysis that detects redundant cast checks. */
028public class CastCheckEliminator extends ForwardBranchedFlowAnalysis<LocalTypeSet> {
029    Map unitToKill = new HashMap();
030    Map unitToGenFallThrough = new HashMap();
031    Map unitToGenBranch = new HashMap();
032    LocalTypeSet emptySet;
033
034    public CastCheckEliminator(BriefUnitGraph cfg) {
035        super(cfg);
036        makeInitialSet();
037        doAnalysis();
038        tagCasts();
039    }
040
041    /** Put the results of the analysis into tags in cast statements. */
042    protected void tagCasts() {
043        for( Iterator<Unit> sIt = ((UnitGraph)graph).getBody().getUnits().iterator(); sIt.hasNext(); ) {
044            final Stmt s = (Stmt) sIt.next();
045            if( s instanceof AssignStmt ) {
046                AssignStmt as = (AssignStmt) s;
047                Value rhs = as.getRightOp();
048                if( rhs instanceof CastExpr ) {
049                    CastExpr cast = (CastExpr) rhs;
050                    Type t = cast.getCastType();
051                    if( t instanceof RefType ) {
052                        if( cast.getOp() instanceof Local ) {
053                            Local l = (Local) cast.getOp(); 
054                            LocalTypeSet set = (LocalTypeSet) getFlowBefore(s);
055                            s.addTag( new CastCheckTag( set.get( set.indexOf(
056                                    l, (RefType) t ) ) ) );
057                        } else {
058                            NullConstant nc = (NullConstant) cast.getOp();
059                            s.addTag( new CastCheckTag( true ) );
060                        }
061                    }
062                }
063            }
064        }
065    }
066
067    /** Find all the locals of reference type and all the types used in casts
068     * to initialize the mapping from locals and types to bits in the bit vector
069     * in LocalTypeSet. */
070    protected void makeInitialSet() {
071        // Find all locals of reference type
072        Collection<Local> locals = ((UnitGraph)graph).getBody().getLocals();
073        List<Local> refLocals = new ArrayList<Local>();
074        for( Iterator<Local> lIt = locals.iterator(); lIt.hasNext(); ) {
075            final Local l = (Local) lIt.next();
076            if( l.getType() instanceof RefType ) {
077                refLocals.add( l );
078            }
079        }
080
081        // Find types of all casts
082        List<Type> types = new ArrayList<Type>();
083        for( Iterator<Unit> sIt = ((UnitGraph)graph).getBody().getUnits().iterator(); sIt.hasNext(); ) {
084            final Stmt s = (Stmt) sIt.next();
085            if( s instanceof AssignStmt ) {
086                AssignStmt as = (AssignStmt) s;
087                Value rhs = as.getRightOp();
088                if( rhs instanceof CastExpr ) {
089                    Type t = ( (CastExpr) rhs ).getCastType();
090                    if( t instanceof RefType && !types.contains( t ) ) {
091                        types.add( t );
092                    }
093                }
094            }
095        }
096        
097        emptySet = new LocalTypeSet( refLocals, types );
098    }
099    
100
101    /** Returns a new, aggressive (local,type) set. */
102    @Override
103    protected LocalTypeSet newInitialFlow() {
104        LocalTypeSet ret = (LocalTypeSet) emptySet.clone();
105        ret.setAllBits();
106        return ret;
107    }
108
109    /** This is the flow function as described in the assignment write-up. */
110    @Override
111    protected void flowThrough( LocalTypeSet in, Unit unit, List<LocalTypeSet> outFallValues,
112                                List<LocalTypeSet> outBranchValues ) 
113    {
114        final LocalTypeSet out = (LocalTypeSet) in.clone();
115        LocalTypeSet outBranch = out; // aliased to out unless unit is IfStmt
116        final Stmt stmt = (Stmt) unit;
117        
118        // First kill all locals defined in this statement
119        for( Iterator<ValueBox> bIt = stmt.getDefBoxes().iterator(); bIt.hasNext(); ) {
120            final ValueBox b = (ValueBox) bIt.next();
121            Value v = b.getValue();
122            if( v instanceof Local && v.getType() instanceof RefType ) {
123                out.killLocal( (Local) v );
124            }
125        }
126        
127        // An AssignStmt may be a new, a simple copy, or a cast
128        if( stmt instanceof AssignStmt ) {
129            AssignStmt astmt = (AssignStmt) stmt;
130            Value rhs = astmt.getRightOp();
131            Value lhs = astmt.getLeftOp();
132            if( lhs instanceof Local && rhs.getType() instanceof RefType ) {
133                Local l = (Local) lhs;
134                if( rhs instanceof NewExpr ) {
135                    out.localMustBeSubtypeOf( l, (RefType) rhs.getType() );
136                } else if( rhs instanceof CastExpr ) {
137                    CastExpr cast = (CastExpr) rhs;
138                    Type castType = cast.getCastType();
139                    if( castType instanceof RefType 
140                    &&  cast.getOp() instanceof Local ) {
141                        RefType refType = (RefType) castType;
142                        Local opLocal = (Local) cast.getOp();
143                        out.localCopy( l, opLocal );
144                        out.localMustBeSubtypeOf( l, refType );
145                        out.localMustBeSubtypeOf( opLocal, refType );
146                    }
147                } else if( rhs instanceof Local ) {
148                    out.localCopy( l, (Local) rhs );
149                }
150            }
151            
152            // Handle if statements
153        } else if( stmt instanceof IfStmt ) {
154            IfStmt ifstmt = (IfStmt) stmt;
155            
156            // This do ... while(false) is here so I can break out of it rather
157            // than having to have seven nested if statements. Silly people who
158            // took goto's out of the language... <grumble> <grumble>
159            do {
160                if( graph.getPredsOf( stmt ).size() != 1 ) break;
161                Object predecessor = graph.getPredsOf( stmt ).get(0);
162                if( !( predecessor instanceof AssignStmt ) ) break;
163                AssignStmt pred = (AssignStmt) predecessor;
164                if( !(pred.getRightOp() instanceof InstanceOfExpr ) ) break;
165                InstanceOfExpr iofexpr = (InstanceOfExpr) pred.getRightOp();
166                if( !(iofexpr.getCheckType() instanceof RefType ) ) break;
167                if( !(iofexpr.getOp() instanceof Local ) ) break;
168                ConditionExpr c = (ConditionExpr) ifstmt.getCondition();
169                if( !c.getOp1().equals( pred.getLeftOp() ) ) break;
170                if( !( c.getOp2() instanceof IntConstant ) ) break;
171                if( ( (IntConstant) c.getOp2() ).value != 0 ) break;
172                if( c instanceof NeExpr ) {
173                    // The IfStmt is like this:
174                    // if x instanceof t goto somewhere_else
175                    // So x is of type t on the taken branch
176                    outBranch = (LocalTypeSet) out.clone();
177                    outBranch.localMustBeSubtypeOf( (Local) iofexpr.getOp(),
178                                                    (RefType) iofexpr.getCheckType() );
179                } else if( c instanceof EqExpr ) {
180                    // The IfStmt is like this:
181                    // if !(x instanceof t) goto somewhere_else
182                    // So x is of type t on the fallthrough branch
183                    outBranch = (LocalTypeSet) out.clone();
184                    out.localMustBeSubtypeOf( (Local) iofexpr.getOp(),
185                                              (RefType) iofexpr.getCheckType() );
186                }
187            } while( false );
188        }
189        
190        // Now copy the computed (local,type) set to all successors
191        for( Iterator<LocalTypeSet> it = outFallValues.iterator(); it.hasNext(); ) {
192            copy( out, it.next() );
193        }
194        for( Iterator<LocalTypeSet> it = outBranchValues.iterator(); it.hasNext(); ) {
195            copy( outBranch, it.next() );
196        }
197    }
198
199    @Override
200    protected void copy( LocalTypeSet s, LocalTypeSet d ) {
201        d.and( s );
202        d.or( s );
203    }
204
205    // The merge operator is set intersection.
206    @Override
207    protected void merge( LocalTypeSet in1, LocalTypeSet in2, LocalTypeSet o ) {
208        o.setAllBits();
209        o.and( in1 );
210        o.and( in2 );
211    }
212    
213    /** Returns a new, aggressive (local,type) set. */
214    @Override
215    protected LocalTypeSet entryInitialFlow() {
216        LocalTypeSet ret = (LocalTypeSet) emptySet.clone();
217        return ret;
218    }
219}
