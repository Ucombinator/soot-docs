041    protected class UniqueRWSets {
042        protected ArrayList<RWSet> l = new ArrayList<RWSet>();
043        RWSet getUnique( RWSet s ) {
044            if( s == null ) return s;
045            for (RWSet ret : l) {
046                if( ret.isEquivTo( s ) ) return ret;
047            }
048            l.add( s );
049            return s;
050        }
051        Iterator<RWSet> iterator() {
052            return l.iterator();
053        }
054        short indexOf( RWSet s ) {
055            short i = 0;
056            for (RWSet ret : l) {
057                if( ret.isEquivTo( s ) ) return i;
058                i++;
059            }
060            return -1;
061        }
062    }

064    protected void initializationStuff( String phaseName ) {
065        if( G.v().Union_factory == null ) {
066            G.v().Union_factory = new UnionFactory() {
067                public Union newUnion() { return FullObjectSet.v(); }
068            };
069        }
070        if( startTime == null ) {
071            startTime = new Date();
072        }
073        cg = Scene.v().getCallGraph();
074    }
075    protected Object keyFor( Stmt s ) {
076        if( s.containsInvokeExpr() ) {
077            if( optionNaive ) throw new RuntimeException( "shouldn't get here" );
078            Iterator it = cg.edgesOutOf( s );
079            if( !it.hasNext() ) {
080                return Collections.EMPTY_LIST;
081            }
082            ArrayList ret = new ArrayList();
083            while( it.hasNext() ) {
084                ret.add( it.next() );
085            }
086            return ret;
087        } else {
088            return s;
089        }
090    }
