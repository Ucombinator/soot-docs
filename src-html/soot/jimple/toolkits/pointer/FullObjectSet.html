020package soot.jimple.toolkits.pointer;
021import java.util.*;
022
023import soot.*;
024import soot.jimple.ClassConstant;
025
026public class FullObjectSet extends Union {
027    public FullObjectSet( Singletons.Global g ) {
028        this( RefType.v( "java.lang.Object" ) );
029    }
030    public static FullObjectSet v() { return G.v().soot_jimple_toolkits_pointer_FullObjectSet(); }
031    public static FullObjectSet v( RefType t ) { 
032        if( t.getClassName().equals( "java.lang.Object" ) ) {
033            return v();
034        }
035        return new FullObjectSet( t );
036    }
037    private final Set<Type> types;
038    private FullObjectSet( RefType declaredType ) {
039        Type type = AnySubType.v( declaredType );
040        types = Collections.singleton( type );
041    }
042
043    public Type type() { return types.iterator().next(); }
044
045    /** Returns true if this set contains no run-time objects. */
046    public boolean isEmpty() {
047        return false;
048    }
049    /** Returns true if this set is a subset of other. */
050    public boolean hasNonEmptyIntersection( PointsToSet other ) {
051        return other != null;
052    }
053    
054    /** Set of all possible run-time types of objects in the set. */
055    @Override
056    public Set<Type> possibleTypes() {
057        return types;
058    }
059
060    /** Adds all objects in s into this union of sets, returning true if this
061     * union was changed. */
062    public boolean addAll( PointsToSet s ) {
063        return false;
064    }
065
066    public Set<String> possibleStringConstants() { return null; }
067    public Set<ClassConstant> possibleClassConstants() { return null; }
068        
069    /**
070         * {@inheritDoc}
071         */
072        public int depth() {
073                return 0;
074        }
075}
