036  /**
037   * Implements the abstract method simulateMethod.
038   * It distributes the request to the corresponding methods 
039   * by signatures.
040   */
041  public void simulateMethod(SootMethod method,
042                             ReferenceVariable thisVar,
043                             ReferenceVariable returnVar,
044                             ReferenceVariable params[]){
045
046    String subSignature = method.getSubSignature();
047
048    if (subSignature.equals("java.lang.Class forName0(java.lang.String,boolean,java.lang.ClassLoader)")) {
049      java_lang_Class_forName0(method, thisVar, returnVar, params);
050      return;
051
052    } else if (subSignature.equals("java.lang.Object newInstance0()")) {
053      java_lang_Class_newInstance0(method, thisVar, returnVar, params);
054      return;
055
056    } else if (subSignature.equals("java.lang.String getName()")) {
057      java_lang_Class_getName(method, thisVar, returnVar, params);
058      return;
059
060    } else if (subSignature.equals("java.lang.ClassLoader getClassLoader0()")){
061      java_lang_Class_getClassLoader0(method, thisVar, returnVar, params);
062      return;
063
064    } else if (subSignature.equals("java.lang.Class getSuperclass()")) {
065      java_lang_Class_getSuperclass(method, thisVar, returnVar, params);
066      return;
067
068    } else if (subSignature.equals("java.lang.Class[] getInterfaces()")){
069      java_lang_Class_getInterfaces(method, thisVar, returnVar, params);
070      return;
071
072    } else if (subSignature.equals("java.lang.Class getComponentType()")){
073      java_lang_Class_getComponentType(method, thisVar, returnVar, params);
074      return;
075
076    } else if (subSignature.equals("java.lang.Object[] getSigners()")){
077      java_lang_Class_getSigners(method, thisVar, returnVar, params);
078      return;
079
080    } else if (subSignature.equals("void setSigners(java.lang.Object[])")) {
081      java_lang_Class_setSigners(method, thisVar, returnVar, params);
082      return;
083
084    } else if (subSignature.equals("java.lang.Class getDeclaringClass()")){
085      java_lang_Class_getDeclaringClass(method, thisVar, returnVar, params);
086      return;
087
088    } else if (subSignature.equals("void setProtectionDomain0(java.security.ProtectionDomain)")){
089      java_lang_Class_setProtectionDomain0(method, thisVar, returnVar, params);
090      return;
091
092    } else if (subSignature.equals("java.security.ProtectionDomain getProtectionDomain0()")) {
093      java_lang_Class_getProtectionDomain0(method, thisVar, returnVar, params);
094      return;
095
096    } else if (subSignature.equals("java.lang.Class getPrimitiveClass(java.lang.String)")) {
097      java_lang_Class_getPrimitiveClass(method, thisVar, returnVar, params);
098      return;
099
100    } else if (subSignature.equals("java.lang.reflect.Field[] getFields0(int)")){
101      java_lang_Class_getFields0(method, thisVar, returnVar, params);
102      return;
103
104    } else if (subSignature.equals("java.lang.reflect.Method[] getMethods0(int)")){
105      java_lang_Class_getMethods0(method, thisVar, returnVar, params);
106      return;
107
108    } else if (subSignature.equals("java.lang.reflect.Constructor[] getConstructors0(int)")) {
109      java_lang_Class_getConstructors0(method, thisVar, returnVar, params);
110      return;
111
112    } else if (subSignature.equals("java.lang.reflect.Field getField0(java.lang.String,int)")) {
113      java_lang_Class_getField0(method, thisVar, returnVar, params);
114      return;
115
116    } else if (subSignature.equals("java.lang.reflect.Method getMethod0(java.lang.String,java.lang.Class[],int)")) {
117      java_lang_Class_getMethod0(method, thisVar, returnVar, params);
118      return;
119
120    } else if (subSignature.equals("java.lang.reflect.Constructor getConstructor0(java.lang.Class[],int)")) {
121      java_lang_Class_getConstructor0(method, thisVar, returnVar, params);
122      return;
123   
124    } else if (subSignature.equals("java.lang.Class[] getDeclaredClasses0()")){
125      java_lang_Class_getDeclaredClasses0(method, thisVar, returnVar, params);
126      return;
127
128    } else if (subSignature.equals("java.lang.reflect.Constructor[] getDeclaredConstructors0(boolean)")){
129      java_lang_Class_getDeclaredConstructors0(method, thisVar, returnVar, params);
130      return;
131
132    } else {
133      defaultMethod(method, thisVar, returnVar, params);
134      return;
135
136    }
137  }

140  /* A quick note for simulating java.lang.Class :
141   *
142   * In theory, the same class may have two or more representations 
143   * at the runtime. But statically, we can assume that all variables
144   * of java.lang.Class type are aliased together. By looking at
145   * static class hierarchy, there is only one ReferenceVariable 
146   * variable for a class in the hierarchy.
147   */
148
149  /**
150   * NOTE: the semantic of forName0 follows forName method.  
151   * 
152   * Returns the Class object associated with the class or interface
153   * with the given string name, using the given class loader. Given
154   * the fully qualified name for a class or interface (in the same
155   * format returned by getName) this method attempts to locate,
156   * load, and link the class or interface. The specified class
157   * loader is used to load the class or interface. If the parameter
158   * loader is null, the class is loaded through the bootstrap class
159   * loader. The class is initialized only if the initialize
160   * parameter is true and if it has not been initialized earlier.
161   *
162   * If name denotes a primitive type or void, an attempt will be made
163   * to locate a user-defined class in the unnamed package whose
164   * name is name. Therefore, this method cannot be used to obtain
165   * any of the Class objects representing primitive types or void.
166   * 
167   * If name denotes an array class, the component type of the array
168   * class is loaded but not initialized.
169   *
170   * For example, in an instance method the expression: 
171   *       Class.forName("Foo")       
172   * is equivalent to: 
173   *       Class.forName("Foo", true, this.getClass().getClassLoader()) 
174   *
175   * private static native java.lang.Class forName0(java.lang.String, 
176   *                                                boolean, 
177   *                                                java.lang.ClassLoader) 
178   *                                throws java.lang.ClassNotFoundException;
179   */
180  public void java_lang_Class_forName0(SootMethod method,
181                                              ReferenceVariable thisVar,
182                                              ReferenceVariable returnVar,
183                                              ReferenceVariable params[]){
184    helper.assignObjectTo(returnVar, Environment.v().getClassObject());
185  }
