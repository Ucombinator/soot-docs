111        @Override
112        protected void internalTransform(String phaseName, @SuppressWarnings("rawtypes") Map options) {
113                if(!initialized) {
114                        CGOptions cgOptions = new CGOptions( PhaseOptions.v().getPhaseOptions("cg") );
115                        String logFilePath = cgOptions.reflection_log();
116                        RTI = new ReflectionTraceInfo(logFilePath);
117
118                        Scene.v().getSootClass(SootSig.class.getName()).setApplicationClass();
119                        Scene.v().getSootClass(UnexpectedReflectiveCall.class.getName()).setApplicationClass();
120                        Scene.v().getSootClass(IUnexpectedReflectiveCallHandler.class.getName()).setApplicationClass();
121                        Scene.v().getSootClass(DefaultHandler.class.getName()).setApplicationClass();
122                        Scene.v().getSootClass(OpaquePredicate.class.getName()).setApplicationClass();
123                        Scene.v().getSootClass(ReflectiveCalls.class.getName()).setApplicationClass();
124                        
125                        reflectiveCallsClass = new SootClass("soot.rtlib.tamiflex.ReflectiveCallsWrapper", Modifier.PUBLIC);
126                        Scene.v().addClass(reflectiveCallsClass);
127                        reflectiveCallsClass.setApplicationClass();
128
129                        UNINTERPRETED_METHOD = Scene.v().makeMethodRef(Scene.v().getSootClass("soot.rtlib.tamiflex.OpaquePredicate"), "getFalse", Collections.<Type>emptyList(), BooleanType.v(), true);
130                        
131                        if(useCaching)
132                                addCaching();
133                        
134                        initializeReflectiveCallsTable();
135                        
136                        callSiteId = 0;
137                        callNum = 0;
138                        
139                        initialized = true;
140                }
141                
142                for(SootMethod m: RTI.methodsContainingReflectiveCalls()) {
143                        m.retrieveActiveBody();
144                        Body b = m.getActiveBody();
145                        {
146                                Set<String> classForNameClassNames = RTI.classForNameClassNames(m);
147                                if(!classForNameClassNames.isEmpty()) {
148                                        inlineRelectiveCalls(m,classForNameClassNames, ReflectionTraceInfo.Kind.ClassForName);
149                                        if(Options.v().validate()) b.validate();
150                                }
151                        }{
152                                Set<String> classNewInstanceClassNames = RTI.classNewInstanceClassNames(m);
153                                if(!classNewInstanceClassNames.isEmpty()) {
154                                        inlineRelectiveCalls(m,classNewInstanceClassNames, ReflectionTraceInfo.Kind.ClassNewInstance);
155                                        if(Options.v().validate()) b.validate();
156                                }
157                        }{
158                                Set<String> constructorNewInstanceSignatures = RTI.constructorNewInstanceSignatures(m);
159                                if(!constructorNewInstanceSignatures.isEmpty()) {
160                                        inlineRelectiveCalls(m, constructorNewInstanceSignatures, ReflectionTraceInfo.Kind.ConstructorNewInstance);
161                                        if(Options.v().validate()) b.validate();
162                                }
163                        }{
164                                Set<String> methodInvokeSignatures = RTI.methodInvokeSignatures(m);
165                                if(!methodInvokeSignatures.isEmpty()) {
166                                        inlineRelectiveCalls(m, methodInvokeSignatures, ReflectionTraceInfo.Kind.MethodInvoke);
167                                        if(Options.v().validate()) b.validate();
168                                }
169                        }{
170                                Set<String> fieldSetSignatures = RTI.fieldSetSignatures(m);
171                                if(!fieldSetSignatures.isEmpty()) {
172                                        inlineRelectiveCalls(m, fieldSetSignatures, ReflectionTraceInfo.Kind.FieldSet);
173                                        if(Options.v().validate()) b.validate();
174                                }
175                        }{
176                                Set<String> fieldGetSignatures = RTI.fieldGetSignatures(m);
177                                if(!fieldGetSignatures.isEmpty()) {
178                                        inlineRelectiveCalls(m, fieldGetSignatures, ReflectionTraceInfo.Kind.FieldGet);
179                                        if(Options.v().validate()) b.validate();
180                                }
181                        }
182                        //clean up after us
183                        cleanup(b);                     
184                }
185        }

187        private void cleanup(Body b) {
188                CopyPropagator.v().transform(b);
189                DeadAssignmentEliminator.v().transform(b);
190                UnusedLocalEliminator.v().transform(b);
191                NopEliminator.v().transform(b);
192        }

194        private void initializeReflectiveCallsTable() {
195                int callSiteId = 0;
196                
197                SootClass reflCallsClass = Scene.v().getSootClass("soot.rtlib.tamiflex.ReflectiveCalls");
198                SootMethod clinit = reflCallsClass.getMethodByName(SootMethod.staticInitializerName);
199                Body body = clinit.retrieveActiveBody();
200                PatchingChain<Unit> units = body.getUnits();
201                LocalGenerator localGen = new LocalGenerator(body);
202                Chain<Unit> newUnits = new HashChain<Unit>();
203                SootClass setClass = Scene.v().getSootClass("java.util.Set");
204                SootMethodRef addMethodRef = setClass.getMethodByName("add").makeRef();
205                for(SootMethod m: RTI.methodsContainingReflectiveCalls()) {
206                        {       
207                                if(!RTI.classForNameClassNames(m).isEmpty()) {
208                                        SootFieldRef fieldRef = Scene.v().makeFieldRef(reflCallsClass, "classForName", RefType.v("java.util.Set"), true);
209                                        Local setLocal = localGen.generateLocal(RefType.v("java.util.Set"));
210                                        newUnits.add(Jimple.v().newAssignStmt(setLocal, Jimple.v().newStaticFieldRef(fieldRef)));
211                                        for(String className: RTI.classForNameClassNames(m)) {
212                                                InterfaceInvokeExpr invokeExpr = Jimple.v().newInterfaceInvokeExpr(setLocal, addMethodRef,StringConstant.v(callSiteId+className));
213                                                newUnits.add(Jimple.v().newInvokeStmt(invokeExpr));
214                                        }
215                                        callSiteId++;
216                                }
217                        }       
218                        {       
219                                if(!RTI.classNewInstanceClassNames(m).isEmpty()) {
220                                        SootFieldRef fieldRef = Scene.v().makeFieldRef(reflCallsClass, "classNewInstance", RefType.v("java.util.Set"), true);
221                                        Local setLocal = localGen.generateLocal(RefType.v("java.util.Set"));
222                                        newUnits.add(Jimple.v().newAssignStmt(setLocal, Jimple.v().newStaticFieldRef(fieldRef)));                       
223                                        for(String className: RTI.classNewInstanceClassNames(m)) {
224                                                InterfaceInvokeExpr invokeExpr = Jimple.v().newInterfaceInvokeExpr(setLocal, addMethodRef,StringConstant.v(callSiteId+className));
225                                                newUnits.add(Jimple.v().newInvokeStmt(invokeExpr));
226                                        }
227                                        callSiteId++;
228                                }
229                        }
230                        {       
231                                if(!RTI.constructorNewInstanceSignatures(m).isEmpty()) {
232                                        SootFieldRef fieldRef = Scene.v().makeFieldRef(reflCallsClass, "constructorNewInstance", RefType.v("java.util.Set"), true);
233                                        Local setLocal = localGen.generateLocal(RefType.v("java.util.Set"));
234                                        newUnits.add(Jimple.v().newAssignStmt(setLocal, Jimple.v().newStaticFieldRef(fieldRef)));                       
235                                        for(String constrSig: RTI.constructorNewInstanceSignatures(m)) {
236                                                InterfaceInvokeExpr invokeExpr = Jimple.v().newInterfaceInvokeExpr(setLocal, addMethodRef,StringConstant.v(callSiteId+constrSig));
237                                                newUnits.add(Jimple.v().newInvokeStmt(invokeExpr));
238                                        }
239                                        callSiteId++;
240                                }
241                        }
242                        {       
243                                if(!RTI.methodInvokeSignatures(m).isEmpty()) {
244                                        SootFieldRef fieldRef = Scene.v().makeFieldRef(reflCallsClass, "methodInvoke", RefType.v("java.util.Set"), true);
245                                        Local setLocal = localGen.generateLocal(RefType.v("java.util.Set"));
246                                        newUnits.add(Jimple.v().newAssignStmt(setLocal, Jimple.v().newStaticFieldRef(fieldRef)));                       
247                                        for(String methodSig: RTI.methodInvokeSignatures(m)) {
248                                                InterfaceInvokeExpr invokeExpr = Jimple.v().newInterfaceInvokeExpr(setLocal, addMethodRef,StringConstant.v(callSiteId+methodSig));
249                                                newUnits.add(Jimple.v().newInvokeStmt(invokeExpr));
250                                        }
251                                        callSiteId++;
252                                }
253                        }
254                }       
255                
256                Unit secondLastStmt = units.getPredOf(units.getLast());
257                units.insertAfter(newUnits, secondLastStmt);
258                
259                if(Options.v().validate()) body.validate();
260        }

262        private void addCaching() {
263                SootClass method = Scene.v().getSootClass("java.lang.reflect.Method");
264                method.addField(new SootField(ALREADY_CHECKED_FIELDNAME, BooleanType.v()));
265                SootClass constructor = Scene.v().getSootClass("java.lang.reflect.Constructor");
266                constructor.addField(new SootField(ALREADY_CHECKED_FIELDNAME, BooleanType.v()));
267                SootClass clazz = Scene.v().getSootClass("java.lang.Class");
268                clazz.addField(new SootField(ALREADY_CHECKED_FIELDNAME, BooleanType.v()));
269                
270                for(Kind k: Kind.values()) {
271                        addCaching(k);
272                }
273        }

275        private void addCaching(Kind kind) {
276                
277                SootClass c;
278                String methodName;
279                switch(kind) {
280                case ClassNewInstance:
281                        c = Scene.v().getSootClass("java.lang.Class");
282                        methodName = "knownClassNewInstance";
283                        break;
284                case ConstructorNewInstance: 
285                        c = Scene.v().getSootClass("java.lang.reflect.Constructor");
286                        methodName = "knownConstructorNewInstance";
287                        break;
288                case MethodInvoke: 
289                        c = Scene.v().getSootClass("java.lang.reflect.Method");
290                        methodName = "knownMethodInvoke";
291                        break;
292                case ClassForName:
293                        //Cannot implement caching in this case because we can add no field to the String argument
294                        return;
295                default:
296                        throw new IllegalStateException("unknown kind: "+kind);
297                }
298                
299                SootClass reflCallsClass = Scene.v().getSootClass("soot.rtlib.tamiflex.ReflectiveCalls");
300                
301                SootMethod m = reflCallsClass.getMethodByName(methodName);
302                JimpleBody body = (JimpleBody) m.retrieveActiveBody();
303                LocalGenerator localGen = new LocalGenerator(body);
304                Unit firstStmt = body.getFirstNonIdentityStmt();
305                firstStmt = body.getUnits().getPredOf(firstStmt);
306                
307                Stmt jumpTarget = Jimple.v().newNopStmt();
308                
309                Chain<Unit> newUnits = new HashChain<Unit>();
310                
311                //alreadyCheckedLocal = m.alreadyChecked
312                InstanceFieldRef fieldRef = Jimple.v().newInstanceFieldRef(body.getParameterLocal(m.getParameterCount()-1), Scene.v().makeFieldRef(c, ALREADY_CHECKED_FIELDNAME, BooleanType.v(), false));
313                Local alreadyCheckedLocal = localGen.generateLocal(BooleanType.v());
314                newUnits.add(Jimple.v().newAssignStmt(alreadyCheckedLocal, fieldRef));
315                
316                //if(!alreadyChecked) goto jumpTarget
317                newUnits.add(Jimple.v().newIfStmt(Jimple.v().newEqExpr(alreadyCheckedLocal, IntConstant.v(0)), jumpTarget));
318                
319                //return
320                newUnits.add(Jimple.v().newReturnVoidStmt());
321                
322                //jumpTarget: nop               
323                newUnits.add(jumpTarget);
324                
325                //m.alreadyChecked = true
326                InstanceFieldRef fieldRef2 = Jimple.v().newInstanceFieldRef(body.getParameterLocal(m.getParameterCount()-1), Scene.v().makeFieldRef(c, ALREADY_CHECKED_FIELDNAME, BooleanType.v(), false));
327                newUnits.add(Jimple.v().newAssignStmt(fieldRef2, IntConstant.v(1)));
328                
329                body.getUnits().insertAfter(newUnits, firstStmt);
330                
331                if(Options.v().validate()) body.validate();
332        }

334        private void inlineRelectiveCalls(SootMethod m, Set<String> targets, Kind callKind) {
335                if(!m.hasActiveBody()) m.retrieveActiveBody();
336                Body b = m.getActiveBody();
337                PatchingChain<Unit> units = b.getUnits();
338                Iterator<Unit> iter = units.snapshotIterator();
339                LocalGenerator localGen = new LocalGenerator(b);
340                
341                //for all units
342                while(iter.hasNext()) {
343                        Chain<Unit> newUnits = new HashChain<Unit>();
344                        Stmt s = (Stmt) iter.next();
345                        
346                        //if we have an invoke expression, test to see if it is a reflective invoke expression
347                        if(s.containsInvokeExpr()) {
348                                InvokeExpr ie = s.getInvokeExpr();
349                                boolean found = false;
350                                Type fieldSetGetType = null;
351                                
352                                if(callKind==Kind.ClassForName && (ie.getMethodRef().getSignature().equals("<java.lang.Class: java.lang.Class forName(java.lang.String)>") ||
353                                                ie.getMethodRef().getSignature().equals("<java.lang.Class: java.lang.Class forName(java.lang.String,boolean,java.lang.ClassLoader)>"))) {
354                                        found = true;
355                                        Value classNameValue = ie.getArg(0);                                    
356                                        newUnits.add(Jimple.v().newInvokeStmt(Jimple.v().newStaticInvokeExpr(Scene.v().getMethod("<soot.rtlib.tamiflex.ReflectiveCalls: void knownClassForName(int,java.lang.String)>").makeRef(),IntConstant.v(callSiteId),classNameValue)));
357                                } else if(callKind==Kind.ClassNewInstance && ie.getMethodRef().getSignature().equals("<java.lang.Class: java.lang.Object newInstance()>")) {
358                                        found = true;
359                                        Local classLocal = (Local) ((InstanceInvokeExpr)ie).getBase();
360                                        newUnits.add(Jimple.v().newInvokeStmt(Jimple.v().newStaticInvokeExpr(Scene.v().getMethod("<soot.rtlib.tamiflex.ReflectiveCalls: void knownClassNewInstance(int,java.lang.Class)>").makeRef(),IntConstant.v(callSiteId),classLocal)));
361                                } else if(callKind==Kind.ConstructorNewInstance && ie.getMethodRef().getSignature().equals("<java.lang.reflect.Constructor: java.lang.Object newInstance(java.lang.Object[])>")) {
362                                        found = true;
363                                        Local constrLocal = (Local) ((InstanceInvokeExpr)ie).getBase();
364                                        newUnits.add(Jimple.v().newInvokeStmt(Jimple.v().newStaticInvokeExpr(Scene.v().getMethod("<soot.rtlib.tamiflex.ReflectiveCalls: void knownConstructorNewInstance(int,java.lang.reflect.Constructor)>").makeRef(),IntConstant.v(callSiteId),constrLocal)));
365                                } else if(callKind==Kind.MethodInvoke && ie.getMethodRef().getSignature().equals("<java.lang.reflect.Method: java.lang.Object invoke(java.lang.Object,java.lang.Object[])>")) {
366                                        found = true;
367                                        Local methodLocal = (Local) ((InstanceInvokeExpr)ie).getBase();
368                                        Value recv = ie.getArg(0);
369                                        newUnits.add(Jimple.v().newInvokeStmt(Jimple.v().newStaticInvokeExpr(Scene.v().getMethod("<soot.rtlib.tamiflex.ReflectiveCalls: void knownMethodInvoke(int,java.lang.Object,java.lang.reflect.Method)>").makeRef(),IntConstant.v(callSiteId),recv,methodLocal)));
370                                } else if(callKind==Kind.FieldSet) {
371                                        SootMethod sootMethod = ie.getMethodRef().resolve();
372                                        if(sootMethod.getDeclaringClass().getName().equals("java.lang.reflect.Field") && fieldSets.contains(sootMethod.getName())) {                                            
373                                                found = true;
374                                                fieldSetGetType = sootMethod.getParameterType(1); //assign type of 2nd parameter (1st is receiver object)
375                                                Value recv = ie.getArg(0);
376                                                Value field = ((InstanceInvokeExpr)ie).getBase();
377                                                newUnits.add(Jimple.v().newInvokeStmt(Jimple.v().newStaticInvokeExpr(Scene.v().getMethod("<soot.rtlib.tamiflex.ReflectiveCalls: void knownFieldSet(int,java.lang.Object,java.lang.reflect.Field)>").makeRef(),IntConstant.v(callSiteId),recv,field)));                                            
378                                        }
379                                } else if(callKind==Kind.FieldGet) {
380                                        SootMethod sootMethod = ie.getMethodRef().resolve();
381                                        if(sootMethod.getDeclaringClass().getName().equals("java.lang.reflect.Field") && fieldGets.contains(sootMethod.getName())) {
382                                                found = true;
383                                                fieldSetGetType = sootMethod.getReturnType(); //assign return type of get
384                                                Value recv = ie.getArg(0);
385                                                Value field = ((InstanceInvokeExpr)ie).getBase();
386                                                newUnits.add(Jimple.v().newInvokeStmt(Jimple.v().newStaticInvokeExpr(Scene.v().getMethod("<soot.rtlib.tamiflex.ReflectiveCalls: void knownFieldSet(int,java.lang.Object,java.lang.reflect.Field)>").makeRef(),IntConstant.v(callSiteId),recv,field)));                                            
387                                        }
388                                }
389                                
390                                if(!found) continue;
391                                
392                                NopStmt endLabel = Jimple.v().newNopStmt();
393                                
394                                //for all recorded targets
395                                for(String target : targets) {
396                                        
397                                        NopStmt jumpTarget = Jimple.v().newNopStmt();
398
399                                        //boolean predLocal = Opaque.getFalse();
400                                        Local predLocal = localGen.generateLocal(BooleanType.v());
401                                        StaticInvokeExpr staticInvokeExpr = Jimple.v().newStaticInvokeExpr(UNINTERPRETED_METHOD);
402                                        newUnits.add(Jimple.v().newAssignStmt(predLocal, staticInvokeExpr));
403                                        //if predLocal == 0 goto <original reflective call>
404                                        newUnits.add(Jimple.v().newIfStmt(Jimple.v().newEqExpr(IntConstant.v(0), predLocal), jumpTarget));
405
406                                        SootMethod newMethod = createNewMethod(callKind, target, fieldSetGetType);
407                                        
408                                        List<Value> args = new LinkedList<Value>();
409                                        switch(callKind) {
410                                        case ClassForName:
411                                        case ClassNewInstance:
412                                                //no arguments
413                                                break;
414                                        case ConstructorNewInstance:
415                                                //add Object[] argument
416                                                args.add((Value) ie.getArgs().get(0));                                  
417                                                break;
418                                        case MethodInvoke:
419                                                //add Object argument
420                                                args.add((Value) ie.getArgs().get(0));                                  
421                                                //add Object[] argument
422                                                args.add((Value) ie.getArgs().get(1));                                  
423                                                break;
424                                        case FieldSet:
425                                                //add Object argument
426                                                args.add((Value) ie.getArgs().get(0));                                  
427                                                //add value argument
428                                                args.add((Value) ie.getArgs().get(1));
429                                                break;
430                                        case FieldGet:
431                                                //add Object argument
432                                                args.add((Value) ie.getArgs().get(0));                                  
433                                                break;
434                                        default:
435                                                throw new IllegalStateException();      
436                                        }
437                                        
438                                        StaticInvokeExpr methodInvokeExpr = Jimple.v().newStaticInvokeExpr(newMethod.makeRef(), args);
439                                        Local retLocal = localGen.generateLocal(newMethod.getReturnType());
440                                        newUnits.add(Jimple.v().newAssignStmt(retLocal, methodInvokeExpr));
441                                        
442                                        if(s instanceof AssignStmt) {
443                                                AssignStmt assignStmt = (AssignStmt) s;
444                                                Value leftOp = assignStmt.getLeftOp();
445                                                AssignStmt newAssignStmt = Jimple.v().newAssignStmt(leftOp, retLocal);
446                                                newUnits.add(newAssignStmt);
447                                        }
448                                        
449                                        GotoStmt gotoStmt = Jimple.v().newGotoStmt(endLabel);
450                                        newUnits.add(gotoStmt);
451                                        
452                                        newUnits.add(jumpTarget);
453                                }
454                                
455                                Unit end = newUnits.getLast();
456                                units.insertAfter(newUnits, s);
457                                units.remove(s);
458                                units.insertAfter(s, end);
459                                units.insertAfter(endLabel, s);
460                                
461                        }
462                }
463                callSiteId++;
464        }

466        @SuppressWarnings("unchecked")
467        private SootMethod createNewMethod(Kind callKind, String target, Type fieldSetGetType) {
468                List<Type> parameterTypes = new LinkedList<Type>();
469                Type returnType=null;
470                switch(callKind) {
471                case ClassForName:
472                        returnType = RefType.v("java.lang.Class");
473                        break;
474                case ClassNewInstance:
475                        returnType = RefType.v("java.lang.Object");
476                        break;
477                case ConstructorNewInstance:
478                        parameterTypes.add(ArrayType.v(RefType.v("java.lang.Object"), 1));
479                        returnType = RefType.v("java.lang.Object");
480                        break;
481                case MethodInvoke:
482                        parameterTypes.add(RefType.v("java.lang.Object"));
483                        parameterTypes.add(ArrayType.v(RefType.v("java.lang.Object"), 1));
484                        returnType = RefType.v("java.lang.Object");
485                        break;
486                case FieldSet:
487                        parameterTypes.add(RefType.v("java.lang.Object"));
488                        parameterTypes.add(fieldSetGetType);
489                        returnType = VoidType.v();
490                        break;
491                case FieldGet:
492                        parameterTypes.add(RefType.v("java.lang.Object"));
493                        returnType = fieldSetGetType;
494                        break;
495                default:
496                        throw new IllegalStateException();      
497                }
498
499                SootMethod newMethod = new SootMethod("reflectiveCall"+(callNum++), parameterTypes, returnType, Modifier.PUBLIC | Modifier.STATIC);
500                Body newBody = Jimple.v().newBody(newMethod);
501                newMethod.setActiveBody(newBody);
502                reflectiveCallsClass.addMethod(newMethod);
503                
504                PatchingChain<Unit> newUnits = newBody.getUnits();
505
506                LocalGenerator localGen = new LocalGenerator(newBody);
507                
508                Local freshLocal;
509                Value replacement=null;
510                Local[] paramLocals=null;
511                switch(callKind) {
512                case ClassForName: 
513                {
514                        //replace by: <Class constant for <target>>
515                        freshLocal = localGen.generateLocal(RefType.v("java.lang.Class"));
516                        replacement = ClassConstant.v(target.replace('.','/'));
517                        break;
518                }
519                case ClassNewInstance:
520                {
521                        //replace by: new <target>
522                        RefType targetType = RefType.v(target);
523                        freshLocal = localGen.generateLocal(targetType);
524                        replacement = Jimple.v().newNewExpr(targetType);
525                        break;
526                }
527                case ConstructorNewInstance:
528                {
529                        /* replace r=constr.newInstance(args) by:
530                         * Object p0 = args[0];
531                         * ...
532                         * Object pn = args[n];
533                         * T0 a0 = (T0)p0;
534                         * ...
535                         * Tn an = (Tn)pn;
536                         */
537                        
538                        SootMethod constructor = Scene.v().getMethod(target);
539                        paramLocals = new Local[constructor.getParameterCount()];
540                        if(constructor.getParameterCount()>0) {
541                                //argArrayLocal = @parameter-0
542                                ArrayType arrayType = ArrayType.v(RefType.v("java.lang.Object"),1);
543                                Local argArrayLocal = localGen.generateLocal(arrayType);
544                                newUnits.add(Jimple.v().newIdentityStmt(argArrayLocal, Jimple.v().newParameterRef(arrayType, 0)));
545                                int i=0;
546                                for(Type paramType: ((Collection<Type>)constructor.getParameterTypes())) {
547                                        paramLocals[i] = localGen.generateLocal(paramType);
548                                        unboxParameter(argArrayLocal, i, paramLocals, paramType, newUnits, localGen);
549                                        i++;
550                                }
551                        } 
552                        RefType targetType = constructor.getDeclaringClass().getType();
553                        freshLocal = localGen.generateLocal(targetType);
554                        replacement = Jimple.v().newNewExpr(targetType);
555                        
556                        break;
557                }
558                case MethodInvoke: 
559                {
560                        /* replace r=m.invoke(obj,args) by:
561                         * T recv = (T)obj;
562                         * Object p0 = args[0];
563                         * ...
564                         * Object pn = args[n];
565                         * T0 a0 = (T0)p0;
566                         * ...
567                         * Tn an = (Tn)pn;
568                         */
569                        SootMethod method = Scene.v().getMethod(target);
570                        //recvObject = @parameter-0
571                        RefType objectType = RefType.v("java.lang.Object");
572                        Local recvObject = localGen.generateLocal(objectType);
573                        newUnits.add(Jimple.v().newIdentityStmt(recvObject, Jimple.v().newParameterRef(objectType, 0)));
574                        paramLocals = new Local[method.getParameterCount()];
575                        if(method.getParameterCount()>0) {
576                                //argArrayLocal = @parameter-1
577                                ArrayType arrayType = ArrayType.v(RefType.v("java.lang.Object"),1);
578                                Local argArrayLocal = localGen.generateLocal(arrayType);
579                                newUnits.add(Jimple.v().newIdentityStmt(argArrayLocal, Jimple.v().newParameterRef(arrayType, 1)));
580                                int i=0;
581                                for(Type paramType: ((Collection<Type>)method.getParameterTypes())) {
582                                        paramLocals[i] = localGen.generateLocal(paramType);
583                                        unboxParameter(argArrayLocal, i, paramLocals, paramType, newUnits, localGen);                                                   
584                                        i++;
585                                }
586                        } 
587                        RefType targetType = method.getDeclaringClass().getType();
588                        freshLocal = localGen.generateLocal(targetType);
589                        replacement = Jimple.v().newCastExpr(recvObject, method.getDeclaringClass().getType());
590                        
591                        break;
592                }
593                case FieldSet:
594                case FieldGet:
595                {
596                        /* replace f.set(o,v) by:
597                         * Object obj = @parameter-0; 
598                         * T freshLocal = (T)obj;
599                         */
600                        RefType objectType = RefType.v("java.lang.Object");
601                        Local recvObject = localGen.generateLocal(objectType);
602                        newUnits.add(Jimple.v().newIdentityStmt(recvObject, Jimple.v().newParameterRef(objectType, 0)));
603                        
604                        SootField field = Scene.v().getField(target);
605                        freshLocal = localGen.generateLocal(field.getDeclaringClass().getType());                       
606                        replacement = Jimple.v().newCastExpr(recvObject, field.getDeclaringClass().getType());
607
608                        break;
609                }
610                default:
611                        throw new InternalError("Unknown kind of reflective call "+callKind);
612                }
613                
614                AssignStmt replStmt = Jimple.v().newAssignStmt(freshLocal, replacement);
615                newUnits.add(replStmt);
616                
617                Local retLocal = localGen.generateLocal(returnType);
618                
619                switch(callKind) {
620                case ClassForName:
621                {
622                        //add: retLocal = freshLocal;
623                        newUnits.add(Jimple.v().newAssignStmt(retLocal, freshLocal));
624                        break;
625                }
626                case ClassNewInstance:
627                {
628                        //add: freshLocal.<init>()
629                        SootClass targetClass = Scene.v().getSootClass(target);
630                        SpecialInvokeExpr constrCallExpr = Jimple.v().newSpecialInvokeExpr(freshLocal, Scene.v().makeMethodRef(targetClass, SootMethod.constructorName, Collections.<Type>emptyList(), VoidType.v(), false));
631                        InvokeStmt constrCallStmt2 = Jimple.v().newInvokeStmt(constrCallExpr);
632                        newUnits.add(constrCallStmt2);
633                        //add: retLocal = freshLocal
634                        newUnits.add(Jimple.v().newAssignStmt(retLocal, freshLocal));
635                        break;
636                }
637                case ConstructorNewInstance:
638                {
639                        //add: freshLocal.<target>(a0,...,an);
640                        SootMethod constructor = Scene.v().getMethod(target);
641                        SpecialInvokeExpr constrCallExpr = Jimple.v().newSpecialInvokeExpr(freshLocal, constructor.makeRef(), Arrays.asList(paramLocals));
642                        InvokeStmt constrCallStmt2 = Jimple.v().newInvokeStmt(constrCallExpr);
643                        newUnits.add(constrCallStmt2);
644                        //add: retLocal = freshLocal
645                        newUnits.add(Jimple.v().newAssignStmt(retLocal, freshLocal));
646                        break;
647                }
648                case MethodInvoke:
649                {
650                        //add: freshLocal=recv.<target>(a0,...,an);
651                        SootMethod method = Scene.v().getMethod(target);
652                        InvokeExpr invokeExpr;
653                        if(method.isStatic())
654                                invokeExpr = Jimple.v().newStaticInvokeExpr(method.makeRef(), Arrays.asList(paramLocals));
655                        else
656                                invokeExpr = Jimple.v().newVirtualInvokeExpr(freshLocal, method.makeRef(), Arrays.<Value>asList(paramLocals));
657                        if(method.getReturnType().equals(VoidType.v())) {
658                                //method returns null; simply invoke it and return null
659                                InvokeStmt invokeStmt = Jimple.v().newInvokeStmt(invokeExpr);
660                                newUnits.add(invokeStmt);
661                                AssignStmt assignStmt = Jimple.v().newAssignStmt(retLocal,NullConstant.v());
662                                newUnits.add(assignStmt);
663                        } else {
664                                AssignStmt assignStmt = Jimple.v().newAssignStmt(retLocal,invokeExpr);
665                                newUnits.add(assignStmt);
666                        }
667                        break;
668                }
669                case FieldSet:
670                {
671                         // add freshLocal.<f> = v;
672                        Local value = localGen.generateLocal(fieldSetGetType);
673                        newUnits.insertBeforeNoRedirect(Jimple.v().newIdentityStmt(value, Jimple.v().newParameterRef(fieldSetGetType, 1)), replStmt);
674
675                        SootField field = Scene.v().getField(target);
676
677                        Local boxedOrCasted = localGen.generateLocal(field.getType()); 
678                        
679                        insertCastOrUnboxingCode(boxedOrCasted, value, newUnits);
680                        
681                        FieldRef fieldRef;
682                        if(field.isStatic()) {
683                                fieldRef = Jimple.v().newStaticFieldRef(field.makeRef());
684                        } else {
685                                fieldRef = Jimple.v().newInstanceFieldRef(freshLocal, field.makeRef());
686                        }
687                        newUnits.add(Jimple.v().newAssignStmt(fieldRef, boxedOrCasted));
688                        break;
689                }
690                case FieldGet:
691                {
692                        /* add:
693                         * T2 temp = recv.<f>;
694                         * return temp;
695                         */
696                        SootField field = Scene.v().getField(target);
697                        
698                        Local value = localGen.generateLocal(field.getType());
699                        
700                        FieldRef fieldRef;
701                        if(field.isStatic()) {
702                                fieldRef = Jimple.v().newStaticFieldRef(field.makeRef());
703                        } else {
704                                fieldRef = Jimple.v().newInstanceFieldRef(freshLocal, field.makeRef());
705                        }
706                        newUnits.add(Jimple.v().newAssignStmt(value, fieldRef));
707                        
708                        insertCastOrBoxingCode(retLocal, value, newUnits);
709                        
710                        break;
711                }
712                } 
713                
714                if(!returnType.equals(VoidType.v()))
715                        newUnits.add(Jimple.v().newReturnStmt(retLocal));
716                
717                if(Options.v().validate()) newBody.validate();
718                
719                cleanup(newBody);
720                
721                return newMethod;
722        }

725        private void insertCastOrUnboxingCode(Local lhs, Local rhs, Chain<Unit> newUnits) {
726                //if assigning to a reference type then there's nothing to do
727                if(lhs.getType() instanceof PrimType) {
728                        if((rhs.getType() instanceof PrimType)) {
729                                //insert cast
730                                newUnits.add(Jimple.v().newAssignStmt(lhs, Jimple.v().newCastExpr(rhs, lhs.getType())));
731                        } else {
732                                //reference type in rhs; insert unboxing code
733                            RefType boxedType = (RefType) rhs.getType();
734                                SootMethodRef ref = Scene.v().makeMethodRef(
735                                              boxedType.getSootClass(),
736                                              lhs.getType().toString() + "Value",
737                                              Collections.<Type>emptyList(),
738                                              lhs.getType(),
739                                              false
740                                            );
741                                newUnits.add(Jimple.v().newAssignStmt(lhs, Jimple.v().newVirtualInvokeExpr(rhs, ref)));
742                        }
743                }
744        }
745
746        private void insertCastOrBoxingCode(Local lhs, Local rhs, Chain<Unit> newUnits) {
747                //if assigning to a primitive type then there's nothing to do
748                if(lhs.getType() instanceof RefLikeType) {
749                        if((rhs.getType() instanceof RefLikeType)) {
750                                //insert cast
751                                newUnits.add(Jimple.v().newAssignStmt(lhs, Jimple.v().newCastExpr(rhs, lhs.getType())));
752                        } else {
753                                //primitive type in rhs; insert boxing code
754                            RefType boxedType = ((PrimType)rhs.getType()).boxedType();
755                                SootMethodRef ref = Scene.v().makeMethodRef(
756                                              boxedType.getSootClass(),
757                                              "valueOf",
758                                              Collections.<Type>singletonList(rhs.getType()),
759                                              boxedType,
760                                              true
761                                            );
762                                newUnits.add(Jimple.v().newAssignStmt(lhs, Jimple.v().newStaticInvokeExpr(ref,rhs)));
763                        }
764                }
765        }

767        /** Auto-unboxes an argument array.
768         * @param argsArrayLocal a local holding the argument Object[] array
769         * @param paramIndex the index of the parameter to unbox
770         * @param paramType the (target) type of the parameter
771         * @param newUnits the Unit chain to which the unboxing code will be appended
772         * @param localGen a {@link LocalGenerator} for the body holding the units
773         */
774        private void unboxParameter(Local argsArrayLocal, int paramIndex, Local[] paramLocals, Type paramType, Chain<Unit> newUnits, LocalGenerator localGen) {
775                ArrayRef arrayRef = Jimple.v().newArrayRef(argsArrayLocal, IntConstant.v(paramIndex));
776                AssignStmt assignStmt;
777                if(paramType instanceof PrimType) {
778                    PrimType primType = (PrimType) paramType;
779                        // Unbox the value if needed
780                    RefType boxedType = primType.boxedType();
781                        SootMethodRef ref = Scene.v().makeMethodRef(
782                      boxedType.getSootClass(),
783                      paramType + "Value",
784                      Collections.<Type>emptyList(),
785                      paramType,
786                      false
787                    );
788                    Local boxedLocal = localGen.generateLocal(RefType.v("java.lang.Object"));
789                    AssignStmt arrayLoad = Jimple.v().newAssignStmt(boxedLocal, arrayRef);
790                    newUnits.add(arrayLoad);
791                    Local castedLocal = localGen.generateLocal(boxedType);
792                    AssignStmt cast = Jimple.v().newAssignStmt(castedLocal, Jimple.v().newCastExpr(boxedLocal, boxedType));
793                    newUnits.add(cast);
794                    VirtualInvokeExpr unboxInvokeExpr = Jimple.v().newVirtualInvokeExpr(castedLocal,ref);
795                        assignStmt = Jimple.v().newAssignStmt(paramLocals[paramIndex], unboxInvokeExpr);
796                } else {
797                    Local boxedLocal = localGen.generateLocal(RefType.v("java.lang.Object"));
798                    AssignStmt arrayLoad = Jimple.v().newAssignStmt(boxedLocal, arrayRef);
799                    newUnits.add(arrayLoad);
800                    Local castedLocal = localGen.generateLocal(paramType);
801                    AssignStmt cast = Jimple.v().newAssignStmt(castedLocal, Jimple.v().newCastExpr(boxedLocal, paramType));
802                    newUnits.add(cast);
803                        assignStmt = Jimple.v().newAssignStmt(paramLocals[paramIndex], castedLocal);
804                }
805                newUnits.add(assignStmt);
806        }
