031package soot.jimple.internal;
032
033import soot.*;
034import soot.jimple.*;
035
036import java.util.*;
037
038@SuppressWarnings("serial")
039abstract public class AbstractJimpleFloatBinopExpr
040    extends AbstractFloatBinopExpr implements ConvertToBaf
041{
042    AbstractJimpleFloatBinopExpr(Value op1, Value op2)
043    {
044        this(Jimple.v().newArgBox(op1),
045             Jimple.v().newArgBox(op2));
046    }
047
048    protected AbstractJimpleFloatBinopExpr(ValueBox op1Box, ValueBox op2Box)
049    {
050        this.op1Box = op1Box;
051        this.op2Box = op2Box;
052    }
053
054    public void convertToBaf(JimpleToBafContext context, List<Unit> out)
055    {
056        ((ConvertToBaf) this.getOp1()).convertToBaf(context, out);
057        ((ConvertToBaf) this.getOp2()).convertToBaf(context, out);
058        Unit u = (Unit)makeBafInst(this.getOp1().getType());
059        out.add(u);
060        u.addAllTagsOf(context.getCurrentUnit());
061    }
062
063    abstract Object makeBafInst(Type opType);
064}
