001package soot.jimple.internal;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import soot.Value;
007import soot.ValueBox;
008
009@SuppressWarnings("serial")
010public abstract class AbstractOpStmt extends AbstractStmt {
011
012        final ValueBox opBox;
013        
014        protected AbstractOpStmt(ValueBox opBox)
015    {
016        this.opBox = opBox;
017    }
018        
019        
020    final public Value getOp()
021    {
022        return opBox.getValue();
023    }
024        
025        
026        final public void setOp(Value op)
027    {
028        opBox.setValue(op);
029    }
030
031        
032        final public ValueBox getOpBox()
033    {
034        return opBox;
035    }
036        
037        @Override
038    final public List<ValueBox> getUseBoxes()
039    {
040        List<ValueBox> list = new ArrayList<ValueBox>();
041
042        list.addAll(opBox.getValue().getUseBoxes());
043        list.add(opBox);
044    
045        return list;
046    }
047}
