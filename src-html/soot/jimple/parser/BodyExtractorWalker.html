026package soot.jimple.parser;
027
028import soot.options.*;
029
030import soot.*;
031import soot.jimple.*;
032
033import soot.jimple.parser.node.*;
034import java.util.*;
035
036/**
037 * Walks a jimple AST and constructs the method bodies for all the methods of
038 * the SootClass associated with this walker (see constructor). note: Contrary
039 * to the plain "Walker", this walker does not create a SootClass, or interact
040 * with the scene. It merely adds method bodies for each of the methods of the
041 * SootClass it was initialized with.
042 */
043
044/* Modified By Marc Berndl May 17th */
045
046public class BodyExtractorWalker extends Walker {
047        Map<SootMethod, JimpleBody> methodToParsedBodyMap;
048
049        /**
050         * Constructs a walker, and attaches it to the given SootClass, sending
051         * bodies to the given methodToParsedBodyMap.
052         */
053        public BodyExtractorWalker(SootClass sc, SootResolver resolver,
054                        Map<SootMethod, JimpleBody> methodToParsedBodyMap) {
055                super(sc, resolver);
056                this.methodToParsedBodyMap = methodToParsedBodyMap;
057        }
058
059        /*
060         * file = modifier* file_type class_name extends_clause? implements_clause?
061         * file_body;
062         */
063        public void caseAFile(AFile node) {
064                inAFile(node);
065                {
066                        Object temp[] = node.getModifier().toArray();
067                        for (Object element : temp) {
068                                ((PModifier) element).apply(this);
069                        }
070                }
071                if (node.getFileType() != null) {
072                        node.getFileType().apply(this);
073                }
074                if (node.getClassName() != null) {
075                        node.getClassName().apply(this);
076                }
077
078                String className = (String) mProductions.removeLast();
079                if (!className.equals(Scene.v().quotedNameOf(mSootClass.getName())))
080                        throw new RuntimeException("expected:  " + className
081                                        + ", but got: " + mSootClass.getName());
082
083                if (node.getExtendsClause() != null) {
084                        node.getExtendsClause().apply(this);
085                }
086                if (node.getImplementsClause() != null) {
087                        node.getImplementsClause().apply(this);
088                }
089                if (node.getFileBody() != null) {
090                        node.getFileBody().apply(this);
091                }
092                outAFile(node);
093        }
094
095        public void outAFile(AFile node) {
096                if (node.getImplementsClause() != null)
097                        mProductions.removeLast(); // implements_clause
098
099                if (node.getExtendsClause() != null)
100                        mProductions.removeLast(); // extends_clause
101
102                mProductions.removeLast(); // file_type
103                mProductions.addLast(mSootClass);
104        }
105
106        /*
107         * member = {field} modifier* type name semicolon | {method} modifier* type
108         * name l_paren parameter_list? r_paren throws_clause? method_body;
109         */
110        public void outAFieldMember(AFieldMember node) {
111                mProductions.removeLast(); // name
112                mProductions.removeLast(); // type
113        }
114
115        public void outAMethodMember(AMethodMember node) {
116                Type type;
117                String name;
118                List<Type> parameterList = new ArrayList<Type>();
119                List throwsClause = null;
120                JimpleBody methodBody = null;
121
122                if (node.getMethodBody() instanceof AFullMethodBody)
123                        methodBody = (JimpleBody) mProductions.removeLast();
124
125                if (node.getThrowsClause() != null)
126                        throwsClause = (List) mProductions.removeLast();
127
128                if (node.getParameterList() != null)
129                        parameterList = (List) mProductions.removeLast();
130
131                name = (String) mProductions.removeLast(); // name
132                type = (Type) mProductions.removeLast(); // type
133                SootMethod sm = mSootClass.getMethodUnsafe(SootMethod.getSubSignature(
134                                name, parameterList, type));
135                if (sm != null) {
136                        if (Options.v().verbose())
137                                G.v().out
138                                                .println("[Jimple parser] "
139                                                                + SootMethod.getSubSignature(name,
140                                                                                parameterList, type));
141                } else {
142                        G.v().out.println("[!!! Couldn't parse !!] "
143                                        + SootMethod.getSubSignature(name, parameterList, type));
144
145                        G.v().out.println("[!] Methods in class are:");
146                        for (SootMethod next : mSootClass.getMethods()) {
147                                G.v().out.println(next.getSubSignature());
148                        }
149
150                }
151
152                if (sm.isConcrete() && methodBody != null) {
153                        if (Options.v().verbose())
154                                G.v().out.println("[Parsed] " + sm.getDeclaration());
155
156                        methodBody.setMethod(sm);
157                        methodToParsedBodyMap.put(sm, methodBody);
158                } else if (node.getMethodBody() instanceof AFullMethodBody) {
159                        if (sm.isPhantom() && Options.v().verbose())
160                                G.v().out.println("[jimple parser] phantom method!");
161                        throw new RuntimeException("Impossible: !concrete => ! instanceof "
162                                        + sm.getName());
163                }
164        }
165
166}
