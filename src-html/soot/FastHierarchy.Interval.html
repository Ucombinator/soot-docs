036/** Represents the class hierarchy.  It is closely linked to a Scene,
037 * and must be recreated if the Scene changes. 
038 *
039 * This version supercedes the old soot.Hierarchy class.
040 *
041 * @author Ondrej Lhotak
042 */
043public class FastHierarchy
044{    
045    /** This map holds all key,value pairs such that 
046     * value.getSuperclass() == key. This is one of the three maps that hold
047     * the inverse of the relationships given by the getSuperclass and
048     * getInterfaces methods of SootClass. */
049    protected MultiMap<SootClass,SootClass> classToSubclasses =
050                new ConcurrentHashMultiMap<SootClass,SootClass>();
051
052    /** This map holds all key,value pairs such that value is an interface 
053     * and key is in value.getInterfaces(). This is one of the three maps 
054     * that hold the inverse of the relationships given by the getSuperclass 
055     * and getInterfaces methods of SootClass. */
056    protected MultiMap<SootClass,SootClass> interfaceToSubinterfaces =
057                new ConcurrentHashMultiMap<SootClass,SootClass>();
058
059    /** This map holds all key,value pairs such that value is a class 
060     * (NOT an interface) and key is in value.getInterfaces(). This is one of 
061     * the three maps that hold the inverse of the relationships given by the 
062     * getSuperclass and getInterfaces methods of SootClass. */
063    protected MultiMap<SootClass,SootClass> interfaceToImplementers =
064                new ConcurrentHashMultiMap<SootClass,SootClass>();
065
066    /** This map is a transitive closure of interfaceToSubinterfaces,
067     * and each set contains its superinterface itself. */
068    protected MultiMap<SootClass,SootClass> interfaceToAllSubinterfaces =
069                new ConcurrentHashMultiMap<SootClass,SootClass>();
070
071    /** This map gives, for an interface, all concrete classes that
072     * implement that interface and all its subinterfaces, but
073     * NOT their subclasses. */
074    protected MultiMap<SootClass,SootClass> interfaceToAllImplementers =
075                new ConcurrentHashMultiMap<SootClass,SootClass>();
076
077    /** For each class (NOT interface), this map contains a Interval, which is
078     * a pair of numbers giving a preorder and postorder ordering of classes
079     * in the inheritance tree. */
080    protected Map<SootClass, Interval> classToInterval = new HashMap<SootClass, Interval>();
081
082    protected Scene sc;
083
084    protected class Interval {
085        int lower;
086        int upper;
087        boolean isSubrange( Interval potentialSubrange ) {
088                if (potentialSubrange == null) return false;
089            if( lower > potentialSubrange.lower ) return false;
090            if( upper < potentialSubrange.upper ) return false;
091            return true;
092        }
093    }
094    protected int dfsVisit( int start, SootClass c ) {
095        Interval r = new Interval();
096        r.lower = start++;
097        Collection<SootClass> col = classToSubclasses.get(c);
098        if( col != null ) {
099            for (SootClass sc : col) {
100                // For some awful reason, Soot thinks interface are subclasses
101                // of java.lang.Object
102                if( sc.isInterface() ) continue;
103                start = dfsVisit( start, sc );
104            }
105        }
106        r.upper = start++;
107        if( c.isInterface() ) {
108            throw new RuntimeException( "Attempt to dfs visit interface "+c );
109        }
110        if(!classToInterval.containsKey(c))
111                classToInterval.put( c, r );
112        return start;
113    }

115    /** Constructs a hierarchy from the current scene. */
116    public FastHierarchy()
117    {
118        this.sc = Scene.v();
119
120        /* First build the inverse maps. */
121        for (SootClass cl : sc.getClasses().getElementsUnsorted()) {
122            if( cl.resolvingLevel() < SootClass.HIERARCHY ) continue;
123            if( !cl.isInterface() && cl.hasSuperclass() ) {
124                classToSubclasses.put(cl.getSuperclass(), cl);
125            }
126            for( final SootClass supercl : cl.getInterfaces() ) {
127                if( cl.isInterface() ) {
128                    interfaceToSubinterfaces.put( supercl, cl );
129                } else {
130                    interfaceToImplementers.put( supercl, cl );
131                }
132            }
133        }
134
135        /* Now do a dfs traversal to get the Interval numbers. */
136        dfsVisit( 0, Scene.v().getSootClass( "java.lang.Object" ) );
137        /* also have to traverse for all phantom classes because they also
138         * can be roots of the type hierarchy
139         */
140        for (final Iterator<SootClass> phantomClassIt = Scene.v().getPhantomClasses().snapshotIterator();
141                        phantomClassIt.hasNext(); ) {
142                SootClass phantomClass = phantomClassIt.next();
143                if(!phantomClass.isInterface())
144                        dfsVisit( 0, phantomClass );
145        }
146    }

148    /** Return true if class child is a subclass of class parent, neither of
149     * them being allowed to be interfaces. */
150    public boolean isSubclass( SootClass child, SootClass parent ) {
151        child.checkLevel(SootClass.HIERARCHY);
152        parent.checkLevel(SootClass.HIERARCHY);
153        Interval parentInterval = classToInterval.get( parent );
154        Interval childInterval = classToInterval.get( child );
155        return parentInterval.isSubrange( childInterval );
156    }

158    /** For an interface parent (MUST be an interface), returns set of all
159     * implementers of it but NOT their subclasses. */
160    public Set<SootClass> getAllImplementersOfInterface( SootClass parent ) {
161        parent.checkLevel(SootClass.HIERARCHY);
162        if( !interfaceToAllImplementers.containsKey( parent ) ) {
163            for(SootClass subinterface : getAllSubinterfaces( parent )) {
164                if( subinterface == parent ) continue;
165                interfaceToAllImplementers.putAll(parent,
166                    getAllImplementersOfInterface( subinterface ) );
167            }
168            interfaceToAllImplementers.putAll(parent, 
169                    interfaceToImplementers.get( parent ) );
170        }
171        return interfaceToAllImplementers.get( parent );
172    }

174    /**
175     * For an interface parent (MUST be an interface), returns set of all subinterfaces.
176     * @param parent the parent interface.
177     * @return an set, possibly empty
178     * */
179    public Set<SootClass> getAllSubinterfaces( SootClass parent ) {
180        parent.checkLevel(SootClass.HIERARCHY);
181        if (!parent.isInterface()) return Collections.<SootClass>emptySet();
182        if( !interfaceToAllSubinterfaces.containsKey( parent ) ) {
183            interfaceToAllSubinterfaces.put( parent, parent );
184            for(SootClass si : interfaceToSubinterfaces.get( parent )) {
185                interfaceToAllSubinterfaces.putAll(parent, getAllSubinterfaces( si ) );
186            }
187        }
188        return interfaceToAllSubinterfaces.get( parent );
189    }

191    /** Given an object of declared type child, returns true if the object
192     * can be stored in a variable of type parent. If child is an interface
193     * that is not a subinterface of parent, this method will return false
194     * even though some objects implementing the child interface may also
195     * implement the parent interface. */
196    public boolean canStoreType( Type child, Type parent ) {
197        if( child.equals( parent ) ) return true;
198        if( parent instanceof NullType ) {
199            return false;
200        }
201        if( child instanceof NullType ) {
202                return parent instanceof RefLikeType;
203        }
204        if( child instanceof RefType ) {
205                if (parent.equals(Scene.v().getObjectType()))
206                        return true;
207            if( parent instanceof RefType) {
208                return canStoreClass( ((RefType) child).getSootClass(),
209                    ((RefType) parent).getSootClass() );
210            } else {
211                return false;
212            }
213        } else if( child instanceof AnySubType ) {
214            if( !(parent instanceof RefLikeType ) ) {
215                throw new RuntimeException( "Unhandled type "+parent );
216            } else if(parent instanceof ArrayType) {
217                Type base = ((AnySubType)child).getBase();
218                // From Java Language Spec 2nd ed., Chapter 10, Arrays
219                return base.equals( RefType.v( "java.lang.Object" ) )
220                || base.equals( RefType.v( "java.io.Serializable" ) )
221                || base.equals( RefType.v( "java.lang.Cloneable" ) );
222            } else {
223                SootClass base = ((AnySubType)child).getBase().getSootClass();
224                SootClass parentClass = ((RefType) parent).getSootClass();
225                LinkedList<SootClass> worklist = new LinkedList<SootClass>();
226                if( base.isInterface() ) worklist.addAll(getAllImplementersOfInterface(base));
227                else worklist.add(base);
228                Set<SootClass> workset = new HashSet<SootClass>();
229                while(!worklist.isEmpty()) {
230                    SootClass cl = (SootClass) worklist.removeFirst();
231                    if( !workset.add(cl) ) continue;
232                    if( cl.isConcrete() 
233                    &&  canStoreClass(cl, parentClass) ) return true;
234                    worklist.addAll(getSubclassesOf(cl));
235                }
236                return false;
237            }
238        } else if (child instanceof ArrayType) {
239            ArrayType achild = (ArrayType) child;
240            if( parent instanceof RefType ) {
241                // From Java Language Spec 2nd ed., Chapter 10, Arrays
242                return parent.equals( RefType.v( "java.lang.Object" ) )
243                || parent.equals( RefType.v( "java.io.Serializable" ) )
244                || parent.equals( RefType.v( "java.lang.Cloneable" ) );
245            }
246            if (!(parent instanceof ArrayType))
247                return false;
248            ArrayType aparent = (ArrayType) parent;
249                                                
250            // You can store a int[][] in a Object[]. Yuck!
251            // Also, you can store a Interface[] in a Object[]
252            if( achild.numDimensions == aparent.numDimensions ) {
253                if( achild.baseType.equals( aparent.baseType ) ) return true;
254                if( !(achild.baseType instanceof RefType ) ) return false;
255                if( !(aparent.baseType instanceof RefType ) ) return false;
256                return canStoreType( achild.baseType, aparent.baseType );
257            } else if( achild.numDimensions > aparent.numDimensions ) {
258                if( aparent.baseType.equals( RefType.v( "java.lang.Object" ) ) )
259                    return true;
260                if( aparent.baseType.equals( RefType.v( "java.io.Serializable" ) ) )
261                    return true;
262                if( aparent.baseType.equals( RefType.v( "java.lang.Cloneable" ) ) )
263                    return true;
264                return false;
265            } else return false;
266        } else
267                return false;
268    }

270    /** Given an object of declared type child, returns true if the object
271     * can be stored in a variable of type parent. If child is an interface
272     * that is not a subinterface of parent, this method will return false
273     * even though some objects implementing the child interface may also
274     * implement the parent interface. */
275    protected boolean canStoreClass( SootClass child, SootClass parent ) {
276        parent.checkLevel(SootClass.HIERARCHY);
277        child.checkLevel(SootClass.HIERARCHY);
278        Interval parentInterval = classToInterval.get( parent );
279        Interval childInterval = classToInterval.get( child );
280        if( parentInterval != null && childInterval != null ) {
281            return parentInterval.isSubrange( childInterval );
282        }
283        if( childInterval == null ) { // child is interface
284            if( parentInterval != null ) { // parent is not interface
285                return parent.equals( Scene.v().getObjectType().getSootClass());
286            } else {
287                return getAllSubinterfaces( parent ).contains( child );
288            }
289        } else {
290            Set<SootClass> impl = getAllImplementersOfInterface( parent );
291            for( Iterator<SootClass> it = impl.iterator(); it.hasNext(); ) {
292                parentInterval = classToInterval.get( it.next() );
293                if( parentInterval != null && parentInterval.isSubrange( childInterval ) ) {
294                    return true;
295                }
296            }
297            return false;
298        }
299    }

301    public Collection<SootMethod> resolveConcreteDispatchWithoutFailing(Collection<Type> concreteTypes, SootMethod m, RefType declaredTypeOfBase ) {
302
303        Set<SootMethod> ret = new HashSet<SootMethod>();
304        SootClass declaringClass = declaredTypeOfBase.getSootClass();
305        declaringClass.checkLevel(SootClass.HIERARCHY);
306        for( final Type t : concreteTypes ) {
307            if( t instanceof AnySubType ) {
308                HashSet<SootClass> s = new HashSet<SootClass>();
309                s.add( declaringClass );
310                while( !s.isEmpty() ) {
311                    SootClass c = (SootClass) s.iterator().next();
312                    s.remove( c );
313                    if( !c.isInterface() && !c.isAbstract()
314                            && canStoreClass( c, declaringClass ) ) {
315                        SootMethod concreteM = resolveConcreteDispatch( c, m );
316                        if( concreteM != null )
317                            ret.add( concreteM );
318                    }
319                    if( classToSubclasses.containsKey( c ) ) {
320                        s.addAll( classToSubclasses.get( c ) );
321                    }
322                    if( interfaceToSubinterfaces.containsKey( c ) ) {
323                        s.addAll( interfaceToSubinterfaces.get( c ) );
324                    }
325                    if( interfaceToImplementers.containsKey( c ) ) {
326                        s.addAll( interfaceToImplementers.get( c ) );
327                    }
328                }
329                return ret;
330            } else if( t instanceof RefType ) {
331                RefType concreteType = (RefType) t;
332                SootClass concreteClass = concreteType.getSootClass();
333                if( !canStoreClass( concreteClass, declaringClass ) ) {
334                    continue;
335                }
336                SootMethod concreteM = null;
337                try {
338                    concreteM = resolveConcreteDispatch( concreteClass, m );
339                } catch( Exception e ) {
340                    concreteM = null;
341                }
342                if( concreteM != null ) ret.add( concreteM );
343            } else if( t instanceof ArrayType ) {
344                SootMethod concreteM = null;
345                try {
346                    concreteM = resolveConcreteDispatch( 
347                        RefType.v( "java.lang.Object" ).getSootClass(), m );
348                } catch( Exception e ) {
349                    concreteM = null;
350                }
351                if( concreteM != null ) ret.add( concreteM );
352            } else throw new RuntimeException( "Unrecognized reaching type "+t );
353        }
354        return ret;
355    }

357    public Collection<SootMethod> resolveConcreteDispatch(Collection<Type> concreteTypes, SootMethod m, RefType declaredTypeOfBase ) {
358
359        Set<SootMethod> ret = new HashSet<SootMethod>();
360        SootClass declaringClass = declaredTypeOfBase.getSootClass();
361        declaringClass.checkLevel(SootClass.HIERARCHY);
362        for( final Type t : concreteTypes ) {
363            if( t instanceof AnySubType ) {
364                HashSet<SootClass> s = new HashSet<SootClass>();
365                s.add( declaringClass );
366                while( !s.isEmpty() ) {
367                    SootClass c = (SootClass) s.iterator().next();
368                    s.remove( c );
369                    if( !c.isInterface() && !c.isAbstract()
370                            && canStoreClass( c, declaringClass ) ) {
371                        SootMethod concreteM = resolveConcreteDispatch( c, m );
372                        if( concreteM != null )
373                            ret.add( concreteM );
374                    }
375                    if( classToSubclasses.containsKey( c ) ) {
376                        s.addAll( classToSubclasses.get( c ) );
377                    }
378                    if( interfaceToSubinterfaces.containsKey( c ) ) {
379                        s.addAll( interfaceToSubinterfaces.get( c ) );
380                    }
381                    if( interfaceToImplementers.containsKey( c ) ) {
382                        s.addAll( interfaceToImplementers.get( c ) );
383                    }
384                }
385                return ret;
386            } else if( t instanceof RefType ) {
387                RefType concreteType = (RefType) t;
388                SootClass concreteClass = concreteType.getSootClass();
389                if( !canStoreClass( concreteClass, declaringClass ) ) {
390                    continue;
391                }
392                SootMethod concreteM = resolveConcreteDispatch( concreteClass, m );
393                if( concreteM != null ) ret.add( concreteM );
394            } else if( t instanceof ArrayType ) {
395                SootMethod concreteM = resolveConcreteDispatch( 
396                        RefType.v( "java.lang.Object" ).getSootClass(), m );
397                if( concreteM != null ) ret.add( concreteM );
398            } else throw new RuntimeException( "Unrecognized reaching type "+t );
399        }
400        return ret;
401    }

405    /** Returns true if the method m is visible from code in the class from. */
406    private boolean isVisible( SootClass from, SootMethod m ) {
407        from.checkLevel(SootClass.HIERARCHY);
408        if( m.isPublic() ) return true;
409        if( m.isPrivate() ) {
410            return from.equals( m.getDeclaringClass() );
411        }
412        if( m.isProtected() ) {
413            return canStoreClass( from, m.getDeclaringClass() );
414        }
415        // m is package
416        return from.getJavaPackageName().equals(
417                m.getDeclaringClass().getJavaPackageName() );
418            //|| canStoreClass( from, m.getDeclaringClass() );
419    }

421    /** Given an object of declared type C, returns the methods which could
422     * be called on an o.f() invocation. */
423    public Set<SootMethod> resolveAbstractDispatch(SootClass abstractType, SootMethod m )
424    {
425        String methodSig = m.getSubSignature();
426        HashSet<SootClass> resolved = new HashSet<SootClass>();
427        HashSet<SootMethod> ret = new HashSet<SootMethod>();
428        LinkedList<SootClass> worklist = new LinkedList<SootClass>();
429        worklist.add( abstractType );
430        while( !worklist.isEmpty() ) {
431            SootClass concreteType = (SootClass) worklist.removeFirst();
432            SootClass savedConcreteType = concreteType;
433            if( concreteType.isInterface() ) {
434                worklist.addAll( getAllImplementersOfInterface( concreteType ) );
435                continue;
436            }
437            Collection<SootClass> c = classToSubclasses.get( concreteType );
438            if( c != null ) worklist.addAll( c );
439            if( !concreteType.isAbstract() ) {
440                while( true ) {
441                    if( resolved.contains( concreteType ) ) break;
442                    resolved.add( concreteType );
443                    SootMethod method = concreteType.getMethodUnsafe(methodSig);
444                    if( method != null ) {
445                        if ( isVisible(concreteType, m) ) {
446                                                        if (method.isAbstract())
447                                                                throw new RuntimeException("abstract dispatch resolved to abstract method!\nAbstract Type: "+abstractType+"\nConcrete Type: "+savedConcreteType+"\nMethod: "+m);
448                                                        else {
449                                                            ret.add( method );
450                                                            break;
451                                                        }                                                               
452                                                }
453                    }
454                    if( !concreteType.hasSuperclass() ) {
455                        if(concreteType.isPhantom())
456                                                        break;
457                                                else
458                                                        throw new RuntimeException("could not resolve abstract dispatch!\nAbstract Type: "+abstractType+"\nConcrete Type: "+savedConcreteType+"\nMethod: "+m);
459                    }
460                    concreteType = concreteType.getSuperclass();
461                }
462            }
463        }
464        return ret;
465    }

467    /** Given an object of actual type C (o = new C()), returns the method which will be called
468        on an o.f() invocation. */
469    public SootMethod resolveConcreteDispatch(SootClass concreteType, SootMethod m)
470    {
471        concreteType.checkLevel(SootClass.HIERARCHY);
472        if( concreteType.isInterface() ) {
473            throw new RuntimeException(
474                "A concrete type cannot be an interface: "+concreteType );
475        }
476
477        String methodSig = m.getSubSignature();
478        while( true ) {
479                SootMethod method = concreteType.getMethodUnsafe(methodSig);
480            if( method != null ) {
481                if( isVisible( concreteType, m ) ) {
482                    if(method.isAbstract()) {
483                        throw new RuntimeException("Error: Method call resolves to abstract method!");
484                    }
485                                        return method;
486                }
487            }
488            if( !concreteType.hasSuperclass() ) break;
489            concreteType = concreteType.getSuperclass();
490        }
491     // When there is no proper dispatch found, we simply return null to let the caller decide what to do 
492        return null;
493//        throw new RuntimeException("could not resolve concrete dispatch!\nType: "+concreteType+"\nMethod: "+m);
494    }

496    /** Returns the target for the given SpecialInvokeExpr. */
497    public SootMethod resolveSpecialDispatch(SpecialInvokeExpr ie, SootMethod container)
498    {
499        SootMethod target = ie.getMethod();
500
501        /* This is a bizarre condition!  Hopefully the implementation is correct.
502           See VM Spec, 2nd Edition, Chapter 6, in the definition of invokespecial. */
503        if (target.getName().equals("<init>") || target.isPrivate())
504            return target;
505        else if (isSubclass(target.getDeclaringClass(), container.getDeclaringClass()))
506            return resolveConcreteDispatch(container.getDeclaringClass(), target );
507        else
508            return target;
509    }

511  /**
512   * Gets the direct subclasses of a given class. The class needs to be resolved at least at the HIERARCHY level.
513   * @param c the class
514   * @return a collection (possibly empty) of the direct subclasses
515   */
516    public Collection<SootClass> getSubclassesOf( SootClass c ) {
517        c.checkLevel(SootClass.HIERARCHY);
518        Collection<SootClass> ret = classToSubclasses.get(c);
519        if( ret == null ) return Collections.emptyList();
520        return ret;
521    }
