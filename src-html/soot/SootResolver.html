047/** Loads symbols for SootClasses from either class files or jimple files. */
048public class SootResolver {
049        /** Maps each resolved class to a list of all references in it. */
050        private final Map<SootClass, Collection<Type>> classToTypesSignature = new HashMap<SootClass, Collection<Type>>();
051
052        /** Maps each resolved class to a list of all references in it. */
053        private final Map<SootClass, Collection<Type>> classToTypesHierarchy = new HashMap<SootClass, Collection<Type>>();
054
055        /** SootClasses waiting to be resolved. */
056        @SuppressWarnings("unchecked")
057        private final Deque<SootClass>[] worklist = new Deque[4];
058
059        private Program program = null;
060
061        public SootResolver(Singletons.Global g) {
062                worklist[SootClass.HIERARCHY] = new ArrayDeque<SootClass>();
063                worklist[SootClass.SIGNATURES] = new ArrayDeque<SootClass>();
064                worklist[SootClass.BODIES] = new ArrayDeque<SootClass>();
065                
066                if (Options.v().src_prec() != Options.src_prec_apk_c_j) {
067                        program = new Program();
068                        program.state().reset();
069        
070                        program.initBytecodeReader(new BytecodeParser());
071                        program.initJavaParser(new JavaParser() {
072                                public CompilationUnit parse(InputStream is, String fileName)
073                                                throws IOException, beaver.Parser.Exception {
074                                        return new JastAddJavaParser().parse(is, fileName);
075                                }
076                        });
077        
078                        program.options().initOptions();
079                        program.options().addKeyValueOption("-classpath");
080                        program.options().setValueForOption(Scene.v().getSootClassPath(),
081                                        "-classpath");
082                        if (Options.v().src_prec() == Options.src_prec_java)
083                                program.setSrcPrec(Program.SRC_PREC_JAVA);
084                        else if (Options.v().src_prec() == Options.src_prec_class)
085                                program.setSrcPrec(Program.SRC_PREC_CLASS);
086                        else if (Options.v().src_prec() == Options.src_prec_only_class)
087                                program.setSrcPrec(Program.SRC_PREC_CLASS);
088                        program.initPaths();
089                }
090        }
091
092        public static SootResolver v() {
093                return G.v().soot_SootResolver();
094        }

096        /** Returns true if we are resolving all class refs recursively. */
097        private boolean resolveEverything() {
098                if (Options.v().on_the_fly())
099                        return false;
100                return (Options.v().whole_program() || Options.v().whole_shimple()
101                                || Options.v().full_resolver() || Options.v().output_format() == Options.output_format_dava);
102        }
103
104        /**
105         * Returns a (possibly not yet resolved) SootClass to be used in references
106         * to a class. If/when the class is resolved, it will be resolved into this
107         * SootClass.
108         * */
109        public SootClass makeClassRef(String className) {
110                // If this class name is escaped, we need to un-escape it
111                className = Scene.v().unescapeName(className);
112                
113                if (Scene.v().containsClass(className))
114                        return Scene.v().getSootClass(className);
115
116                SootClass newClass;
117                newClass = new SootClass(className);
118                newClass.setResolvingLevel(SootClass.DANGLING);
119                Scene.v().addClass(newClass);
120
121                return newClass;
122        }
123
124        /**
125         * Resolves the given class. Depending on the resolver settings, may decide
126         * to resolve other classes as well. If the class has already been resolved,
127         * just returns the class that was already resolved.
128         * */
129        public SootClass resolveClass(String className, int desiredLevel) {
130                SootClass resolvedClass = null;
131                try {
132                        resolvedClass = makeClassRef(className);
133                        addToResolveWorklist(resolvedClass, desiredLevel);
134                        processResolveWorklist();
135                        return resolvedClass;
136                } catch (SootClassNotFoundException e) {
137                        // remove unresolved class and rethrow
138                        if (resolvedClass != null) {
139                                assert resolvedClass.resolvingLevel() == SootClass.DANGLING;
140                                Scene.v().removeClass(resolvedClass);
141                        }
142                        throw e;
143                }
144        }
145
146        /** Resolve all classes on toResolveWorklist. */
147        private void processResolveWorklist() {
148                for (int i = SootClass.BODIES; i >= SootClass.HIERARCHY; i--) {
149                        while (!worklist[i].isEmpty()) {
150                                SootClass sc = worklist[i].pop();
151                                if (resolveEverything()) { // Whole program mode
152                                        boolean onlySignatures = sc.isPhantom()
153                                                        || (Options.v().no_bodies_for_excluded()
154                                                                        && Scene.v().isExcluded(sc) && !Scene.v()
155                                                                        .getBasicClasses().contains(sc.getName()));
156                                        if (onlySignatures) {
157                                                bringToSignatures(sc);
158                                                sc.setPhantomClass();
159                                                for (SootMethod m : sc.getMethods()) {
160                                                        m.setPhantom(true);
161                                                }
162                                                for (SootField f : sc.getFields()) {
163                                                        f.setPhantom(true);
164                                                }
165                                        } else
166                                                bringToBodies(sc);
167                                } else { // No transitive
168                                        switch (i) {
169                                        case SootClass.BODIES:
170                                                bringToBodies(sc);
171                                                break;
172                                        case SootClass.SIGNATURES:
173                                                bringToSignatures(sc);
174                                                break;
175                                        case SootClass.HIERARCHY:
176                                                bringToHierarchy(sc);
177                                                break;
178                                        }
179                                }
180                        }
181                }
182        }
183
184        private void addToResolveWorklist(Type type, int level) {
185                // We go from Type -> SootClass directly, since RefType.getSootClass
186                // calls makeClassRef anyway
187                if (type instanceof RefType)
188                        addToResolveWorklist(((RefType) type).getSootClass(), level);
189                else if (type instanceof ArrayType)
190                        addToResolveWorklist(((ArrayType) type).baseType, level);
191                // Other types ignored
192        }
193
194        private void addToResolveWorklist(SootClass sc, int desiredLevel) {
195                if (sc.resolvingLevel() >= desiredLevel)
196                        return;
197                worklist[desiredLevel].add(sc);
198        }
199
200        /**
201         * Hierarchy - we know the hierarchy of the class and that's it requires at
202         * least Hierarchy for all supertypes and enclosing types.
203         * */
204        private void bringToHierarchy(SootClass sc) {
205                if (sc.resolvingLevel() >= SootClass.HIERARCHY)
206                        return;
207                if (Options.v().debug_resolver())
208                        G.v().out.println("bringing to HIERARCHY: " + sc);
209                sc.setResolvingLevel(SootClass.HIERARCHY);
210
211                String className = sc.getName();
212                ClassSource is = SourceLocator.v().getClassSource(className);
213                try {
214                        boolean modelAsPhantomRef = is == null;
215                        // || (
216                        // Options.v().no_jrl() &&
217                        // Scene.v().isExcluded(sc) &&
218                        // !Scene.v().getBasicClasses().contains(sc.getName())
219                        // );
220                        if (modelAsPhantomRef) {
221                                if (!Scene.v().allowsPhantomRefs()) {
222                                        String suffix = "";
223                                        if (className.equals("java.lang.Object")) {
224                                                suffix = " Try adding rt.jar to Soot's classpath, e.g.:\n"
225                                                                + "java -cp sootclasses.jar soot.Main -cp "
226                                                                + ".:/path/to/jdk/jre/lib/rt.jar <other options>";
227                                        } else if (className.equals("javax.crypto.Cipher")) {
228                                                suffix = " Try adding jce.jar to Soot's classpath, e.g.:\n"
229                                                                + "java -cp sootclasses.jar soot.Main -cp "
230                                                                + ".:/path/to/jdk/jre/lib/rt.jar:/path/to/jdk/jre/lib/jce.jar <other options>";
231                                        }
232                                        throw new SootClassNotFoundException("couldn't find class: "
233                                                        + className
234                                                        + " (is your soot-class-path set properly?)" + suffix);
235                                } else {
236                                        G.v().out.println("Warning: " + className
237                                                        + " is a phantom class!");
238                                        sc.setPhantomClass();
239                                        classToTypesSignature.put(sc, Collections.<Type> emptyList());
240                                        classToTypesHierarchy.put(sc, Collections.<Type> emptyList());
241                                }
242                        } else {
243                                Dependencies dependencies = is.resolve(sc);
244                                if (!dependencies.typesToSignature.isEmpty())
245                                        classToTypesSignature.put(sc, dependencies.typesToSignature);
246                                if (!dependencies.typesToHierarchy.isEmpty())
247                                        classToTypesHierarchy.put(sc, dependencies.typesToHierarchy);
248                        }
249                }
250                finally {
251                        if (is != null)
252                                is.close();
253                }
254                reResolveHierarchy(sc);
255        }
256
257        public void reResolveHierarchy(SootClass sc) {
258                // Bring superclasses to hierarchy
259                if (sc.hasSuperclass())
260                        addToResolveWorklist(sc.getSuperclass(), SootClass.HIERARCHY);
261                if (sc.hasOuterClass())
262                        addToResolveWorklist(sc.getOuterClass(), SootClass.HIERARCHY);
263                for (SootClass iface : sc.getInterfaces()) {
264                        addToResolveWorklist(iface, SootClass.HIERARCHY);
265                }
266        }
267
268        /**
269         * Signatures - we know the signatures of all methods and fields requires at
270         * least Hierarchy for all referred to types in these signatures.
271         * */
272        private void bringToSignatures(SootClass sc) {
273                if (sc.resolvingLevel() >= SootClass.SIGNATURES)
274                        return;
275                bringToHierarchy(sc);
276                if (Options.v().debug_resolver())
277                        G.v().out.println("bringing to SIGNATURES: " + sc);
278                sc.setResolvingLevel(SootClass.SIGNATURES);
279
280                for (SootField f : sc.getFields()) {
281                        addToResolveWorklist(f.getType(), SootClass.HIERARCHY);
282                }
283                for (SootMethod m : sc.getMethods()) {
284                        addToResolveWorklist(m.getReturnType(), SootClass.HIERARCHY);
285                        for (Type ptype : m.getParameterTypes()) {
286                                addToResolveWorklist(ptype, SootClass.HIERARCHY);
287                        }
288                        for (SootClass exception : m.getExceptions()) {
289                                addToResolveWorklist(exception, SootClass.HIERARCHY);
290                        }
291                }
292
293                // Bring superclasses to signatures
294                if (sc.hasSuperclass())
295                        addToResolveWorklist(sc.getSuperclass(), SootClass.SIGNATURES);
296                for (SootClass iface : sc.getInterfaces()) {
297                        addToResolveWorklist(iface, SootClass.SIGNATURES);
298                }
299        }
300
301        /**
302         * Bodies - we can now start loading the bodies of methods for all referred
303         * to methods and fields in the bodies, requires signatures for the method
304         * receiver and field container, and hierarchy for all other classes
305         * referenced in method references. Current implementation does not
306         * distinguish between the receiver and other references. Therefore, it is
307         * conservative and brings all of them to signatures. But this could/should
308         * be improved.
309         * */
310        private void bringToBodies(SootClass sc) {
311                if (sc.resolvingLevel() >= SootClass.BODIES)
312                        return;
313                bringToSignatures(sc);
314                if (Options.v().debug_resolver())
315                        G.v().out.println("bringing to BODIES: " + sc);
316                sc.setResolvingLevel(SootClass.BODIES);
317
318                {
319                        Collection<Type> references = classToTypesHierarchy.get(sc);
320                        if (references != null) {
321                                // This must be an iterator, not a for-all since the underlying
322                                // collection may change as we go
323                                Iterator<Type> it = references.iterator();
324                                while (it.hasNext()) {
325                                        final Type t = it.next();
326                                        addToResolveWorklist(t, SootClass.HIERARCHY);
327                                }
328                        }
329                }
330
331                {
332                        Collection<Type> references = classToTypesSignature.get(sc);
333                        if (references != null) {
334                                // This must be an iterator, not a for-all since the underlying
335                                // collection may change as we go
336                                Iterator<Type> it = references.iterator();
337                                while (it.hasNext()) {
338                                        final Type t = it.next();
339                                        addToResolveWorklist(t, SootClass.SIGNATURES);
340                                }
341                        }
342                }
343        }
344
345        public void reResolve(SootClass cl, int newResolvingLevel) {
346                int resolvingLevel = cl.resolvingLevel();
347                if (resolvingLevel >= newResolvingLevel)
348                        return;
349                reResolveHierarchy(cl);
350                cl.setResolvingLevel(newResolvingLevel);
351                addToResolveWorklist(cl, resolvingLevel);
352                processResolveWorklist();
353        }
354
355        public void reResolve(SootClass cl) {
356                reResolve(cl, SootClass.HIERARCHY);
357        }

360                return program;
361        }
