187                StackFrame frame = frames.get(insn);
188                if (frame == null) {
189                        frame = new StackFrame(this);
190                        frames.put(insn, frame);

195        private Local getLocal(int idx) {
196                if (idx >= maxLocals)
197                        throw new IllegalArgumentException("Invalid local index: " + idx);
198                Integer i = idx;
199                Local l = locals.get(i);
200                if (l == null) {
201                        String name;
202                        if (localVars != null) {
203                                name = null;
204                                for (LocalVariableNode lvn : localVars) {
205                                        if (lvn.index == idx) {
206                                                name = lvn.name;
207                                                break;
208                                        }
209                                }
210                                /* normally for try-catch blocks */
211                                if (name == null)
212                                        name = "l" + idx;
213                        } else {
214                                name = "l" + idx;
215                        }
216                        l = Jimple.v().newLocal(name, UnknownType.v());
217                        locals.put(i, l);
218                }
219                return l;
220        }

242        private Operand pop() {
243                if (stack.isEmpty())
244                        throw new RuntimeException("Stack underrun");
245                return stack.remove(stack.size() - 1);
246        }

248        private Operand popDual() {
249                Operand o = pop();
250                Operand o2 = pop();
251                if (o2 != DWORD_DUMMY && o2 != o)
252                        throw new AssertionError("Not dummy operand, " + o2.value + " -- " + o.value);
253                return o;
254        }

260        private Operand popLocal(Operand o) {
261                Value v = o.value;
262                Local l = o.stack;
263                if (l == null && !(v instanceof Local)) {
264                        l = o.stack = newStackLocal();
265                        setUnit(o.insn, Jimple.v().newAssignStmt(l, v));
266                        o.updateBoxes();
267                }
268                return o;
269        }

331        void setUnit(AbstractInsnNode insn, Unit u) {
332                if (Options.v().keep_line_number() && lastLineNumber >= 0) {
333                        Tag lineTag = u.getTag("LineNumberTag");
334                        if (lineTag == null) {
335                                lineTag = new LineNumberTag(lastLineNumber);
336                                u.addTag(lineTag);
337                        }
338                        else if (((LineNumberTag) lineTag).getLineNumber() != lastLineNumber)
339                                throw new RuntimeException("Line tag mismatch");
340                }
341                
342                Unit o = units.put(insn, u);
343                if (o != null)
344                        throw new AssertionError(insn.getOpcode() + " already has a unit, " + o);
345        }

347        void mergeUnits(AbstractInsnNode insn, Unit u) {
348                Unit prev = units.put(insn, u);
349                if (prev != null) {
350                        Unit merged = new UnitContainer(prev, u);
351                        units.put(insn, merged);
352                }
353        }

367        private void assignReadOps(Local l) {
368                if (stack.isEmpty())
369                        return;
370                for (Operand opr : stack) {
371                        if (opr == DWORD_DUMMY ||
372                                        opr.stack != null ||
373                                        (l == null && opr.value instanceof Local))
374                                continue;
375                        if (l != null && !opr.value.equivTo(l)) {
376                                List<ValueBox> uses = opr.value.getUseBoxes();
377                                boolean noref = true;
378                                for (ValueBox use : uses) {
379                                        Value val = use.getValue();
380                                        if (val.equivTo(l)) {
381                                                noref = false;
382                                                break;
383                                        }
384                                }
385                                if (noref)
386                                        continue;
387                        }
388                        int op = opr.insn.getOpcode();
389                        if (l == null && op != GETFIELD && op != GETSTATIC &&
390                                        (op < IALOAD && op > SALOAD))
391                                continue;
392                        Local stack = newStackLocal();
393                        opr.stack = stack;
394                        AssignStmt as = Jimple.v().newAssignStmt(stack, opr.value);
395                        opr.updateBoxes();
396                        setUnit(opr.insn, as);
397                }
398        }

400        private void convertGetFieldInsn(FieldInsnNode insn) {
401                StackFrame frame = getFrame(insn);
402                Operand[] out = frame.out();
403                Operand opr;
404                Type type;
405                if (out == null) {
406                        SootClass declClass = Scene.v().getSootClass(
407                                        AsmUtil.toQualifiedName(insn.owner));
408                        type = AsmUtil.toJimpleType(insn.desc);
409                        Value val;
410                        SootFieldRef ref;
411                        if (insn.getOpcode() == GETSTATIC) {
412                                ref = Scene.v().makeFieldRef(declClass, insn.name, type, true);
413                                val = Jimple.v().newStaticFieldRef(ref);
414                        } else {
415                                Operand base = popLocal();
416                                ref = Scene.v().makeFieldRef(declClass, insn.name, type, false);
417                                InstanceFieldRef ifr =
418                                                Jimple.v().newInstanceFieldRef(
419                                                                base.stackOrValue(), ref);
420                                val = ifr;
421                                base.addBox(ifr.getBaseBox());
422                                frame.in(base);
423                                frame.boxes(ifr.getBaseBox());
424                        }
425                        opr = new Operand(insn, val);
426                        frame.out(opr);
427                } else {
428                        opr = out[0];
429                        type = opr.<FieldRef>value().getFieldRef().type();
430                        if (insn.getOpcode() == GETFIELD)
431                                frame.mergeIn(pop());
432                }
433                push(type, opr);
434        }

436        private void convertPutFieldInsn(FieldInsnNode insn) {
437                boolean instance = insn.getOpcode() == PUTFIELD;
438                StackFrame frame = getFrame(insn);
439                Operand[] out = frame.out();
440                Operand opr, rvalue;
441                Type type;
442                if (out == null) {
443                        SootClass declClass = Scene.v().getSootClass(
444                                        AsmUtil.toQualifiedName(insn.owner));
445                        type = AsmUtil.toJimpleType(insn.desc);
446                        Value val;
447                        SootFieldRef ref;
448                        rvalue = popImmediate(type);
449                        if (!instance) {
450                                ref = Scene.v().makeFieldRef(declClass, insn.name, type, true);
451                                val = Jimple.v().newStaticFieldRef(ref);
452                                frame.in(rvalue);
453                        } else {
454                                Operand base = popLocal();
455                                ref = Scene.v().makeFieldRef(declClass, insn.name, type, false);
456                                InstanceFieldRef ifr =
457                                                Jimple.v().newInstanceFieldRef(
458                                                                base.stackOrValue(), ref);
459                                val = ifr;
460                                base.addBox(ifr.getBaseBox());
461                                frame.in(rvalue, base);
462                        }
463                        opr = new Operand(insn, val);
464                        frame.out(opr);
465                        AssignStmt as = Jimple.v().newAssignStmt(val, rvalue.stackOrValue()); 
466                        rvalue.addBox(as.getRightOpBox());
467                        if (!instance) {
468                                frame.boxes(as.getRightOpBox());
469                        } else {
470                                frame.boxes(as.getRightOpBox(),
471                                                ((InstanceFieldRef) val).getBaseBox());
472                        }
473                        setUnit(insn, as);
474                } else {
475                        opr = out[0];
476                        type = opr.<FieldRef>value().getFieldRef().type();
477                        rvalue = pop(type);
478                        if (!instance) {
479                                /* PUTSTATIC only needs one operand on the stack, the rvalue */
480                                frame.mergeIn(rvalue);
481                        } else {
482                                /* PUTFIELD has a rvalue and a base */
483                                frame.mergeIn(rvalue, pop());
484                        }
485                }
486                /*
487                 * in case any static field or array is read from, and the static constructor
488                 * or the field this instruction writes to, modifies that field, write out any
489                 * previous read from field/array
490                 */
491                assignReadOps(null);
492        }

511        private void convertConstInsn(InsnNode insn) {
512                int op = insn.getOpcode();
513                StackFrame frame = getFrame(insn);
514                Operand[] out = frame.out();
515                Operand opr;
516                if (out == null) {
517                        Value v;
518                        if (op == ACONST_NULL)
519                                v = NullConstant.v();
520                        else if (op >= ICONST_M1 && op <= ICONST_5)
521                                v = IntConstant.v(op - ICONST_0);
522                        else if (op == LCONST_0 || op == LCONST_1)
523                                v = LongConstant.v(op - LCONST_0);
524                        else if (op >= FCONST_0 && op <= FCONST_2)
525                                v = FloatConstant.v(op - FCONST_0);
526                        else if (op == DCONST_0 || op == DCONST_1)
527                                v = DoubleConstant.v(op - DCONST_0);
528                        else
529                                throw new AssertionError("Unknown constant opcode: " + op);
530                        opr = new Operand(insn, v);
531                        frame.out(opr);
532                } else {
533                        opr = out[0];
534                }
535                if (op == LCONST_0 || op == LCONST_1 ||
536                                op == DCONST_0 || op == DCONST_1) {
537                        pushDual(opr);
538                } else {
539                        push(opr);
540                }
541        }

543        private void convertArrayLoadInsn(InsnNode insn) {
544                StackFrame frame = getFrame(insn);
545                Operand[] out = frame.out();
546                Operand opr;
547                if (out == null) {
548                        Operand indx = popImmediate();
549                        Operand base = popImmediate();
550                        ArrayRef ar = Jimple.v().newArrayRef(
551                                        base.stackOrValue(), indx.stackOrValue());
552                        indx.addBox(ar.getIndexBox());
553                        base.addBox(ar.getBaseBox());
554                        opr = new Operand(insn, ar);
555                        frame.in(indx, base);
556                        frame.boxes(ar.getIndexBox(), ar.getBaseBox());
557                        frame.out(opr);
558                } else {
559                        opr = out[0];
560                        frame.mergeIn(pop(), pop());
561                }
562                int op = insn.getOpcode();
563                if (op == DALOAD || op == LALOAD)
564                        pushDual(opr);
565                else
566                        push(opr);
567        }

569        private void convertArrayStoreInsn(InsnNode insn) {
570                int op = insn.getOpcode();
571                boolean dword = op == LASTORE || op == DASTORE;
572                StackFrame frame = getFrame(insn);
573                if (!units.containsKey(insn)) {
574                        Operand valu = dword ? popImmediateDual() : popImmediate();
575                        Operand indx = popImmediate();
576                        Operand base = popLocal();
577                        ArrayRef ar = Jimple.v().newArrayRef(
578                                        base.stackOrValue(), indx.stackOrValue());
579                        indx.addBox(ar.getIndexBox());
580                        base.addBox(ar.getBaseBox());
581                        AssignStmt as = Jimple.v().newAssignStmt(ar, valu.stackOrValue());
582                        valu.addBox(as.getRightOpBox());
583                        frame.in(valu, indx, base);
584                        frame.boxes(as.getRightOpBox(),
585                                        ar.getIndexBox(), ar.getBaseBox());
586                        setUnit(insn, as);
587                } else {
588                        frame.mergeIn(dword ? popDual() : pop(), pop(), pop());
589                }
590        }

631        private void convertDupInsn(InsnNode insn) {
632                int op = insn.getOpcode();
633                
634                // Get the top stack value which we need in either case
635                Operand dupd = popImmediate();
636                Operand dupd2 = null;
637                
638                // Some instructions allow operands that take two registers
639                boolean dword = op == DUP2 || op == DUP2_X1 || op == DUP2_X2;
640                if (dword) {
641                        if (peek() == DWORD_DUMMY) {
642                                pop();
643                                dupd2 = dupd;
644                        } else
645                                dupd2 = popImmediate();
646                }
647                
648                if (op == DUP) {
649                        // val -> val, val
650                        push(dupd);
651                        push(dupd);
652                } else if (op == DUP_X1) {
653                        // val2, val1 -> val1, val2, val1
654                        // value1, value2 must not be of type double or long
655                        Operand o2 = popImmediate();
656                        push(dupd);
657                        push(o2);
658                        push(dupd);
659                } else if (op == DUP_X2) {
660                        // value3, value2, value1 -> value1, value3, value2, value1
661                        Operand o2 = popImmediate();
662                        Operand o3 = peek() == DWORD_DUMMY ? pop() : popImmediate();
663                        push(dupd);
664                        push(o3);
665                        push(o2);
666                        push(dupd);
667                } else if (op == DUP2) {
668                        // value2, value1 -> value2, value1, value2, value1
669                        push(dupd2);
670                        push(dupd);
671                        push(dupd2);
672                        push(dupd);
673                } else if (op == DUP2_X1) {
674                        // value3, value2, value1 -> value2, value1, value3, value2, value1
675                        // Attention: value2 may be
676                        Operand o2 = popImmediate();
677                        push(dupd2);
678                        push(dupd);
679                        push(o2);
680                        push(dupd2);
681                        push(dupd);
682                } else if (op == DUP2_X2) {
683                        // (value4, value3), (value2, value1) -> (value2, value1), (value4, value3), (value2, value1) 
684                        Operand o2 = popImmediate();
685                        Operand o2h = peek() == DWORD_DUMMY ? pop() : popImmediate();
686                        push(dupd2);
687                        push(dupd);
688                        push(o2h);
689                        push(o2);
690                        push(dupd2);
691                        push(dupd);
692                }
693        }

695        private void convertBinopInsn(InsnNode insn) {
696                int op = insn.getOpcode();
697                boolean dword = op == DADD || op == LADD ||
698                                op == DSUB || op == LSUB ||
699                                op == DMUL || op == LMUL ||
700                                op == DDIV || op == LDIV ||
701                                op == DREM || op == LREM ||
702                                op == LSHL || op == LSHR ||
703                                op == LUSHR || op == LAND ||
704                                op == LOR || op == LXOR ||
705                                op == LCMP || op == DCMPL ||
706                                op == DCMPG;
707                StackFrame frame = getFrame(insn);
708                Operand[] out = frame.out();
709                Operand opr;
710                if (out == null) {
711                        Operand op2 = (dword && op != LSHL && op != LSHR && op != LUSHR) ?
712                                        popImmediateDual() : popImmediate();
713                        Operand op1 = dword ? popImmediateDual() : popImmediate();
714                        Value v1 = op1.stackOrValue();
715                        Value v2 = op2.stackOrValue();
716                        BinopExpr binop;
717                        if (op >= IADD && op <= DADD)
718                                binop = Jimple.v().newAddExpr(v1, v2);
719                        else if (op >= ISUB && op <= DSUB)
720                                binop = Jimple.v().newSubExpr(v1, v2);
721                        else if (op >= IMUL && op <= DMUL)
722                                binop = Jimple.v().newMulExpr(v1, v2);
723                        else if (op >= IDIV && op <= DDIV)
724                                binop = Jimple.v().newDivExpr(v1, v2);
725                        else if (op >= IREM && op <= DREM)
726                                binop = Jimple.v().newRemExpr(v1, v2);
727                        else if (op >= ISHL && op <= LSHL)
728                                binop = Jimple.v().newShlExpr(v1, v2);
729                        else if (op >= ISHR && op <= LSHR)
730                                binop = Jimple.v().newShrExpr(v1, v2);
731                        else if (op >= IUSHR && op <= LUSHR)
732                                binop = Jimple.v().newUshrExpr(v1, v2);
733                        else if (op >= IAND && op <= LAND)
734                                binop = Jimple.v().newAndExpr(v1, v2);
735                        else if (op >= IOR && op <= LOR)
736                                binop = Jimple.v().newOrExpr(v1, v2);
737                        else if (op >= IXOR && op <= LXOR)
738                                binop = Jimple.v().newXorExpr(v1, v2);
739                        else if (op == LCMP)
740                                binop = Jimple.v().newCmpExpr(v1, v2);
741                        else if (op == FCMPL || op == DCMPL)
742                                binop = Jimple.v().newCmplExpr(v1, v2);
743                        else if (op == FCMPG || op == DCMPG)
744                                binop = Jimple.v().newCmpgExpr(v1, v2);
745                        else
746                                throw new AssertionError("Unknown binop: " + op);
747                        op1.addBox(binop.getOp1Box());
748                        op2.addBox(binop.getOp2Box());
749                        opr = new Operand(insn, binop);
750                        frame.in(op2, op1);
751                        frame.boxes(binop.getOp2Box(), binop.getOp1Box());
752                        frame.out(opr);
753                } else {
754                        opr = out[0];
755                        if (dword) {
756                                if (op != LSHL && op != LSHR && op != LUSHR)
757                                        frame.mergeIn(popDual(), popDual());
758                                else
759                                        frame.mergeIn(pop(), popDual());
760                        } else {
761                                frame.mergeIn(pop(), pop());
762                        }
763                }
764                if (dword && (op < LCMP || op > DCMPG))
765                        pushDual(opr);
766                else
767                        push(opr);
768        }

770        private void convertUnopInsn(InsnNode insn) {
771                int op = insn.getOpcode();
772                boolean dword = op == LNEG || op == DNEG;
773                StackFrame frame = getFrame(insn);
774                Operand[] out = frame.out();
775                Operand opr;
776                if (out == null) {
777                        Operand op1 = dword ? popImmediateDual() : popImmediate();
778                        Value v1 = op1.stackOrValue();
779                        UnopExpr unop;
780                        if (op >= INEG && op <= DNEG)
781                                unop = Jimple.v().newNegExpr(v1);
782                        else if (op == ARRAYLENGTH)

784                        else
785                                throw new AssertionError("Unknown unop: " + op);
786                        op1.addBox(unop.getOpBox());
787                        opr = new Operand(insn, unop);
788                        frame.in(op1);
789                        frame.boxes(unop.getOpBox());
790                        frame.out(opr);
791                } else {
792                        opr = out[0];
793                        frame.mergeIn(dword ? popDual() : pop());
794                }
795                if (dword)
796                        pushDual(opr);
797                else
798                        push(opr);
799        }

863        private void convertInsn(InsnNode insn) {
864                int op = insn.getOpcode();
865                if (op == NOP) {
866                        /*
867                         * We can ignore NOP instructions, but
868                         * for completeness, we handle them
869                         */
870                        if (!units.containsKey(insn))
871                                units.put(insn, Jimple.v().newNopStmt());
872                } else if (op >= ACONST_NULL && op <= DCONST_1) {
873                        convertConstInsn(insn);
874                } else if (op >= IALOAD && op <= SALOAD) {
875                        convertArrayLoadInsn(insn);
876                } else if (op >= IASTORE && op <= SASTORE) {
877                        convertArrayStoreInsn(insn);
878                } else if (op == POP) {
879                        popImmediate();
880                } else if (op == POP2) {
881                        popImmediate();
882                        if (peek() == DWORD_DUMMY)
883                                pop();
884                        else
885                                popImmediate();
886                } else if (op >= DUP && op <= DUP2_X2) {
887                        convertDupInsn((InsnNode) insn);
888                } else if (op == SWAP) {
889                        Operand o1 = popImmediate();
890                        Operand o2 = popImmediate();
891                        push(o1);
892                        push(o2);
893                } else if ((op >= IADD && op <= DREM) ||
894                                (op >= ISHL && op <= LXOR) ||
895                                (op >= LCMP && op <= DCMPG)) {
896                        convertBinopInsn((InsnNode) insn);
897                } else if ((op >= INEG && op <= DNEG) ||
898                                op == ARRAYLENGTH) {
899                        convertUnopInsn(insn);
900                } else if (op >= I2L && op <= I2S) {
901                        convertPrimCastInsn(insn);
902                } else if (op >= IRETURN && op <= ARETURN) {
903                        convertReturnInsn(insn);
904                } else if (op == RETURN) {
905                        if (!units.containsKey(insn))
906                                setUnit(insn, Jimple.v().newReturnVoidStmt());
907                } else if (op == ATHROW) {
908                        StackFrame frame = getFrame(insn);
909                        Operand opr;
910                        if (!units.containsKey(insn)) {
911                                opr = popImmediate();
912                                ThrowStmt ts = Jimple.v().newThrowStmt(opr.stackOrValue());
913                                opr.addBox(ts.getOpBox());
914                                frame.in(opr);
915                                frame.out(opr);
916                                frame.boxes(ts.getOpBox());
917                                setUnit(insn, ts);
918                        } else {
919                                opr = pop();
920                                frame.mergeIn(opr);
921                        }
922                        push(opr);
923                } else if (op == MONITORENTER || op == MONITOREXIT) {
924                        StackFrame frame = getFrame(insn);
925                        if (!units.containsKey(insn)) {
926                                Operand opr = popStackConst();
927                                MonitorStmt ts = op == MONITORENTER ?
928                                                Jimple.v().newEnterMonitorStmt(opr.stackOrValue()) :
929                                                Jimple.v().newExitMonitorStmt(opr.stackOrValue());
930                                opr.addBox(ts.getOpBox());
931                                frame.in(opr);
932                                frame.boxes(ts.getOpBox());
933                                setUnit(insn, ts);
934                        } else {
935                                frame.mergeIn(pop());
936                        }
937                } else {
938                        throw new AssertionError("Unknown insn op: " + op);
939                }
940        }

942        private void convertIntInsn(IntInsnNode insn) {
943                int op = insn.getOpcode();
944                StackFrame frame = getFrame(insn);
945                Operand[] out = frame.out();
946                Operand opr;
947                if (out == null) {
948                        Value v;
949                        if (op == BIPUSH || op == SIPUSH) {
950                                v = IntConstant.v(insn.operand);
951                        } else {
952                                Type type;
953                                switch (insn.operand) {
954                                case T_BOOLEAN:
955                                        type = BooleanType.v();
956                                        break;
957                                case T_CHAR:
958                                        type = CharType.v();
959                                        break;
960                                case T_FLOAT:
961                                        type = FloatType.v();
962                                        break;
963                                case T_DOUBLE:
964                                        type = DoubleType.v();
965                                        break;
966                                case T_BYTE:
967                                        type = ByteType.v();
968                                        break;
969                                case T_SHORT:
970                                        type = ShortType.v();
971                                        break;
972                                case T_INT:
973                                        type = IntType.v();
974                                        break;
975                                case T_LONG:
976                                        type = LongType.v();
977                                        break;
978                                default:
979                                        throw new AssertionError("Unknown NEWARRAY type!");
980                                }
981                                Operand size = popImmediate();
982                                NewArrayExpr anew = Jimple.v().newNewArrayExpr(type, size.stackOrValue());
983                                size.addBox(anew.getSizeBox());
984                                frame.in(size);
985                                frame.boxes(anew.getSizeBox());
986                                v = anew;
987                        }
988                        opr = new Operand(insn, v);
989                        frame.out(opr);
990                } else {
991                        opr = out[0];
992                        if (op == NEWARRAY)
993                                frame.mergeIn(pop());
994                }
995                push(opr);
996        }

998        private void convertJumpInsn(JumpInsnNode insn) {
999                int op = insn.getOpcode();
1000                if (op == GOTO) {
1001                        if (!units.containsKey(insn)) {
1002                                UnitBox box = Jimple.v().newStmtBox(null);
1003                                labels.put(insn.label, box);
1004                                setUnit(insn, Jimple.v().newGotoStmt(box));
1005                        }
1006                        return;
1007                }
1008                /* must be ifX insn */
1009                StackFrame frame = getFrame(insn);
1010                if (!units.containsKey(insn)) {
1011                        Operand val = popImmediate();
1012                        Value v = val.stackOrValue();
1013                        ConditionExpr cond;
1014                        if (op >= IF_ICMPEQ && op <= IF_ACMPNE) {
1015                                Operand val1 = popImmediate();
1016                                Value v1 = val1.stackOrValue();
1017                                if (op == IF_ICMPEQ)
1018                                        cond = Jimple.v().newEqExpr(v1, v);
1019                                else if (op == IF_ICMPNE)
1020                                        cond = Jimple.v().newNeExpr(v1, v);
1021                                else if (op == IF_ICMPLT)
1022                                        cond = Jimple.v().newLtExpr(v1, v);
1023                                else if (op == IF_ICMPGE)
1024                                        cond = Jimple.v().newGeExpr(v1, v);
1025                                else if (op == IF_ICMPGT)
1026                                        cond = Jimple.v().newGtExpr(v1, v);
1027                                else if (op == IF_ICMPLE)
1028                                        cond = Jimple.v().newLeExpr(v1, v);
1029                                else if (op == IF_ACMPEQ)
1030                                        cond = Jimple.v().newEqExpr(v1, v);
1031                                else if (op == IF_ACMPNE)
1032                                        cond = Jimple.v().newNeExpr(v1, v);
1033                                else
1034                                        throw new AssertionError("Unknown if op: " + op);
1035                                val1.addBox(cond.getOp1Box());
1036                                val.addBox(cond.getOp2Box());
1037                                frame.boxes(cond.getOp2Box(), cond.getOp1Box());
1038                                frame.in(val, val1);
1039                        } else {
1040                                if (op == IFEQ)
1041                                        cond = Jimple.v().newEqExpr(v, IntConstant.v(0));
1042                                else if (op == IFNE)
1043                                        cond = Jimple.v().newNeExpr(v, IntConstant.v(0));
1044                                else if (op == IFLT)
1045                                        cond = Jimple.v().newLtExpr(v, IntConstant.v(0));
1046                                else if (op == IFGE)
1047                                        cond = Jimple.v().newGeExpr(v, IntConstant.v(0));
1048                                else if (op == IFGT)
1049                                        cond = Jimple.v().newGtExpr(v, IntConstant.v(0));
1050                                else if (op == IFLE)
1051                                        cond = Jimple.v().newLeExpr(v, IntConstant.v(0));
1052                                else if (op == IFNULL)
1053                                        cond = Jimple.v().newEqExpr(v, NullConstant.v());
1054                                else if (op == IFNONNULL)
1055                                        cond = Jimple.v().newNeExpr(v, NullConstant.v());
1056                                else
1057                                        throw new AssertionError("Unknown if op: " + op);
1058                                val.addBox(cond.getOp1Box());
1059                                frame.boxes(cond.getOp1Box());
1060                                frame.in(val);
1061                        }
1062                        UnitBox box = Jimple.v().newStmtBox(null);
1063                        labels.put(insn.label, box);
1064                        setUnit(insn, Jimple.v().newIfStmt(cond, box));
1065                } else {
1066                        if (op >= IF_ICMPEQ && op <= IF_ACMPNE)
1067                                frame.mergeIn(pop(), pop());
1068                        else
1069                                frame.mergeIn(pop());
1070                }
1071        }

1113        private void convertLookupSwitchInsn(LookupSwitchInsnNode insn) {
1114                StackFrame frame = getFrame(insn);
1115                if (units.containsKey(insn)) {
1116                        frame.mergeIn(pop());
1117                        return;
1118                }
1119                Operand key = popImmediate();
1120                UnitBox dflt = Jimple.v().newStmtBox(null);
1121                
1122                List<UnitBox> targets = new ArrayList<UnitBox>(insn.labels.size());
1123                labels.put(insn.dflt, dflt);
1124                for (LabelNode ln : insn.labels) {
1125                        UnitBox box = Jimple.v().newStmtBox(null);
1126                        targets.add(box);
1127                        labels.put(ln, box);
1128                }
1129                
1130                List<IntConstant> keys = new ArrayList<IntConstant>(insn.keys.size());
1131                for (Integer i : insn.keys)
1132                        keys.add(IntConstant.v(i));

1134                LookupSwitchStmt lss = Jimple.v().newLookupSwitchStmt(key.stackOrValue(),
1135                                keys, targets, dflt);
1136                key.addBox(lss.getKeyBox());
1137                frame.in(key);
1138                frame.boxes(lss.getKeyBox());
1139                setUnit(insn, lss);
1140        }

1142        private void convertMethodInsn(MethodInsnNode insn) {
1143                int op = insn.getOpcode();
1144                boolean instance = op != INVOKESTATIC;
1145                StackFrame frame = getFrame(insn);
1146                Operand[] out = frame.out();
1147                Operand opr;
1148                Type returnType;
1149                if (out == null) {
1150                        String clsName = AsmUtil.toQualifiedName(insn.owner);
1151                        if (clsName.charAt(0) == '[')
1152                                clsName = "java.lang.Object";
1153                        SootClass cls = Scene.v().getSootClass(clsName);
1154                        List<Type> sigTypes = AsmUtil.toJimpleDesc(insn.desc);
1155                        returnType = sigTypes.remove(sigTypes.size() - 1);
1156                        SootMethodRef ref =
1157                                        Scene.v().makeMethodRef(cls, insn.name, sigTypes, returnType, !instance);
1158                        int nrArgs = sigTypes.size();
1159                        final Operand[] args;
1160                        List<Value> argList = Collections.emptyList();
1161                        if (!instance) {
1162                                args = nrArgs == 0 ? null : new Operand[nrArgs];
1163                                if (args != null)
1164                                        argList = new ArrayList<Value>(nrArgs);
1165                        } else {
1166                                args = new Operand[nrArgs + 1];
1167                                if (nrArgs != 0)
1168                                        argList = new ArrayList<Value>(nrArgs);
1169                        }
1170                        while (nrArgs-- != 0) {
1171                                args[nrArgs] = popImmediate(sigTypes.get(nrArgs));
1172                                argList.add(args[nrArgs].stackOrValue());
1173                        }
1174                        if (argList.size() > 1)
1175                                Collections.reverse(argList);
1176                        if (instance)
1177                                args[args.length - 1] = popLocal();
1178                        ValueBox[] boxes = args == null ? null : new ValueBox[args.length];
1179                        InvokeExpr invoke;
1180                        if (!instance) {
1181                                invoke = Jimple.v().newStaticInvokeExpr(ref, argList);
1182                        } else {
1183                                Local base = (Local) args[args.length - 1].stackOrValue();
1184                                InstanceInvokeExpr iinvoke;
1185                                if (op == INVOKESPECIAL)
1186                                        iinvoke = Jimple.v().newSpecialInvokeExpr(base, ref, argList);
1187                                else if (op == INVOKEVIRTUAL)
1188                                        iinvoke = Jimple.v().newVirtualInvokeExpr(base, ref, argList);
1189                                else if (op == INVOKEINTERFACE)
1190                                        iinvoke = Jimple.v().newInterfaceInvokeExpr(base, ref, argList);
1191                                else
1192                                        throw new AssertionError("Unknown invoke op:" + op);
1193                                boxes[boxes.length - 1] = iinvoke.getBaseBox();
1194                                args[args.length - 1].addBox(boxes[boxes.length - 1]);
1195                                invoke = iinvoke;
1196                        }
1197                        if (boxes != null) {
1198                                for (int i = 0; i != sigTypes.size(); i++) {
1199                                        boxes[i] = invoke.getArgBox(i);
1200                                        args[i].addBox(boxes[i]);
1201                                }
1202                                frame.boxes(boxes);
1203                                frame.in(args);
1204                        }
1205                        opr = new Operand(insn, invoke);
1206                        frame.out(opr);
1207                } else {
1208                        opr = out[0];
1209                        InvokeExpr expr = (InvokeExpr) opr.value;
1210                        List<Type> types = expr.getMethodRef().parameterTypes();
1211                        Operand[] oprs;
1212                        int nrArgs = types.size();
1213                        if (expr.getMethodRef().isStatic())
1214                                oprs = nrArgs == 0 ? null : new Operand[nrArgs];
1215                        else
1216                                oprs = new Operand[nrArgs + 1];
1217                        if (oprs != null) {
1218                                while (nrArgs-- != 0) {
1219                                        oprs[nrArgs] = pop(types.get(nrArgs));
1220                                }
1221                                if (!expr.getMethodRef().isStatic())
1222                                        oprs[oprs.length - 1] = pop();
1223                                frame.mergeIn(oprs);
1224                                nrArgs = types.size();
1225                        }
1226                        returnType = expr.getMethodRef().returnType();
1227                }
1228                if (AsmUtil.isDWord(returnType))
1229                        pushDual(opr);
1230                else if (!(returnType instanceof VoidType))
1231                        push(opr);
1232                else if (!units.containsKey(insn))
1233                        setUnit(insn, Jimple.v().newInvokeStmt(opr.value));
1234                /*
1235                 * assign all read ops in case the method modifies any of the fields
1236                 */
1237                assignReadOps(null);
1238        }

1240        private void convertInvokeDynamicInsn(InvokeDynamicInsnNode insn) {
1241                StackFrame frame = getFrame(insn);
1242                Operand[] out = frame.out();
1243                Operand opr;
1244                Type returnType;
1245                if (out == null) {
1246                        //convert info on bootstrap method
1247                        SootMethodRef bsmMethodRef = toSootMethodRef(insn.bsm);
1248                        List<Value> bsmMethodArgs = new ArrayList<Value>(insn.bsmArgs.length);
1249                        for(Object bsmArg: insn.bsmArgs) {
1250                                bsmMethodArgs.add(toSootValue(bsmArg));
1251                        }
1252                        
1253                        // create ref to actual method
1254                        SootClass bclass = Scene.v().getSootClass(SootClass.INVOKEDYNAMIC_DUMMY_CLASS_NAME);
1255                        
1256                        // Generate parameters & returnType & parameterTypes
1257                        Type[] types = Util.v().jimpleTypesOfFieldOrMethodDescriptor(insn.desc);
1258                        List<Type> parameterTypes = new ArrayList<Type>(types.length);
1259                        List<Value> methodArgs = new ArrayList<Value>(types.length);
1260
1261                        Operand[] args = new Operand[types.length - 1];
1262                        ValueBox[] boxes = new ValueBox[args.length];
1263
1264                        int nrArgs = args.length;
1265                        while (nrArgs-- != 0) {
1266                                parameterTypes.add(types[nrArgs]);
1267                                args[nrArgs] = popImmediate(types[nrArgs]);
1268                                methodArgs.add(args[nrArgs].stackOrValue());                            
1269                        }
1270                        if (methodArgs.size() > 1)
1271                                Collections.reverse(methodArgs);        // Call stack is FIFO, Jimple is linear
1272                        
1273                        returnType = types[types.length - 1];
1274                        
1275                        // we always model invokeDynamic method refs as static method references
1276                        // of methods on the type SootClass.INVOKEDYNAMIC_DUMMY_CLASS_NAME
1277                        SootMethodRef methodRef = Scene.v().makeMethodRef(bclass, insn.name, parameterTypes, returnType, true);         
1278                        
1279                        DynamicInvokeExpr indy = Jimple.v().newDynamicInvokeExpr(bsmMethodRef,
1280                                        bsmMethodArgs, methodRef, insn.bsm.getTag(), methodArgs);
1281                        
1282                        for (int i = 0; i < args.length - 1; i++) {
1283                                boxes[i] = indy.getArgBox(i);
1284                                args[i].addBox(boxes[i]);
1285                        }
1286                        
1287                        opr = new Operand(insn,indy);
1288                        frame.boxes(boxes);
1289                        frame.in(args);
1290                        frame.out(opr);
1291                } else {
1292                        opr = out[0];
1293                        InvokeExpr expr = (InvokeExpr) opr.value;
1294                        List<Type> types = expr.getMethodRef().parameterTypes();
1295                        Operand[] oprs;
1296                        int nrArgs = types.size();
1297                        if (expr.getMethodRef().isStatic())
1298                                oprs = nrArgs == 0 ? null : new Operand[nrArgs];
1299                        else
1300                                oprs = new Operand[nrArgs + 1];
1301                        if (oprs != null) {
1302                                while (nrArgs-- != 0) {
1303                                        oprs[nrArgs] = pop(types.get(types.size() - nrArgs - 1));
1304                                }
1305                                if (!expr.getMethodRef().isStatic())
1306                                        oprs[oprs.length - 1] = pop();
1307                                frame.mergeIn(oprs);
1308                                nrArgs = types.size();
1309                        }
1310                        returnType = expr.getMethodRef().returnType();
1311                }
1312                if (AsmUtil.isDWord(returnType))
1313                        pushDual(opr);
1314                else if (!(returnType instanceof VoidType))
1315                        push(opr);
1316                else if (!units.containsKey(insn))
1317                        setUnit(insn, Jimple.v().newInvokeStmt(opr.value));
1318                /*
1319                 * assign all read ops in case the method modifies any of the fields
1320                 */
1321                assignReadOps(null);
1322        }

1367        private void convertTableSwitchInsn(TableSwitchInsnNode insn) {
1368                StackFrame frame = getFrame(insn);
1369                if (units.containsKey(insn)) {
1370                        frame.mergeIn(pop());
1371                        return;
1372                }
1373                Operand key = popImmediate();
1374                UnitBox dflt = Jimple.v().newStmtBox(null);

1390        private void convertTypeInsn(TypeInsnNode insn) {
1391                int op = insn.getOpcode();
1392                StackFrame frame = getFrame(insn);
1393                Operand[] out = frame.out();
1394                Operand opr;
1395                if (out == null) {
1396                        Type t = AsmUtil.toJimpleRefType(insn.desc);
1397                        Value val;
1398                        if (op == NEW) {
1399                                val = Jimple.v().newNewExpr((RefType) t);
1400                        } else {
1401                                Operand op1 = popImmediate();
1402                                Value v1 = op1.stackOrValue();
1403                                ValueBox vb;
1404                                if (op == ANEWARRAY) {
1405                                        NewArrayExpr expr = Jimple.v().newNewArrayExpr(t, v1);
1406                                        vb = expr.getSizeBox();
1407                                        val = expr;
1408                                } else if (op == CHECKCAST) {
1409                                        CastExpr expr = Jimple.v().newCastExpr(v1, t);
1410                                        vb = expr.getOpBox();
1411                                        val = expr;
1412                                } else if (op == INSTANCEOF) {
1413                                        InstanceOfExpr expr = Jimple.v().newInstanceOfExpr(v1, t);
1414                                        vb = expr.getOpBox();
1415                                        val = expr;
1416                                } else {
1417                                        throw new AssertionError("Unknown type op: " + op);
1418                                }
1419                                op1.addBox(vb);
1420                                frame.in(op1);
1421                                frame.boxes(vb);
1422                        }
1423                        opr = new Operand(insn, val);
1424                        frame.out(opr);
1425                } else {
1426                        opr = out[0];
1427                        if (op != NEW)
1428                                frame.mergeIn(pop());
1429                }
1430                push(opr);
1431        }

1522        private final class Edge {
1523                /* edge endpoint */
1524                final AbstractInsnNode insn;
1525                /* previous stacks at edge */
1526                final LinkedList<Operand[]> prevStacks;
1527                /* current stack at edge */
1528                ArrayList<Operand> stack;

1530                Edge(AbstractInsnNode insn, ArrayList<Operand> stack) {
1531                        this.insn = insn;
1532                        this.prevStacks = new LinkedList<Operand[]>();
1533                        this.stack = stack;
1534                }

1536                Edge(AbstractInsnNode insn) {
1537                        this(insn, new ArrayList<Operand>(AsmMethodSource.this.stack));

1544        private void addEdges(AbstractInsnNode cur,
1545                        AbstractInsnNode tgt1, List<LabelNode> tgts) {
1546                int lastIdx = tgts == null ? -1 : tgts.size() - 1;
1547                Operand[] stackss = (new ArrayList<Operand>(stack)).toArray(new Operand[stack.size()]);
1548                AbstractInsnNode tgt = tgt1;
1549                int i = 0;
1550                tgt_loop:
1551                do {
1552                        Edge edge = edges.get(cur, tgt);
1553                        if (edge == null) {
1554                                edge = new Edge(tgt);
1555                                edge.prevStacks.add(stackss);
1556                                edges.put(cur, tgt, edge);
1557                                conversionWorklist.add(edge);
1558                                continue;
1559                        }
1560                        if (edge.stack != null) {
1561                                ArrayList<Operand> stackTemp = edge.stack;
1562                                if (stackTemp.size() != stackss.length){
1563                                        throw new AssertionError("Multiple un-equal stacks!");
1564                                }
1565                                for (int j = 0; j != stackss.length; j++) {
1566                                        if (!stackTemp.get(j).equivTo(stackss[j]))
1567                                                throw new AssertionError("Multiple un-equal stacks!");
1568                                }
1569                                continue;
1570                        }
1571                        for (Operand[] ps : edge.prevStacks) {
1572                                if (Arrays.equals(ps, stackss))
1573                                        continue tgt_loop;
1574                        }
1575                        edge.stack = new ArrayList<Operand>(stack);
1576                        edge.prevStacks.add(stackss);
1577                        conversionWorklist.add(edge);
1578                } while (i <= lastIdx && (tgt = tgts.get(i++)) != null);
1579        }

1581        private void convert() {
1582                ArrayDeque<Edge> worklist = new ArrayDeque<Edge>();
1583                for (LabelNode ln : trapHandlers.keySet()) {
1584                        if (checkInlineExceptionHandler(ln))
1585                                handleInlineExceptionHandler(ln, worklist);
1586                        else
1587                                worklist.add(new Edge(ln, new ArrayList<Operand>()));
1588                }
1589                worklist.add(new Edge(instructions.getFirst(), new ArrayList<Operand>()));
1590                conversionWorklist = worklist;
1591                edges = HashBasedTable.create(1,1);
1592                
1593                do {
1594                        Edge edge = worklist.pollLast();
1595                        AbstractInsnNode insn = edge.insn;
1596                        stack = edge.stack;
1597                        edge.stack = null;
1598                        do {
1599                                int type = insn.getType();
1600                                if (type == FIELD_INSN) {
1601                                        convertFieldInsn((FieldInsnNode) insn);
1602                                } else if (type == IINC_INSN) {
1603                                        convertIincInsn((IincInsnNode) insn);
1604                                } else if (type == INSN) {
1605                                        convertInsn((InsnNode) insn);
1606                                        int op = insn.getOpcode();
1607                                        if ((op >= IRETURN && op <= RETURN) ||
1608                                                        op == ATHROW) {
1609                                                break;
1610                                        }
1611                                } else if (type == INT_INSN) {
1612                                        convertIntInsn((IntInsnNode) insn);
1613                                } else if (type == LDC_INSN) {
1614                                        convertLdcInsn((LdcInsnNode) insn);
1615                                } else if (type == JUMP_INSN) {
1616                                        JumpInsnNode jmp = (JumpInsnNode) insn;
1617                                        convertJumpInsn(jmp);
1618                                        int op = jmp.getOpcode();
1619                                        if (op == JSR)
1620                                                throw new UnsupportedOperationException("JSR!");
1621                                        if (op != GOTO) {
1622                                                /* ifX opcode, i.e. two successors */
1623                                                AbstractInsnNode next = insn.getNext();
1624                                                addEdges(insn, next, Collections.singletonList(jmp.label));
1625                                        } else {
1626                                                addEdges(insn, jmp.label, null);
1627                                        }
1628                                        break;
1629                                } else if (type == LOOKUPSWITCH_INSN) {
1630                                        LookupSwitchInsnNode swtch = (LookupSwitchInsnNode) insn;
1631                                        convertLookupSwitchInsn(swtch);
1632                                        LabelNode dflt = swtch.dflt;
1633                                        addEdges(insn, dflt, swtch.labels);
1634                                        break;
1635                                } else if (type == METHOD_INSN) {
1636                                        convertMethodInsn((MethodInsnNode) insn);
1637                                } else if (type == INVOKE_DYNAMIC_INSN) {
1638                                        convertInvokeDynamicInsn((InvokeDynamicInsnNode) insn);
1639                                } else if (type == MULTIANEWARRAY_INSN) {
1640                                        convertMultiANewArrayInsn((MultiANewArrayInsnNode) insn);
1641                                } else if (type == TABLESWITCH_INSN) {
1642                                        TableSwitchInsnNode swtch = (TableSwitchInsnNode) insn;
1643                                        convertTableSwitchInsn(swtch);
1644                                        LabelNode dflt = swtch.dflt;
1645                                        addEdges(insn, dflt, swtch.labels);
1646                                } else if (type == TYPE_INSN) {
1647                                        convertTypeInsn((TypeInsnNode) insn);
1648                                } else if (type == VAR_INSN) {
1649                                        if (insn.getOpcode() == RET)
1650                                                throw new UnsupportedOperationException("RET!");
1651                                        convertVarInsn((VarInsnNode) insn);
1652                                } else if (type == LABEL) {
1653                                        convertLabel((LabelNode) insn);
1654                                } else if (type == LINE) {
1655                                        convertLine((LineNumberNode) insn);
1656                                } else if (type == FRAME) {
1657                                                // we can ignore it
1658                                }
1659                                else
1660                                        throw new RuntimeException("Unknown instruction type: " + type);
1661                        } while ((insn = insn.getNext()) != null);
1662                } while (!worklist.isEmpty());
1663                conversionWorklist = null;
1664                edges = null;
1665        }

1667        private void handleInlineExceptionHandler(LabelNode ln, ArrayDeque<Edge> worklist) {
1668                // Catch the exception
1669                CaughtExceptionRef ref = Jimple.v().newCaughtExceptionRef();
1670                Local local = newStackLocal();
1671                DefinitionStmt as = Jimple.v().newIdentityStmt(local, ref);
1672

1674                opr.stack = local;
1675

1677                stack.add(opr);
1678                worklist.add(new Edge(ln, stack));
1679                
1680                // Save the statements
1681                inlineExceptionHandlers.put(ln, as);
1682        }

1684        private boolean checkInlineExceptionHandler(LabelNode ln) {
1685                // If this label is reachable through an exception and through normal
1686                // code, we have to split the exceptional case (with the exception on
1687                // the stack) from the normal fall-through case without anything on the
1688                // stack.
1689                for (Iterator<AbstractInsnNode> it = instructions.iterator(); it.hasNext(); ) {
1690                        AbstractInsnNode node = it.next();
1691                        if (node instanceof JumpInsnNode) {
1692                                if (((JumpInsnNode) node).label == ln) {
1693                                        inlineExceptionLabels.add(ln);
1694                                        return true;
1695                                }
1696                        }
1697                        else if (node instanceof LookupSwitchInsnNode) {
1698                                if (((LookupSwitchInsnNode) node).labels.contains(ln)) {
1699                                        inlineExceptionLabels.add(ln);
1700                                        return true;
1701                                }
1702                        }
1703                        else if (node instanceof TableSwitchInsnNode) {
1704                                if (((TableSwitchInsnNode) node).labels.contains(ln)) {
1705                                        inlineExceptionLabels.add(ln);
1706                                        return true;
1707                                }
1708                        }
1709                }
1710                return false;
1711        }

1771        private void emitUnits() {
1772                AbstractInsnNode insn = instructions.getFirst();
1773                ArrayDeque<LabelNode> labls = new ArrayDeque<LabelNode>();
1774                
1775                while (insn != null) {
1776                        // Save the label to assign it to the next real unit
1777                        if (insn instanceof LabelNode)
1778                                labls.add((LabelNode) insn);
1779                        
1780                        // Get the unit associated with the current instruction
1781                        Unit u = units.get(insn);
1782                        if (u == null) {
1783                                insn = insn.getNext();
1784                                continue;
1785                        }
1786                        
1787                        emitUnits(u);
1788                        
1789                        // If this is an exception handler, register the starting unit for it
1790                        {
1791                                IdentityStmt caughtEx = null;
1792                                if (u instanceof IdentityStmt)
1793                                        caughtEx = (IdentityStmt) u;
1794                                else if (u instanceof UnitContainer)
1795                                        caughtEx = getIdentityRefFromContrainer((UnitContainer) u);
1796                                
1797                                if (insn instanceof LabelNode
1798                                                && caughtEx != null
1799                                                && caughtEx.getRightOp() instanceof CaughtExceptionRef) {
1800                                        // We directly place this label
1801                                        Collection<UnitBox> traps = trapHandlers.get((LabelNode) insn);
1802                                        for (UnitBox ub : traps)
1803                                                ub.setUnit(caughtEx);
1804                                }
1805                        }
1806                        
1807                        // Register this unit for all targets of the labels ending up at it
1808                        while (!labls.isEmpty()) {
1809                                LabelNode ln = labls.poll();
1810                                Collection<UnitBox> boxes = labels.get(ln);
1811                                if (boxes != null) {
1812                                        for (UnitBox box : boxes) {
1813                                                box.setUnit(u instanceof UnitContainer ? ((UnitContainer) u).getFirstUnit() : u);
1814                                        }
1815                                }
1816                        }
1817                        insn = insn.getNext();
1818                }
1819                
1820                // Emit the inline exception handlers
1821                for (LabelNode ln : this.inlineExceptionHandlers.keySet()) {
1822                        Unit handler = this.inlineExceptionHandlers.get(ln);
1823                        emitUnits(handler);
1824                        
1825                        Collection<UnitBox> traps = trapHandlers.get(ln);
1826                        for (UnitBox ub : traps)
1827                                ub.setUnit(handler);
1828                        
1829                        // We need to jump to the original implementation
1830                        Unit targetUnit = units.get(ln);
1831                        GotoStmt gotoImpl = Jimple.v().newGotoStmt(targetUnit);
1832                        body.getUnits().add(gotoImpl);
1833                }
1834                
1835                /* set remaining labels & boxes to last unit of chain */
1836                if (labls.isEmpty())
1837                        return;
1838                Unit end = Jimple.v().newNopStmt();
1839                body.getUnits().add(end);
1840                while (!labls.isEmpty()) {
1841                        LabelNode ln = labls.poll();
1842                        Collection<UnitBox> boxes = labels.get(ln);
1843                        if (boxes != null) {
1844                                for (UnitBox box : boxes)
1845                                        box.setUnit(end);
1846                        }
1847                }
1848        }

1850        private IdentityStmt getIdentityRefFromContrainer(UnitContainer u) {
1851                for (Unit uu : ((UnitContainer) u).units) {
1852                        if (uu instanceof IdentityStmt) {
1853                                return (IdentityStmt) uu;
1854                        }
1855                        else if (uu instanceof UnitContainer)
1856                                return getIdentityRefFromContrainer((UnitContainer) uu);
1857                }
1858                return null;
1859        }

1861        public Body getBody(SootMethod m, String phaseName) {
1862                if (!m.isConcrete())
1863                        return null;
1864                JimpleBody jb = Jimple.v().newBody(m);
1865                /* initialize */
1866                int nrInsn = instructions.size();
1867                nextLocal = maxLocals;
1868                locals = new HashMap<Integer, Local>(maxLocals + (maxLocals / 2));
1869                labels = ArrayListMultimap.create(4, 1);
1870                units = new HashMap<AbstractInsnNode, Unit>(nrInsn);
1871                frames = new HashMap<AbstractInsnNode, StackFrame>(nrInsn);
1872                trapHandlers = ArrayListMultimap.create(tryCatchBlocks.size(),1);
1873                body = jb;
1874                /* retrieve all trap handlers */
1875                for (TryCatchBlockNode tc : tryCatchBlocks)
1876                        trapHandlers.put(tc.handler, Jimple.v().newStmtBox(null));
1877                /* convert instructions */
1878                try {
1879                        convert();
1880                } catch (Throwable t) {
1881                        throw new RuntimeException("Failed to convert " + m, t);
1882                }
1883                
1884                /* build body (add units, locals, traps, etc.) */
1885                emitLocals();
1886                emitTraps();
1887                emitUnits();
1888                
1889                /* clean up */
1890                locals = null;
1891                labels = null;
1892                units = null;
1893                stack = null;
1894                frames = null;
1895                body = null;
1896                
1897                // Make sure to inline patterns of the form to enable proper variable
1898                // splitting and type assignment:
1899                // a = new A();
1900                // goto l0;
1901                // l0:
1902                //      b = (B) a;
1903                //      return b;
1904                castAndReturnInliner.transform(jb);
1905                
1906                try {
1907                PackManager.v().getPack("jb").apply(jb);
1908                } catch (Throwable t) {
