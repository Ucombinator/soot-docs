019package soot.toolkits.graph.pdg;
020
021
022import java.util.ArrayList;
023import java.util.Iterator;
024import java.util.LinkedList;
025import java.util.List;
026
027import soot.SootClass;
028import soot.SootMethod;
029import soot.Unit;
030import soot.toolkits.graph.Block;
031import soot.toolkits.graph.UnitGraph;
032
033
034/**
035 * This class was originally designed to represent a weak region. Later,
036 * PDGRegion was designed to represent a richer region representation but 
037 * since there were cases that we wanted to enforce the use of Region instead
038 * of PDGRegion, and for some other compatibility issues, we chose not to eliminate
039 * this class and even not to factor it into a common abstract class. 
040 * 
041 * One major contributing factor to the above is that, a Region can exist 
042 * independent from a PDG since they are generated by RegionAnalysis, whereas
043 * a PDGRegion is generated by a PDG.
044 * 
045 * @author Hossein Sadat-Mohtasham
046 * Jan 2009
047 */
048
049public class Region implements IRegion{
050        
051        private SootClass m_class = null;;
052        private SootMethod m_method = null;
053        private List<Block> m_blocks = null;
054        private List<Unit> m_units = null;
055        private int m_id = -1;
056        private UnitGraph m_unitGraph = null;
057        
058        //The following are needed to create a tree of regions based on the containment (dependency)
059        //relation between regions.
060        private IRegion m_parent = null;
061        private List<IRegion> m_children = new ArrayList<IRegion>();
062        
063        
064        public Region(int id, SootMethod m, SootClass c, UnitGraph ug)
065        {
066                this(id, new ArrayList<Block>(), m, c, ug);
067                
068        }
069        
070        public Region(int id, List<Block> blocks, SootMethod m, SootClass c, UnitGraph ug)
071        {
072                
073                this.m_blocks = blocks;
074                this.m_id = id;
075                this.m_method = m;
076                this.m_class = c;
077                this.m_unitGraph = ug;
078                this.m_units = null;
079                
080                
081        }
082                
083        @SuppressWarnings("unchecked")
084        public Object clone()
085        {
086                Region r = new Region(this.m_id, this.m_method, this.m_class, this.m_unitGraph);
087                r.m_blocks = (List<Block>)((ArrayList<Block>)this.m_blocks).clone();
088                
089                return r;
090                
091        }
092        public SootMethod getSootMethod()
093        {
094                return this.m_method;
095        }
096        
097        public SootClass getSootClass()
098        {
099                return this.m_class;
100        }
101        
102        public List<Block> getBlocks()
103        {
104                return this.m_blocks;
105        }
106        
107        
108        public UnitGraph getUnitGraph()
109        {
110                return this.m_unitGraph;
111        }
112        public List<Unit> getUnits()
113        {
114                if(this.m_units == null)
115                {
116                        this.m_units = new LinkedList<Unit>();
117                        for(Iterator<Block> itr = this.m_blocks.iterator(); itr.hasNext();)
118                        {
119                                Block b = itr.next();
120                                for(Iterator<Unit> itr1 = b.iterator(); itr1.hasNext();)
121                                {
122                                        Unit u = itr1.next();
123                                        ((LinkedList<Unit>)this.m_units).addLast(u);
124
125                                }
126                        }
127                        
128                }
129                
130                return this.m_units;
131        }
132        
133        public List<Unit> getUnits(Unit from, Unit to)
134        {
135                
136                return m_units.subList(m_units.indexOf(from), m_units.indexOf(to));
137                
138        }
139        
140        public Unit getLast()
141        {
142                if(this.m_units != null)
143                        if(this.m_units.size() > 0)
144                                return ((LinkedList<Unit>)this.m_units).getLast();
145                        
146                                
147                return null;
148        }
149        
150        public Unit getFirst()
151        {
152                if(this.m_units != null)
153                        if(this.m_units.size() > 0)
154                                return ((LinkedList<Unit>)this.m_units).getFirst();
155                
156                
157                return null;
158        }
159
160        
161        
162        public void add(Block b)
163        {
164                //Add to the front
165                this.m_blocks.add(0, b);
166                
167        }
168        
169        public void add2Back(Block b)
170        {
171                //Add to the last
172                this.m_blocks.add(this.m_blocks.size(), b);
173                
174        }
175        
176        public void remove(Block b)
177        {
178                this.m_blocks.remove(b);
179                //make the units be recalculated.
180                this.m_units = null;
181        }
182        
183        public int getID()
184        {
185                return this.m_id;
186        }
187        
188        public boolean occursBefore(Unit u1, Unit u2)
189        {
190                int i = this.m_units.lastIndexOf(u1);
191                int j = this.m_units.lastIndexOf(u2);
192                
193                if(i == -1 || j == -1)
194                        throw new RuntimeException("These units don't exist in the region!");
195                
196                return i < j;
197        }
198        
199        
200        public void setParent(IRegion pr)
201        {
202                this.m_parent = pr;
203        }
204        public IRegion getParent()
205        {
206                return this.m_parent;
207        }
208        
209        public void addChildRegion(IRegion chr)
210        {
211                if(!this.m_children.contains(chr))
212                        this.m_children.add(chr);
213        }
214        
215        public List<IRegion> getChildRegions()
216        {
217                return this.m_children;
218        }
219        public String toString()
220        {
221                String str = new String();
222                str += "Begin-----------Region:  " + this.m_id + "-------------\n";
223                
224                List<Unit> regionUnits = this.getUnits();         
225                for (Iterator<Unit> itr = regionUnits.iterator(); itr.hasNext();) 
226                {
227                        Unit u = itr.next();
228                        str += u + "\n";
229                        
230                }
231                str += "End Region " + this.m_id + " -----------------------------\n";
232                
233                return str;
234                
235        }
236        
237        
238}
