043/**
044 *
045 * This class implements a Program Dependence Graph as defined in 
046 * 
047 * Ferrante, J., Ottenstein, K. J., and Warren, J. D. 1987. 
048 * The program dependence graph and its use in optimization. 
049 * ACM Trans. Program. Lang. Syst. 9, 3 (Jul. 1987), 319-349. 
050 * DOI= http://doi.acm.org/10.1145/24039.24041
051 * 
052 * Note: the implementation is not exactly as in the above paper. It first finds
053 * the regions of control dependence then uses part of the algorithm given in
054 * the above paper to build the graph.
055 * 
056 * The constructor accepts a UnitGraph, which can be a BriefUnitGraph, an ExceptionalUnitGraph,
057 * or an EnhancedUnitGraph. At the absence of exception handling constructs in
058 * a method, all of these work the same. However, at the presence of exception handling
059 * constructs, BriefUnitGraph is multi-headed and potentially multi-tailed which makes
060 * the results of RegionAnalysis and PDG construction unreliable (It's not clear if
061 * it would be useful anyway); Also, ExceptionalGraph's usefulness when exception handling
062 * is present is not so clear since almost every unit can throw exception hence the 
063 * dependency is affected. Currently, the PDG is based on a UnitGraph (BlockGraph) and does
064 * not care whether flow is exceptional or not.
065 * 
066 * The nodes in a PDG are of type PDGNode and the edges can have three labels: "dependency",
067 * "dependency-back", and "controlflow"; however, the "controlflow" edges are auxiliary
068 * and the dependencies are represented by the labels beginning with "dependency". 
069 * Other labels can be added later for application
070 * or domain-specific cases.
071 * 
072 * 
073 * To support methods that contain exception-handling and multiple-heads or tails, use EnhancedUnitGraph.
074 * It does not represent exceptional flow in the way ExceptionalUnitGraph does, but it integrates 
075 * them in a concise way. Also, it adds START/STOP nodes to graph if necessary to make the graph
076 * single entry single exit.
077 * 
078 *
079 * @author Hossein Sadat-Mohtasham
080 * Sep 2009
081 */
082
083public class HashMutablePDG extends HashMutableEdgeLabelledDirectedGraph<PDGNode, String> implements ProgramDependenceGraph {
084        
085        protected Body m_body = null;
086        protected SootClass m_class = null;
087        protected UnitGraph m_cfg = null;
088        protected BlockGraph m_blockCFG = null;
089        protected Hashtable<Object, PDGNode> m_obj2pdgNode = new Hashtable<Object, PDGNode>();
090        protected List<Region> m_weakRegions = null;
091        protected List<Region> m_strongRegions = null;
092        protected PDGNode m_startNode = null;
093        protected List<PDGRegion> m_pdgRegions = null;
094        private RegionAnalysis m_regionAnalysis = null;
095        private int m_strongRegionStartID;
096        
097        public HashMutablePDG(UnitGraph cfg)
098        {
099                this.m_body = cfg.getBody();
100                this.m_class = this.m_body.getMethod().getDeclaringClass();
101                this.m_cfg = cfg;
102                                
103                this.m_regionAnalysis = new RegionAnalysis(this.m_cfg, this.m_body.getMethod(), this.m_class);
104                
105                /*
106                 * Get the weak regions and save a copy. Note that the strong regions list is
107                 * initially cloned from the weak region to be later modified.
108                 */
109                this.m_strongRegions = this.m_regionAnalysis.getRegions();
110                this.m_weakRegions = this.cloneRegions(this.m_strongRegions);
111                this.m_blockCFG = this.m_regionAnalysis.getBlockCFG();
112                
113                //Construct the PDG
114                this.constructPDG();
115                this.m_pdgRegions = HashMutablePDG.computePDGRegions(this.m_startNode);
116
117                
118                /*This is needed to convert the initially Region-typed inner node of the PDG's head
119                        to a PDGRegion-typed one after the whole graph is computed.
120                        The root PDGRegion is the one with no parent.
121                */
122
123                IRegion r = this.m_pdgRegions.get(0);
124                while(r.getParent() != null)
125                        r = r.getParent();
126                
127                this.m_startNode.setNode(r);
128                
129        }

131    public BlockGraph getBlockGraph(){
132      return m_blockCFG;

135        /**
136         * This is the heart of the PDG contruction. It is huge and definitely needs 
137         * some refactorings, but since it's been evlovong to cover some boundary cases
138         * it has become hard to rafactor.
139         * 
140         * It uses the list of weak regions, along with the dominator and post-dominator
141         * trees to construct the PDG nodes.
142         */
143        protected void constructPDG()
144        {
145                Hashtable<Block, Region> block2region = this.m_regionAnalysis.getBlock2RegionMap();
146                DominatorTree<Block> pdom = this.m_regionAnalysis.getPostDominatorTree();
147                DominatorTree<Block> dom = this.m_regionAnalysis.getDominatorTree();
148                                
149                List<Region> regions2process = new LinkedList<Region>();
150                Region topLevelRegion = this.m_regionAnalysis.getTopLevelRegion();
151                m_strongRegionStartID = m_weakRegions.size();
152                
153                //This becomes the top-level region (or ENTRY region node)
154                PDGNode pdgnode = new PDGNode(topLevelRegion, PDGNode.Type.REGION);
155                this.addNode(pdgnode);
156                this.m_obj2pdgNode.put(topLevelRegion, pdgnode);
157                this.m_startNode = pdgnode;
158                topLevelRegion.setParent(null);
159                
160        Set<Region> processedRegions = new HashSet<Region>();
161                regions2process.add(topLevelRegion);
162                
163                //while there's a (weak) region to process
164                while(!regions2process.isEmpty())
165                {
166                        Region r = regions2process.remove(0);
167            processedRegions.add(r);
168                        
169                        //get the corresponding pdgnode
170                        pdgnode = this.m_obj2pdgNode.get(r);
171                        
172                        /*
173                         * For all the CFG nodes in the region, create the corresponding PDG node and edges, and
174                         * process them if they are in the dependence set of other regions, i.e. other regions
175                         * depend on them.
176                         */
177                        List<Block> blocks = r.getBlocks();
178                        Hashtable<Region, List<Block>> toBeRemoved = new Hashtable<Region, List<Block>>();
179                        PDGNode prevPDGNodeInRegion = null;
180                        PDGNode curNodeInRegion = null;
181                        for(Iterator<Block> itr = blocks.iterator(); itr.hasNext();)
182                        {
183                                /*
184                                 * Add the PDG node corresponding to the CFG block node.
185                                 */
186                                Block a = itr.next();
187                                PDGNode pdgNodeOfA = null;
188                                if(!this.m_obj2pdgNode.containsKey(a))
189                                {
190                                        pdgNodeOfA = new PDGNode(a, PDGNode.Type.CFGNODE);
191                                        this.addNode(pdgNodeOfA);
192                                        this.m_obj2pdgNode.put(a, pdgNodeOfA);
193                                }
194                                else
195                                        pdgNodeOfA = this.m_obj2pdgNode.get(a);
196                                                                
197                                this.addEdge(pdgnode, pdgNodeOfA, "dependency");
198                                pdgnode.addDependent(pdgNodeOfA);
199                                //
200                                
201                                curNodeInRegion = pdgNodeOfA;
202                                
203                                /*
204                                 * For each successor B of A, if B does not post-dominate A, add all the
205                                 * nodes on the path from B to the L in the post-dominator tree, where L is the least
206                                 * common ancestor of A and B in the post-dominator tree (L will be either A itself or 
207                                 * the parent of A.).
208                                 */
209                                
210                                List<Block> bs = this.m_blockCFG.getSuccsOf(a);
211                                for(Iterator<Block> bItr = bs.iterator(); bItr.hasNext();)
212                                {
213                                        List<Block> dependents = new ArrayList<Block>();
214                                        
215                                        Block b = bItr.next();
216                                        
217                                        if(b.equals(a))
218                                                throw new RuntimeException("PDG construction: A and B are not supposed to be the same node!");
219                                        
220                                        
221                                        DominatorNode<Block> aDode = pdom.getDode(a);
222                                        DominatorNode<Block> bDode = pdom.getDode(b);
223                                        
224                                        //If B post-dominates A, go to the next successor.
225                                        if(pdom.isDominatorOf(bDode, aDode))
226                                                continue;
227                                        
228                                        //FIXME: what if the parent is null?!!
229                                        DominatorNode<Block> aParentDode = aDode.getParent();
230                                        
231                                        DominatorNode<Block> dode = bDode;
232                                        while(dode != aParentDode)
233                                        {
234                                                dependents.add((Block)dode.getGode());
235                                                
236                                                //This occurs if the CFG is multi-tailed and therefore the pdom is a forest.
237                                                if(dode.getParent() == null)
238                                                        //throw new RuntimeException("parent dode in pdom is null: dode is " + aDode);
239                                                        break;
240                                                dode = dode.getParent();                                                
241                                        }
242                                        
243                                        /*
244                                         * If node A is in the dependent list of A, then A is the header of a loop.
245                                         * Otherwise, A could still be the header of a loop or just a simple predicate.
246                                         */
247                                        
248                                        //first make A's pdg node be a conditional (predicate) pdgnode, if it's not already.                                    
249                                        if(pdgNodeOfA.getAttrib() != PDGNode.Attribute.CONDHEADER)
250                                        {
251                                                PDGNode oldA = pdgNodeOfA;
252                                                pdgNodeOfA = new ConditionalPDGNode(pdgNodeOfA);
253                                                this.replaceInGraph(pdgNodeOfA, oldA);
254                                                pdgnode.removeDependent(oldA);
255                                                this.m_obj2pdgNode.put(a, pdgNodeOfA);
256                                                pdgnode.addDependent(pdgNodeOfA);
257                                                pdgNodeOfA.setAttrib(PDGNode.Attribute.CONDHEADER);
258                                                
259                                                curNodeInRegion = pdgNodeOfA;
260                                        }               
261                                        
262                                        List<Block> copyOfDependents = new ArrayList<Block>();
263                                        copyOfDependents.addAll(dependents);
264                                        
265                                        //First, add the dependency for B and its corresponding region.
266                                        Region regionOfB = block2region.get(b);
267                                        PDGNode pdgnodeOfBRegion = null;
268                                        if(!this.m_obj2pdgNode.containsKey(regionOfB))
269                                        {
270                                                pdgnodeOfBRegion = new PDGNode(regionOfB, PDGNode.Type.REGION);
271                                                this.addNode(pdgnodeOfBRegion);
272                                                this.m_obj2pdgNode.put(regionOfB, pdgnodeOfBRegion);
273                                                
274                                        }
275                                        else
276                                                pdgnodeOfBRegion = this.m_obj2pdgNode.get(regionOfB);
277                                        
278                                        //set the region hierarchy
279                                        regionOfB.setParent(r);
280                                        r.addChildRegion(regionOfB);
281                                        
282                                        //add the dependency edges
283                                        this.addEdge(pdgNodeOfA, pdgnodeOfBRegion, "dependency");
284                                        pdgNodeOfA.addDependent(pdgnodeOfBRegion);
285                    if(!processedRegions.contains(regionOfB)){
286                                          regions2process.add(regionOfB);
287                    }
288                                        //now remove b and all the nodes in the same weak region from the list of dependents
289                                        copyOfDependents.remove(b);
290                                        copyOfDependents.removeAll(regionOfB.getBlocks());
291                                
292                                        
293                                        /* What remains here in the dependence set needs to be processed separately. For
294                                         * each node X remained in the dependency set, find the corresponding PDG region node
295                                         * and add a dependency edge from the region of B to the region of X. If X's weak 
296                                         * region contains other nodes not in the dependency set of A, create a new region
297                                         * for X and add the proper dependency edges (this actually happens if X is the header
298                                         * of a loop and B is a predicate guarding a break/continue.)
299                                         * 
300                                         * Note: it seems the only case that there is a node remained in the dependents is when there 
301                                         * is a path from b to the header of a loop. 
302                                         */
303                                        
304                                        while(!copyOfDependents.isEmpty())
305                                        {
306                                                Block depB = copyOfDependents.remove(0);
307                                                Region rdepB = block2region.get(depB);
308                                                
309                                                /* 
310                                                 * Actually, there are cases when depB is not the header of a loop and therefore would not dominate the current node
311                                                 * (A) and therefore might not have been created yet. This has happened when an inner loop breaks out of the outer 
312                                                 * loop but could have other cases too.
313                                                 */
314                                                PDGNode depBPDGNode = this.m_obj2pdgNode.get(depB);
315                                                if(depBPDGNode == null)
316                                                {                                               
317                                                        //First, add the dependency for depB and its corresponding region.
318                                                        
319                                                        PDGNode pdgnodeOfdepBRegion = null;
320                                                        if(!this.m_obj2pdgNode.containsKey(rdepB))
321                                                        {
322                                                                pdgnodeOfdepBRegion = new PDGNode(rdepB, PDGNode.Type.REGION);
323                                                                this.addNode(pdgnodeOfdepBRegion);
324                                                                this.m_obj2pdgNode.put(rdepB, pdgnodeOfdepBRegion);
325                                                                
326                                                        }
327                                                        else
328                                                                pdgnodeOfdepBRegion = this.m_obj2pdgNode.get(rdepB);
329                                                        
330                                                        //set the region hierarchy
331                                                        rdepB.setParent(regionOfB);
332                                                        regionOfB.addChildRegion(rdepB);
333                                                        
334                                                        //add the dependency edges
335                                                        this.addEdge(pdgnodeOfBRegion, pdgnodeOfdepBRegion, "dependency");
336                                                        pdgnodeOfBRegion.addDependent(pdgnodeOfdepBRegion);
337                            if(!processedRegions.contains(rdepB)){
338                                                          regions2process.add(rdepB);
339                            }
340                                                        
341                                                        //now remove all the nodes in the same weak region from the list of dependents
342                                                        
343                                                        copyOfDependents.removeAll(rdepB.getBlocks());
344                                                        
345                                                        continue;
346
347                                                }
348                                                
349                                                /**
350                                                 * If all the nodes in the weak region of depB are dependent on A, then add 
351                                                 * an edge from the region of B to the region of depB.
352                                                 * 
353                                                 * else, a new region has to be created to contain the dependences of depB, if
354                                                 * not already created.
355                                                 */
356                                                if(dependents.containsAll(rdepB.getBlocks()))
357                                                {
358                                                        /*
359                                                         * Just add an edge to the pdg node of the existing depB region.
360                                                         * 
361                                                         */
362                                                        //add the dependency edges
363                                                        //First, add the dependency for depB and its corresponding region.
364                                                        
365                                                        PDGNode pdgnodeOfdepBRegion = null;
366                                                        if(!this.m_obj2pdgNode.containsKey(rdepB))
367                                                        {
368                                                                pdgnodeOfdepBRegion = new PDGNode(rdepB, PDGNode.Type.REGION);
369                                                                this.addNode(pdgnodeOfdepBRegion);
370                                                                this.m_obj2pdgNode.put(rdepB, pdgnodeOfdepBRegion);
371                                                                
372                                                        }
373                                                        else
374                                                                pdgnodeOfdepBRegion = this.m_obj2pdgNode.get(rdepB);
375                                                        
376                                                        //set the region hierarchy
377                                                        //Do not set this because the region was created before so must have the
378                                                        //proper parent already.
379                                                        //rdepB.setParent(regionOfB);
380                                                        //regionOfB.addChildRegion(rdepB);
381                                                        
382                                                        this.addEdge(pdgnodeOfBRegion, pdgnodeOfdepBRegion, "dependency");
383                                                        pdgnodeOfBRegion.addDependent(pdgnodeOfdepBRegion);
384                            if(!processedRegions.contains(rdepB)){
385                                                          regions2process.add(rdepB);
386                            }
387                                                        
388                                                        //now remove all the nodes in the same weak region from the list of dependents
389                                                        
390                                                        copyOfDependents.removeAll(rdepB.getBlocks());
391        
392                                                        continue;
393                                                }
394                                                else
395                                                {
396                                                        PDGNode predPDGofdepB = (PDGNode) this.getPredsOf(depBPDGNode).get(0);
397                                                        assert(this.m_obj2pdgNode.containsKey(rdepB));                                                  
398                                                        PDGNode pdgnodeOfdepBRegion = this.m_obj2pdgNode.get(rdepB);
399                                                        //If the loop header has not been separated from its weak region yet
400                                                        if(predPDGofdepB == pdgnodeOfdepBRegion)
401                                                        {
402                                                                /*
403                                                                 * Create a new region to represent the whole loop. In fact, this is a strong
404                                                                 * region as opposed to the weak regions that were created in the RegionAnalysis.
405                                                                 * This strong region only contains the header of the loop, A, and is dependent 
406                                                                 * on it. Also, A is dependent on this strong region as well.
407                                                                 */
408                                                                
409                                                                Region newRegion = new Region(this.m_strongRegionStartID++, topLevelRegion.getSootMethod(), topLevelRegion.getSootClass(), this.m_cfg);
410                                                                newRegion.add(depB);
411                                                                
412                                                                this.m_strongRegions.add(newRegion);
413                                                                                                                                
414                                                                //toBeRemoved.add(depB);
415                                                                List<Block> blocks2BRemoved;
416                                                                if(toBeRemoved.contains(predPDGofdepB))
417                                                                        blocks2BRemoved = toBeRemoved.get(predPDGofdepB);
418                                                                else
419                                                                {
420                                                                        blocks2BRemoved = new ArrayList<Block>();
421                                                                        toBeRemoved.put(rdepB, blocks2BRemoved);
422                                                                }
423                                                                blocks2BRemoved.add(depB);
424                                                                
425                                                                
426                                                                
427                                                                PDGNode newpdgnode = new LoopedPDGNode(newRegion, PDGNode.Type.REGION, depBPDGNode);
428                                                                this.addNode(newpdgnode);
429                                                                this.m_obj2pdgNode.put(newRegion, newpdgnode);
430                                                                newpdgnode.setAttrib(PDGNode.Attribute.LOOPHEADER);
431                                                                depBPDGNode.setAttrib(PDGNode.Attribute.LOOPHEADER);
432                                                                
433                                                                this.removeEdge(pdgnodeOfdepBRegion, depBPDGNode, "dependency");
434                                                                pdgnodeOfdepBRegion.removeDependent(depBPDGNode);
435                                                                this.addEdge(pdgnodeOfdepBRegion, newpdgnode, "dependency");
436                                                                this.addEdge(newpdgnode, depBPDGNode, "dependency");
437                                                                pdgnodeOfdepBRegion.addDependent(newpdgnode);
438                                                                newpdgnode.addDependent(depBPDGNode);
439                                                                
440                                                                
441                                                                
442                                                                //If a is dependent on itself (simple loop)
443                                                                if(depB == a)
444                                                                {
445                                                                        PDGNode loopBodyPDGNode = (PDGNode) this.getSuccsOf(depBPDGNode).get(0);
446                                                                        this.addEdge(depBPDGNode, newpdgnode, "dependency-back");
447                                                                        ((LoopedPDGNode)newpdgnode).setBody(loopBodyPDGNode);
448                                                                        
449                                                                        depBPDGNode.addBackDependent(newpdgnode);
450                                                                        
451                                                                        //This is needed to correctly adjust the prev/next pointers for the new loop node. We should not need
452                                                                        //to adjust the old loop header node because the prev/next should not have been set previously for it.
453                                                                        curNodeInRegion = newpdgnode;
454                                                                        
455                                                                }
456                                                                else
457                                                                {
458                                                                        
459                                                                        //this is a back-dependency
460                                                                        pdgnodeOfBRegion.addBackDependent(newpdgnode);
461                                                                        this.addEdge(pdgnodeOfBRegion, newpdgnode, "dependency-back");
462                                                                        
463                                                                        //DEtermine which dependent of the loop header is actually the loop body region
464                                                                        PDGNode loopBodyPDGNode = null;
465                                                                        List<PDGNode> successors = this.getSuccsOf(depBPDGNode);
466                                                                        Iterator<PDGNode> succItr = successors.iterator();
467                                                                        while(succItr.hasNext())
468                                                                        {
469                                                                                PDGNode succRPDGNode = succItr.next();
470                                                                                
471                                                                                assert(succRPDGNode.getType() == PDGNode.Type.REGION);
472                                                                                Region succR = (Region) succRPDGNode.getNode();
473                                                                                Block h = succR.getBlocks().get(0);
474                                                                                
475                                                                                DominatorNode<Block> hdode = dom.getDode(h);
476                                                                                DominatorNode<Block> adode = dom.getDode(a);
477                                                                                
478                                                                                if(dom.isDominatorOf(hdode, adode))
479                                                                                {
480                                                                                        loopBodyPDGNode = succRPDGNode;
481                                                                                        break;
482                                                                                }
483                                                                                
484                                                                        }
485                                                                        assert(loopBodyPDGNode != null);
486                                                                        ((LoopedPDGNode)newpdgnode).setBody(loopBodyPDGNode);
487                                                                        
488                                                                        
489                                                                        PDGNode prev = depBPDGNode.getPrev();
490                                                                        if(prev != null)
491                                                                        {
492                                                                                this.removeEdge(prev, depBPDGNode, "controlflow");
493                                                                                this.addEdge(prev, newpdgnode, "controlflow");
494                                                                                prev.setNext(newpdgnode);
495                                                                                newpdgnode.setPrev(prev);
496                                                                                depBPDGNode.setPrev(null);
497                                                                                
498                                                                        }
499                                                                        
500                                                                        PDGNode next = depBPDGNode.getNext();
501                                                                        if(next != null)
502                                                                        {
503                                                                                this.removeEdge(depBPDGNode, next, "controlflow");
504                                                                                this.addEdge(newpdgnode, next, "controlflow");
505                                                                                newpdgnode.setNext(next);
506                                                                                next.setPrev(newpdgnode);
507                                                                                depBPDGNode.setNext(null);
508                                                                                
509                                                                        }
510                                                                }
511                                                                
512                                                        }
513                                                        else
514                                                        {
515                                                                /*
516                                                                 * The strong region for the header has already been created and its 
517                                                                 * corresponding PDGNode exist. Just add the dependency edge.
518                                                                 */
519                                                                this.addEdge(pdgnodeOfBRegion, predPDGofdepB, "dependency-back");
520                                                                //this is a back-dependency
521                                                                pdgnodeOfBRegion.addBackDependent(predPDGofdepB);
522                                                                
523                                                        }
524                                                        
525                                                }               
526                                                
527                                        }
528                
529                                }       
530                        
531                        
532
533                                /* If there is a previous node in this region, add a control flow edge to indicate the
534                                 * the correct direction of control flow in the region.
535                                 */
536                                if(prevPDGNodeInRegion != null)
537                                {
538                                        this.addEdge(prevPDGNodeInRegion, curNodeInRegion, "controlflow");
539                                        prevPDGNodeInRegion.setNext(curNodeInRegion);
540                                        curNodeInRegion.setPrev(prevPDGNodeInRegion);
541                                        
542                                }
543                                
544                                prevPDGNodeInRegion = curNodeInRegion;
545
546                        }
547                        
548                        //remove all the blocks marked to be removed from the region (to change a weak region
549                        //to a strong region.)
550                        
551                        
552                        Enumeration<Region> itr1 = toBeRemoved.keys();
553                        while(itr1.hasMoreElements())
554                        {
555                                Region region = itr1.nextElement();
556                                
557                                Iterator<Block> blockItr = toBeRemoved.get(region).iterator();
558                                while(blockItr.hasNext())
559                                        region.remove(blockItr.next());
560                                        
561                        }
562
563                }
564                
565        }

567        private List<Region> cloneRegions(List<Region> weak)
568        {
569                List<Region> strong = new ArrayList<Region>();
570                Iterator<Region> itr = weak.iterator();
571                while(itr.hasNext())
572                {
573                        Region r = itr.next();
574                        strong.add((Region)r.clone());
575                }
576                return strong;

622        public static List<IRegion> getPreorderRegionList(IRegion r)
623        {
624                List<IRegion> list = new ArrayList<IRegion>();
625
626                
627                Queue<IRegion> toProcess = new LinkedList<IRegion>();
628                toProcess.add(r);
629                while(!toProcess.isEmpty())
630                {
631                        IRegion reg = toProcess.poll();
632                        list.add(reg);
633                        for(Iterator<IRegion> itr = reg.getChildRegions().iterator(); itr.hasNext(); )
634                                toProcess.add((Region)itr.next());
635                        
636                }
637                
638                return list;

649        private static List<IRegion> postorder(IRegion r, List<IRegion> list)
650        {
651                //If there are children, push the children to the stack
652                if(!r.getChildRegions().isEmpty())
653                        for(Iterator<IRegion> itr = r.getChildRegions().iterator(); itr.hasNext(); )
654                                postorder(itr.next(), list);
655                
656                list.add(r);
657                return list;

685        private static Hashtable<PDGNode, PDGRegion> node2Region = new Hashtable<PDGNode, PDGRegion>();
686        //compute the pdg region list with in post order 
687        private static List<PDGRegion> computePDGRegions(PDGNode root)
688        {

693

699        private static PDGRegion pdgpostorder(PDGNode node, List<PDGRegion> list)
700        {
701        if(node.getVisited()){
702          return null;
703        }
704                node.setVisited(true);
705
706                PDGRegion region = null;
707                if(!node2Region.containsKey(node))
708                {
709                        region = new PDGRegion(node);
710                        node2Region.put(node, region);
711                }
712                else
713                        region = node2Region.get(node);
714
715                //If there are children, push the children to the stack
716                List<PDGNode> dependents = node.getDependents();
717                if(!dependents.isEmpty())
718                        for(Iterator<PDGNode> itr = dependents.iterator(); itr.hasNext(); )
719                        {
720                                PDGNode curNode = (PDGNode) itr.next();
721                                if(curNode.getVisited())
722                                        continue;
723                                
724                                region.addPDGNode(curNode);
725                                
726                                if(curNode instanceof LoopedPDGNode)
727                                {
728                                        PDGNode body = ((LoopedPDGNode)curNode).getBody();
729                                        PDGRegion kid = pdgpostorder(body, list);
730                    if(kid != null){
731                                          kid.setParent(region);
732                                          region.addChildRegion(kid);
733                                        
734                                          //This sets the node from the old Region into a PDGRegion
735                                          body.setNode(kid);
736                    }
737                                }
738                                else if(curNode instanceof ConditionalPDGNode)
739                                {
740                                        List<PDGNode> childs = curNode.getDependents();
741                                        Iterator<PDGNode> condItr = childs.iterator();
742                                        while(condItr.hasNext())
743                                        {
744                                                PDGNode child = (PDGNode)condItr.next();
745                                                PDGRegion kid = pdgpostorder(child, list);
746                        if(kid != null){
747                                                  kid.setParent(region);
748                                                  region.addChildRegion(kid);
749                                                  //This sets the node from the old Region into a PDGRegion
750
751                                                  child.setNode(kid);
752                                                }
753                                        }
754
755                                }
756                                        
757                        }
758                
759                list.add(region);
760                return region;
761        }

797        @SuppressWarnings("unchecked")
798        private void replaceInGraph(PDGNode newnode, PDGNode oldnode)
799        {               
800                this.addNode(newnode);
801                
802                HashMutableEdgeLabelledDirectedGraph graph = (HashMutableEdgeLabelledDirectedGraph) this.clone();
803                                
804                List<PDGNode> succs = graph.getSuccsOf(oldnode);
805                List<PDGNode> preds = graph.getPredsOf(oldnode);
806                
807                for (PDGNode succ : succs) {
808                        List<Object> labels = graph.getLabelsForEdges(oldnode, succ);
809                        for(Iterator<Object> labelItr = labels.iterator(); labelItr.hasNext(); )
810                        {
811                                Object label = labelItr.next();
812                                this.addEdge(newnode, succ, new String(label.toString()));      
813                        }       
814                }
815                        
816                for (PDGNode pred : preds) {
817                        List<Object> labels = graph.getLabelsForEdges(pred, oldnode);
818                        for(Iterator<Object> labelItr = labels.iterator(); labelItr.hasNext(); )
819                        {
820                                Object label = labelItr.next();
821                                this.addEdge(pred, newnode, new String(label.toString()));
822                        }               
823                }
824                
825                this.removeNode(oldnode);
826        
827        }

829         * The existing removeAllEdges in the parent class seems to be throwing concurrentmodification exception
830         * most of the time. Here is a version that doesn't throw that exception.
831         * 
832         */
833        public void removeAllEdges(PDGNode from, PDGNode to)
834        {
835        if (!containsAnyEdge(from, to))
836            return;
837            
838        List<String> labels = new ArrayList<String>(this.getLabelsForEdges(from, to));        
839        for(String label : labels) {        
840                this.removeEdge(from, to, label);
841            }

844        /**
845         * {@inheritDoc}
846         */
847    @Override
848        public String toString()
849        {
850                String s = new String("\nProgram Dependence Graph for Method " + this.m_body.getMethod().getName());
851                s += "\n*********CFG******** \n" + RegionAnalysis.CFGtoString(this.m_blockCFG, true);
852                s += "\n*********PDG******** \n";
853                
854                List<PDGNode> processed = new ArrayList<PDGNode>();
855                Queue<PDGNode> nodes = new LinkedList<PDGNode>();
856                nodes.offer(this.m_startNode);
857                
858                while(nodes.peek() != null)
859                {
860                        PDGNode node = (PDGNode) nodes.remove();
861                        processed.add(node);
862                        
863                        s += "\n Begin PDGNode: " + node;
864                        List<PDGNode> succs = this.getSuccsOf(node);
865                        s += "\n has " + succs.size() + " successors:\n";
866                        
867                        int i = 0;
868                        for(PDGNode succ : succs) {
869                                List<String> labels = this.getLabelsForEdges(node, succ);
870                                
871                                s += i++;
872                                s += ": Edge's label: " + labels + " \n";
873                                s += "   Target: " + succ.toShortString();
874                                s += "\n";
875                                
876                                if(labels.get(0).equals("dependency"))
877                                if(!processed.contains(succ))
878                                        nodes.offer(succ);
879                                
880                        }
881                        
882                        s += "\n End PDGNode.";
883                        
884                }
885                return s;
886        }
