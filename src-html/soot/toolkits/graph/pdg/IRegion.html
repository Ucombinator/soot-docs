019package soot.toolkits.graph.pdg;
020
021
022import java.util.List;
023import soot.SootClass;
024import soot.SootMethod;
025import soot.Unit;
026import soot.toolkits.graph.Block;
027import soot.toolkits.graph.UnitGraph;
028
029/**
030 * This interface represents a region of control dependence in the control flow graph.
031 * There are different kinds of region representations that may slightly differ in 
032 * definition or implementation; here is an interface that is expected to be supported
033 * by all these different regions.
034 * 
035 * 
036 * @author Hossein Sadat-Mohtasham
037 * Jan 2009
038 */
039
040public interface IRegion {
041        
042        public SootMethod getSootMethod();
043        public SootClass getSootClass();        
044        public UnitGraph getUnitGraph();
045        public List<Unit> getUnits();
046        public List<Unit> getUnits(Unit from, Unit to);
047        public List<Block> getBlocks();
048        public Unit getLast();
049        public Unit getFirst();
050        public int getID();
051        public boolean occursBefore(Unit u1, Unit u2);
052        public void setParent(IRegion pr);
053        public IRegion getParent();
054        public void addChildRegion(IRegion chr);
055        public List<IRegion> getChildRegions();
056
057}
