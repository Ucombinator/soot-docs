034/**
035 * <p>
036 * Represents the control flow graph of a {@link Body} at the basic block level.
037 * Each node of the graph is a {@link Block} while the edges represent the flow
038 * of control from one basic block to the next.
039 * </p>
040 *
041 * <p>
042 * This is an abstract base class for different variants of {@link BlockGraph},
043 * where the variants differ in how they analyze the control flow between
044 * individual units (represented by passing different variants of
045 * {@link UnitGraph} to the <code>BlockGraph</code> constructor) and in how they
046 * identify block leaders (represented by overriding <code>BlockGraph</code>'s
047 * definition of {@link computeLeaders()}.
048 */
049public abstract class BlockGraph implements DirectedGraph<Block> {
050        protected Body mBody;
051        protected Chain<Unit> mUnits;
052        protected List<Block> mBlocks;
053        protected List<Block> mHeads = new ArrayList<Block>();
054        protected List<Block> mTails = new ArrayList<Block>();
055
056        /**
057         * Create a <code>BlockGraph</code> representing at the basic block level
058         * the control flow specified, at the <code>Unit</code> level, by a given
059         * {@link UnitGraph}.
060         *
061         * @param unitGraph
062         *            A representation of the control flow at the level of
063         *            individual {@link Unit}s.
064         */
065        protected BlockGraph(UnitGraph unitGraph) {
066                mBody = unitGraph.getBody();
067                mUnits = mBody.getUnits();
068                Set<Unit> leaders = computeLeaders(unitGraph);
069                buildBlocks(leaders, unitGraph);
070        }
071
072        /**
073         * <p>
074         * Utility method for computing the basic block leaders for a {@link Body},
075         * given its {@link UnitGraph} (i.e., the instructions which begin new basic
076         * blocks).
077         * </p>
078         *
079         * <p>
080         * This implementation designates as basic block leaders :
081         *
082         * <ul>
083         *
084         * <li>Any <code>Unit</code> which has zero predecessors (e.g. the
085         * <code>Unit</code> following a return or unconditional branch) or more
086         * than one predecessor (e.g. a merge point).</li>
087         *
088         * <li><code>Unit</code>s which are the target of any branch (even if they
089         * have no other predecessors and the branch has no other successors, which
090         * is possible for the targets of unconditional branches or degenerate
091         * conditional branches which both branch and fall through to the same
092         * <code>Unit</code>).</li>
093         *
094         * <li>All successors of any <code>Unit</code> which has more than one
095         * successor (this includes the successors of <code>Unit</code>s which may
096         * throw an exception that gets caught within the <code>Body</code>, as well
097         * the successors of conditional branches).</li>
098         *
099         * <li>The first <code>Unit</code> in any <code>Trap</code> handler.
100         * (Strictly speaking, if <code>unitGraph</code> were a
101         * <code>ExceptionalUnitGraph</code> that included only a single
102         * unexceptional predecessor for some handler&mdash;because no trapped unit
103         * could possibly throw the exception that the handler catches, while the
104         * code preceding the handler fell through to the handler's code&mdash;then
105         * you could merge the handler into the predecessor's basic block; but such
106         * situations occur only in carefully contrived bytecode.)
107         *
108         * </ul>
109         * </p>
110         *
111         * @param unitGraph
112         *            is the <code>Unit</code>-level CFG which is to be split into
113         *            basic blocks.
114         *
115         * @return the {@link Set} of {@link Unit}s in <code>unitGraph</code> which
116         *         are block leaders.
117         */
118        protected Set<Unit> computeLeaders(UnitGraph unitGraph) {
119                Body body = unitGraph.getBody();
120                if (body != mBody) {
121                        throw new RuntimeException(
122                                        "BlockGraph.computeLeaders() called with a UnitGraph that doesn't match its mBody.");
123                }
124                Set<Unit> leaders = new HashSet<Unit>();
125
126                // Trap handlers start new basic blocks, no matter how many
127                // predecessors they have.
128                Chain<Trap> traps = body.getTraps();
129                for (Iterator<Trap> trapIt = traps.iterator(); trapIt.hasNext();) {
130                        Trap trap = trapIt.next();
131                        leaders.add(trap.getHandlerUnit());
132                }
133
134                for (Iterator<Unit> unitIt = body.getUnits().iterator(); unitIt.hasNext();) {
135                        Unit u = unitIt.next();
136                        List<Unit> predecessors = unitGraph.getPredsOf(u);
137                        int predCount = predecessors.size();
138                        List<Unit> successors = unitGraph.getSuccsOf(u);
139                        int succCount = successors.size();
140
141                        if (predCount != 1) { // If predCount == 1 but the predecessor
142                                leaders.add(u); // is a branch, u will get added by that
143                        } // branch's successor test.
144                        if ((succCount > 1) || (u.branches())) {
145                                for (Iterator<Unit> it = successors.iterator(); it.hasNext();) {
146                                        leaders.add((Unit) it.next()); // The cast is an
147                                } // assertion check.
148                        }
149                }
150                return leaders;
151        }
152
153        /**
154         * <p>
155         * A utility method that does most of the work of constructing basic blocks,
156         * once the set of block leaders has been determined, and which designates
157         * the heads and tails of the graph.
158         * </p>
159         *
160         * <p>
161         * <code>BlockGraph</code> provides an implementation of
162         * <code>buildBlocks()</code> which splits the {@link Unit}s in
163         * <code>unitGraph</code> so that each <code>Unit</code> in the passed set
164         * of block leaders is the first unit in a block. It defines as heads the
165         * blocks which begin with <code>Unit</code>s which are heads in
166         * <code>unitGraph</code>, and defines as tails the blocks which end with
167         * <code>Unit</code>s which are tails in <code>unitGraph</code>. Subclasses
168         * might override this behavior.
169         *
170         * @param leaders
171         *            Contains <code>Unit</code>s which are to be block leaders.
172         *
173         * @param unitGraph
174         *            Provides information about the predecessors and successors of
175         *            each <code>Unit</code> in the <code>Body</code>, for
176         *            determining the predecessors and successors of each created
177         *            {@link Block}.
178         *
179         * @return a {@link Map} from {@link Unit}s which begin or end a block to
180         *         the block which contains them.
181         */
182        protected Map<Unit, Block> buildBlocks(Set<Unit> leaders, UnitGraph unitGraph) {
183                List<Block> blockList = new ArrayList<Block>(leaders.size());
184                Map<Unit, Block> unitToBlock = new HashMap<Unit, Block>(); // Maps head
185                                                                                                                                        // and tail
186                                                                                                                                        // units to
187                // their blocks, for building
188                // predecessor and successor lists.
189                Unit blockHead = null;
190                int blockLength = 0;
191                Iterator<Unit> unitIt = mUnits.iterator();
192                if (unitIt.hasNext()) {
193                        blockHead = unitIt.next();
194                        if (!leaders.contains(blockHead)) {
195                                throw new RuntimeException("BlockGraph: first unit not a leader!");
196                        }
197                        blockLength++;
198                }
199                Unit blockTail = blockHead;
200                int indexInMethod = 0;
201
202                while (unitIt.hasNext()) {
203                        Unit u = unitIt.next();
204                        if (leaders.contains(u)) {
205                                addBlock(blockHead, blockTail, indexInMethod, blockLength, blockList, unitToBlock);
206                                indexInMethod++;
207                                blockHead = u;
208                                blockLength = 0;
209                        }
210                        blockTail = u;
211                        blockLength++;
212                }
213                if (blockLength > 0) {
214                        // Add final block.
215                        addBlock(blockHead, blockTail, indexInMethod, blockLength, blockList, unitToBlock);
216                }
217
218                // The underlying UnitGraph defines heads and tails.
219                for (Iterator<Unit> it = unitGraph.getHeads().iterator(); it.hasNext();) {
220                        Unit headUnit = (Unit) it.next();
221                        Block headBlock = unitToBlock.get(headUnit);
222                        if (headBlock.getHead() == headUnit) {
223                                mHeads.add(headBlock);
224                        } else {
225                                throw new RuntimeException("BlockGraph(): head Unit is not the first unit in the corresponding Block!");
226                        }
227                }
228                for (Iterator<Unit> it = unitGraph.getTails().iterator(); it.hasNext();) {
229                        Unit tailUnit = (Unit) it.next();
230                        Block tailBlock = unitToBlock.get(tailUnit);
231                        if (tailBlock.getTail() == tailUnit) {
232                                mTails.add(tailBlock);
233                        } else {
234                                throw new RuntimeException("BlockGraph(): tail Unit is not the last unit in the corresponding Block!");
235                        }
236                }
237
238                for (Iterator<Block> blockIt = blockList.iterator(); blockIt.hasNext();) {
239                        Block block = blockIt.next();
240
241                        List<Unit> predUnits = unitGraph.getPredsOf(block.getHead());
242                        List<Block> predBlocks = new ArrayList<Block>(predUnits.size());
243                        for (Iterator<Unit> predIt = predUnits.iterator(); predIt.hasNext();) {
244                                Unit predUnit = predIt.next();
245                                Block predBlock = unitToBlock.get(predUnit);
246                                if (predBlock == null) {
247                                        throw new RuntimeException("BlockGraph(): block head mapped to null block!");
248                                }
249                                predBlocks.add(predBlock);
250                        }
251
252                        if (predBlocks.size() == 0) {
253                                block.setPreds(Collections.<Block> emptyList());
254
255                                // If the UnreachableCodeEliminator is not eliminating
256                                // unreachable handlers, then they will have no
257                                // predecessors, yet not be heads.
258                                /*
259                                 * if (! mHeads.contains(block)) { throw new
260                                 * RuntimeException("Block with no predecessors is not a head!"
261                                 * );
262                                 * 
263                                 * // Note that a block can be a head even if it has //
264                                 * predecessors: a handler that might catch an exception //
265                                 * thrown by the first Unit in the method. }
266                                 */
267                        } else {
268                                block.setPreds(Collections.unmodifiableList(predBlocks));
269                                if (block.getHead() == mUnits.getFirst()) {
270                                        mHeads.add(block); // Make the first block a head
271                                        // even if the Body is one huge loop.
272                                }
273                        }
274
275                        List<Unit> succUnits = unitGraph.getSuccsOf(block.getTail());
276                        List<Block> succBlocks = new ArrayList<Block>(succUnits.size());
277                        for (Iterator<Unit> succIt = succUnits.iterator(); succIt.hasNext();) {
278                                Unit succUnit = succIt.next();
279                                Block succBlock = unitToBlock.get(succUnit);
280                                if (succBlock == null) {
281                                        throw new RuntimeException("BlockGraph(): block tail mapped to null block!");
282                                }
283                                succBlocks.add(succBlock);
284                        }
285                        
286                        if (succBlocks.size() == 0) {
287                                block.setSuccs(Collections.<Block> emptyList());
288                                if (!mTails.contains(block)) {
289                                        // if this block is totally empty and unreachable, we remove it
290                                        if (block.getPreds().isEmpty()
291                                                        && block.getHead() == block.getTail()
292                                                        && block.getHead() instanceof NopStmt)
293                                                blockIt.remove();
294                                        else
295                                                throw new RuntimeException("Block with no successors is not a tail!: " + block.toString());
296                                        // Note that a block can be a tail even if it has
297                                        // successors: a return that throws a caught exception.
298                                }
299                        } else {
300                                block.setSuccs(Collections.unmodifiableList(succBlocks));
301                        }
302                }
303                mBlocks = Collections.unmodifiableList(blockList);
304                mHeads = Collections.unmodifiableList(mHeads);
305                if (mTails.size() == 0) {
306                        mTails = Collections.emptyList();
307                } else {
308                        mTails = Collections.unmodifiableList(mTails);
309                }
310                return unitToBlock;
311        }

313        /**
314         * A utility method which creates a new block and adds information about it
315         * to data structures used to build the graph.
316         *
317         * @param head
318         *            The first unit in the block.
319         * @param tail
320         *            The last unit in the block.
321         * @param index
322         *            The index of this block this {@link Body}.
323         * @param length
324         *            The number of units in this block.
325         * @param blockList
326         *            The list of blocks for this method. <code>addBlock()</code>
327         *            will add the newly created block to this list.
328         * @param unitToBlock
329         *            A map from units to blocks. <code>addBlock()</code> will add
330         *            mappings from <code>head</code> and <code>tail</code> to the
331         *            new block
332         */
333        private void addBlock(Unit head, Unit tail, int index, int length, List<Block> blockList,
334                        Map<Unit, Block> unitToBlock) {
335                Block block = new Block(head, tail, mBody, index, length, this);
336                blockList.add(block);
337                unitToBlock.put(tail, block);
338                unitToBlock.put(head, block);
339        }

341        /**
342         * Returns the {@link Body} this {@link BlockGraph} is derived from.
343         * 
344         * @return The {@link Body} this {@link BlockGraph} is derived from.
345         */
346        public Body getBody() {
347                return mBody;
348        }
349
350        /**
351         * Returns a list of the Blocks composing this graph.
352         * 
353         * @return A list of the blocks composing this graph in the same order as
354         *         they partition underlying Body instance's unit chain.
355         * @see Block
356         */
357        public List<Block> getBlocks() {
358                return mBlocks;
359        }
360
361        public String toString() {
362
363                Iterator<Block> it = mBlocks.iterator();
364                StringBuffer buf = new StringBuffer();
365                while (it.hasNext()) {
366                        Block someBlock = it.next();
367
368                        buf.append(someBlock.toString() + '\n');
369                }
370
371                return buf.toString();
372        }

374        /* DirectedGraph implementation */
375        public List<Block> getHeads() {
376                return mHeads;
377        }
378
379        public List<Block> getTails() {
380                return mTails;
381        }
382
383        public List<Block> getPredsOf(Block b) {
384                return b.getPreds();
385        }
386
387        public List<Block> getSuccsOf(Block b) {
388                return b.getSuccs();
389        }
390
391        public int size() {
392                return mBlocks.size();
393        }
394
395        public Iterator<Block> iterator() {
396                return mBlocks.iterator();
397        }
