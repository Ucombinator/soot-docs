043/**
044 *  <p>Represents a CFG where the nodes are {@link Block}s and the
045 *  edges are derived from control flow.  Control flow associated with
046 *  exceptions is taken into account: when a {@link Unit} may throw
047 *  an exception that is caught by a {@link Trap} within the
048 *  <code>Body</code>, the excepting <code>Unit</code> starts a new basic
049 *  block (<code>Unit</code>s do not start a new block when all the exceptions
050 *  they might throw would escape the method without being caught).
051 *  </p>
052 */
053
054public class ExceptionalBlockGraph extends BlockGraph implements ExceptionalGraph<Block>
055{
056    // Maps for distinguishing exceptional and unexceptional control flow.
057    // We follow two conventions to save space (and runtime, if no client ever 
058    // asks for the exceptional information):  if the graph contains no
059    // exceptional edges (e.g. there are no traps in the method) we leave
060    // all these map references as NULL, while if an individual block has only 
061    // unexceptional successors or predecessors, it is not added to the 
062    // relevant map. When the access methods are asked about such blocks, 
063    // they return empty lists for the exceptional predecessors and successors,
064    // and the complete list of predecessors or successors for 
065    // the unexceptional predecessors and successors.
066    Map<Block, List<Block>> blockToExceptionalPreds;
067    Map<Block, List<Block>> blockToExceptionalSuccs;
068    Map<Block, List<Block>> blockToUnexceptionalPreds;
069    Map<Block, List<Block>> blockToUnexceptionalSuccs;
070    Map<Block, Collection<ExceptionDest>> blockToExceptionDests;
071
072    // When the graph has no traps (thus no exceptional CFG edges); we cache the
073    // throwAnalysis for generating ExceptionDests on demand.  If there
074    // are traps, throwAnalysis remains null.
075    ThrowAnalysis throwAnalysis ;
076
077    /**
078     *   <p>Constructs an <code>ExceptionalBlockGraph</code> for the
079     *   blocks found by partitioning the the units of the provided
080     *   <code>Body</code> instance into basic blocks.</p>
081     *
082     *   <p>Note that this constructor builds an {@link
083     *   ExceptionalUnitGraph} internally when splitting <code>body</code>'s
084     *   {@link Unit}s into {@link Block}s.  Callers who already have
085     *   an <code>ExceptionalUnitGraph</code> to hand can use the constructor
086     *   taking an <code>ExceptionalUnitGraph</code> as a parameter, as a
087     *   minor optimization.
088     *
089     *   @param body    The underlying body we want to make a graph for.
090     */
091    public ExceptionalBlockGraph(Body body)
092    {
093        this(new ExceptionalUnitGraph(body));
094    }

112    /**
113     * {@inheritDoc}
114     *
115     * This implementation calls the inherited implementation to split
116     * units into blocks, before adding the distinctions between
117     * exceptional and unexceptional control flow.
118     *
119     * @param {@inheritDoc}
120     *
121     * @return {@inheritDoc}
122     */
123
124    @Override
125        protected Map<Unit,Block> buildBlocks(Set<Unit> leaders, UnitGraph uncastUnitGraph) {
126        ExceptionalUnitGraph unitGraph = (ExceptionalUnitGraph) uncastUnitGraph;
127        Map<Unit,Block> unitToBlock = super.buildBlocks(leaders, unitGraph);
128
129        if (unitGraph.getBody().getTraps().size() == 0) {
130            // All exceptions escape the method.  Cache the ThrowAnalysis
131            // to respond to getExceptionDests() on demand.
132            throwAnalysis = unitGraph.getThrowAnalysis();
133            if (throwAnalysis == null) {
134                throw new IllegalStateException("ExceptionalUnitGraph lacked a cached ThrowAnalysis for a Body with no Traps.");
135            }
136
137        } else {
138            int initialMapSize = (mBlocks.size() * 2) / 3;
139            blockToUnexceptionalPreds = new HashMap<Block, List<Block>>(initialMapSize);
140            blockToUnexceptionalSuccs = new HashMap<Block, List<Block>>(initialMapSize);
141            blockToExceptionalPreds = new HashMap<Block, List<Block>>(initialMapSize);
142            blockToExceptionalSuccs = new HashMap<Block, List<Block>>(initialMapSize);
143
144            for (Block block : mBlocks) {
145                Unit blockHead = block.getHead();
146                List<Unit> exceptionalPredUnits = unitGraph.getExceptionalPredsOf(blockHead);
147                if (exceptionalPredUnits.size() != 0) {
148                    List<Block> exceptionalPreds = mappedValues(exceptionalPredUnits,
149                                                         unitToBlock);
150                    exceptionalPreds = Collections.unmodifiableList(exceptionalPreds);
151                    blockToExceptionalPreds.put(block, exceptionalPreds);
152                    List<Unit> unexceptionalPredUnits = unitGraph.getUnexceptionalPredsOf(blockHead);
153                    List<Block> unexceptionalPreds = null;
154                    if (unexceptionalPredUnits.size() == 0) {
155                        unexceptionalPreds = Collections.emptyList();
156                    } else {
157                        unexceptionalPreds = mappedValues(unexceptionalPredUnits,
158                                                          unitToBlock);
159                        unexceptionalPreds = Collections.unmodifiableList(unexceptionalPreds);
160                    }
161                    blockToUnexceptionalPreds.put(block, unexceptionalPreds);
162                }
163
164                Unit blockTail = block.getTail();
165                List<Unit> exceptionalSuccUnits = unitGraph.getExceptionalSuccsOf(blockTail);
166                if (exceptionalSuccUnits.size() != 0) {
167                    List<Block> exceptionalSuccs = mappedValues(exceptionalSuccUnits,
168                                                         unitToBlock);
169                    exceptionalSuccs = Collections.unmodifiableList(exceptionalSuccs);
170                      blockToExceptionalSuccs.put(block, exceptionalSuccs);
171                    List<Unit> unexceptionalSuccUnits = unitGraph.getUnexceptionalSuccsOf(blockTail);
172                    List<Block> unexceptionalSuccs = null;
173                    if (unexceptionalSuccUnits.size() == 0) {
174                        unexceptionalSuccs = Collections.emptyList();
175                    } else {
176                        unexceptionalSuccs = mappedValues(unexceptionalSuccUnits,
177                                                          unitToBlock);
178                        unexceptionalSuccs = Collections.unmodifiableList(unexceptionalSuccs);
179                    }
180                    blockToUnexceptionalSuccs.put(block,unexceptionalSuccs);
181                }
182            }
183            blockToExceptionDests = buildExceptionDests(unitGraph, unitToBlock);
184        }
185        return unitToBlock;
186    }

189    /**
190     * Utility method which, given a {@link List} of objects and a {@link Map}
191     * where those objects appear as keys, returns a <code>List</code> of the
192     * values to which the keys map, in the corresponding order.
193     *
194     * @param keys the keys to be looked up.
195     *
196     * @param keyToValue the map in which to look up the keys.
197     *
198     * @throws IllegalStateException if one of the elements in <code>keys</code>
199     * does not appear in <code>keyToValue</code>
200     */
201    private <K,V> List<V> mappedValues(List<K> keys, Map<K,V> keyToValue) {
202        List<V> result = new ArrayList<V>(keys.size());
203        for (K key : keys) {
204            V value = keyToValue.get(key);
205            if (value == null) {
206                throw new IllegalStateException("No value corresponding to key: " + key.toString());
207            }
208            result.add(value);
209        }
210        return result;
211    }

213    private Map<Block, Collection<ExceptionDest>> buildExceptionDests(ExceptionalUnitGraph unitGraph, 
214                                    Map<Unit,Block> unitToBlock) {
215        Map<Block, Collection<ExceptionDest>> result = new HashMap<Block, Collection<ExceptionDest>>(mBlocks.size() * 2 + 1, 0.7f);
216        for (Block block : mBlocks) {
217            result.put(block, collectDests(block, unitGraph, unitToBlock));
218        }
219        return result;
220    }

223    /**
224     * Utility method which, given a {@link Block} and the 
225     * {@link ExceptionalUnitGraph} from which it was constructed,
226     * returns the {@link ExceptionDest}s representing the exceptions
227     * which may be thrown by units in the block.
228     *
229     * @param block the {@link Block} whose exceptions are to be collected.
230     *
231     * @param unitGraph the {@link ExceptionalUnitGraph} from which this 
232     *                  graph is constructed.
233     *
234     * @param unitToBlock a {@link Map} from the units which are block heads 
235     *                    or tails to the blocks that they belong to.
236     *
237     * @return a {@link Collection} of {@link ExceptionDest}s representing 
238     *           the exceptions that may be thrown by this block, together
239     *           with their catchers.
240     */
241    private Collection<ExceptionDest> collectDests(Block block, ExceptionalUnitGraph unitGraph,
242                                    Map<Unit,Block> unitToBlock) {
243        Unit blockHead = block.getHead();
244        Unit blockTail = block.getTail();
245        ArrayList<ExceptionDest> blocksDests = null;
246        ThrowableSet escapingThrowables = ThrowableSet.Manager.v().EMPTY;
247        Map<Trap, ThrowableSet> trapToThrowables = null;  // Don't allocate unless we need it.
248        int caughtCount = 0;
249
250        for (Unit unit2 : block) {
251            Unit unit = unit2;
252            Collection<ExceptionalUnitGraph.ExceptionDest> unitDests = unitGraph.getExceptionDests(unit);
253            if (unitDests.size() != 1 && unit != blockHead && unit != blockTail) {
254                throw new IllegalStateException("Multiple ExceptionDests associated with a unit which does not begin or end its block.");
255            }
256            for (soot.toolkits.graph.ExceptionalUnitGraph.ExceptionDest unitDest : unitDests) {
257                if (unitDest.getTrap() == null) {
258                    try {
259                        escapingThrowables = escapingThrowables.add(unitDest.getThrowables());
260                    } catch (ThrowableSet.AlreadyHasExclusionsException e) {
261                        if (escapingThrowables != ThrowableSet.Manager.v().EMPTY) {
262                            // Return multiple escaping ExceptionDests,
263                            // since ThrowableSet's limitations do not permit us
264                            // to add all the escaping type descriptions together.
265                            if (blocksDests == null) {
266                                blocksDests = new ArrayList<ExceptionDest>(10);
267                            }
268                            blocksDests.add(new ExceptionDest(null, 
269                                                              escapingThrowables,
270                                                              null));
271                        }
272                        escapingThrowables = unitDest.getThrowables();
273                    }
274                } else {
275                    if (unit != blockHead && unit != blockTail) {
276                        // Assertion failure.
277                        throw new IllegalStateException("Unit " 
278                                                        + unit.toString() 
279                                                        + " is not a block head or tail, yet it throws " 
280                                                        + unitDest.getThrowables() 
281                                                        + " to "
282                                                        + unitDest.getTrap());
283                    }
284                    caughtCount++;
285                    if (trapToThrowables == null) {
286                        trapToThrowables = new HashMap<Trap, ThrowableSet>(unitDests.size() * 2);
287                    }
288                    Trap trap = unitDest.getTrap();
289                    ThrowableSet throwables = trapToThrowables.get(trap);
290                    if (throwables == null) {
291                        throwables = unitDest.getThrowables();
292                    } else {
293                        throwables = throwables.add(unitDest.getThrowables());
294                    }
295                    trapToThrowables.put(trap, throwables);
296                }
297            }
298        }
299        
300        if (blocksDests == null) {
301            blocksDests = new ArrayList<ExceptionDest>(caughtCount + 1);
302        } else {
303            blocksDests.ensureCapacity(blocksDests.size() + caughtCount);
304        }
305
306        if (escapingThrowables != ThrowableSet.Manager.v().EMPTY) {
307            ExceptionDest escapingDest = new ExceptionDest(null, escapingThrowables, 
308                                                           null);
309            blocksDests.add(escapingDest);
310        }
311        if (trapToThrowables != null) {
312            for (Map.Entry<Trap,ThrowableSet> entry : trapToThrowables.entrySet()) {
313                Trap trap = entry.getKey();
314                Block trapBlock = unitToBlock.get(trap.getHandlerUnit());
315                if (trapBlock == null) {
316                    throw new IllegalStateException("catching unit is not recorded as a block leader.");
317                }
318                ThrowableSet throwables = entry.getValue();
319                ExceptionDest blockDest = new ExceptionDest(trap, throwables, trapBlock);
320                blocksDests.add(blockDest);
321            }
322        }
323        return blocksDests;
324    }

327    @Override
328        public List<Block> getUnexceptionalPredsOf(Block b) {
329        if ((blockToUnexceptionalPreds == null) 
330            || (! blockToUnexceptionalPreds.containsKey(b))) {
331            Block block = b;
332            return block.getPreds();
333        } else {
334            return blockToUnexceptionalPreds.get(b);
335        }
336    }
337
338
339    @Override
340        public List<Block> getUnexceptionalSuccsOf(Block b) {
341        if ((blockToUnexceptionalSuccs == null) 
342            || (! blockToUnexceptionalSuccs.containsKey(b))) {
343            Block block = b;
344            return block.getSuccs();
345        } else {
346            return blockToUnexceptionalSuccs.get(b);
347        }
348    }

351    @Override
352        public List<Block> getExceptionalPredsOf(Block  b) {
353        if (blockToExceptionalPreds == null ||
354            (!blockToExceptionalPreds.containsKey(b))) {
355            return Collections.emptyList();
356        } else {
357            return blockToExceptionalPreds.get(b);
358        }
359    }
360
361
362    @Override
363        public List<Block> getExceptionalSuccsOf(Block b) {
364        if (blockToExceptionalSuccs == null ||
365            (!blockToExceptionalSuccs.containsKey(b))) {
366            return Collections.emptyList();
367        } else {
368            return blockToExceptionalSuccs.get(b);
369        }
370    }

373        @Override
374        public Collection<ExceptionDest> getExceptionDests(final Block b) {
375                if (blockToExceptionDests == null) {
376                        ExceptionDest e = new ExceptionDest(null, null, null) {
377                                private ThrowableSet throwables;
378                                
379                                @Override
380                                public ThrowableSet getThrowables() {
381                                        if (null == throwables) {
382                                                throwables = ThrowableSet.Manager.v().EMPTY;
383                                                for (Unit unit : b) {
384                                                        throwables = throwables.add(throwAnalysis.mightThrow(unit));
385                                                }
386                                        }
387                                        return throwables;
388                                }
389                        };
390                        return Collections.singletonList(e);
391                }
392                return blockToExceptionDests.get(b);
393        }

396    public static class ExceptionDest implements ExceptionalGraph.ExceptionDest<Block> {
397        private Trap trap;
398        private ThrowableSet throwables;
399        private Block handler;
400
401        protected ExceptionDest(Trap trap, ThrowableSet throwables, Block handler) {
402            this.trap = trap;
403            this.throwables = throwables;
404            this.handler = handler;
405        }
406        
407        @Override
408        public Trap getTrap() {
409            return trap;
410        }
411
412        @Override
413        public ThrowableSet getThrowables() {
414            return throwables;
415        }
416
417        @Override
418        public Block getHandlerNode() {
419            return handler;
420        }
421
422        @Override
423        public String toString() {
424            StringBuffer buf = new StringBuffer();
425            buf.append(getThrowables());
426            buf.append(" -> ");
427            if (trap == null) {
428                buf.append("(escapes)");
429            } else {
430                buf.append(trap.toString());
431                buf.append("handler: ");
432                buf.append(getHandlerNode().toString());
433            }
434            return buf.toString();
435        }
436    }
