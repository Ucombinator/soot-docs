045/**
046 * Analysis that provides an implementation of the LocalDefs interface.
047 */
048public class SimpleLocalDefs implements LocalDefs {
049        static private class StaticSingleAssignment implements LocalDefs {
050                final Map<Local, List<Unit>> result;
051
052                StaticSingleAssignment(Local[] locals, List<Unit>[] unitList) {
053                        assert locals.length == unitList.length;
054
055                        final int N = locals.length;
056                        result = new HashMap<Local, List<Unit>>((N * 3) / 2 + 7);
057
058                        for (int i = 0; i < N; i++) {
059                                if (unitList[i].isEmpty())
060                                        continue;
061                                assert unitList[i].size() == 1;
062                                result.put(locals[i], unitList[i]);
063                        }
064                }
065
066                @Override
067                public List<Unit> getDefsOfAt(Local l, Unit s) {
068                        List<Unit> lst = result.get(l);
069                        if (lst == null)
070                                return emptyList();
071
072                        // singleton-lists are immutable
073                        return lst;
074                }
075
076                @Override
077                public List<Unit> getDefsOf(Local l) {
078                        return getDefsOfAt(l, null);
079                }
080                
081        }
082
083        static private class FlowAssignment extends
084                        ForwardFlowAnalysis<Unit, FlowAssignment.FlowBitSet> implements
085                        LocalDefs {
086                class FlowBitSet extends BitSet {
087                        private static final long serialVersionUID = -8348696077189400377L;
088
089                        FlowBitSet() {
090                                super(universe.length);
091                        }
092
093                        List<Unit> asList(int fromIndex, int toIndex) {
094                                BitSet bits = this;
095                                if (universe.length < toIndex || toIndex < fromIndex
096                                                || fromIndex < 0)
097                                        throw new IndexOutOfBoundsException();
098
099                                if (fromIndex == toIndex) {
100                                        return emptyList();
101                                }
102
103                                if (fromIndex == toIndex - 1) {
104                                        if (bits.get(fromIndex)) {
105                                                return singletonList(universe[fromIndex]);
106                                        }
107                                        return emptyList();
108                                }
109
110                                int i = bits.nextSetBit(fromIndex);
111                                if (i < 0 || i >= toIndex)
112                                        return emptyList();
113
114                                if (i == toIndex - 1)
115                                        return singletonList(universe[i]);
116
117                                List<Unit> elements = new ArrayList<Unit>(toIndex - i);
118
119                                for (;;) {
120                                        int endOfRun = Math.min(toIndex, bits.nextClearBit(i + 1));
121                                        do {
122                                                elements.add(universe[i++]);
123                                        } while (i < endOfRun);
124                                        if (i >= toIndex)
125                                                break;
126                                        i = bits.nextSetBit(i + 1);
127                                        if (i < 0 || i >= toIndex)
128                                                break;
129                                }
130                                return elements;
131                        }
132                }
133
134                final Map<Local, Integer> locals;
135                final List<Unit>[] unitList;
136                final int[] localRange;
137                final Unit[] universe;
138
139                private Map<Unit, Integer> indexOfUnit;
140
141                FlowAssignment(DirectedGraph<Unit> graph, Local[] locals,
142                                List<Unit>[] unitList, int units, boolean omitSSA) {
143                        super(graph);
144
145                        final int N = locals.length;
146
147                        this.locals = new HashMap<Local, Integer>((N * 3) / 2 + 7);
148                        this.unitList = unitList;
149
150                        universe = new Unit[units];
151                        indexOfUnit = new HashMap<Unit, Integer>(units);
152
153                        localRange = new int[N + 1];
154                        for (int j = 0, i = 0; i < N; localRange[++i] = j) {
155                                if (unitList[i].isEmpty())
156                                        continue;
157
158                                this.locals.put(locals[i], i);
159
160                                if (unitList[i].size() >= 2) {
161                                        for (Unit u : unitList[i]) {
162                                                indexOfUnit.put(u, j);
163                                                universe[j++] = u;
164                                        }
165                                } else if (omitSSA) {
166                                        universe[j++] = unitList[i].get(0);
167                                }
168                        }
169                        assert localRange[N] == units;
170
171                        doAnalysis();
172
173                        indexOfUnit.clear();
174                        indexOfUnit = null;
175                }
176
177                @Override
178                public List<Unit> getDefsOfAt(Local l, Unit s) {
179                        Integer lno = locals.get(l);
180                        if (lno == null)
181                                return emptyList();
182
183                        int from = localRange[lno];
184                        int to = localRange[lno + 1];
185                        assert from <= to;
186
187                        if (from == to) {
188                                assert unitList[lno].size() == 1;
189                                // both singletonList is immutable
190                                return unitList[lno];
191                        }
192
193                        return getFlowBefore(s).asList(from, to);
194                }
195
196                @Override
197                protected boolean omissible(Unit u) {
198                        // avoids temporary creation of iterators (more like micro-tuning)
199                        if (u.getDefBoxes().isEmpty())
200                                return true;
201                        for (ValueBox vb : u.getDefBoxes()) {
202                                Value v = vb.getValue();
203                                if (v instanceof Local) {
204                                        Local l = (Local) v;
205                                        int lno = l.getNumber();
206                                        return (localRange[lno] == localRange[lno + 1]);
207                                }
208                        }
209                        return true;
210                }
211
212                @Override
213                protected Flow getFlow(Unit from, Unit to) {
214                        // QND
215                        if (to instanceof IdentityUnit) {
216                                if (graph instanceof ExceptionalGraph) {
217                                        ExceptionalGraph<Unit> g = (ExceptionalGraph<Unit>) graph;
218                                        if (!g.getExceptionalPredsOf(to).isEmpty()) {
219                                                // look if there is a real exception edge
220                                                for (ExceptionDest<Unit> exd : g
221                                                                .getExceptionDests(from)) {
222                                                        Trap trap = exd.getTrap();
223                                                        if (null == trap)
224                                                                continue;
225
226                                                        if (trap.getHandlerUnit() == to)
227                                                                return Flow.IN;
228                                                }
229                                        }
230                                }
231                        }
232                        return Flow.OUT;
233                }
234
235                @Override
236                protected void flowThrough(FlowBitSet in, Unit unit, FlowBitSet out) {
237                        copy(in, out);
238
239                        // reassign all definitions
240                        for (ValueBox vb : unit.getDefBoxes()) {
241                                Value v = vb.getValue();
242                                if (v instanceof Local) {
243                                        Local l = (Local) v;
244                                        int lno = l.getNumber();
245
246                                        int from = localRange[lno];
247                                        int to = localRange[1 + lno];
248
249                                        if (from == to)
250                                                continue;
251
252                                        assert from <= to;
253
254                                        if (to - from == 1) {
255                                                // special case: this local has only one def point
256                                                out.set(from);
257                                        } else {
258                                                out.clear(from, to);
259                                                out.set(indexOfUnit.get(unit));
260                                        }
261                                }
262                        }
263                }
264
265                @Override
266                protected void copy(FlowBitSet source, FlowBitSet dest) {
267                        if (dest == source)
268                                return;
269                        dest.clear();
270                        dest.or(source);
271                }
272
273                @Override
274                protected FlowBitSet newInitialFlow() {
275                        return new FlowBitSet();
276                }
277
278                @Override
279                protected void mergeInto(Unit succNode, FlowBitSet inout, FlowBitSet in) {
280                        inout.or(in);
281                }
282
283                @Override
284                protected void merge(FlowBitSet in1, FlowBitSet in2, FlowBitSet out) {
285                        throw new UnsupportedOperationException("should never be called");
286                }
287
288                @Override
289                public List<Unit> getDefsOf(Local l) {
290                        List<Unit> defs = new ArrayList<Unit>();
291                        for (Unit u : graph) {
292                                List<Unit> defsOf = getDefsOfAt(l, u);
293                                if (defsOf != null)
294                                        defs.addAll(defsOf);
295                        }
296                        return defs;
297                }
298                
299        }

308                this(graph, FlowAnalysisMode.Automatic);

312                this(graph, graph.getBody().getLocals(), mode);

315        SimpleLocalDefs(DirectedGraph<Unit> graph, Collection<Local> locals,
316                        FlowAnalysisMode mode) {
317                this(graph, locals.toArray(new Local[locals.size()]), mode);
318        }
319
320        SimpleLocalDefs(DirectedGraph<Unit> graph, Local[] locals, boolean omitSSA) {
321                this(graph, locals, omitSSA ? FlowAnalysisMode.OmitSSA : FlowAnalysisMode.Automatic);
322        }
323        
324        /**
325         * The different modes in which the flow analysis can run
326         */
327        enum FlowAnalysisMode {
328                /**
329                 * Automatically detect the mode to use
330                 */
331                Automatic,
332                /**
333                 * Never use the SSA form, even if the unit graph would allow for a
334                 * flow-insensitive analysis without losing precision
335                 */
336                OmitSSA,
337                /**
338                 * Always conduct a flow-insensitive analysis
339                 */
340                FlowInsensitive
341        }
342        
343        SimpleLocalDefs(DirectedGraph<Unit> graph, Local[] locals, FlowAnalysisMode mode) {
344                if (Options.v().time())
345                        Timers.v().defsTimer.start();
346
347                final int N = locals.length;
348
349                // reassign local numbers
350                int[] oldNumbers = new int[N];
351                for (int i = 0; i < N; i++) {
352                        oldNumbers[i] = locals[i].getNumber();
353                        locals[i].setNumber(i);
354                }
355
356                init(graph, locals, mode);
357
358                // restore local numbering
359                for (int i = 0; i < N; i++) {
360                        locals[i].setNumber(oldNumbers[i]);
361                }
362
363                if (Options.v().time())
364                        Timers.v().defsTimer.end();
365        }
366
367        private void init(DirectedGraph<Unit> graph, Local[] locals, FlowAnalysisMode mode) {
368                @SuppressWarnings("unchecked")
369                List<Unit>[] unitList = (List<Unit>[]) new List[locals.length];
370
371                Arrays.fill(unitList, emptyList());
372
373                boolean omitSSA = mode == FlowAnalysisMode.OmitSSA;
374                boolean doFlowAnalsis = omitSSA;
375
376                int units = 0;
377
378                // collect all def points
379                for (Unit unit : graph) {
380                        for (ValueBox box : unit.getDefBoxes()) {
381                                Value v = box.getValue();
382                                if (v instanceof Local) {
383                                        Local l = (Local) v;
384                                        int lno = l.getNumber();
385
386                                        switch (unitList[lno].size()) {
387                                        case 0:
388                                                unitList[lno] = singletonList(unit);
389                                                if (omitSSA)
390                                                        units++;
391                                                break;
392                                        case 1:
393                                                if (!omitSSA)
394                                                        units++;
395                                                unitList[lno] = new ArrayList<Unit>(unitList[lno]);
396                                                doFlowAnalsis = true;
397                                                // fallthrough
398                                        default:
399                                                unitList[lno].add(unit);
400                                                units++;
401                                                break;
402                                        }
403                                }
404                        }
405                }
406
407                if (doFlowAnalsis && mode != FlowAnalysisMode.FlowInsensitive) {
408                        def = new FlowAssignment(graph, locals, unitList, units, omitSSA);
409                } else {
410                        def = new StaticSingleAssignment(locals, unitList);
411                }
412        }
413
414        @Override
415        public List<Unit> getDefsOfAt(Local l, Unit s) {
416                return def.getDefsOfAt(l, s);
417        }
418
419        @Override
420        public List<Unit> getDefsOf(Local l) {
421                return def.getDefsOf(l);
422        }
