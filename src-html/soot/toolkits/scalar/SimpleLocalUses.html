031package soot.toolkits.scalar;
032import java.util.ArrayList;
033import java.util.Collections;
034import java.util.HashMap;
035import java.util.HashSet;
036import java.util.List;
037import java.util.Map;
038import java.util.Set;
039
040import soot.Body;
041import soot.G;
042import soot.Local;
043import soot.Timers;
044import soot.Unit;
045import soot.Value;
046import soot.ValueBox;
047import soot.options.Options;
048import soot.toolkits.graph.UnitGraph;
049
050
051/**
052 *  Analysis that implements the LocalUses interface.
053 *  Uses for a Local defined at a given Unit are returned as 
054 *  a list of UnitValueBoxPairs each containing a Unit that use the
055 *  local and the Local itself wrapped in a ValueBox.
056 */
057public class SimpleLocalUses implements LocalUses
058{
059        final Body body;
060    private Map<Unit, List<UnitValueBoxPair>> unitToUses;
061
062    /**
063     * Construct the analysis from a UnitGraph representation
064     * of a method body and a LocalDefs interface. This supposes that
065     * a LocalDefs analysis must have been computed prior.
066     *
067     * <p> Note: If you do not already have a UnitGraph, it may be
068     * cheaper to use the constructor which only requires a Body.
069     */
070    public SimpleLocalUses(UnitGraph graph, LocalDefs localDefs)
071    {
072        this(graph.getBody(), localDefs);
073    }
074
075    /**
076     * Construct the analysis from a method body and a LocalDefs
077     * interface. This supposes that a LocalDefs analysis must have
078     * been computed prior.
079     */
080    public SimpleLocalUses(Body body, LocalDefs localDefs)
081    {
082        this.body = body;
083        if(Options.v().time())
084           Timers.v().usesTimer.start();
085    
086        if(Options.v().time())
087           Timers.v().usePhase1Timer.start();
088        
089        if(Options.v().verbose())
090            G.v().out.println("[" + body.getMethod().getName() +
091                "]     Constructing SimpleLocalUses...");
092        
093        unitToUses = new HashMap<Unit, List<UnitValueBoxPair>>(body.getUnits().size() * 2 + 1, 0.7f);
094    
095        // Initialize this map to empty sets
096
097        if(Options.v().time())
098           Timers.v().usePhase1Timer.end();
099    
100        if(Options.v().time())
101           Timers.v().usePhase2Timer.start();
102    
103        // Traverse units and associate uses with definitions
104        for (Unit unit : body.getUnits()) {
105            for (ValueBox useBox : unit.getUseBoxes()) {
106                Value v = useBox.getValue();
107                if (v instanceof Local) {
108                    // Add this statement to the uses of the definition of the local
109                    Local l = (Local) v;
110                                        
111                    UnitValueBoxPair newPair = new UnitValueBoxPair(unit, useBox);
112                    
113                    List<Unit> defs = localDefs.getDefsOfAt(l, unit);
114                    if (defs != null) {
115                            for (Unit def : defs) {
116                                List<UnitValueBoxPair> lst = unitToUses.get(def);
117                                if (lst == null) {
118                                        unitToUses.put(def, lst = new ArrayList<UnitValueBoxPair>());
119                                }                       
120                                lst.add(newPair);
121                            }
122                    }
123                }
124            }
125        }
126        
127
128        if(Options.v().time())
129           Timers.v().usePhase2Timer.end();
130        
131        if(Options.v().time())
132            Timers.v().usesTimer.end();
133
134        if(Options.v().verbose())
135            G.v().out.println("[" + body.getMethod().getName() +
136                "]     finished SimpleLocalUses...");
137    }
138
139    /**
140     *  Uses for a Local defined at a given Unit are returned as 
141     *  a list of UnitValueBoxPairs each containing a Unit that use the
142     *  local and the Local itself wrapped in a ValueBox.
143     *  @param s a unit that we want to query for the uses of the Local it (may) define.
144     *  @return a UnitValueBoxPair of the Units that use the Local.
145     */
146    @Override
147    public List<UnitValueBoxPair> getUsesOf(Unit s)
148    {
149        List<UnitValueBoxPair> l = unitToUses.get(s);
150        if (l == null)
151                return Collections.emptyList();
152                
153        return Collections.unmodifiableList(l);
154    }
155    
156    /**
157     * Gets all variables that are used in this body
158     * @return The list of variables used in this body
159     */
160    public Set<Local> getUsedVariables() {
161        Set<Local> res = new HashSet<Local>();
162        for (List<UnitValueBoxPair> vals : unitToUses.values())
163                for (UnitValueBoxPair val : vals)
164                        res.add((Local) val.valueBox.getValue());
165        return res;
166    }
167    
168    /**
169     * Gets all variables that are not used in this body
170     * @return The list of variables declared, but not used in this body
171     */
172    public Set<Local> getUnusedVariables() {
173        Set<Local> res = new HashSet<Local>(body.getLocals());
174        res.retainAll(getUsedVariables());
175        return res;
176    }
