051/**
052 * An abstract class providing a framework for carrying out dataflow analysis.
053 * Subclassing either BackwardFlowAnalysis or ForwardFlowAnalysis and providing
054 * implementations for the abstract methods will allow Soot to compute the
055 * corresponding flow analysis.
056 */
057public abstract class FlowAnalysis<N, A> extends AbstractFlowAnalysis<N, A> {
058        public enum Flow {
059                IN {
060                        @Override
061                        <F> F getFlow(Entry<?, F> e) {
062                                return e.inFlow;
063                        }
064                },
065                OUT {
066                        @Override
067                        <F> F getFlow(Entry<?, F> e) {
068                                return e.outFlow;
069                        }

072                abstract <F> F getFlow(Entry<?, F> e);
073        }

077                int number;

079                /**

117                /**
118                 * Creates a new {@code Entry} graph based on a {@code DirectedGraph}. This includes
119                 * pseudo topological order, local access for predecessors and successors, a graph
120                 * entry-point, a {@code Numberable} interface and a real strongly connected component marker.
121                 * @param g
122                 * @param gv
123                 * @param entryFlow
124                 * @return
125                 */
126                <D, F> List<Entry<D, F>> newUniverse (DirectedGraph<D> g, GraphView gv, F entryFlow, boolean isForward) {
127                        final int n = g.size();
128
129                        Deque<Entry<D, F>> s = new ArrayDeque<Entry<D, F>>(n);
130                        List<Entry<D, F>> universe = new ArrayList<Entry<D, F>>(n);
131                        Map<D, Entry<D, F>> visited = new HashMap<D, Entry<D, F>>(((n+1) * 4) / 3);
132
133                        // out of universe node
134                        Entry<D, F> superEntry = new Entry<D, F>(null, null);
135
136                        List<D> entries = null;
137                        List<D> actualEntries = gv.getEntries(g);
138
139                        if (!actualEntries.isEmpty()) {
140                                // normal cases: there is at least 
141                                // one return statement for a backward analysis
142                                // or one entry statement for a forward analysis
143                                entries = actualEntries;
144                        } else {
145                                // cases without any entry statement
146
147                                if (isForward) {
148                                        // case of a forward flow analysis on 
149                                        // a method without any entry point
150                                        throw new RuntimeException("error: no entry point for method in forward analysis");
151                                } else {
152                                        // case of backward analysis on 
153                                        // a method which potentially has 
154                                        // an infinite loop and no return statement
155                                        entries = new ArrayList<D>();
156
157                                        // a single head is expected
158                                        assert g.getHeads().size() == 1;
159                                        D head = g.getHeads().get(0);
160
161                                        Set<D> visitedNodes = new HashSet<D>();
162                                        List<D> workList = new ArrayList<D>();
163                                        D current = null;
164
165                                        // collect all 'goto' statements to catch the 'goto'
166                                        // from the infinite loop
167                                        workList.add(head);
168                                        while (!workList.isEmpty()) {
169                                                current = workList.remove(0);
170                                                visitedNodes.add(current);
171
172                                                // only add 'goto' statements
173                                                if (current instanceof GotoInst || current instanceof GotoStmt) {
174                                                        entries.add(current);
175                                                }
176
177                                                for (D next: g.getSuccsOf(current)) {
178                                                        if (visitedNodes.contains(next)) {
179                                                                continue;
180                                                        }
181                                                        workList.add(next);
182                                                }
183                                        }
184
185                                        //
186                                        if (entries.isEmpty()) {
187                                                throw new RuntimeException("error: backward analysis on an empty entry set.");
188                                        }
189
190                                }
191
192                        }
193
194                        visitEntry(visited, superEntry, entries);
195                        superEntry.inFlow = entryFlow;
196                        superEntry.outFlow = entryFlow;
197
198                        
199                        @SuppressWarnings("unchecked")
200                        Entry<D, F>[] sv = new Entry[g.size()];
201                        int[] si = new int[g.size()];
202                        int index = 0;
203                        
204                        int i = 0;
205                        Entry<D, F> v = superEntry;
206                
207                        for (;;) {
208                                if (i < v.out.length) {
209                                        Entry<D, F> w = v.out[i++];
210                                        
211                                        // an unvisited child node
212                                        if (w.number == Integer.MIN_VALUE) {
213                                                w.number = s.size();
214                                                s.add(w);
215                                                
216                                                visitEntry(visited, w, gv.getOut(g, w.data));
217                                                
218                                                // save old
219                                                si[index] = i;
220                                                sv[index] = v;
221                                                index++;
222                                                
223                                                i = 0;
224                                                v = w;
225                                        }
226                                } else {
227                                        if (index == 0) {
228                                                assert universe.size() <= g.size();
229                                                Collections.reverse(universe);
230                                                return universe;
231                                        }
232                                        
233                                        universe.add(v);
234                                        sccPop(s, v);
235                                        
236                                        // restore old
237                                        index--;
238                                        v = sv[index];
239                                        i = si[index];
240                                }
241                        }
242                }

245                @SuppressWarnings("unchecked")
246                private <D, F> Entry<D, F>[] visitEntry(Map<D, Entry<D, F>> visited, Entry<D, F> v, List<D> out) {
247                        int n = out.size();
248                        Entry<D, F>[] a = new Entry[n];
249
250                        assert (out instanceof RandomAccess);
251                        
252                        for (int i = 0; i < n; i++) {
253                                a[i] = getEntryOf(visited, out.get(i), v);
254                        }
255
256                        return v.out = a;
257                }
258
259                private <D, F> Entry<D, F> getEntryOf(Map<D, Entry<D, F>> visited, D d, Entry<D, F> v) {
260                        // either we reach a new node or a merge node, the latter one is rare
261                        // so put and restore should be better that a lookup
262                        // putIfAbsent would be the ideal strategy
263                        
264                        // add and restore if required
265                        Entry<D, F> newEntry = new Entry<D, F>(d, v);
266                        Entry<D, F> oldEntry = visited.put(d, newEntry);
267                        
268                        // no restore required
269                        if (oldEntry == null)
270                                return newEntry;
271                        
272                        // false prediction, restore the entry
273                        visited.put(d, oldEntry);
274                        
275                        // adding self ref (real strongly connected with itself)
276                        if (oldEntry == v)
277                                oldEntry.isRealStronglyConnected = true;
278                        
279                        // merge nodes are rare, so this is ok
280                        int l = oldEntry.in.length;
281                        oldEntry.in = Arrays.copyOf(oldEntry.in, l + 1);
282                        oldEntry.in[l] = v;             
283                        return oldEntry;
284                }

286                private <D, F> void sccPop(Deque<Entry<D, F>> s, Entry<D, F> v) {
287                        int min = v.number;
288                        for (Entry<D, F> e : v.out) {
289                                assert e.number > Integer.MIN_VALUE;
290                                min = Math.min(min, e.number);
291                        }
292                        
293                        // not our SCC
294                        if (min != v.number) {
295                                v.number = min;
296                                return;
297                        }
298                        
299                        // we only want real SCCs (size > 1)
300                        Entry<D, F> w = s.removeLast();
301                        w.number = Integer.MAX_VALUE;
302                        if (w == v) {
303                                return;
304                        }
305                        
306                        w.isRealStronglyConnected = true;
307                        for (;;) {
308                                w = s.removeLast();
309                                assert w.number >= v.number;
310                                w.isRealStronglyConnected = true;
311                                w.number = Integer.MAX_VALUE;
312                                if (w == v) {
313                                        assert w.in.length >= 2;
314                                        return;
315                                }
316                        }
317                }
318        }

320        enum InteractionFlowHandler {
321                NONE, 
322                FORWARD {
323                        @Override
324                        public <A, N> void handleFlowIn(FlowAnalysis<N, A> a, N s) {
325                                beforeEvent(stop(s), a, s);
326                        }
327
328                        @Override
329                        public <A, N> void handleFlowOut(FlowAnalysis<N, A> a, N s) {
330                                afterEvent(InteractionHandler.v(), a, s);
331                        }
332                },
333                BACKWARD {
334                        @Override
335                        public <A, N> void handleFlowIn(FlowAnalysis<N, A> a, N s) {
336                                afterEvent(stop(s), a, s);
337                        }
338
339                        @Override
340                        public <A, N> void handleFlowOut(FlowAnalysis<N, A> a, N s) {
341                                beforeEvent(InteractionHandler.v(), a, s);
342                        }
343                };
344
345                <A, N> void beforeEvent(InteractionHandler i, FlowAnalysis<N, A> a, N s) {
346                        A savedFlow = a.filterUnitToBeforeFlow.get(s);
347                        if (savedFlow == null)
348                                savedFlow = a.newInitialFlow();
349                        a.copy(a.unitToBeforeFlow.get(s), savedFlow);
350                        i.handleBeforeAnalysisEvent(new FlowInfo<A, N>(savedFlow, s, true));
351                }
352
353                <A, N> void afterEvent(InteractionHandler i, FlowAnalysis<N, A> a, N s) {
354                        A savedFlow = a.filterUnitToAfterFlow.get(s);
355                        if (savedFlow == null)
356                                savedFlow = a.newInitialFlow();
357                        a.copy(a.unitToAfterFlow.get(s), savedFlow);
358                        i.handleAfterAnalysisEvent(new FlowInfo<A, N>(savedFlow, s, false));
359                }
360
361                InteractionHandler stop(Object s) {
362                        InteractionHandler h = InteractionHandler.v();
363                        List<?> stopList = h.getStopUnitList();
364                        if (stopList != null && stopList.contains(s)) {
365                                h.handleStopAtNodeEvent(s);
366                        }
367                        return h;
368                }
369
370                public <A, N> void handleFlowIn(FlowAnalysis<N, A> a, N s) {}
371                public <A, N> void handleFlowOut(FlowAnalysis<N, A> a, N s) {}
372        }

374        enum GraphView {
375                BACKWARD {
376                        @Override
377                        <N> List<N> getEntries(DirectedGraph<N> g) {
378                                return g.getTails();
379                        }
380
381                        @Override
382                        <N> List<N> getOut(DirectedGraph<N> g, N s) {
383                                return g.getPredsOf(s);
384                        }
385                },
386                FORWARD {
387                        @Override
388                        <N> List<N> getEntries(DirectedGraph<N> g) {
389                                return g.getHeads();
390                        }
391
392                        @Override
393                        <N> List<N> getOut(DirectedGraph<N> g, N s) {
394                                return g.getSuccsOf(s);
395                        }
396                };

398                abstract <N> List<N> getEntries(DirectedGraph<N> g);
399                abstract <N> List<N> getOut(DirectedGraph<N> g, N s);
400        }

415        /**
416         * Given the merge of the <code>out</code> sets, compute the <code>in</code>
417         * set for <code>s</code> (or in to out, depending on direction).
418         *
419         * This function often causes confusion, because the same interface is used
420         * for both forward and backward flow analyses. The first parameter is
421         * always the argument to the flow function (i.e. it is the "in" set in a
422         * forward analysis and the "out" set in a backward analysis), and the third
423         * parameter is always the result of the flow function (i.e. it is the "out"
424         * set in a forward analysis and the "in" set in a backward analysis).
425         *
426         * @param in
427         *            the input flow
428         * @param d
429         *            the current node
430         * @param out
431         *            the returned flow
432         **/
433        protected abstract void flowThrough(A in, N d, A out);

438                A a = unitToAfterFlow.get(s);
439                return a == null ? newInitialFlow() : a;

444                A a = unitToBeforeFlow.get(s);

448        private void initFlow(Iterable<Entry<N, A>> universe, Map<N, A> in, Map<N, A> out) {
449                assert universe != null;
450                assert in != null;
451                assert out != null;
452
453                // If a node has only a single in-flow, the in-flow is always equal
454                // to the out-flow if its predecessor, so we use the same object.
455                // this saves memory and requires less object creation and copy calls.
456
457                // Furthermore a node can be marked as omissible, this allows us to use
458                // the same "flow-set" for out-flow and in-flow. A merge node with within
459                // a real scc cannot be omitted, as it could cause endless loops within
460                // the fixpoint-iteration!
461
462                for (Entry<N, A> n : universe) {
463                        boolean omit = true;
464                        if (n.in.length > 1) {
465                                n.inFlow = newInitialFlow();
466                                
467                                // no merge points in loops
468                                omit = !n.isRealStronglyConnected;
469                        } else {
470                                assert n.in.length == 1 : "missing superhead";
471                                n.inFlow = getFlow(n.in[0], n);
472                                assert n.inFlow != null : "topological order is broken";
473                        }
474
475                        if (omit && omissible(n.data)) {
476                                // We could recalculate the graph itself but thats more expensive than
477                                // just falling through such nodes.
478                                n.outFlow = n.inFlow;
479                        } else {
480                                n.outFlow = newInitialFlow();
481                        }
482
483                        // for legacy api
484                        in.put(n.data, n.inFlow);
485                        out.put(n.data, n.outFlow);
486                }
487        }

489        /**
490         * If a flow node can be omitted return <code>true</code>, otherwise
491         * <code>false</code>. There is no guarantee a node will be omitted. A
492         * omissible node does not influence the result of an analysis.
493         * 
494         * If you are unsure, don't overwrite this method
495         * 
496         * @param n the node to check
497         * @return <code>false</code>
498         */
499        protected boolean omissible(N n) {
500                return false;
501        }

513        private A getFlow(Entry<N, A> o, Entry<N, A> e) {
514                return (o.inFlow == o.outFlow) ? o.outFlow : getFlow(o.data, e.data).getFlow(o);
515        }

517        private void meetFlows(Entry<N, A> e) {
518                assert e.in.length >= 1;
519
520                if (e.in.length > 1) {
521                        boolean copy = true;
522                        for (Entry<N, A> o : e.in) {
523                                A flow = getFlow(o, e);
524                                if (copy) {
525                                        copy = false;
526                                        copy(flow, e.inFlow);
527                                } else {
528                                        mergeInto(e.data, e.inFlow, flow);
529                                }
530                        }
531                }
532        }

534        final int doAnalysis(GraphView gv, InteractionFlowHandler ifh, Map<N, A> inFlow, Map<N, A> outFlow) {
535                assert gv != null;
536                assert ifh != null;
537
538                ifh = Options.v().interactive_mode() ? ifh : InteractionFlowHandler.NONE;
539
540                final List<Entry<N, A>> universe = Orderer.INSTANCE.newUniverse(graph, gv, entryInitialFlow(), isForward());
541                initFlow(universe, inFlow, outFlow);
542
543                Queue<Entry<N, A>> q = PriorityQueue.of(universe, true);
544
545                // Perform fixed point flow analysis
546                for (int numComputations = 0;; numComputations++) {
547                        Entry<N, A> e = q.poll();
548                        if (e == null)
549                                return numComputations;
550
551                        meetFlows(e);
552
553                        // Compute beforeFlow and store it.
554                        ifh.handleFlowIn(this, e.data);
555                        boolean hasChanged = flowThrough(e);
556                        ifh.handleFlowOut(this, e.data);
557
558                        // Update queue appropriately
559                        if (hasChanged) {
560                                q.addAll(Arrays.asList(e.out));
561                        }
562                }
563        }
564
565        private boolean flowThrough(Entry<N, A> d) {
566                // omitted, just fall through
567                if (d.inFlow == d.outFlow) {
568                        assert !d.isRealStronglyConnected || d.in.length == 1;
569                        return true;
570                }
571                
572                if (d.isRealStronglyConnected) {
573                        // A flow node that is influenced by at least one back-reference.
574                        // It's essential to check if "flowThrough" changes the result.
575                        // This requires the calculation of "equals", which itself
576                        // can be really expensive - depending on the used flow-model.
577                        // Depending on the "merge"+"flowThrough" costs, it can be cheaper
578                        // to fall through. Only nodes with real back-references always 
579                        // need to be checked for changes
580                        A out = newInitialFlow();
581                        flowThrough(d.inFlow, d.data, out);
582                        if (out.equals(d.outFlow)) {
583                                return false;
584                        }
585                        // copy back the result, as it has changed
586                        copy(out, d.outFlow);
587                        return true;
588                }
589
590                // no back-references, just calculate "flowThrough"
591                flowThrough(d.inFlow, d.data, d.outFlow);
592                return true;
593        }
