020package soot.toolkits.scalar;
021
022import soot.PointsToAnalysis;
023import soot.SootMethod;
024
025/** Just a pair of arbitrary objects.
026 * 
027 * @author Ondrej Lhotak
028 * @author Manu Sridharan (genericized it)
029 * @author xiao, extend it with more functions
030 */
031public class Pair<T, U>
032{
033        public Pair() { o1 = null; o2 = null; }
034    public Pair( T o1, U o2 ) { this.o1 = o1; this.o2 = o2; }
035    
036    @Override
037        public int hashCode() {
038                final int prime = 31;
039                int result = 1;
040                result = prime * result + ((o1 == null) ? 0 : o1.hashCode());
041                result = prime * result + ((o2 == null) ? 0 : o2.hashCode());
042                return result;
043        }
044    
045    @Override
046        public boolean equals(Object obj) {
047                if (this == obj)
048                        return true;
049                if (obj == null)
050                        return false;
051                if (getClass() != obj.getClass())
052                        return false;
053                @SuppressWarnings("rawtypes")
054                Pair other = (Pair) obj;
055                if (o1 == null) {
056                        if (other.o1 != null)
057                                return false;
058                } else if (!o1.equals(other.o1))
059                        return false;
060                if (o2 == null) {
061                        if (other.o2 != null)
062                                return false;
063                } else if (!o2.equals(other.o2))
064                        return false;
065                return true;
066        }
067
068        /**
069         * Decide if this pair represents a method parameter.
070     */
071        public boolean isParameter()
072        {
073                if ( o1 instanceof SootMethod && o2 instanceof Integer ) return true;
074                return false;
075        }
076        
077        /**
078         * Decide if this pair stores the THIS parameter for a method.
079         */
080        public boolean isThisParameter()
081        {
082                return (o1 instanceof SootMethod &&
083                                o2.equals(PointsToAnalysis.THIS_NODE)) ? true : false;
084        }
085
086    public String toString() {
087        return "Pair "+o1+","+o2;
088    }
089    public T getO1() { return o1; }
090    public U getO2() { return o2; }
091        public void setO1( T no1 ) { o1 = no1; }
092        public void setO2( U no2 ) { o2 = no2; }
093        public void setPair( T no1, U no2 ) { o1 = no1; o2 = no2; }
094    protected T o1;
095    protected U o2;
096}
