026package soot.toolkits.scalar;
027
028import java.util.Arrays;
029import java.util.HashSet;
030import java.util.Iterator;
031import java.util.List;
032
033/**
034 * Reference implementation for a FlowSet. Items are stored in an Array.
035 */
036public class ArraySparseSet<T> extends AbstractFlowSet<T> {
037        protected static final int DEFAULT_SIZE = 8;
038
039        protected int numElements;
040        protected int maxElements;
041        protected T[] elements;
042
043        @SuppressWarnings("unchecked")
044        public ArraySparseSet() {
045                maxElements = DEFAULT_SIZE;
046                elements = (T[]) new Object[DEFAULT_SIZE];
047                numElements = 0;
048        }
049
050        private ArraySparseSet(ArraySparseSet<T> other) {
051                numElements = other.numElements;
052                maxElements = other.maxElements;
053                elements = other.elements.clone();
054        }
055
056        /** Returns true if flowSet is the same type of flow set as this. */
057        private boolean sameType(Object flowSet) {
058                return (flowSet instanceof ArraySparseSet);
059        }
060
061        public ArraySparseSet<T> clone() {
062                return new ArraySparseSet<T>(this);
063        }
064
065        public FlowSet<T> emptySet() {
066                return new ArraySparseSet<T>();
067        }
068
069        public void clear() {
070                numElements = 0;
071                Arrays.fill(elements, null);
072        }
073
074        public int size() {
075                return numElements;
076        }
077
078        public boolean isEmpty() {
079                return numElements == 0;
080        }
081
082        /** Returns a unbacked list of elements in this set. */
083        public List<T> toList() {        
084        return Arrays.asList(Arrays.copyOf(elements, numElements));
085        }
086
087        /*
088         * Expand array only when necessary, pointed out by Florian Loitsch March
089         * 08, 2002
090         */
091        public void add(T e) {
092                /* Expand only if necessary! and removes one if too:) */
093                // Add element
094                if (!contains(e)) {
095                        // Expand array if necessary
096                        if (numElements == maxElements)
097                                doubleCapacity();
098                        elements[numElements++] = e;
099                }
100        }
101
102        private void doubleCapacity() {
103                int newSize = maxElements * 2;
104
105                @SuppressWarnings("unchecked")
106                T[] newElements = (T[]) new Object[newSize];
107
108                System.arraycopy(elements, 0, newElements, 0, numElements);
109                elements = newElements;
110                maxElements = newSize;
111        }
112
113        public void remove(Object obj) {
114                for (int i = 0; i < numElements; i++)
115                        if (elements[i].equals(obj)) {
116                                remove(i);
117                                break;
118                        }
119        }
120
121        public void remove(int idx) {
122                numElements--;
123                // copy last element to deleted position
124                elements[idx] = elements[numElements];
125                // delete reference in last cell so that
126                // we only retain a single reference to the
127                // "old last" element, for memory safety
128                elements[numElements] = null;
129                return;
130        }
131
132        public void union(FlowSet<T> otherFlow, FlowSet<T> destFlow) {
133                if (sameType(otherFlow) && sameType(destFlow)) {
134                        ArraySparseSet<T> other = (ArraySparseSet<T>) otherFlow;
135                        ArraySparseSet<T> dest = (ArraySparseSet<T>) destFlow;
136
137                        // For the special case that dest == other
138                        if (dest == other) {
139                                for (int i = 0; i < this.numElements; i++)
140                                        dest.add(this.elements[i]);
141                        }
142
143                        // Else, force that dest starts with contents of this
144                        else {
145                                if (this != dest)
146                                        copy(dest);
147
148                                for (int i = 0; i < other.numElements; i++)
149                                        dest.add(other.elements[i]);
150                        }
151                } else
152                        super.union(otherFlow, destFlow);
153        }
154
155        public void intersection(FlowSet<T> otherFlow, FlowSet<T> destFlow) {
156                if (sameType(otherFlow) && sameType(destFlow)) {
157                        ArraySparseSet<T> other = (ArraySparseSet<T>) otherFlow;
158                        ArraySparseSet<T> dest = (ArraySparseSet<T>) destFlow;
159                        ArraySparseSet<T> workingSet;
160
161                        if (dest == other || dest == this)
162                                workingSet = new ArraySparseSet<T>();
163                        else {
164                                workingSet = dest;
165                                workingSet.clear();
166                        }
167
168                        for (int i = 0; i < this.numElements; i++) {
169                                if (other.contains(this.elements[i]))
170                                        workingSet.add(this.elements[i]);
171                        }
172
173                        if (workingSet != dest)
174                                workingSet.copy(dest);
175                } else
176                        super.intersection(otherFlow, destFlow);
177        }
178
179        public void difference(FlowSet<T> otherFlow, FlowSet<T> destFlow) {
180                if (sameType(otherFlow) && sameType(destFlow)) {
181                        ArraySparseSet<T> other = (ArraySparseSet<T>) otherFlow;
182                        ArraySparseSet<T> dest = (ArraySparseSet<T>) destFlow;
183                        ArraySparseSet<T> workingSet;
184
185                        if (dest == other || dest == this)
186                                workingSet = new ArraySparseSet<T>();
187                        else {
188                                workingSet = dest;
189                                workingSet.clear();
190                        }
191
192                        for (int i = 0; i < this.numElements; i++) {
193                                if (!other.contains(this.elements[i]))
194                                        workingSet.add(this.elements[i]);
195                        }
196
197                        if (workingSet != dest)
198                                workingSet.copy(dest);
199                } else
200                        super.difference(otherFlow, destFlow);
201        }
202
203        /**
204         * @deprecated This method uses linear-time lookup. For better performance,
205         *             consider using a {@link HashSet} instead, if you require this
206         *             operation.
207         */
208        @Deprecated
209        public boolean contains(Object obj) {
210                for (int i = 0; i < numElements; i++)
211                        if (elements[i].equals(obj))
212                                return true;
213
214                return false;
215        }
216
217        @SuppressWarnings("unchecked")
218        public boolean equals(Object otherFlow) {
219                if (sameType(otherFlow)) {
220                        ArraySparseSet<T> other = (ArraySparseSet<T>) otherFlow;
221
222                        if (other.numElements != this.numElements)
223                                return false;
224
225                        int size = this.numElements;
226
227                        // Make sure that thisFlow is contained in otherFlow
228                        for (int i = 0; i < size; i++)
229                                if (!other.contains(this.elements[i]))
230                                        return false;
231
232                        /*
233                         * both arrays have the same size, no element appears twice in one
234                         * array, all elements of ThisFlow are in otherFlow -> they are
235                         * equal! we don't need to test again! // Make sure that otherFlow
236                         * is contained in ThisFlow for(int i = 0; i < size; i++)
237                         * if(!this.contains(other.elements[i])) return false;
238                         */
239
240                        return true;
241                } else
242                        return super.equals(otherFlow);
243        }
244
245        public void copy(FlowSet<T> destFlow) {
246                if (sameType(destFlow)) {
247                        ArraySparseSet<T> dest = (ArraySparseSet<T>) destFlow;
248
249                        while (dest.maxElements < this.maxElements)
250                                dest.doubleCapacity();
251
252                        dest.numElements = this.numElements;
253
254                        System.arraycopy(this.elements, 0, dest.elements, 0,
255                                        this.numElements);
256                } else
257                        super.copy(destFlow);
258        }
259
260        @Override
261        public Iterator<T> iterator() {
262                return new Iterator<T>() {
263                        
264                        int lastIdx = 0;
265                        
266                        @Override
267                        public boolean hasNext() {
268                                return lastIdx < numElements;
269                        }
270
271                        @Override
272                        public T next() {
273                                return elements[lastIdx++];
274                        }
275
276                        @Override
277                        public void remove() {
278                                ArraySparseSet.this.remove(lastIdx);
279                                lastIdx--;
280                        }
281                        
282                };
283        }
