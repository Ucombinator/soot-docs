027package soot.toolkits.scalar;
028
029import java.util.*;
030
031/**
032 * gives an injection of Objects to ints. Different instances of
033 * <code>ObjectIntMap</code> may map different ints to the same object.
034 */
035public class ObjectIntMapper<E> {
036  private Vector<E> intToObjects;
037  private int counter;
038  private Map<E, Integer> objectToInts;
039  
040  public ObjectIntMapper() {
041    intToObjects = new Vector<E>();
042    objectToInts = new HashMap<E, Integer>();
043    counter = 0;
044  }
045
046  public ObjectIntMapper(FlowUniverse<E> flowUniverse) {
047    this(flowUniverse.iterator(), flowUniverse.size());
048  }
049
050  public ObjectIntMapper(Collection<E> collection) {
051    this(collection.iterator(), collection.size());
052  }
053
054  private ObjectIntMapper(Iterator<E> it, int initSize) {
055    intToObjects = new Vector<E>(initSize);
056    objectToInts = new HashMap<E, Integer>(initSize);
057    counter = 0;
058    while (it.hasNext())
059      add(it.next());
060  }
061
062  /**
063   * adds <code>o</code> into the map. no test are made, if it is already in the
064   * map.
065   */
066  public int add(E o) {
067    objectToInts.put(o, counter);
068    intToObjects.add(o);
069    return counter++;
070  }
071
072  /**
073   * returns the mapping of <code>o</code>. if there has been a call to
074   * <code>objectToInt</code> with the same <code>o</code> before, the same
075   * value will be returned.
076   *
077   * @param o
078   * @return <code>o</code>'s mapping
079   */
080  public int getInt(E o) {
081    Integer i = objectToInts.get(o);
082    if (i != null) return i;
083    return add(o);
084  }
085
086  /**
087   * returns the object associated to <code>i</code>.
088   *
089   * @param i
090   * @return <code>i</code>'s object
091   */
092  public E getObject(int i) {
093    return intToObjects.get(i);
094  }
095
096  /**
097   * returns true, if <code>o</code> has already been mapped.
098   *
099   * @param o
100   * @return true if <code>o</code> has already a number.
101   */
102  public boolean contains(Object o) {
103    return objectToInts.containsKey(o);
104  }
105
106  /**
107   * returns the number of mapped objects.
108   */
109  public int size() {
110    return counter;
111  }
112
113}
