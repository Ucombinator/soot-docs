026package soot.toolkits.scalar;
027
028import java.util.Iterator;
029import java.util.List;
030
031/**
032 * provides functional code for most of the methods. Subclasses are invited to
033 * provide a more efficient version. Most often this will be done in the
034 * following way:<br>
035 * 
036 * <pre>
037 * public void yyy(FlowSet dest) {
038 *   if (dest instanceof xxx) {
039 *     blahblah;
040 *   } else
041 *     super.yyy(dest)
042 * }
043 * </pre>
044 */
045public abstract class AbstractFlowSet<T> implements FlowSet<T> {
046        public abstract AbstractFlowSet<T> clone();
047
048        /**
049         * implemented, but inefficient.
050         */
051        public FlowSet<T> emptySet() {
052                FlowSet<T> t = clone();
053                t.clear();
054                return t;
055        }
056
057        public void copy(FlowSet<T> dest) {
058                if (this == dest)
059                        return;
060                dest.clear();
061                for (T t : this)
062                        dest.add(t);
063        }
064
065        /**
066         * implemented, but *very* inefficient.
067         */
068        public void clear() {
069                for (T t : this)
070                        remove(t);
071        }
072
073        public void union(FlowSet<T> other) {
074                if (this == other)
075                        return;
076                union(other, this);
077        }
078
079        public void union(FlowSet<T> other, FlowSet<T> dest) {
080                if (dest != this && dest != other)
081                        dest.clear();
082
083                if (dest != null && dest != this) {
084                        for (T t : this)
085                                dest.add(t);
086                }
087
088                if (other != null && dest != other) {
089                        for (T t : other)
090                                dest.add(t);
091                }
092        }
093
094        public void intersection(FlowSet<T> other) {
095                if (this == other)
096                        return;
097                intersection(other, this);
098        }
099
100        public void intersection(FlowSet<T> other, FlowSet<T> dest) {
101                if (dest == this && dest == other)
102                        return;
103                FlowSet<T> elements = null;
104                FlowSet<T> flowSet = null;
105                if (dest == this) {
106                        /*
107                         * makes automaticly a copy of <code>this</code>, as it will be
108                         * cleared
109                         */
110                        elements = this;
111                        flowSet = other;
112                } else {
113                        /* makes a copy o <code>other</code>, as it might be cleared */
114                        elements = other;
115                        flowSet = this;
116                }
117                dest.clear();
118                for (T t : elements) {
119                        if (flowSet.contains(t))
120                                dest.add(t);
121                }
122        }
123
124        public void difference(FlowSet<T> other) {
125                difference(other, this);
126        }
127
128        public void difference(FlowSet<T> other, FlowSet<T> dest) {
129                if (dest == this && dest == other) {
130                        dest.clear();
131                        return;
132                }
133
134                FlowSet<T> flowSet = (other == dest) ? other.clone() : other;
135                dest.clear(); // now safe, since we have copies of this & other
136
137                for (T t : this)
138                        if (!flowSet.contains(t))
139                                dest.add(t);
140        }
141
142        public abstract boolean isEmpty();
143
144        public abstract int size();
145
146        public abstract void add(T obj);
147
148        public void add(T obj, FlowSet<T> dest) {
149                if (dest != this)
150                        copy(dest);
151                dest.add(obj);
152        }
153
154        public abstract void remove(T obj);
155
156        public void remove(T obj, FlowSet<T> dest) {
157                if (dest != this)
158                        copy(dest);
159                dest.remove(obj);
160        }
161
162        @Override
163        public boolean isSubSet(FlowSet<T> other) {
164                if (other == this)
165                        return true;
166                
167                for (T t : other) {
168                        if (!contains(t))
169                                return false;
170                }
171                return true;
172        }
173
174        public abstract boolean contains(T obj);
175
176        public abstract Iterator<T> iterator();
177
178        public abstract List<T> toList();
179
180        @SuppressWarnings("unchecked")
181        public boolean equals(Object o) {
182                if (!(o instanceof FlowSet))
183                        return false;
184                FlowSet<T> other = (FlowSet<T>) o;
185                if (size() != other.size())
186                        return false;
187                for (T t : this)
188                        if (!other.contains(t))
189                                return false;
190                return true;
191        }
192
193        public int hashCode() {
194                int result = 1;
195                for (T t : this)
196                        result += t.hashCode();
197                return result;
198        }
199
200        public String toString() {
201                StringBuffer buffer = new StringBuffer("{");
202
203                boolean isFirst = true;
204                for (T t : this) {
205                        if (!isFirst)
206                                buffer.append(", ");
207                        isFirst = false;
208
209                        buffer.append(t);
210                }
211                buffer.append("}");
212                return buffer.toString();
213        }
214}
