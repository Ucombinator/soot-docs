189    /**
190     * If this name is in the set of reserved names, then return a quoted
191     * version of it.  Else pass it through. If the name consists of multiple
192     * parts separated by dots, the individual names are checked as well.
193     */
194    public String quotedNameOf(String s)
195    {
196        // Pre-check: Is there a chance that we need to escape something?
197        // If not, skip the transformation altogether.
198        boolean found = s.contains("-");
199        for (String token : reservedNames)
200                if (s.contains(token)) {
201                        found = true;
202                        break;
203                }
204        if (!found)
205                return s;
206        
207        StringBuilder res = new StringBuilder(s.length());
208        for (String part : s.split("\\.")) {
209                if (res.length() > 0)
210                        res.append('.');
211                if(part.startsWith("-") || reservedNames.contains(part)) {
212                    res.append('\'');
213                    res.append(part);
214                    res.append('\'');
215                }
216                else
217                    res.append(part);
218        }
219        return res.toString();
220    }

222    /**
223     * This method is the inverse of quotedNameOf(). It takes a possible escaped
224     * class and reconstructs the original version of it.
225     * @param s The possibly escaped name
226     * @return The original, non-escaped name
227     */
228    public String unescapeName(String s) {
229        // If the name is not escaped, there is nothing to do here
230        if (!s.contains("'"))
231                return s;
232        
233        StringBuilder res = new StringBuilder(s.length());
234        for (String part : s.split("\\.")) {
235                if (res.length() > 0)
236                        res.append('.');
237                if(part.startsWith("'") && part.endsWith("'")) {
238                    res.append(part.substring(1, part.length() - 1));
239                }
240                else
241                    res.append(part);
242        }
243        return res.toString();
244    }

282    public String getSootClassPath()
283    {
284        if( sootClassPath == null ) {
285            String optionscp = Options.v().soot_classpath();
286            if( optionscp != null && optionscp.length() > 0 )
287                sootClassPath = optionscp;
288            
289                //if no classpath is given on the command line, take the default
290                if( sootClassPath == null || sootClassPath.isEmpty() ) {
291                        sootClassPath = defaultClassPath();
292                } else {
293                        //if one is given...
294                    if(Options.v().prepend_classpath()) {
295                        //if the prepend flag is set, append the default classpath
296                        sootClassPath += File.pathSeparator + defaultClassPath();
297                    } 
298                    //else, leave it as it is
299                }   
300                
301                //add process-dirs
302                List<String> process_dir = Options.v().process_dir();
303                StringBuffer pds = new StringBuffer();
304                for (String path : process_dir) {
305                        if(!sootClassPath.contains(path)) {
306                                pds.append(path);
307                                pds.append(File.pathSeparator);
308                        }
309                        }
310                sootClassPath = pds + sootClassPath;
311        }
312
313
314
315        return sootClassPath;
316    }

318    /**
319     * Returns the max Android API version number available
320     * in directory 'dir'
321     * @param dir
322     * @return
323     */
324    private int getMaxAPIAvailable(String dir) {
325        Integer mapi = this.maxAPIs.get(dir);
326        if (mapi != null)
327                return mapi;
328        
329        File d = new File(dir);
330        if (!d.exists())
331                throw new RuntimeException("The Android platform directory you have"
332                                + "specified (" + dir + ") does not exist. Please check.");
333        
334        File[] files = d.listFiles();
335        if (files == null)
336                return -1;
337        
338        int maxApi = -1;
339        for (File f: files) {
340            String name = f.getName();
341            if (f.isDirectory() && name.startsWith("android-")) {
342                try {
343                int v = Integer.decode(name.split("android-")[1]);
344                if (v > maxApi)
345                    maxApi = v;
346                }
347                catch (NumberFormatException ex) {
348                        // We simply ignore directories that do not follow the
349                        // Android naming structure
350                }
351            }
352        }
353        this.maxAPIs.put(dir, maxApi);
354        return maxApi;
355    }

375        private int getAndroidAPIVersion(String jars, String apk) {
376                // Do we already have an API version?
377                if (androidAPIVersion > 0)
378                        return androidAPIVersion;
379                
380                // get path to appropriate android.jar
381                File jarsF = new File(jars);
382                File apkF = apk == null ? null : new File(apk);
383
384                if (!jarsF.exists())
385                        throw new RuntimeException("file '" + jars + "' does not exist!");
386
387                if (apkF != null && !apkF.exists())
388                        throw new RuntimeException("file '" + apk + "' does not exist!");
389
390                // Use the default if we don't have any other information
391                androidAPIVersion = defaultSdkVersion;
392
393                // Do we have an explicit API version?
394                if (Options.v().android_api_version() > 0)
395                        androidAPIVersion = Options.v().android_api_version();
396                // Look into the manifest file
397                else if (apk != null)
398                        if (apk.toLowerCase().endsWith(".apk"))
399                                androidAPIVersion = getTargetSDKVersion(apk, jars);
400                
401                // If we don't have that API version installed, we take the most recent
402                // one we have
403                final int maxAPI = getMaxAPIAvailable(jars);
404                if (androidAPIVersion > maxAPI)
405                        androidAPIVersion = maxAPI;
406                
407                return androidAPIVersion;
408        }

410        private int getTargetSDKVersion(String apkFile, String platformJARs) {
411                // get AndroidManifest
412                InputStream manifestIS = null;
413                ZipFile archive = null;
414                try {
415                        try {
416                                archive = new ZipFile(apkFile);
417                                for (Enumeration<? extends ZipEntry> entries = archive.entries(); entries.hasMoreElements();) {
418                                        ZipEntry entry = entries.nextElement();
419                                        String entryName = entry.getName();
420                                        // We are dealing with the Android manifest
421                                        if (entryName.equals("AndroidManifest.xml")) {
422                                                manifestIS = archive.getInputStream(entry);
423                                                break;
424                                        }
425                                }
426                        } catch (Exception e) {
427                                throw new RuntimeException("Error when looking for manifest in apk: " + e);
428                        }
429                
430                if (manifestIS == null) {
431                        G.v().out.println("Could not find sdk version in Android manifest! Using default: "+defaultSdkVersion);
432                        return defaultSdkVersion;
433                }
434                
435                // process AndroidManifest.xml
436                int maxAPI = getMaxAPIAvailable(platformJARs);
437                int sdkTargetVersion = -1;
438                int minSdkVersion = -1;
439                int platformBuildVersionCode = -1;
440                try {
441                        AXmlResourceParser parser = new AXmlResourceParser();
442                        parser.open(manifestIS);
443                        int depth = 0;
444                        loop: while (true) {
445                                int type = parser.next();
446                                switch (type) {
447                                        case XmlPullParser.START_DOCUMENT: {
448                                                break;
449                                        }
450                                        case XmlPullParser.END_DOCUMENT:
451                                                break loop;
452                                        case XmlPullParser.START_TAG: {
453                                                depth++;
454                                                String tagName = parser.getName();
455                                                if (depth == 1 && tagName.equals("manifest")) {
456                                                        for (int i = 0; i != parser.getAttributeCount(); ++i) {
457                                                                String attributeName = parser.getAttributeName(i);
458                                                                String attributeValue = AXMLPrinter.getAttributeValue(parser, i);
459                                                                if (attributeName.equals("platformBuildVersionCode")) {
460                                                                        platformBuildVersionCode = Integer.parseInt(attributeValue);
461                                                                }
462                                                        }
463                                                }
464                                                else if (depth == 2 && tagName.equals("uses-sdk")) {
465                                                        for (int i = 0; i != parser.getAttributeCount(); ++i) {
466                                                                String attributeName = parser.getAttributeName(i);
467                                                                String attributeValue = AXMLPrinter.getAttributeValue(parser, i);
468                                                                if (attributeName.equals("targetSdkVersion")) {
469                                                                        sdkTargetVersion = Integer.parseInt(attributeValue);
470                                                                } else if (attributeName.equals("minSdkVersion")) {
471                                                                        minSdkVersion = Integer.parseInt(attributeValue);
472                                                                }
473                                                        }
474                                                }
475                                                break;
476                                        }
477                                        case XmlPullParser.END_TAG:
478                                                depth--;
479                                                break;
480                                        case XmlPullParser.TEXT:
481                                                break;
482                                        }
483                                }
484                        } catch (Exception e) {
485                                e.printStackTrace();
486                        }
487                
488                        int APIVersion = -1;
489                        if (sdkTargetVersion != -1) {
490                            if (sdkTargetVersion > maxAPI
491                                    && minSdkVersion != -1
492                                    && minSdkVersion <= maxAPI) {
493                                G.v().out.println("warning: Android API version '"+ sdkTargetVersion +"' not available, using minApkVersion '"+ minSdkVersion +"' instead");
494                                APIVersion = minSdkVersion;
495                            } else {
496                                APIVersion = sdkTargetVersion;
497                            }
498                        } else if (platformBuildVersionCode != -1) {
499                            if (platformBuildVersionCode > maxAPI
500                                    && minSdkVersion != -1
501                                    && minSdkVersion <= maxAPI) {
502                                G.v().out.println("warning: Android API version '"+ platformBuildVersionCode +"' not available, using minApkVersion '"+ minSdkVersion +"' instead");
503                                APIVersion = minSdkVersion;
504                            } else {
505                                APIVersion = platformBuildVersionCode;
506                            }
507                        } else if (minSdkVersion != -1) {
508                                APIVersion = minSdkVersion;
509                        } else {
510                                G.v().out.println("Could not find sdk version in Android manifest! Using default: "+defaultSdkVersion);
511                                APIVersion = defaultSdkVersion;
512                        }
513                        
514                        if (APIVersion <= 2)
515                                        APIVersion = 3;
516                        return APIVersion;
517                }
518                finally {
519                        if (archive != null)
520                                try {
521                                        archive.close();
522                                } catch (IOException e) {
523                                        throw new RuntimeException("Error when looking for manifest in apk: " + e);
524                                }
525                }
526        }

547    private String defaultAndroidClassPath() {
548                // check that android.jar is not in classpath
549                String androidJars = Options.v().android_jars();
550                String forceAndroidJar = Options.v().force_android_jar();
551                if ((androidJars == null || androidJars.equals(""))
552                                && (forceAndroidJar == null || forceAndroidJar.equals(""))) {
553                        throw new RuntimeException("You are analyzing an Android application but did "
554                                        + "not define android.jar. Options -android-jars or -force-android-jar should be used.");
555                }
556                
557                // Get the platform JAR file. It either directly specified, or
558                // we detect it from the target version of the APK we are
559                // analyzing
560                String jarPath = "";
561                if (forceAndroidJar != null && !forceAndroidJar.isEmpty()) {
562                        jarPath = forceAndroidJar;
563                        
564                        if (Options.v().android_api_version() > 0)
565                                androidAPIVersion = Options.v().android_api_version();
566                        else if (forceAndroidJar.contains("android-")) {
567                                Pattern pt = Pattern.compile("\\" + File.separatorChar + "android-(\\d+)"
568                                                + "\\" + File.separatorChar);
569                                Matcher m = pt.matcher(forceAndroidJar);
570                                if (m.find())
571                                        androidAPIVersion = Integer.valueOf(m.group(1));
572                        }
573                        else
574                                androidAPIVersion = defaultSdkVersion;
575                }
576                else if (androidJars != null && !androidJars.isEmpty()) {
577                        List<String> classPathEntries = new LinkedList<String>(Arrays.asList(
578                                        Options.v().soot_classpath().split(File.pathSeparator)));
579                        classPathEntries.addAll(Options.v().process_dir());
580                        
581                        String targetApk = "";
582                        Set<String> targetDexs = new HashSet<String>();
583                        for (String entry : classPathEntries) {
584                                if(entry.toLowerCase().endsWith(".apk")) {      // on Windows, file names are case-insensitive
585                                        // We cannot have multiple APKs, because this would give us multiple
586                                        // manifests which we do not support right now
587                                        if (targetApk != null && !targetApk.isEmpty())
588                                                throw new RuntimeException("only one Android application can be analyzed when using option -android-jars.");
589                                        targetApk = entry;
590                                }
591                                if(entry.toLowerCase().endsWith(".dex"))        // on Windows, file names are case-insensitive
592                                        targetDexs.add(entry);
593                        }
594                        
595                        // We need at least one file to process
596                        if (targetApk == null || targetApk.isEmpty()) {
597                                if (targetDexs.isEmpty())
598                                        throw new RuntimeException("no apk file given");
599                                jarPath = getAndroidJarPath(androidJars, null);
600                        }
601                        else
602                                jarPath = getAndroidJarPath(androidJars, targetApk);
603                }
604                
605                // We must have a platform JAR file when analyzing Android apps
606                if (jarPath.equals(""))
607                        throw new RuntimeException("android.jar not found.");
608                
609                // Check the platform JAR file
610                File f = new File (jarPath);
611                if (!f.exists())
612                        throw new RuntimeException("file '"+ jarPath +"' does not exist!");
613                else
614                        G.v().out.println("Using '"+ jarPath +"' as android.jar");
615                
616                return jarPath;
617        }

619        private String defaultJavaClassPath() {
620                StringBuilder sb = new StringBuilder();
621        if(System.getProperty("os.name").equals("Mac OS X")) {
622                //in older Mac OS X versions, rt.jar was split into classes.jar and ui.jar
623                sb.append(System.getProperty("java.home"));
624                sb.append(File.separator);
625                sb.append("..");
626                sb.append(File.separator);
627                sb.append("Classes");
628                sb.append(File.separator);
629                sb.append("classes.jar");
630        
631                sb.append(File.pathSeparator);
632                sb.append(System.getProperty("java.home"));
633                sb.append(File.separator);
634                sb.append("..");
635                sb.append(File.separator);
636                sb.append("Classes");
637                sb.append(File.separator);
638                sb.append("ui.jar");
639                sb.append(File.pathSeparator);
640        }
641
642        File rtJar = new File(System.getProperty("java.home") + File.separator + "lib" + File.separator + "rt.jar");
643        if (rtJar.exists() && rtJar.isFile()) {
644            // G.v().out.println("Using JRE runtime: " + rtJar.getAbsolutePath());
645            sb.append(rtJar.getAbsolutePath());
646        } else {
647            // in case we're not in JRE environment, try JDK
648            rtJar = new File(System.getProperty("java.home") + File.separator + "jre" + File.separator + "lib" + File.separator + "rt.jar");
649            if (rtJar.exists() && rtJar.isFile()) {
650                // G.v().out.println("Using JDK runtime: " + rtJar.getAbsolutePath());
651                sb.append(rtJar.getAbsolutePath());
652            } else {
653                // not in JDK either
654                throw new RuntimeException("Error: cannot find rt.jar.");
655            }
656        }
657
658                if(Options.v().whole_program() || Options.v().output_format()==Options.output_format_dava) {
659                        //add jce.jar, which is necessary for whole program mode
660                        //(java.security.Signature from rt.jar import javax.crypto.Cipher from jce.jar                  
661                        sb.append(File.pathSeparator+
662                                System.getProperty("java.home")+File.separator+"lib"+File.separator+"jce.jar");
663                }

678    public void addClass(SootClass c) 
679    {
680        if(c.isInScene())
681            throw new RuntimeException("already managed: "+c.getName());
682
683        if(containsClass(c.getName()))
684            throw new RuntimeException("duplicate class: "+c.getName());
685
686        classes.add(c);
687        c.setLibraryClass();
688
689        nameToClass.put(c.getName(), c.getType());
690        c.getType().setSootClass(c);
691        c.setInScene(true);
692        
693        // Phantom classes are not really part of the hierarchy anyway, so
694        // we can keep the old one
695        if (!c.isPhantom)
696                modifyHierarchy();
697    }

716        modifyHierarchy();

863     * @throws RuntimeException if the Type for this name cannot be found.
864     * Use {@link #getRefTypeUnsafe(String)} to check if type is an registered RefType.
865     */
866    public Type getType(String arg) {
867        String type = arg.replaceAll("([^\\[\\]]*)(.*)", "$1");
868        int arrayCount = arg.contains("[") ? arg.replaceAll("([^\\[\\]]*)(.*)", "$2").length() / 2 : 0;
869        
870        Type result = getRefTypeUnsafe(type);
871        
872        if (result == null) {
873                if (type.equals("long"))
874              result = LongType.v();
875                else if (type.equals("short"))
876              result = ShortType.v();
877                else if (type.equals("double"))
878              result = DoubleType.v();
879                else if (type.equals("int"))
880              result = IntType.v();
881                else if (type.equals("float"))
882              result = FloatType.v();
883                else if (type.equals("byte"))
884              result = ByteType.v();
885                else if (type.equals("char"))
886              result = CharType.v();
887                else if (type.equals("void"))
888              result = VoidType.v();
889                else if (type.equals("boolean"))
890              result = BooleanType.v();
891                else
892              throw new RuntimeException("unknown type: '" + type + "'");
893        }
894        
895        if (arrayCount != 0) {
896                result = ArrayType.v(result, arrayCount);
897        }
898        return result;
899    }

1496        CGOptions options = new CGOptions( PhaseOptions.v().getPhaseOptions("cg") );
1497        String log = options.reflection_log();
1498        
1499        Set<String> classNames = new HashSet<String>();
1500        if(log!=null && log.length()>0) {
1501                        BufferedReader reader = null;
1502                        String line="";
1503                        try {
1504                                reader = new BufferedReader(new InputStreamReader(new FileInputStream(log)));
1505                                while((line=reader.readLine())!=null) {
1506                                        if(line.length()==0) continue;
1507                                        String[] portions = line.split(";",-1);
1508                                        String kind = portions[0];
1509                                        String target = portions[1];
1510                                        String source = portions[2];
1511                                        String sourceClassName = source.substring(0,source.lastIndexOf("."));
1512                                        classNames.add(sourceClassName);
1513                                        if(kind.equals("Class.forName")) {
1514                                                classNames.add(target);
1515                                        } else if(kind.equals("Class.newInstance")) {
1516                                                classNames.add(target);
1517                                        } else if(kind.equals("Method.invoke") || kind.equals("Constructor.newInstance")) {
1518                                                classNames.add(signatureToClass(target));
1519                                        } else if(kind.equals("Field.set*") || kind.equals("Field.get*")) {
1520                                                classNames.add(signatureToClass(target));
1521                                        } else throw new RuntimeException("Unknown entry kind: "+kind);
1522                                }
1523                        } catch (Exception e) {
1524                                throw new RuntimeException("Line: '"+line+"'", e);
1525                        }
1526                        finally {
1527                                if (reader != null)
1528                                        try {
1529                                                reader.close();
1530                                        } catch (IOException e) {
1531                                                throw new RuntimeException(e);
1532                                        }
1533                        }
1534        }
1535        
1536        for (String c : classNames) {
1537                addBasicClass(c, SootClass.BODIES);

1619    /* Generate classes to process, adding or removing package marked by
1620     * command line options. 
1621     */
1622    private void prepareClasses() {
1623        // Remove/add all classes from packageInclusionMask as per -i option
1624        Chain<SootClass> processedClasses = new HashChain<SootClass>();
1625        while(true) {
1626            Chain<SootClass> unprocessedClasses = new HashChain<SootClass>(getClasses());
1627            unprocessedClasses.removeAll(processedClasses);
1628            if( unprocessedClasses.isEmpty() ) break;
1629            processedClasses.addAll(unprocessedClasses);
1630            for (SootClass s : unprocessedClasses) {
1631                if( s.isPhantom() ) continue;
1632                if(Options.v().app()) {
1633                    s.setApplicationClass();
1634                }
1635                if (Options.v().classes().contains(s.getName())) {
1636                    s.setApplicationClass();
1637                    continue;
1638                }
1639                if(s.isApplicationClass() && isExcluded(s)){
1640                    s.setLibraryClass();
1641                }
1642                if(isIncluded(s)){
1643                    s.setApplicationClass();
1644                }
1645                if(s.isApplicationClass()) {
1646                    // make sure we have the support
1647                    loadClassAndSupport(s.getName());
1648                }
1649            }
1650        }
1651    }

1657                        return !isIncluded(sc);

1659        }

1669        }
