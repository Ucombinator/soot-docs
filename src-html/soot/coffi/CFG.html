163    public static HashMap<SootMethod, int[]> methodsToVEM = new HashMap<SootMethod, int[]>();
164    private void complexity() 
165    {
166      // ignore all non-app classes
167      if (!method.jmethod.getDeclaringClass().isApplicationClass()) return;
168      
169      BasicBlock b = this.cfg;
170      HashMap<BasicBlock, Integer> block2exc = new HashMap<BasicBlock, Integer>();
171      int tmp, nodes = 0, edges = 0, highest = 0;
172      
173      while (b != null) {
174        tmp = 0;
175        for (exception_table_entry element : method.code_attr.exception_table) {
176          Instruction start = element.start_inst;
177          Instruction end = element.start_inst;
178          if ((start.label >= b.head.label && start.label <= b.tail.label) ||
179                (end.label > b.head.label && (b.tail.next == null || end.label <= b.tail.next.label)))
180            tmp++;
181        }
182        block2exc.put(b, new Integer(tmp));
183        b = b.next;
184      }
185      
186      b = this.cfg;
187      while ( b!= null )
188      {
189        nodes++;
190        tmp = b.succ.size() + block2exc.get(b).intValue();
191        
192        // exceptions are not counted in succs and preds so we need to do so manually
193        int deg = b.pred.size() + tmp + (b.beginException ? 1 : 0);
194        if (deg > highest)
195          highest = deg;
196        edges += tmp;
197        b = b.next;
198      }
199      methodsToVEM.put(method.jmethod,new int[]{nodes, edges, highest});
200    }

202    // Constructs the actual control flow graph. Assumes the hash table
203    // currently associates leaders with BasicBlocks, this function
204    // builds the next[] and prev[] pointer arrays.
205    private void buildBBCFG() 
206    {
207        Object branches[];
208        Code_attribute ca = method.locate_code_attribute();
209
210        {
211            h2bb = new Hashtable<Instruction, BasicBlock>(100,25);
212            t2bb = new Hashtable<Instruction, BasicBlock>(100,25);
213
214            Instruction insn = this.sentinel.next;
215            BasicBlock blast = null;
216            if (insn != null)
217            {
218                Instruction tail = buildBasicBlock(insn);
219                cfg = new BasicBlock(insn, tail);
220                h2bb.put(insn, cfg);
221                t2bb.put(tail, cfg);
222                insn = tail.next;
223                blast = cfg;
224            }   
225
226            while (insn != null)
227            {
228                Instruction tail = buildBasicBlock(insn);
229                BasicBlock block = new BasicBlock(insn, tail);
230                blast.next = block;
231                blast = block;
232                h2bb.put(insn, block);
233                t2bb.put(tail, block);
234                insn = tail.next;
235            }
236        }
237
238        BasicBlock block = cfg;
239
240        while (block != null) 
241        {
242            Instruction insn = block.tail;
243
244            if (insn.branches) 
245            {
246                if (insn instanceof Instruction_Athrow)
247                {
248                    // see how many targets it can reach.  Note that this is a
249                    // subset of the exception_table.
250                    HashSet<Instruction> ethandlers = new HashSet<Instruction>();
251
252                    // not quite a subset---could also be that control 
253                    // exits this method, so start icount at 1
254                    for (int i=0; i<ca.exception_table_length; i++) 
255                    {
256                        exception_table_entry etentry =
257                            ca.exception_table[i];
258
259                        if (insn.label >= etentry.start_inst.label 
260                            && (etentry.end_inst==null 
261                                || insn.label < etentry.end_inst.label)) 
262                        {
263                            ethandlers.add(etentry.handler_inst);
264                        }
265                    }
266
267                    branches = ethandlers.toArray();
268                } 
269                else
270                {
271                    branches = insn.branchpoints(insn.next);          
272                }
273
274                if (branches != null)
275                {
276                    block.succ.ensureCapacity(block.succ.size()+branches.length);
277
278                    for (Object element : branches) {
279                        if ( element!=null ) {
280                            BasicBlock bb = h2bb.get(element);
281                 
282                            if (bb == null)
283                            {                 
284                                G.v().out.println("Warning: "
285                                               +"target of a branch is null");
286                                G.v().out.println ( insn );
287                            }
288                            else 
289                            {
290                                block.succ.addElement(bb);
291                                bb.pred.addElement(block);
292                            }
293                        }
294                    }
295                }
296            } 
297            else 
298            if (block.next!=null) 
299            { // BB ended not with a branch, so just go to next
300                block.succ.addElement(block.next);
301                block.next.pred.addElement(block);
302            }
303            block = block.next;
304        }
305
306        // One final step, run through exception handlers and mark which
307        // basic blocks begin their code
308        for (int i=0; i<ca.exception_table_length; i++) 
309        {
310            BasicBlock bb = h2bb.get(
311                                         ca.exception_table[i].handler_inst);
312            if ( bb == null )
313            {
314                G.v().out.println("Warning: No basic block found for" +
315                                   " start of exception handler code.");
316            }
317            else 
318            {
319                bb.beginException = true;
320                ca.exception_table[i].b = bb;
321            }
322        }
323    }

325    /* given the list of instructions head, this pulls off the front
326     * basic block, terminates it with a null, and returns the next
327     * instruction after.
328     */
329    private static Instruction buildBasicBlock(Instruction head) 
330    {
331        Instruction insn, next;
332        insn = head;
333        next = insn.next;
334
335        if (next == null)
336            return insn;
337
338        do 
339        {           
340            if (insn.branches || next.labelled)
341                break;
342            else
343            {
344                insn = next;
345                next = insn.next;
346            }
347        } while (next != null);
348
349        return insn;
350    }

358    /* Eliminate subroutines ( JSR/RET instructions ) by inlining the 
359       routine bodies. */
360    private boolean eliminateJsrRets()
361    {
362        Instruction insn = this.sentinel;
363
364        // find the last instruction, for copying blocks.
365        while (insn.next != null) {
366            insn = insn.next;
367        }
368        this.lastInstruction = insn;
369
370        HashMap<Instruction, Instruction> todoBlocks = new HashMap<Instruction, Instruction>();
371        todoBlocks.put(this.sentinel.next, this.lastInstruction);
372        LinkedList<Instruction> todoList = new LinkedList<Instruction>();
373        todoList.add(this.sentinel.next);
374
375        while (!todoList.isEmpty()) {
376            Instruction firstInsn = todoList.removeFirst();
377            Instruction lastInsn  = todoBlocks.get(firstInsn);
378
379            jsrorder.clear();
380            jsr2astore.clear();
381            astore2ret.clear();
382
383            if (findOutmostJsrs(firstInsn, lastInsn)) {
384                HashMap<Instruction, Instruction> newblocks = inliningJsrTargets();
385                todoBlocks.putAll(newblocks);
386                todoList.addAll(newblocks.keySet());
387            }
388        }
389
390        /* patch exception table and others.*/
391        {
392            method.instructions = this.sentinel.next;
393
394            adjustExceptionTable();
395            adjustLineNumberTable();
396            adjustBranchTargets();
397        }
398
399        // we should prune the code and exception table here.
400        // remove any exception handler whose region is in a jsr/ret block.
401        // pruneExceptionTable();
402
403        return true;
404    }

406    // find outmost jsr/ret pairs in a code area, all information is
407    // saved in jsr2astore, and astore2ret
408    // start : start instruction, inclusively.
409    // end   : the last instruction, inclusively.
410    // return the last instruction encounted ( before end )
411    // the caller cleans jsr2astore, astore2ret
412    private boolean findOutmostJsrs(Instruction start, Instruction end) {
413        // use to put innerJsrs.
414        HashSet<Instruction> innerJsrs = new HashSet<Instruction>();
415        boolean unusual = false;
416
417        Instruction insn = start;
418        do {
419            if (insn instanceof Instruction_Jsr
420                || insn instanceof Instruction_Jsr_w)
421            {
422                if (innerJsrs.contains(insn)) {
423                    // skip it
424                    insn = insn.next;
425                    continue;
426                }    
427
428                Instruction astore = ((Instruction_branch)insn).target;
429                if (! (astore instanceof Interface_Astore))
430                {
431                    unusual = true;
432                    break;
433                }
434                
435                Instruction ret = findMatchingRet(astore, insn, innerJsrs);
436
437                /*
438                if (ret == null)
439                {
440                    unusual = true;
441                    break;
442                }
443                */
444
445                jsrorder.addLast(insn);
446                jsr2astore.put(insn, astore);
447                astore2ret.put(astore, ret);
448            }
449 
450            insn = insn.next;
451           
452        } while (insn != end.next);
453
454        if (unusual)
455        {
456            G.v().out.println("Sorry, I cannot handle this method.");
457            return false;
458        }
459        
460        return true;
461    }

463    private Instruction findMatchingRet(Instruction astore, 
464                                        Instruction jsr,
465                                        HashSet<Instruction>     innerJsrs)
466    {
467        int astorenum = ((Interface_Astore)astore).getLocalNumber();
468        
469        Instruction insn = astore.next;
470        while (insn != null)
471        {
472            if (insn instanceof Instruction_Ret
473                || insn instanceof Instruction_Ret_w)
474            {
475                int retnum = ((Interface_OneIntArg)insn).getIntArg();
476                if (astorenum == retnum)
477                    return insn;
478            }
479            else
480            /* adjust the jsr inlining order. */
481            if (insn instanceof Instruction_Jsr
482                || insn instanceof Instruction_Jsr_w)
483            {
484                innerJsrs.add(insn);
485            }
486
487            insn = insn.next;
488        }
489
490        return null;
491    }

543    /* make a copy of code between from and to exclusively, 
544     * fixup targets of branch instructions in the code.
545     */
546    private Instruction makeCopyOf(Instruction astore,
547                                   Instruction ret,
548                                   Instruction target)
549    {
550        // do a quick hacker for ret == null
551        if (ret == null) {
552            return astore.next;
553        }
554
555        Instruction last = this.lastInstruction;
556        Instruction headbefore = last;
557
558        int curlabel = this.lastInstruction.label;
559
560        // mapping from original instructions to new instructions.
561        HashMap<Instruction, Instruction> insnmap = new HashMap<Instruction, Instruction>(); 
562        Instruction insn = astore.next;
563        
564        while (insn != ret && insn != null)
565        {
566            try {
567                Instruction newone = (Instruction)insn.clone();
568
569                newone.label = ++curlabel;
570                newone.prev = last;
571                last.next = newone;
572                last = newone;
573
574                insnmap.put(insn, newone);
575            } catch (CloneNotSupportedException e)
576            {
577                G.v().out.println("Error !");
578            }
579            insn = insn.next;   
580        }
581
582        // replace ret by a goto
583        Instruction_Goto togo = new Instruction_Goto();
584        togo.target = target;
585        target.labelled = true;
586        togo.label = ++curlabel;
587        last.next = togo;
588        togo.prev = last;
589        last = togo;
590
591        this.lastInstruction = last;
592
593        // The ret instruction is removed, 
594        insnmap.put(astore, headbefore.next);
595        insnmap.put(ret, togo);
596
597        // fixup targets in new instruction (only in the scope of 
598        //  new instructions).
599        // do not forget set target labelled as TRUE
600        insn = headbefore.next;
601        while (insn != last)
602        {
603            if (insn instanceof Instruction_branch)
604            {
605                Instruction oldtgt = ((Instruction_branch)insn).target;
606                Instruction newtgt = insnmap.get(oldtgt);
607                if (newtgt != null)
608                {
609                    ((Instruction_branch)insn).target = newtgt;
610                    newtgt.labelled = true;
611                } 
612            }
613            else
614            if (insn instanceof Instruction_Lookupswitch)
615            {
616                Instruction_Lookupswitch switchinsn = 
617                    (Instruction_Lookupswitch)insn;
618                
619                Instruction newdefault = insnmap.get(switchinsn.default_inst);
620                if (newdefault != null)
621                {
622                    switchinsn.default_inst = newdefault;
623                    newdefault.labelled = true;
624                }
625
626                for (int i=0; i<switchinsn.match_insts.length; i++)
627                {
628                    Instruction newtgt = insnmap.get(switchinsn.match_insts[i]);
629                    if (newtgt != null)
630                    {
631                        switchinsn.match_insts[i] = newtgt;
632                        newtgt.labelled = true;
633                    }
634                }
635            }
636            else
637            if (insn instanceof Instruction_Tableswitch)
638            {
639                Instruction_Tableswitch switchinsn = 
640                    (Instruction_Tableswitch)insn;
641                
642                Instruction newdefault = insnmap.get(switchinsn.default_inst);
643                if (newdefault != null)
644                {
645                    switchinsn.default_inst = newdefault;
646                    newdefault.labelled = true;
647                }
648
649                for (int i=0; i<switchinsn.jump_insts.length; i++)
650                {
651                    Instruction newtgt = insnmap.get(switchinsn.jump_insts[i]);
652                    if (newtgt != null)
653                    {
654                        switchinsn.jump_insts[i] = newtgt;
655                        newtgt.labelled = true;
656                    }
657                }
658            }
659
660            insn = insn.next;
661        }
662        
663        // do we need to copy a new exception table entry? 
664        // new exception table has new exception range, 
665        // and the new exception handler.
666        {
667            Code_attribute ca = method.locate_code_attribute();
668
669            LinkedList<exception_table_entry> newentries = new LinkedList<exception_table_entry>();
670
671            int orig_start_of_subr = astore.next.originalIndex; // inclusive
672            int orig_end_of_subr = ret.originalIndex; // again, inclusive
673
674            for (int i=0; i<ca.exception_table_length; i++) 
675            {
676                exception_table_entry etentry =
677                    ca.exception_table[i];
678
679                int orig_start_of_trap = etentry.start_pc; // inclusive
680                int orig_end_of_trap = etentry.end_pc; // exclusive
681                if ( orig_start_of_trap < orig_end_of_subr &&
682                     orig_end_of_trap > orig_start_of_subr) {
683                    // At least a portion of the cloned subroutine is trapped.
684                    exception_table_entry newone = 
685                        new exception_table_entry();
686                    if (orig_start_of_trap <= orig_start_of_subr) {
687                        newone.start_inst = headbefore.next;
688                    } else {
689                        Instruction ins = insnmap.get(etentry.start_inst);
690                        if(ins!=null)
691                                newone.start_inst = insnmap.get(etentry.start_inst);
692                        else 
693                                newone.start_inst = etentry.start_inst;
694                    }
695                    if (orig_end_of_trap > orig_end_of_subr) {
696                        newone.end_inst = null; // Representing the insn after
697                                                // the last instruction in the
698                                                // subr; we need to fix it if
699                                                // we inline another subr.
700                    } else {
701                        newone.end_inst = insnmap.get(etentry.end_inst);
702                    }
703
704                    newone.handler_inst = insnmap.get(etentry.handler_inst);
705                    if (newone.handler_inst == null)
706                        newone.handler_inst = etentry.handler_inst;
707
708                    // We can leave newone.start_pc == 0 and newone.end_pc == 0.
709                    // since that cannot overlap the range of any other
710                    // subroutines that get inlined later.
711
712                    newentries.add(newone);
713                }
714                // Finally, fix up the old entry if its protected area
715                // ran to the end of the method we have just lengthened:
716                // patch its end marker to be the first
717                // instruction in the subroutine we've just inlined.
718                if (etentry.end_inst == null) {
719                    etentry.end_inst = headbefore.next;
720                }
721            }
722
723            if (newentries.size() > 0)
724            {
725                ca.exception_table_length += newentries.size();
726                exception_table_entry[] newtable = new exception_table_entry[ca.exception_table_length];
727                System.arraycopy(ca.exception_table, 0, newtable, 0, ca.exception_table.length);
728                for (int i=0, j=ca.exception_table.length; i<newentries.size(); i++, j++)
729                {
730                    newtable[j] = newentries.get(i);
731                }
732                
733                ca.exception_table = newtable;
734            }
735        }
736        
737        return headbefore.next;
738    }

744    /* do not forget set the target labelled as TRUE.*/
745    private void adjustBranchTargets()
746    {
747        Instruction insn = this.sentinel.next;
748        while (insn != null)
749        {
750            if (insn instanceof Instruction_branch)
751            {
752                Instruction_branch binsn = (Instruction_branch)insn;

773

775                {
776                    Instruction newtgt = 
777                        replacedInsns.get(switchinsn.match_insts[i]);
778                    if (newtgt != null)
779                    {
780                        switchinsn.match_insts[i] = newtgt;

782                    }
783                }
784            }

787            {
788                Instruction_Tableswitch switchinsn = 
789                    (Instruction_Tableswitch)insn;

792                if (newdefault != null)
793                {
794                    switchinsn.default_inst = newdefault;
795                    newdefault.labelled = true;
796                }
797
798                for (int i=0; i<=switchinsn.high-switchinsn.low; i++)
799                {
800                    Instruction newtgt = 
801                        replacedInsns.get(switchinsn.jump_insts[i]);
802                    if (newtgt != null)
803                    {
804                        switchinsn.jump_insts[i] = newtgt;
805                        newtgt.labelled = true;
806                    }
807                }
808            }

1467                while (stmt != null

1469                {
1470                    stmt.addTag(tag);

1472                }
1473            }
1474        }
