019package soot.coffi;
020
021import soot.Value;
022
023/** A constant pool entry of type CONSTANT_MethodHandle
024 * @see cp_info
025 * @author Eric Bodden
026 */class CONSTANT_MethodHandle_info extends cp_info {
027         
028         public int kind;
029         
030         public int target_index;
031
032         public int size() { return 4; }
033         
034           /** Returns a String representation of this entry.
035            * @param constant_pool constant pool of ClassFile.
036            * @return String representation of this entry.
037            * @see cp_info#toString
038            */
039           public String toString(cp_info constant_pool[]) {
040                   //currently neglects field "kind"               
041                   cp_info target = constant_pool[target_index];
042               return target.toString(constant_pool);
043           }
044           /** Returns a String description of what kind of entry this is.
045            * @return the String "methodhandle".
046            * @see cp_info#typeName
047            */
048           public String typeName() { return "methodhandle"; }
049           
050           /** Compares this entry with another cp_info object (which may reside
051            * in a different constant pool).
052            * @param constant_pool constant pool of ClassFile for this.
053            * @param cp constant pool entry to compare against.
054            * @param cp_constant_pool constant pool of ClassFile for cp.
055            * @return a value <0, 0, or >0 indicating whether this is smaller,
056            * the same or larger than cp.
057            * @see cp_info#compareTo
058            */
059           public int compareTo(cp_info constant_pool[],cp_info cp,cp_info cp_constant_pool[]) {
060              int i;
061              if (tag!=cp.tag) return tag-cp.tag;
062              CONSTANT_MethodHandle_info cu = (CONSTANT_MethodHandle_info)cp;
063              i = constant_pool[target_index].
064                 compareTo(constant_pool,cp_constant_pool[cu.target_index],cp_constant_pool);
065              if (i!=0) return i;
066              return kind-cu.kind;
067           }
068
069 
070           public Value createJimpleConstantValue(cp_info[] constant_pool) {
071                   //FIXME may need to determine static-ness based on "kind" field
072                   return constant_pool[target_index].createJimpleConstantValue(constant_pool);
073           }
074}
