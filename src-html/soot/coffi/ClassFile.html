185      
186      try

188        DataInputStream classDataStream = new DataInputStream(classFileStream);
189        data = new byte[classDataStream.available()];
190        classDataStream.readFully(data);
191        f = new ByteArrayInputStream(data);
192         
193      } catch(IOException e)
194      {
195      }
196
197      if (Options.v().time())
198        Timers.v().readTimer.end();
199

204        classFileStream.close();
205        d.close();
206        if (f != null)
207                f.close();
208      } catch(IOException e) {
209         G.v().out.println("IOException with " + fn + ": " + e.getMessage());
210         return false;
211      }
212      
213      if (!b) return false;
214      //parse();        // parse all methods & builds CFGs
215      //G.v().out.println("-- Read " + cf + " --");
216      return true;
217   }

230         f = new FileOutputStream(fn);
231      } catch(FileNotFoundException e) {
232         if (fn.indexOf(".class")>=0) {

235         }

238            f = new FileOutputStream(fn);

240            G.v().out.println("Can't find " + fn);
241            return false;
242         }
243      }
244      d = new DataOutputStream(f);
245      b = writeClass(d);

247         d.close();
248         f.close();
249      } catch(IOException e) {
250         G.v().out.println("IOException with " + fn + ": " + e.getMessage());
251         return false;
252      }
253      return b;
254   }

339   /** Builds the internal representation of this Class by reading in the
340    * given class file.
341    * @param d Stream forming the <tt>.class</tt> file.
342    * @return <i>true</i> if read was successful, <i>false</i> on some error.
343    */
344   public boolean readClass(DataInputStream d) {
345      try {
346         // first read in magic number
347         magic = d.readInt() & 0xFFFFFFFFL;
348         if (magic != MAGIC) {
349            G.v().out.println("Wrong magic number in " + fn + ": " + magic);
350            return false;
351         }
352         minor_version = d.readUnsignedShort();
353         major_version = d.readUnsignedShort();
354         constant_pool_count = d.readUnsignedShort();
355
356         if (!readConstantPool(d))
357            return false;
358
359         access_flags = d.readUnsignedShort();
360         this_class = d.readUnsignedShort();
361         super_class = d.readUnsignedShort();
362         interfaces_count = d.readUnsignedShort();
363         if (interfaces_count>0) {
364            interfaces = new int[interfaces_count];
365            int j;
366            for (j=0; j<interfaces_count; j++)
367               interfaces[j] = d.readUnsignedShort();
368         }
369
370         if (Options.v().time())
371           Timers.v().fieldTimer.start();
372         
373         fields_count = d.readUnsignedShort();
374         readFields(d);
375
376         if (Options.v().time())
377           Timers.v().fieldTimer.end();
378
379         if (Options.v().time())
380           Timers.v().methodTimer.start();
381         methods_count = d.readUnsignedShort();
382         readMethods(d);
383
384         if (Options.v().time())
385           Timers.v().methodTimer.end();
386
387         if (Options.v().time())
388           Timers.v().attributeTimer.start();
389         
390         attributes_count = d.readUnsignedShort();
391         if (attributes_count>0) {
392            attributes =  new attribute_info[attributes_count];
393            readAttributes(d,attributes_count,attributes);
394         }
395         if (Options.v().time())
396           Timers.v().attributeTimer.end();
397         
398      } catch(IOException e) {
399         throw new RuntimeException("IOException with " + fn + ": " + e.getMessage(), e);
400      }
401
402      return true;
403   }

405   /** Reads in the constant pool from the given stream.
406    * @param d Stream forming the <tt>.class</tt> file.
407    * @return <i>true</i> if read was successful, <i>false</i> on some error.
408    * @exception java.io.IOException on error.
409    */
410   protected boolean readConstantPool(DataInputStream d) throws IOException {
411      byte tag;
412      cp_info cp;
413      int i;
414      boolean skipone;   // set if next cp entry is to be skipped
415
416      constant_pool = new cp_info[constant_pool_count];
417      //Instruction.constant_pool = constant_pool;
418      skipone = false;
419
420      for (i=1;i<constant_pool_count;i++) {
421         if (skipone) {
422            skipone = false;
423            continue;
424         }
425         tag = (byte)d.readUnsignedByte();
426         switch(tag) {
427         case cp_info.CONSTANT_Class:
428            cp = new CONSTANT_Class_info();
429            ((CONSTANT_Class_info)cp).name_index = d.readUnsignedShort();
430            if (debug) G.v().out.println("Constant pool[" + i + "]: Class");
431            break;
432         case cp_info.CONSTANT_Fieldref:
433            cp = new CONSTANT_Fieldref_info();
434            ((CONSTANT_Fieldref_info)cp).class_index = d.readUnsignedShort();
435            ((CONSTANT_Fieldref_info)cp).name_and_type_index =
436                d.readUnsignedShort();
437            if (debug) G.v().out.println("Constant pool[" + i + "]: Fieldref");
438            break;
439         case cp_info.CONSTANT_Methodref:
440            cp = new CONSTANT_Methodref_info();
441            ((CONSTANT_Methodref_info)cp).class_index = d.readUnsignedShort();
442            ((CONSTANT_Methodref_info)cp).name_and_type_index =
443               d.readUnsignedShort();
444            if (debug) G.v().out.println("Constant pool[" + i + "]: Methodref");
445            break;
446         case cp_info.CONSTANT_InterfaceMethodref:
447            cp = new CONSTANT_InterfaceMethodref_info();
448            ((CONSTANT_InterfaceMethodref_info)cp).class_index =
449               d.readUnsignedShort();
450            ((CONSTANT_InterfaceMethodref_info)cp).name_and_type_index =
451               d.readUnsignedShort();
452            if (debug)
453               G.v().out.println("Constant pool[" + i + "]: MethodHandle");
454            break;
455         case cp_info.CONSTANT_String:
456            cp = new CONSTANT_String_info();
457            ((CONSTANT_String_info)cp).string_index =
458                d.readUnsignedShort();
459            if (debug) G.v().out.println("Constant pool[" + i + "]: String");
460            break;
461         case cp_info.CONSTANT_Integer:
462            cp = new CONSTANT_Integer_info();
463            ((CONSTANT_Integer_info)cp).bytes = d.readInt();
464            if (debug) G.v().out.println("Constant pool[" + i + "]: Integer = " +
465                                          ((CONSTANT_Integer_info)cp).bytes);
466            break;
467         case cp_info.CONSTANT_Float:
468            cp = new CONSTANT_Float_info();
469            ((CONSTANT_Float_info)cp).bytes = d.readInt();
470            if (debug) G.v().out.println("Constant pool[" + i + "]: Float = " +
471                                          ((CONSTANT_Float_info)cp).convert());
472            break;
473         case cp_info.CONSTANT_Long:
474            cp = new CONSTANT_Long_info();
475            ((CONSTANT_Long_info)cp).high = d.readInt() & 0xFFFFFFFFL;
476            ((CONSTANT_Long_info)cp).low = d.readInt() & 0xFFFFFFFFL;
477            
478            if (debug) {
479               String temp = cp.toString(constant_pool);
480               G.v().out.println("Constant pool[" + i + "]: Long = " + temp);
481               /*G.v().out.println("Constant pool[" + i + "]: that's " +
482                 cp.printBits(((CONSTANT_Long_info)cp).high) + " <<32 + " +
483                 cp.printBits(((CONSTANT_Long_info)cp).low) + " = " +
484                 cp.printBits(((CONSTANT_Long_info)cp).convert()));*/
485            }
486            skipone = true;  // next entry needs to be skipped
487            break;
488         case cp_info.CONSTANT_Double:
489            cp = new CONSTANT_Double_info();
490            ((CONSTANT_Double_info)cp).high = d.readInt() & 0xFFFFFFFFL;
491            ((CONSTANT_Double_info)cp).low = d.readInt() & 0xFFFFFFFFL;
492            if (debug) G.v().out.println("Constant pool[" + i + "]: Double = " +
493                                          ((CONSTANT_Double_info)cp).convert());
494            skipone = true;  // next entry needs to be skipped
495            break;
496         case cp_info.CONSTANT_NameAndType:
497            cp = new CONSTANT_NameAndType_info();
498            ((CONSTANT_NameAndType_info)cp).name_index =
499               d.readUnsignedShort();
500            ((CONSTANT_NameAndType_info)cp).descriptor_index =
501               d.readUnsignedShort();
502            if (debug) G.v().out.println("Constant pool[" + i + "]: Name and Type");
503            break;
504         case cp_info.CONSTANT_Utf8:
505            CONSTANT_Utf8_info cputf8 = new CONSTANT_Utf8_info(d);
506            // If an equivalent CONSTANT_Utf8 already exists, we return
507            // the pre-existing one and allow cputf8 to be GC'd.
508            cp = (cp_info) CONSTANT_Utf8_collector.v().add(cputf8);
509            if (debug)
510               G.v().out.println("Constant pool[" + i + "]: Utf8 = \"" +
511                                  cputf8.convert() + "\"");
512            break;
513         case cp_info.CONSTANT_MethodHandle:
514             cp = new CONSTANT_MethodHandle_info();
515             ((CONSTANT_MethodHandle_info)cp).kind = d.readByte();
516             ((CONSTANT_MethodHandle_info)cp).target_index = d.readUnsignedShort();
517             break;
518         case cp_info.CONSTANT_InvokeDynamic:
519             cp = new CONSTANT_InvokeDynamic_info();
520             ((CONSTANT_InvokeDynamic_info)cp).bootstrap_method_index = d.readUnsignedShort();
521             ((CONSTANT_InvokeDynamic_info)cp).name_and_type_index = d.readUnsignedShort();
522             break;
523         default:
524            G.v().out.println("Unknown tag in constant pool: " +
525                               tag + " at entry " + i);
526            return false;
527         }
528         cp.tag = tag;
529         constant_pool[i] = cp;
530      }
531      return true;
532   }

534   private void readAllBytes(byte[] dest, DataInputStream d) throws IOException {
535     int total_len = dest.length;
536     int read_len = 0;
537     while(read_len < total_len){
538       int to_read = total_len - read_len;
539       int curr_read = d.read(dest, read_len, to_read);
540       read_len += curr_read;
541     }
542   }

544   /** Reads in the given number of attributes from the given stream.
545    * @param d Stream forming the <tt>.class</tt> file.
546    * @param attributes_count number of attributes to read in.
547    * @param ai pre-allocated array of attributes to be filled in.
548    * @return <i>true</i> if read was successful, <i>false</i> on some error.
549    * @exception java.io.IOException on error.
550    */
551   protected boolean readAttributes(DataInputStream d,int attributes_count,
552                                    attribute_info[] ai) throws IOException {
553      attribute_info a=null;
554      int i;
555      int j;
556      long len;
557      String s;
558
559      for (i=0;i<attributes_count;i++) {
560         
561         j = d.readUnsignedShort();  // read attribute name before allocating
562         len = d.readInt() & 0xFFFFFFFFL;
563         s = ((CONSTANT_Utf8_info)(constant_pool[j])).convert();
564         if (s.compareTo(attribute_info.SourceFile)==0) 
565         {
566            SourceFile_attribute sa = new SourceFile_attribute();
567            sa.sourcefile_index = d.readUnsignedShort();
568            a = (attribute_info)sa;
569         } else if(s.compareTo(attribute_info.ConstantValue)==0) 
570         {
571            ConstantValue_attribute ca = new ConstantValue_attribute();
572            ca.constantvalue_index = d.readUnsignedShort();
573            a = (attribute_info)ca;
574         } else if(s.compareTo(attribute_info.Code)==0) 
575         {
576            Code_attribute ca = new Code_attribute();
577            ca.max_stack = d.readUnsignedShort();
578            ca.max_locals = d.readUnsignedShort();
579            ca.code_length = d.readInt() & 0xFFFFFFFFL;
580            ca.code = new byte[(int) ca.code_length];
581            readAllBytes(ca.code, d);
582            ca.exception_table_length = d.readUnsignedShort();
583            ca.exception_table = new exception_table_entry[ca.exception_table_length];
584            int k;
585            exception_table_entry e;
586            for (k=0; k<ca.exception_table_length; k++) {
587               e = new exception_table_entry();
588               e.start_pc = d.readUnsignedShort();
589               e.end_pc = d.readUnsignedShort();
590               e.handler_pc = d.readUnsignedShort();
591               e.catch_type = d.readUnsignedShort();
592               ca.exception_table[k] = e;
593            }
594            ca.attributes_count = d.readUnsignedShort();
595            ca.attributes = new attribute_info[ca.attributes_count];
596            readAttributes(d,ca.attributes_count,ca.attributes);
597            a = (attribute_info)ca;
598            
599            
600         } else if(s.compareTo(attribute_info.Exceptions)==0) 
601         {
602            Exception_attribute ea = new Exception_attribute();
603            ea.number_of_exceptions = d.readUnsignedShort();
604            if (ea.number_of_exceptions>0) {
605               int k;
606               ea.exception_index_table = new int[ea.number_of_exceptions];
607               for (k=0; k<ea.number_of_exceptions; k++)
608                  ea.exception_index_table[k]  = d.readUnsignedShort();
609            }
610            a = (attribute_info)ea;
611         } else if(s.compareTo(attribute_info.LineNumberTable)==0) 
612         {
613             LineNumberTable_attribute la = new LineNumberTable_attribute();
614             la.line_number_table_length = d.readUnsignedShort();
615             int k;
616             line_number_table_entry e;
617             la.line_number_table = new
618                 line_number_table_entry[la.line_number_table_length];
619             for (k=0; k<la.line_number_table_length; k++) {
620                 e = new line_number_table_entry();
621                 e.start_pc = d.readUnsignedShort();
622                 e.line_number = d.readUnsignedShort();
623                 la.line_number_table[k] = e;
624             }
625             a = (attribute_info)la;
626         } 
627         else if(s.compareTo(attribute_info.LocalVariableTable)==0) 
628         {
629            LocalVariableTable_attribute la = new LocalVariableTable_attribute();
630            la.local_variable_table_length = d.readUnsignedShort();
631            int k;
632            local_variable_table_entry e;
633            la.local_variable_table =
634               new local_variable_table_entry[la.local_variable_table_length];
635            for (k=0; k<la.local_variable_table_length; k++) {
636               e = new local_variable_table_entry();
637               e.start_pc = d.readUnsignedShort();
638               e.length = d.readUnsignedShort();
639               e.name_index = d.readUnsignedShort();
640               e.descriptor_index = d.readUnsignedShort();
641               e.index = d.readUnsignedShort();
642               la.local_variable_table[k] = e;
643            }
644            a = (attribute_info)la;
645         } 
646         else if(s.compareTo(attribute_info.LocalVariableTypeTable)==0) 
647         {

807    private element_value [] readElementValues(int count, DataInputStream d, boolean needName, int name_index)
808        throws IOException {
809        element_value [] list = new element_value[count];
810        for (int x = 0; x < count; x++){
811            if (needName){
812                name_index = d.readUnsignedShort();
813            }
814            int tag = d.readUnsignedByte();
815            char kind = (char)tag;
816            if (kind == 'B' || kind == 'C' || kind == 'D' || kind == 'F' || kind == 'I' || kind == 'J' || kind == 'S' || kind == 'Z' || kind == 's'){
817                constant_element_value elem = new constant_element_value();
818                elem.name_index = name_index;
819                elem.tag = kind;
820                elem.constant_value_index = d.readUnsignedShort();
821                list[x] = elem;
822            }
823            else if (kind == 'e'){
824                enum_constant_element_value elem = new enum_constant_element_value();
825                elem.name_index = name_index;
826                elem.tag = kind;
827                elem.type_name_index = d.readUnsignedShort();
828                elem.constant_name_index = d.readUnsignedShort();
829                list[x] = elem;
830            }
831            else if (kind == 'c'){
832                class_element_value elem = new class_element_value();
833                elem.name_index = name_index;
834                elem.tag = kind;
835                elem.class_info_index = d.readUnsignedShort();
836                list[x] = elem;
837            }
838            else if (kind == '['){
839                array_element_value elem = new array_element_value();
840                elem.name_index = name_index;
841                elem.tag = kind;
842                elem.num_values = d.readUnsignedShort();
843                elem.values = readElementValues(elem.num_values, d, false, name_index);
844                list[x] = elem;
845            }
846            else if (kind == '@'){
847                annotation_element_value elem = new annotation_element_value();
848                elem.name_index = name_index;
849                elem.tag = kind;
850                annotation annot = new annotation();
851                annot.type_index = d.readUnsignedShort();
852                annot.num_element_value_pairs = d.readUnsignedShort();
853                annot.element_value_pairs = readElementValues(annot.num_element_value_pairs, d, true, 0);
854                elem.annotation_value = annot;
855                list[x] = elem;
856            }
857            else {
858                throw new RuntimeException("Unknown element value pair kind: "+kind);
859            }
860        }

1420    * @return String of types parsed.
1421    * @see ClassFile#parseDesc

1428      boolean didone = false;
1429
1430      len = desc.length();
1431      for (i=0;i<len;i++) {
1432         c = desc.charAt(i);
1433         if (c==DESC_BYTE.charAt(0)) {
1434            param = "byte";
1435         } else if (c==DESC_CHAR.charAt(0)) {
1436            param = "char";
1437         } else if (c==DESC_DOUBLE.charAt(0)) {
1438            param = "double";
1439         } else if (c==DESC_FLOAT.charAt(0)) {
1440            param = "float";
1441         } else if (c==DESC_INT.charAt(0)) {
1442            param = "int";
1443         } else if (c==DESC_LONG.charAt(0)) {
1444            param = "long";
1445         } else if (c==DESC_SHORT.charAt(0)) {
1446            param = "short";
1447         } else if (c==DESC_BOOLEAN.charAt(0)) {
1448            param = "boolean";
1449         } else if (c==DESC_VOID.charAt(0)) {
1450            param = "void";
1451         } else if (c==DESC_ARRAY.charAt(0)) {
1452            arraylevel++;
1453            continue;
1454         } else if (c==DESC_OBJECT.charAt(0)) {
1455            int j;
1456            j = desc.indexOf(';',i+1);
1457            if (j<0) {
1458               G.v().out.println("Warning: Parse error -- can't find a ; in " +
1459                                  desc.substring(i+1));
1460               param = "<error>";
1461            } else {
1462               if (j-i>10 && desc.substring(i+1,i+11).compareTo("java/lang/")==0)
1463                  i = i+10;
1464               param = desc.substring(i+1,j);
1465               // replace '/'s with '.'s
1466               param = param.replace('/','.');
1467               i = j;
1468            }
1469         } else    {
1470            param = "???";
1471         }
1472         if (didone) params = params + sep;
1473         params = params + param;
1474         while (arraylevel>0) {
1475            params = params + "[]";
1476            arraylevel--;
1477         }
1478         didone = true;
1479      }
1480      return params;
1481   }

1788   // or quicksort rather cumbersome, so since it is quite close to the
1789   // limits of efficient insertion/selection sort, we'll use that anyway.
1790    void sortConstantPool() {

1792      short redirect[] = new short[constant_pool_count];
1793      newcp[0] = constant_pool[0];  // the 0-entry stays put

1796      for (int i=1;i<constant_pool_count;i++) redirect[i] = (short)0;
1797      for (int i=1;i<constant_pool_count;i++) {
1798         for (smallest = 1;smallest<constant_pool_count;smallest++)
1799            if (redirect[smallest]==(short)0) break;
1800         //G.v().out.println(" smallest = " + smallest);
1801         j = (constant_pool[smallest].tag==cp_info.CONSTANT_Double ||
1802              constant_pool[smallest].tag==cp_info.CONSTANT_Long) ? smallest+2 : smallest+1;
1803         for (;j<constant_pool_count;j++) {

1805                compareTo(constant_pool,constant_pool[smallest],constant_pool)<0) {
1806               smallest = j;

1809                constant_pool[j].tag==cp_info.CONSTANT_Long) j++;
