032package soot.coffi;
033
034/** Generic superclass for all attributes.
035 * @author Clark Verbrugge
036 */
037class attribute_info {
038   /** String by which a SourceFile attribute is recognized.
039    * @see SourceFile_attribute
040    */
041   public static final String SourceFile = "SourceFile";
042   /** String by which a ConstantValue attribute is recognized.
043    * @see ConstantValue_attribute
044    */
045   public static final String ConstantValue = "ConstantValue";
046   /** String by which a Code attribute is recognized.
047    * @see Code_attribute
048    */
049   public static final String Code = "Code";
050   /** String by which an Exceptions attribute is recognized.
051    * @see Exception_attribute
052    */
053   public static final String Exceptions = "Exceptions";
054   /** String by which a LineNumberTable attribute is recognized.
055    * @see LineNumberTable_attribute
056    */
057   public static final String LineNumberTable = "LineNumberTable";
058   /** String by which a LocalVariableTable attribute is recognized.
059    * @see LocalVariableTable_attribute
060    */
061   public static final String LocalVariableTable = "LocalVariableTable";
062   /** String by which a InnerClasses attribute is recognized.
063    * @see InnerClasses_attribute
064    */
065   public static final String InnerClasses = "InnerClasses";
066   /** String by which a Synthetic attribute is recognized.
067    * @see Synthetic_attribute
068    */
069   public static final String Synthetic = "Synthetic";
070   /** String by which a BootstrapMethods attribute is recognized.
071    * @see BootstrapMethods_attribute
072    */
073   public static final String BootstrapMethods = "BootstrapMethods";
074
075   
076   /**
077    * the following tags are added for java1.5
078    */
079   
080   /** String by which a Synthetic attribute is recognized.
081    * @see Signature_attribute
082    */
083   public static final String Signature = "Signature";
084
085   /** String by which a Deprecated attribute is recognized.
086    * @see Deprecated_attribute
087    */
088   public static final String Deprecated = "Deprecated";
089
090   /** String by which a EnclosingMethod attribute is recognized.
091    * @see EnclosingMethod_attribute
092    */
093   public static final String EnclosingMethod = "EnclosingMethod";
094   
095   /** String by which a LocalVariableTypeTable attribute is recognized.
096    * @see LocalVariableTypeTable_attribute
097    */
098   public static final String LocalVariableTypeTable = "LocalVariableTypeTable";
099   
100   /** String by which a runtime visible annotation attribute is recognized.
101    * @see RuntimeVisibleAnnotations_attribute
102    */
103   public static final String RuntimeVisibleAnnotations = "RuntimeVisibleAnnotations";
104
105   /** String by which a runtime invisible annotation attribute is recognized.
106    * @see RuntimeInvisibleAnnotations_attribute
107    */
108   public static final String RuntimeInvisibleAnnotations = "RuntimeInvisibleAnnotations";
109
110   /** String by which a runtime visible parameter annotation attribute 
111    * is recognized.
112    * @see RuntimeVisibleParameterAnnotations_attribute
113    */
114   public static final String RuntimeVisibleParameterAnnotations = "RuntimeVisibleParameterAnnotations";
115
116   /** String by which a runtime invisible parameter annotation attribute 
117    * is recognized.
118    * @see RuntimeInvisibleParameterAnnotations_attribute
119    */
120   public static final String RuntimeInvisibleParameterAnnotations = "RuntimeInvisibleParameterAnnotations";
121
122   /** String by which an annotation default attribute 
123    * is recognized.
124    * @see AnnotationDefault_attribute
125    */
126   public static final String AnnotationDefault = "AnnotationDefault";
127
128   /** Constant pool index of the name of this attribute; should be a utf8 String
129    * matching one of the constant Strings define here.
130    * @see attribute_info#SourceFile
131    * @see attribute_info#ConstantValue
132    * @see attribute_info#Code
133    * @see attribute_info#Exceptions
134    * @see attribute_info#LineNumberTable
135    * @see attribute_info#LocalVariableTable
136    * @see attribute_info#InnerClasses
137    * @see CONSTANT_Utf8_info
138    */
139   public int attribute_name;
140   /** Length of attribute in bytes. */
141   public long attribute_length;
142}
