032package soot.coffi;
033import soot.*;
034
035import java.util.*;
036
037/** Represents a single method_info object.
038 * @see ClassFile
039 * @author Clark Verbrugge
040 */
041public class method_info {
042   /** Access flags for this field. */
043    public int access_flags;
044   /** Constant pool index of the name of this method.
045    * @see ClassFile#constant_pool
046    * @see CONSTANT_Utf8_info
047    */
048    public int name_index;
049   /** Constant pool index of the type descriptor of this method.
050    * @see ClassFile#constant_pool
051    * @see CONSTANT_Utf8_info
052    */
053    public int descriptor_index;
054   /** Count of attributes this method contains. */
055    public int attributes_count;
056   /** Array of attribute_info objects for this method.
057    * @see attribute_info
058    */
059    public attribute_info attributes[];
060
061    /** A shortcut into attributes array for Code_attribute
062     * @see Code_attribute
063     */
064    public Code_attribute code_attr;
065
066   /** List of Instructions constructed when the method is parsed.
067    * @see Instruction
068    */
069    public Instruction instructions;
070   /** Control Flow Graph constructed when the method is parsed.
071    * @see CFG
072    */
073    public CFG cfg;
074
075    public soot.SootMethod jmethod;
076
077    List instructionList;
078
079   /** Returns the name of this method.
080    * @param constant_pool the constant_pool for this class.
081    * @return the name of this method.
082    */
083    public String toName(cp_info constant_pool[]) {
084      CONSTANT_Utf8_info ci;
085      ci = (CONSTANT_Utf8_info)(constant_pool[name_index]);
086      return ci.convert();
087   }
088
089   /** Locates and returns the code attribute for this method.
090    * @return the single code attribute, or null if not found.
091    * @see Code_attribute
092    */
093    Code_attribute locate_code_attribute() {
094      attribute_info ai;
095      int i;
096
097      for (i=0; i<attributes_count; i++) {
098         ai = attributes[i];
099         if (ai instanceof Code_attribute)
100            return (Code_attribute)ai;
101      }
102      return null;
103   }
104
105   /** Returns the prototype of this field.
106    * @param constant_pool the constant_pool for this class.
107    * @return the prototype (access + return + name + parameters) of this method.
108    */
109    public String prototype(cp_info constant_pool[]) {
110      String access,rt,name,params;
111      Code_attribute c = locate_code_attribute();
112
113      access = ClassFile.access_string(access_flags," ");
114      rt = ClassFile.parseMethodDesc_return(cp_info.getTypeDescr(constant_pool,
115                                                                 descriptor_index));
116      name = toName(constant_pool);
117      params = ClassFile.parseMethodDesc_params(cp_info.
118                                                getTypeDescr(constant_pool,
119                                                             descriptor_index));
120      if (access.length()>0)
121         return access + " " + rt + " " + name + "(" + params + ")";
122      return rt + " " + name + "(" + params + ")";
123   }
124
125   /** Displays this method, printing a prototype followed by list of Instructions.
126    * @param constant_pool the constant_pool for this class.
127    * @see prototype
128    * @see ByteCode#showCode
129    */
130    void print(cp_info constant_pool[]) {
131      G.v().out.println(prototype(constant_pool));
132      ByteCode.showCode(instructions,constant_pool);
133   }
134}
