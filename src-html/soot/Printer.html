079    public void printTo(SootClass cl, PrintWriter out) {
080        // add jimple line number tags
081        setJimpleLnNum(1);
082
083        // Print class name + modifiers
084        {
085            StringTokenizer st =
086                new StringTokenizer(Modifier.toString(cl.getModifiers()));
087            while (st.hasMoreTokens()) {
088                String tok = st.nextToken();
089                if( cl.isInterface() && tok.equals("abstract") ) continue;
090                out.print(tok + " ");
091            }
092
093            String classPrefix = "";
094
095            if (!cl.isInterface()) {
096                classPrefix = classPrefix + " class";
097                classPrefix = classPrefix.trim();
098            }
099
100            out.print(
101                classPrefix + " " + Scene.v().quotedNameOf(cl.getName()) + "");
102        }
103
104        // Print extension
105        {
106            if (cl.hasSuperclass())
107                out.print(
108                    " extends "
109                        + Scene.v().quotedNameOf(cl.getSuperclass().getName())
110                        + "");
111        }
112
113        // Print interfaces
114        {
115            Iterator<SootClass> interfaceIt = cl.getInterfaces().iterator();
116
117            if (interfaceIt.hasNext()) {
118                out.print(" implements ");
119
120                out.print(
121                    ""
122                        + Scene.v().quotedNameOf(
123                            interfaceIt.next().getName())
124                        + "");
125
126                while (interfaceIt.hasNext()) {
127                    out.print(",");
128                    out.print(
129                        " "
130                            + Scene.v().quotedNameOf(
131                                interfaceIt.next().getName())
132                            + "");
133                }
134            }
135        }
136
137        out.println();
138        incJimpleLnNum();
139/*        if (!addJimpleLn()) {
140            Iterator clTagsIt = cl.getTags().iterator();
141            while (clTagsIt.hasNext()) {
142                final Tag t = (Tag)clTagsIt.next();
143                out.println(t);
144            }
145        }*/
146        out.println("{");
147        incJimpleLnNum();
148        if (Options.v().print_tags_in_output()){
149            Iterator<Tag> cTagIterator = cl.getTags().iterator();
150            while (cTagIterator.hasNext()) {
151                Tag t = cTagIterator.next();
152                out.print("/*");
153                out.print(t.toString());
154                out.println("*/");
155            }
156        }
157
158        // Print fields
159        {
160            Iterator<SootField> fieldIt = cl.getFields().iterator();
161
162            if (fieldIt.hasNext()) {
163                while (fieldIt.hasNext()) {
164                    SootField f = fieldIt.next();
165
166                    if (f.isPhantom())
167                        continue;
168
169                    if (Options.v().print_tags_in_output()){
170                        Iterator<Tag> fTagIterator = f.getTags().iterator();
171                        while (fTagIterator.hasNext()) {
172                            Tag t = fTagIterator.next();
173                            out.print("/*");
174                            out.print(t.toString());
175                            out.println("*/");
176                        }
177                    }
178                    out.println("    " + f.getDeclaration() + ";");
179                    if (addJimpleLn()) {
180                        setJimpleLnNum(addJimpleLnTags(getJimpleLnNum(), f));           
181                    }
182
183                    //incJimpleLnNum();
184                }
185            }
186        }
187
188        // Print methods
189        {
190            Iterator<SootMethod> methodIt = cl.methodIterator();
191
192            if (methodIt.hasNext()) {
193                if (cl.getMethodCount() != 0) {
194                    out.println();
195                    incJimpleLnNum();
196                }
197
198                while (methodIt.hasNext()) {
199                    SootMethod method = methodIt.next();
200
201                    if (method.isPhantom())
202                        continue;
203
204                    if (!Modifier.isAbstract(method.getModifiers())
205                        && !Modifier.isNative(method.getModifiers())) {
206                        if (!method.hasActiveBody()){
207                            method.retrieveActiveBody(); //force loading the body
208                            if (!method.hasActiveBody()) //in case we really don't have it
209                                throw new RuntimeException("method "+ method.getName() + " has no active body!");
210                        }
211                        else
212                            if (Options.v().print_tags_in_output()){
213                                Iterator<Tag> mTagIterator = method.getTags().iterator();
214                                while (mTagIterator.hasNext()) {
215                                    Tag t = mTagIterator.next();
216                                    out.print("/*");
217                                    out.print(t.toString());
218                                    out.println("*/");
219                                }
220                            }
221                            printTo(method.getActiveBody(), out);
222
223                        if (methodIt.hasNext()) {
224                            out.println();
225                            incJimpleLnNum();
226                        }
227                    } else {
228                           
229                        if (Options.v().print_tags_in_output()){
230                            Iterator<Tag> mTagIterator = method.getTags().iterator();
231                            while (mTagIterator.hasNext()) {
232                                Tag t = mTagIterator.next();
233                                out.print("/*");
234                                out.print(t.toString());
235                                out.println("*/");
236                            }
237                        }
238                        
239                        out.print("    ");
240                        out.print(method.getDeclaration());
241                        out.println(";");
242                        incJimpleLnNum();
243                        if (methodIt.hasNext()) {
244                            out.println();
245                            incJimpleLnNum();
246                        }
247                    }
248                }
249            }
250        }
251        out.println("}");
252        incJimpleLnNum();
253    }

255    /**
256     *   Prints out the method corresponding to b Body, (declaration and body),
257     *   in the textual format corresponding to the IR used to encode b body.
258     *
259     *   @param out a PrintWriter instance to print to.
260     */
261    public void printTo(Body b, PrintWriter out) {
262//        b.validate();
263
264        boolean isPrecise = !useAbbreviations();
265
266        String decl = b.getMethod().getDeclaration();
267
268        out.println("    " + decl);
269        //incJimpleLnNum();
270    
271        // only print tags if not printing attributes in a file 
272        if (!addJimpleLn()) {
273            /*for( Iterator tIt = b.getMethod().getTags().iterator(); tIt.hasNext(); ) {    final Tag t = (Tag) tIt.next();
274                out.println(t);
275                incJimpleLnNum();
276
277            }*/
278        }
279       
280        if (addJimpleLn()) {
281            setJimpleLnNum(addJimpleLnTags(getJimpleLnNum(), b.getMethod()));           
282            //G.v().out.println("added jimple ln tag for method: "+b.getMethod().toString()+" "+b.getMethod().getDeclaringClass().getName());
283        }
284
285        out.println("    {");
286        incJimpleLnNum();
287        
288        UnitGraph unitGraph = new soot.toolkits.graph.BriefUnitGraph(b);
289
290        LabeledUnitPrinter up;
291        if( isPrecise ) up = new NormalUnitPrinter(b);
292        else up = new BriefUnitPrinter(b);
293
294        if (addJimpleLn()) {
295            up.setPositionTagger( new AttributesUnitPrinter(getJimpleLnNum()) );
296        }
297        
298        printLocalsInBody(b, up);
299
300        printStatementsInBody(b, out, up, unitGraph);
301
302        out.println("    }");
303        incJimpleLnNum();
304
305    }

307    /** Prints the given <code>JimpleBody</code> to the specified <code>PrintWriter</code>. */
308    private void printStatementsInBody(Body body, java.io.PrintWriter out, LabeledUnitPrinter up, UnitGraph unitGraph ) {
309        Chain<Unit> units = body.getUnits();
310        Unit previousStmt;
311
312        for (Unit currentStmt : units) {
313            previousStmt = currentStmt;
314            
315            // Print appropriate header.
316            {
317                // Put an empty line if the previous node was a branch node, the current node is a join node
318                //   or the previous statement does not have body statement as a successor, or if
319                //   body statement has a label on it
320
321                if (currentStmt != units.getFirst()) {
322                    if (unitGraph.getSuccsOf(previousStmt).size() != 1
323                        || unitGraph.getPredsOf(currentStmt).size() != 1
324                        || up.labels().containsKey(currentStmt)) {
325                        up.newline();
326                    } else {
327                        // Or if the previous node does not have body statement as a successor.
328
329                        List<Unit> succs = unitGraph.getSuccsOf(previousStmt);
330
331                        if (succs.get(0) != currentStmt) {
332                            up.newline();
333                        }
334                    }
335                }
336
337                if (up.labels().containsKey(currentStmt)) {
338                    up.unitRef( currentStmt, true );
339                    up.literal(":");
340                    up.newline();
341                }
342
343                if (up.references().containsKey(currentStmt)) {
344                    up.unitRef( currentStmt, false );
345                }
346            }
347
348            up.startUnit(currentStmt);
349            currentStmt.toString(up);
350            up.endUnit(currentStmt);
351
352            up.literal(";");
353            up.newline();
354
355            // only print them if not generating attributes files 
356            // because they mess up line number
357            //if (!addJimpleLn()) {
358            if (Options.v().print_tags_in_output()){
359                Iterator<Tag> tagIterator = currentStmt.getTags().iterator();
360                while (tagIterator.hasNext()) {
361                    Tag t = tagIterator.next();
362                    up.noIndent();
363                    up.literal("/*");
364                    up.literal(t.toString());
365                    up.literal("*/");
366                    up.newline();
367                }
368                /*Iterator udIt = currentStmt.getUseAndDefBoxes().iterator();
369                while (udIt.hasNext()) {
370                    ValueBox temp = (ValueBox)udIt.next();
371                    Iterator vbtags = temp.getTags().iterator();
372                    while (vbtags.hasNext()) {
373                        Tag t = (Tag) vbtags.next();
374                        up.noIndent();
375                        up.literal("VB Tag: "+t.toString());
376                        up.newline();
377                    }
378                }*/
379            }
380        }
381
382        out.print(up.toString());
383                if (addJimpleLn()){
384                        setJimpleLnNum(up.getPositionTagger().getEndLn());
385                }
386
387
388        // Print out exceptions
389        {
390            Iterator<Trap> trapIt = body.getTraps().iterator();
391
392            if (trapIt.hasNext()) {
393                out.println();
394                incJimpleLnNum();
395            }
396
397            while (trapIt.hasNext()) {
398                Trap trap = trapIt.next();
399
400                out.println(
401                    "        catch "
402                        + Scene.v().quotedNameOf(trap.getException().getName())
403                        + " from "
404                        + up.labels().get(trap.getBeginUnit())
405                        + " to "
406                        + up.labels().get(trap.getEndUnit())
407                        + " with "
408                        + up.labels().get(trap.getHandlerUnit())
409                        + ";");
410
411                incJimpleLnNum();
412
413            }
414        }
415
416    }

418    private int addJimpleLnTags(int lnNum, SootMethod meth) {
419        meth.addTag(new JimpleLineNumberTag(lnNum));
420        lnNum++;
421        return lnNum;
422    }

424    private int addJimpleLnTags(int lnNum, SootField f) {
425        f.addTag(new JimpleLineNumberTag(lnNum));
426        lnNum++;
427        return lnNum;

430    /** Prints the given <code>JimpleBody</code> to the specified <code>PrintWriter</code>. */
431    private void printLocalsInBody(
432        Body body,
433        UnitPrinter up) {
434        // Print out local variables
435        {
436            Map<Type, List<Local>> typeToLocals =
437                new DeterministicHashMap<Type, List<Local>>(body.getLocalCount() * 2 + 1, 0.7f);
438
439            // Collect locals
440            {
441                Iterator<Local> localIt = body.getLocals().iterator();
442
443                while (localIt.hasNext()) {
444                    Local local = localIt.next();
445
446                    List<Local> localList;
447
448                    Type t = local.getType();
449
450                    if (typeToLocals.containsKey(t))
451                        localList = typeToLocals.get(t);
452                    else {
453                        localList = new ArrayList<Local>();
454                        typeToLocals.put(t, localList);
455                    }
456
457                    localList.add(local);
458                }
459            }
460
461            // Print locals
462            {
463                Iterator<Type> typeIt = typeToLocals.keySet().iterator();
464
465                while (typeIt.hasNext()) {
466                    Type type = typeIt.next();
467
468                    List<Local> localList = typeToLocals.get(type);
469                    Object[] locals = localList.toArray();
470                    up.type( type );
471                    up.literal( " " );
472
473                    for (int k = 0; k < locals.length; k++) {
474                        if (k != 0)
475                            up.literal( ", " );
476
477                        up.local( (Local) locals[k] );
478                    }
479
480                    up.literal(";");
481                    up.newline();
482                }
483            }
484
485            if (!typeToLocals.isEmpty()) {
486                up.newline();
487            }
488        }
