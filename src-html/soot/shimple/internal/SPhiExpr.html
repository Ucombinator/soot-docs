031/**
032 * Internal implementation of Phi nodes.
033 *
034 * @author Navindra Umanee
035 * @see soot.shimple.PhiExpr

037public class SPhiExpr implements PhiExpr
038{
039    protected List<ValueUnitPair> argPairs = new ArrayList<ValueUnitPair>();
040    protected Map<Unit, ValueUnitPair> predToPair = new HashMap<Unit, ValueUnitPair>();  // cache
041    protected Type type;
042    
043    /**
044     * Create a trivial Phi expression for leftLocal.  preds is an ordered
045     * list of the control flow predecessor Blocks of the PhiExpr.
046     **/
047    public SPhiExpr(Local leftLocal, List<Block> preds)
048    {
049        type = leftLocal.getType();
050
051        for (Block pred : preds) {
052                addArg(leftLocal, pred);
053        }
054    }
055
056    /**
057     * Create a Phi expression from the given list of Values and Blocks.
058     **/
059    public SPhiExpr(List<Value> args, List<Unit> preds)
060    {
061        if(args.size() == 0)
062            throw new RuntimeException("Arg list may not be empty");
063        
064        if(args.size() != preds.size())
065            throw new RuntimeException("Arg list does not match Pred list");
066
067        type = args.get(0).getType();
068        Iterator<Value> argsIt = args.iterator();
069        Iterator<Unit> predsIt = preds.iterator();
070
071        while(argsIt.hasNext()){
072            Value arg = argsIt.next();
073            Object pred = predsIt.next();
074
075            if(pred instanceof Block)
076                addArg(arg, (Block)pred);
077            else if(pred instanceof Unit)
078                addArg(arg, (Unit)pred);
079            else
080                throw new RuntimeException("Must be a CFG block or tail unit.");
081        }
082    }

141    public int getArgIndex(Unit predTailUnit)

147    public ValueUnitPair getArgBox(Unit predTailUnit)
148    {
149        ValueUnitPair vup = predToPair.get(predTailUnit);
150
151        // we pay a penalty for misses but hopefully the common case
152        // is faster than an iteration over argPairs every time
153        if(vup == null || vup.getUnit() != predTailUnit){
154            updateCache();
155            vup = predToPair.get(predTailUnit);
156            if((vup != null) && (vup.getUnit() != predTailUnit))
157                throw new RuntimeException("Assertion failed.");
158        }
159
160        // (null if not found)
161        return vup;
162    }
163    
164    public Value getValue(Unit predTailUnit)
165    {
166        ValueBox vb = getArgBox(predTailUnit);
167        if(vb == null)
168            return null;
169        return vb.getValue();
170    }

172    public int getArgIndex(Block pred)
173    {
174        ValueUnitPair box = getArgBox(pred);
175        return argPairs.indexOf(box); // (-1 on null)
176    }
177
178    public ValueUnitPair getArgBox(Block pred)
179    {
180        Unit predTailUnit = pred.getTail();
181        ValueUnitPair box = getArgBox(predTailUnit);
182
183        // workaround added for internal cases where the predTailUnit
184        // may not be at the end of the predecessor block
185        // (eg, fall-through pointer not moved)
186        while(box == null){
187            predTailUnit = pred.getPredOf(predTailUnit);
188            if(predTailUnit == null)
189                break;
190            box = getArgBox(predTailUnit);
191        }
192
193        return box;
194    }
195
196    public Value getValue(Block pred)
197    {
198        ValueBox vb = getArgBox(pred);
199        if(vb == null)
200            return null;
201        return vb.getValue();
202    }

207    {
208        boolean ret1 = setValue(index, arg);
209        boolean ret2 = setPred(index, predTailUnit);
210        if(ret1 != ret2)
211            throw new RuntimeException("Assertion failed.");
212        return ret1;
213    }

241    public boolean setPred(int index, Unit predTailUnit)
242    {
243        ValueUnitPair argPair = getArgBox(index);
244        if(argPair == null)
245            return false;
246
247        int other = getArgIndex(predTailUnit);
248        if(other != -1){
249            G.v().out.println("WARNING: An argument with control flow predecessor " +  predTailUnit + " already exists in " + this + "!");
250            G.v().out.println("WARNING: setPred resulted in deletion of " + argPair + " from " + this + ".");
251            removeArg(argPair);
252            return false;
253        }
254
255        argPair.setUnit(predTailUnit);
256        return true;
257    }
258
259    public boolean setPred(int index, Block pred)
260    {
261        return setPred(index, pred.getTail());
262    }

284    public boolean removeArg(ValueUnitPair arg)
285    {
286        if(argPairs.remove(arg)){
287            // update cache
288            predToPair.remove(arg.getUnit());
289            // remove back-pointer
290            arg.getUnit().removeBoxPointingToThis(arg);
291            return true;
292        }
293        
294        return false;
295    }
296
297    public boolean addArg(Value arg, Block pred)
298    {
299        return addArg(arg, pred.getTail());
300    }
301    
302    public boolean addArg(Value arg, Unit predTailUnit)
303    {
304        // Do not allow phi nodes for dummy blocks
305        if (predTailUnit == null)
306                return false;
307        
308        // we disallow duplicate arguments
309        if(predToPair.keySet().contains(predTailUnit))
310            return false;
311
312        ValueUnitPair vup = new SValueUnitPair(arg, predTailUnit);
313
314        // add and update cache
315        argPairs.add(vup);
316        predToPair.put(predTailUnit, vup);
317        return true;
318    }

336    /**
337     * Update predToPair cache map which could be out-of-sync due to
338     * external setUnit or clone operations on the UnitBoxes.
339     **/
340    protected void updateCache()
341    {
342        int needed = argPairs.size();
343        predToPair = new HashMap<Unit, ValueUnitPair>(needed << 1, 1.0F); //Always attempt to allocate the next power of 2 sized map
344        for (ValueUnitPair vup : argPairs) {
345            predToPair.put(vup.getUnit(), vup);
346        }
347    }

349    public boolean equivTo(Object o)
350    {
351        if(o instanceof SPhiExpr){
352            SPhiExpr pe = (SPhiExpr) o;
353
354            if(getArgCount() != pe.getArgCount())
355                return false;
356
357            for(int i = 0; i < getArgCount(); i++){
358                if(!getArgBox(i).equivTo(pe.getArgBox(i)))
359                    return false;
360            }
361
362            return true;
363        }
364
365        return false;
366    }
367
368    public int equivHashCode()
369    {
370        int hashcode = 1;
371        
372        for(int i = 0; i < getArgCount(); i++){
373            hashcode = hashcode * 17 + getArgBox(i).equivHashCode();
374        }
375
376        return hashcode;
377    }

379    @Override
380    public List<UnitBox> getUnitBoxes()
381    {
382        Set<UnitBox> boxes = new HashSet<UnitBox>(argPairs.size());
383        for (ValueUnitPair up : argPairs)
384                boxes.add(up);
385        return new ArrayList<UnitBox>(boxes);
386    }
387
388    public void clearUnitBoxes()
389    {
390        for (UnitBox box : getUnitBoxes()) {
391            box.setUnit(null);
392        }
393    }

395    public List<ValueBox> getUseBoxes()
396    {
397        Set<ValueBox> set = new HashSet<ValueBox>();
398
399        for (ValueUnitPair argPair : argPairs) {
400            set.addAll(argPair.getValue().getUseBoxes());
401            set.add(argPair);
402        }
403
404        return new ArrayList<ValueBox>(set);
405    }

412    public String toString()
413    {
414        StringBuffer expr = new StringBuffer(Shimple.PHI + "(");
415        boolean isFirst = true;
416        for (ValueUnitPair vuPair : argPairs) {
417            if(!isFirst)
418                expr.append(", ");
419            Value arg = vuPair.getValue();
420            expr.append(arg.toString());
421            isFirst = false;
422        }
423
424        expr.append(")");
425
426        return expr.toString();
427    }
428    
429    public void toString(UnitPrinter up)
430    {
431        up.literal(Shimple.PHI);
432        up.literal("(");
433        
434        boolean isFirst = true;
435        for (ValueUnitPair vuPair : argPairs) {
436            if (!isFirst)
437                up.literal(", ");
438            vuPair.toString(up);
439            isFirst = false;
440        }
441
442        up.literal(")");
443    }

450    public Object clone()
451    {
452        // Note to self: Do not try to "fix" this *again*.  Yes, it
453        // should be a shallow copy in order to conform with the rest
454        // of Soot.  When a body is cloned, the Values are cloned
455        // explicitly and replaced, and UnitBoxes are explicitly
456        // patched.  See Body.importBodyContentsFrom for details.
457        return new SPhiExpr(getValues(), getPreds());
458    }
