063    /**
064     * Phi node Insertion Algorithm from Cytron et al 91, P24-5,
065     *
066     * <p>Special Java case: If a variable is not defined along all
067     * paths of entry to a node, a Phi node is not needed.</p>
068     **/
069    public boolean insertTrivialPhiNodes()
070    {
071                update();
072                boolean change = false;
073                varToBlocks = new HashMultiMap<Local, Block>();
074                Map<Local, List<Block>> localsToDefPoints = new HashMap<Local, List<Block>>();
075
076                // compute localsToDefPoints and varToBlocks
077                for (Block block : cfg) {
078                        for (Unit unit : block) {
079                                List<ValueBox> defBoxes = unit.getDefBoxes();
080                                for (ValueBox vb : defBoxes) {
081                                        Value def = vb.getValue();
082                                        if (def instanceof Local) {
083                                                Local local = (Local) def;
084                                                List<Block> def_points = null;
085                                                if (localsToDefPoints.containsKey(local)) {
086                                                        def_points = localsToDefPoints.get(local);
087                                                } else {
088                                                        def_points = new ArrayList<Block>();
089                                                        localsToDefPoints.put(local, def_points);
090                                                }
091                                                def_points.add(block);
092                                        }
093                                }
094
095                                if (Shimple.isPhiNode(unit))
096                                        varToBlocks.put(Shimple.getLhsLocal(unit), block);
097                        }
098                }
099
100        /* Routine initialisations. */
101        
102        int[] workFlags = new int[cfg.size()];
103        int iterCount = 0;
104        Stack<Block> workList = new Stack<Block>();
105
106        Map<Integer, Integer> has_already = new HashMap<Integer, Integer>();
107        for(Iterator<Block> blocksIt = cfg.iterator(); blocksIt.hasNext(); ){
108          Block block = blocksIt.next();
109          has_already.put(block.getIndexInMethod(), 0);
110        }
111
112        /* Main Cytron algorithm. */
113        
114        {
115                for (Local local : localsToDefPoints.keySet()) {
116                iterCount++;
117
118                // initialise worklist
119                {
120                    List<Block> def_points = localsToDefPoints.get(local);
121                    //if the local is only defined once, no need for phi nodes
122                    if(def_points.size() == 1){
123                      continue;
124                    }
125                    for(Block block : def_points){
126                        workFlags[block.getIndexInMethod()] = iterCount;
127                        workList.push(block);
128                    }
129                }
130
131                while(!workList.empty()){
132                    Block block = workList.pop();
133                    DominatorNode<Block> node = dt.getDode(block);
134                    Iterator<DominatorNode<Block>> frontierNodes = df.getDominanceFrontierOf(node).iterator();
135
136                    while(frontierNodes.hasNext()){
137                        Block frontierBlock = frontierNodes.next().getGode();
138                        int fBIndex = frontierBlock.getIndexInMethod();
139
140                        Iterator<Unit> unitsIt = frontierBlock.iterator();
141                        if(!unitsIt.hasNext()){
142                          continue;
143                        }
144                        
145                        if(has_already.get(frontierBlock.getIndexInMethod()) < iterCount)
146                        {
147                            has_already.put(frontierBlock.getIndexInMethod(), iterCount);
148                            prependTrivialPhiNode(local, frontierBlock);
149                            change = true;
150
151                            if(workFlags[fBIndex] < iterCount){
152                                workFlags[fBIndex] = iterCount;
153                                workList.push(frontierBlock);
154                            }
155                        } 
156                    }
157                }
158            }
159        }
160
161        return change;
162    }

164    /**
165     * Inserts a trivial Phi node with the appropriate number of
166     * arguments.
167     **/
168    public void prependTrivialPhiNode(Local local, Block frontierBlock)
169    {
170        List<Block> preds = frontierBlock.getPreds();
171        PhiExpr pe = Shimple.v().newPhiExpr(local, preds);
172        pe.setBlockId(frontierBlock.getIndexInMethod());
173        Unit trivialPhi = Jimple.v().newAssignStmt(local, pe);
174        Unit blockHead = frontierBlock.getHead();
175
176        // is it a catch block?
177        if(blockHead instanceof IdentityUnit)
178            frontierBlock.insertAfter(trivialPhi, frontierBlock.getHead());
179        else
180            frontierBlock.insertBefore(trivialPhi, frontierBlock.getHead());
181
182        varToBlocks.put(local, frontierBlock);
183    }

185    /**
186     * Exceptional Phi nodes have a huge number of arguments and control
187     * flow predecessors by default.  Since it is useless trying to keep
188     * the number of arguments and control flow predecessors in synch,
189     * we might as well trim out all redundant arguments and eliminate
190     * a huge number of copy statements when we get out of SSA form in
191     * the process.
192     **/
193    public void trimExceptionalPhiNodes()
194    {
195        Set<Unit> handlerUnits = new HashSet<Unit>();
196        Iterator<Trap> trapsIt = body.getTraps().iterator();
197
198        while(trapsIt.hasNext()) {
199            Trap trap = trapsIt.next();
200            handlerUnits.add(trap.getHandlerUnit());
201        }
202
203        for (Block block : cfg) {
204            // trim relevant Phi expressions
205            if(handlerUnits.contains(block.getHead())){
206                for (Unit unit : block) {
207                    //if(!(newPhiNodes.contains(unit)))
208                    PhiExpr phi = Shimple.getPhiExpr(unit);
209
210                    if(phi == null)
211                        continue;
212
213                    trimPhiNode(phi);
214                }
215            }
216        }
217    }

219    /**
220     * @see #trimExceptionalPhiNodes()
221     **/
222    public void trimPhiNode(PhiExpr phiExpr)
223    {
224        /* A value may appear many times in an exceptional Phi. Hence,
225           the same value may be associated with many UnitBoxes. We
226           build the MultiMap valueToPairs for convenience.  */
227
228        MultiMap<Value, ValueUnitPair> valueToPairs = new HashMultiMap<Value, ValueUnitPair>();
229        for (ValueUnitPair argPair : phiExpr.getArgs()) {
230            Value value = argPair.getValue();
231            valueToPairs.put(value, argPair);
232        }
233
234        /* Consider each value and see if we can find the dominating
235           UnitBoxes.  Once we have found all the dominating
236           UnitBoxes, the rest of the redundant arguments can be
237           trimmed.  */
238        
239        Iterator<Value> valuesIt = valueToPairs.keySet().iterator();
240        while(valuesIt.hasNext()){
241            Value value = valuesIt.next();
242
243            // although the champs list constantly shrinks, guaranteeing
244            // termination, the challengers list never does.  This could
245            // be optimised.
246            Set<ValueUnitPair> pairsSet = valueToPairs.get(value);
247            List<ValueUnitPair> champs = new LinkedList<ValueUnitPair>(pairsSet);
248            List<ValueUnitPair> challengers = new LinkedList<ValueUnitPair>(pairsSet);
249            
250            // champ is the currently assumed dominator
251            ValueUnitPair champ = champs.remove(0);
252            Unit champU = champ.getUnit();
253
254            // hopefully everything will work out the first time, but
255            // if not, we will have to try a new champion just in case
256            // there is more that can be trimmed.
257            boolean retry = true;
258            while(retry){
259                retry = false;
260
261                // go through each challenger and see if we dominate them
262                // if not, the challenger becomes the new champ
263                for (Iterator<ValueUnitPair> iterator = challengers.iterator(); iterator.hasNext(); ) {
264                    ValueUnitPair challenger = iterator.next();
265                    if (challenger.equals(champ))
266                        continue;
267                    Unit challengerU = challenger.getUnit();
268
269                    // kill the challenger
270                    if (dominates(champU, challengerU)) {
271                        phiExpr.removeArg(challenger);
272                        iterator.remove();
273                    }
274                    // we die, find a new champ
275                    else if (dominates(challengerU, champU)) {
276                        phiExpr.removeArg(champ);
277                        champ = challenger;
278                        champU = champ.getUnit();
279                    }
280
281                    // neither wins, oops!  we'll have to try the next
282                    // available champ at the next pass.  It may very
283                    // well be inevitable that we will have two
284                    // identical value args in an exceptional PhiExpr,
285                    // but the more we can trim the better.
286                    else
287                        retry = true;
288                }
289
290                if(retry) {
291                    if(champs.isEmpty())
292                        break;
293                    champ = champs.remove(0);
294                    champU = champ.getUnit();
295                }
296            }
297        }
298
299        /*
300        {
301            List preds = phiExpr.getPreds();
302
303            for(int i = 0; i < phiExpr.getArgCount(); i++){
304                ValueUnitPair vup = phiExpr.getArgBox(i);
305                Value value = vup.getValue();
306                Unit unit = vup.getUnit();
307
308                PhiExpr innerPhi = Shimple.getPhiExpr(unit);
309                if(innerPhi == null)
310                    continue;
311                
312                Value innerValue = Shimple.getLhsLocal(unit);
313                if(!innerValue.equals(value))
314                    continue;
315
316                boolean canRemove = true;
317                for(int j = 0; j < innerPhi.getArgCount(); j++){
318                    Unit innerPred = innerPhi.getPred(j);
319                    if(!preds.contains(innerPred)){
320                        canRemove = false;
321                        break;
322                    }
323                }
324
325                if(canRemove)
326                    phiExpr.removeArg(vup);
327            }
328        }
329        */
330    }

334    /**
335     * Returns true if champ dominates challenger.  Note that false
336     * doesn't necessarily mean that challenger dominates champ.
337     **/
338    public boolean dominates(Unit champ, Unit challenger)
339    {
340        if(champ == null || challenger == null)
341            throw new RuntimeException("Assertion failed.");
342        
343        // self-domination
344        if(champ.equals(challenger))
345            return true;
346        
347        if(unitToBlock == null)
348            unitToBlock = getUnitToBlockMap(cfg);
349
350        Block champBlock = unitToBlock.get(champ);
351        Block challengerBlock = unitToBlock.get(challenger);
352
353        if(champBlock.equals(challengerBlock)){
354            Iterator<Unit> unitsIt = champBlock.iterator();
355
356            while(unitsIt.hasNext()){
357                Unit unit = unitsIt.next();
358                if(unit.equals(champ))
359                    return true;
360                if(unit.equals(challenger))
361                    return false;
362            }
363
364            throw new RuntimeException("Assertion failed.");
365        }
366
367        DominatorNode<Block> champNode = dt.getDode(champBlock);
368        DominatorNode<Block> challengerNode = dt.getDode(challengerBlock);
369
370        // *** FIXME: System.out.println("champ: " + champNode);
371        // System.out.println("chall: " + challengerNode);
372        
373        return(dt.isDominatorOf(champNode, challengerNode));
374    }

376    /**
377     * Eliminate Phi nodes in block by naively replacing them with
378     * shimple assignment statements in the control flow predecessors.
379     * Returns true if new locals were added to the body during the
380     * process, false otherwise.
381     **/
382    public boolean doEliminatePhiNodes()
383    {
384        // flag that indicates whether we created new locals during the
385        // elimination process
386        boolean addedNewLocals = false;
387        
388        // List of Phi nodes to be deleted.
389        List<Unit> phiNodes = new ArrayList<Unit>();
390
391        // This stores the assignment statements equivalent to each
392        // (and every) Phi.  We use lists instead of a Map of
393        // non-determinate order since we prefer to preserve the order
394        // of the assignment statements, i.e. if a block has more than
395        // one Phi expression, we prefer that the equivalent
396        // assignments be placed in the same order as the Phi expressions.
397        List<AssignStmt> equivStmts = new ArrayList<AssignStmt>();
398
399        // Similarly, to preserve order, instead of directly storing
400        // the pred, we store the pred box so that we follow the
401        // pointers when SPatchingChain moves them.
402        List<ValueUnitPair> predBoxes = new ArrayList<ValueUnitPair>();
403        
404        Chain<Unit> units = body.getUnits();
405        for (Unit unit : units) {
406            PhiExpr phi = Shimple.getPhiExpr(unit);
407
408            if(phi == null)
409                continue;
410
411            Local lhsLocal = Shimple.getLhsLocal(unit);
412
413            for(int i = 0; i < phi.getArgCount(); i++){
414                Value phiValue = phi.getValue(i);
415                AssignStmt convertedPhi =
416                    Jimple.v().newAssignStmt(lhsLocal, phiValue);
417
418                equivStmts.add(convertedPhi);
419                predBoxes.add(phi.getArgBox(i));
420            }
421
422            phiNodes.add(unit);
423        }
424
425        if(equivStmts.size() != predBoxes.size())
426            throw new RuntimeException("Assertion failed.");
427        
428        /* Avoid Concurrent Modification exceptions. */
429
430        for(int i = 0; i < equivStmts.size(); i++){
431            AssignStmt stmt = equivStmts.get(i);
432            Unit pred = predBoxes.get(i).getUnit();
433
434            if(pred == null)
435                throw new RuntimeException("Assertion failed.");
436
437            // if we need to insert the copy statement *before* an
438            // instruction that happens to be *using* the Local being
439            // defined, we need to do some extra work to make sure we
440            // don't overwrite the old value of the local
441            if(pred.branches()){
442                boolean needPriming = false;
443                Local lhsLocal = (Local) stmt.getLeftOp();
444                Local savedLocal = Jimple.v().newLocal(lhsLocal.getName()+"_",
445                                                       lhsLocal.getType());
446                Iterator<ValueBox> useBoxesIt = pred.getUseBoxes().iterator();
447
448                while(useBoxesIt.hasNext()){
449                    ValueBox useBox = useBoxesIt.next();
450                    if(lhsLocal.equals(useBox.getValue())){
451                        needPriming = true;
452                        addedNewLocals = true;
453                        useBox.setValue(savedLocal);
454                    }
455                }
456
457                if(needPriming){
458                    body.getLocals().add(savedLocal);
459                    AssignStmt copyStmt = Jimple.v().newAssignStmt(savedLocal, lhsLocal);
460                    units.insertBefore(copyStmt, pred);
461                }
462
463                // this is all we really wanted to do!
464                units.insertBefore(stmt, pred);
465            }
466            else
467                units.insertAfter(stmt, pred);
468        }
469        
470        Iterator<Unit> phiNodesIt = phiNodes.iterator();
471
472        while(phiNodesIt.hasNext()){
473            Unit removeMe = phiNodesIt.next();
474            units.remove(removeMe);
475            removeMe.clearUnitBoxes();
476        }
477
478        return addedNewLocals;
479    }

481    /**
482     * Convenience function that maps units to blocks.  Should
483     * probably be in BlockGraph.
484     **/
485    public Map<Unit, Block> getUnitToBlockMap(BlockGraph blocks)
486    {
487        Map<Unit, Block> unitToBlock = new HashMap<Unit, Block>();
488
489        Iterator<Block> blocksIt = blocks.iterator();
490        while(blocksIt.hasNext()){
491            Block block = blocksIt.next();
492            Iterator<Unit> unitsIt = block.iterator();
493
494            while(unitsIt.hasNext()){
495                Unit unit = unitsIt.next();
496                unitToBlock.put(unit, block);
497            }
498        }
499
500        return unitToBlock;
501    }
