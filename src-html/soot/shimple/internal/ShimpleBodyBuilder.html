033/**
034 * This class does the real high-level work.  It takes a Jimple body
035 * or Jimple/Shimple hybrid body and produces pure Shimple.
036 *
037 * <p> The work is done in two main steps:
038 *
039 * <ol>
040 * <li> Trivial Phi nodes are added.
041 * <li> A renaming algorithm is executed.
042 * </ol>
043 *
044 * <p> This class can also translate out of Shimple by producing an
045 * equivalent Jimple body with all Phi nodes removed.
046 *
047 * <p> Note that this is an internal class, understanding it should
048 * not be necessary from a user point-of-view and relying on it
049 * directly is not recommended.
050 *
051 * @author Navindra Umanee
052 * @see soot.shimple.ShimpleBody
053 * @see <a
054 * href="http://citeseer.nj.nec.com/cytron91efficiently.html">Efficiently
055 * Computing Static Single Assignment Form and the Control Dependence
056 * Graph</a>
057 **/
058public class ShimpleBodyBuilder
059{
060    protected ShimpleBody body;
061    protected ShimpleFactory sf;
062    protected DominatorTree<Block> dt;
063    protected BlockGraph cfg;
064    
065    /**
066     * A fixed list of all original Locals.
067     **/
068    protected List<Local> origLocals;
069
070    public PhiNodeManager phi;
071    public PiNodeManager pi;
072
073    ShimpleOptions options;

075    /**
076     * Transforms the provided body to pure SSA form.
077     **/
078    public ShimpleBodyBuilder(ShimpleBody body)
079    {
080        this.body = body;
081        sf = new DefaultShimpleFactory(body);
082        sf.clearCache();
083        phi = new PhiNodeManager(body, sf);
084        pi = new PiNodeManager(body, false, sf);
085        options = body.getOptions();
086        makeUniqueLocalNames();
087    }
088    
089    public void update()
090    {
091        cfg = sf.getBlockGraph();
092        dt = sf.getDominatorTree();
093        origLocals = new ArrayList<Local>(body.getLocals());
094    }
095
096    public void transform()
097    {
098        phi.insertTrivialPhiNodes();
099
100        boolean change = false;
101        if(options.extended()){
102            change = pi.insertTrivialPiNodes();
103        
104            while(change){
105                if(phi.insertTrivialPhiNodes()){
106                    change = pi.insertTrivialPiNodes();
107                }
108                else{
109                    break;
110                }
111            }
112        }
113
114        renameLocals();
115        phi.trimExceptionalPhiNodes();
116        makeUniqueLocalNames();
117    }

132        if(optElim){
133            DeadAssignmentEliminator.v().transform(body);
134            UnreachableCodeEliminator.v().transform(body);
135            UnconditionalBranchFolder.v().transform(body);
136            Aggregator.v().transform(body);
137            UnusedLocalEliminator.v().transform(body);
138        }
139    }

141    /**
142     * Remove Phi nodes from current body, high probablity this
143     * destroys SSA form.
144     *
145     * <p> Dead code elimination + register aggregation are performed

147     * use some improvements.

151    public void eliminatePhiNodes()
152    {
153        if(phi.doEliminatePhiNodes())
154            makeUniqueLocalNames();
155
156    }

159    {
160        boolean optElim = options.node_elim_opt();
161        pi.eliminatePiNodes(optElim);
162    }

164    /**
165     * Maps new name Strings to Locals.
166     **/
167    protected Map<String, Local> newLocals;
168
169    /**
170     * Maps renamed Locals to original Locals.
171     **/
172    protected Map<Local,Local> newLocalsToOldLocal;
173
174    protected int[] assignmentCounters;
175    protected Stack<Integer>[] namingStacks;
176    
177    /**
178     * Variable Renaming Algorithm from Cytron et al 91, P26-8,
179     * implemented in various bits and pieces by the next functions.
180     * Must be called after trivial nodes have been added.
181     **/
182    public void renameLocals()
183    {
184        update();
185        newLocals = new HashMap<String, Local>();
186        newLocalsToOldLocal = new HashMap<Local, Local>();
187
188        assignmentCounters = new int[origLocals.size()];
189        namingStacks = new Stack[origLocals.size()];
190
191        for(int i = 0; i < namingStacks.length; i++)
192            namingStacks[i] = new Stack<Integer>();
193
194        List<Block> heads = cfg.getHeads();
195
196        if(heads.isEmpty())
197            return;
198
199        if(heads.size() != 1)
200            throw new RuntimeException("Assertion failed:  Only one head expected.");
201        
202        Block entry = heads.get(0);
203        renameLocalsSearch(entry);
204    }
205
206    /**
207     * Driven by renameLocals().
208     **/
209    public void renameLocalsSearch(Block block)
210    {
211        // accumulated in Step 1 to be re-processed in Step 4
212        List<Local> lhsLocals = new ArrayList<Local>();
213        
214        // Step 1 of 4 -- Rename block's uses (ordinary) and defs
215        {
216            // accumulated and re-processed in a later loop
217                for (Unit unit : block) {
218                // Step 1/2 of 1
219                {
220                    List<ValueBox> useBoxes = new ArrayList<ValueBox>();
221
222                    if(!Shimple.isPhiNode(unit))
223                        useBoxes.addAll(unit.getUseBoxes());
224
225                    for (ValueBox useBox : useBoxes) {
226                        Value use = useBox.getValue();
227                        int localIndex = indexOfLocal(use);
228
229                        // not one of our locals
230                        if(localIndex == -1)
231                            continue;
232
233                        Local localUse = (Local) use;
234
235                        if(namingStacks[localIndex].empty())
236                            continue;
237
238                        Integer subscript = namingStacks[localIndex].peek();
239
240                        Local renamedLocal = fetchNewLocal(localUse, subscript);
241                        useBox.setValue(renamedLocal);
242                    }
243                }
244
245                // Step 1 of 1
246                {
247                    if(!(unit instanceof DefinitionStmt))
248                        continue;
249                
250                    DefinitionStmt defStmt = (DefinitionStmt) unit;
251                    
252                    Value lhsValue = defStmt.getLeftOp();
253                    
254                    // not something we're interested in
255                    if(!origLocals.contains(lhsValue))
256                        continue;
257
258                    ValueBox lhsLocalBox = defStmt.getLeftOpBox();
259                    Local lhsLocal = (Local) lhsValue;
260
261                    // re-processed in Step 4
262                    lhsLocals.add(lhsLocal);
263
264                    int localIndex = indexOfLocal(lhsLocal);
265                    if(localIndex == -1)
266                        throw new RuntimeException("Assertion failed.");
267                
268                    Integer subscript = assignmentCounters[localIndex];
269
270                    Local newLhsLocal = fetchNewLocal(lhsLocal, subscript);
271                    lhsLocalBox.setValue(newLhsLocal);
272
273                    namingStacks[localIndex].push(subscript);
274                    assignmentCounters[localIndex]++;                    
275                    
276                }
277            }
278        }
279
280        // Step 2 of 4 -- Rename Phi node uses in Successors
281        {
282                for (Block succ : cfg.getSuccsOf(block)) {
283                        // Ignore dummy blocks
284                        if (block.getHead() == null && block.getTail() == null)
285                                continue;
286                        
287                        for (Unit unit : succ) {
288                    PhiExpr phiExpr = Shimple.getPhiExpr(unit);
289
290                    if(phiExpr == null)
291                        continue;
292
293                    // simulate whichPred
294                    int argIndex = phiExpr.getArgIndex(block);
295                    if(argIndex == -1)
296                        throw new RuntimeException("Assertion failed.");
297                        
298                    ValueBox phiArgBox = phiExpr.getArgBox(argIndex);
299
300                    Local phiArg = (Local) phiArgBox.getValue();
301                    
302                    int localIndex = indexOfLocal(phiArg);
303                    if(localIndex == -1)
304                        throw new RuntimeException("Assertion failed.");
305                    
306                    if(namingStacks[localIndex].empty())
307                        continue;
308
309                    Integer subscript = namingStacks[localIndex].peek();
310                    
311                    Local newPhiArg = fetchNewLocal(phiArg, subscript);
312                    phiArgBox.setValue(newPhiArg);
313                }
314            }
315        }
316
317        // Step 3 of 4 -- Recurse over children.
318        {
319            DominatorNode<Block> node = dt.getDode(block);
320
321            // now we recurse over children
322
323            Iterator<DominatorNode<Block>> childrenIt = dt.getChildrenOf(node).iterator();
324
325            while(childrenIt.hasNext()){
326                DominatorNode<Block> childNode = childrenIt.next();
327
328                renameLocalsSearch(childNode.getGode());
329            }
330        }
331
332        // Step 4 of 4 -- Tricky name stack updates.
333        {
334            Iterator<Local> lhsLocalsIt = lhsLocals.iterator();
335
336            while(lhsLocalsIt.hasNext()){
337                Local lhsLocal = lhsLocalsIt.next();
338
339                int lhsLocalIndex = indexOfLocal(lhsLocal);
340                if(lhsLocalIndex == -1)
341                    throw new RuntimeException("Assertion failed.");
342                
343                namingStacks[lhsLocalIndex].pop();
344            }
345        }
346
347        /* And we're done.  The renaming process is complete. */
348    }

350    /**
351     * Clever convenience function to fetch or create new Local's
352     * given a Local and the desired subscript.
353     **/
354    protected Local fetchNewLocal(Local local, Integer subscript)
355    {
356        Local oldLocal = local;
357        
358        if(!origLocals.contains(local))
359            oldLocal = newLocalsToOldLocal.get(local);
360        
361        if(subscript.intValue() == 0)
362            return oldLocal;
363
364        // If the name already exists, makeUniqueLocalNames() will
365        // take care of it.
366        String name = oldLocal.getName() + "_" + subscript;
367
368        Local newLocal = newLocals.get(name);
369
370        if(newLocal == null){
371            newLocal = new JimpleLocal(name, oldLocal.getType());
372            newLocals.put(name, newLocal);
373            newLocalsToOldLocal.put(newLocal, oldLocal);
374
375            // add proper Local declation
376            body.getLocals().add(newLocal);
377        }
378
379        return newLocal;
380    }
381
382    /**
383     * Convenient function that maps new Locals to the originating
384     * Local, and finds the appropriate array index into the naming
385     * structures.
386     **/
387    protected int indexOfLocal(Value local)
388    {
389        int localIndex = origLocals.indexOf(local);
390
391        if(localIndex == -1){
392            // might be null
393            Local oldLocal = newLocalsToOldLocal.get(local);
394
395            localIndex = origLocals.indexOf(oldLocal);
396        }
397        
398        return localIndex;
399    }

401    /**
402     * Make sure the locals in the given body all have unique String
403     * names.  Renaming is done if necessary.
404     **/
405    public void makeUniqueLocalNames()
406    {
407        if(options.standard_local_names()){
408            LocalNameStandardizer.v().transform(body);
409            return;
410        }
411
412        Set<String> localNames = new HashSet<String>();
413        Iterator<Local> localsIt = body.getLocals().iterator();
414
415        while(localsIt.hasNext()){
416            Local local = localsIt.next();
417            String localName = local.getName();
418            
419            if(localNames.contains(localName)){
420                String uniqueName = makeUniqueLocalName(localName, localNames);
421                local.setName(uniqueName);
422                localNames.add(uniqueName);
423            }
424            else
425                localNames.add(localName);
426        }
427    }
428
429    /**
430     * Given a set of Strings, return a new name for dupName that is
431     * not currently in the set.
432     **/
433    public String makeUniqueLocalName(String dupName, Set<String> localNames)
434    {
435        int counter = 1;
436        String newName = dupName;
437
438        while(localNames.contains(newName))
439            newName = dupName + "_" + counter++;
440
441        return newName;
442    }
