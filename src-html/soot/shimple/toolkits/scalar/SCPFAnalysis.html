036/**
037 * A powerful constant propagator and folder based on an algorithm
038 * sketched by Cytron et al that takes conditional control flow into
039 * account.  This optimization demonstrates some of the benefits of
040 * SSA -- particularly the fact that Phi nodes represent natural merge
041 * points in the control flow.
042 *
043 * @author Navindra Umanee
044 * @see <a
045 * href="http://citeseer.nj.nec.com/cytron91efficiently.html">Efficiently
046 * Computing Static Single Assignment Form and the Control Dependence
047 * Graph</a>
048 **/
049public class SConstantPropagatorAndFolder extends BodyTransformer
050{
051    public SConstantPropagatorAndFolder(Singletons.Global g) {}
052
053    public static SConstantPropagatorAndFolder v()
054    { return G.v().soot_shimple_toolkits_scalar_SConstantPropagatorAndFolder(); }
055
056    protected ShimpleBody sb;
057    protected boolean debug;
058    
059    protected void internalTransform(Body b, String phaseName, Map<String,String> options)
060    {
061        if(!(b instanceof ShimpleBody))
062            throw new RuntimeException("SConstantPropagatorAndFolder requires a ShimpleBody.");
063        
064        this.sb = (ShimpleBody) b;
065
066        if(!sb.isSSA())
067            throw new RuntimeException("ShimpleBody is not in proper SSA form as required by SConstantPropagatorAndFolder.  You may need to rebuild it or use ConstantPropagatorAndFolder instead.");
068
069        boolean pruneCFG = PhaseOptions.getBoolean(options, "prune-cfg");
070        debug = Options.v().debug();
071        debug |= sb.getOptions().debug();
072        
073        if (Options.v().verbose())
074            G.v().out.println("[" + sb.getMethod().getName() +
075                              "] Propagating and folding constants (SSA)...");
076
077        // *** FIXME: What happens when Shimple is built with another UnitGraph?
078        SCPFAnalysis scpf = new SCPFAnalysis(new ExceptionalUnitGraph(sb));
079
080        propagateResults(scpf.getResults());
081        if(pruneCFG){
082            removeStmts(scpf.getDeadStmts());
083            replaceStmts(scpf.getStmtsToReplace());
084        }
085    }
086
087    /**
088     * Propagates constants to the definition and uses of the relevant
089     * locals given a mapping.  Notice that we use the Shimple
090     * implementation of LocalDefs and LocalUses.
091     **/
092    protected void propagateResults(Map<Local, Constant> localToConstant)
093    {
094        Chain<Unit> units = sb.getUnits();
095        Collection<Local> locals = sb.getLocals();
096        ShimpleLocalDefs localDefs = new ShimpleLocalDefs(sb);
097        ShimpleLocalUses localUses = new ShimpleLocalUses(sb);
098        
099        Iterator<Local> localsIt = locals.iterator();
100        while(localsIt.hasNext()){
101            Local local = localsIt.next();
102            Constant constant = localToConstant.get(local);
103            
104            if(constant instanceof MetaConstant)
105                continue;
106
107            // update definition
108            {
109                DefinitionStmt stmt =
110                    (DefinitionStmt) localDefs.getDefsOf(local).get(0);
111
112                ValueBox defSrcBox = stmt.getRightOpBox();
113                Value defSrc = defSrcBox.getValue();
114                
115                if(defSrcBox.canContainValue(constant)){
116                    defSrcBox.setValue(constant);
117
118                    // remove dangling pointers
119                    if(defSrc instanceof UnitBoxOwner)
120                        ((UnitBoxOwner)defSrc).clearUnitBoxes();
121                }
122                else if(debug)
123                    G.v().out.println("Warning: Couldn't propagate constant " + constant + " to box " + defSrcBox.getValue() + " in unit " + stmt);
124            }
125            
126            // update uses
127            {
128                Iterator usesIt = localUses.getUsesOf(local).iterator();
129                
130                while(usesIt.hasNext()){
131                    UnitValueBoxPair pair = (UnitValueBoxPair) usesIt.next();
132                    ValueBox useBox = pair.getValueBox();
133
134                    if(useBox.canContainValue(constant))
135                       useBox.setValue(constant);
136                    else if(debug)
137                        G.v().out.println("Warning: Couldn't propagate constant " + constant + " to box " + useBox.getValue() + " in unit " + pair.getUnit());
138                }
139            }
140        }
141    }
142
143    /**
144     * Removes the given list of fall through IfStmts from the body.
145     **/
146    protected void removeStmts(List<IfStmt> deadStmts)
147    {
148        Chain units = sb.getUnits();
149        Iterator<IfStmt> deadIt = deadStmts.iterator();
150        while(deadIt.hasNext()){
151            Unit dead = deadIt.next();
152            units.remove(dead);
153            dead.clearUnitBoxes();
154        }
155    }
156
157    /**
158     * Replaces conditional branches by unconditional branches as
159     * given by the mapping.
160     **/
161    protected void replaceStmts(Map<Stmt, GotoStmt> stmtsToReplace)
162    {
163        Chain units = sb.getUnits();
164        Iterator<Stmt> stmtsIt = stmtsToReplace.keySet().iterator();
165        while(stmtsIt.hasNext()){
166            // important not to call clearUnitBoxes() on booted since
167            // replacement uses the same UnitBox
168            Unit booted = stmtsIt.next();
169            Unit replacement = stmtsToReplace.get(booted);
170            units.swapWith(booted, replacement);
171        }
172    }
173}

175/**
176 * The actual branching flow analysis implementation.  Briefly, a
177 * sketch of the sketch from the Cytron et al paper:
178 *
179 * <p> Initially the algorithm assumes that each edge is unexecutable
180 * (the entry nodes are reachable) and that each variable is constant
181 * with an unknown value, Top.  Assumptions are corrected until they
182 * stabilise.
183 *
184 * <p> For example, if <tt>q</tt> is found to be not a constant (Bottom)
185 * in <tt>if(q == 0) goto label1</tt> then both edges leaving the the
186 * statement are considered executable, if <tt>q</tt> is found to be a
187 * constant then only one of the edges are executable.
188 *
189 * <p> Whenever a reachable definition statement such as "x = 3" is
190 * found, the information is propagated to all uses of x (this works
191 * due to the SSA property).
192 *
193 * <p> Perhaps the crucial point is that if a node such as <tt>x =
194 * Phi(x_1, x_2)</tt> is ever found, information on <tt>x</tt> is
195 * assumed as follows:
196 *
197 * <ul>
198 *  <li>If <tt>x_1</tt> and <tt>x_2</tt> are the same assumed
199 *  constant, <tt>x</tt> is assumed to be that constant.  If they are
200 *  not the same constant, <tt>x</tt> is Bottom.</li>
201 *
202 *  <li>If either one is Top and the other is a constant, <tt>x</tt>
203 *  is assumed to be the same as the known constant.</li>
204 *
205 *  <li>If either is Bottom, <tt>x</tt> is assumed to be Bottom.</li>
206 * </ul>
207 *
208 * <p> The crucial point about the crucial point is that if
209 * definitions of <tt>x_1</tt> or <tt>x_2</tt> are never reached, the
210 * Phi node will still assume them to be Top and hence they will not
211 * influence the decision as to whether <tt>x</tt> is a constant or not.
212 **/
213class SCPFAnalysis extends ForwardBranchedFlowAnalysis
214{
215    protected FlowSet emptySet;
216
217    /**
218     * A mapping of the locals to their current assumed constant value
219     * (which may be Top or Bottom).
220     **/
221    protected Map<Local, Constant> localToConstant;
222
223    /**
224     * A map from conditional branches to their possible replacement 
225     * unit, an unconditional branch.
226     **/
227    protected Map<Stmt, GotoStmt> stmtToReplacement;
228
229    /**
230     * A list of IfStmts that always fall through.
231     **/
232    protected List<IfStmt> deadStmts;
233
234
235    /**
236     * Returns the localToConstant map.
237     **/
238    public Map<Local, Constant> getResults()
239    {
240        return localToConstant;
241    }
242
243    /**
244     * Returns the list of fall through IfStmts.
245     **/
246    public List<IfStmt> getDeadStmts()
247    {
248        return deadStmts;
249    }
250
251    /**
252     * Returns a Map from conditional branches to the unconditional branches
253     * that can replace them.
254     **/
255    public Map<Stmt, GotoStmt> getStmtsToReplace()
256    {
257        return stmtToReplacement;
258    }
259    
260    public SCPFAnalysis(UnitGraph graph)
261    {
262        super(graph);
263        emptySet = new ArraySparseSet();
264        stmtToReplacement = new HashMap<Stmt, GotoStmt>();
265        deadStmts = new ArrayList<IfStmt>();
266        
267        // initialise localToConstant map -- assume all scalars are
268        // constant (Top)
269        {
270                Collection<Local> locals = graph.getBody().getLocals();
271            Iterator<Local> localsIt = locals.iterator();
272            localToConstant = new HashMap<Local, Constant>(graph.size() * 2 + 1, 0.7f);
273
274            while(localsIt.hasNext()){
275                Local local = (Local) localsIt.next();
276                localToConstant.put(local, TopConstant.v());
277            }
278        }
279
280        doAnalysis();
281    }

283    // *** NOTE: this is here because ForwardBranchedFlowAnalysis does
284    // *** not handle exceptional control flow properly in the
285    // *** dataflow analysis.  this should be removed when
286    // *** ForwardBranchedFlowAnalysis is fixed.
287    protected boolean treatTrapHandlersAsEntries()
288    {
289        return true;
290    }

292    /**
293     * If a node has empty IN sets we assume that it is not reachable.
294     * Hence, we initialise the entry sets to be non-empty to indicate
295     * that they are reachable.
296     **/
297    protected Object entryInitialFlow()
298    {
299        FlowSet entrySet = (FlowSet) emptySet.emptySet();
300        entrySet.add(TopConstant.v());
301        return entrySet;
302    }
303
304    /**
305     * All other nodes are assumed to be unreachable by default.
306     **/
307    protected Object newInitialFlow()
308    {
309        return emptySet.emptySet();
310    }
311
312    /**
313     * Since we are interested in control flow from all branches,
314     * take the union.
315     **/
316    protected void merge(Object in1, Object in2, Object out)
317    {
318        FlowSet fin1 = (FlowSet) in1;
319        FlowSet fin2 = (FlowSet) in2;
320        FlowSet fout = (FlowSet) out;
321
322        fin1.union(fin2, fout);
323    }
324
325    /**
326     * Defer copy to FlowSet.
327     **/
328    protected void copy(Object source, Object dest)
329    {
330        FlowSet fource = (FlowSet) source;
331        FlowSet fest = (FlowSet) dest;
332
333        fource.copy(fest);
334    }

336    /**
337     * If a node has an empty in set, it is considered unreachable.
338     * Otherwise the node is examined and if any assumptions have to
339     * be corrected, a Pair containing the corrected assumptions is
340     * flowed to the reachable nodes.  If no assumptions have to be
341     * corrected then no information other than the in set is
342     * propagated to the reachable nodes.
343     *
344     * <p> Pair serves no other purpose than to keep the analysis
345     * flowing for as long as needed.  The final results are
346     * accumulated in the localToConstant map.
347     **/
348    protected void flowThrough(Object in, Unit s, List fallOut, List branchOuts)
349    {
350        FlowSet fin = ((FlowSet)in).clone();
351
352        // not reachable
353        if(fin.isEmpty())
354            return;
355        
356        // If s is a definition, check if any assumptions have to be
357        // corrected.
358        Pair pair = processDefinitionStmt(s);
359
360        if(pair != null)
361            fin.add(pair);
362        
363        // normal, non-branching statement
364        if(!s.branches() && s.fallsThrough()){
365            Iterator fallOutIt = fallOut.iterator();
366            while(fallOutIt.hasNext()){
367                FlowSet fallSet = (FlowSet) fallOutIt.next();
368                fallSet.union(fin);
369            }
370
371            return;
372        }
373
374        /* determine which nodes are reachable. */
375        
376        boolean conservative = true;
377        boolean fall = false;
378        boolean branch = false;
379        FlowSet oneBranch = null;
380        
381        IFSTMT:
382        {
383        if(s instanceof IfStmt){
384            IfStmt ifStmt = (IfStmt) s;
385            Value cond = ifStmt.getCondition();
386            Constant constant =
387                SEvaluator.getFuzzyConstantValueOf(cond, localToConstant);
388            
389            // flow both ways
390            if(constant instanceof BottomConstant){
391                deadStmts.remove(ifStmt);
392                stmtToReplacement.remove(ifStmt);
393                break IFSTMT;
394            }
395
396            // no flow
397            if(constant instanceof TopConstant)
398                return;
399
400            /* determine whether to flow through or branch */
401            
402            conservative = false;
403
404            Constant trueC = IntConstant.v(1);
405            Constant falseC = IntConstant.v(0);
406
407            if(constant.equals(trueC)){
408                branch = true;
409                GotoStmt gotoStmt =
410                    Jimple.v().newGotoStmt(ifStmt.getTargetBox());
411                stmtToReplacement.put(ifStmt, gotoStmt);
412            }
413
414            if(constant.equals(falseC)){
415                fall = true;
416                deadStmts.add(ifStmt);
417            }
418        }
419        } // end IFSTMT
420
421        TABLESWITCHSTMT:
422        {
423        if(s instanceof TableSwitchStmt){
424            TableSwitchStmt table = (TableSwitchStmt) s;
425            Value keyV = table.getKey();
426            Constant keyC =
427                SEvaluator.getFuzzyConstantValueOf(keyV, localToConstant);
428
429            // flow all branches
430            if(keyC instanceof BottomConstant){
431                stmtToReplacement.remove(table);
432                break TABLESWITCHSTMT;
433            }
434
435            // no flow
436            if(keyC instanceof TopConstant)
437                return;
438
439            // flow all branches
440            if(!(keyC instanceof IntConstant))
441                break TABLESWITCHSTMT;
442
443            /* find the one branch we need to flow to */
444
445            conservative = false;
446            
447            int key = ((IntConstant)keyC).value;
448            int low = table.getLowIndex();
449            int high = table.getHighIndex();
450            int index = key - low;
451
452            UnitBox branchBox = null;
453            if(index < 0 || index > high)
454                branchBox = table.getDefaultTargetBox();
455            else
456                branchBox = table.getTargetBox(index);
457
458            GotoStmt gotoStmt = Jimple.v().newGotoStmt(branchBox);
459            stmtToReplacement.put(table, gotoStmt);
460            
461            List unitBoxes = table.getUnitBoxes();
462            int setIndex = unitBoxes.indexOf(branchBox);
463            oneBranch = (FlowSet) branchOuts.get(setIndex);
464        }
465        } // end TABLESWITCHSTMT
466
467        LOOKUPSWITCHSTMT:
468        {
469        if(s instanceof LookupSwitchStmt){
470            LookupSwitchStmt lookup = (LookupSwitchStmt) s;
471            Value keyV = lookup.getKey();
472            Constant keyC =
473                SEvaluator.getFuzzyConstantValueOf(keyV, localToConstant);
474
475            // flow all branches
476            if(keyC instanceof BottomConstant){
477                stmtToReplacement.remove(lookup);
478                break LOOKUPSWITCHSTMT;
479            }
480
481            // no flow
482            if(keyC instanceof TopConstant)
483                return;
484
485            // flow all branches
486            if(!(keyC instanceof IntConstant))
487                break LOOKUPSWITCHSTMT;
488
489            /* find the one branch we need to flow to */
490
491            conservative = false;
492            
493            int index = lookup.getLookupValues().indexOf(keyC);
494
495            UnitBox branchBox = null;
496            if(index == -1)
497                branchBox = lookup.getDefaultTargetBox();
498            else
499                branchBox = lookup.getTargetBox(index);
500
501            GotoStmt gotoStmt = Jimple.v().newGotoStmt(branchBox);
502            stmtToReplacement.put(lookup, gotoStmt);
503            
504            List unitBoxes = lookup.getUnitBoxes();
505            int setIndex = unitBoxes.indexOf(branchBox);
506            oneBranch = (FlowSet) branchOuts.get(setIndex);
507        }
508        } // end LOOKUPSWITCHSTMT
509        
510        // conservative control flow estimates
511        if(conservative){
512            fall = s.fallsThrough();
513            branch = s.branches();
514        }
515
516        if(fall){
517            Iterator fallOutIt = fallOut.iterator();
518            while(fallOutIt.hasNext()){
519                FlowSet fallSet = (FlowSet) fallOutIt.next();
520                fallSet.union(fin);
521            }
522        }
523        
524        if(branch){
525            Iterator branchOutsIt = branchOuts.iterator();
526            while(branchOutsIt.hasNext()){
527                FlowSet branchSet = (FlowSet) branchOutsIt.next();
528                branchSet.union(fin);
529            }
530        }
531
532        if(oneBranch != null){
533            oneBranch.union(fin);
534        }
535    }

537    /**
538     * Returns (Unit, Constant) pair if an assumption has  changed 
539     * due to the fact that u is reachable.  Else returns null.
540     **/
541    protected Pair processDefinitionStmt(Unit u)
542    {
543        if(!(u instanceof DefinitionStmt))
544            return null;
545
546        DefinitionStmt dStmt = (DefinitionStmt) u;
547        
548        Local local;
549
550        {
551            Value value = dStmt.getLeftOp();
552            if(!(value instanceof Local))
553                return null;
554            local = (Local) value;
555        }
556
557        /* update assumptions */
558
559        Value rightOp = dStmt.getRightOp();
560        Constant constant =
561            SEvaluator.getFuzzyConstantValueOf(rightOp, localToConstant);
562        
563        if(!merge(local, constant))
564            return null;
565
566        return new Pair(u, localToConstant.get(local));
567    }
568    
569    /**
570     * Verifies if the given assumption "constant" changes the
571     * previous assumption about "local" and merges the information
572     * into the localToConstant map.  Returns true if something
573     * changed.
574     **/
575    protected boolean merge(Local local, Constant constant)
576    {
577        Constant current = localToConstant.get(local);
578
579        if(current instanceof BottomConstant) 
580            return false;
581
582        if(current instanceof TopConstant){
583            localToConstant.put(local, constant);
584            return true;
585        }
586
587        if(current.equals(constant))
588            return false;
589
590        // not equal
591        localToConstant.put(local, BottomConstant.v());
592        return true;
593    }
