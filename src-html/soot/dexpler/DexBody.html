148    PseudoInstruction isAddressInData (int a) {
149      for (PseudoInstruction pi: pseudoInstructionData) {
150        int fb = pi.getDataFirstByte();
151        int lb = pi.getDataLastByte();
152        if (fb <= a && a <= lb)
153          return pi;
154      }
155      return null;
156    }

158    /**
159     * @param code the codeitem that is contained in this body
160     * @param method the method that is associated with this body
161     */
162    public DexBody(DexFile dexFile, Method method, RefType declaringClassType) {
163        MethodImplementation code = method.getImplementation();
164        if (code == null)
165            throw new RuntimeException("error: no code for method "+ method.getName());
166        this.declaringClassType = declaringClassType;
167        tries = code.getTryBlocks();
168        methodSignature = method.getDefiningClass() +": "+ method.getReturnType() +" "+ method.getName() +"(";
169        for (MethodParameter mp: method.getParameters())
170            methodSignature += mp.getType() +",";
171
172        List<? extends CharSequence> paramTypes = method.getParameterTypes();
173        if (paramTypes != null) {
174            parameterTypes = new ArrayList<Type>();
175            for (CharSequence type : paramTypes)
176                parameterTypes.add(DexType.toSoot(type.toString()));
177        } else {
178                parameterTypes = Collections.emptyList();
179        }
180        
181        isStatic = Modifier.isStatic(method.getAccessFlags());
182        numRegisters = code.getRegisterCount();
183        numParameterRegisters = MethodUtil.getParameterRegisterCount(method);
184        if (!isStatic)
185            numParameterRegisters--;
186
187        instructions = new ArrayList<DexlibAbstractInstruction>();
188        instructionAtAddress = new HashMap<Integer, DexlibAbstractInstruction>();
189
190        registerLocals = new Local[numRegisters];
191
192        int address = 0;
193        
194        for (Instruction instruction : code.getInstructions()) {
195            DexlibAbstractInstruction dexInstruction = fromInstruction(instruction, address);
196            instructions.add(dexInstruction);
197            instructionAtAddress.put(address, dexInstruction);
198            Debug.printDbg(" put instruction '", dexInstruction ,"' at 0x", Integer.toHexString(address));
199            address += instruction.getCodeUnits();
200        }
201        
202        // Check taken from Android's dalvik/libdex/DexSwapVerify.cpp
203        if (numParameterRegisters > numRegisters)
204                throw new RuntimeException("Malformed dex file: insSize (" + numParameterRegisters
205                                + ") > registersSize (" + numRegisters + ")");
206        
207        for (DebugItem di: code.getDebugItems()) {
208            if (di instanceof ImmutableLineNumber) {
209                ImmutableLineNumber ln = (ImmutableLineNumber)di;
210                DexlibAbstractInstruction ins = instructionAtAddress(ln.getCodeAddress());
211                if (ins == null) {
212                        Debug.printDbg("Line number tag pointing to invalid offset: " + ln.getCodeAddress());
213                        continue;
214                }
215                ins.setLineNumber(ln.getLineNumber());
216                Debug.printDbg("Add line number tag " + ln.getLineNumber() + " for instruction: "
217                        + instructionAtAddress(ln.getCodeAddress()));
218            }
219        }
220
221        this.dexFile = dexFile;
222        this.method = method;
223    }

225    /**
226     * Return the types that are used in this body.
227     */
228    public Set<Type> usedTypes() {
229        Set<Type> types = new HashSet<Type>();
230        for (DexlibAbstractInstruction i : instructions)
231            types.addAll(i.introducedTypes());
232
233        if(tries!=null) {
234                for (TryBlock<? extends ExceptionHandler> tryItem : tries) {
235                    List<? extends ExceptionHandler> hList = tryItem.getExceptionHandlers();
236                        for (ExceptionHandler handler: hList) {
237                            String exType = handler.getExceptionType();
238                            if (exType == null) // for handler which capture all Exceptions
239                                continue;
240                            types.add(DexType.toSoot(exType));
241                        }
242                }
243        }
244
245        return types;
246    }

324    /**
325     * Return the instruction that is present at the byte code address.
326     *
327     * @param address the byte code address.
328     * @throws RuntimeException if address is not part of this body.
329     */
330    public DexlibAbstractInstruction instructionAtAddress(int address) {
331        DexlibAbstractInstruction i = null;
332        while (i == null && address >= 0) {
333            // catch addresses can be in the middlde of last instruction. Ex. in com.letang.ldzja.en.apk:
334            //
335            //          042c46: 7020 2a15 0100                         |008f: invoke-direct {v1, v0}, Ljavax/mi...
336            //          042c4c: 2701                                   |0092: throw v1
337            //          catches       : 4
338            //              <any> -> 0x0065
339            //            0x0069 - 0x0093
340                //
341                // SA, 14.05.2014: We originally scanned only two code units back. This is not sufficient
342                // if we e.g., have a wide constant and the line number in the debug sections points to
343                // some address the middle.
344                i = instructionAtAddress.get(address);
345                address--;
346        }
347        return i;
348    }

350    /**
351     * Return the jimple equivalent of this body.
352     *
353     * @param m the SootMethod that contains this body
354     */
355    public Body jimplify(Body b, SootMethod m) {
356
357                Timer t_whole_jimplification = new Timer();
358                Timer t_num = new Timer();
359                Timer t_null = new Timer();
360
361                t_whole_jimplification.start();
362
363        jBody = (JimpleBody)b;
364        localGenerator = new LocalGenerator(jBody);
365        deferredInstructions = new ArrayList<DeferableInstruction>();
366        instructionsToRetype = new HashSet<RetypeableInstruction>();
367
368        if (IDalvikTyper.ENABLE_DVKTYPER) {
369            Debug.printDbg(IDalvikTyper.DEBUG, "clear dalvik typer");
370            DalvikTyper.v().clear();
371        }
372
373        Debug.printDbg("\n[jimplify] start for: ", methodSignature);
374
375        // process method parameters and generate Jimple locals from Dalvik registers
376        List<Local> paramLocals = new LinkedList<Local>();
377        if (!isStatic) {
378            int thisRegister = numRegisters - numParameterRegisters - 1;
379
380            Local thisLocal = Jimple.v().newLocal("$u"+ thisRegister, UnknownType.v()); //generateLocal(UnknownType.v());
381            jBody.getLocals().add(thisLocal);
382
383            registerLocals[thisRegister] = thisLocal;
384            JIdentityStmt idStmt = (JIdentityStmt) Jimple.v().newIdentityStmt(thisLocal, Jimple.v().newThisRef(declaringClassType));
385            add(idStmt);
386            paramLocals.add(thisLocal);
387            if (IDalvikTyper.ENABLE_DVKTYPER) {
388                Debug.printDbg(IDalvikTyper.DEBUG, "constraint: ", idStmt);
389                DalvikTyper.v().setType(idStmt.leftBox, jBody.getMethod().getDeclaringClass().getType(), false);
390            }
391
392        }
393        {
394                int i = 0; // index of parameter type
395                int parameterRegister = numRegisters - numParameterRegisters; // index of parameter register
396                for (Type t: parameterTypes) {
397
398                    Local gen = Jimple.v().newLocal("$u"+ parameterRegister, UnknownType.v()); //may only use UnknownType here because the local may be reused with a different type later (before splitting)
399                    jBody.getLocals().add(gen);
400
401                    Debug.printDbg ("add local for parameter register number: ", parameterRegister);
402                    registerLocals[parameterRegister] = gen;
403                    JIdentityStmt idStmt = (JIdentityStmt) Jimple.v().newIdentityStmt(gen, Jimple.v().newParameterRef(t, i++));
404                    add(idStmt);
405                    paramLocals.add(gen);
406                    if (IDalvikTyper.ENABLE_DVKTYPER) {
407                        Debug.printDbg(IDalvikTyper.DEBUG, "constraint: "+ idStmt);
408                        DalvikTyper.v().setType(idStmt.leftBox, t, false);
409                    }
410
411                    // some parameters may be encoded on two registers.
412                    // in Jimple only the first Dalvik register name is used
413                    // as the corresponding Jimple Local name. However, we also add
414                    // the second register to the registerLocals array since it could be
415                    // used later in the Dalvik bytecode
416                    if (t instanceof LongType || t instanceof DoubleType) {
417                      parameterRegister++;
418                      Local g = Jimple.v().newLocal("$u"+ parameterRegister, UnknownType.v()); //may only use UnknownType here because the local may be reused with a different type later (before splitting)
419                      jBody.getLocals().add (g);
420                      registerLocals[parameterRegister] = g;
421                    }
422
423                    parameterRegister++;
424                }
425        }
426
427        for (int i = 0; i < (numRegisters - numParameterRegisters - (isStatic?0:1)); i++) {
428            Debug.printDbg ("add local for register number: ", i);
429            registerLocals[i] = Jimple.v().newLocal("$u"+ i, UnknownType.v());
430            jBody.getLocals().add(registerLocals[i]);
431        }
432
433        // add local to store intermediate results
434        storeResultLocal = Jimple.v().newLocal("$u-1", UnknownType.v());
435        jBody.getLocals().add (storeResultLocal);
436        
437        // process bytecode instructions
438        final boolean isOdex = dexFile instanceof DexBackedDexFile ?
439                        ((DexBackedDexFile) dexFile).isOdexFile() : false;
440        
441        ClassPath cp = null;
442        if (isOdex) {
443                String[] sootClasspath = Options.v().soot_classpath().split(File.pathSeparator);
444                List<String> classpathList = new ArrayList<String>();
445                for (String str : sootClasspath)
446                        classpathList.add(str);
447                try{
448                        ClassPathResolver resolver = new ClassPathResolver(classpathList, classpathList, classpathList, dexFile);
449                        cp = new ClassPath(resolver.getResolvedClassProviders().toArray(new ClassProvider[0]));
450                }catch(IOException e){
451                        throw new RuntimeException(e);
452                }
453        }
454
455        int prevLineNumber = -1;
456        for(DexlibAbstractInstruction instruction : instructions) {
457                if (isOdex && instruction instanceof OdexInstruction)
458                        ((OdexInstruction) instruction).deOdex(dexFile, method, cp);
459            if (dangling != null) {
460                dangling.finalize(this, instruction);
461                dangling = null;
462            }
463            //Debug.printDbg(" current op to jimplify: 0x", Integer.toHexString(instruction.getInstruction().opcode.value) ," instruction: ", instruction );
464            instruction.jimplify(this);
465            if (instruction.getLineNumber() > 0)
466                                prevLineNumber = instruction.getLineNumber();
467                        else {
468                                instruction.setLineNumber(prevLineNumber);
469                        }
470        }
471        for(DeferableInstruction instruction : deferredInstructions) {
472            instruction.deferredJimplify(this);
473        }
474        
475        if (tries != null)
476            addTraps();
477        
478        // At this point Jimple code is generated
479        // Cleaning...
480
481        instructions = null;
482        //registerLocals = null;
483        //storeResultLocal = null;
484        instructionAtAddress.clear();
485        //localGenerator = null;
486        deferredInstructions = null;
487        //instructionsToRetype = null;
488        dangling = null;
489        tries = null;
490
491        /* We eliminate dead code. Dead code has been shown to occur under the following
492         * circumstances.
493         *
494         *  0006ec: 0d00                                   |00a2: move-exception v0
495            ...
496                        0006f2: 0d00                                   |00a5: move-exception v0
497                        ...
498                0x0041 - 0x008a
499                  Ljava/lang/Throwable; -> 0x00a5
500                  <any> -> 0x00a2
501
502               Here there are two traps both over the same region. But the same always fires, hence
503               rendering the code at a2 unreachable.
504               Dead code yields problems during local splitting because locals within dead code
505               will not be split. Hence we remove all dead code here.
506         */
507
508                Debug.printDbg("body before any transformation : ", m, "\n", jBody);
509        
510        Debug.printDbg("\nbefore splitting");
511        Debug.printDbg("",(Body)jBody);
512        
513        // Fix traps that do not catch exceptions
514        DexTrapStackFixer.v().transform(jBody);
515        
516        // Sort out jump chains
517        DexJumpChainShortener.v().transform(jBody);
518        
519        // Make sure that we don't have any overlapping uses due to returns
520        DexReturnInliner.v().transform(jBody);    
521        
522        // Shortcut: Reduce array initializations
523        DexArrayInitReducer.v().transform(jBody);
524        
525        // split first to find undefined uses
526        getLocalSplitter().transform(jBody);
527        
528                // Remove dead code and the corresponding locals before assigning types
529                getUnreachableCodeEliminator().transform(jBody);
530                DeadAssignmentEliminator.v().transform(jBody);
531                UnusedLocalEliminator.v().transform(jBody);
532        
533        Debug.printDbg("\nafter splitting");
534        Debug.printDbg("",(Body)jBody);
535        
536                for (RetypeableInstruction i : instructionsToRetype)
537            i.retype(jBody);
538
539//        {
540//          // remove instructions from instructions list
541//          List<DexlibAbstractInstruction> iToRemove = new ArrayList<DexlibAbstractInstruction>();
542//          for (DexlibAbstractInstruction i: instructions)
543//            if (!jBody.getUnits().contains(i.getUnit()))
544//              iToRemove.add(i);
545//          for (DexlibAbstractInstruction i: iToRemove) {
546//            Debug.printDbg("removing dexinstruction containing unit '", i.getUnit() ,"'");
547//            instructions.remove(i);
548//          }
549//        }
550                
551        if (IDalvikTyper.ENABLE_DVKTYPER) {
552
553                        DexReturnValuePropagator.v().transform(jBody);
554                        getCopyPopagator().transform(jBody);
555                        DexNullThrowTransformer.v().transform(jBody);
556                        DalvikTyper.v().typeUntypedConstrantInDiv(jBody);
557                        DeadAssignmentEliminator.v().transform(jBody);
558                        UnusedLocalEliminator.v().transform(jBody);
559
560          Debug.printDbg("[DalvikTyper] resolving typing constraints...");
561          DalvikTyper.v().assignType(jBody);
562          Debug.printDbg("[DalvikTyper] resolving typing constraints... done.");
563          //jBody.validate();
564          jBody.validateUses();
565          jBody.validateValueBoxes();
566          //jBody.checkInit();
567                        //Validate.validateArrays(jBody);
568          //jBody.checkTypes();
569          //jBody.checkLocals();
570          Debug.printDbg("\nafter Dalvik Typer");
571
572        } else {
573                        t_num.start();
574                DexNumTransformer.v().transform (jBody);
575                        t_num.end();
576                
577                DexReturnValuePropagator.v().transform(jBody);
578            getCopyPopagator().transform(jBody);
579                
580                DexNullThrowTransformer.v().transform(jBody);
581
582                        t_null.start();
583                DexNullTransformer.v().transform(jBody);
584                        t_null.end();
585
586                DexIfTransformer.v().transform(jBody);
587                
588                DeadAssignmentEliminator.v().transform(jBody);
589                UnusedLocalEliminator.v().transform(jBody);
590                
591                //DexRefsChecker.v().transform(jBody);
592            DexNullArrayRefTransformer.v().transform(jBody);
593                
594                Debug.printDbg("\nafter Num and Null transformers");
595        }
596        Debug.printDbg("",(Body)jBody);
597        
598        if (IDalvikTyper.ENABLE_DVKTYPER) {
599            for (Local l: jBody.getLocals()) {
600                l.setType(UnknownType.v());
601            }
602        }
603        
604        TypeAssigner.v().transform(jBody);
605        
606                if (IDalvikTyper.ENABLE_DVKTYPER) {
607            for (Unit u: jBody.getUnits()) {
608                if (u instanceof IfStmt) {
609                    ConditionExpr expr = (ConditionExpr) ((IfStmt) u).getCondition();
610                    if (((expr instanceof EqExpr) || (expr instanceof NeExpr))) {
611                        Value op1 = expr.getOp1();
612                        Value op2 = expr.getOp2();
613                        if (op1 instanceof Constant && op2 instanceof Local) {
614                            Local l = (Local)op2;
615                            Type ltype = l.getType();
616                            if (ltype instanceof PrimType)
617                                continue;
618                            if (!(op1 instanceof IntConstant)) // by default null is IntConstant(0) in Dalvik
619                                continue;
620                            IntConstant icst = (IntConstant)op1;
621                            int val = icst.value;
622                            if (val != 0)
623                                continue;
624                            expr.setOp1(NullConstant.v());
625                        } else if (op1 instanceof Local && op2 instanceof Constant) {
626                            Local l = (Local)op1;
627                            Type ltype = l.getType();
628                            if (ltype instanceof PrimType)
629                                continue;
630                            if (!(op2 instanceof IntConstant)) // by default null is IntConstant(0) in Dalvik
631                                continue;
632                            IntConstant icst = (IntConstant)op2;
633                            int val = icst.value;
634                            if (val != 0)
635                                continue;
636                            expr.setOp2(NullConstant.v());
637                        } else if (op1 instanceof Local && op2 instanceof Local) {
638                                                        // nothing to do
639                                                } else if (op1 instanceof Constant && op2 instanceof Constant) {
640
641                                                        if (op1 instanceof NullConstant && op2 instanceof NumericConstant) {
642                                                                IntConstant nc = (IntConstant) op2;
643                                                                if (nc.value != 0)
644                                                                        throw new RuntimeException("expected value 0 for int constant. Got " + expr);
645                                                                expr.setOp2(NullConstant.v());
646                                                        } else if (op2 instanceof NullConstant && op1 instanceof NumericConstant) {
647                                                                IntConstant nc = (IntConstant) op1;
648                                                                if (nc.value != 0)
649                                                                        throw new RuntimeException("expected value 0 for int constant. Got " + expr);
650                                                                expr.setOp1(NullConstant.v());
651                                                        }
652                        } else {
653                            throw new RuntimeException("error: do not handle if: "+ u);
654                        }
655
656                    }
657                }
658            }
659            
660            // For null_type locals: replace their use by NullConstant()
661            List<ValueBox> uses = jBody.getUseBoxes();
662            //List<ValueBox> defs = jBody.getDefBoxes();
663            List<ValueBox> toNullConstantify = new ArrayList<ValueBox>();
664            List<Local> toRemove = new ArrayList<Local>();
665            for (Local l: jBody.getLocals()) {
666                
667                if (l.getType() instanceof NullType) {
668                    toRemove.add(l);
669                    for (ValueBox vb: uses) {
670                        Value v = vb.getValue();
671                        if (v == l)
672                            toNullConstantify.add(vb);
673                    }
674                }
675            }
676            for (ValueBox vb: toNullConstantify) {
677                System.out.println("replace valuebox '"+ vb +" with null constant");
678                vb.setValue(NullConstant.v());
679            }
680            for (Local l: toRemove) {
681                System.out.println("removing null_type local "+ l);
682                l.setType(RefType.v("java.lang.Object"));
683            }
684            
685
686        }
687
688        // We pack locals that are not used in overlapping regions. This may
689        // again lead to unused locals which we have to remove.
690        LocalPacker.v().transform(jBody);
691        UnusedLocalEliminator.v().transform(jBody);
692        LocalNameStandardizer.v().transform(jBody);
693
694        Debug.printDbg("\nafter type assigner localpacker and name standardizer");
695        Debug.printDbg("",(Body)jBody);
696        
697        // Inline PackManager.v().getPack("jb").apply(jBody);
698        // Keep only transformations that have not been done
699        // at this point.
700        TrapTightener.v().transform(jBody);
701                TrapMinimizer.v().transform(jBody);
702        //LocalSplitter.v().transform(jBody);
703        Aggregator.v().transform(jBody);
704        //UnusedLocalEliminator.v().transform(jBody);
705        //TypeAssigner.v().transform(jBody);
706        //LocalPacker.v().transform(jBody);
707        //LocalNameStandardizer.v().transform(jBody);
708        
709        // Remove if (null == null) goto x else <madness>. We can only do this
710        // after we have run the constant propagation as we might not be able
711        // to statically decide the conditions earlier.
712        ConditionalBranchFolder.v().transform(jBody);
713        
714        // Remove unnecessary typecasts
715        ConstantCastEliminator.v().transform(jBody);
716        
717        // We need to run this transformer since the conditional branch folder
718        // might have rendered some code unreachable (well, it was unreachable
719        // before as well, but we didn't know).
720        UnreachableCodeEliminator.v().transform(jBody);
721        
722        // Not sure whether we need this even though we do it earlier on as
723        // the earlier pass does not have type information
724        //CopyPropagator.v().transform(jBody);
725
726        // we might have gotten new dead assignments and unused locals through
727        // copy propagation and unreachable code elimination, so we have to do
728        // this again
729        DeadAssignmentEliminator.v().transform(jBody);
730        UnusedLocalEliminator.v().transform(jBody);
731        NopEliminator.v().transform(jBody);
732        
733        // Remove unnecessary chains of return statements
734        DexReturnPacker.v().transform(jBody);
735        
736        for (Unit u: jBody.getUnits()) {
737            if (u instanceof AssignStmt) {
738                AssignStmt ass = (AssignStmt)u;
739                if (ass.getRightOp() instanceof CastExpr) {
740                    CastExpr c = (CastExpr)ass.getRightOp();
741                    if (c.getType() instanceof NullType) {
742                        Debug.printDbg("replacing cast to null_type by nullConstant assignment in ", u);
743                        ass.setRightOp(NullConstant.v());
744                    }
745                }
746            }
747            if (u instanceof DefinitionStmt) {
748                DefinitionStmt def = (DefinitionStmt) u;
749                // If the body references a phantom class in a CaughtExceptionRef,
750                // we must manually fix the hierarchy
751                if (def.getLeftOp() instanceof Local
752                                && def.getRightOp() instanceof CaughtExceptionRef) {
753                        Type t = def.getLeftOp().getType();
754                        if (t instanceof RefType) {
755                                RefType rt = (RefType) t;
756                                if (rt.getSootClass().isPhantom()
757                                                && !rt.getSootClass().hasSuperclass()
758                                                && !rt.getSootClass().getName().equals("java.lang.Throwable"))
759                                        rt.getSootClass().setSuperclass(Scene.v().getSootClass("java.lang.Throwable"));
760                        }
761                }
762            }
763        }
764        
765        Debug.printDbg("\nafter jb pack");
766        Debug.printDbg("",(Body)jBody);
767
768        // Replace local type null_type by java.lang.Object.
769        //
770        // The typing engine cannot find correct type for such code:
771        //
772        // null_type $n0;
773        // $n0 = null;
774        // $r4 = virtualinvoke $n0.<java.lang.ref.WeakReference: java.lang.Object get()>();
775        //
776        for(Local l: jBody.getLocals()) {
777            Type t = l.getType();
778            if (t instanceof NullType) {
779                Debug.printDbg("replacing null_type by java.lang.Object for local ", l);
780                l.setType(RefType.v("java.lang.Object"));
781            }
782        }
783        
784                t_whole_jimplification.end();
785                Debug.printDbg("timer whole jimlification: ", t_whole_jimplification.getTime());
786                Debug.printDbg("timer num: ", t_num.getTime());
787                Debug.printDbg("timer null: ", t_null.getTime());
788
789        return jBody;
790    }

830     * Return the instructions that appear (lexically) after the given instruction.
831     *
832     * @param instruction the instruction which successors will be returned.
833     */
834    public List<DexlibAbstractInstruction> instructionsAfter(DexlibAbstractInstruction instruction) {
835        int i = instructions.indexOf(instruction);
836        if (i == -1)
837            throw new IllegalArgumentException("Instruction" + instruction + "not part of this body.");
838
839        return instructions.subList(i + 1, instructions.size());
840    }

842    /**
843     * Return the instructions that appear (lexically) before the given instruction.
844     *
845     * The instruction immediately before the given is the first instruction and so on.
846     *
847     * @param instruction the instruction which successors will be returned.
848     */
849    public List<DexlibAbstractInstruction> instructionsBefore(DexlibAbstractInstruction instruction) {
850        int i = instructions.indexOf(instruction);
851        if (i == -1)
852            throw new IllegalArgumentException("Instruction " + instruction + " not part of this body.");
853
854        List<DexlibAbstractInstruction> l = new ArrayList<DexlibAbstractInstruction>();
855        l.addAll(instructions.subList(0 , i));
856        Collections.reverse(l);
857        return l;
858    }

860    /**
861     * Add the traps of this body.
862     *
863     * Should only be called at the end jimplify.
864     */
865    private void addTraps() {
866      for (TryBlock<? extends ExceptionHandler> tryItem : tries) {
867            int startAddress = tryItem.getStartCodeAddress();
868            Debug.printDbg(" start : 0x", Integer.toHexString(startAddress));
869            int length = tryItem.getCodeUnitCount();//.getTryLength();
870            Debug.printDbg(" length: 0x", Integer.toHexString(length));
871            Debug.printDbg(" end   : 0x", Integer.toHexString(startAddress + length));
872            int endAddress = startAddress + length;// - 1;
873            Unit beginStmt = instructionAtAddress(startAddress).getUnit();
874            // (startAddress + length) typically points to the first byte of the first instruction after the try block
875            // except if there is no instruction after the try block in which case it points to the last byte of the last
876            // instruction of the try block. Removing 1 from (startAddress + length) always points to "somewhere" in
877            // the last instruction of the try block since the smallest instruction is on two bytes (nop = 0x0000).
878            Unit endStmt =  instructionAtAddress (endAddress).getUnit();
879                        // if the try block ends on the last instruction of the body, add a
880                        // nop instruction so Soot can include
881                        // the last instruction in the try block.
882                        if (jBody.getUnits().getLast() == endStmt
883                                        && instructionAtAddress(endAddress - 1).getUnit() == endStmt) {
884                                Unit nop = Jimple.v().newNopStmt();
885                                jBody.getUnits().insertAfter(nop, endStmt);
886                                endStmt = nop;
887                        }
888                        Debug.printDbg("begin instruction (0x",
889                                        Integer.toHexString(startAddress), "): ",
890                                        instructionAtAddress(startAddress).getUnit(), " --- ",
891                                        beginStmt);
892                        Debug.printDbg("end instruction   (0x",
893                                        Integer.toHexString(endAddress), "): ",
894                                        instructionAtAddress(endAddress).getUnit(), " --- ",
895                                        endStmt);
896
897
898            List<? extends ExceptionHandler> hList = tryItem.getExceptionHandlers();
899            for (ExceptionHandler handler: hList) {
900              int handlerAddress = handler.getHandlerCodeAddress();
901              Debug.printDbg("handler   (0x",
902                          Integer.toHexString(handlerAddress),
903                          "): ",
904                          instructionAtAddress (handlerAddress).getUnit(),
905                          " --- ",
906                          handlerAddress > 0 ? instructionAtAddress (handlerAddress-1).getUnit() : "<unknown>");
907              String exceptionType = handler.getExceptionType();
908              if (exceptionType == null)
909                  exceptionType = "Ljava/lang/Throwable;";
910              Type t = DexType.toSoot(exceptionType);
911                // exceptions can only be of RefType
912                if (t instanceof RefType) {
913                    SootClass exception = ((RefType) t).getSootClass();
914                    DexlibAbstractInstruction instruction = instructionAtAddress(handler.getHandlerCodeAddress());
915                    if (! (instruction instanceof MoveExceptionInstruction))
916                        Debug.printDbg("First instruction of trap handler unit not MoveException but " , instruction.getClass());
917                    else
918                      ((MoveExceptionInstruction) instruction).setRealType(this, exception.getType());
919
920                    Trap trap = Jimple.v().newTrap(exception, beginStmt, endStmt, instruction.getUnit());
921                    jBody.getTraps().add(trap);
922                }
923            }
924        }
925    }
