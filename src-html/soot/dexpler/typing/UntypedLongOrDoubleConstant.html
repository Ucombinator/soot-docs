001package soot.dexpler.typing;
002
003import soot.DoubleType;
004import soot.LongType;
005import soot.Type;
006import soot.Value;
007import soot.jimple.DoubleConstant;
008import soot.jimple.LongConstant;
009
010public class UntypedLongOrDoubleConstant extends UntypedConstant {
011
012    /**
013         * 
014         */
015        private static final long serialVersionUID = -3970057807907204253L;
016        public final long value;
017
018    private UntypedLongOrDoubleConstant(long value)
019    {
020        this.value = value;
021    }
022
023    public static UntypedLongOrDoubleConstant v(long value)
024    {
025        return new UntypedLongOrDoubleConstant(value);
026    }
027
028    public boolean equals(Object c)
029    {
030        return c instanceof UntypedLongOrDoubleConstant && ((UntypedLongOrDoubleConstant) c).value == this.value;
031    }
032
033    /** Returns a hash code for this DoubleConstant object. */
034    public int hashCode()
035    {
036        return (int)(value^(value>>>32));
037    }
038    
039    public DoubleConstant toDoubleConstant() {
040        return DoubleConstant.v(Double.longBitsToDouble(value));
041    }
042    
043    public LongConstant toLongConstant() {
044        return LongConstant.v(value);
045    }
046    
047        @Override
048        public Value defineType(Type t) {
049                if (t instanceof DoubleType) {
050                        return this.toDoubleConstant();
051                } else if (t instanceof LongType) {
052                        return this.toLongConstant();
053                } else {
054                        throw new RuntimeException("error: expected Double type or Long type. Got " + t);
055                }
056        }
057
058}
