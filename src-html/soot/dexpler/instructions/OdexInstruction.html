001package soot.dexpler.instructions;
002
003import org.jf.dexlib2.analysis.ClassPath;
004import org.jf.dexlib2.iface.DexFile;
005import org.jf.dexlib2.iface.Method;
006
007/**
008 * Interface for instructions that are only valid in optimized dex files (ODEX).
009 * These instructions require special handling for de-odexing.
010 * 
011 * @author Steven Arzt
012 */
013public interface OdexInstruction {
014        
015        /**
016         * De-odexes the current instruction.
017         * @param parentFile The parent file to which the current ODEX instruction
018         * belongs
019         */
020        public void deOdex(DexFile parentFile, Method m, ClassPath cp);
021
022}
