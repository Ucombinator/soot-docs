025package soot.dexpler.instructions;
026
027import org.jf.dexlib2.iface.instruction.Instruction;
028import org.jf.dexlib2.iface.instruction.OneRegisterInstruction;
029
030import soot.Local;
031import soot.RefType;
032import soot.dexpler.Debug;
033import soot.dexpler.DexBody;
034import soot.dexpler.IDalvikTyper;
035import soot.dexpler.typing.DalvikTyper;
036import soot.jimple.EnterMonitorStmt;
037import soot.jimple.Jimple;
038
039public class MonitorEnterInstruction extends DexlibAbstractInstruction {
040
041    EnterMonitorStmt enterMonitorStmt = null;
042  
043    public MonitorEnterInstruction (Instruction instruction, int codeAdress) {
044        super(instruction, codeAdress);
045    }
046
047    public void jimplify (DexBody body) {
048        int reg = ((OneRegisterInstruction) instruction).getRegisterA();
049        Local object = body.getRegisterLocal(reg);
050        enterMonitorStmt = Jimple.v().newEnterMonitorStmt(object);
051        setUnit(enterMonitorStmt);
052        addTags(enterMonitorStmt);
053        body.add(enterMonitorStmt);
054        
055                if (IDalvikTyper.ENABLE_DVKTYPER) {
056                        Debug.printDbg(IDalvikTyper.DEBUG, "constraint: "+ enterMonitorStmt);
057                        DalvikTyper.v().setType(enterMonitorStmt.getOpBox(), RefType.v("java.lang.Object"), true);
058        }
059    }
060}
