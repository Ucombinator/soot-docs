060/**
061 * BodyTransformer to find and change initialization type of Jimple variables.
062 * Dalvik bytecode does not provide enough information regarding the type of
063 * initialized variables. For instance, using the dexdump disassembler on some
064 * Dalvik bytecode can produce the following (wrong) output:
065 * 
066 * 006c : const -wide v6 , #double 0.000000 // #0014404410000000 0071: and-long
067 * /2 addr v6 , v4
068 * 
069 * At 0x6c, the initialized register is not of type double, but of type long
070 * because it is used in a long and operation at 0x71. Thus, one need to check
071 * how the register is used to deduce its type. By default, and since the
072 * dexdump disassembler does not perform such analysis, it supposes the register
073 * is of type double.
074 * 
075 * Dalvik comes with the following instructions to initialize constants: 0x12
076 * const/4 vx,lit4 0x13 const/16 vx,lit16 0x14 const vx, lit32 0x15 const/high16
077 * v0, lit16 0x16 const-wide/16 vx, lit16 0x17 const-wide/32 vx, lit32 0x18
078 * const-wide vx, lit64 0x19 const-wide/high16 vx,lit16 0x1A const-string
079 * vx,string id 0x1B const-string-jumbo vx,string 0x1C const-class vx,type id
080 *
081 * Instructions 0x12, 0x1A, 0x1B, 0x1C can not produce wrong initialized
082 * registers. The other instructions are converted to the following Jimple
083 * statement: JAssignStmt ( Local, rightValue ). Since at the time of the
084 * statement creation the no analysis can be performed, a default type is given
085 * to rightValue. This default type is "int" for registers whose size is less or
086 * equal to 32bits and "long" to registers whose size is 64bits. The problem is
087 * that 32bits registers could be either "int" or "float" and 64bits registers
088 * "long" or "double". If the analysis concludes that an "int" has to be changed
089 * to a "float", rightValue has to change from IntConstant.v(literal) to
090 * Float.intBitsToFloat((int) literal). If the analysis concludes that an "long"
091 * has to be changed to a "double, rightValue has to change from
092 * LongConstant.v(literal) to
093 * DoubleConstant.v(Double.longBitsToDouble(literal)).
094 */
095public class DexNumTransformer extends DexTransformer {
096        // Note: we need an instance variable for inner class access, treat this as
097        // a local variable (including initialization before use)
098
099        private boolean usedAsFloatingPoint;
100        boolean doBreak = false;
101
102        public static DexNumTransformer v() {
103                return new DexNumTransformer();
104        }
105        
106        protected void internalTransform(final Body body, String phaseName, Map<String,String> options) {
107                final DexDefUseAnalysis localDefs = new DexDefUseAnalysis(body);
108                
109        for (Local loc : getNumCandidates(body)) {
110            Debug.printDbg("\n[num candidate] ", loc);
111                        usedAsFloatingPoint = false;
112                        Set<Unit> defs = localDefs.collectDefinitionsWithAliases(loc);
113                        
114                // process normally
115                        doBreak = false;
116                        for (Unit u : defs) {
117                                // put correct local in l
118                                final Local l = u instanceof DefinitionStmt ? (Local) ((DefinitionStmt) u).getLeftOp()
119                                                : null;
120                                
121                        Debug.printDbg("    def  : ", u);
122                                Debug.printDbg("    local: ", l);
123
124                                // check defs
125                                u.apply(new AbstractStmtSwitch() {
126                                        public void caseAssignStmt(AssignStmt stmt) {
127                                                Value r = stmt.getRightOp();
128                                                if (r instanceof BinopExpr && !(r instanceof CmpExpr)) {
129                                                        usedAsFloatingPoint = examineBinopExpr((Unit) stmt);
130                                                        doBreak = true;
131                                                } else if (r instanceof FieldRef) {
132                                                        usedAsFloatingPoint = isFloatingPointLike(((FieldRef) r)
133                                                                        .getFieldRef().type());
134                                                        doBreak = true;
135                                                } else if (r instanceof NewArrayExpr) {
136                                                        NewArrayExpr nae = (NewArrayExpr) r;
137                                                        Type t = nae.getType();
138                                                        Debug.printDbg("new array expr: ", nae, " type: ",
139                                                                        t);
140                                                        usedAsFloatingPoint = isFloatingPointLike(t);
141                                                        doBreak = true;
142                                                } else if (r instanceof ArrayRef) {
143                                                        ArrayRef ar = (ArrayRef) r;
144                                                        Type arType = ar.getType();
145                                                        Debug.printDbg("ar: ", r, " ", arType);
146                                                        if (arType instanceof UnknownType) {
147                                                                Type t = findArrayType(localDefs,
148                                                                                stmt, 0, Collections.<Unit> emptySet()); // TODO:
149                                                                                                                                                                        // check
150                                                                                                                                                                        // where
151                                                                                                                                                                        // else
152                                                                                                                                                                        // to
153                                                                                                                                                                        // update
154                                                                                                                                                                        // if(ArrayRef...
155                                                                Debug.printDbg(" array type:", t);
156                                                                usedAsFloatingPoint = isFloatingPointLike(t);
157                                                        } else {
158                                                                usedAsFloatingPoint = isFloatingPointLike(ar
159                                                                                .getType());
160                                                        }
161                                                        doBreak = true;
162                                                } else if (r instanceof CastExpr) {
163                                                        usedAsFloatingPoint = isFloatingPointLike(((CastExpr) r)
164                                                                        .getCastType());
165                                                        doBreak = true;
166                                                } else if (r instanceof InvokeExpr) {
167                                                        usedAsFloatingPoint = isFloatingPointLike(((InvokeExpr) r)
168                                                                        .getType());
169                                                        doBreak = true;
170                                                } else if (r instanceof LengthExpr) {
171                                                        usedAsFloatingPoint = false;
172                                                        doBreak = true;
173                                                }
174                                                // introduces alias
175                                                else if (r instanceof Local) {
176                                                }
177
178                                        }
179
180                                        public void caseIdentityStmt(IdentityStmt stmt) {
181                                                Debug.printDbg("h");
182                                                if (stmt.getLeftOp() == l) {
183                                                        usedAsFloatingPoint = isFloatingPointLike(stmt
184                                                                        .getRightOp().getType());
185                                                        doBreak = true;
186                                                }
187                                        }
188                                });
189
190                                if (doBreak) {
191                                        break;
192                                }
193
194                                // check uses
195                                for (Unit use : localDefs.getUsesOf(l)) {
196                                        Debug.printDbg("    use: ", use);
197
198                                        use.apply(new AbstractStmtSwitch() {
199                                                private boolean examineInvokeExpr(InvokeExpr e) {
200                                                        List<Value> args = e.getArgs();
201                                                        List<Type> argTypes = e.getMethodRef()
202                                                                        .parameterTypes();
203                                                        assert args.size() == argTypes.size();
204                                                        for (int i = 0; i < args.size(); i++) {
205                                                                if (args.get(i) == l
206                                                                                && isFloatingPointLike(argTypes.get(i))) {
207                                                                        return true;
208                                                                }
209                                                        }
210                                                        return false;
211                                                }
212
213                                                public void caseInvokeStmt(InvokeStmt stmt) {
214                                                        InvokeExpr e = stmt.getInvokeExpr();
215                                                        usedAsFloatingPoint = examineInvokeExpr(e);
216                                                }
217
218                                                public void caseAssignStmt(AssignStmt stmt) {
219                                                        // only case where 'l' could be on the left side is
220                                                        // arrayRef with 'l' as the index
221                                                        Value left = stmt.getLeftOp();
222                                                        if (left instanceof ArrayRef) {
223                                                                ArrayRef ar = (ArrayRef) left;
224                                                                if (ar.getIndex() == l) {
225                                                                        doBreak = true;
226                                                                        return;
227                                                                }
228                                                        }
229
230                                                        // from this point, we only check the right hand
231                                                        // side of the assignment
232                                                        Value r = stmt.getRightOp();
233                                                        if (r instanceof ArrayRef) {
234                                                                if (((ArrayRef) r).getIndex() == l) {
235                                                                        doBreak = true;
236                                                                        return;
237                                                                }
238                                                        } else if (r instanceof InvokeExpr) {
239                                                                usedAsFloatingPoint = examineInvokeExpr((InvokeExpr) r);
240                                                                doBreak = true;
241                                                                return;
242                                                        } else if (r instanceof BinopExpr) {
243                                                                usedAsFloatingPoint = examineBinopExpr((Unit) stmt);
244                                                                doBreak = true;
245                                                                return;
246                                                        } else if (r instanceof CastExpr) {
247                                                                usedAsFloatingPoint = stmt.hasTag("FloatOpTag")
248                                                                                || stmt.hasTag("DoubleOpTag");
249                                                                doBreak = true;
250                                                                return;
251                                                        } else if (r instanceof Local && r == l) {
252                                                                if (left instanceof FieldRef) {
253                                                                        FieldRef fr = (FieldRef) left;
254                                                                        if (isFloatingPointLike(fr.getType())) {
255                                                                                usedAsFloatingPoint = true;
256                                                                        }
257                                                                        doBreak = true;
258                                                                        return;
259                                                                } else if (left instanceof ArrayRef) {
260                                                                        ArrayRef ar = (ArrayRef) left;
261                                                                        Type arType = ar.getType();
262                                                                        Debug.printDbg("ar: ", r, " ", arType);
263                                                                        if (arType instanceof UnknownType) {
264                                                                                arType = findArrayType(localDefs, stmt, 0, Collections.<Unit> emptySet());
265                                                                        }
266                                                                        Debug.printDbg(" array type:", arType);
267                                                                        usedAsFloatingPoint = isFloatingPointLike(arType);
268                                                                        doBreak = true;
269                                                                        return;
270                                                                }
271                                                        }
272
273                                                }
274
275                                                public void caseReturnStmt(ReturnStmt stmt) {
276                                                        usedAsFloatingPoint = stmt.getOp() == l
277                                                                        && isFloatingPointLike(body.getMethod()
278                                                                                        .getReturnType());
279                                                        Debug.printDbg(" [return stmt] ", stmt,
280                                                                        " usedAsObject: ", usedAsFloatingPoint,
281                                                                        ", return type: ", body.getMethod()
282                                                                                        .getReturnType());
283                                                        Debug.printDbg(" class: ", body.getMethod()
284                                                                        .getReturnType().getClass());
285                                                        doBreak = true;
286                                                        return;
287                                                }
288                                        });
289
290                                        if (doBreak)
291                                                break;
292
293                                } // for uses
294                                if (doBreak)
295                                        break;
296                        } // for defs
297
298                        // change values
299                        if (usedAsFloatingPoint) {
300                                for (Unit u : defs) {
301                                        replaceWithFloatingPoint(u);
302                                }
303                        } // end if
304
305                }
306        }
307
308        protected boolean examineBinopExpr(Unit u) {
309                if (u.hasTag("FloatOpTag") || u.hasTag("DoubleOpTag")) {
310                        return true;
311                }
312                return false;
313        }
314
315        private boolean isFloatingPointLike(Type t) {
316                return (t instanceof FloatType || t instanceof DoubleType);
317        }
318
319        /**
320         * Collect all the locals which are assigned a IntConstant(0) or are used
321         * within a zero comparison.
322         *
323         * @param body
324         *            the body to analyze
325         */
326        private Set<Local> getNumCandidates(Body body) {
327                Set<Local> candidates = new HashSet<Local>();
328                for (Unit u : body.getUnits()) {
329                        if (u instanceof AssignStmt) {
330                                AssignStmt a = (AssignStmt) u;
331                                if (!(a.getLeftOp() instanceof Local))
332                                        continue;
333                                Local l = (Local) a.getLeftOp();
334                                Value r = a.getRightOp();
335                                if ((r instanceof IntConstant || r instanceof LongConstant)) {
336                                        candidates.add(l);
337                                        Debug.printDbg("[add num candidate: ", u);
338                                }
339                        }
340                }
341
342                return candidates;
343        }
344
345        /**
346         * Replace 0 with null in the given unit.
347         *
348         * @param u
349         *            the unit where 0 will be replaced with null.
350         */
351        private void replaceWithFloatingPoint(Unit u) {
352                if (u instanceof AssignStmt) {
353                        AssignStmt s = (AssignStmt) u;
354                        Value v = s.getRightOp();
355                        if ((v instanceof IntConstant)) {
356                                int vVal = ((IntConstant) v).value;
357                                s.setRightOp(FloatConstant.v(Float.intBitsToFloat((int) vVal)));
358                                Debug.printDbg("[floatingpoint] replacing with float in ", u);
359                        } else if (v instanceof LongConstant) {
360                                long vVal = ((LongConstant) v).value;
361                                s.setRightOp(DoubleConstant.v(Double
362                                                .longBitsToDouble((long) vVal)));
363                                Debug.printDbg("[floatingpoint] replacing with double in ", u);
364                        }
365                }
366
367        }
