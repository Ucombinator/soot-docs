020package soot;
021import soot.SourceLocator.FoundFile;
022import soot.jimple.parser.lexer.LexerException;
023import soot.jimple.parser.parser.ParserException;
024import soot.options.*;
025import soot.javaToJimple.IInitialResolver.Dependencies;
026import soot.jimple.*;
027
028import java.io.*;
029import java.util.*;
030
031/** A class source for resolving from .jimple files using the Jimple parser.
032 */
033public class JimpleClassSource extends ClassSource {
034        
035        private FoundFile foundFile;
036        
037    public JimpleClassSource(String className, FoundFile foundFile) {
038        super(className);
039        if(foundFile == null)
040                        throw new IllegalStateException("Error: The FoundFile must not be null.");
041        this.foundFile = foundFile;
042    }
043    public Dependencies resolve( SootClass sc ) {
044        if(Options.v().verbose())
045            G.v().out.println("resolving [from .jimple]: " + className );
046        
047        InputStream classFile = null;
048        try {
049                classFile = foundFile.inputStream();
050                soot.jimple.parser.JimpleAST jimpAST = new soot.jimple.parser.JimpleAST(classFile);
051                jimpAST.getSkeleton(sc);
052                JimpleMethodSource mtdSrc = new JimpleMethodSource(jimpAST);
053
054                Iterator<SootMethod> mtdIt = sc.methodIterator();
055                while(mtdIt.hasNext()) {
056                        SootMethod sm = mtdIt.next();
057                        sm.setSource(mtdSrc);
058                }
059
060                Dependencies deps = new Dependencies();
061                //The method documentation states it returns RefTypes only, so this is a transformation safe
062                for (String t : jimpAST.getCstPool()){
063                        deps.typesToSignature.add(RefType.v(t));
064                }
065
066                return deps;
067        } catch (IOException e) { 
068                throw new RuntimeException("Error: Failed to create JimpleAST from source input stream for class " + className + ".",e); 
069        } catch (ParserException e) {
070                throw new RuntimeException("Error: Failed when parsing class " + className + ".",e);
071        } catch (LexerException e) {
072                throw new RuntimeException("Error: Failed when lexing class " + className + ".", e);
073        } finally {
074                try {
075                        if(classFile != null){
076                                classFile.close();
077                                classFile = null;
078                        }
079                } catch(IOException e) {
080                        throw new RuntimeException("Error: Failed to close source input stream.",e);
081                } finally {
082                        close();
083                }
084        }
085    }
086    
087        @Override
088        public void close() {
089                if(foundFile != null){
090                        foundFile.close();
091                        foundFile = null;
092                }
093        }
094    
095}
