059/**
060 * Abstract super-class for ASM-based back-ends. Generates byte-code for
061 * everything except the method bodies, as they are dependent on the IR.
062 * 
063 * @author Tobias Hamann, Florian Kuebler, Dominik Helm, Lukas Sommer
064 * 
065 */
066public abstract class AbstractASMBackend {
067
068        // An ASM ClassVisitor that is used to emit the bytecode to
069        protected ClassVisitor cv;
070        // The SootClass that is to be converted into bytecode
071        protected SootClass sc;
072        // The Java version to be used for generating this class
073        protected int javaVersion;

075        private final Map<SootMethod, BafBody> bafBodyCache = new HashMap<SootMethod, BafBody>();

077        /**
078         * Creates a new ASM backend
079         * @param sc The SootClass that is to be converted into bytecode
080         * @param javaVersion A particular Java version enforced by the user, may be 0 for automatic detection, must not be lower than necessary for all features used
081         */
082        public AbstractASMBackend(SootClass sc, int javaVersion) {
083                this.sc = sc;
084                
085                int minVersion = getMinJavaVersion(sc);         
086
087                if (javaVersion == 0)
088                        javaVersion = Options.java_version_default;
089                
090                if(javaVersion != Options.java_version_default && javaVersion < minVersion){
091                        throw new IllegalArgumentException("Enforced Java version " + translateJavaVersion(javaVersion) + " too low to support required features (" + translateJavaVersion(minVersion) + " required)");
092                }
093                
094                javaVersion = Math.max(javaVersion, minVersion);
095
096                switch (javaVersion) {
097                case Options.java_version_1_1:
098                        this.javaVersion = Opcodes.V1_1;
099                        break;
100                case Options.java_version_1_2:
101                        this.javaVersion = Opcodes.V1_2;
102                        break;
103                case Options.java_version_1_3:
104                        this.javaVersion = Opcodes.V1_3;
105                        break;
106                case Options.java_version_1_4:
107                        this.javaVersion = Opcodes.V1_4;
108                        break;
109                case Options.java_version_1_5:
110                        this.javaVersion = Opcodes.V1_5;
111                        break;
112                case Options.java_version_1_6:
113                        this.javaVersion = Opcodes.V1_6;
114                        break;
115                case Options.java_version_1_7:
116                        this.javaVersion = Opcodes.V1_7;
117                        break;
118                case Options.java_version_1_8:
119                        this.javaVersion = Opcodes.V1_8;
120                        break;
121                }
122        }

124        /**
125         * Gets the baf body for the given SootMethod. This method will first check
126         * whether the method already has a baf body. If not, it will query the local
127         * cache. If this fails as well, it will construct a new baf body.
128         * @param method The method for which to obtain a baf body
129         * @return The baf body for the given method
130         */
131        protected BafBody getBafBody(SootMethod method) {
132                final Body activeBody = method.getActiveBody();
133                if (activeBody instanceof BafBody)
134                        return (BafBody) activeBody;
135
136                BafBody body = bafBodyCache.get(method);
137                if (body != null)
138                        return body;
139                
140                if (activeBody instanceof JimpleBody) {
141                        body = PackManager.v().convertJimpleBodyToBaf(method);
142                } else {
143                        throw new RuntimeException(
144                                        "ASM-backend can only translate Baf- and JimpleBodies!");
145                }
146                
147                bafBodyCache.put(method, body);
148                return body;
149        }

151        /**
152         * Determines the minimum Java version required for the bytecode of the given SootClass
153         * @param sc The SootClass the minimum Java version is to be determined for
154         * @return The minimum Java version required for the given SootClass
155         */
156        private int getMinJavaVersion(SootClass sc) {
157                int minVersion = Options.java_version_1_1;
158                
159                if(sc.hasTag("SignatureTag")){
160                        if(((SignatureTag)sc.getTag("SignatureTag")).getSignature().contains("<")){
161                                minVersion=Options.java_version_1_5;
162                        }
163                }
164                if(sc.hasTag("VisibilityAnnotationTag")){
165                        minVersion = Options.java_version_1_5;
166                }
167                if(Modifier.isAnnotation(sc.getModifiers())){
168                        // Class is defining an annotation
169                        minVersion = Options.java_version_1_5;
170                }
171                
172                for(SootField sf : sc.getFields()){
173                        if(minVersion >= Options.java_version_1_5){
174                                break;
175                        }
176                        if(sf.hasTag("SignatureTag")){
177                                if(((SignatureTag)sf.getTag("SignatureTag")).getSignature().contains("<")){
178                                        minVersion=Options.java_version_1_5;
179                                }
180                        }
181                        if(sf.hasTag("VisibilityAnnotationTag")){
182                                minVersion = Options.java_version_1_5;
183                        }
184                }
185                
186                for(SootMethod sm : sc.getMethods()){
187                        if(minVersion >= Options.java_version_1_8){
188                                break;
189                        }
190                        if(sm.hasTag("SignatureTag")){
191                                if(((SignatureTag)sm.getTag("SignatureTag")).getSignature().contains("<")){
192                                        minVersion = Math.max(minVersion, Options.java_version_1_5);
193                                }
194                        }
195                        if(sm.hasTag("VisibilityAnnotationTag") || sm.hasTag("VisibilityParameterAnnotationTag")){
196                                minVersion = Math.max(minVersion, Options.java_version_1_5);
197                        }
198                        if(sm.hasActiveBody()){
199                                minVersion = Math.max(minVersion, getMinJavaVersion(sm));
200                        }
201                }
202                return minVersion;
203        }

215        /**
216         * Outputs the bytecode generated as a class file
217         * @param os The OutputStream the class file is written to
218         */
219        public void generateClassFile(OutputStream os) {
220                ClassWriter cw = new SootASMClassWriter(ClassWriter.COMPUTE_FRAMES);
221                cv = cw;
222                generateByteCode();
223                try {
224                        os.write(cw.toByteArray());
225                } catch (IOException e) {
226                        throw new RuntimeException(
227                                        "Could not write class file in the ASM-backend!", e);
228                }
229        }

240        /**
241         * Emits the bytecode for the complete class 
242         */
243        protected void generateByteCode() {
244                generateClassHeader();
245
246                // Retrieve information about the source of the class
247                if (sc.hasTag("SourceFileTag")
248                                && !Options.v().no_output_source_file_attribute()) {
249                        String srcName = ((SourceFileTag) sc.getTag("SourceFileTag"))
250                                        .getSourceFile();
251                        cv.visitSource(srcName, null); // TODO Correct value for the debug
252                                                                                        // argument
253                }
254
255                // Retrieve information about outer class if present
256                if (sc.hasOuterClass() || sc.hasTag("EnclosingMethodTag")
257                                || sc.hasTag("OuterClassTag")) {
258                        generateOuterClassReference();
259                }
260
261                // Retrieve information about annotations
262                generateAnnotations(cv, sc);
263
264                // Retrieve information about attributes
265                generateAttributes();
266
267                // Retrieve information about inner classes
268                generateInnerClassReferences();
269
270                // Generate fields
271                generateFields();
272
273                // Generate methods
274                generateMethods();
275
276                cv.visitEnd();
277        }

279        /**
280         * Comparatator that is used to sort the methods before they are written
281         * out. This is mainly used to enforce a deterministic output between runs
282         * which we need for testing.
283         * 
284         * @author Steven Arzt
285         *
286         */
287        private class SootMethodComparator implements Comparator<SootMethod> {
288
289                @Override
290                public int compare(SootMethod o1, SootMethod o2) {
291                        return o1.getName().compareTo(o2.getName());
292                }
293                
294        }

296        /**
297         * Emits the bytecode for all methods of the class
298         */
299        protected void generateMethods() {
300                List<SootMethod> sortedMethods = new ArrayList<SootMethod>(sc.getMethods());
301                Collections.sort(sortedMethods, new SootMethodComparator());
302                for (SootMethod sm : sortedMethods) {
303                        int access = getModifiers(sm.getModifiers(), sm);
304                        String name = sm.getName();
305                        StringBuilder descBuilder = new StringBuilder(5);
306                        descBuilder.append('(');
307                        for (Type t : sm.getParameterTypes()) {
308                                descBuilder.append(toTypeDesc(t));
309                        }
310                        descBuilder.append(')');
311                        descBuilder.append(toTypeDesc(sm.getReturnType()));
312
313                        String sig = null;
314                        if (sm.hasTag("SignatureTag")) {
315                                SignatureTag genericSignature = (SignatureTag) sm
316                                                .getTag("SignatureTag");
317                                sig = genericSignature.getSignature();
318                        }
319
320                        String[] exceptions = new String[sm.getExceptions().size()];
321                        int i = 0;
322                        for (SootClass exc : sm.getExceptions()) {
323                                exceptions[i] = slashify(exc.getName());
324                                ++i;
325                        }
326                        MethodVisitor mv = cv.visitMethod(access, name,
327                                        descBuilder.toString(), sig, exceptions);
328                        if (mv != null) {
329                                // Visit parameter annotations
330                                for (Tag t : sm.getTags()) {
331                                        if (t instanceof VisibilityParameterAnnotationTag) {
332                                                VisibilityParameterAnnotationTag vpt = (VisibilityParameterAnnotationTag) t;
333                                                ArrayList<VisibilityAnnotationTag> tags = vpt
334                                                                .getVisibilityAnnotations();
335                                                if (tags != null) {
336                                                        for (int j = 0; j < tags.size(); ++j) {
337                                                                VisibilityAnnotationTag va = tags.get(j);
338                                                                if (va == null) {
339                                                                        continue;
340                                                                }
341                                                                for (AnnotationTag at : va.getAnnotations()) {
342                                                                        AnnotationVisitor av = mv
343                                                                                        .visitParameterAnnotation(
344                                                                                                        j,
345                                                                                                        at.getType(),
346                                                                                                        (va.getVisibility() == AnnotationConstants.RUNTIME_VISIBLE));
347                                                                        generateAnnotationElems(av, at.getElems(), true);
348                                                                }
349                                                        }
350                                                }
351                                        }
352                                }
353
354                                generateAnnotations(mv, sm);
355
356                                generateAttributes(mv, sm);
357                                if (sm.hasActiveBody()) {
358                                        mv.visitCode();
359                                        generateMethodBody(mv, sm);
360                                        /*
361                                         * Correct values are computed automatically by ASM,
362                                         * but we need the call anyways.
363                                         */
364                                        mv.visitMaxs(0, 0);
365                                }
366                                mv.visitEnd();
367                        }
368                }
369        }

371        /**
372         * Emits the bytecode for all fields of the class
373         */
374        protected void generateFields() {
375        for (SootField f : sc.getFields()) {
376            String name = f.getName();
377            String desc = toTypeDesc(f.getType());
378            String sig = null;
379            if (f.hasTag("SignatureTag")) {
380                SignatureTag genericSignature = (SignatureTag) f
381                        .getTag("SignatureTag");
382                sig = genericSignature.getSignature();
383            }
384            Object value = getDefaultValue(f);
385            int access = getModifiers(f.getModifiers(), f);
386            FieldVisitor fv = cv.visitField(access, name, desc, sig, value);
387            if (fv != null) {
388                generateAnnotations(fv, f);
389                generateAttributes(fv, f);
390                fv.visitEnd();
391            }
392        }
393        }

413        /**
414         * Emits the bytecode for all references to inner classes if present
415         */
416        protected void generateInnerClassReferences() {
417                if (sc.hasTag("InnerClassAttribute")
418                                && !Options.v().no_output_inner_classes_attribute()) {
419                        InnerClassAttribute ica = (InnerClassAttribute) sc
420                                        .getTag("InnerClassAttribute");
421                        List<InnerClassTag> sortedTags = new ArrayList<InnerClassTag>(ica.getSpecs());
422                        Collections.sort(sortedTags, new SootInnerClassComparator());
423                        for (InnerClassTag ict : sortedTags) {
424                                String name = slashify(ict.getInnerClass());
425                                String outerClassName = slashify(ict.getOuterClass());
426                                String innerName = slashify(ict.getShortName());
427                                int access = ict.getAccessFlags();
428                                cv.visitInnerClass(name, outerClassName, innerName, access);
429                        }
430                }
431        }

472    /**
473     * Emits the bytecode for all annotations of a class, field or method
474     * @param visitor A ClassVisitor, FieldVisitor or MethodVisitor to emit the bytecode to
475     * @param host A Host (SootClass, SootField or SootMethod) the bytecode is to be emitted for, has to match the visitor
476     */
477        protected void generateAnnotations(Object visitor, Host host) {
478                for (Tag t : host.getTags()) {
479                        // Find all VisibilityAnnotationTags
480                        if (t instanceof VisibilityAnnotationTag) {
481                                VisibilityAnnotationTag vat = (VisibilityAnnotationTag) t;
482                                boolean runTimeVisible = (vat.getVisibility() == AnnotationConstants.RUNTIME_VISIBLE);
483                                for (AnnotationTag at : vat.getAnnotations()) {
484                                        AnnotationVisitor av = null;
485                                        if (visitor instanceof ClassVisitor) {
486                                                av = ((ClassVisitor) visitor).visitAnnotation(
487                                                                at.getType(), runTimeVisible);
488                                        } else if (visitor instanceof FieldVisitor) {
489                                                av = ((FieldVisitor) visitor).visitAnnotation(
490                                                                at.getType(), runTimeVisible);
491                                        } else if (visitor instanceof MethodVisitor) {
492                                                av = ((MethodVisitor) visitor).visitAnnotation(
493                                                                at.getType(), runTimeVisible);
494                                        }
495
496                                        generateAnnotationElems(av, at.getElems(), true);
497                                }
498                        }
499                        /*
500                         * Here TypeAnnotations could be visited potentially. Currently
501                         * (2015/02/03) they are not supported by the ASM-front-end and
502                         * their information is not accessible.
503                         */
504
505                        // Visit AnnotationDefault on methods
506                        else if (host instanceof SootMethod
507                                        && t instanceof AnnotationDefaultTag) {
508                                AnnotationDefaultTag adt = (AnnotationDefaultTag) t;
509                                AnnotationVisitor av = ((MethodVisitor) visitor)
510                                                .visitAnnotationDefault();
511                                generateAnnotationElems(av,
512                                                Collections.singleton(adt.getDefaultVal()), true);
513                        }
514
515                }
516        }

518        /**
519         * Emits the bytecode for the values of an annotation
520         * @param av The AnnotationVisitor to emit the bytecode to
521         * @param elements A collection of AnnatiotionElem that are the values of the annotation
522         * @param addName True, if the name of the annotation has to be added, false otherwise (should be false only in recursive calls!)
523         */
524        protected void generateAnnotationElems(AnnotationVisitor av,
525                        Collection<AnnotationElem> elements, boolean addName) {
526                if (av != null) {
527                        Iterator<AnnotationElem> it = elements.iterator();
528                        while (it.hasNext()) {
529                                AnnotationElem elem = it.next();
530                                if (elem instanceof AnnotationEnumElem) {
531                                        AnnotationEnumElem enumElem = (AnnotationEnumElem) elem;
532                                        av.visitEnum(enumElem.getName(), enumElem.getTypeName(),
533                                                        enumElem.getConstantName());
534                                } else if (elem instanceof AnnotationArrayElem) {
535                                        AnnotationArrayElem arrayElem = (AnnotationArrayElem) elem;
536                                        AnnotationVisitor arrayVisitor = av.visitArray(arrayElem
537                                                        .getName());
538                                        generateAnnotationElems(arrayVisitor, arrayElem.getValues(), false);
539                                } else if (elem instanceof AnnotationAnnotationElem) {
540                                        AnnotationAnnotationElem aElem = (AnnotationAnnotationElem) elem;
541                                        AnnotationVisitor aVisitor = av.visitAnnotation(
542                                                        aElem.getName(), aElem.getValue().getType());
543                                        generateAnnotationElems(aVisitor, aElem.getValue()
544                                                        .getElems(), true);
545                                } else {
546                                        Object val = null;
547                                        if (elem instanceof AnnotationIntElem) {
548                                                AnnotationIntElem intElem = (AnnotationIntElem) elem;
549                                                int value = intElem.getValue();
550                                                switch (intElem.getKind()) {
551                                                case 'B':
552                                                        val = (byte) value;
553                                                        break;
554                                                case 'Z':
555                                                        val = (value == 1);
556                                                        break;
557                                                case 'I':
558                                                        val = value;
559                                                        break;
560                                                case 'S':
561                                                        val = (short) value;
562                                                        break;
563                                                }
564                                        } else if (elem instanceof AnnotationBooleanElem) {
565                                                AnnotationBooleanElem booleanElem = (AnnotationBooleanElem) elem;
566                                                val = booleanElem.getValue();
567                                        } else if (elem instanceof AnnotationFloatElem) {
568                                                AnnotationFloatElem floatElem = (AnnotationFloatElem) elem;
569                                                val = floatElem.getValue();
570                                        } else if (elem instanceof AnnotationLongElem) {
571                                                AnnotationLongElem longElem = (AnnotationLongElem) elem;
572                                                val = longElem.getValue();
573                                        } else if (elem instanceof AnnotationDoubleElem) {
574                                                AnnotationDoubleElem doubleElem = (AnnotationDoubleElem) elem;
575                                                val = doubleElem.getValue();
576                                        } else if (elem instanceof AnnotationStringElem) {
577                                                AnnotationStringElem stringElem = (AnnotationStringElem) elem;
578                                                val = stringElem.getValue();
579                                        } else if (elem instanceof AnnotationClassElem) {
580                                                AnnotationClassElem classElem = (AnnotationClassElem) elem;
581                                                val = org.objectweb.asm.Type.getType(classElem
582                                                                .getDesc());
583                                        }
584                                        if(addName){
585                                                av.visit(elem.getName(), val);
586                                        }
587                                        else{
588                                                av.visit(null, val);
589                                        }
590                                        
591                                }
592                        }
593                        av.visitEnd();
594                }
595        }

597        /**
598         * Emits the bytecode for a reference to an outer class if necessary
599         */
600        protected void generateOuterClassReference() {
601                SootClass outerClass = sc.getOuterClass();
602                String outerClassName = slashify(outerClass.getName());
603                String enclosingMethod = null;
604                String enclosingMethodSig = null;
605                if (sc.hasTag("EnclosingMethodTag")) {
606                        EnclosingMethodTag emTag = (EnclosingMethodTag) sc
607                                        .getTag("EnclosingMethodTag");
608                        if (!sc.hasOuterClass()) {
609                                outerClassName = slashify(emTag.getEnclosingClass());
610                        }
611                        enclosingMethod = emTag.getEnclosingMethod();
612                        enclosingMethodSig = emTag.getEnclosingMethodSig();
613                }
614                if (!sc.hasOuterClass() && sc.hasTag("OuterClassTag")) {
615                        outerClassName = slashify(((OuterClassTag) sc
616                                        .getTag("OuterClassTag")).getName());
617                }
618                cv.visitOuterClass(outerClassName, enclosingMethod, enclosingMethodSig);
619        }

621        /**
622         * Emits the bytecode for the class itself, including its signature
623         */
624        protected void generateClassHeader() {
625                /*
626                 * Retrieve all modifiers
627                 */
628                int classModifiers = sc.getModifiers();
629                int modifier = getModifiers(classModifiers, sc);
630
631                // Retrieve class-name
632                String className = slashify(sc.getName());
633                // Retrieve generics
634                String signature = null;
635                if (sc.hasTag("SignatureTag")) {
636                        signature = ((SignatureTag) sc.getTag("SignatureTag"))
637                                        .getSignature();
638                }
639                /*
640                 * Retrieve super-class If no super-class is explicitly given, the
641                 * default is java.lang.Object.
642                 */
643                String superClass = "java/lang/Object";
644                if (sc.hasSuperclass()) {
645                        superClass = slashify(sc.getSuperclass().getName());
646                }
647
648                // Retrieve directly implemented interfaces
649                String[] interfaces = new String[sc.getInterfaceCount()];
650                Iterator<SootClass> implementedInterfaces = sc.getInterfaces()
651                                .iterator();
652                int i = 0;
653                while (implementedInterfaces.hasNext()) {
654                        SootClass interf = implementedInterfaces.next();
655                        interfaces[i] = slashify(interf.getName());
656                        ++i;
657                }
658
659                cv.visit(javaVersion, modifier, className, signature, superClass,
660                                interfaces);
661        }

663        /**
664         * Utility method to get the access modifiers of a Host
665         * @param modVal The bitset representation of the Host's modifiers
666         * @param host The Host (SootClass, SootField or SootMethod) the modifiers are to be retrieved from
667         * @return A bitset representation of the Host's modifiers in ASM's internal representation
668         */
669        protected static int getModifiers(int modVal, Host host) {
670                int modifier = 0;
671                // Retrieve visibility-modifier
672                if (Modifier.isPublic(modVal)) {
673                        modifier |= Opcodes.ACC_PUBLIC;
674                } else if (Modifier.isPrivate(modVal)) {
675                        modifier |= Opcodes.ACC_PRIVATE;
676                } else if (Modifier.isProtected(modVal)) {
677                        modifier |= Opcodes.ACC_PROTECTED;
678                }
679                // Retrieve static-modifier
680                if (Modifier.isStatic(modVal)
681                                && ((host instanceof SootField) || (host instanceof SootMethod))) {
682                        modifier |= Opcodes.ACC_STATIC;
683                }
684                // Retrieve final-modifier
685                if (Modifier.isFinal(modVal)) {
686                        modifier |= Opcodes.ACC_FINAL;
687                }
688                // Retrieve synchronized-modifier
689                if (Modifier.isSynchronized(modVal) && host instanceof SootMethod) {
690                        modifier |= Opcodes.ACC_SYNCHRONIZED;
691                }
692                // Retrieve volatile/bridge-modifier
693                if (Modifier.isVolatile(modVal) && !(host instanceof SootClass)) {
694                        modifier |= Opcodes.ACC_VOLATILE;
695                }
696                // Retrieve transient/varargs-modifier
697                if (Modifier.isTransient(modVal) && !(host instanceof SootClass)) {
698                        modifier |= Opcodes.ACC_TRANSIENT;
699                }
700                // Retrieve native-modifier
701                if (Modifier.isNative(modVal) && host instanceof SootMethod) {
702                        modifier |= Opcodes.ACC_NATIVE;
703                }
704                // Retrieve interface-modifier
705                if (Modifier.isInterface(modVal) && host instanceof SootClass) {
706                        modifier |= Opcodes.ACC_INTERFACE;
707                } else if (host instanceof SootClass) {
708                        /*
709                         * For all classes except for interfaces the super-flag should be
710                         * set. See JVM 8-Specification section 4.1, page 72.
711                         */
712                        modifier |= Opcodes.ACC_SUPER;
713                }
714                // Retrieve abstract-modifier
715                if (Modifier.isAbstract(modVal) && !(host instanceof SootField)) {
716                        modifier |= Opcodes.ACC_ABSTRACT;
717                }
718                // Retrieve strictFP-modifier
719                if (Modifier.isStrictFP(modVal) && host instanceof SootMethod) {
720                        modifier |= Opcodes.ACC_STRICT;
721                }
722                /*
723                 * Retrieve synthetic-modifier. Class not present in source-code but
724                 * generated by e.g. compiler TODO Do we need both checks?
725                 */
726                if (Modifier.isSynthetic(modVal) || host.hasTag("SyntheticTag")) {
727                        modifier |= Opcodes.ACC_SYNTHETIC;
728                }
729                // Retrieve annotation-modifier
730                if (Modifier.isAnnotation(modVal) && host instanceof SootClass) {
731                        modifier |= Opcodes.ACC_ANNOTATION;
732                }
733                // Retrieve enum-modifier
734                if (Modifier.isEnum(modVal) && !(host instanceof SootMethod)) {
735                        modifier |= Opcodes.ACC_ENUM;
736                }
737                return modifier;
738        }

740        /**
741         * Emits the bytecode for the body of a single method
742         * Has to be implemented by subclasses to suit their needs
743         * @param mv The MethodVisitor to emit the bytecode to 
744         * @param method The SootMethod the bytecode is to be emitted for
745         */
746        abstract protected void generateMethodBody(MethodVisitor mv,
747                        SootMethod method);
