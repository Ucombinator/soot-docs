042/** Represents the class hierarchy.  It is closely linked to a Scene,
043 * and must be recreated if the Scene changes. 
044 *
045 * The general convention is that if a method name contains 
046 * "Including", then it returns the non-strict result; otherwise,
047 * it does a strict query (e.g. strict superclass).  */
048public class Hierarchy
049{
050    // These two maps are not filled in the constructor.
051    private Map<SootClass, List<SootClass>> classToSubclasses;
052    private Map<SootClass, List<SootClass>> interfaceToSubinterfaces;
053    private Map<SootClass, List<SootClass>> interfaceToSuperinterfaces;
054
055    private Map<SootClass, List<SootClass>> classToDirSubclasses;
056    private Map<SootClass, List<SootClass>> interfaceToDirSubinterfaces;
057    private Map<SootClass, List<SootClass>> interfaceToDirSuperinterfaces;
058
059    // This holds the direct implementers.
060    private Map<SootClass, List<SootClass>> interfaceToDirImplementers;
061
062    int state;
063    Scene sc;

065    /** Constructs a hierarchy from the current scene. */
066    public Hierarchy()
067    {
068        this.sc = Scene.v();
069        state = sc.getState();
070
071        // Well, this used to be describable by 'Duh'.
072        // Construct the subclasses hierarchy and the subinterfaces hierarchy.
073        {
074            Chain<SootClass> allClasses = sc.getClasses();
075
076            classToSubclasses = new HashMap<SootClass, List<SootClass>>(allClasses.size() * 2 + 1, 0.7f);
077            interfaceToSubinterfaces = new HashMap<SootClass, List<SootClass>>(allClasses.size() * 2 + 1, 0.7f);
078            interfaceToSuperinterfaces = new HashMap<SootClass, List<SootClass>>(allClasses.size() * 2 + 1, 0.7f);
079            
080            classToDirSubclasses = new HashMap<SootClass, List<SootClass>>
081                (allClasses.size() * 2 + 1, 0.7f);
082            interfaceToDirSubinterfaces = new HashMap<SootClass, List<SootClass>>
083                (allClasses.size() * 2 + 1, 0.7f);
084            interfaceToDirSuperinterfaces = new HashMap<SootClass, List<SootClass>>
085                (allClasses.size() * 2 + 1, 0.7f);
086            interfaceToDirImplementers = new HashMap<SootClass, List<SootClass>>
087                (allClasses.size() * 2 + 1, 0.7f);
088            
089            for (SootClass c : allClasses) {
090                if( c.resolvingLevel() < SootClass.HIERARCHY ) continue;
091
092                if (c.isInterface())
093                {
094                    interfaceToDirSubinterfaces.put(c, new ArrayList<SootClass>());
095                    interfaceToDirSuperinterfaces.put(c, new ArrayList<SootClass>());
096                    interfaceToDirImplementers.put(c, new ArrayList<SootClass>());
097                }
098                else
099                    classToDirSubclasses.put(c, new ArrayList<SootClass>());
100            }
101
102            for (SootClass c : allClasses) {
103                if( c.resolvingLevel() < SootClass.HIERARCHY ) continue;
104                
105
106                if (c.hasSuperclass()) {
107                    if (c.isInterface()) {
108                        List<SootClass> l2 = interfaceToDirSuperinterfaces.get(c);
109                        for (SootClass i : c.getInterfaces()) {
110                            if( c.resolvingLevel() < SootClass.HIERARCHY ) continue;
111                            List<SootClass> l = interfaceToDirSubinterfaces.get(i);
112                            if (l != null) l.add(c);
113                            if (l2 != null) l2.add(i);
114                        }
115                    }
116                    else {
117                        List<SootClass> l = classToDirSubclasses.get(c.getSuperclass());                        
118                        if (l != null) l.add(c);
119
120                        for (SootClass i : c.getInterfaces()) {
121                            if( c.resolvingLevel() < SootClass.HIERARCHY ) continue;
122                            l = interfaceToDirImplementers.get(i);
123                            if (l != null) l.add(c);
124                        }
125                    }
126                }
127            }
128
129            // Fill the directImplementers lists with subclasses.
130            for (SootClass c : allClasses) {
131                if( c.resolvingLevel() < SootClass.HIERARCHY ) continue;
132                if (c.isInterface()) {
133                        List<SootClass> imp = interfaceToDirImplementers.get(c);
134                        Set<SootClass> s = new ArraySet<SootClass>();                       
135                        for (SootClass c0 : imp) {
136                                if( c.resolvingLevel() < SootClass.HIERARCHY ) continue;
137                                s.addAll(getSubclassesOfIncluding(c0));
138                        }
139                        imp.clear();
140                        imp.addAll(s);
141                }
142            }
143        }
144    }

146    private void checkState()
147    {
148        if (state != sc.getState())
149            throw new ConcurrentModificationException("Scene changed for Hierarchy!");
150    }

167    /** Returns a list of subclasses of c, excluding itself. */
168    public List<SootClass> getSubclassesOf(SootClass c)
169    {
170        c.checkLevel(SootClass.HIERARCHY);
171        if (c.isInterface())
172            throw new RuntimeException("class needed!");
173
174        checkState();
175
176        // If already cached, return the value.
177        if (classToSubclasses.get(c) != null)
178            return classToSubclasses.get(c);        
179        
180        // Otherwise, build up the hashmap.
181        List<SootClass> l = new ArrayList<SootClass>();
182
183        for (SootClass cls : classToDirSubclasses.get(c)) {
184            if( cls.resolvingLevel() < SootClass.HIERARCHY ) continue;
185            l.addAll(getSubclassesOfIncluding(cls));
186        }
187        
188        l = Collections.unmodifiableList(l);
189        classToSubclasses.put(c, l);
190
191        return l;
192    }

203    /** Returns a list of strict superclasses of c, starting with c's parent. */
204    public List<SootClass> getSuperclassesOf(SootClass c)
205    {
206        c.checkLevel(SootClass.HIERARCHY);
207        if (c.isInterface())
208            throw new RuntimeException("class needed!");
209
210        checkState();
211
212        ArrayList<SootClass> l = new ArrayList<SootClass>();
213        SootClass cl = c;
214
215        while (cl.hasSuperclass())
216        {
217            l.add(cl.getSuperclass());
218            cl = cl.getSuperclass();
219        }
220
221        return Collections.unmodifiableList(l);
222    }

404    /**
405     * Returns true if child is a subclass of possibleParent. If one of the
406     * known parent classes is phantom, we conservatively assume that the
407     * current class might be a child.
408     */
409    public boolean isClassSubclassOf(SootClass child, SootClass possibleParent)
410    {
411        child.checkLevel(SootClass.HIERARCHY);
412        possibleParent.checkLevel(SootClass.HIERARCHY);
413        List<SootClass> parentClasses = getSuperclassesOf(child);
414        if (parentClasses.contains(possibleParent))
415                return true;
416        
417        for (SootClass sc : parentClasses)
418                if (sc.isPhantom())
419                        return true;
420        
421        return false;
422    }

511    /** Checks whether check is a visible class in view of the from class.
512     *  It assumes that protected and private classes do not exit.
513     *  If they exist and check is either protected or private,
514     *  the check will return false. */
515    public boolean isVisible( SootClass from, SootClass check) {
516        if (check.isPublic())
517                return true;
518        
519        if( check.isProtected() || check.isPrivate())
520                return false;
521                
522        //package visibility
523        return from.getJavaPackageName().equals(
524                check.getJavaPackageName() );
525    }

527    /** Returns true if the classmember m is visible from code in the class from. */
528    public boolean isVisible( SootClass from, ClassMember m ) {
529        from.checkLevel(SootClass.HIERARCHY);
530        m.getDeclaringClass().checkLevel(SootClass.HIERARCHY);
531        
532        if (!isVisible(from, m.getDeclaringClass()))
533                return false;
534        
535        if( m.isPublic() ) return true;
536        if( m.isPrivate() ) {
537            return from.equals( m.getDeclaringClass() );
538        }
539        if( m.isProtected() ) {
540            return isClassSubclassOfIncluding( from, m.getDeclaringClass() );
541        }
542        // m is package
543        return from.getJavaPackageName().equals(
544                m.getDeclaringClass().getJavaPackageName() );
545            //|| isClassSubclassOfIncluding( from, m.getDeclaringClass() );
546    }

549        /**
550         * Given an object of actual type C (o = new C()), returns the method which
551         * will be called on an o.f() invocation.
552         */
553        public SootMethod resolveConcreteDispatch(SootClass concreteType,
554                        SootMethod m) {
555                concreteType.checkLevel(SootClass.HIERARCHY);
556                m.getDeclaringClass().checkLevel(SootClass.HIERARCHY);
557                checkState();
558
559                if (concreteType.isInterface())
560                        throw new RuntimeException("class needed!");
561
562                String methodSig = m.getSubSignature();
563
564                for (SootClass c : getSuperclassesOfIncluding(concreteType)) {
565                        SootMethod sm = c.getMethodUnsafe(methodSig); 
566                        if (sm != null && isVisible(c, m)) {
567                                return sm;
568                        }
569                }
570                throw new RuntimeException(
571                                "could not resolve concrete dispatch!\nType: " + concreteType
572                                                + "\nMethod: " + m);
573        }

575    /** Given a set of definite receiver types, returns a list of possible targets. */
576    public List<SootMethod> resolveConcreteDispatch(List<Type> classes, SootMethod m)
577    {
578        m.getDeclaringClass().checkLevel(SootClass.HIERARCHY);
579        checkState();
580
581        Set<SootMethod> s = new ArraySet<SootMethod>();
582        for (Type cls : classes) {
583            if (cls instanceof RefType)
584                s.add(resolveConcreteDispatch(((RefType)cls).getSootClass(), m));
585            else if (cls instanceof ArrayType) {
586                s.add(resolveConcreteDispatch((RefType.v("java.lang.Object")).getSootClass(), m));
587            }
588            else throw new RuntimeException("Unable to resolve concrete dispatch of type "+ cls);
589        }
590
591        return Collections.unmodifiableList(new ArrayList<SootMethod>(s));
592    }

594    // what can get called for c & all its subclasses
595    /** Given an abstract dispatch to an object of type c and a method m, gives
596     * a list of possible receiver methods. */
597    public List<SootMethod> resolveAbstractDispatch(SootClass c, SootMethod m) 
598    {
599        c.checkLevel(SootClass.HIERARCHY);
600        m.getDeclaringClass().checkLevel(SootClass.HIERARCHY);
601        checkState();
602
603        Set<SootMethod> s = new ArraySet<SootMethod>();
604        Collection<SootClass> classesIt;
605        
606        if (c.isInterface()) {
607            Set<SootClass> classes = new HashSet<SootClass>();
608            for (SootClass sootClass : getImplementersOf(c)) {
609                classes.addAll(getSubclassesOfIncluding(sootClass));
610            }
611            classesIt = classes;
612        }
613        else
614            classesIt = getSubclassesOfIncluding(c);
615
616        for (SootClass cl : classesIt) {
617            if( !Modifier.isAbstract( cl.getModifiers() ) ) {
618                s.add(resolveConcreteDispatch(cl, m));
619            }
620        }
621
622        return Collections.unmodifiableList(new ArrayList<SootMethod>(s));
623    }

625    // what can get called if you have a set of possible receiver types
626    /** Returns a list of possible targets for the given method and set of receiver types. */
627    public List<SootMethod> resolveAbstractDispatch(List<SootClass> classes, SootMethod m)
628    {
629        m.getDeclaringClass().checkLevel(SootClass.HIERARCHY);
630        Set<SootMethod> s = new ArraySet<SootMethod>();
631        for (SootClass sootClass : classes ) {
632            s.addAll(resolveAbstractDispatch(sootClass, m));
633        }
634        
635        return Collections.unmodifiableList(new ArrayList<SootMethod>(s));
636    }

638    /** Returns the target for the given SpecialInvokeExpr. */
639    public SootMethod resolveSpecialDispatch(SpecialInvokeExpr ie, SootMethod container)
640    {
641        container.getDeclaringClass().checkLevel(SootClass.HIERARCHY);
642        SootMethod target = ie.getMethod();
643        target.getDeclaringClass().checkLevel(SootClass.HIERARCHY);
644
645        /* This is a bizarre condition!  Hopefully the implementation is correct.
646           See VM Spec, 2nd Edition, Chapter 6, in the definition of invokespecial. */
647        if ("<init>".equals(target.getName()) || target.isPrivate())
648            return target;
649        else if (isClassSubclassOf(target.getDeclaringClass(), container.getDeclaringClass()))
650            return resolveConcreteDispatch(container.getDeclaringClass(), target);
651        else
652            return target;
653    }
