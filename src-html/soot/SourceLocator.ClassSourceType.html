062    /** Given a class name, uses the soot-class-path to return a ClassSource for the given class. */
063        public ClassSource getClassSource(String className) 
064    {
065                if(classesToLoad==null) {
066                        classesToLoad = new HashSet<String>();
067                        classesToLoad.addAll(Scene.v().getBasicClasses());
068                        for(SootClass c: Scene.v().getApplicationClasses()) {
069                                classesToLoad.add(c.getName());
070                        }
071                }
072        
073        if( classPath == null ) {
074            classPath = explodeClassPath(Scene.v().getSootClassPath());
075        }
076        if( classProviders == null ) {
077            setupClassProviders();
078        }
079        JarException ex = null;
080        for (ClassProvider cp : classProviders) {
081            try {
082                        ClassSource ret = cp.find(className);
083                    if( ret != null ) return ret;
084            } catch(JarException e) {
085                ex = e;
086            }
087        }
088        if(ex!=null) throw ex;
089        for(final ClassLoader cl: additionalClassLoaders) {
090            try {
091                ClassSource ret = new ClassProvider() {
092                                        
093                                        public ClassSource find(String className) {
094                                        String fileName = className.replace('.', '/') + ".class";
095                                                InputStream stream = cl.getResourceAsStream(fileName);
096                                                if(stream==null) return null;
097                                                return new CoffiClassSource(className, stream, fileName);
098                                        }
099
100                }.find(className);
101                    if( ret != null ) return ret;
102            } catch(JarException e) {
103                ex = e;
104            }
105        }
106        if(ex!=null) throw ex;
107        if(className.startsWith("soot.rtlib.tamiflex.")) {
108                String fileName = className.replace('.', '/') + ".class";
109                ClassLoader cl = getClass().getClassLoader();
110                if (cl == null)
111                        return null;
112                InputStream stream = cl.getResourceAsStream(fileName);
113                if(stream!=null) {
114                                return new CoffiClassSource(className, stream, fileName);
115                }
116        }
117        return null;
118    }

124    private void setupClassProviders() {
125        classProviders = new LinkedList<ClassProvider>();
126        ClassProvider classFileClassProvider = Options.v().coffi() ? new CoffiClassProvider() : new AsmClassProvider();
127                switch( Options.v().src_prec() ) {
128            case Options.src_prec_class:
129                classProviders.add(classFileClassProvider);
130                classProviders.add(new JimpleClassProvider());
131                classProviders.add(new JavaClassProvider());
132                break;
133            case Options.src_prec_only_class:
134                classProviders.add(classFileClassProvider);
135                break;
136            case Options.src_prec_java:
137                classProviders.add(new JavaClassProvider());
138                classProviders.add(classFileClassProvider);
139                classProviders.add(new JimpleClassProvider());
140                break;
141            case Options.src_prec_jimple:
142                classProviders.add(new JimpleClassProvider());
143                classProviders.add(classFileClassProvider);
144                classProviders.add(new JavaClassProvider());
145                break;
146            case Options.src_prec_apk:
147                classProviders.add(new DexClassProvider());
148                                classProviders.add(classFileClassProvider);
149                                classProviders.add(new JavaClassProvider());
150                                classProviders.add(new JimpleClassProvider());
151                break;
152            case Options.src_prec_apk_c_j:
153                classProviders.add(new DexClassProvider());
154                                classProviders.add(classFileClassProvider);
155                                classProviders.add(new JimpleClassProvider());
156                break;
157            default:
158                throw new RuntimeException("Other source precedences are not currently supported.");
159        }
160    }

179                ClassSourceType cst = getClassSourceType(dir);

189    private LoadingCache<String, ClassSourceType> pathToSourceType = CacheBuilder.newBuilder()
190                .initialCapacity(60)
191                .maximumSize(500)
192                .softValues()
193                .concurrencyLevel(Runtime.getRuntime().availableProcessors())
194                .build(
195                                new CacheLoader<String, ClassSourceType>(){
196                                                @Override
197                                                public ClassSourceType load(String path) throws Exception {
198                                                        File f = new File(path);
199                                                        if(!f.exists() && !Options.v().ignore_classpath_errors())
200                                                                throw new Exception("Error: The path '" + path + "' does not exist.");
201                                                        if(!f.canRead() && !Options.v().ignore_classpath_errors())
202                                                                throw new Exception("Error: The path '" + path + "' exists but is not readable.");
203                                                        if(f.isFile()) {
204                                                                if (path.endsWith(".zip"))
205                                                        return ClassSourceType.zip;
206                                                    else if (path.endsWith(".jar"))
207                                                        return ClassSourceType.jar;
208                                                    else if (path.endsWith(".apk"))
209                                                        return ClassSourceType.apk;
210                                                    else if (path.endsWith(".dex"))
211                                                        return ClassSourceType.dex;
212                                                    else
213                                                        return ClassSourceType.unknown;
214                                                        }
215                                                        return ClassSourceType.directory;
216                                                }
217                                }
218    );

220    private ClassSourceType getClassSourceType(String path) {
221        try {
222                        return pathToSourceType.get(path);
223                } catch (Exception e) {
224                        throw new RuntimeException(e);
225                }

232    private List<String> getClassesUnder(String aPath, String prefix) {
233                List<String> classes = new ArrayList<String>();
234                ClassSourceType cst = getClassSourceType(aPath);
235                
236                // Get the dex file from an apk
237                if (cst == ClassSourceType.apk) {
238                        ZipFile archive = null;
239                        try {
240                                archive = new ZipFile(aPath);
241                                for (Enumeration<? extends ZipEntry> entries = archive.entries(); entries.hasMoreElements();) {
242                                        ZipEntry entry = entries.nextElement();
243                                        String entryName = entry.getName();
244                                        // We are dealing with an apk file
245                                        if (entryName.endsWith(".dex"))
246                                                classes.addAll(DexClassProvider.classesOfDex(new File(aPath)));
247                                }               
248                        } catch (IOException e) {
249                                throw new CompilationDeathException("Error reasing archive '" + aPath + "'",e);
250                        }finally{
251                                try{
252                                        if(archive != null)
253                                                archive.close();
254                                }catch(Throwable t) {}
255                        }
256                }
257                // Directly load a dex file
258                else if (cst == ClassSourceType.dex) {
259                        try {
260                                classes.addAll(DexClassProvider.classesOfDex(new File(aPath)));
261                        } catch (IOException e) {
262                                throw new CompilationDeathException("Error reasing '" + aPath + "'",e);
263                        }
264                }
265                // load Java class files from ZIP and JAR
266                else if (cst == ClassSourceType.jar || cst == ClassSourceType.zip) {
267                        Set<String> dexEntryNames = new HashSet<String>();
268                        ZipFile archive = null;
269                        try {
270                                archive = new ZipFile(aPath);
271                                for (Enumeration<? extends ZipEntry> entries = archive.entries(); entries.hasMoreElements();) {
272                                        ZipEntry entry = entries.nextElement();
273                                        String entryName = entry.getName();
274                                        if(entryName.endsWith(".class") || entryName.endsWith(".jimple")){
275                                                int extensionIndex = entryName.lastIndexOf('.');
276                                                entryName = entryName.substring(0, extensionIndex);
277                                                entryName = entryName.replace('/', '.');
278                                                classes.add(prefix + entryName);
279                                        }
280                                        else if(entryName.endsWith(".dex")){
281                                                dexEntryNames.add(entryName);
282                                        }
283                                }
284                        } catch (Throwable e) {
285                                throw new CompilationDeathException("Error reading archive '" + aPath + "'", e);
286                        }finally{
287                                try{
288                                        if(archive != null)
289                                                archive.close();
290                                }catch(Throwable t) {}
291                        }
292                        
293                        if(!dexEntryNames.isEmpty()){
294                                File file = new File(aPath);
295                                if(Options.v().process_multiple_dex()){
296                                        for(String dexEntryName : dexEntryNames){
297                                                try {
298                                                        classes.addAll(DexClassProvider.classesOfDex(file,dexEntryName));
299                                                } catch (Throwable e) {} /* Ignore unreadable files */
300                                        }
301                                }else{
302                                        try {
303                                                classes.addAll(DexClassProvider.classesOfDex(file));
304                                        } catch (Throwable e) {} /* Ignore unreadable files */
305                                }
306                        }
307                }
308                else if (cst == ClassSourceType.directory) {
309                        File file = new File(aPath);
310
311                        File[] files = file.listFiles();
312                        if (files == null) {
313                                files = new File[1];
314                                files[0] = file;
315                        }
316
317                        for (File element : files) {
318                                if (element.isDirectory()) {
319                                        classes.addAll(getClassesUnder(aPath + File.separatorChar + element.getName(),
320                                                        prefix + element.getName() + "."));
321                                } else {
322                                        String fileName = element.getName();
323
324                                        if (fileName.endsWith(".class")) {
325                                                int index = fileName.lastIndexOf(".class");
326                                                classes.add(prefix + fileName.substring(0, index));
327                                        }else if (fileName.endsWith(".jimple")) {
328                                                int index = fileName.lastIndexOf(".jimple");
329                                                classes.add(prefix + fileName.substring(0, index));
330                                        }else if (fileName.endsWith(".java")) {
331                                                int index = fileName.lastIndexOf(".java");
332                                                classes.add(prefix + fileName.substring(0, index));
333                                        }else if (fileName.endsWith(".dex")) {
334                                                try {
335                                                        classes.addAll(DexClassProvider.classesOfDex(element));
336                                                } catch (IOException e) { /* Ignore unreadable files */
337                                                }
338                                        }
339                                }
340                        }
341                }
342                else
343                        throw new RuntimeException("Invalid class source type");
344                return classes;
345        }

347    public String getFileNameFor(SootClass c, int rep) {
348        if (rep == Options.output_format_none)
349            return null;
350
351        StringBuffer b = new StringBuffer();
352
353        if( !Options.v().output_jar() ) {
354            b.append(getOutputDir());
355        }
356
357        if ((b.length() > 0) && (b.charAt(b.length() - 1) != File.separatorChar))
358            b.append(File.separatorChar);
359
360        if (rep != Options.output_format_dava) {
361            if(rep == Options.output_format_class) {
362                b.append(c.getName().replace('.', File.separatorChar));
363            } else if(rep == Options.output_format_template) {
364                b.append(c.getName().replace('.', '_'));
365                b.append("_Maker");
366            } else {
367                b.append(c.getName());
368            }
369            b.append(getExtensionFor(rep));
370
371            return b.toString();
372        }
373
374        return getDavaFilenameFor(c, b);
375    }

377    private String getDavaFilenameFor(SootClass c, StringBuffer b) {
378        b.append("dava");
379        b.append(File.separatorChar);
380        ensureDirectoryExists(new File(b.toString() + "classes"));
381
382        b.append("src");
383        b.append(File.separatorChar);
384        String fixedPackageName = c.getJavaPackageName();
385        if (!fixedPackageName.equals("")) {
386            b.append(fixedPackageName.replace('.', File.separatorChar));
387            b.append(File.separatorChar);
388        }
389        
390        ensureDirectoryExists(new File(b.toString()));
391        
392        b.append(c.getShortJavaStyleName());
393        b.append(".java");
394
395        return b.toString();
396    }

398    /* This is called after sootClassPath has been defined. */
399    public Set<String> classesInDynamicPackage(String str) {
400        HashSet<String> set = new HashSet<String>(0);
401        StringTokenizer strtok = new StringTokenizer(
402                Scene.v().getSootClassPath(), String.valueOf(File.pathSeparatorChar));
403        while (strtok.hasMoreTokens()) {
404            String path = strtok.nextToken();
405
406            // For jimple files
407            List<String> l = getClassesUnder(path);
408            for (String filename : l) {
409                if (filename.startsWith(str))
410                    set.add(filename);
411            }
412
413            // For class files;
414            path = path + File.pathSeparatorChar;
415            StringTokenizer tokenizer = new StringTokenizer(str, ".");
416            while (tokenizer.hasMoreTokens()) {
417                path = path + tokenizer.nextToken();
418                if (tokenizer.hasMoreTokens())
419                    path = path + File.pathSeparatorChar;
420            }
421            l = getClassesUnder(path);
422            for (String string : l)
423                                set.add(str + "." + string);
424        }
425        return set;
426    }

467     * Returns the output directory given by {@link Options} or a default if not
468     * set. Also ensures that all directories in the path exist.
469     *
470     * @return the output directory from {@link Options} or a default if not set
471     */
472    public String getOutputDir() {
473        File dir;
474        if (Options.v().output_dir().length() == 0) {
475            //Default if -output-dir was not set
476            dir = new File("sootOutput");
477        } else {
478            dir = new File(Options.v().output_dir());
479            //If a Jar name was given as the output dir
480            //  get its parent path (possibly empty)
481            if (dir.getPath().endsWith(".jar")) {
482                dir = dir.getParentFile();
483                if (dir == null) {
484                    dir = new File("");
485                }
486            }
487        }
488
489        ensureDirectoryExists(dir);
490        return dir.getPath();
491    }

494     * If {@link Options#v()#output_jar()} is set, returns the name of the jar
495     * file to which the output will be written. The name of the jar file can be
496     * given with the -output-dir option or a default will be used. Also ensures
497     * that all directories in the path exist.
498     *
499     * @return the name of the Jar file to which outputs are written
500     */
501    public String getOutputJarName() {
502        if (!Options.v().output_jar()) {
503            return "";
504        }
505
506        File dir;
507        if (Options.v().output_dir().length() == 0) {
508            //Default if -output-dir was not set
509            dir = new File("sootOutput/out.jar");
510        } else {
511            dir = new File(Options.v().output_dir());
512            //If a Jar name was not given, then supply default
513            if (!dir.getPath().endsWith(".jar")) {
514                dir = new File(dir.getPath(), "out.jar");
515            }
516        }
517
518        ensureDirectoryExists(dir.getParentFile());
519        return dir.getPath();
520    }

522    /** Explodes a class path into a list of individual class path entries. */
523    public static List<String> explodeClassPath( String classPath ) {
524        List<String> ret = new ArrayList<String>();
525
526        StringTokenizer tokenizer = 
527            new StringTokenizer(classPath, File.pathSeparator);
528        while( tokenizer.hasMoreTokens() ) {
529            String originalDir = tokenizer.nextToken();
530            String canonicalDir;
531            try {
532                canonicalDir = new File(originalDir).getCanonicalPath();
533                ret.add(canonicalDir);
534            } catch( IOException e ) {
535                throw new CompilationDeathException( "Couldn't resolve classpath entry "+originalDir+": "+e );
536            }
537        }
538        return ret;
539    }

541    public static class FoundFile {
542        private File file;
543        private String entryName;
544        private ZipFile zipFile;
545        private ZipEntry zipEntry;
546        private List<InputStream> openedInputStreams;
547        
548        FoundFile(String archivePath, String entryName) {
549                this();
550                if(archivePath == null || entryName == null)
551                        throw new IllegalArgumentException("Error: The archive path and entry name cannot be null.");
552                this.file = new File(archivePath);
553                this.entryName = entryName;
554        }
555        
556        FoundFile(File file) {
557                this();
558                if(file == null)
559                        throw new IllegalArgumentException("Error: The file cannot be null.");
560            this.file = file;
561            this.entryName = null;
562        }
563        
564        private FoundFile() {
565                this.zipFile = null;
566                this.zipEntry = null;
567                this.openedInputStreams = new ArrayList<InputStream>();
568        }
569        
570        public String getFilePath() {
571                return file.getPath();
572        }
573        
574        public boolean isZipFile() {
575                return entryName != null;
576        }
577        
578        public File getFile() {
579                return file;
580        }
581        
582        public InputStream inputStream() {
583                InputStream ret = null;
584                if(!isZipFile()) {
585                        try{
586                                ret = new FileInputStream(file);
587                        } catch(Exception e) {
588                                throw new RuntimeException("Error: Failed to open a InputStream for the file at path '" + file.getPath() + "'.",e);
589                        }
590                } else {
591                        if(zipFile == null) {
592                                try {
593                                        zipFile = new ZipFile(file);
594                                        zipEntry = zipFile.getEntry(entryName);
595                                        if(zipEntry == null) {
596                                                silentClose();
597                                                throw new RuntimeException("Error: Failed to find entry '" + entryName + "' in the archive file at path '" + 
598                                                                file.getPath() + "'.");
599                                        }
600                                } catch(Exception e) {
601                                        silentClose();
602                                        throw new RuntimeException("Error: Failed to open the archive file at path '" + file.getPath() + "' for entry '" + 
603                                                        entryName + "'.",e);
604                                }
605                        }
606                        try{
607                                ret = doJDKBugWorkaround(zipFile.getInputStream(zipEntry), zipEntry.getSize());
608                        } catch(Exception e){
609                                throw new RuntimeException("Error: Failed to open a InputStream for the entry '" + zipEntry.getName() + 
610                                                "' of the archive at path '" + zipFile.getName() + "'.",e);
611                        }
612                }
613                
614                openedInputStreams.add(ret);
615                return ret;
616        }
617        
618        public void silentClose() {
619                try {
620                        close();
621                } catch(Exception e) {}
622        }
623        
624        public void close(){
625                //Try to close all opened input streams
626                List<Exception> errs = new ArrayList<Exception>();
627                for(Iterator<InputStream> it = openedInputStreams.iterator(); it.hasNext();){
628                        InputStream is = it.next();
629                        try {
630                                is.close();
631                        } catch(Exception e) {
632                                errs.add(e);//record errors for later
633                        }
634                        it.remove();//remove the stream no matter what
635                }
636                //Try to close the opened zip file if it exists
637                if(zipFile != null) {
638                        try {
639                                zipFile.close();
640                                errs.clear();//Successfully closed the archive so all input streams were closed successfully also
641                        } catch(Exception e) {
642                                errs.add(e);
643                        }
644                        zipFile = null;//set to null no matter what
645                        zipEntry = null;//set to null no matter what
646                }
647                //Throw single exception combining all errors
648                if(!errs.isEmpty()) {
649                        String msg = null;
650                        ByteArrayOutputStream baos = new ByteArrayOutputStream();
651                        PrintStream ps = null;
652                        try {
653                                ps = new PrintStream(baos,true,"utf-8");
654                                ps.println("Error: Failed to close all opened resources. The following exceptions were thrown in the process: ");
655                                int i = 0;
656                                for(Throwable t : errs){
657                                        ps.print("Exception ");
658                                        ps.print(i++);
659                                        ps.print(": ");
660                                        t.printStackTrace(ps);
661                                }
662                                msg = new String(baos.toByteArray(), StandardCharsets.UTF_8);
663                        } catch(Exception e) {
664                                //Do nothing as this will never occur
665                        } finally {
666                                ps.close();
667                        }
668                        throw new RuntimeException(msg);
669                }
670        }
671        
672        private InputStream doJDKBugWorkaround(InputStream is, long size) throws IOException {
673                int sz = (int) size;
674                byte[] buf = new byte[sz];                                      
675                final int N = 1024;
676                int ln = 0;
677                int count = 0;
678                while (sz > 0 && (ln = is.read(buf, count, Math.min(N, sz))) != -1) {
679                        count += ln;
680                        sz -= ln;
681                }
682                return  new ByteArrayInputStream(buf);          
683        }
684    }

686    /** Searches for a file with the given name in the exploded classPath. */
687    public FoundFile lookupInClassPath( String fileName ) {
688        for (String dir : classPath) {
689            FoundFile ret = null;
690            ClassSourceType cst = getClassSourceType(dir);
691            if(cst == ClassSourceType.zip || cst == ClassSourceType.jar) {
692                ret = lookupInArchive(dir, fileName);
693            }
694            else if (cst == ClassSourceType.directory) {
695                ret = lookupInDir(dir, fileName);
696            }
697            if( ret != null )
698                return ret;
699        }
700        return null;
701    }
702    private FoundFile lookupInDir(String dir, String fileName) {
703        File f = new File( dir+File.separatorChar+fileName );
704        if( f.canRead() ) {
705            return new FoundFile(f);
706        }
707        return null;
708    }

710    private LoadingCache<String, Set<String>> archivePathsToEntriesCache = CacheBuilder.newBuilder()
711                .initialCapacity(60)
712                .maximumSize(500)
713                .softValues()
714                .concurrencyLevel(Runtime.getRuntime().availableProcessors())
715                .build(
716                                new CacheLoader<String, Set<String>>(){
717                                                @Override
718                                                public Set<String> load(String archivePath) throws Exception {
719                                                        ZipFile archive = null;
720                                                        try {
721                                                                archive = new ZipFile(archivePath);
722                                                                Set<String> ret = new HashSet<String>();
723                                                                Enumeration<? extends ZipEntry> it = archive.entries();
724                                                                while(it.hasMoreElements()){
725                                                                        ret.add(it.nextElement().getName());
726                                                                }
727                                                                return ret;
728                                                        } finally {
729                                                                if(archive != null)
730                                                                        archive.close();
731                                                        }
732                                                }
733                                }
734    );
735    
736    private FoundFile lookupInArchive(String archivePath, String fileName) {
737        Set<String> entryNames = null;
738        try {
739                entryNames = archivePathsToEntriesCache.get(archivePath);
740        } catch(Exception e) {
741                throw new RuntimeException("Error: Failed to retrieve the archive entries list for the archive at path '" + archivePath + "'.",e);
742        }
743        if(entryNames.contains(fileName)){
744                return new FoundFile(archivePath, fileName);
745        }
746        return null;
747    }

749    /** Returns the name of the class in which the (possibly inner) class
750     * className appears. */
751    public String getSourceForClass( String className ) {
752        String javaClassName = className;
753        if (className.indexOf("$") != -1) {
754            // class is an inner class and will be in
755            // Outer of Outer$Inner
756            javaClassName = className.substring(0, className.indexOf("$"));
757        }
758        return javaClassName;
759    }
