147        }
148        Scene.v().getMethodNumberer().add(this);
149        subsignature =
150            Scene.v().getSubSigNumberer().findOrAdd(getSubSignature());
151        
152

290        Retrieves the active body for this method.
291     */
292    public Body getActiveBody() {
293        if (declaringClass!=null && declaringClass.isPhantomClass())
294            throw new RuntimeException(
295                "cannot get active body for phantom class: " + getSignature());
296
297                // ignore empty body exceptions if we are just computing coffi metrics
298        if (!soot.jbco.Main.metrics && !hasActiveBody())
299            throw new RuntimeException(
300                "no active body present for method " + getSignature());
301
302        return activeBody;
303    }

305    /**
306     * Returns the active body if present, else constructs an active body and returns that.
307     *
308     * If you called Scene.v().loadClassAndSupport() for a class yourself, it will
309     * not be an application class, so you cannot get retrieve its active body.
310     * Please call setApplicationClass() on the relevant class.
311     */
312
313    public Body retrieveActiveBody() {
314        // If we already have a body for some reason, we just take it. In this case,
315        // we don't care about resolving levels or whatever.
316        if (hasActiveBody())
317                return getActiveBody();
318        
319        declaringClass.checkLevel(SootClass.BODIES);
320        if (declaringClass.isPhantomClass())
321            throw new RuntimeException(
322                "cannot get resident body for phantom class : "
323                    + getSignature()
324                    + "; maybe you want to call c.setApplicationClass() on this class!");
325        
326        Body b = this.getBodyFromMethodSource("jb");
327        setActiveBody(b);
328        return b;
329    }

331    /**
332        Sets the active body for this method. 
333     */
334    public void setActiveBody(Body body) {
335        if ((declaringClass != null)
336            && declaringClass.isPhantomClass())
337            throw new RuntimeException(
338                "cannot set active body for phantom class! " + this);
339        
340        // If someone sets a body for a phantom method, this method then is no
341        // longer phantom
342        isPhantom = false;
343        
344        if (!isConcrete())
345            throw new RuntimeException(
346                "cannot set body for non-concrete method! " + this);
347
348        if (body!= null && body.getMethod() != this)
349            body.setMethod(this);
350
351        activeBody = body;
352    }

369    /** Adds the given exception to the list of exceptions thrown by this method. */
370    public void addException(SootClass e) {
371        if(DEBUG)
372                System.out.println("Adding exception "+e);
373        
374        if (exceptions == null)
375            exceptions = new ArrayList<SootClass>();
376        else if (exceptions.contains(e))
377            throw new RuntimeException(
378                "already throws exception " + e.getName());
379
380        exceptions.add(e);
381    }

383    /** Removes the given exception from the list of exceptions thrown by this method. */
384    public void removeException(SootClass e) {
385        if(DEBUG)
386                System.out.println("Removing exception "+e);
387
388        if (exceptions == null)
389            exceptions = new ArrayList<SootClass>();
390
391        if (!exceptions.contains(e))
392            throw new RuntimeException(
393                "does not throw exception " + e.getName());
394
395        exceptions.remove(e);
396    }

598    private static String getSubSignatureImpl(
599        String name,
600        List<Type> params,
601        Type returnType) {
602        StringBuilder buffer = new StringBuilder();
603        
604        buffer.append(returnType.getEscapedName());
605        
606        buffer.append(" ");
607        buffer.append(Scene.v().quotedNameOf(name));
608        buffer.append("(");
609
610        for (int i = 0; i < params.size(); i++) {
611            buffer.append(params.get(i).getEscapedName());
612            if (i < params.size() - 1)
613                buffer.append(",");
614        }
615        buffer.append(")");
616
617        return buffer.toString().intern();
618    }

635     * DavaBody

637    public String getDavaDeclaration() {
638        if (getName().equals(staticInitializerName))
639            return "static";
640
641        StringBuffer buffer = new StringBuffer();
642
643        // modifiers
644        StringTokenizer st =
645            new StringTokenizer(Modifier.toString(this.getModifiers()));
646        if (st.hasMoreTokens())
647            buffer.append(st.nextToken());
648
649        while (st.hasMoreTokens())
650            buffer.append(" " + st.nextToken());
651
652        if (buffer.length() != 0)
653            buffer.append(" ");
654
655        // return type + name
656
657        if (getName().equals(constructorName))
658            buffer.append(getDeclaringClass().getShortJavaStyleName());
659        else {
660            Type t = this.getReturnType();
661
662            String tempString = t.toString();    
663            
664            /*
665             * Added code to handle RuntimeExcepotion thrown by getActiveBody
666             */
667            if(hasActiveBody()){
668                DavaBody body = (DavaBody) getActiveBody();
669                IterableSet<String> importSet = body.getImportList();
670
671                if(!importSet.contains(tempString)){
672                        body.addToImportList(tempString);
673                }
674                tempString = RemoveFullyQualifiedName.getReducedName(importSet,tempString,t);
675            }
676                                                                        
677                        buffer.append(tempString + " ");
678
679            buffer.append(Scene.v().quotedNameOf(this.getName()));
680        }
681
682        buffer.append("(");
683
684        // parameters
685        Iterator<Type> typeIt = this.getParameterTypes().iterator();
686        int count = 0;
687        while (typeIt.hasNext()) {
688            Type t = (Type) typeIt.next();
689                        String tempString = t.toString();
690            
691            /*
692                         *  Nomair A. Naeem 7th Feb 2006
693                         *  It is nice to remove the fully qualified type names
694                         *  of parameters if the package they belong to have been imported
695                         *  javax.swing.ImageIcon should be just ImageIcon if javax.swing is imported
696                         *  If not imported WHY NOT..import it!! 
697                         */
698                        if(hasActiveBody()){
699                                DavaBody body = (DavaBody) getActiveBody();
700                                IterableSet<String> importSet = body.getImportList();
701
702                                if(!importSet.contains(tempString)){
703                                        body.addToImportList(tempString);
704                                }
705                                tempString = RemoveFullyQualifiedName.getReducedName(importSet,tempString,t);
706                        }
707                                                                        
708                        buffer.append(tempString + " ");
709                        
710            buffer.append(" ");
711            if (hasActiveBody()){
712                buffer.append(((DavaBody) getActiveBody()).get_ParamMap().get(new Integer(count++)));
713            }
714            else {
715                if (t == BooleanType.v())
716                    buffer.append("z" + count++);
717                else if (t == ByteType.v())
718                    buffer.append("b" + count++);
719                else if (t == ShortType.v())
720                    buffer.append("s" + count++);
721                else if (t == CharType.v())
722                    buffer.append("c" + count++);
723                else if (t == IntType.v())
724                    buffer.append("i" + count++);
725                else if (t == LongType.v())
726                    buffer.append("l" + count++);
727                else if (t == DoubleType.v())
728                    buffer.append("d" + count++);
729                else if (t == FloatType.v())
730                    buffer.append("f" + count++);
731                else if (t == StmtAddressType.v())
732                    buffer.append("a" + count++);
733                else if (t == ErroneousType.v())
734                    buffer.append("e" + count++);
735                else if (t == NullType.v())
736                    buffer.append("n" + count++);
737                else
738                    buffer.append("r" + count++);
739            }
740
741            if (typeIt.hasNext())
742                buffer.append(", ");
743
744        }
745
746        buffer.append(")");
747
748        // Print exceptions
749        if (exceptions != null) {
750            Iterator<SootClass> exceptionIt = this.getExceptions().iterator();
751
752            if (exceptionIt.hasNext()) {
753                buffer.append(
754                    " throws " + exceptionIt.next().getName());
755
756                while (exceptionIt.hasNext()) {

766    /**
767     * Returns the declaration of this method, as used at the top of textual body representations 
768     *  (before the {}'s containing the code for representation.)
769     */
770    public String getDeclaration() {
771        StringBuffer buffer = new StringBuffer();
772
773        // modifiers
774        StringTokenizer st =
775            new StringTokenizer(Modifier.toString(this.getModifiers()));
776        if (st.hasMoreTokens())
777            buffer.append(st.nextToken());
778
779        while (st.hasMoreTokens())
780            buffer.append(" " + st.nextToken());
781
782        if (buffer.length() != 0)
783            buffer.append(" ");
784
785        // return type + name
786
787        buffer.append(this.getReturnType() + " ");
788        buffer.append(Scene.v().quotedNameOf(this.getName()));
789
790        buffer.append("(");
791
792        // parameters
793        Iterator<Type> typeIt = this.getParameterTypes().iterator();
794        //int count = 0;
795        while (typeIt.hasNext()) {
796            Type t = (Type) typeIt.next();
797
798            buffer.append(t);
799
800            if (typeIt.hasNext())
801                buffer.append(", ");
802
803        }
804
805        buffer.append(")");
806
807        // Print exceptions
808        if (exceptions != null) {
809            Iterator<SootClass> exceptionIt = this.getExceptions().iterator();
810
811            if (exceptionIt.hasNext()) {
812                buffer.append(
813                    " throws " + exceptionIt.next().getName());
814
815                while (exceptionIt.hasNext()) {
816                    buffer.append(
817                        ", " + exceptionIt.next().getName());
818                }
819            }
820        }
821
822        return buffer.toString().intern();
823    }

837    @Override
838    public int getJavaSourceStartLineNumber() {
839        super.getJavaSourceStartLineNumber();
840        //search statements for first line number
841        if(line==-1 && hasActiveBody()) {
842                PatchingChain<Unit> unit = getActiveBody().getUnits();
843                for (Unit u : unit) {
844                        int l = u.getJavaSourceStartLineNumber();
845                        if(l>-1) {
846                                //store l-1, as method header is usually one line before 1st statement
847                                line = l-1;
848                                break;
849                        }
850                        }
851        } 
852        return line;
853    }
