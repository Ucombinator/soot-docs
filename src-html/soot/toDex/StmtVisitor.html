085/**
086 * A visitor that builds a list of instructions from the Jimple statements it visits.<br>
087 * <br>
088 * Use {@link Switchable#apply(soot.util.Switch)} with this visitor to add statements
089 * and {@link #getFinalInsns()} to get the final dexlib instructions.<br>
090 * <br>
091 * These final instructions do have correct offsets, jump targets and register numbers.
092 * 
093 * @see Insn intermediate representation of an instruction
094 * @see Instruction final representation of an instruction
095 */
096class StmtVisitor implements StmtSwitch {
097

105                oppositeIfs.put(Opcode.IF_EQZ, Opcode.IF_NEZ);

192    protected void addInsn(Insn insn, Stmt s) {
193                int highestIndex = insns.size();
194                addInsn(highestIndex, insn);
195                if (s != null)
196                        if (insnStmtMap.put(insn, s) != null)
197                                throw new RuntimeException("Duplicate instruction");
198        }

217        /**
218         * Reduces the instruction list by removing unnecessary instruction pairs
219         * such as move v0 v1; move v1 v0;
220         */
221        private void reduceInstructions() {
222                for (int i = 0; i < this.insns.size() - 1; i++) {
223                        Insn curInsn = this.insns.get(i);
224                        // Only consider real instructions
225                        if (curInsn instanceof AddressInsn)
226                                continue;
227                        if (!isReducableMoveInstruction(curInsn.getOpcode().name))
228                                continue;
229                        
230                        // Skip over following address instructions
231                        Insn nextInsn = null;
232                        int nextIndex = -1;
233                        for (int j = i + 1; j < this.insns.size(); j++) {
234                                Insn candidate = this.insns.get(j);
235                                if (candidate instanceof AddressInsn)
236                                        continue;
237                                nextInsn = candidate;
238                                nextIndex = j;
239                                break;
240                        }
241                        if (nextInsn == null || !isReducableMoveInstruction(nextInsn.getOpcode().name))
242                                continue;
243                        
244                        // Do not remove the last instruction in the body as we need to remap
245                        // jump targets to the successor
246                        if (nextIndex == this.insns.size() - 1)
247                                continue;
248                        
249                        // Check if we have a <- b; b <- a;
250                        Register firstTarget = curInsn.getRegs().get(0);
251                        Register firstSource = curInsn.getRegs().get(1);
252                        Register secondTarget = nextInsn.getRegs().get(0);
253                        Register secondSource = nextInsn.getRegs().get(1);
254                        if (firstTarget.equals(secondSource) && secondTarget.equals(firstSource)) {
255                                Stmt origStmt = insnStmtMap.get(nextInsn);
256                                
257                                // Remove the second instruction as it does not change any
258                                // state. We cannot remove the first instruction as other
259                                // instructions may depend on the register being set.
260                                if (origStmt == null || !isJumpTarget(origStmt)) {
261                                        Insn nextStmt = this.insns.get(nextIndex + 1);
262                                        insns.remove(nextIndex);
263                                        
264                                        if (origStmt != null) {
265                                                insnStmtMap.remove(nextInsn);
266                                                insnStmtMap.put(nextStmt, origStmt);
267                                        }
268                                }
269                        }
270                }
271        }
272        
273        private boolean isReducableMoveInstruction(String name) {
274                return name.startsWith("move/")
275                                || name.startsWith("move-object/")
276                                || name.startsWith("move-wide/");
277        }
278
279        private boolean isJumpTarget(Stmt target) {
280                for (Insn insn : this.insns)
281                        if (insn instanceof InsnWithOffset)
282                                if (((InsnWithOffset) insn).getTarget() == target)
283                                        return true;
284                return false;
285        }

287        private void addPayloads() {
288                // add switch payloads to the end of the insns
289                for (AbstractPayload payload : payloads) {
290            addInsn(new AddressInsn(payload), null);
291            addInsn(payload, null);
292                }
293        }

295        public List<BuilderInstruction> getRealInsns(LabelAssigner labelAssigner) {
296                List<BuilderInstruction> finalInsns = new ArrayList<BuilderInstruction>();
297                for (Insn i : insns) {
298                        if (i instanceof AddressInsn) {
299                                continue; // skip non-insns
300                        }
301                        BuilderInstruction realInsn = i.getRealInsn(labelAssigner);
302                        finalInsns.add(realInsn);
303            if (insnStmtMap.containsKey(i)) { // get tags
304                instructionInsnMap.put(realInsn, i);
305            }
306            if (insnRegisterMap.containsKey(i)) {
307                instructionRegisterMap.put(realInsn, insnRegisterMap.get(i));
308            }
309            if (i instanceof AbstractPayload)
310                instructionPayloadMap.put(realInsn, (AbstractPayload) i);
311                }
312                return finalInsns;
313        }

362        private Insn buildMonitorInsn(MonitorStmt stmt, Opcode opc) {
363                Value lockValue = stmt.getOp();
364        constantV.setOrigStmt(stmt);
365        
366        // When leaving a monitor, we must make sure to re-use the old
367        // register. If we assign the same class constant to a new register
368        // before leaving the monitor, Android's bytecode verifier will assume
369        // that this constant assignment can throw an exception, leaving us
370        // with a dangling monitor. Imprecise static analyzers ftw.
371                Register lockReg = null;
372                if (lockValue instanceof Constant)
373                        if ((lockReg = monitorRegs.get(lockValue)) != null)
374                                lockReg = lockReg.clone();
375                if (lockReg == null) {
376                        lockReg = regAlloc.asImmediate(lockValue, constantV);
377                        regAlloc.lockRegister(lockReg);
378                        if (lockValue instanceof Constant) {
379                                monitorRegs.put((Constant) lockValue, lockReg);
380                                regAlloc.lockRegister(lockReg);
381                        }
382                }
383                return new Insn11x(opc, lockReg);
384        }

394        @Override
395        public void caseAssignStmt(AssignStmt stmt) {
396                // If this is the beginning of an array initialization, we shortcut the
397                // normal translation process
398                List<Value> arrayValues = arrayInitDetector.getValuesForArrayInit(stmt);
399                if (arrayValues != null) {
400                        Insn insn = buildArrayFillInsn((ArrayRef) stmt.getLeftOp(), arrayValues);
401                        if (insn != null) {
402                                addInsn(insn, stmt);
403                                return;
404                        }
405                }
406                if (arrayInitDetector.getIgnoreUnits().contains(stmt)) {
407                        return;
408                }
409                
410                constantV.setOrigStmt(stmt);
411        exprV.setOrigStmt(stmt);
412                Value lhs = stmt.getLeftOp();
413                if (lhs instanceof ConcreteRef) {
414                        // special cases that lead to *put* opcodes
415                        Value source = stmt.getRightOp();
416            addInsn(buildPutInsn((ConcreteRef) lhs, source), stmt);
417                        return;
418                }
419                // other cases, where lhs is a local
420                if (!(lhs instanceof Local)) {
421                        throw new Error("left-hand side of AssignStmt is not a Local: " + lhs.getClass());
422                }
423                Local lhsLocal = (Local) lhs;
424                
425                Register lhsReg = regAlloc.asLocal(lhsLocal);
426                
427                Value rhs = stmt.getRightOp();
428                if (rhs instanceof Local) {
429                        // move rhs local to lhs local, if different
430                        String lhsName = ((Local)lhs).getName();
431                        String rhsName = ((Local)rhs).getName();
432                        if (lhsName.equals(rhsName)) {
433                                return;
434                        }
435                        Register sourceReg = regAlloc.asLocal((Local) rhs);
436            addInsn(buildMoveInsn(lhsReg, sourceReg), stmt);
437                } else if (rhs instanceof Constant) {
438                        // move rhs constant into the lhs local
439                        constantV.setDestination(lhsReg);
440                        rhs.apply(constantV);
441                } else if (rhs instanceof ConcreteRef) {
442            addInsn(buildGetInsn((ConcreteRef) rhs, lhsReg), stmt);
443                } else {
444                        // evaluate rhs expression, saving the result in the lhs local
445                        exprV.setDestinationReg(lhsReg);
446                        rhs.apply(exprV);
447                        if (rhs instanceof InvokeExpr) {
448                                // do the actual "assignment" for an invocation: move its result to the lhs reg (it was not used yet)
449                                Insn moveResultInsn = buildMoveResultInsn(lhsReg);
450                                int invokeInsnIndex = insns.indexOf(getLastInvokeInsn());
451                                addInsn(invokeInsnIndex + 1, moveResultInsn);
452                        }
453                }
454
455                this.insnRegisterMap.put(insns.get(insns.size() - 1), LocalRegisterAssignmentInformation.v(lhsReg, lhsLocal));
456        }

458        private Insn buildGetInsn(ConcreteRef sourceRef, Register destinationReg) {
459                if (sourceRef instanceof StaticFieldRef) {
460                        return buildStaticFieldGetInsn(destinationReg, (StaticFieldRef) sourceRef);
461                } else if (sourceRef instanceof InstanceFieldRef) {
462                        return buildInstanceFieldGetInsn(destinationReg, (InstanceFieldRef) sourceRef);
463                } else if (sourceRef instanceof ArrayRef) {
464                        return buildArrayGetInsn(destinationReg, (ArrayRef) sourceRef);
465                } else {
466                        throw new RuntimeException("unsupported type of ConcreteRef: " + sourceRef.getClass());
467                }
468        }

470        private Insn buildPutInsn(ConcreteRef destRef, Value source) {
471                if (destRef instanceof StaticFieldRef) {
472                        return buildStaticFieldPutInsn((StaticFieldRef) destRef, source);
473                } else if (destRef instanceof InstanceFieldRef) {
474                        return buildInstanceFieldPutInsn((InstanceFieldRef) destRef, source);
475                } else if (destRef instanceof ArrayRef) {
476                        return buildArrayPutInsn((ArrayRef) destRef, source);
477                } else {
478                        throw new RuntimeException("unsupported type of ConcreteRef: " + destRef.getClass());
479                }
480        }

482        protected static Insn buildMoveInsn(Register destinationReg, Register sourceReg) {
483                // get the opcode type, depending on the source reg (we assume that the destination has the same type)
484                String opcType;
485                if (sourceReg.isObject()) {
486                        opcType = "MOVE_OBJECT";
487                } else if (sourceReg.isWide()) {
488                        opcType = "MOVE_WIDE";
489                } else {
490                        opcType = "MOVE";
491                }
492                // get the optional opcode suffix, depending on the sizes of the regs
493                if (!destinationReg.fitsShort()) {
494                        Opcode opc = Opcode.valueOf(opcType + "_16");
495                        return new Insn32x(opc, destinationReg, sourceReg);
496                } else if (!destinationReg.fitsByte() || !sourceReg.fitsByte()) {
497                        Opcode opc = Opcode.valueOf(opcType + "_FROM16");
498                        return new Insn22x(opc, destinationReg, sourceReg);
499                }
500                Opcode opc = Opcode.valueOf(opcType);
501                return new Insn12x(opc, destinationReg, sourceReg);
502        }

504        private Insn buildStaticFieldPutInsn(StaticFieldRef destRef, Value source) {
505                SootField destSootField = destRef.getField();
506                Register sourceReg = regAlloc.asImmediate(source, constantV);
507                BuilderFieldReference destField = DexPrinter.toFieldReference(destSootField, belongingFile);
508                Opcode opc = getPutGetOpcodeWithTypeSuffix("sput", destField.getType());
509                return new Insn21c(opc, sourceReg, destField);
510        }

512        private Insn buildInstanceFieldPutInsn(InstanceFieldRef destRef, Value source) {
513                SootField destSootField = destRef.getField();
514                BuilderFieldReference destField = DexPrinter.toFieldReference(destSootField, belongingFile);
515                Local instance = (Local) destRef.getBase();
516                Register instanceReg = regAlloc.asLocal(instance);
517                Register sourceReg = regAlloc.asImmediate(source, constantV);
518                Opcode opc = getPutGetOpcodeWithTypeSuffix("iput", destField.getType());
519                return new Insn22c(opc, sourceReg, instanceReg, destField);
520        }

522        private Insn buildArrayPutInsn(ArrayRef destRef, Value source) {
523                Local array = (Local) destRef.getBase();
524                Register arrayReg = regAlloc.asLocal(array);
525                Value index = destRef.getIndex();
526                Register indexReg = regAlloc.asImmediate(index, constantV);
527                Register sourceReg  = regAlloc.asImmediate(source, constantV);
528                String arrayTypeDescriptor = SootToDexUtils.getArrayTypeDescriptor((ArrayType) array.getType());
529                Opcode opc = getPutGetOpcodeWithTypeSuffix("aput", arrayTypeDescriptor);
530                return new Insn23x(opc, sourceReg, arrayReg, indexReg);
531        }

533        private Insn buildArrayFillInsn(ArrayRef destRef, List<Value> values) {
534                Local array = (Local) destRef.getBase();
535                Register arrayReg = regAlloc.asLocal(array);
536                
537                // Convert the list of values into a list of numbers
538                int elementSize = 0;
539                List<Number> numbers = new ArrayList<Number>(values.size());
540                for (Value val : values) {
541                        if (val instanceof IntConstant) {
542                                elementSize = Math.max(elementSize, 4);
543                                numbers.add(((IntConstant) val).value);
544                        }
545                        else if (val instanceof LongConstant) {
546                                elementSize = Math.max(elementSize, 8);
547                                numbers.add(((LongConstant) val).value);                                
548                        }
549                        else if (val instanceof FloatConstant) {
550                                elementSize = Math.max(elementSize, 4);
551                                numbers.add(((FloatConstant) val).value);                               
552                        }
553                        else if (val instanceof DoubleConstant) {
554                                elementSize = Math.max(elementSize, 8);
555                                numbers.add(((DoubleConstant) val).value);                              
556                        }
557                        else
558                                return null;
559                }
560                
561                // For some local types, we know the size upfront
562                if (destRef.getType() instanceof BooleanType)
563                        elementSize = 1;
564                else if (destRef.getType() instanceof ByteType)
565                        elementSize = 1;
566                else if (destRef.getType() instanceof CharType)
567                        elementSize = 2;
568                else if (destRef.getType() instanceof ShortType)
569                        elementSize = 2;
570                else if (destRef.getType() instanceof IntType)
571                        elementSize = 4;
572                else if (destRef.getType() instanceof FloatType)
573                        elementSize = 4;
574                else if (destRef.getType() instanceof LongType)
575                        elementSize = 8;
576                else if (destRef.getType() instanceof DoubleType)
577                        elementSize = 8;
578                
579                ArrayDataPayload payload = new ArrayDataPayload(elementSize, numbers);
580                payloads.add(payload);
581                Insn31t insn = new Insn31t(Opcode.FILL_ARRAY_DATA, arrayReg);
582                insn.setPayload(payload);
583                return insn;
584        }

602        private Insn buildArrayGetInsn(Register destinationReg, ArrayRef sourceRef) {
603                Value index = sourceRef.getIndex();
604                Register indexReg = regAlloc.asImmediate(index, constantV);
605                Local array = (Local) sourceRef.getBase();
606                Register arrayReg = regAlloc.asLocal(array);
607                String arrayTypeDescriptor = SootToDexUtils.getArrayTypeDescriptor((ArrayType) array.getType());
608                Opcode opc = getPutGetOpcodeWithTypeSuffix("aget", arrayTypeDescriptor);
609                return new Insn23x(opc, destinationReg, arrayReg, indexReg);
610        }
611
612        private Opcode getPutGetOpcodeWithTypeSuffix(String prefix, String fieldType) {
613                prefix = prefix.toUpperCase();
614                if (fieldType.equals("Z")) {
615                        return Opcode.valueOf(prefix + "_BOOLEAN");
616                } else if (fieldType.equals("I") || fieldType.equals("F")) {
617                        return Opcode.valueOf(prefix);
618                } else if (fieldType.equals("B")) {
619                        return Opcode.valueOf(prefix + "_BYTE");
620                } else if (fieldType.equals("C")) {
621                        return Opcode.valueOf(prefix + "_CHAR");
622                } else if (fieldType.equals("S")) {
623                        return Opcode.valueOf(prefix + "_SHORT");
624                } else if (SootToDexUtils.isWide(fieldType)) {
625                        return Opcode.valueOf(prefix + "_WIDE");
626                } else if (SootToDexUtils.isObject(fieldType)) {
627                        return Opcode.valueOf(prefix + "_OBJECT");
628                } else {
629                        throw new RuntimeException("unsupported field type for *put*/*get* opcode: " + fieldType);
630                }
631        }

633        private Insn getLastInvokeInsn() {
634                // traverse backwards through the instructions, searching for "invoke-"
635                ListIterator<Insn> listIterator = insns.listIterator(insns.size());
636                while (listIterator.hasPrevious()) {
637                        Insn inst = listIterator.previous();
638                        if (inst.getOpcode().name.startsWith("invoke-")) {
639                                return inst;
640                        }
641                }
642                throw new Error("tried to get last invoke-* instruction, but there was none!");
643        }

645        private Insn buildMoveResultInsn(Register destinationReg) {
646                // build it right after the invoke instruction (more than one instruction could have been generated)
647                Opcode opc;
648                if (SootToDexUtils.isObject(lastReturnTypeDescriptor)) {
649                        opc = Opcode.MOVE_RESULT_OBJECT;
650                } else if (SootToDexUtils.isWide(lastReturnTypeDescriptor)) {
651                        opc = Opcode.MOVE_RESULT_WIDE;
652                } else {
653                        opc = Opcode.MOVE_RESULT;
654                }
655                return new Insn11x(opc, destinationReg);
656        }

670        public void caseReturnStmt(ReturnStmt stmt) {
671                Value returnValue = stmt.getOp();
672                constantV.setOrigStmt(stmt);
673                Register returnReg = regAlloc.asImmediate(returnValue, constantV);
674                Opcode opc;
675                Type retType = returnValue.getType();
676                if (SootToDexUtils.isObject(retType)) {
677                        opc = Opcode.RETURN_OBJECT;
678                } else if (SootToDexUtils.isWide(retType)) {
679                        opc = Opcode.RETURN_WIDE;
680                } else {
681                        opc = Opcode.RETURN;
682                }
683        addInsn(new Insn11x(opc, returnReg), stmt);
684        }

686        @Override
687        public void caseIdentityStmt(IdentityStmt stmt) {
688                Local lhs = (Local) stmt.getLeftOp();
689                Value rhs = stmt.getRightOp();
690                if (rhs instanceof CaughtExceptionRef) {
691                        // save the caught exception with move-exception
692                        Register localReg = regAlloc.asLocal(lhs);
693                        
694            addInsn(new Insn11x(Opcode.MOVE_EXCEPTION, localReg), stmt);
695
696            this.insnRegisterMap.put(insns.get(insns.size() - 1), LocalRegisterAssignmentInformation.v(localReg, (Local)lhs));
697                } else if (rhs instanceof ThisRef || rhs instanceof ParameterRef) {
698                        /* 
699                         * do not save the ThisRef or ParameterRef in a local, because it always has a parameter register already.
700                         * at least use the local for further reference in the statements
701                         */
702                        Local localForThis = (Local) lhs;
703                        regAlloc.asParameter(belongingMethod, localForThis);
704                        
705                        parameterInstructionsList.add(LocalRegisterAssignmentInformation.v(regAlloc.asLocal(localForThis).clone(), localForThis));
706                } else {
707                        throw new Error("unknown Value as right-hand side of IdentityStmt: " + rhs);
708                }
709        }

726        @Override
727        public void caseLookupSwitchStmt(LookupSwitchStmt stmt) {
728        exprV.setOrigStmt(stmt);
729        constantV.setOrigStmt(stmt);
730                // create payload that references the switch's targets
731                List<IntConstant> keyValues = stmt.getLookupValues();
732                int[] keys = new int[keyValues.size()];
733                for (int i = 0; i < keys.length; i++) {
734                        keys[i] = keyValues.get(i).value;
735                }
736                List<Unit> targets = stmt.getTargets();
737                SparseSwitchPayload payload = new SparseSwitchPayload(keys, targets);
738                payloads.add(payload);
739                // create sparse-switch instruction that references the payload
740                Value key = stmt.getKey();
741                Stmt defaultTarget = (Stmt) stmt.getDefaultTarget();
742                if (defaultTarget == stmt)
743                        throw new RuntimeException("Looping switch block detected");
744        addInsn(buildSwitchInsn(Opcode.SPARSE_SWITCH, key, defaultTarget,
745                        payload, stmt), stmt);
746        }

748        @Override
749        public void caseTableSwitchStmt(TableSwitchStmt stmt) {
750        exprV.setOrigStmt(stmt);
751        constantV.setOrigStmt(stmt);
752                // create payload that references the switch's targets
753                int firstKey = stmt.getLowIndex();              
754                List<Unit> targets = stmt.getTargets();
755                PackedSwitchPayload payload = new PackedSwitchPayload(firstKey, targets);
756                payloads.add(payload);
757                // create packed-switch instruction that references the payload
758                Value key = stmt.getKey();
759                Stmt defaultTarget = (Stmt) stmt.getDefaultTarget();
760        addInsn(buildSwitchInsn(Opcode.PACKED_SWITCH, key, defaultTarget,
761                        payload, stmt), stmt);
762        }
763        
764        private Insn buildSwitchInsn(Opcode opc, Value key, Stmt defaultTarget,
765                        SwitchPayload payload, Stmt stmt) {
766                Register keyReg = regAlloc.asImmediate(key, constantV);
767                Insn31t switchInsn = new Insn31t(opc, keyReg);
768                switchInsn.setPayload(payload);
769                payload.setSwitchInsn(switchInsn);
770        addInsn(switchInsn, stmt);
771                // create instruction to jump to the default target, always follows the switch instruction
772                return buildGotoInsn(defaultTarget);
773        }
