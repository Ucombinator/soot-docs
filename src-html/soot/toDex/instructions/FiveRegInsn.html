001package soot.toDex.instructions;
002
003import soot.toDex.Register;
004
005/**
006 * Interface for instructions that need five registers.<br>
007 * <br>
008 * Note that the interface does not inherit from {@link ThreeRegInsn} due to the unusual register naming - the register indices cannot be overwritten here.
009 */
010public interface FiveRegInsn extends Insn {
011        
012        static final int REG_D_IDX = 0;
013        
014        static final int REG_E_IDX = REG_D_IDX + 1;
015        
016        static final int REG_F_IDX = REG_E_IDX + 1;
017        
018        static final int REG_G_IDX = REG_F_IDX + 1;
019        
020        static final int REG_A_IDX = REG_G_IDX + 1;
021        
022        Register getRegD();
023
024        Register getRegE();
025
026        Register getRegF();
027
028        Register getRegG();
029
030        Register getRegA();
031}
