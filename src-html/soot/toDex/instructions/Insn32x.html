001package soot.toDex.instructions;
002
003import java.util.BitSet;
004
005import org.jf.dexlib2.Opcode;
006import org.jf.dexlib2.builder.BuilderInstruction;
007import org.jf.dexlib2.builder.instruction.BuilderInstruction32x;
008
009import soot.toDex.LabelAssigner;
010import soot.toDex.Register;
011
012/**
013 * The "32x" instruction format: It needs three 16-bit code units, has two registers
014 * and is used for general purposes (hence the "x").<br>
015 * <br>
016 * It is used by the opcodes "move/16", "move-wide/16" and "move-object/16".
017 */
018public class Insn32x extends AbstractInsn implements TwoRegInsn {
019
020        public Insn32x(Opcode opc, Register regA, Register regB) {
021                super(opc);
022                regs.add(regA);
023                regs.add(regB);
024        }
025        
026        public Register getRegA() {
027                return regs.get(REG_A_IDX);
028        }
029
030        public Register getRegB() {
031                return regs.get(REG_B_IDX);
032        }
033        
034        @Override
035        protected BuilderInstruction getRealInsn0(LabelAssigner assigner) {
036                return new BuilderInstruction32x(opc, getRegA().getNumber(),
037                                getRegB().getNumber());
038        }
039        
040        @Override
041        public BitSet getIncompatibleRegs() {
042                BitSet incompatRegs = new BitSet(2);
043                if (!getRegA().fitsUnconstrained()) {
044                        incompatRegs.set(REG_A_IDX);
045                }
046                if (!getRegB().fitsUnconstrained()) {
047                        incompatRegs.set(REG_B_IDX);
048                }
049                return incompatRegs;
050        }
051}
