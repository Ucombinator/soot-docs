001package soot.toDex.instructions;
002
003import org.jf.dexlib2.Opcode;
004
005import soot.jimple.Stmt;
006
007/**
008 * An abstract implementation for instructions that have a jump label.
009 */
010public abstract class InsnWithOffset extends AbstractInsn {
011        
012        protected Stmt target;
013        
014        public InsnWithOffset(Opcode opc) {
015                super(opc);
016        }
017        
018        public void setTarget(Stmt target) {
019                if (target == null)
020                        throw new RuntimeException("Cannot jump to a NULL target");
021                this.target = target;
022        }
023        
024        public Stmt getTarget() {
025                return this.target;
026        }
027        
028        /**
029         * Gets the maximum number of words available for the jump offset
030         * @return The maximum number of words available for the jump offset
031         */
032        public abstract int getMaxJumpOffset();
033        
034}
