077/**
078 * A visitor that builds a list of instructions from the Jimple expressions it visits.<br>
079 * <br>
080 * Use {@link Switchable#apply(soot.util.Switch)} with this visitor to add statements.
081 * These are added to the instructions in the {@link StmtVisitor}.<br>
082 * If the expression is part of an assignment or an if statement, use {@link #setDestinationReg(Register)}
083 * and {@link #setTargetForOffset(Stmt)}, respectively.
084 * 
085 * @see StmtVisitor
086 */

134        @Override
135        public void caseSpecialInvokeExpr(SpecialInvokeExpr sie) {
136                BuilderMethodReference method = DexPrinter.toMethodReference
137                                (sie.getMethodRef(), dexFile);
138                List<Register> arguments = getInstanceInvokeArgumentRegs(sie);
139                if (isCallToConstructor(sie) || isCallToPrivate(sie)) {
140            stmtV.addInsn(buildInvokeInsn("INVOKE_DIRECT", method, arguments), origStmt);
141                } else if (isCallToSuper(sie)) {
142            stmtV.addInsn(buildInvokeInsn("INVOKE_SUPER", method, arguments), origStmt);
143                } else {
144                        // This should normally never happen, but if we have such a
145                        // broken call (happens in malware for instance), we fix it.
146            stmtV.addInsn(buildInvokeInsn("INVOKE_VIRTUAL", method, arguments), origStmt);
147                }
148        }

150        private Insn buildInvokeInsn(String invokeOpcode, BuilderMethodReference method,
151                        List<Register> argumentRegs) {
152                Insn invokeInsn;
153                int regCountForArguments = SootToDexUtils.getRealRegCount(argumentRegs);
154                if (regCountForArguments <= 5) {
155                        Register[] paddedArray = pad35cRegs(argumentRegs);
156                        Opcode opc = Opcode.valueOf(invokeOpcode);
157                        invokeInsn = new Insn35c(opc, regCountForArguments, paddedArray[0],
158                                        paddedArray[1], paddedArray[2], paddedArray[3], paddedArray[4], method);
159                } else if (regCountForArguments <= 255) {
160                        Opcode opc = Opcode.valueOf(invokeOpcode + "_RANGE");
161                        invokeInsn = new Insn3rc(opc, argumentRegs, (short) regCountForArguments, method);
162                } else {
163                        throw new Error("too many parameter registers for invoke-* (> 255): " + regCountForArguments + " or registers too big (> 4 bits)");
164                }
165                // save the return type for the move-result insn
166                stmtV.setLastReturnTypeDescriptor(method.getReturnType());
167                return invokeInsn;
168        }
169
170        private boolean isCallToPrivate(SpecialInvokeExpr sie) {
171                return sie.getMethod().isPrivate();
172        }
173
174        private boolean isCallToConstructor(SpecialInvokeExpr sie) {
175                return sie.getMethod().isConstructor();
176        }

178        private boolean isCallToSuper(SpecialInvokeExpr sie) {
179                SootClass classWithInvokation = sie.getMethod().getDeclaringClass();
180                SootClass currentClass = stmtV.getBelongingClass();
181                while (currentClass.hasSuperclass()) {
182                        currentClass = currentClass.getSuperclass();
183                        if (currentClass.equals(classWithInvokation)) {
184                                return true;
185                        }
186                }
187                
188                // If we're dealing with phantom classes, we might not actually
189                // arrive at java.lang.Object. In this case, we should not fail
190                // the check
191                if (currentClass.isPhantom() && !currentClass.getName().equals("java.lang.Object"))
192                        return true;
193                
194                return false; // we arrived at java.lang.Object and did not find a declaration
195        }

197        @Override
198        public void caseVirtualInvokeExpr(VirtualInvokeExpr vie) {
199                /*
200                 * for final methods we build an invoke-virtual opcode, too, although the dex spec says that a virtual method is not final.
201                 * An alternative would be the invoke-direct opcode, but this is inconsistent with dx's output...
202                 */
203                BuilderMethodReference method = DexPrinter.toMethodReference
204                                (vie.getMethodRef(), dexFile);
205                List<Register> argumentRegs = getInstanceInvokeArgumentRegs(vie);
206        stmtV.addInsn(buildInvokeInsn("INVOKE_VIRTUAL", method, argumentRegs), origStmt);
207        }

209        private List<Register> getInvokeArgumentRegs(InvokeExpr ie) {
210                constantV.setOrigStmt(origStmt);
211                List<Register> argumentRegs = new ArrayList<Register>();
212                for (Value arg : ie.getArgs()) {
213                        Register currentReg = regAlloc.asImmediate(arg, constantV);
214                        argumentRegs.add(currentReg);
215                }
216                return argumentRegs;
217        }
218
219        private List<Register> getInstanceInvokeArgumentRegs(InstanceInvokeExpr iie) {
220                constantV.setOrigStmt(origStmt);
221                List<Register> argumentRegs = getInvokeArgumentRegs(iie);
222                // always add reference to callee as first parameter (instance != static)
223                Local callee = (Local) iie.getBase();
224                Register calleeRegister = regAlloc.asLocal(callee);
225                argumentRegs.add(0, calleeRegister);
226                return argumentRegs;
227        }

229        private Register[] pad35cRegs(List<Register> realRegs) {
230                // pad real registers to an array of five bytes, as required by the instruction format "35c"
231                Register[] paddedArray = new Register[5];
232                int nextReg = 0;
233                for (Register realReg : realRegs) {
234                        paddedArray[nextReg] = realReg;
235                        /*
236                         * we include the second half of a wide with an empty reg for the "gap".
237                         * this will be made explicit for dexlib later on.
238                         */
239                        if (realReg.isWide()) {
240                                nextReg++;
241                                paddedArray[nextReg] = Register.EMPTY_REGISTER;
242                        }
243                        nextReg++;
244                }
245                // do the real padding with empty regs
246                for (; nextReg < 5; nextReg++) {
247                        paddedArray[nextReg] = Register.EMPTY_REGISTER;
248                }
249                return paddedArray;
250        }

268        private Insn buildCalculatingBinaryInsn(final String binaryOperation,
269                        Value firstOperand, Value secondOperand) {
270                /*
271                 * it is assumed (but not enforced!) that the types of firstOperand, secondOperand and destinationRegister are compatible
272                 */
273                constantV.setOrigStmt(origStmt);
274                Register firstOpReg = regAlloc.asImmediate(firstOperand, constantV);
275                
276                // use special "/lit"-opcodes if int is the destination's type, the second op is an integer literal and the opc is not sub (no sub*/lit opc available)
277                if (destinationReg.getType() instanceof IntType && secondOperand instanceof IntConstant && !binaryOperation.equals("SUB")) {
278                        int secondOpConstant = ((IntConstant) secondOperand).value;
279                        if (SootToDexUtils.fitsSigned8(secondOpConstant)) {
280                                return buildLit8BinaryInsn(binaryOperation, firstOpReg, (byte) secondOpConstant);
281                        }
282                        if (SootToDexUtils.fitsSigned16(secondOpConstant)) {
283                                if (!binaryOperation.equals("SHL") && !binaryOperation.equals("SHR") && !binaryOperation.equals("USHR")) {
284                                        // no shift opc available for /lit16
285                                        return buildLit16BinaryInsn(binaryOperation, firstOpReg, (short) secondOpConstant);
286                                }
287                        }
288                        // constant is too big, so use it as second op and build normally
289                }
290                
291                // Double-check that we are not carrying out any arithmetic operations
292                // on non-primitive values
293                if (!(secondOperand.getType() instanceof PrimType))
294                        throw new RuntimeException("Invalid value type for primitibe operation");
295                
296                Register secondOpReg = regAlloc.asImmediate(secondOperand, constantV);
297                // use special "/2addr"-opcodes if destination equals first op
298                if (destinationReg.getNumber() == firstOpReg.getNumber()) {
299                        return build2AddrBinaryInsn(binaryOperation, secondOpReg);
300                }
301                // no optimized binary opcode possible
302                return buildNormalBinaryInsn(binaryOperation, firstOpReg, secondOpReg);
303        }

305        private String fixIntTypeString(String typeString) {
306                if (typeString.equals("boolean") || typeString.equals("byte") || typeString.equals("char") || typeString.equals("short")) {
307                        return "int";
308                }
309                return typeString;
310        }

312        private Insn build2AddrBinaryInsn(final String binaryOperation, Register secondOpReg) {
313                String localTypeString = destinationReg.getTypeString();
314                localTypeString = fixIntTypeString(localTypeString);
315                Opcode opc = Opcode.valueOf(binaryOperation + "_" + localTypeString.toUpperCase() + "_2ADDR");
316                return new Insn12x(opc, destinationReg, secondOpReg);
317        }

319        private Insn buildNormalBinaryInsn(String binaryOperation, Register firstOpReg, Register secondOpReg) {
320                String localTypeString = firstOpReg.getTypeString();
321                localTypeString = fixIntTypeString(localTypeString);
322                Opcode opc = Opcode.valueOf(binaryOperation + "_" + localTypeString.toUpperCase());
323                return new Insn23x(opc, destinationReg, firstOpReg, secondOpReg);
324        }

373                Value firstOperand = xe.getOp1();
374                Value secondOperand = xe.getOp2();
375                constantV.setOrigStmt(origStmt);
376                // see for unary ones-complement shortcut, may be a result of Dexpler's conversion
377                if (secondOperand.equals(IntConstant.v(-1))) {
378                        Register sourceReg = regAlloc.asImmediate(firstOperand, constantV);
379            stmtV.addInsn(new Insn12x(Opcode.NOT_INT, destinationReg, sourceReg), origStmt);
380                } else if (secondOperand.equals(LongConstant.v(-1))) {
381                        Register sourceReg = regAlloc.asImmediate(firstOperand, constantV);
382            stmtV.addInsn(new Insn12x(Opcode.NOT_LONG, destinationReg, sourceReg), origStmt);
383                } else {
384            stmtV.addInsn(buildCalculatingBinaryInsn("XOR", firstOperand, secondOperand), origStmt);
385                }
386        }

403        private Insn buildComparingBinaryInsn(String binaryOperation, Value firstOperand, Value secondOperand) {
404                constantV.setOrigStmt(origStmt);
405                Value realFirstOperand = fixNullConstant(firstOperand);
406                Value realSecondOperand = fixNullConstant(secondOperand);
407                Register firstOpReg = regAlloc.asImmediate(realFirstOperand, constantV);
408                // select fitting opcode ("if" or "if-zero")
409                InsnWithOffset comparingBinaryInsn;
410                String opcodeName = "IF_" + binaryOperation;
411                boolean secondOpIsInt = realSecondOperand instanceof IntConstant;
412                boolean secondOpIsZero = secondOpIsInt && ((IntConstant) realSecondOperand).value == 0;
413                if (secondOpIsZero) {
414                        Opcode opc = Opcode.valueOf(opcodeName.concat("Z"));
415                        comparingBinaryInsn = new Insn21t(opc, firstOpReg);
416                        comparingBinaryInsn.setTarget(targetForOffset);
417                } else {
418                        Opcode opc = Opcode.valueOf(opcodeName);
419                        Register secondOpReg = regAlloc.asImmediate(realSecondOperand, constantV);
420                        comparingBinaryInsn = new Insn22t(opc, firstOpReg, secondOpReg);
421                        comparingBinaryInsn.setTarget(targetForOffset);
422                }
423                return comparingBinaryInsn;
424        }
425        
426        private Value fixNullConstant(Value potentialNullConstant) {
427                /*
428                 * The bytecode spec says: "In terms of bitwise representation, (Object) null == (int) 0."
429                 * So use an IntConstant(0) for null comparison in if-*z opcodes.
430                 */
431                if (potentialNullConstant instanceof NullConstant) {
432                        return IntConstant.v(0);
433                }
434                return potentialNullConstant;
435        }

482        private Insn buildCmpInsn(String opcodePrefix, Value firstOperand, Value secondOperand) {
483                constantV.setOrigStmt(origStmt);
484                Register firstReg = regAlloc.asImmediate(firstOperand, constantV);
485                Register secondReg = regAlloc.asImmediate(secondOperand, constantV);
486                // select fitting opcode according to the prefix and the type of the operands
487                Opcode opc = null;
488                // we assume that the type of the operands are equal and use only the first one
489                if (opcodePrefix.equals("CMP_LONG")) {
490                        opc = Opcode.CMP_LONG;
491                } else if (firstReg.isFloat()) {
492                        opc = Opcode.valueOf(opcodePrefix + "_FLOAT");
493                } else if (firstReg.isDouble()) {
494                        opc = Opcode.valueOf(opcodePrefix + "_DOUBLE");
495                } else {
496                        throw new RuntimeException("unsupported type of operands for cmp* opcode: " + firstOperand.getType());
497                }
498                return new Insn23x(opc, destinationReg, firstReg, secondReg);
499        }

503                Value array = le.getOp();
504                constantV.setOrigStmt(origStmt);
505                // In buggy code, the parameter could be a NullConstant.
506                // This is why we use .asImmediate() and not .asLocal()
507                Register arrayReg = regAlloc.asImmediate(array, constantV);
508        stmtV.addInsn(new Insn12x(Opcode.ARRAY_LENGTH, destinationReg, arrayReg), origStmt);
509        }

511        @Override
512        public void caseNegExpr(NegExpr ne) {
513                Value source = ne.getOp();
514                constantV.setOrigStmt(origStmt);
515                Register sourceReg = regAlloc.asImmediate(source, constantV);
516                Opcode opc;
517                Type type = source.getType();
518                if (type instanceof IntegerType) {
519                        opc = Opcode.NEG_INT;
520                } else if (type instanceof FloatType) {
521                        opc = Opcode.NEG_FLOAT;
522                } else if (type instanceof DoubleType) {
523                        opc = Opcode.NEG_DOUBLE;
524                } else if (type instanceof LongType) {
525                        opc = Opcode.NEG_LONG;
526                } else {
527                        throw new RuntimeException("unsupported value type for neg-* opcode: " + type);
528                }
529        stmtV.addInsn(new Insn12x(opc, destinationReg, sourceReg), origStmt);
530        }

546        @Override
547        public void caseCastExpr(CastExpr ce) {
548                Type castType = ce.getCastType();
549                Value source = ce.getOp();
550                constantV.setOrigStmt(origStmt);
551                Register sourceReg = regAlloc.asImmediate(source, constantV);
552                if (SootToDexUtils.isObject(castType)) {
553                        castObject(sourceReg, castType);
554                } else {
555                        castPrimitive(sourceReg, source, castType);
556                }
557        }

559        private void castObject(Register sourceReg, Type castType) {
560                /*
561                 * No real "cast" is done: move the object to a tmp reg, check the cast with check-cast and finally move to the "cast" object location.
562                 * This way a) the old reg is not touched by check-cast, and b) the new reg does change its type only after a successful check-cast.
563                 * 
564                 * a) is relevant e.g. for "r1 = (String) r0" if r0 contains null, since the internal type of r0
565                 * would change from Null to String after the check-cast opcode, which alerts the verifier in future uses of r0
566                 * although nothing should change during execution.
567                 * 
568                 * b) is relevant for exceptional control flow: if we move to the new reg and do the check-cast there, an exception between the end of
569                 * the move's execution and the end of the check-cast execution leaves the new reg with the type of the old reg.
570                 */
571                BuilderReference castTypeItem = DexPrinter.toTypeReference
572                                (castType, stmtV.getBelongingFile());
573                if (sourceReg.getNumber() == destinationReg.getNumber()) {
574                        // simplyfied case if reg numbers do not differ
575            stmtV.addInsn(new Insn21c(Opcode.CHECK_CAST, destinationReg, castTypeItem), origStmt);
576                } else {
577                        // move to tmp reg, check cast, move to destination
578                        Register tmp = regAlloc.asTmpReg(sourceReg.getType());
579            stmtV.addInsn(StmtVisitor.buildMoveInsn(tmp, sourceReg), origStmt);
580            stmtV.addInsn(new Insn21c(Opcode.CHECK_CAST, tmp.clone(), castTypeItem), origStmt);
581            stmtV.addInsn(StmtVisitor.buildMoveInsn(destinationReg, tmp.clone()), origStmt);
582                }
583        }
584        
585        private void castPrimitive(Register sourceReg, Value source, Type castSootType) {
586                PrimitiveType castType = PrimitiveType.getByName(castSootType.toString());
587
588                // Fix null_types on the fly. This should not be necessary, but better be safe
589                // than sorry and it's easy to fix it here.
590                if (castType == PrimitiveType.INT && source.getType() instanceof NullType)
591                        source = IntConstant.v(0);
592                
593                // select fitting conversion opcode, depending on the source and cast type
594                PrimitiveType sourceType = PrimitiveType.getByName(source.getType().toString());
595                if (castType == PrimitiveType.BOOLEAN) {
596                        // there is no "-to-boolean" opcode, so just pretend to move an int to an int
597                        castType = PrimitiveType.INT;
598                        sourceType = PrimitiveType.INT;
599                }
600                if (shouldCastFromInt(sourceType, castType)) {
601                        // pretend to cast from int since that is OK
602                        sourceType = PrimitiveType.INT;
603                        Opcode opc = getCastOpc(sourceType, castType);
604            stmtV.addInsn(new Insn12x(opc, destinationReg, sourceReg), origStmt);
605                } else if (isMoveCompatible(sourceType, castType)) {
606                        /*
607                         * no actual cast needed, just move the reg content if regs differ
608                         */
609                        if (destinationReg.getNumber() != sourceReg.getNumber()) {
610                stmtV.addInsn(StmtVisitor.buildMoveInsn(destinationReg, sourceReg), origStmt);
611                        }
612                } else if (needsCastThroughInt(sourceType, castType)) {
613                        /* 
614                         * an unsupported "dest = (cast) src" is broken down to
615                         * "tmp = (int) src" and
616                         * "dest = (cast) tmp",
617                         * using a tmp reg to not mess with the original reg types
618                         */
619                        Opcode castToIntOpc = getCastOpc(sourceType, PrimitiveType.INT);
620                        Opcode castFromIntOpc = getCastOpc(PrimitiveType.INT, castType);
621                        Register tmp = regAlloc.asTmpReg(IntType.v());
622            stmtV.addInsn(new Insn12x(castToIntOpc, tmp, sourceReg), origStmt);
623            stmtV.addInsn(new Insn12x(castFromIntOpc, destinationReg, tmp.clone()), origStmt);
624                } else {
625                        // the leftover simple cases, where we just cast as stated
626                        Opcode opc = getCastOpc(sourceType, castType);
627            stmtV.addInsn(new Insn12x(opc, destinationReg, sourceReg), origStmt);
628                }
629        }
630
631        private boolean needsCastThroughInt(PrimitiveType sourceType, PrimitiveType castType) {
632                // source >= long && cast < int -> too far away to cast directly
633                return isEqualOrBigger(sourceType, PrimitiveType.LONG) && !isEqualOrBigger(castType, PrimitiveType.INT);
634        }
635
636        private boolean isMoveCompatible(PrimitiveType sourceType, PrimitiveType castType) {
637                if (sourceType == castType) {
638                        // at this point, the types are "bigger" or equal to int, so no "should cast from int" is needed 
639                        return true;
640                }
641                if (castType == PrimitiveType.INT && !isEqualOrBigger(sourceType, PrimitiveType.INT)) {
642                        // there is no "upgrade" cast from "smaller than int" to int, so move it
643                        return true;
644                }
645                return false;
646        }
647
648        private boolean shouldCastFromInt(PrimitiveType sourceType, PrimitiveType castType) {
649                if (isEqualOrBigger(sourceType, PrimitiveType.INT)) {
650                        // source is already "big" enough
651                        return false;
652                }
653                if (castType == PrimitiveType.INT) {
654                        // would lead to an int-to-int cast, so leave it as it is
655                        return false;
656                }
657                return true;
658        }
659
660        private boolean isEqualOrBigger(PrimitiveType type, PrimitiveType relativeTo) {
661                return type.compareTo(relativeTo) >= 0;
662        }
663
664        private Opcode getCastOpc(PrimitiveType sourceType, PrimitiveType castType) {
665                Opcode opc = Opcode.valueOf(sourceType.getName().toUpperCase() + "_TO_"
666                                + castType.getName().toUpperCase());
667                if (opc == null) {
668                        throw new RuntimeException("unsupported cast from " + sourceType + " to " + castType);
669                }
670                return opc;
671        }

673        @Override
674        public void caseNewArrayExpr(NewArrayExpr nae) {
675                Value size = nae.getSize();
676                constantV.setOrigStmt(origStmt);
677                Register sizeReg = regAlloc.asImmediate(size, constantV);
678                ArrayType arrayType = nae.getBaseType().getArrayType();
679                BuilderReference arrayTypeItem = DexPrinter.toTypeReference
680                                (arrayType, stmtV.getBelongingFile());
681        stmtV.addInsn(new Insn22c(Opcode.NEW_ARRAY, destinationReg, sizeReg, arrayTypeItem), origStmt);
682        }

684        @Override
685        public void caseNewMultiArrayExpr(NewMultiArrayExpr nmae) {
686                constantV.setOrigStmt(origStmt);
687                // get array dimensions
688                if (nmae.getSizeCount() > 255) {
689                        throw new RuntimeException("number of dimensions is too high (> 255) for the filled-new-array* opcodes: " + nmae.getSizeCount());
690                }
691                short dimensions = (short) nmae.getSizeCount();
692                // get array base type
693                ArrayType arrayType = ArrayType.v(nmae.getBaseType().baseType, dimensions);
694                BuilderReference arrayTypeItem = DexPrinter.toTypeReference
695                                (arrayType, stmtV.getBelongingFile());
696                // get the dimension size registers
697                List<Register> dimensionSizeRegs = new ArrayList<Register>();
698                for (int i = 0; i < dimensions; i++) {
699                        Value currentDimensionSize = nmae.getSize(i);
700                        Register currentReg = regAlloc.asImmediate(currentDimensionSize, constantV);
701                        dimensionSizeRegs.add(currentReg);
702                }
703                // create filled-new-array instruction, depending on the dimension sizes
704                if (dimensions <= 5) {
705                        Register[] paddedRegs = pad35cRegs(dimensionSizeRegs);
706            stmtV.addInsn(new Insn35c(Opcode.FILLED_NEW_ARRAY, dimensions, paddedRegs[0],
707                    paddedRegs[1], paddedRegs[2], paddedRegs[3], paddedRegs[4], arrayTypeItem),
708                    null);
709                } else {
710            stmtV.addInsn(new Insn3rc(Opcode.FILLED_NEW_ARRAY_RANGE, dimensionSizeRegs, dimensions,
711                    arrayTypeItem), null);
712                } // check for > 255 is done already
713                // move the resulting array into the destination register
714        stmtV.addInsn(new Insn11x(Opcode.MOVE_RESULT_OBJECT, destinationReg), origStmt);
715        }
