136/**
137 * Main entry point for the "dex" output format.<br>
138 * <br>
139 * Use {@link #add(SootClass)} to add classes that should be printed as dex output and {@link #print()} to finally print the classes.<br>
140 * If the printer has found the original APK of an added class (via {@link SourceLocator#dexClassIndex()}),
141 * the files in the APK are copied to a new one, replacing it's classes.dex and excluding the signature files.
142 * Note that you have to sign and align the APK yourself, with jarsigner and zipalign, respectively.<br>
143 * If there is no original APK, the printer just emits a classes.dex.
144 * 
145 * @see <a href="http://docs.oracle.com/javase/7/docs/technotes/tools/windows/jarsigner.html">jarsigner documentation</a>
146 * @see <a href="http://developer.android.com/tools/help/zipalign.html">zipalign documentation</a>
147 */
148public class DexPrinter {

161        private void printApk(String outputDir, File originalApk) throws IOException {
162                ZipOutputStream outputApk = null;
163                if(Options.v().output_jar()) {
164                        outputApk = PackManager.v().getJarFile();
165                        G.v().out.println("Writing APK to: " + Options.v().output_dir());
166                } else {
167                        String outputFileName = outputDir + File.separatorChar + originalApk.getName();
168                
169                        File outputFile = new File(outputFileName);
170                        if(outputFile.exists() && !Options.v().force_overwrite()) {
171                                throw new CompilationDeathException("Output file "+outputFile+" exists. Not overwriting.");
172                        } 
173                        outputApk = new ZipOutputStream(new FileOutputStream(outputFile));
174                        G.v().out.println("Writing APK to: " + outputFileName);
175                }
176                G.v().out.println("do not forget to sign the .apk file with jarsigner and to align it with zipalign");
177                
178                // Copy over additional resources from original APK
179                ZipFile original = null;
180                try {
181                        original = new ZipFile(originalApk);
182                        copyAllButClassesDexAndSigFiles(original, outputApk);
183                }
184                finally {
185                        if (original != null)
186                                original.close();
187                }
188                
189                // put our classes.dex into the zip archive
190                File tmpFile = File.createTempFile("toDex", null);
191                FileInputStream fis = new FileInputStream(tmpFile);
192                try {
193                        outputApk.putNextEntry(new ZipEntry(CLASSES_DEX));
194                        writeTo(tmpFile.getAbsolutePath());
195                        while (fis.available() > 0) {
196                                byte[] data = new byte[fis.available()];
197                                fis.read(data);
198                                outputApk.write(data);
199                        }
200                        outputApk.closeEntry();
201                }
202                finally {
203                        fis.close();
204                        tmpFile.delete();
205                        if (outputApk != null)
206                                outputApk.close();
207                }
208        }

210        private void copyAllButClassesDexAndSigFiles(ZipFile source, ZipOutputStream destination) throws IOException {
211                Enumeration<? extends ZipEntry> sourceEntries = source.entries();
212                while (sourceEntries.hasMoreElements()) {
213                        ZipEntry sourceEntry = sourceEntries.nextElement();
214                        String sourceEntryName = sourceEntry.getName();
215                        if (sourceEntryName.equals(CLASSES_DEX) || isSignatureFile(sourceEntryName)) {
216                                continue;
217                        }
218                        // separate ZipEntry avoids compression problems due to encodings
219                        ZipEntry destinationEntry = new ZipEntry(sourceEntryName);
220                        // use the same compression method as the original (certain files are stored, not compressed)
221                        destinationEntry.setMethod(sourceEntry.getMethod());
222                        // copy other necessary fields for STORE method
223                        destinationEntry.setSize(sourceEntry.getSize());
224                        destinationEntry.setCrc(sourceEntry.getCrc());
225                        // finally craft new entry
226                        destination.putNextEntry(destinationEntry);
227                        InputStream zipEntryInput = source.getInputStream(sourceEntry);
228                        byte[] buffer = new byte[2048];
229                        int bytesRead = zipEntryInput.read(buffer);
230                        while (bytesRead > 0) {
231                                destination.write(buffer, 0, bytesRead);
232                                bytesRead = zipEntryInput.read(buffer);
233                        }
234                        zipEntryInput.close();
235                }
236        }

238        private static boolean isSignatureFile(String fileName) {
239                StringBuilder sigFileRegex = new StringBuilder();
240                // file name must start with META-INF...
241                sigFileRegex.append("META\\-INF");
242                // ...followed by a zip file separator...
243                sigFileRegex.append('/');
244                // ...followed by anything but a zip file separator...
245                sigFileRegex.append("[^/]+");
246                // ...ending with .SF, .DSA, .RSA or .EC
247                sigFileRegex.append("(\\.SF|\\.DSA|\\.RSA|\\.EC)$");
248                return fileName.matches(sigFileRegex.toString());
249        }

251        private void writeTo(String fileName) throws IOException {
252                FileDataStore fds = new FileDataStore(new File(fileName));
253                dexFile.writeTo(fds);
254                fds.close();

257    /**
258     * Encodes Annotations Elements from Jimple to Dexlib
259     * @param elem Jimple Element
260     * @return Dexlib encoded element
261     */
262    private EncodedValue buildEncodedValueForAnnotation(AnnotationElem elem){
263        switch (elem.getKind()) {
264        case 'Z': {
265                if (elem instanceof AnnotationIntElem) {
266                    AnnotationIntElem e = (AnnotationIntElem)elem;
267                    if (e.getValue() == 0) {
268                        return ImmutableBooleanEncodedValue.FALSE_VALUE;
269                    } else if (e.getValue() == 1) {
270                        return ImmutableBooleanEncodedValue.TRUE_VALUE;
271                    } else {
272                        throw new RuntimeException("error: boolean value from int with value != 0 or 1.");
273                    }
274                }
275                else if (elem instanceof AnnotationBooleanElem) {
276                        AnnotationBooleanElem e = (AnnotationBooleanElem) elem;
277                        if (e.getValue())
278                                return ImmutableBooleanEncodedValue.TRUE_VALUE;
279                        else
280                                return ImmutableBooleanEncodedValue.FALSE_VALUE;
281                }
282                else
283                        throw new RuntimeException("Annotation type incompatible with target type boolean");
284        }
285        case 'S': {
286            AnnotationIntElem e = (AnnotationIntElem)elem;
287            return new ImmutableShortEncodedValue((short)e.getValue());
288        }
289        case 'B': {
290            AnnotationIntElem e = (AnnotationIntElem)elem;
291            return new ImmutableByteEncodedValue((byte)e.getValue());
292        }
293        case 'C': {
294            AnnotationIntElem e = (AnnotationIntElem)elem;
295            return new ImmutableCharEncodedValue((char)e.getValue());
296        }
297        case 'I': {
298            AnnotationIntElem e = (AnnotationIntElem)elem;
299            return new ImmutableIntEncodedValue(e.getValue());
300        }
301        case 'J': {
302            AnnotationLongElem e = (AnnotationLongElem)elem;
303            return new ImmutableLongEncodedValue(e.getValue());
304        }
305        case 'F': {
306            AnnotationFloatElem e = (AnnotationFloatElem)elem;
307            return new ImmutableFloatEncodedValue(e.getValue());
308        }
309        case 'D': {
310            AnnotationDoubleElem e = (AnnotationDoubleElem)elem;
311            return new ImmutableDoubleEncodedValue(e.getValue());
312        }
313        case 's': {
314            AnnotationStringElem e = (AnnotationStringElem)elem;
315            return new ImmutableStringEncodedValue(e.getValue());
316        }
317        case 'e': {
318            AnnotationEnumElem e = (AnnotationEnumElem)elem;
319            
320            String classT = SootToDexUtils.getDexClassName(e.getTypeName());
321            String fieldT = classT;
322            
323            FieldReference fref = dexFile.internFieldReference(new ImmutableFieldReference
324                        (classT, e.getConstantName(), fieldT));
325            
326            return new ImmutableEnumEncodedValue(fref);
327        }
328        case 'c': {
329            AnnotationClassElem e = (AnnotationClassElem)elem;
330            return new ImmutableTypeEncodedValue(e.getDesc());
331        }
332        case '[': {
333            AnnotationArrayElem e = (AnnotationArrayElem)elem;
334            List<EncodedValue> values = new ArrayList<EncodedValue>();
335            for (int i = 0; i < e.getNumValues(); i++){
336                EncodedValue val = buildEncodedValueForAnnotation(e.getValueAt(i));
337                values.add(val);
338            }
339            return new ImmutableArrayEncodedValue(values);
340        }
341        case '@': {
342                AnnotationAnnotationElem e = (AnnotationAnnotationElem)elem;
343                
344                Set<String> alreadyWritten = new HashSet<String>();
345            List<AnnotationElement> elements = null;
346            if (!e.getValue().getElems().isEmpty()) {
347                elements = new ArrayList<AnnotationElement>();
348                    for (AnnotationElem ae : e.getValue().getElems()) {
349                        if (!alreadyWritten.add(ae.getName()))
350                                throw new RuntimeException("Duplicate annotation attribute: " + ae.getName());
351                        
352                        AnnotationElement element = new ImmutableAnnotationElement(ae.getName(),
353                                        buildEncodedValueForAnnotation(ae));
354                        elements.add(element);
355                    }
356            }
357                        
358            return new ImmutableAnnotationEncodedValue
359                        (SootToDexUtils.getDexClassName(e.getValue().getType()),
360                        elements);
361        }
362        case 'f': { // field (Dalvik specific?)
363            AnnotationStringElem e = (AnnotationStringElem)elem;
364            
365            String fSig = e.getValue();
366            String[] sp = fSig.split(" ");
367            String classString = SootToDexUtils.getDexClassName(sp[0].split(":")[0]);
368            if (classString.isEmpty())
369                throw new RuntimeException("Empty class name in annotation");
370
371            String typeString = sp[1];
372            if (typeString.isEmpty())
373                throw new RuntimeException("Empty type string in annotation");
374            
375            String fieldName = sp[2];
376            
377            FieldReference fref = dexFile.internFieldReference(new ImmutableFieldReference
378                        (classString, fieldName, typeString));
379            return new ImmutableFieldEncodedValue(fref);
380        }
381        case 'M': { // method (Dalvik specific?)
382            AnnotationStringElem e = (AnnotationStringElem)elem;
383            
384            String[] sp = e.getValue().split(" ");
385            String classString = SootToDexUtils.getDexClassName(sp[0].split(":")[0]);
386            if (classString.isEmpty())
387                throw new RuntimeException("Empty class name in annotation");
388
389            String returnType = sp[1];
390            String[] sp2 = sp[2].split("\\(");
391            String methodNameString = sp2[0];
392            
393            String parameters = sp2[1].replaceAll("\\)", "");
394            List<String> paramTypeList = null;
395            if (!parameters.isEmpty()) {
396                paramTypeList = new ArrayList<String>();
397                    if (parameters.length() > 0)
398                        for (String p: parameters.split(",")) {
399                            paramTypeList.add(p);
400                        }
401            }
402            
403            MethodReference mref = dexFile.internMethodReference(new ImmutableMethodReference
404                        (classString, methodNameString, paramTypeList, returnType));
405            return new ImmutableMethodEncodedValue(mref);
406        }
407        case 'N': { // null (Dalvik specific?)
408                return ImmutableNullEncodedValue.INSTANCE;
409        }
410        default :
411            throw new RuntimeException("Unknown Elem Attr Kind: "+elem.getKind());
412        }
413    }

415    private EncodedValue makeConstantItem(SootField sf, Tag t) {
416        if (!(t instanceof ConstantValueTag))
417            throw new RuntimeException("error: t not ConstantValueTag.");
418
419        if (t instanceof IntegerConstantValueTag) {
420            Type sft = sf.getType();
421            IntegerConstantValueTag i = (IntegerConstantValueTag) t;
422            if (sft instanceof BooleanType) {
423                int v = i.getIntValue();
424                if (v == 0) {
425                    return ImmutableBooleanEncodedValue.FALSE_VALUE;
426                } else if (v == 1) {
427                        return ImmutableBooleanEncodedValue.TRUE_VALUE;
428                } else {
429                    throw new RuntimeException(
430                            "error: boolean value from int with value != 0 or 1.");
431                }
432            } else if (sft instanceof CharType) {
433                return new ImmutableCharEncodedValue((char) i.getIntValue());
434            } else if (sft instanceof ByteType) {
435                return new ImmutableByteEncodedValue((byte) i.getIntValue());
436            } else if (sft instanceof IntType) {
437                return new ImmutableIntEncodedValue(i.getIntValue());
438            } else if (sft instanceof ShortType) {
439                return new ImmutableShortEncodedValue((short) i.getIntValue());
440            } else {
441                throw new RuntimeException("error: unexpected constant tag type: " + t
442                        + " for field " + sf);
443            }
444        } else if (t instanceof LongConstantValueTag) {
445            LongConstantValueTag l = (LongConstantValueTag) t;
446            return new ImmutableLongEncodedValue(l.getLongValue());
447        } else if (t instanceof DoubleConstantValueTag) {
448            DoubleConstantValueTag d = (DoubleConstantValueTag) t;
449            return new ImmutableDoubleEncodedValue(d.getDoubleValue());
450        } else if (t instanceof FloatConstantValueTag) {
451            FloatConstantValueTag f = (FloatConstantValueTag) t;
452            return new ImmutableFloatEncodedValue(f.getFloatValue());
453        } else if (t instanceof StringConstantValueTag) {
454            StringConstantValueTag s = (StringConstantValueTag) t;
455            return new ImmutableStringEncodedValue(s.getStringValue());
456        } else
457                throw new RuntimeException("Unexpected constant type");
458    }

460    private void addAsClassDefItem(SootClass c) {
461        // add source file tag if any
462        String sourceFile = null;
463        if (c.hasTag("SourceFileTag")) {
464            SourceFileTag sft = (SourceFileTag) c.getTag("SourceFileTag");
465            sourceFile = sft.getSourceFile();
466        }
467        
468        String classType = SootToDexUtils.getDexTypeDescriptor(c.getType());
469        int accessFlags = c.getModifiers();
470        String superClass = c.hasSuperclass() ?
471                        SootToDexUtils.getDexTypeDescriptor(c.getSuperclass().getType()) : null;
472        
473        List<String> interfaces = null;
474        if (!c.getInterfaces().isEmpty()) {
475                interfaces = new ArrayList<String>();
476            for (SootClass ifc : c.getInterfaces())
477                interfaces.add(SootToDexUtils.getDexTypeDescriptor(ifc.getType()));
478        }
479        
480        List<BuilderField> fields = null;
481        if (!c.getFields().isEmpty()) {
482                fields = new ArrayList<BuilderField>();
483                for (SootField f : c.getFields()) {             
484                        // Look for a static initializer
485                    EncodedValue staticInit = null;
486                    for (Tag t : f.getTags()) {
487                        if (t instanceof ConstantValueTag) {
488                            if (staticInit != null) {
489                                G.v().out.println("warning: more than one constant tag for field: " + f + ": "
490                                        + t);
491                            } else {
492                                staticInit = makeConstantItem(f, t);
493                            }
494                        }
495                    }
496                    if (staticInit == null)
497                        staticInit = BuilderEncodedValues.defaultValueForType
498                                        (SootToDexUtils.getDexTypeDescriptor(f.getType()));
499                    
500                    // Build field annotations
501                    Set<Annotation> fieldAnnotations = buildFieldAnnotations(f);
502                    
503                        BuilderField field = dexFile.internField(classType,
504                                        f.getName(),
505                                        SootToDexUtils.getDexTypeDescriptor(f.getType()),
506                                        f.getModifiers(),
507                                        staticInit,
508                                        fieldAnnotations);
509                        fields.add(field);
510                }
511        }
512                
513        dexFile.internClassDef(classType,
514                        accessFlags,
515                        superClass,
516                        interfaces,
517                        sourceFile,
518                        buildClassAnnotations(c),
519                        fields,
520                        toMethods(c));
521        }

523    private Set<Annotation> buildClassAnnotations(SootClass c) {
524        Set<String> skipList = new HashSet<String>();
525        Set<Annotation> annotations = buildCommonAnnotations(c, skipList);
526        
527        // Classes can have either EnclosingMethod or EnclosingClass tags. Soot
528        // sets the outer class for both "normal" and anonymous inner classes,
529        // so we test for enclosing methods first. 
530        if (c.hasTag("EnclosingMethodTag")) {
531                EnclosingMethodTag eMethTag = (EnclosingMethodTag)c.getTag("EnclosingMethodTag");
532                Annotation enclosingMethodItem = buildEnclosingMethodTag(eMethTag, skipList);
533                if (enclosingMethodItem != null)
534                  annotations.add(enclosingMethodItem);
535        }
536        else if (c.hasOuterClass()) {
537                        if (skipList.add("Ldalvik/annotation/EnclosingClass;")) {
538                        // EnclosingClass annotation
539                        ImmutableAnnotationElement enclosingElement = new ImmutableAnnotationElement
540                                        ("value", new ImmutableTypeEncodedValue
541                                                        (SootToDexUtils.getDexClassName(c.getOuterClass().getName())));
542                        annotations.add(new ImmutableAnnotation(AnnotationVisibility.SYSTEM,
543                                        "Ldalvik/annotation/EnclosingClass;",
544                                        Collections.singleton(enclosingElement)));
545                }
546        }
547        
548        // If we have an outer class, we also pick up the InnerClass annotations
549        // from there. Note that Java and Soot associate InnerClass annotations
550        // with the respective outer classes, while Dalvik puts them on the
551        // respective inner classes.
552        if (c.hasOuterClass()) {
553                InnerClassAttribute icTag = (InnerClassAttribute) c.getOuterClass().getTag("InnerClassAttribute");
554                if (icTag != null) {
555                        List<Annotation> innerClassItem = buildInnerClassAttribute(c, icTag, skipList);
556                        if (innerClassItem != null)
557                          annotations.addAll(innerClassItem);
558                }
559        }
560        
561        // Write the MemberClasses tag
562        InnerClassAttribute icTag = (InnerClassAttribute) c.getTag("InnerClassAttribute");
563        if (icTag != null) {
564                List<Annotation> memberClassesItem = buildMemberClassesAttribute(c, icTag, skipList);
565                if (memberClassesItem != null)
566                  annotations.addAll(memberClassesItem);
567        }
568        
569        
570        for (Tag t : c.getTags()) {
571            if (t.getName().equals("VisibilityAnnotationTag")){
572                List<ImmutableAnnotation> visibilityItems = buildVisibilityAnnotationTag
573                                ((VisibilityAnnotationTag) t, skipList);
574                annotations.addAll(visibilityItems);
575            }
576        }
577        
578
579        //Write default-annotation tags
580                List<AnnotationElem> defaults = new ArrayList<AnnotationElem>();
581        for (SootMethod method : c.getMethods()) {
582                AnnotationDefaultTag tag = (AnnotationDefaultTag) method.getTag("AnnotationDefaultTag");
583                if (tag != null) {
584                        tag.getDefaultVal().setName(method.getName());
585                        defaults.add(tag.getDefaultVal());
586                }
587        }
588        if (defaults.size() > 0) {
589                VisibilityAnnotationTag defaultAnnotationTag = new VisibilityAnnotationTag(AnnotationConstants.RUNTIME_INVISIBLE);
590                AnnotationTag a = new AnnotationTag("Ldalvik/annotation/AnnotationDefault;");
591                defaultAnnotationTag.addAnnotation(a);
592
593                AnnotationTag at = new AnnotationTag(SootToDexUtils.getDexClassName(c.getName()));
594                AnnotationAnnotationElem ae = new AnnotationAnnotationElem(at, '@', "value");
595                a.addElem(ae);
596
597                
598                for (AnnotationElem aelem : defaults)
599                        at.addElem(aelem);
600                
601            List<ImmutableAnnotation> visibilityItems = buildVisibilityAnnotationTag
602                        (defaultAnnotationTag, skipList);
603                annotations.addAll(visibilityItems);
604        }
605        
606        return annotations;
607    }

609    private Set<Annotation> buildFieldAnnotations(SootField f) {
610        Set<String> skipList = new HashSet<String>();

719    private List<ImmutableAnnotation> buildVisibilityAnnotationTag
720                        (VisibilityAnnotationTag t, Set<String> skipList) {
721        if (t.getAnnotations() == null)
722                return Collections.emptyList();
723        
724        List<ImmutableAnnotation> annotations = new ArrayList<ImmutableAnnotation>();
725        for (AnnotationTag at: t.getAnnotations()) {
726            String type = at.getType();
727            if (!skipList.add(type))
728                continue;
729            
730            Set<String> alreadyWritten = new HashSet<String>();
731            List<AnnotationElement> elements = null;
732            if (!at.getElems().isEmpty()) {
733                elements = new ArrayList<AnnotationElement>();
734                    for (AnnotationElem ae : at.getElems()) {
735                        if (ae.getName() == null || ae.getName().isEmpty())
736                                throw new RuntimeException("Null or empty annotation name encountered");
737                        if (!alreadyWritten.add(ae.getName()))
738                                throw new RuntimeException("Duplicate annotation attribute: " + ae.getName());
739                        
740                        EncodedValue value = buildEncodedValueForAnnotation(ae);
741                        ImmutableAnnotationElement element = new ImmutableAnnotationElement
742                                        (ae.getName(), value);
743                        elements.add(element);
744                    }
745            }
746            
747            String typeName = SootToDexUtils.getDexClassName(at.getType());
748            ImmutableAnnotation ann = new ImmutableAnnotation(getVisibility(t.getVisibility()),
749                        typeName, elements);
750            annotations.add(ann);
751        }
752        return annotations;
753        }

829    private List<Annotation> buildInnerClassAttribute(SootClass parentClass,
830                InnerClassAttribute t, Set<String> skipList) {
831        if (t.getSpecs() == null)
832                return null;
833        
834        List<Annotation> anns = null;
835        for (Tag t2 : t.getSpecs()) {
836                InnerClassTag icTag = (InnerClassTag) t2;
837                
838                // In Dalvik, both the EnclosingMethod/EnclosingClass tag and the
839                // InnerClass tag are written to the inner class which is different
840                // to Java. We thus check whether this tag actually points to our
841                // outer class.
842                String outerClass = getOuterClassNameFromTag(icTag);
843                        String innerClass = icTag.getInnerClass().replaceAll("/", ".");
844                                                
845                        // Only write the InnerClass tag to the inner class itself, not
846                        // the other one. If the outer class points to our parent, but
847                        // this is simply the wrong inner class, we also continue with the
848                        // next tag.
849                if (!parentClass.hasOuterClass()
850                                || !innerClass.equals(parentClass.getName()))
851                        continue;
852                
853                // If the outer class points to the very same class, we null it
854                if (parentClass.getName().equals(outerClass)
855                                && icTag.getOuterClass() == null)
856                        outerClass = null;
857                
858                // Do not write garbage. Never.
859                if (parentClass.getName().equals(outerClass))
860                        continue;
861                
862                // This is an actual inner class. Write the annotation
863                if (skipList.add("Ldalvik/annotation/InnerClass;")) {
864                        // InnerClass annotation
865                        List<AnnotationElement> elements = new ArrayList<AnnotationElement>();
866                        
867                        ImmutableAnnotationElement flagsElement = new ImmutableAnnotationElement
868                                        ("accessFlags", new ImmutableIntEncodedValue(icTag.getAccessFlags()));
869                        elements.add(flagsElement);
870                        
871                        ImmutableEncodedValue nameValue;
872                        if (icTag.getShortName() != null && !icTag.getShortName().isEmpty())
873                                nameValue = new ImmutableStringEncodedValue(icTag.getShortName());
874                        else
875                                nameValue = ImmutableNullEncodedValue.INSTANCE;
876                        
877                        ImmutableAnnotationElement nameElement = new ImmutableAnnotationElement
878                                        ("name", nameValue);
879                            elements.add(nameElement);
880                        
881                        if (anns == null) anns = new ArrayList<Annotation>();
882                        anns.add(new ImmutableAnnotation(AnnotationVisibility.SYSTEM,
883                                        "Ldalvik/annotation/InnerClass;",
884                                        elements));
885                }
886        }
887                
888        return anns;
889    }

923                        && !memberClasses.isEmpty()

962        private Collection<BuilderMethod> toMethods(SootClass clazz) {
963                if (clazz.getMethods().isEmpty())
964                        return null;
965                
966        String classType = SootToDexUtils.getDexTypeDescriptor(clazz.getType());
967        List<BuilderMethod> methods = new ArrayList<BuilderMethod>();
968        for (SootMethod sm : clazz.getMethods()) {
969            if (sm.isPhantom()) {
970                // Do not print method bodies for inherited methods
971                continue;
972            }
973            
974                MethodImplementation impl = toMethodImplementation(sm);
975                
976                List<String> parameterNames = null;
977                if (sm.hasTag("ParamNamesTag"))
978                        parameterNames = ((ParamNamesTag) sm.getTag("ParamNamesTag")).getNames();
979                
980                int paramIdx = 0;
981                List<MethodParameter> parameters = null;
982                if (sm.getParameterCount() > 0) {
983                        parameters = new ArrayList<MethodParameter>();
984                        for (Type tp : sm.getParameterTypes()) {
985                                String paramType = SootToDexUtils.getDexTypeDescriptor(tp);
986                                parameters.add(new ImmutableMethodParameter(paramType,
987                                                buildMethodParameterAnnotations(sm, paramIdx),
988                                                sm.isConcrete() && parameterNames != null ?
989                                                                parameterNames.get(paramIdx) : null));
990                                paramIdx++;
991                        }
992                }
993                
994            String returnType = SootToDexUtils.getDexTypeDescriptor(sm.getReturnType());
995            
996                        int accessFlags = SootToDexUtils.getDexAccessFlags(sm);
997            BuilderMethod meth = dexFile.internMethod(classType,
998                                        sm.getName(),
999                                        parameters,
1000                                        returnType,
1001                                        accessFlags,
1002                                        buildMethodAnnotations(sm),
1003                                        impl);
1004            methods.add(meth);
1005        }
1006                return methods;
1007        }

1037        private MethodImplementation toMethodImplementation(SootMethod m) {
1038                if (m.isAbstract() || m.isNative()) {
1039                        return null;
1040                }
1041                Body activeBody = m.retrieveActiveBody();
1042                
1043                // check the method name to make sure that dexopt won't get into trouble
1044                // when installing the app
1045                if (m.getName().contains("<") || m.getName().equals(">"))
1046                        if (!m.getName().equals("<init>") && !m.getName().equals("<clinit>"))
1047                                throw new RuntimeException("Invalid method name: " + m.getName());
1048                
1049                // Switch statements may not be empty in dex, so we have to fix this first
1050                EmptySwitchEliminator.v().transform(activeBody);
1051                
1052                // Dalvik requires synchronized methods to have explicit monitor calls,
1053                // so we insert them here. See http://milk.com/kodebase/dalvik-docs-mirror/docs/debugger.html
1054                // We cannot place this upon the developer since it is only required
1055                // for Dalvik, but not for other targets.
1056                SynchronizedMethodTransformer.v().transform(activeBody);
1057                
1058                // Tries may not start or end at units which have no corresponding Dalvik
1059                // instructions such as IdentityStmts. We reduce the traps to start at the
1060                // first "real" instruction. We could also use a TrapTigthener, but that
1061                // would be too expensive for what we need here.
1062                FastDexTrapTightener.v().transform(activeBody);
1063                
1064                // Look for sequences of array element assignments that we can collapse
1065                // into bulk initializations
1066                DexArrayInitDetector initDetector = new DexArrayInitDetector();
1067                initDetector.constructArrayInitializations(activeBody);
1068                initDetector.fixTraps(activeBody);
1069                
1070                // Split the tries since Dalvik does not supported nested try/catch blocks
1071                TrapSplitter.v().transform(activeBody);
1072
1073                // word count of incoming parameters
1074                int inWords = SootToDexUtils.getDexWords(m.getParameterTypes());
1075                if (!m.isStatic()) {
1076                        inWords++; // extra word for "this"
1077                }
1078                // word count of max outgoing parameters
1079                Collection<Unit> units = activeBody.getUnits();
1080                // register count = parameters + additional registers, depending on the dex instructions generated (e.g. locals used and constants loaded)
1081                StmtVisitor stmtV = new StmtVisitor(m, dexFile, initDetector);
1082                
1083                toInstructions(units, stmtV);
1084                
1085                int registerCount = stmtV.getRegisterCount();
1086                if (inWords > registerCount) {
1087                        /*
1088                         * as the Dalvik VM moves the parameters into the last registers, the "in" word count must be at least equal to the register count.
1089                         * a smaller register count could occur if soot generated the method body, see e.g. the handling of phantom refs in SootMethodRefImpl.resolve(StringBuffer):
1090                         * the body has no locals for the ParameterRefs, it just throws an error.
1091                         * 
1092                         * we satisfy the verifier by just increasing the register count, since calling phantom refs will lead to an error anyway.
1093                         */
1094                        registerCount = inWords;
1095                }
1096                
1097                MethodImplementationBuilder builder = new MethodImplementationBuilder(registerCount);
1098                LabelAssigner labelAssinger = new LabelAssigner(builder);
1099                List<BuilderInstruction> instructions = stmtV.getRealInsns(labelAssinger);
1100                
1101                fixLongJumps(instructions, labelAssinger, stmtV);
1102                
1103                Map<Local, Integer> seenRegisters = new HashMap<Local, Integer>();
1104                Map<Instruction, LocalRegisterAssignmentInformation> instructionRegisterMap = stmtV.getInstructionRegisterMap();
1105                
1106                if (Options.v().write_local_annotations()) {
1107                        for (LocalRegisterAssignmentInformation assignment : stmtV.getParameterInstructionsList()) {
1108                                //The "this" local gets added automatically, so we do not need to add it explicitly
1109                                //(at least not if it exists with exactly this name)
1110                                if (assignment.getLocal().getName().equals("this"))
1111                                        continue;
1112                                addRegisterAssignmentDebugInfo(assignment, seenRegisters, builder);
1113                        }
1114                }
1115        
1116        for (BuilderInstruction ins : instructions) {
1117            Stmt origStmt = stmtV.getStmtForInstruction(ins);
1118            
1119            // If this is a switch payload, we need to place the label
1120            if (stmtV.getInstructionPayloadMap().containsKey(ins))
1121                builder.addLabel(labelAssinger.getLabelName(stmtV.getInstructionPayloadMap().get(ins)));
1122            
1123            if (origStmt != null) {
1124                    // Do we need a label here because this a trap handler?
1125                    for (Trap t : m.getActiveBody().getTraps()) {
1126                        if (t.getBeginUnit() == origStmt
1127                                        || t.getEndUnit() == origStmt
1128                                        || t.getHandlerUnit() == origStmt) {
1129                                labelAssinger.getOrCreateLabel(origStmt);
1130                                break;
1131                        }
1132                    }
1133                    
1134                        // Add the label if the statement has one
1135                                String labelName = labelAssinger.getLabelName(origStmt);
1136                                if (labelName != null && !builder.getLabel(labelName).isPlaced())
1137                                        builder.addLabel(labelName);
1138                                
1139                                // Add the tags
1140                                if (stmtV.getStmtForInstruction(ins) != null) {
1141                        List<Tag> tags = origStmt.getTags();
1142                        for (Tag t : tags) {
1143                            if (t instanceof LineNumberTag) {
1144                                LineNumberTag lnt = (LineNumberTag) t;
1145                                        builder.addLineNumber(lnt.getLineNumber());
1146                            }
1147                            else if (t instanceof SourceFileTag) {
1148                                SourceFileTag sft = (SourceFileTag) t;
1149                                builder.addSetSourceFile(dexFile.internStringReference
1150                                                (sft.getSourceFile()));
1151                            }
1152                        }
1153                    }
1154            }
1155            
1156            builder.addInstruction(ins);
1157            LocalRegisterAssignmentInformation registerAssignmentTag = instructionRegisterMap.get(ins);
1158            if (registerAssignmentTag != null)
1159            {
1160                                //Add start local debugging information: Register -> Local assignment
1161                        addRegisterAssignmentDebugInfo(registerAssignmentTag, seenRegisters, builder);
1162            }
1163                }
1164                
1165                for (int registersLeft : seenRegisters.values())
1166                        builder.addEndLocal(registersLeft);
1167                
1168                toTries(activeBody.getTraps(), stmtV, builder, labelAssinger);
1169        
1170        // Make sure that all labels have been placed by now
1171        for (Label lbl : labelAssinger.getAllLabels())
1172                if (!lbl.isPlaced())
1173                        throw new RuntimeException("Label not placed: " + lbl);
1174        
1175        return builder.getMethodImplementation();
1176        }

1178        /**
1179         * Fixes long jumps that exceed the maximum distance for the respective jump
1180         * type
1181         * @param instructions The list of generated dalvik instructions
1182         * @param labelAssigner The label assigner that maps statements to labels
1183         * @param stmtV The statement visitor used to produce the dalvik instructions
1184         */
1185        private void fixLongJumps(List<BuilderInstruction> instructions,
1186                        LabelAssigner labelAssigner, StmtVisitor stmtV) {
1187                // Only construct the maps once and update them afterwards
1188                Map<Instruction, Integer> instructionsToIndex = new HashMap<Instruction, Integer>();
1189                List<Integer> instructionsToOffsets = new ArrayList<Integer>();
1190                Map<Label, Integer> labelsToOffsets = new HashMap<Label, Integer>();
1191                Map<Label, Integer> labelsToIndex = new HashMap<Label, Integer>();
1192                
1193                boolean hasChanged;
1194                l0 : do {
1195                        // Look for changes anew every time
1196                        hasChanged = false;
1197                        instructionsToOffsets.clear();
1198                        
1199                        // Build a mapping between instructions and offsets
1200                        {
1201                        int offset = 0;
1202                        int idx = 0;
1203                        for (BuilderInstruction bi : instructions) {
1204                                instructionsToIndex.put(bi, idx);
1205                                instructionsToOffsets.add(offset);
1206                    Stmt origStmt = stmtV.getStmtForInstruction(bi);
1207                    if (origStmt != null) {
1208                        Label lbl = labelAssigner.getLabelUnsafe(origStmt);
1209                        if (lbl != null) {
1210                                labelsToOffsets.put(lbl, offset);
1211                                labelsToIndex.put(lbl, idx);
1212                        }
1213                    }
1214                    offset += (bi.getFormat().size / 2);
1215                    idx++;
1216                        }
1217                        }
1218                        
1219                        // Look for references to labels
1220                        for (int j = 0; j < instructions.size(); j++) {
1221                                BuilderInstruction bj = instructions.get(j);
1222                                if (bj instanceof BuilderOffsetInstruction) {
1223                                        BuilderOffsetInstruction boj = (BuilderOffsetInstruction) bj;
1224                                        
1225                                        // Compute the distance between the instructions
1226                                        Insn jumpInsn = stmtV.getInsnForInstruction(boj);
1227                                        if (jumpInsn instanceof InsnWithOffset) {
1228                                                InsnWithOffset offsetInsn = (InsnWithOffset) jumpInsn;
1229                                                Integer targetOffset = labelsToOffsets.get(boj.getTarget());
1230                                                if (targetOffset == null)
1231                                                        continue;
1232                                                
1233                                                int distance = instructionsToOffsets.get(j) - targetOffset;
1234                                                if (Math.abs(distance) > offsetInsn.getMaxJumpOffset()) {
1235                                                        // We need intermediate jumps
1236                                                        insertIntermediateJump(labelsToIndex.get(boj.getTarget()),
1237                                                                        j, stmtV, instructions, labelAssigner);
1238                                                        hasChanged = true;
1239                                                        continue l0;
1240                                                }
1241                                        }
1242                                }
1243                        }
1244                } while (hasChanged);
1245        }

1247        /**
1248         * Creates an intermediate jump instruction between the original jump
1249         * instruction and its target
1250         * @param targetInsPos The jump target index
1251         * @param jumpInsPos The position of the jump instruction
1252         * @param stmtV The statement visitor used for constructing the instructions
1253         * @param instructions The list of Dalvik instructions
1254         * @param labelAssigner The label assigner to be used for creating new labels
1255         */
1256        private void insertIntermediateJump(int targetInsPos, int jumpInsPos,
1257                        StmtVisitor stmtV, List<BuilderInstruction> instructions,
1258                        LabelAssigner labelAssigner) {
1259                // Get the original jump instruction
1260                BuilderInstruction originalJumpInstruction = instructions.get(jumpInsPos);
1261                Insn originalJumpInsn = stmtV.getInsnForInstruction(originalJumpInstruction);
1262                if (originalJumpInsn == null)
1263                        return;
1264                if (!(originalJumpInsn instanceof InsnWithOffset))
1265                        throw new RuntimeException("Unexpected jump instruction target");
1266                InsnWithOffset offsetInsn = (InsnWithOffset) originalJumpInsn;
1267                
1268                // If this is goto instruction, we can just replace it
1269                if (originalJumpInsn instanceof Insn10t) {
1270                        if (originalJumpInsn.getOpcode() == Opcode.GOTO) {
1271                                Insn30t newJump = new Insn30t(Opcode.GOTO_32);
1272                                newJump.setTarget(((Insn10t) originalJumpInsn).getTarget());
1273                                BuilderInstruction newJumpInstruction = newJump.getRealInsn(labelAssigner);
1274                                instructions.remove(jumpInsPos);
1275                                instructions.add(jumpInsPos, newJumpInstruction);
1276                                stmtV. fakeNewInsn(stmtV.getStmtForInstruction(originalJumpInstruction),
1277                                                newJump, newJumpInstruction);
1278                                return;
1279                        }
1280                }
1281                
1282                // Find a position where we can jump to
1283                int distance = Math.max(targetInsPos, jumpInsPos) - Math.min(targetInsPos, jumpInsPos);
1284                if (distance == 0)
1285                        return;
1286                int newJumpIdx = Math.min(targetInsPos, jumpInsPos) + (distance / 2);
1287                int sign = (int) Math.signum(targetInsPos - jumpInsPos);
1288                
1289                // There must be a statement at the instruction after the jump target.
1290                // This statement must not appear at an earlier statement as the jump
1291                // label may otherwise be attached to the wrong statement
1292                do {
1293                        Stmt newStmt = stmtV.getStmtForInstruction(instructions.get(newJumpIdx));
1294                        Stmt prevStmt = newJumpIdx > 0 ? stmtV.getStmtForInstruction(instructions.get(newJumpIdx - 1)) : null;
1295                        
1296                        if (newStmt == null || newStmt == prevStmt) {
1297                                newJumpIdx -= sign;
1298                                if (newJumpIdx < 0 || newJumpIdx >= instructions.size())
1299                                        throw new RuntimeException("No position for inserting intermediate "
1300                                                        + "jump instruction found");
1301                        }
1302                        else
1303                                break;
1304                } while (true);
1305                
1306                // Create a jump instruction from the middle to the end
1307                NopStmt nop = Jimple.v().newNopStmt();
1308                Insn30t newJump = new Insn30t(Opcode.GOTO_32);
1309                newJump.setTarget(stmtV.getStmtForInstruction(instructions.get(targetInsPos)));
1310                BuilderInstruction newJumpInstruction = newJump.getRealInsn(labelAssigner);
1311                instructions.add(newJumpIdx, newJumpInstruction);
1312                stmtV.fakeNewInsn(nop, newJump, newJumpInstruction);
1313                
1314                // We have added something, so we need to fix indices
1315                if (newJumpIdx <= jumpInsPos)
1316                        jumpInsPos++;
1317                if (newJumpIdx <= targetInsPos)
1318                        targetInsPos++;
1319                
1320                // Jump from the original instruction to the new one in the middle
1321                offsetInsn.setTarget(nop);
1322                BuilderInstruction replacementJumpInstruction = offsetInsn.getRealInsn(labelAssigner);
1323                assert instructions.get(jumpInsPos) == originalJumpInstruction;
1324                instructions.remove(jumpInsPos);
1325                instructions.add(jumpInsPos, replacementJumpInstruction);
1326                stmtV.fakeNewInsn(stmtV.getStmtForInstruction(originalJumpInstruction),
1327                                originalJumpInsn, replacementJumpInstruction);
1328                
1329                // Our indices are still fine, because we just replaced something
1330                Stmt afterNewJump = stmtV.getStmtForInstruction(instructions.get(newJumpIdx + 1));
1331                
1332                // Make the original control flow jump around the new artificial jump instruction
1333                Insn10t jumpAround = new Insn10t(Opcode.GOTO);
1334                jumpAround.setTarget(afterNewJump);
1335                BuilderInstruction jumpAroundInstruction = jumpAround.getRealInsn(labelAssigner);
1336                instructions.add(newJumpIdx, jumpAroundInstruction);
1337                stmtV.fakeNewInsn(Jimple.v().newNopStmt(), jumpAround, jumpAroundInstruction);
1338        }

1390        private static class CodeRange {
1391                int startAddress;
1392                int endAddress;
1393                
1394                public CodeRange(int startAddress, int endAddress) {
1395                        this.startAddress = startAddress;
1396                        this.endAddress = endAddress;
1397                }
1398                
1399                /**
1400                 * Checks whether the given code range r is fully enclosed by this code
1401                 * range.
1402                 * @param r The other code range
1403                 * @return True if the given code range r is fully enclosed by this code
1404                 * range, otherwise false.
1405                 */
1406                public boolean containsRange(CodeRange r) {
1407                        return (r.startAddress >= this.startAddress && r.endAddress <= this.endAddress);
1408                }
1409                
1410                /**
1411                 * Checks whether this range overlaps with the given one
1412                 * @param r The region to check for overlaps
1413                 * @return True if this region has a non-empty overlap with the given one
1414                 */
1415                public boolean overlaps(CodeRange r) {
1416                        return (r.startAddress >= this.startAddress && r.startAddress < this.endAddress)
1417                                        || (r.startAddress <= this.startAddress && r.endAddress > this.startAddress);
1418                }
1419                
1420                @Override
1421                public String toString() {
1422                        return this.startAddress + "-" + this.endAddress;
1423                }
1424                
1425                @Override
1426                public boolean equals(Object other) {
1427                        if (other == this)
1428                                return true;
1429                        if (other == null && !(other instanceof CodeRange))
1430                                return false;
1431                        CodeRange cr = (CodeRange) other;
1432                        return (this.startAddress == cr.startAddress && this.endAddress == cr.endAddress);
1433                }
1434                
1435                @Override
1436                public int hashCode() {
1437                        return 17 * startAddress + 13 * endAddress;
1438                }
1439        }

1441        private void toTries(Collection<Trap> traps, StmtVisitor stmtV,
1442                        MethodImplementationBuilder builder, LabelAssigner labelAssigner) {
1443                // Original code: assume that the mapping startCodeAddress -> TryItem is enough for
1444                //              a "code range", ignore different end Units / try lengths
1445                // That's definitely not enough since we can have two handlers H1, H2 with
1446                //              H1:240-322, H2:242-322. There is no valid ordering for such overlapping traps
1447                //              in dex. Current solution: If there is already a trap T' for a subrange of the
1448                //              current trap T, merge T and T' on the fully range of T. This is not a 100%
1449                //              correct since we extend traps over the requested range, but it's better than
1450                //              the previous code that produced APKs which failed Dalvik's bytecode verification.
1451                //              (Steven Arzt, 09.08.2013)
1452                // There are cases in which we need to split traps, e.g. in cases like
1453                //              ( (t1) ... (t2) )<big catch all around it> where the all three handlers do
1454                //              something different. That's why we run the TrapSplitter before we get here.
1455                //              (Steven Arzt, 25.09.2013)
1456                Map<CodeRange, List<ExceptionHandler>> codeRangesToTryItem =
1457                                new LinkedHashMap<CodeRange, List<ExceptionHandler>>();
1458                for (Trap t : traps) {
1459                        // see if there is old handler info at this code range
1460                        Stmt beginStmt = (Stmt) t.getBeginUnit();
1461                        Stmt endStmt = (Stmt) t.getEndUnit();
1462                        
1463                        int startCodeAddress = labelAssigner.getLabel(beginStmt).getCodeAddress();
1464                        int endCodeAddress = labelAssigner.getLabel(endStmt).getCodeAddress();
1465                        CodeRange range = new CodeRange(startCodeAddress, endCodeAddress);
1466                        
1467                String exceptionType = SootToDexUtils.getDexTypeDescriptor(t.getException().getType());
1468                
1469                int codeAddress = labelAssigner.getLabel((Stmt) t.getHandlerUnit()).getCodeAddress();
1470                        ImmutableExceptionHandler exceptionHandler = new ImmutableExceptionHandler
1471                                        (exceptionType, codeAddress);
1472                        
1473                        List<ExceptionHandler> newHandlers = new ArrayList<ExceptionHandler>();
1474                        for (CodeRange r : codeRangesToTryItem.keySet()) {
1475                                // Check whether this range is contained in some other range. We then extend our
1476                                // trap over the bigger range containing this range
1477                                if (r.containsRange(range)) {
1478                                        range.startAddress = r.startAddress;
1479                                        range.endAddress = r.endAddress;
1480                                        
1481                                        // copy the old handlers to a bigger array (the old one cannot be modified...)
1482                                        List<ExceptionHandler> oldHandlers = codeRangesToTryItem.get(r);
1483                                        if (oldHandlers != null)
1484                                                newHandlers.addAll(oldHandlers);
1485                                        break;
1486                                }
1487                                // Check whether the other range is contained in this range. In this case,
1488                                // a smaller range is already in the list. We merge the two over the larger
1489                                // range.
1490                                else if (range.containsRange(r)) {
1491                                        range.startAddress = r.startAddress;
1492                                        range.endAddress = r.endAddress;
1493
1494                                        // just use the newly found handler info
1495                                        List<ExceptionHandler> oldHandlers = codeRangesToTryItem.get(range);
1496                                        if (oldHandlers != null)
1497                                                newHandlers.addAll(oldHandlers);
1498                                        
1499                                        // remove the old range, the new one will be added anyway and contain
1500                                        // the merged handlers
1501                                        codeRangesToTryItem.remove(r);
1502                                        break;
1503                                }
1504                        }
1505                        
1506                        if (!newHandlers.contains(exceptionHandler))
1507                                newHandlers.add(exceptionHandler);
1508                        codeRangesToTryItem.put(range, newHandlers);
1509                }
1510                
1511                // Check for overlaps
1512                for (CodeRange r1 : codeRangesToTryItem.keySet()) {
1513                        for (CodeRange r2 : codeRangesToTryItem.keySet()) {
1514                                if (r1 != r2 && r1.overlaps(r2))
1515                                        System.out.println("WARNING: Trap region overlap detected");
1516                        }
1517                }
1518                
1519                for (CodeRange range : codeRangesToTryItem.keySet()) {
1520                        boolean allCaughtForRange = false;
1521                        for (ExceptionHandler handler : codeRangesToTryItem.get(range)) {
1522                                // If we have a catchall directive for a range and then some follow-up
1523                                // exception handler, we can discard the latter as it will never be used
1524                                // anyway.
1525                                if (allCaughtForRange)
1526                                        continue;
1527                                
1528                                // Normally, we would model catchall as real catchall directives. For
1529                                // some reason, this however fails with an invalid handler index. We
1530                                // therefore hack it using java.lang.Throwable.
1531                                if (handler.getExceptionType().equals("Ljava/lang/Throwable;")) {
1532                                        /*
1533                                        builder.addCatch(labelAssigner.getLabelAtAddress(range.startAddress),
1534                                                        labelAssigner.getLabelAtAddress(range.endAddress),
1535                                                        labelAssigner.getLabelAtAddress(handler.getHandlerCodeAddress()));
1536                                                        */
1537                                        allCaughtForRange = true;
1538                                }
1539//                              else
1540                                        builder.addCatch(dexFile.internTypeReference(handler.getExceptionType()),
1541                                                        labelAssigner.getLabelAtAddress(range.startAddress),
1542                                                        labelAssigner.getLabelAtAddress(range.endAddress),
1543                                                        labelAssigner.getLabelAtAddress(handler.getHandlerCodeAddress()));
1544                        }
1545                }
1546        }

1549                if (c.isPhantom())
1550                        return;
1551                                
1552                addAsClassDefItem(c);
1553                // save original APK for this class, needed to copy all the other files inside
1554                Map<String, File> dexClassIndex = SourceLocator.v().dexClassIndex();
1555        if (dexClassIndex == null) {
1556                return; // no dex classes were loaded
1557        }
1558                File sourceForClass = dexClassIndex.get(c.getName());
1559        if (sourceForClass == null || sourceForClass.getName().endsWith(".dex")) {
1560                return; // a class was written that was not a dex class or the class originates from a .dex file, not an APK
1561        }
1562        if (originalApk != null && !originalApk.equals(sourceForClass)) {
1563                throw new CompilationDeathException("multiple APKs as source of an application are not supported");
1564        }
1565        originalApk = sourceForClass;
1566        }
