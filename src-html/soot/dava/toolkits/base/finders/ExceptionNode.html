020package soot.dava.toolkits.base.finders;
021
022import java.util.Collection;
023import java.util.HashMap;
024import java.util.LinkedList;
025import java.util.List;
026import java.util.Map;
027
028import soot.G;
029import soot.SootClass;
030import soot.dava.internal.asg.AugmentedStmt;
031import soot.dava.internal.asg.AugmentedStmtGraph;
032import soot.util.IterableSet;
033
034public class ExceptionNode {
035        private final IterableSet<AugmentedStmt> body;
036        private IterableSet<AugmentedStmt> tryBody, catchBody;
037        private boolean dirty;
038        private List<AugmentedStmt> exitList;
039        private LinkedList<IterableSet<AugmentedStmt>> catchList;
040        private SootClass exception;
041        private HashMap<IterableSet<AugmentedStmt>, SootClass> catch2except;
042        private AugmentedStmt handlerAugmentedStmt;
043
044        public ExceptionNode(IterableSet<AugmentedStmt> tryBody, SootClass exception,
045                        AugmentedStmt handlerAugmentedStmt) {
046                this.tryBody = tryBody;
047                this.catchBody = null;
048                this.exception = exception;
049                this.handlerAugmentedStmt = handlerAugmentedStmt;
050
051                body = new IterableSet<AugmentedStmt>(tryBody);
052
053                dirty = true;
054                exitList = null;
055                catchList = null;
056
057                catch2except = null;
058        }
059
060        public boolean add_TryStmts(Collection<AugmentedStmt> c) {
061                for (AugmentedStmt as : c)
062                        if (add_TryStmt(as) == false)
063                                return false;
064
065                return true;
066        }
067
068        public boolean add_TryStmt(AugmentedStmt as) {
069                if ((body.contains(as)) || (tryBody.contains(as)))
070                        return false;
071
072                body.add(as);
073                tryBody.add(as);
074
075                return true;
076        }
077
078        public void refresh_CatchBody(ExceptionFinder ef) {
079                if (catchBody != null)
080                        body.removeAll(catchBody);
081
082                catchBody = ef.get_CatchBody(handlerAugmentedStmt);
083                body.addAll(catchBody);
084        }
085
086        public IterableSet<AugmentedStmt> get_Body() {
087                return body;
088        }
089
090        public IterableSet<AugmentedStmt> get_TryBody() {
091                return tryBody;
092        }
093
094        public IterableSet<AugmentedStmt> get_CatchBody() {
095                return catchBody;
096        }
097
098        public boolean remove(AugmentedStmt as) {
099                if (body.contains(as) == false)
100                        return false;
101
102                if (tryBody.contains(as))
103                        tryBody.remove(as);
104                else if ((catchBody != null) && (catchBody.contains(as))) {
105                        catchBody.remove(as);
106                        dirty = true;
107                } else
108                        return false;
109
110                body.remove(as);
111
112                return true;
113        }
114
115        public List<AugmentedStmt> get_CatchExits() {
116                if (catchBody == null)
117                        return null;
118
119                if (dirty) {
120                        exitList = new LinkedList<AugmentedStmt>();
121                        dirty = false;
122
123                        for (AugmentedStmt as : catchBody) {
124                                for (AugmentedStmt succ : as.bsuccs)
125                                        if (catchBody.contains(succ) == false) {
126                                                exitList.add(as);
127                                                break;
128                                        }
129                        }
130                }
131
132                return exitList;
133        }
134
135        public void splitOff_ExceptionNode(IterableSet<AugmentedStmt> newTryBody,
136                        AugmentedStmtGraph asg, IterableSet<ExceptionNode> enlist) {
137                IterableSet<AugmentedStmt> oldTryBody = new IterableSet<AugmentedStmt>();
138                oldTryBody.addAll(tryBody);
139
140                IterableSet<AugmentedStmt> oldBody = new IterableSet<AugmentedStmt>();
141                oldBody.addAll(body);
142
143                for (AugmentedStmt as : newTryBody) {
144                        if (remove(as) == false) {
145
146                                StringBuffer b = new StringBuffer();
147                                for (AugmentedStmt auBody : newTryBody)
148                                        b.append("\n" + auBody.toString());
149                                b.append("\n-");
150
151                                for (AugmentedStmt auBody : oldTryBody)
152                                        b.append("\n" + auBody.toString());
153                                b.append("\n-");
154
155                                for (AugmentedStmt auBody : oldBody)
156                                        b.append("\n" + auBody.toString());
157                                b.append("\n-");
158
159                                throw new RuntimeException(
160                                                "Tried to split off a new try body that isn't in the old one.\n"
161                                                                + as + "\n - " + b.toString());
162                        }
163                }
164
165                asg.clone_Body(catchBody);
166
167                AugmentedStmt oldCatchTarget = handlerAugmentedStmt, newCatchTarget = asg
168                                .get_CloneOf(handlerAugmentedStmt);
169
170                for (AugmentedStmt as : newTryBody) {
171                        as.remove_CSucc(oldCatchTarget);
172                        oldCatchTarget.remove_CPred(as);
173                }
174
175                for (AugmentedStmt as : tryBody) {
176                        as.remove_CSucc(newCatchTarget);
177                        newCatchTarget.remove_CPred(as);
178                }
179
180                for (ExceptionNode en : enlist) {
181                        if (this == en)
182                                continue;
183
184                        if (catchBody.isSupersetOf(en.get_Body())) {
185
186                                IterableSet<AugmentedStmt> clonedTryBody = new IterableSet<AugmentedStmt>();
187
188                                for (AugmentedStmt au : en.get_TryBody())
189                                        clonedTryBody.add(asg.get_CloneOf(au));
190
191                                enlist.addLast(new ExceptionNode(clonedTryBody, en.exception,
192                                                asg.get_CloneOf(en.handlerAugmentedStmt)));
193                        }
194                }
195
196                enlist.addLast(new ExceptionNode(newTryBody, exception, asg
197                                .get_CloneOf(handlerAugmentedStmt)));
198
199                for (ExceptionNode en : enlist)
200                        en.refresh_CatchBody(ExceptionFinder.v());
201
202                asg.find_Dominators();
203        }
204
205        public void add_CatchBody(ExceptionNode other) {
206                if (other.get_CatchList() == null) {
207                        add_CatchBody(other.get_CatchBody(), other.get_Exception());
208                        return;
209                }
210
211                for (IterableSet<AugmentedStmt> c : other.get_CatchList()) {
212                        add_CatchBody(c, other.get_Exception(c));
213                }
214        }
215
216        public void add_CatchBody(IterableSet<AugmentedStmt> newCatchBody, SootClass except) {
217                if (catchList == null) {
218                        catchList = new LinkedList<IterableSet<AugmentedStmt>>();
219                        catchList.addLast(catchBody);
220
221                        catch2except = new HashMap<IterableSet<AugmentedStmt>, SootClass>();
222                        catch2except.put(catchBody, exception);
223                }
224
225                body.addAll(newCatchBody);
226                catchList.addLast(newCatchBody);
227                catch2except.put(newCatchBody, except);
228        }
229
230        public List<IterableSet<AugmentedStmt>> get_CatchList() {
231                List<IterableSet<AugmentedStmt>> l = catchList;
232
233                if (l == null) {
234                        l = new LinkedList<IterableSet<AugmentedStmt>>();
235                        l.add(catchBody);
236                }
237
238                return l;
239        }
240
241        public Map<IterableSet<AugmentedStmt>, SootClass> get_ExceptionMap() {
242                Map<IterableSet<AugmentedStmt>, SootClass> m = catch2except;
243
244                if (m == null) {
245                        m = new HashMap<IterableSet<AugmentedStmt>, SootClass>();
246                        m.put(catchBody, exception);
247                }
248
249                return m;
250        }
251
252        public SootClass get_Exception() {
253                return exception;
254        }
255
256        public SootClass get_Exception(IterableSet<AugmentedStmt> catchBody) {
257                if (catch2except == null)
258                        return exception;
259
260                return catch2except.get(catchBody);
261        }
262
263        public void dump() {
264                G.v().out.println("try {");
265                for (AugmentedStmt au : get_TryBody())
266                        G.v().out.println("\t" + au);
267                G.v().out.println("}");
268
269                for (IterableSet<AugmentedStmt> catchBody : get_CatchList()) {
270                        G.v().out.println("catch " + get_ExceptionMap().get(catchBody)
271                                        + " {");
272                        
273                        for (AugmentedStmt au : catchBody)
274                                G.v().out.println("\t" + au);
275                        G.v().out.println("}");
276
277                }
278        }
279}
