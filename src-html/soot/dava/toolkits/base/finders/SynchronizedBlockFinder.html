052        private final Integer WHITE = new Integer(0);// never visited in DFS
053        private final Integer GRAY = new Integer(1);// visited but not finished
054        private final Integer BLACK = new Integer(2);// finished

062        public void find(DavaBody body, AugmentedStmtGraph asg, SETNode SET)
063                        throws RetriggerAnalysisException {
064                davaBody = body;
065                Dava.v().log("SynchronizedBlockFinder::find()");
066
067                as2ml = new HashMap();
068
069                IterableSet synchronizedBlockFacts = body.get_SynchronizedBlockFacts();
070                synchronizedBlockFacts.clear();
071
072                set_MonitorLevels(asg);
073
074                Map<AugmentedStmt, IterableSet> as2synchSet = build_SynchSets();
075
076                IterableSet usedMonitors = new IterableSet();
077
078                AugmentedStmt previousStmt = null;
079
080                Iterator asgit = asg.iterator();
081                while (asgit.hasNext()) {
082                        // going through all augmentedStmts
083
084                        AugmentedStmt as = (AugmentedStmt) asgit.next();
085                        if (as.get_Stmt() instanceof EnterMonitorStmt) {
086                                // for each monitor enter stmt found
087
088                                IterableSet synchSet = as2synchSet.get(as);
089                                if (synchSet != null) {
090                                        IterableSet synchBody = get_BodyApproximation(as, synchSet);
091                                        Value local = ((EnterMonitorStmt) as.get_Stmt()).getOp();
092                                        Value copiedLocal = null;
093
094                                        /*
095                                         * Synch bug due to copy stmts r2 = r1 enter monitor r1
096                                         * synch body exit monitor r2 we need to realize that r1 and
097                                         * r2 are the same implemented as a hack, the correct
098                                         * approach is reaching copies
099                                         */
100                                        if (previousStmt != null) {
101                                                Stmt previousS = previousStmt.get_Stmt();
102                                                if (previousS instanceof DefinitionStmt) {
103                                                        DefinitionStmt previousDef = (DefinitionStmt) previousS;
104                                                        Value rightPrevious = previousDef.getRightOp();
105                                                        if (rightPrevious.toString().compareTo(
106                                                                        local.toString()) == 0) {
107                                                                copiedLocal = previousDef.getLeftOp();
108                                                        }
109                                                }
110                                        }
111
112                                        Integer level = as2ml.get(as).get(local);
113                                        Iterator enit = body.get_ExceptionFacts().iterator();
114
115                                        // going through all exception nodes in the DavaBody
116                                        boolean done = false;
117                                        IterableSet origSynchBody = synchBody;
118                                        while (enit.hasNext()) {
119                                                ExceptionNode en = (ExceptionNode) enit.next();
120                                                synchBody = (IterableSet) origSynchBody.clone();
121                                                // see if the two bodies match exactly
122
123                                                if (verify_CatchBody(en, synchBody, local, copiedLocal)) {
124                                                        if (SET.nest(new SETSynchronizedBlockNode(en, local))) {
125                                                                done = true;
126                                                                // System.out.println("synch block created");
127                                                                Iterator ssit = synchSet.iterator();
128                                                                while (ssit.hasNext()) {
129                                                                        AugmentedStmt ssas = (AugmentedStmt) ssit
130                                                                                        .next();
131                                                                        Stmt sss = ssas.get_Stmt();
132
133                                                                        /*
134                                                                         * Jeromes Implementation: changed because
135                                                                         * of copy stmt bug
136                                                                         * 
137                                                                         * if ((sss instanceof MonitorStmt) &&
138                                                                         * (((MonitorStmt) sss).getOp() == local) &&
139                                                                         * (((Integer) ((HashMap) as2ml.get(
140                                                                         * ssas)).get( local)).equals( level)) &&
141                                                                         * (usedMonitors.contains( ssas) == false)){
142                                                                         */
143                                                                        if (sss instanceof MonitorStmt) {
144                                                                                if ((((MonitorStmt) sss).getOp() == local)
145                                                                                                && ((as2ml.get(ssas).get(local))
146                                                                                                                .equals(level))
147                                                                                                && (usedMonitors.contains(ssas) == false)) {
148
149                                                                                        usedMonitors.add(ssas);
150                                                                                } else {
151                                                                                        if ((((MonitorStmt) sss).getOp() == copiedLocal)
152                                                                                                        && (usedMonitors
153                                                                                                                        .contains(ssas) == false)) {
154                                                                                                // note we dont check levels in
155                                                                                                // this case
156                                                                                                usedMonitors.add(ssas);
157                                                                                        }
158                                                                                }
159                                                                        }
160                                                                }
161                                                                synchronizedBlockFacts.add(en);
162                                                        }
163                                                        break;
164                                                } else {
165                                                        // throw new
166                                                        // RuntimeException("Could not verify approximated Synchronized body");
167                                                }
168                                        }
169                                        if (!done)
170                                                throw new RuntimeException(
171                                                                "Could not verify approximated Synchronized body!\n"
172                                                                                + "Method:\n"
173                                                                                + body.getMethod()
174                                                                                + "Body:\n"
175                                                                                + "===============================================================\n"
176                                                                                + body.getUnits()
177                                                                                + "===============================================================\n");
178                                }// non null synch set for this enter monitor stmt
179                        }// if the augmented stmt was a enter monitor stmt
180                        previousStmt = as;
181                }// going through all augmented stmts
182
183                IterableSet<AugmentedStmt> monitorFacts = body.get_MonitorFacts();
184                monitorFacts.clear();
185
186                for (AugmentedStmt as : asg) {
187                        if ((as.get_Stmt() instanceof MonitorStmt)
188                                        && (usedMonitors.contains(as) == false)) {
189                                monitorFacts.add(as);
190                        }
191                }
192        }

200        private void find_VariableIncreasing(AugmentedStmtGraph asg,
201                        HashMap local2level_template, LinkedList<AugmentedStmt> viAugStmts,
202                        HashMap<AugmentedStmt, LinkedList<Value>> as2locals) {
203                StronglyConnectedComponentsFast scc = new StronglyConnectedComponentsFast(
204                                asg);
205                IterableSet viSeeds = new IterableSet();
206                HashMap as2color = new HashMap(), as2rml = new HashMap();
207
208                // as2rml contains each augmented stmt as key with a local2Level mapping
209                // as value
210                Iterator asgit = asg.iterator();
211                while (asgit.hasNext()) {
212                        as2rml.put(asgit.next(), local2level_template.clone());
213                }
214
215                // loop through all the strongly connected components in the graph
216                Iterator<List> sccit = scc.getTrueComponents().iterator();
217                while (sccit.hasNext()) {
218                        // componentList contains augmentedstmts belonging to a particular
219                        // scc
220                        List componentList = sccit.next();
221
222                        // component contains augmentedstmts belonging to a particular scc
223                        IterableSet component = new IterableSet();
224                        component.addAll(componentList);
225
226                        // add to as2color each augstmt belonging to the component as the
227                        // key and the color white as the value
228                        Iterator cit = component.iterator();
229                        while (cit.hasNext()) {
230                                as2color.put(cit.next(), WHITE);
231                        }
232
233                        // DFS and mark enough of the variable increasing points to get
234                        // started.
235                        AugmentedStmt seedStmt = (AugmentedStmt) component.getFirst();
236                        // seedStmt contains the first stmt of the scc
237
238                        DFS_Scc(seedStmt, component, as2rml, as2color, seedStmt, viSeeds);
239                        // viSeeds contain augmentedStmts which have unknown level since
240                        // their level was less
241                        // than that of their predecessors
242                }
243
244                // viSeeds contains all the augmentedStmts with unknown level in the
245                // method (for all scc)
246                IterableSet worklist = new IterableSet();
247                worklist.addAll(viSeeds);
248
249                // Propegate the variable increasing property.
250                while (worklist.isEmpty() == false) {
251                        AugmentedStmt as = (AugmentedStmt) worklist.getFirst();
252                        worklist.removeFirst();
253                        HashMap local2level = (HashMap) as2rml.get(as);
254
255                        // get all sucessors of the viSeed
256                        Iterator sit = as.csuccs.iterator();
257                        while (sit.hasNext()) {
258                                AugmentedStmt sas = (AugmentedStmt) sit.next();
259                                HashMap<Value, Integer> slocal2level = (HashMap<Value, Integer>) as2rml
260                                                .get(sas);
261
262                                Iterator mlsit = monitorLocalSet.iterator();
263                                while (mlsit.hasNext()) {
264                                        Value local = (Value) mlsit.next();
265
266                                        // if the level for a local is set to unknown and the level
267                                        // for the same local is not
268                                        // set to unknown in the successor set it and add it to the
269                                        // worklist
270                                        if ((local2level.get(local) == VARIABLE_INCR)
271                                                        && (slocal2level.get(local) != VARIABLE_INCR)) {
272                                                slocal2level.put(local, VARIABLE_INCR);
273
274                                                if (worklist.contains(sas) == false)
275                                                        worklist.addLast(sas);
276                                        }
277                                }
278                        }
279                }
280                /*
281                 * At the end of this the worklist is empty the unknown level of locals
282                 * of an augmentedstmt present in the viSeed has been propagated through
283                 * out the stmts reachable from that stmt
284                 */
285
286                // Summarize the variable increasing information for the
287                // set_MonitorLevels() function.
288                asgit = asg.iterator();
289                while (asgit.hasNext()) {
290                        AugmentedStmt as = (AugmentedStmt) asgit.next();
291                        HashMap local2level = (HashMap) as2rml.get(as);
292
293                        Iterator mlsit = monitorLocalSet.iterator();
294                        while (mlsit.hasNext()) {
295                                // for each local involved in monitor stmts
296                                Value local = (Value) mlsit.next();
297
298                                if (local2level.get(local) == VARIABLE_INCR) {
299
300                                        /*
301                                         * Nomair A. Naeem 08-FEB-2005 BUG IN CODE. The getLast
302                                         * method returns a NoSuchElementException if list is empty.
303                                         * Protect this by checking isEmpty
304                                         */
305                                        if (!viAugStmts.isEmpty()) {
306                                                if (viAugStmts.getLast() != as)
307                                                        viAugStmts.addLast(as);
308                                        } else {
309                                                // System.out.println("List was empty added as");
310                                                viAugStmts.addLast(as);
311                                        }
312
313                                        LinkedList<Value> locals = null;
314
315                                        if ((locals = as2locals.get(as)) == null) {
316                                                locals = new LinkedList<Value>();
317                                                as2locals.put(as, locals);
318                                        }
319
320                                        locals.addLast(local);
321                                }
322                        }
323                }
324        }

326        /*
327         * Inputs: as: the seed stmt, i.e. the first statement of a strongly
328         * connected component component: the strongly connected stmt being
329         * processed as2rml: a hashmap which contains an augmentedstmt as key and a
330         * hashmap with local to level mapping as value as2color: maps each
331         * augmentedstmt of a scc to a color (white initially) viSeeds: variable
332         * increasing levels, initially empty
333         * 
334         * Purpose: makes a traversal of the entire scc and increases the levels of
335         * locals when used in monitor enter decreases the levels of locals when
336         * leaving monitor if a successor has never been visited invoke same method
337         * on it with updated level information if a sucessor has been visited
338         * before and has a lower level than a seed the level of the sucessor is
339         * marked unknown and is added to the viSeeds list
340         */
341        private void DFS_Scc(AugmentedStmt as, IterableSet component,
342                        HashMap as2rml, HashMap as2color, AugmentedStmt seedStmt,
343                        IterableSet viSeeds) {
344                as2color.put(as, GRAY);
345
346                Stmt s = as.get_Stmt();
347                // get local to level mapping of the augmented stmt
348                HashMap<Value, Integer> local2level = (HashMap<Value, Integer>) as2rml
349                                .get(as);
350
351                if (s instanceof MonitorStmt) {
352                        Value local = ((MonitorStmt) s).getOp();
353
354                        if (s instanceof EnterMonitorStmt) {// its an enter hence increase
355                                                                                                // level for this local
356                                local2level.put(local, new Integer(local2level.get(local)
357                                                .intValue() + 1));
358                        } else {// its an exit stmt hence reduce level
359                                local2level.put(local, new Integer(local2level.get(local)
360                                                .intValue() - 1));
361                        }
362                }
363
364                // get all successors of the augmented stmt
365                Iterator sit = as.csuccs.iterator();
366                // going through sucessors
367                while (sit.hasNext()) {
368                        AugmentedStmt sas = (AugmentedStmt) sit.next();
369
370                        if (component.contains(sas) == false)
371                                continue;
372
373                        // get the local2Level hashmap for the successor
374                        HashMap<Value, Integer> slocal2level = (HashMap<Value, Integer>) as2rml
375                                        .get(sas);
376                        // get the color for the sucessor
377                        Integer scolor = (Integer) as2color.get(sas);
378
379                        if (scolor.equals(WHITE)) {
380                                // the augmented stmt hasnt been handled yet
381                                Iterator mlsit = monitorLocalSet.iterator();
382                                while (mlsit.hasNext()) {
383                                        // for all locals which are used in monitor stmts
384                                        Value local = (Value) mlsit.next();
385                                        /*
386                                         * update the sucessors hashtable with level values from the
387                                         * seed stmt The loca2Level is the hashmap of local-->level
388                                         * for the seed The sLocal2Level is the hashmap of local -->
389                                         * level for the successor of the seed
390                                         */
391                                        slocal2level.put(local, local2level.get(local));
392                                }
393
394                                // do recursive call on the sucessor
395                                DFS_Scc(sas, component, as2rml, as2color, seedStmt, viSeeds);
396                        }
397
398                        else {
399                                // if the augmented stmt has been visited before
400                                Iterator mlsit = monitorLocalSet.iterator();
401                                while (mlsit.hasNext()) {
402                                        Value local = (Value) mlsit.next();
403
404                                        if (slocal2level.get(local).intValue() < local2level.get(
405                                                        local).intValue()) {
406                                                // if the sucessors value for the level of a local is
407                                                // less than that of the level of the seed
408                                                // make the level for this local to be unknown-->
409                                                // VARIABLE_INCR
410                                                slocal2level.put(local, VARIABLE_INCR);
411
412                                                // add this to the viSeeds list
413                                                if (viSeeds.contains(sas) == false)
414                                                        viSeeds.add(sas);
415                                        }
416                                }
417                        }
418                }
419
420                // mark augmented stmt as done
421                as2color.put(as, BLACK);
422        }

424        /*
425         * Created a synch set for each enter monitor stmt. The synch set contains
426         * all sucessors of the monitor enter stmt which is dominated by the enter
427         * stmt and the level is greater or equal to that of the enter stmt
428         */
429        private Map<AugmentedStmt, IterableSet> build_SynchSets() {
430                HashMap<AugmentedStmt, IterableSet> as2synchSet = new HashMap<AugmentedStmt, IterableSet>();
431
432                Iterator mesit = monitorEnterSet.iterator();
433                monitorEnterLoop: while (mesit.hasNext()) {
434                        // going through monitor enter stmts
435                        AugmentedStmt headAs = (AugmentedStmt) mesit.next();
436                        Value local = ((EnterMonitorStmt) headAs.get_Stmt()).getOp();
437                        IterableSet synchSet = new IterableSet();
438
439                        // get the monitor level for the local uses in the enter stmt
440                        int monitorLevel = (as2ml.get(headAs).get(local)).intValue();
441                        IterableSet worklist = new IterableSet();
442                        worklist.add(headAs);
443
444                        while (worklist.isEmpty() == false) {
445                                AugmentedStmt as = (AugmentedStmt) worklist.getFirst();
446                                worklist.removeFirst();
447
448                                Stmt s = as.get_Stmt();
449                                if ((s instanceof DefinitionStmt)
450                                                && (((DefinitionStmt) s).getLeftOp() == local))
451                                        continue monitorEnterLoop;
452
453                                synchSet.add(as);
454
455                                Iterator sit = as.csuccs.iterator();
456                                while (sit.hasNext()) {
457                                        AugmentedStmt sas = (AugmentedStmt) sit.next();
458                                        // get the sucessors monitor level
459                                        int sml = (as2ml.get(sas).get(local)).intValue();
460
461                                        /*
462                                         * if the sucessor is dominated by the head stmt and the
463                                         * level is greater or equal to that of the head and is not
464                                         * waiting to be analysed and is not in the synchSet.. then
465                                         * add it to worklist
466                                         */
467                                        if (sas.get_Dominators().contains(headAs)
468                                                        && (sml >= monitorLevel)
469                                                        && (worklist.contains(sas) == false)
470                                                        && (synchSet.contains(sas) == false))
471
472                                                worklist.addLast(sas);
473                                }
474                        }
475
476                        as2synchSet.put(headAs, synchSet);
477                }
478                return as2synchSet;
479        }

481        /*
482         * MonitorLocalSet: contains all locals used in monitor enter statements
483         * MonitorEnterSet: contains all monitor enter statements as2ml : key is
484         * each augmented stmt of the augmented graph, value is a hashmap which
485         * contains a local as key and the level as value
486         */
487        private void set_MonitorLevels(AugmentedStmtGraph asg) {
488                monitorLocalSet = new IterableSet();
489                monitorEnterSet = new IterableSet();
490
491                // Identify the locals that are used in monitor statements, and all the
492                // monitor enters.
493                Iterator<AugmentedStmt> asgit = asg.iterator();
494                while (asgit.hasNext()) {
495                        AugmentedStmt as = asgit.next();
496                        Stmt s = as.get_Stmt();
497
498                        if (s instanceof MonitorStmt) {
499                                Value local = ((MonitorStmt) s).getOp();
500
501                                // if the monitorLocalSet does not contain this local add it
502                                if (monitorLocalSet.contains(local) == false)
503                                        monitorLocalSet.add(local);
504
505                                // add the monitor enter statement to the monitorEnter Set
506                                if (s instanceof EnterMonitorStmt)
507                                        monitorEnterSet.add(as);
508                        }
509                }
510
511                // Set up a base monitor lock level of 0 for all monitor locals.
512                HashMap local2level_template = new HashMap();
513                Iterator mlsit = monitorLocalSet.iterator();
514                while (mlsit.hasNext()) {
515                        // add the local as key and value 0 as value
516                        local2level_template.put(mlsit.next(), new Integer(0));
517                }
518
519                // Give each statement the base monitor lock levels.
520                asgit = asg.iterator();
521                while (asgit.hasNext()) {
522                        // the augmented stmt is key and the whole hashMap with all the
523                        // local-->value mapping is the key
524                        as2ml.put(asgit.next(),
525                                        (Map<Value, Integer>) local2level_template.clone());
526                }
527
528                LinkedList<AugmentedStmt> viAugStmts = new LinkedList<AugmentedStmt>();
529                HashMap<AugmentedStmt, LinkedList<Value>> incrAs2locals = new HashMap<AugmentedStmt, LinkedList<Value>>();
530
531                // setup the variable increasing monitor levels
532                find_VariableIncreasing(asg, local2level_template, viAugStmts,
533                                incrAs2locals);
534                /*
535                 * At this time the viAugStmts contains all augments Stmt which have
536                 * some local that has unknown level the incrAs2locals contains a map of
537                 * augmented stmt to a linkedlist which contains locals with unknown
538                 * level
539                 */
540
541                // going through all augmented stmts with some local with unknown level
542                Iterator<AugmentedStmt> viasit = viAugStmts.iterator();
543                while (viasit.hasNext()) {
544                        AugmentedStmt vias = viasit.next();
545                        Map local2level = as2ml.get(vias);
546
547                        // getting the list of locals with unknown level for this augmented
548                        // stmt
549                        Iterator lit = incrAs2locals.get(vias).iterator();
550                        while (lit.hasNext()) {
551                                // marking the level for this local as unknown
552                                local2level.put(lit.next(), VARIABLE_INCR);
553                        }
554                }
555
556                IterableSet worklist = new IterableSet();
557                worklist.addAll(monitorEnterSet);
558
559                // Flow monitor lock levels.
560                while (worklist.isEmpty() == false) {
561                        // going through all monitor enter stmts
562                        AugmentedStmt as = (AugmentedStmt) worklist.getFirst();
563                        worklist.removeFirst();
564
565                        Map<Value, Integer> cur_local2level = as2ml.get(as);
566
567                        Iterator pit = as.cpreds.iterator();
568                        while (pit.hasNext()) {
569                                // going through preds of an enter monitor stmt
570                                AugmentedStmt pas = (AugmentedStmt) pit.next();
571                                Stmt s = as.get_Stmt();
572
573                                Map pred_local2level = as2ml.get(pas);
574
575                                mlsit = monitorLocalSet.iterator();
576                                while (mlsit.hasNext()) {
577                                        Value local = (Value) mlsit.next();
578
579                                        int predLevel = ((Integer) pred_local2level.get(local))
580                                                        .intValue();
581                                        Stmt ps = pas.get_Stmt();
582
583                                        if (predLevel == UNKNOWN) // Already marked as variable
584                                                                                                // increasing.
585                                                continue;
586
587                                        if (ps instanceof ExitMonitorStmt) {
588
589                                                ExitMonitorStmt ems = (ExitMonitorStmt) ps;
590                                                /*
591                                                 * if the predecessor stmt is a monitor exit and the
592                                                 * local it exits is the same as this local and the
593                                                 * predLevel is greater than 0 decrease the pred level
594                                                 */
595                                                if ((ems.getOp() == local) && (predLevel > 0))
596                                                        predLevel--;
597                                        }
598
599                                        if (s instanceof EnterMonitorStmt) {
600                                                EnterMonitorStmt ems = (EnterMonitorStmt) s;
601                                                /*
602                                                 * if the stmt is a monitor enter (which is true for the
603                                                 * initial worklist) and the local it enters is the same
604                                                 * as this local and the predLevel is greater or equal
605                                                 * to 0 increase the pred level
606                                                 */
607                                                if ((ems.getOp() == local) && (predLevel >= 0))
608                                                        predLevel++;
609                                        }
610
611                                        int curLevel = cur_local2level.get(local).intValue();
612
613                                        /*
614                                         * if the pred level is greater than the current level
615                                         * update current level add the sucessors of the stmt to the
616                                         * worklist
617                                         */
618                                        if (predLevel > curLevel) {
619                                                cur_local2level.put(local, new Integer(predLevel));
620
621                                                Iterator sit = as.csuccs.iterator();
622                                                while (sit.hasNext()) {
623                                                        Object so = sit.next();
624
625                                                        if (worklist.contains(so) == false)
626                                                                worklist.add(so);
627                                                }
628                                        }
629                                }
630                        }
631                }
632        }

634        /*
635         * If a stmt is removed from the synchBody then all other stmts which are
636         * dominated by this stmt should also be removed from the synchBody
637         */
638        private void removeOtherDominatedStmts(IterableSet synchBody,
639                        AugmentedStmt sas) {
640                ArrayList toRemove = new ArrayList();
641
642                Iterator it = synchBody.iterator();
643                while (it.hasNext()) {
644                        AugmentedStmt as = (AugmentedStmt) it.next();
645                        IterableSet doms = as.get_Dominators();
646                        if (doms.contains(sas)) {
647                                // System.out.println("\n\nstmt:"+as+" is dominated by removed stmt"+sas);
648                                toRemove.add(as);
649                        }
650                }
651
652                it = toRemove.iterator();
653                while (it.hasNext()) {
654                        AugmentedStmt as = (AugmentedStmt) it.next();
655                        // System.out.println("Removed dominated stmt: "+as);
656                        synchBody.remove(as);
657                }
658        }

660        private boolean verify_CatchBody(ExceptionNode en, IterableSet synchBody,
661                        Value monitorVariable, Value copiedLocal) {
662                // System.out.println("starting verification");
663                {
664                        /*
665                         * synchBody is a likely superset of exception Node since the
666                         * synchBody contains a goto stmt to the stmt right after the
667                         * "to be created synch block" See if this is the case and remove
668                         * the stmt
669                         */
670
671                        IterableSet tryBodySet = en.get_TryBody();
672                        Iterator tempIt = tryBodySet.iterator();
673                        AugmentedStmt tempas = null;
674
675                        // to compute last stmt in a tryBody one should find the stmt whose
676                        // sucessor is not in the tryBody
677                        outer: while (tempIt.hasNext()) {
678                                tempas = (AugmentedStmt) tempIt.next();
679
680                                // get natural sucessors
681                                Iterator succIt = tempas.bsuccs.iterator();
682                                while (succIt.hasNext()) {
683                                        AugmentedStmt succAS = (AugmentedStmt) succIt.next();
684                                        if (!tryBodySet.contains(succAS)) {
685                                                // tempas has a sucessor which is not part of the
686                                                // trybody hence this is the last stmt
687                                                break outer;
688                                        }
689                                }
690                                // System.out.println(tempas);
691                        }
692
693                        // tempas contains the last augmentedStmt
694                        if (tempas != null) {
695
696                                // retrieving successors
697                                Iterator sit = tempas.bsuccs.iterator();
698                                while (sit.hasNext()) {
699                                        AugmentedStmt sas = (AugmentedStmt) sit.next();
700                                        // System.out.println("Removing: successor of last stmt:"+sas.get_Stmt());
701
702                                        // this is the stmt which is the successor of the try block
703                                        // which shouldnt be in the synchronized block
704                                        // remove the stmt if present in synchBody
705
706                                        synchBody.remove(sas);
707                                        // System.out.println("Here, removed: "+sas);
708                                        // should remove all other stmts in the synchBody which are
709                                        // dominated by this stmt
710                                        removeOtherDominatedStmts(synchBody, sas);
711                                }
712                        }
713                }
714
715                /*
716                 * In the abc created code when a synch body occurs within a try block
717                 * There is an error because the synchBody created has addition
718                 * statements regarding the catching of exception due to the outer try
719                 * block
720                 * 
721                 * Since these statements are not part of the synch body they should be
722                 * removed
723                 */
724                {
725                        Iterator tempIt = en.get_TryBody().iterator();
726                        while (tempIt.hasNext()) {
727                                AugmentedStmt as = (AugmentedStmt) tempIt.next();
728                                if (as.get_Stmt() instanceof ExitMonitorStmt) {
729                                        // System.out.println("All possible sucessors of as (CSUCCS):"+as.csuccs);
730                                        List csuccs = as.csuccs;
731
732                                        // remove the natural sucessors
733                                        csuccs.remove(as.bsuccs);
734                                        // now csuccs have the exception sucessors
735                                        // System.out.println("Exception sucessors of Exit (CSUCCS:"+csuccs);
736
737                                        Iterator succIt = csuccs.iterator();
738                                        while (succIt.hasNext()) {
739                                                AugmentedStmt as1 = (AugmentedStmt) succIt.next();
740                                                if (as1.get_Stmt() instanceof GotoStmt) {
741                                                        Unit target = ((soot.jimple.internal.JGotoStmt) as1
742                                                                        .get_Stmt()).getTarget();
743                                                        if (target instanceof DefinitionStmt) {
744                                                                DefinitionStmt defStmt = (DefinitionStmt) target;
745
746                                                                /*
747                                                                 * These are the caught exceptions (can be more
748                                                                 * than once if there are multiple areas of
749                                                                 * protection Keep the one which is of type
750                                                                 * Throwable
751                                                                 */
752
753                                                                Value asnFrom = defStmt.getRightOp();
754
755                                                                // if not a caught exception of type throwable
756                                                                // remove from synch
757                                                                if (!(asnFrom instanceof CaughtExceptionRef)) {
758                                                                        // System.out.println("REMOVING:"+defStmt+" since this is not a caughtexception def");
759                                                                        synchBody.remove(as1);
760                                                                        // should remove all other stmts in the
761                                                                        // synchBody which are dominated by this
762                                                                        // stmt
763                                                                        removeOtherDominatedStmts(synchBody, as1);
764                                                                } else {
765                                                                        Value leftOp = defStmt.getLeftOp();
766                                                                        // System.out.println("the left op is:"+leftOp);
767
768                                                                        /*
769                                                                         * Only keep this if it is of throwable type
770                                                                         */
771                                                                        HashSet params = new HashSet();
772                                                                        params.addAll(davaBody.get_CaughtRefs());
773
774                                                                        Iterator localIt = davaBody.getLocals()
775                                                                                        .iterator();
776                                                                        String typeName = "";
777                                                                        while (localIt.hasNext()) {
778                                                                                Local local = (Local) localIt.next();
779                                                                                if (local.toString().compareTo(
780                                                                                                leftOp.toString()) == 0) {
781                                                                                        Type t = local.getType();
782
783                                                                                        typeName = t.toString();
784                                                                                        break;
785                                                                                }
786                                                                        }
787
788                                                                        if (!(typeName.compareTo(THROWABLE) == 0)) {
789                                                                                // System.out.println("REMOVING:"+defStmt+
790                                                                                // " since the caughtException not throwable type");
791                                                                                synchBody.remove(as1);
792                                                                                // should remove all other stmts in the
793                                                                                // synchBody which are dominated by this
794                                                                                // stmt
795                                                                                removeOtherDominatedStmts(synchBody,
796                                                                                                as1);
797                                                                        } else {
798                                                                                // System.out.println("KEEPING"+defStmt);
799                                                                                // System.out.println((RefType)((CaughtExceptionRef)
800                                                                                // asnFrom).getType());
801                                                                        }
802                                                                }
803                                                        } else {
804                                                                // System.out.println("\n\nnot definition"+target);
805                                                                // System.out.println("Removed"+as1);
806                                                                synchBody.remove(as1);
807                                                                // System.out.println(as1.bsuccs.get(0));
808                                                                synchBody.remove(as1.bsuccs.get(0));
809                                                                // should remove all other stmts in the
810                                                                // synchBody which are dominated by this stmt
811                                                                removeOtherDominatedStmts(synchBody, as1);
812                                                        }
813                                                }
814                                        }
815                                }
816                        }
817                }
818
819                /*
820                 * There might be some unwanted stmts in the synch body. These are
821                 * likely to be stmts which have no predecessor in the synchbody. Remove
822                 * these stmts. NOTE: the entry of the synchbody is a special case since
823                 * its predecessor is also not going to be in the synchbody but we
824                 * SHOULD not remove the entry point
825                 */
826
827                // find the entry point of synchbody
828                /*
829                 * One way of doing this is going through the synch body and finiding a
830                 * stmt whose predecessor is an enter monitor stmt NOT present in the
831                 * synchBody
832                 */
833                AugmentedStmt synchEnter = null;
834                Iterator synchIt = synchBody.iterator();
835
836                outerLoop: while (synchIt.hasNext()) {
837                        AugmentedStmt as = (AugmentedStmt) synchIt.next();
838
839                        Iterator pit = as.cpreds.iterator();
840                        while (pit.hasNext()) {
841                                AugmentedStmt pas = (AugmentedStmt) pit.next();
842
843                                if (synchBody.contains(pas) == false) {
844                                        // the as stmt has a predecessor which is not part of the
845                                        // synchBody
846                                        Stmt pasStmt = pas.get_Stmt();
847                                        if (pasStmt instanceof EnterMonitorStmt) {
848                                                // found the entry point to the synchBody
849                                                synchEnter = as;
850                                                break outerLoop;
851                                        }
852                                }
853                        }
854                }
855
856                if (synchEnter == null) {
857                        throw new RuntimeException(
858                                        "Could not find enter stmt of the synchBody: "
859                                                        + davaBody.getMethod().getSignature());
860                }
861
862                // System.out.println("Enter stmt of synchBody is:"+synchEnter);
863
864                /*
865                 * Now that we know the exception case we can go through the synchBody
866                 * and remove all stmts whose predecessor does not belong to the
867                 * synchBody
868                 */
869
870                boolean unChanged = false;
871                while (!unChanged) {
872                        unChanged = true;
873                        List<AugmentedStmt> toRemove = new ArrayList<AugmentedStmt>();
874                        synchIt = synchBody.iterator();
875                        while (synchIt.hasNext()) {
876                                AugmentedStmt synchAs = (AugmentedStmt) synchIt.next();
877                                if (synchAs == synchEnter) {
878                                        // entrypoint is an exception so just continue
879                                        continue;
880                                }
881                                Iterator pit = synchAs.cpreds.iterator();
882                                boolean remove = true;
883                                while (pit.hasNext()) {
884                                        AugmentedStmt pAs = (AugmentedStmt) pit.next();
885                                        if (synchBody.contains(pAs)) {
886                                                // one of the preds of this as is in the synchBody so
887                                                // dont remove
888                                                remove = false;
889                                        }
890                                }// going through preds of the synchAs
891                                if (remove) {
892                                        // all preds not present in synchBody
893                                        toRemove.add(synchAs);
894                                }
895                        }// going through the synchBody
896                        if (toRemove.size() > 0) {
897                                // none of the preds of synchAs are in the synchBody hence this
898                                // stmt is unreachable
899                                synchBody.removeAll(toRemove);
900                                // System.out.println("Removing:"+toRemove+" since none of its preds are in the synchBody");
901                                unChanged = false;
902                        }
903                }
904
905                // see if the two bodies match
906                if ((en.get_Body().equals(synchBody) == false)) {
907                        // System.out.println("returning unverified since synchBody does not match");
908                        // System.out.println("\n\nEN BODY:\n"+en.get_Body());
909                        // System.out.println("\n\nSYNCH BODY:\n"+synchBody);
910                        return false;
911                }
912
913                /*
914                 * The two bodies match check if the exception thrown is of type
915                 * "throwable" and that there is only one catchlist The reason for doing
916                 * this is that synchronized blocks get converted to a try catch block
917                 * with the catch , catching the throwable exception and this is the
918                 * ONLY catch
919                 */
920                if ((en.get_Exception().getName().equals(THROWABLE) == false)
921                                || (en.get_CatchList().size() > 1)) {
922                        // System.out.println("returning unverified here");
923                        return false;
924                }
925
926                // at this point we know that the two bodies match and that the en
927                // exception is a throwable exception
928
929                /*
930                 * find the entry point of the catch boody The entry point of the catch
931                 * body is the first stmt in the catchbody This can be found by finiding
932                 * the stmt whose predecessor is not part of the catch body
933                 */
934                IterableSet catchBody = en.get_CatchBody();
935                AugmentedStmt entryPoint = null;
936
937                Iterator it = catchBody.iterator();
938                catchBodyLoop: while (it.hasNext()) {
939                        AugmentedStmt as = (AugmentedStmt) it.next();
940
941                        Iterator pit = as.cpreds.iterator();
942                        while (pit.hasNext()) {
943                                AugmentedStmt pas = (AugmentedStmt) pit.next();
944
945                                if (catchBody.contains(pas) == false) {
946                                        entryPoint = as;
947                                        break catchBodyLoop;
948                                }
949                        }
950                }
951
952                // Horror upon horrors, what follows is a hard coded state machine.
953
954                /*
955                 * To verify the catchbody one has to go through the catchbody ONE stmt
956                 * at a time. This is NOT a good approach as it is using PATTERN
957                 * MATCHING
958                 * 
959                 * FIRST STEP Need to verify that the entrypoint stmt and any goto stmts
960                 * following it all point to the entrypoint
961                 */
962
963                Unit entryPointTarget = null;
964                if (entryPoint.get_Stmt() instanceof GotoStmt) {
965                        // System.out.println("Entry point is a goto stmt getting the target");
966                        entryPointTarget = ((soot.jimple.internal.JGotoStmt) entryPoint
967                                        .get_Stmt()).getTarget();
968                }
969
970                AugmentedStmt as = entryPoint;
971                if (as.bsuccs.size() != 1) {
972                        // System.out.println("here1");
973                        return false;
974                }
975
976                while (as.get_Stmt() instanceof GotoStmt) {
977                        as = (AugmentedStmt) as.bsuccs.get(0);
978                        // if ((as.bsuccs.size() != 1) || ((as != entryPoint) &&
979                        // (as.cpreds.size() != 1))) {
980                        // return false;
981                        // }
982                        if (as.bsuccs.size() != 1) {
983                                // System.out.println("here2a");
984                                return false;
985                        }
986                        if (entryPointTarget != null) {
987                                if ((as.get_Stmt() != entryPointTarget)
988                                                && (as.cpreds.size() != 1)) {
989                                        if (as.cpreds.size() != 1) {
990                                                // System.out.println("here2b");
991                                                return false;
992                                        }
993                                }
994                        } else {
995                                if ((as != entryPoint) && (as.cpreds.size() != 1)) {
996                                        // System.out.println("here2c");
997                                        return false;
998                                }
999                        }
1000                }
1001
1002                // so now we are not at a goto stmt for sure
1003                // according to the creation pattern of the try catch block we should be
1004                // at the definition stmt
1005                // e.g. <var> := @caughtexception
1006
1007                Stmt s = as.get_Stmt();
1008
1009                if (!(s instanceof DefinitionStmt)) {
1010                        // System.out.println("here3");
1011                        return false;
1012                }
1013
1014                DefinitionStmt ds = (DefinitionStmt) s;
1015                Value asnFrom = ds.getRightOp();
1016
1017                // if not a caught exception of type throwable we have a problem
1018                if (!((asnFrom instanceof CaughtExceptionRef) && (((RefType) ((CaughtExceptionRef) asnFrom)
1019                                .getType()).getSootClass().getName().equals(THROWABLE)))) {
1020                        // System.out.println("here4");
1021                        return false;
1022                }
1023
1024                Value throwlocal = ds.getLeftOp();
1025                // System.out.println("Throw local is:"+throwlocal);
1026
1027                /*
1028                 * The escuccs contains all the EXCEPTION SUCESSORS of the
1029                 * caughtexception stmt
1030                 */
1031                IterableSet esuccs = new IterableSet();
1032                esuccs.addAll(as.csuccs);
1033                esuccs.removeAll(as.bsuccs);
1034
1035                // sucessor of definition stmt
1036                as = (AugmentedStmt) as.bsuccs.get(0);
1037                s = as.get_Stmt();
1038
1039                // this COULD be a copy stmt in which case update the throwlocal
1040                while (s instanceof DefinitionStmt
1041                                && (((DefinitionStmt) s).getRightOp().toString()
1042                                                .compareTo(throwlocal.toString()) == 0)) {
1043
1044                        // System.out.println("copy stmt using throwLocal found");
1045                        // System.out.println("Right op is :"+((DefinitionStmt)s).getRightOp());
1046                        // return false;
1047                        throwlocal = ((DefinitionStmt) s).getLeftOp();
1048
1049                        // the sucessor of this stmt MIGHT be the exitmonitor stmt
1050                        as = (AugmentedStmt) as.bsuccs.get(0);
1051                        s = as.get_Stmt();
1052                }
1053
1054                if (as.bsuccs.size() != 1) {
1055                        // should have one true sucessor
1056                        // System.out.println("here5a");
1057                        return false;
1058                }
1059
1060                if (as.cpreds.size() != 1) {
1061                        // should have one true predecessor
1062                        // System.out.println("here5b");
1063                        return false;
1064                }
1065
1066                // need to check whether this stmt is protected by the same exception
1067                // block as the whole catchbody
1068                checkProtectionArea(as, ds);
1069
1070                s = as.get_Stmt();
1071                if (!(s instanceof ExitMonitorStmt)) {
1072                        // System.out.println("Not an exit monitor stmt"+s);
1073                        return false;
1074                }
1075
1076                if ((((ExitMonitorStmt) s).getOp() != monitorVariable)) {
1077
1078                        if ((((ExitMonitorStmt) s).getOp() != copiedLocal)) {
1079                                // System.out.println("exit monitor variable does not match enter monitor variable");
1080                                return false;
1081                        }
1082                }
1083
1084                // next stmt should be a throw stmt
1085                as = (AugmentedStmt) as.bsuccs.get(0);
1086                if ((as.bsuccs.size() != 0) || (as.cpreds.size() != 1)
1087                                || (verify_ESuccs(as, esuccs) == false)) {
1088                        // System.out.println("here7");
1089                        return false;
1090                }
1091
1092                s = as.get_Stmt();
1093
1094                if (!((s instanceof ThrowStmt) && (((ThrowStmt) s).getOp() == throwlocal))) {
1095                        // System.out.println("here8"+s+" Throw local is:"+throwlocal);
1096                        return false;
1097                }
1098
1099                return true;
1100        }

1102        /*

1104         * sucessors of as should directly or indirectly point to s

1106        private boolean checkProtectionArea(AugmentedStmt as, DefinitionStmt s) {
1107                IterableSet esuccs = new IterableSet();
1108                esuccs.addAll(as.csuccs);
1109                esuccs.removeAll(as.bsuccs);
1110
1111                // esuccs contains the exception sucessors of as
1112                // System.out.println("ESUCCS are:"+esuccs);
1113
1114                Iterator it = esuccs.iterator();
1115                while (it.hasNext()) {
1116                        // going through each exception sucessor
1117                        AugmentedStmt tempas = (AugmentedStmt) it.next();
1118                        Stmt temps = tempas.get_Stmt();
1119                        if (temps instanceof GotoStmt) {
1120                                Unit target = ((GotoStmt) temps).getTarget();
1121                                if (target != s) {
1122                                        // System.out.println("DID NOT Match indirectly");
1123                                        return false;
1124                                } else {
1125                                        // System.out.println("Matched indirectly");
1126                                }
1127                        } else {
1128                                if (temps != s) {
1129                                        // System.out.println("DID NOT Match directly");
1130                                        return false;
1131                                } else {
1132                                        // System.out.println("Matched directly");
1133                                }
1134                        }
1135
1136                }
1137                return true;
1138        }

1151        /*
1152         * Input: head: this is the monitor enter stmt synchSet: this contains all
1153         * sucessors of head which have level greater or equal to the head
1154         */
1155        private IterableSet get_BodyApproximation(AugmentedStmt head,
1156                        IterableSet synchSet) {
1157                IterableSet body = (IterableSet) synchSet.clone();
1158                Value local = ((EnterMonitorStmt) head.get_Stmt()).getOp();
1159                Integer level = as2ml.get(head).get(local);
1160
1161                // System.out.println("BODY"+body);
1162                body.remove(head);
1163
1164                Iterator bit = body.snapshotIterator();
1165                while (bit.hasNext()) {
1166                        AugmentedStmt as = (AugmentedStmt) bit.next();
1167                        Stmt s = as.get_Stmt();
1168
1169                        if ((s instanceof ExitMonitorStmt)
1170                                        && (((ExitMonitorStmt) s).getOp() == local)
1171                                        && ((as2ml.get(as).get(local)).equals(level))) {
1172
1173                                Iterator sit = as.csuccs.iterator();
1174                                while (sit.hasNext()) {
1175                                        AugmentedStmt sas = (AugmentedStmt) sit.next();
1176
1177                                        // if not dominated by head continue with next stmt in body
1178                                        if (sas.get_Dominators().contains(head) == false)
1179                                                continue;
1180
1181                                        Stmt ss = sas.get_Stmt();
1182
1183                                        if (((ss instanceof GotoStmt) || (ss instanceof ThrowStmt))
1184                                                        && (body.contains(sas) == false)) {
1185                                                // if (ss instanceof ThrowStmt && (body.contains( sas)
1186                                                // == false)){
1187                                                // System.out.println("adding"+sas);
1188                                                body.add(sas);
1189                                        }
1190                                }
1191                        }
1192                }
1193
1194                return body;
1195        }
