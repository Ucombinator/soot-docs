050        public Renamer(heuristicSet info, ASTMethodNode node) {
051                heuristics = info;
052                locals = null;
053                methodNode = node;
054                
055                changedOrNot = new HashMap<Local, Boolean>();
056                Iterator<Local> localIt = info.getLocalsIterator();
057                while(localIt.hasNext())
058                        changedOrNot.put(localIt.next(),new Boolean(false));
059                
060                
061                forLoopNames = new ArrayList<String>();
062                forLoopNames.add("i");
063                forLoopNames.add("j");
064                forLoopNames.add("k");
065                forLoopNames.add("l");
066        }

068        /*
069         * Add any naming heuristic as a separate method and invoke the method from
070         * this method.
071         * 
072         * HOWEVER, NOTE that the order of naming really really matters
073         */
074        public void rename() {
075                debug("rename","Renaming started");
076
077                // String args
078                mainMethodArgument();
079                
080                //for(i=0;i<bla;i++)
081                forLoopIndexing();
082                
083                //exceptions are named using first letter of each capital char in the class name
084                exceptionNaming();
085                
086                //arrays get <type>Array
087                arraysGetTypeArray();
088                
089                //if a local is assigned a field that name can be used since fields are conserved
090                assignedFromAField();
091                
092                //check if a local is assigned the result of a new invocation
093                newClassName();
094
095                //check if a local is assigned after casting
096                castedObject(); 
097                
098                //if nothing else give a reference the name of the class
099                objectsGetClassName();
100                
101                //atleast remove the ugly dollar signs
102                removeDollarSigns();
103        }

107        /*
108         * if there is an array int[] x. then if no other heuristic matches give it the name intArray 
109         */
110        private void arraysGetTypeArray(){
111                Iterator<Local> it = heuristics.getLocalsIterator();
112                while (it.hasNext()) {
113                        Local tempLocal = it.next();
114                        if(alreadyChanged(tempLocal)){
115                                continue;
116                        }
117                        
118                        debug("arraysGetTypeArray","checking "+tempLocal);
119                        
120                        Type type = tempLocal.getType();
121                        if(type instanceof ArrayType){
122                                debug("arraysGetTypeArray","Local:"+tempLocal+" is an Array Type: "+type.toString());
123                                String tempClassName = type.toString();
124                                //remember that a toString of an array gives you the square brackets
125                                if(tempClassName.indexOf('[')>=0)
126                                        tempClassName = tempClassName.substring(0,tempClassName.indexOf('['));
127                                
128                                //debug("arraysGetTypeArray","type of object is"+tempClassName);
129                                if(tempClassName.indexOf('.')!= -1){
130                                        //contains a dot have to remove that
131                                        tempClassName=tempClassName.substring(tempClassName.lastIndexOf('.')+1);
132                                }
133                                
134                                
135                                String newName = tempClassName.toLowerCase();
136                                newName = newName+"Array";
137                                int count=0;
138                                newName += count;
139                                count++;
140                                
141                                while(!isUniqueName(newName)){
142                                        newName = newName.substring(0,newName.length()-1)+count;
143                                        count++;                                                
144                                }
145                                setName(tempLocal,newName);
146
147                        }
148                }
149                
150                
151        }

155        /*
156         * The method assigns any local whose name hasnt been changed yet to 
157         * the name of the class type it belongs to
158         */
159        private void objectsGetClassName(){
160                Iterator<Local> it = heuristics.getLocalsIterator();
161                while (it.hasNext()) {
162                        Local tempLocal = it.next();
163                        if(alreadyChanged(tempLocal)){
164                                continue;
165                        }
166                        
167                        debug("objectsGetClassName","checking "+tempLocal);
168                        
169                        Type type = tempLocal.getType();
170                        if(type instanceof ArrayType){
171                                //should have been handled by arraysGetTypeArray heuristic
172                                continue;
173                        }
174                        
175                        if(type instanceof RefLikeType){
176                                debug("objectsGetClassName","Local:"+tempLocal+" Type: "+type.toString());
177                                //debug("objectsGetClassName","getting array type"+type.getArrayType());
178                                String tempClassName = type.toString();
179                                //debug("objectsGetClassName","type of object is"+tempClassName);
180                                if(tempClassName.indexOf('.')!= -1){
181                                        //contains a dot have to remove that
182                                        tempClassName=tempClassName.substring(tempClassName.lastIndexOf('.')+1);
183                                }
184                                
185                                
186                                String newName = tempClassName.toLowerCase();
187                                int count=0;
188                                newName += count;
189                                count++;
190                                
191                                while(!isUniqueName(newName)){
192                                        newName = newName.substring(0,newName.length()-1)+count;
193                                        count++;                                                
194                                }
195                                setName(tempLocal,newName);
196
197                        }
198                }

204        /*
205         * If a local is assigned the resullt of a cast expression   temp = (List) object;
206         * then u can use list as the name...however only if its always casted to the same object
207         */
208        private void castedObject(){
209                debug("castedObject","");
210                
211                Iterator<Local> it = heuristics.getLocalsIterator();
212                while (it.hasNext()) {
213                        Local tempLocal = it.next();
214                        if(!alreadyChanged(tempLocal)){
215                                debug("castedObject","checking "+tempLocal);
216                                List<String> classes = heuristics.getCastStrings(tempLocal);
217                                
218                                Iterator<String> itClass = classes.iterator();
219                                String classNameToUse = null;
220                                while(itClass.hasNext()){
221                                        String tempClassName = itClass.next();
222                                        if(tempClassName.indexOf('.')!= -1){
223                                                //contains a dot have to remove that
224                                                tempClassName=tempClassName.substring(tempClassName.lastIndexOf('.')+1);
225                                        }
226                                        if(classNameToUse == null)
227                                                classNameToUse = tempClassName;                 
228                                        else if(!classNameToUse.equals(tempClassName)){
229                                                //different new assignment
230                                                //cant use these classNames
231                                                classNameToUse=null;
232                                                break;
233                                        }
234                                }//going through class names stored
235                                if(classNameToUse!=null){
236                                        debug("castedObject","found a classNametoUse through cast expr");
237                                        /*
238                                         * We should use this classNAme to assign to the local name
239                                         * We are guaranteed that all cast expressions use this type
240                                         */
241                                        String newName = classNameToUse.toLowerCase();
242                                        int count=0;
243                                        newName += count;
244                                        count++;
245                                        
246                                        while(!isUniqueName(newName)){
247                                                newName = newName.substring(0,newName.length()-1)+count;
248                                                count++;                                                
249                                        }
250                                        setName(tempLocal,newName);
251                                }
252                        }//not already changed
253                }//going through locals
254        }

256        /*
257         * See if any local was initialized using the new operator
258         * That name might give us a hint to a name to use for the local
259         */
260        private void newClassName(){
261                
262                debug("newClassName","");
263                //check if CLASSNAME is set
264                //that would mean there was new className invocation
265                Iterator<Local> it = heuristics.getLocalsIterator();
266                while (it.hasNext()) {
267                        Local tempLocal = it.next();
268                        if(!alreadyChanged(tempLocal)){
269                                debug("newClassName","checking "+tempLocal);
270                                List<String> classes = heuristics.getObjectClassName(tempLocal);
271                                Iterator<String> itClass = classes.iterator();
272                                String classNameToUse = null;
273                                while(itClass.hasNext()){
274                                        String tempClassName = itClass.next();
275                                        if(tempClassName.indexOf('.')!= -1){
276                                                //contains a dot have to remove that
277                                                tempClassName=tempClassName.substring(tempClassName.lastIndexOf('.')+1);
278                                        }
279                                        if(classNameToUse == null)
280                                                classNameToUse = tempClassName;                 
281                                        else if(!classNameToUse.equals(tempClassName)){
282                                                //different new assignment
283                                                //cant use these classNames
284                                                classNameToUse=null;
285                                                break;
286                                        }
287                                }//going through class names stored
288                                if(classNameToUse!=null){
289                                        debug("newClassName","found a classNametoUse");
290                                        /*
291                                         * We should use this classNAme to assign to the local name
292                                         * We are guaranteed that all new invocations use this class name
293                                         */
294                                        String newName = classNameToUse.toLowerCase();
295                                        int count=0;
296                                        newName += count;
297                                        count++;
298                                        
299                                        while(!isUniqueName(newName)){
300                                                newName = newName.substring(0,newName.length()-1)+count;
301                                                count++;                                                
302                                        }

304                                }

306                }//going through locals

310        /*
311         * If a local is assigned from a field (static or non staitc) we can use that name
312         * to assign a some what better name for the local
313         * 
314         * If multiple fields are assigned then it might be a better idea to not do
315         * anything since that will only confuse the user
316         * 
317         */
318        private void assignedFromAField(){
319                Iterator<Local> it = heuristics.getLocalsIterator();
320                while (it.hasNext()) {
321                        Local tempLocal = it.next();
322                        if(!alreadyChanged(tempLocal)){
323                                debug("assignedFromField","checking "+tempLocal);
324                                List<String> fieldNames = heuristics.getFieldName(tempLocal);
325                                if(fieldNames.size()>1){
326                                        //more than one fields were assigned to this var
327                                        continue;
328                                }
329                                else if(fieldNames.size()==1){
330                                        //only one field was used
331                                        String fieldName = fieldNames.get(0);
332                                        
333                                        //okkay to use the name of the field if its not in scope
334                                        //eg it was some other classes field
335                                        int count=0;
336                                        while(!isUniqueName(fieldName)){
337                                                if(count==0)
338                                                        fieldName = fieldName+count;
339                                                else
340                                                        fieldName = fieldName.substring(0,fieldName.length()-1)+count;
341                                                count++;
342                                        }
343                                        
344                                        setName(tempLocal,fieldName);
345                                }//only one field assigned to this local
346                        }//not changed
347                }//going through locals
348        }

356        /*
357         * If we cant come up with any better name atleast we should remove the $ signs
358         */
359        private void removeDollarSigns(){
360                Iterator<Local> it = heuristics.getLocalsIterator();
361                while (it.hasNext()) {
362                        Local tempLocal = it.next();
363                        String currentName = tempLocal.getName();
364                        int dollarIndex = currentName.indexOf('$'); 
365                        if(dollarIndex == 0){
366                                //meaning there is a $ sign in the first location
367                                String newName = currentName.substring(1,currentName.length());
368                                
369
370                                if(isUniqueName(newName)){
371                                        setName(tempLocal,newName);
372//                                      System.out.println("Changed "+currentName+" to "+newName);
373                                        //tempLocal.setName(newName);
374                                }
375                        }
376                }
377        }

381        /*
382         * 
383         */
384        private void exceptionNaming(){
385                Iterator<Local> it = heuristics.getLocalsIterator();
386                while (it.hasNext()) {
387                        Local tempLocal = it.next();
388                        Type localType = tempLocal.getType();
389                        String typeString = localType.toString();
390                        if(typeString.indexOf("Exception")>=0){
391                                //the string xception occurs in this type
392                                debug("exceptionNaming","Type is an exception"+ tempLocal);
393                                
394                                //make a new name of all caps characters in typeString
395                                String newName = "";
396                                for(int i=0;i<typeString.length();i++){
397                                        char character = typeString.charAt(i);
398                                        if(Character.isUpperCase(character)){
399                                                newName += Character.toLowerCase(character);
400                                        }
401                                }
402                                int count =0;
403                                if(!isUniqueName(newName)){
404                                        count++;
405                                        while(!isUniqueName(newName+count)){
406                                                count++;
407                                        }                               
408                                }
409                                if(count !=0)
410                                        newName = newName + count;
411                                
412                                setName(tempLocal,newName);
413                        }
414                }
415        }

422        /*
423         * Probably one of the most common programming idioms
424         * for loop indexes are often i j k l
425         */
426        private void forLoopIndexing(){
427                Iterator<Local> it = heuristics.getLocalsIterator();
428                while (it.hasNext()) {
429                        Local tempLocal = it.next();
430                        debug("foeLoopIndexing","Checking local"+tempLocal.getName());
431                        if (heuristics.getHeuristic(tempLocal,
432                                        infoGatheringAnalysis.FORLOOPUPDATE)) {
433                                // this local variable is the main argument
434                                // will like to set it to args if no one has an objection
435                                int count = -1;
436                                
437                                String newName; 
438
439                                do{
440                                        count++;
441                                        if(count>=forLoopNames.size()){
442                                                newName=null;
443                                                break;
444                                        }
445                                        newName = (String)forLoopNames.get(count);                                      
446                                }while (!isUniqueName(newName));
447
448                                if(newName!=null){
449                                        setName(tempLocal,newName);
450                                }
451                        }
452                }
453        }

457        /*
458         * A simple heuristic which sets the mainMethodArgument's name to args
459         */
460        private void mainMethodArgument() {
461                Iterator<Local> it = heuristics.getLocalsIterator();
462                while (it.hasNext()) {
463                        Local tempLocal = it.next();
464                        if (heuristics.getHeuristic(tempLocal,
465                                        infoGatheringAnalysis.MAINARG)) {
466
467                                // this local variable is the main argument
468                                // will like to set it to args if no one has an objection
469                                String newName = "args";
470                                int count = 0;
471                                while (!isUniqueName(newName)) {
472                                        if(count==0)
473                                                newName = newName+count;
474                                        else
475                                                newName = newName.substring(0,newName.length()-1)+count;
476
477                                        count++;
478                                }
479                                setName(tempLocal,newName);
480                                //there cant be a same local with this heuristic set so just return
481                                return;
482                        }
483                }
484
485        }

491        /*
492         * In order to make sure that some previous heuristic which is usually a STRONGER
493         * heuristic has not already changed the name we use this method which checks for
494         * past name changes and only changes the name if the name hasnt been changed previously
495         */
496        private void setName(Local var, String newName){
497                        
498                Object truthValue = changedOrNot.get(var);
499                
500                //if it wasnt in there add it
501                if(truthValue == null)
502                        changedOrNot.put(var,new Boolean(false));
503                else{
504                        if(((Boolean)truthValue).booleanValue()){
505                                //already changed just return
506                                debug("setName","Var: "+var + " had already been renamed");
507                                return;
508                        }
509                }
510                //will only get here if the var had not been changed 
511                
512                debug("setName","Changed "+var.getName()+" to "+newName);
513                var.setName(newName);
514                changedOrNot.put(var,new Boolean(true));
515        }

527        /*
528         * Check if a local has already been changed
529         * @param local to check
530         * @return true if already changed otherwise false
531         */
532        private boolean alreadyChanged(Local var){
533                Object truthValue = changedOrNot.get(var);
534                
535                //if it wasnt in there add it
536                if(truthValue == null){
537                        changedOrNot.put(var,new Boolean(false));
538                        return false;
539                }
540                else{
541                        if(((Boolean)truthValue).booleanValue()){
542                                //already changed just return
543                                debug("alreadyChanged","Var: "+var + " had already been renamed");
544                                return true;
545                        }
546                        else
547                                return false;
548                }
549        }

553        /*
554         * Should return true if the name is unique
555         */
556        private boolean isUniqueName(String name) {
557                Iterator it = getScopedLocals();
558                // check that none of the locals uses this name
559                while (it.hasNext()) {
560                        Local tempLocal = (Local) it.next();
561                        if (tempLocal.getName().equals(name)){
562                                debug("isUniqueName","New Name "+ name+ " is not unique (matches some local)..changing");       
563                                return false;
564                        }
565                        else
566                                debug("isUniqueName","New Name "+ name+ " is different from local "+ tempLocal.getName());
567                }
568
569                it = getScopedFields();
570                // check that none of the fields uses this name
571                while (it.hasNext()) {
572                        SootField tempField = (SootField) it.next();
573                        if (tempField.getName().equals(name)){
574                                debug("isUniqueName","New Name "+ name+ " is not unique (matches field)..changing");
575                                return false;
576                        }
577                        else
578                                debug("isUniqueName","New Name "+ name+ " is different from field "+ tempField.getName());
579                }
580                return true;
581        }

583        /*
584         * Method is responsible to find all names with which there could be a
585         * potential clash The variables are: all the fields of this class and all
586         * the locals defined in this method
587         */
588        private Iterator getScopedFields() {
589                // get the fields for this class and store them
590                SootClass sootClass = methodNode.getDavaBody().getMethod()
591                .getDeclaringClass();
592                fields = sootClass.getFields();
593
594                return fields.iterator();
595        }

597        /*
598         * Method is responsible to find all variable names with which there could
599         * be a potential clash The variables are: all the fields of this class and
600         * all the locals defined in this method
601         */
602        private Iterator getScopedLocals() {
603                Iterator<Local> it = heuristics.getLocalsIterator();
604                
605                locals = new ArrayList();
606                while(it.hasNext())
607                        locals.add(it.next());
608
609                return locals.iterator();
610
611        }
