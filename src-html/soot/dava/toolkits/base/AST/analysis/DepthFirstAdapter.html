252        public void caseASTForLoopNode(ASTForLoopNode node) {
253                inASTForLoopNode(node);
254
255                /*
256                 Apply on init
257                 */
258                for (AugmentedStmt as : node.getInit()) {
259                        Stmt s = as.get_Stmt();
260                        if (s instanceof DefinitionStmt)
261                                caseDefinitionStmt((DefinitionStmt) s);
262                        else if (s instanceof ReturnStmt)
263                                caseReturnStmt((ReturnStmt) s);
264                        else if (s instanceof InvokeStmt)
265                                caseInvokeStmt((InvokeStmt) s);
266                        else if (s instanceof ThrowStmt)
267                                caseThrowStmt((ThrowStmt) s);
268                        else
269                                caseStmt(s);
270                }
271
272                /*
273                 apply on the ASTCondition
274                 */
275                ASTCondition condition = node.get_Condition();
276                condition.apply(this);
277
278                /*
279                 Apply on update
280                 */
281                for (AugmentedStmt as : node.getUpdate()) {
282                        Stmt s = as.get_Stmt();
283
284                        if (s instanceof DefinitionStmt)
285                                caseDefinitionStmt((DefinitionStmt) s);
286                        else if (s instanceof ReturnStmt)
287                                caseReturnStmt((ReturnStmt) s);
288                        else if (s instanceof InvokeStmt)
289                                caseInvokeStmt((InvokeStmt) s);
290                        else if (s instanceof ThrowStmt)
291                                caseThrowStmt((ThrowStmt) s);
292                        else
293                                caseStmt(s);
294                }
295
296                /*
297                 Apply on the for body
298                 */
299                normalRetrieving(node);
300                outASTForLoopNode(node);
301        }

339        public void caseASTTryNode(ASTTryNode node) {
340                inASTTryNode(node);
341
342                //get try body 
343                List<Object> tryBody = node.get_TryBody();
344                Iterator<Object> it = tryBody.iterator();
345
346                //go over the ASTNodes in this tryBody and apply
347                while (it.hasNext())
348                        ((ASTNode) it.next()).apply(this);
349
350                Map<Object, Object> exceptionMap = node.get_ExceptionMap();
351                Map<Object, Object> paramMap = node.get_ParamMap();
352                //get catch list and apply on the following
353                // a, type of exception caught
354                // b, local of exception
355                // c, catchBody
356                List<Object> catchList = node.get_CatchList();
357                Iterator itBody = null;
358                it = catchList.iterator();
359                while (it.hasNext()) {
360                        ASTTryNode.container catchBody = (ASTTryNode.container) it.next();
361
362                        SootClass sootClass = ((SootClass) exceptionMap.get(catchBody));
363                        Type type = sootClass.getType();
364
365                        //apply on type of exception
366                        caseType(type);
367
368                        //apply on local of exception
369                        Local local = (Local) paramMap.get(catchBody);
370                        /*
371                         * March 18th, 2006, Since these are always locals we dont have access to ValueBox
372                         */
373                        decideCaseExprOrRef(local);
374
375                        //apply on catchBody
376                        List body = (List) catchBody.o;
377                        itBody = body.iterator();
378                        while (itBody.hasNext()) {
379                                ((ASTNode) itBody.next()).apply(this);
380                        }
381                }
382                outASTTryNode(node);
383        }

476        public void normalRetrieving(ASTNode node) {
477                //from the Node get the subBodes
478                Iterator<Object> sbit = node.get_SubBodies().iterator();
479                while (sbit.hasNext()) {
480                        Object subBody = sbit.next();
481                        Iterator it = ((List) subBody).iterator();
482
483                        //go over the ASTNodes in this subBody and apply
484                        while (it.hasNext()) {
485                                ASTNode temp = (ASTNode) it.next();
486
487                                temp.apply(this);
488                        }
489                }//end of going over subBodies
490        }

502        public void caseASTStatementSequenceNode(ASTStatementSequenceNode node) {
503                inASTStatementSequenceNode(node);
504                for (AugmentedStmt as : node.getStatements()) {
505                        Stmt s = as.get_Stmt();
506                        /*
507                         Do a case by case check of possible statements and invoke
508                         the case methods from within this method.
509                         
510                         cant use apply since the Statements are defined in some other
511                         package and dont want to change code all over the place
512                         */
513
514                        if (s instanceof DefinitionStmt)
515                                caseDefinitionStmt((DefinitionStmt) s);
516                        else if (s instanceof ReturnStmt)
517                                caseReturnStmt((ReturnStmt) s);
518                        else if (s instanceof InvokeStmt)
519                                caseInvokeStmt((InvokeStmt) s);
520                        else if (s instanceof ThrowStmt)
521                                caseThrowStmt((ThrowStmt) s);
522                        else if (s instanceof DVariableDeclarationStmt)
523                                caseDVariableDeclarationStmt((DVariableDeclarationStmt) s);
524                        else
525                                caseStmt(s);
526
527                }//end of while going through the statement sequence
528                outASTStatementSequenceNode(node);
529        }

768        public void decideCaseExpr(Expr e) {
769                if (e instanceof BinopExpr)
770                        caseBinopExpr((BinopExpr) e);
771                else if (e instanceof UnopExpr)
772                        caseUnopExpr((UnopExpr) e);
773                else if (e instanceof NewArrayExpr)
774                        caseNewArrayExpr((NewArrayExpr) e);
775                else if (e instanceof NewMultiArrayExpr)
776                        caseNewMultiArrayExpr((NewMultiArrayExpr) e);
777                else if (e instanceof InstanceOfExpr)
778                        caseInstanceOfExpr((InstanceOfExpr) e);
779                else if (e instanceof InvokeExpr)
780                        caseInvokeExpr((InvokeExpr) e);
781                else if (e instanceof CastExpr)
782                        caseCastExpr((CastExpr) e);
783        }
