049        public void inASTStatementSequenceNode(ASTStatementSequenceNode node) {
050                debug("inASTStatementSequenceNode");
051                boolean success = false;
052                ArrayList<AugmentedStmt> toRemove = new ArrayList<AugmentedStmt>();
053                for (AugmentedStmt as : node.getStatements()) {
054                        success = false;
055                        Stmt s = as.get_Stmt();
056                        if (!(s instanceof DefinitionStmt))
057                                continue;
058
059                        DefinitionStmt ds = (DefinitionStmt) s;
060                        ValueBox right = ds.getRightOpBox();
061                        Value rightValue = right.getValue();
062
063                        if (!(rightValue instanceof NewArrayExpr))
064                                continue;
065
066                        debug("Found a new ArrayExpr" + rightValue);
067                        debug("Type of array is:" + rightValue.getType());
068
069                        // get type out
070                        Type arrayType = rightValue.getType();
071
072                        // get size....need to know this statically for sure!!!
073                        Value size = ((NewArrayExpr) rightValue).getSize();
074
075                        if (!(size instanceof IntConstant))
076                                continue;
077
078                        if (((IntConstant) size).value == 0) {
079                                debug("Size of array is 0 dont do anything");
080                                continue;
081                        }
082
083                        if (DEBUG)
084                                System.out.println("Size of array is: "
085                                                + ((IntConstant) size).value);
086
087                        Iterator<AugmentedStmt> tempIt = node.getStatements().iterator();
088                        // get to the array creation stmt
089                        while (tempIt.hasNext()) {
090                                AugmentedStmt tempAs = tempIt.next();
091                                Stmt tempS = tempAs.get_Stmt();
092                                if (tempS.equals(s))
093                                        break;
094                        }
095                        // have the size have the type, tempIt is poised at the current def
096                        // stmt
097                        ValueBox[] array = new ValueBox[((IntConstant) size).value];
098                        success = true;
099                        for (int i = 0; i < ((IntConstant) size).value; i++) {
100                                // check these many next stmts they better all be array
101                                // initializations
102
103                                if (!tempIt.hasNext()) {
104                                        // since its end of the stmt seq node just return
105                                        if (DEBUG)
106                                                System.out.println("returning");
107                                        return;
108                                }
109
110                                AugmentedStmt aug = tempIt.next();
111                                Stmt augS = aug.get_Stmt();
112                                if (!isInSequenceAssignment(augS, ds.getLeftOp(), i)) {
113                                        // cant create shortcut since we dont have all necessary
114                                        // initializations
115                                        if (DEBUG)
116                                                System.out
117                                                                .println("Out of order assignment aborting attempt");
118
119                                        success = false;
120                                        break;
121                                } else {
122                                        if (DEBUG)
123                                                System.out
124                                                                .println("Assignment stmt in order adding to array");
125                                        // the augS is the next assignment in the sequence add to
126                                        // ValueBox array
127                                        array[i] = ((DefinitionStmt) augS).getRightOpBox();
128                                        toRemove.add(aug);
129                                }
130                        }// end checking for 1D pattern
131
132                        if (success) {
133                                DArrayInitExpr tempExpr = new DArrayInitExpr(array, arrayType);
134                                DArrayInitValueBox tempValueBox = new DArrayInitValueBox(
135                                                tempExpr);
136                                DAssignStmt newStmt = new DAssignStmt(ds.getLeftOpBox(),
137                                                tempValueBox);
138                                // cant do array initialization without declaration being part
139                                // of the stmt!!!!!
140                                // have to prove that this array is never utilized before i.e.
141                                // from start of method to this point there is no use
142                                // or def of this array then only can we create this decl/init
143                                // stmt
144                                if (DEBUG)
145                                        System.out
146                                                        .println("Created new DAssignStmt and replacing it");
147
148                                InitializationDeclarationShortcut shortcutChecker = new InitializationDeclarationShortcut(
149                                                as);
150                                methodNode.apply(shortcutChecker);
151                                boolean possible = shortcutChecker.isShortcutPossible();
152
153                                if (possible) {
154                                        if (DEBUG)
155                                                System.out.println("Shortcut is possible");
156
157                                        // create shortcut stmt
158                                        DShortcutAssignStmt newShortcutStmt = new DShortcutAssignStmt(
159                                                        newStmt, arrayType);
160                                        as.set_Stmt(newShortcutStmt);
161                                        // make sure to mark the local in the DVariableDeclarations
162                                        // so that its not printed
163                                        markLocal(ds.getLeftOp());
164
165                                }
166                                break;
167                        }
168                }// end going through stmt seq node
169                if (success) {
170                        // means we did a transformation remove the stmts
171                        List<AugmentedStmt> newStmtList = new ArrayList<AugmentedStmt>();
172                        for (AugmentedStmt as : node.getStatements()) {
173                                if (toRemove.contains(as)) {
174                                        toRemove.remove(as);
175                                } else {
176                                        newStmtList.add(as);
177                                }
178                        }
179                        node.setStatements(newStmtList);
180
181                        // make sure any other possible simplifications are done
182                        inASTStatementSequenceNode(node);
183                        G.v().ASTTransformations_modified = true;
184                }
185
186                // try the second pattern also
187                secondPattern(node);
188        }

190        /*
191         * Check that s is a definition stmt which assigns to array leftOp and index
192         * location index
193         */
194        public boolean isInSequenceAssignment(Stmt s, Value leftOp, int index) {
195                // DEBUG=false;
196                if (!(s instanceof DefinitionStmt))
197                        return false;
198
199                DefinitionStmt ds = (DefinitionStmt) s;
200                Value leftValue = ds.getLeftOp();
201                if (!(leftValue instanceof ArrayRef))
202                        return false;
203
204                if (DEBUG) {
205                        System.out.println("Stmt number " + index
206                                        + " is an array ref assignment" + leftValue);
207                        System.out.println("Array is" + leftOp);
208                }
209
210                ArrayRef leftRef = (ArrayRef) leftValue;
211                if (!(leftOp.equals(leftRef.getBase()))) {
212                        if (DEBUG)
213                                System.out.println("Not assigning to same array");
214                        return false;
215                }
216
217                if (!(leftRef.getIndex() instanceof IntConstant)) {
218                        if (DEBUG)
219                                System.out.println("Cant determine index of assignment");
220                        return false;
221                }
222
223                IntConstant leftIndex = (IntConstant) leftRef.getIndex();
224                if (leftIndex.value != index) {
225                        if (DEBUG)
226                                System.out.println("Out of order assignment");
227                        return false;
228                }
229
230                return true;
231        }

233        /*
234         * Maybe its a multi-D array then we need to look for a DAssignStmt followed
235         * by a definition Stmt
236         */
237        public void secondPattern(ASTStatementSequenceNode node) {
238                boolean success = false;
239                ArrayList<AugmentedStmt> toRemove = new ArrayList<AugmentedStmt>();
240                for (AugmentedStmt as : node.getStatements()) {
241                        success = false;
242                        Stmt s = as.get_Stmt();
243                        if (!(s instanceof DefinitionStmt))
244                                continue;
245
246                        DefinitionStmt ds = (DefinitionStmt) s;
247                        ValueBox right = ds.getRightOpBox();
248                        Value rightValue = right.getValue();
249
250                        if (!(rightValue instanceof NewArrayExpr))
251                                continue;
252
253                        if (DEBUG) {
254                                System.out.println("Found a new ArrayExpr" + rightValue);
255                                System.out.println("Type of array is:" + rightValue.getType());
256                        }
257
258                        // get type out
259                        Type arrayType = rightValue.getType();
260
261                        // get size....need to know this statically for sure!!!
262                        Value size = ((NewArrayExpr) rightValue).getSize();
263
264                        if (!(size instanceof IntConstant))
265                                continue;
266
267                        if (((IntConstant) size).value == 0) {
268                                debug("Found value to be 0 doing nothing");
269                                continue;
270                        }
271                        if (DEBUG)
272                                System.out.println("Size of array is: "
273                                                + ((IntConstant) size).value);
274
275                        Iterator<AugmentedStmt> tempIt = node.getStatements().iterator();
276                        // get to the array creation stmt
277                        while (tempIt.hasNext()) {
278                                AugmentedStmt tempAs = (AugmentedStmt) tempIt.next();
279                                Stmt tempS = tempAs.get_Stmt();
280                                if (tempS.equals(s))
281                                        break;
282                        }
283                        // have the size have the type, tempIt is poised at the current def
284                        // stmt
285                        ValueBox[] array = new ValueBox[((IntConstant) size).value];
286                        success = true;
287                        for (int i = 0; i < ((IntConstant) size).value; i++) {
288                                // check for each iteration there is one DAssignStmt followed by
289                                // a DefinitionStmt
290                                if (!tempIt.hasNext()) {
291                                        // since its end of the stmt seq node just return
292                                        if (DEBUG)
293                                                System.out.println("returning");
294                                        return;
295                                }
296
297                                AugmentedStmt augOne = tempIt.next();
298                                Stmt augSOne = augOne.get_Stmt();
299
300                                if (!tempIt.hasNext()) {
301                                        // since its end of the stmt seq node just return
302                                        if (DEBUG)
303                                                System.out.println("returning");
304                                        return;
305                                }
306
307                                AugmentedStmt augTwo = tempIt.next();
308                                Stmt augSTwo = augTwo.get_Stmt();
309
310                                if (!isInSequenceAssignmentPatternTwo(augSOne, augSTwo,
311                                                ds.getLeftOp(), i)) {
312                                        // cant create shortcut since we dont have all necessary
313                                        // initializations
314                                        if (DEBUG)
315                                                System.out
316                                                                .println("Out of order assignment aborting attempt");
317                                        success = false;
318                                        break;
319                                } else {
320                                        if (DEBUG)
321                                                System.out
322                                                                .println("Assignment stmt in order adding to array");
323                                        // the RHS of augSOne is the next assignment in the sequence
324                                        // add to ValueBox array
325                                        array[i] = ((DShortcutAssignStmt) augSOne).getRightOpBox();
326                                        toRemove.add(augOne);
327                                        toRemove.add(augTwo);
328                                }
329                        }// end checking for 1D pattern
330
331                        if (success) {
332                                DArrayInitExpr tempExpr = new DArrayInitExpr(array, arrayType);
333                                DArrayInitValueBox tempValueBox = new DArrayInitValueBox(
334                                                tempExpr);
335                                DAssignStmt newStmt = new DAssignStmt(ds.getLeftOpBox(),
336                                                tempValueBox);
337                                // cant do array initialization without declaration being part
338                                // of the stmt!!!!!
339                                // have to prove that this array is never utilized before i.e.
340                                // from start of method to this point there is no use
341                                // or def of this array then only can we create this decl/init
342                                // stmt
343                                if (DEBUG)
344                                        System.out
345                                                        .println("Created new DAssignStmt and replacing it");
346
347                                InitializationDeclarationShortcut shortcutChecker = new InitializationDeclarationShortcut(
348                                                as);
349                                methodNode.apply(shortcutChecker);
350                                boolean possible = shortcutChecker.isShortcutPossible();
351
352                                if (possible) {
353                                        if (DEBUG)
354                                                System.out.println("Shortcut is possible");
355
356                                        // create shortcut stmt
357                                        DShortcutAssignStmt newShortcutStmt = new DShortcutAssignStmt(
358                                                        newStmt, arrayType);
359                                        as.set_Stmt(newShortcutStmt);
360                                        // make sure to mark the local in the DVariableDeclarations
361                                        // so that its not printed
362                                        markLocal(ds.getLeftOp());
363                                }
364
365                                break;
366                        }
367                }// end going through stmt seq node
368                if (success) {
369                        // means we did a transformation remove the stmts
370                        List<AugmentedStmt> newStmtList = new ArrayList<AugmentedStmt>();
371                        for (AugmentedStmt as : node.getStatements()) {
372                                if (toRemove.contains(as)) {
373                                        toRemove.remove(as);
374                                } else {
375                                        newStmtList.add(as);
376                                }
377                        }
378                        node.setStatements(newStmtList);
379
380                        // make sure any other possible simplifications are done
381                        inASTStatementSequenceNode(node);
382                        G.v().ASTTransformations_modified = true;
383                }
384
385        }

387        /*
388         * Check that one is a definition stmt which declares an array Check that
389         * two is a definition of leftOp at index location
390         */
391        public boolean isInSequenceAssignmentPatternTwo(Stmt one, Stmt two,
392                        Value leftOp, int index) {
393                if (!(two instanceof DefinitionStmt))
394                        return false;
395
396                DefinitionStmt ds = (DefinitionStmt) two;
397                Value leftValue = ds.getLeftOp();
398                if (!(leftValue instanceof ArrayRef))
399                        return false;
400
401                ArrayRef leftRef = (ArrayRef) leftValue;
402                if (!(leftOp.equals(leftRef.getBase()))) {
403                        if (DEBUG)
404                                System.out.println("Not assigning to same array");
405                        return false;
406                }
407
408                if (!(leftRef.getIndex() instanceof IntConstant)) {
409                        if (DEBUG)
410                                System.out.println("Cant determine index of assignment");
411                        return false;
412                }
413
414                IntConstant leftIndex = (IntConstant) leftRef.getIndex();
415                if (leftIndex.value != index) {
416                        if (DEBUG)
417                                System.out.println("Out of order assignment");
418                        return false;
419                }
420
421                Value rightOp = ds.getRightOp();
422
423                if (!(one instanceof DShortcutAssignStmt))
424                        return false;
425
426                DShortcutAssignStmt shortcut = (DShortcutAssignStmt) one;
427                Value shortcutVar = shortcut.getLeftOp();
428                if (!shortcutVar.equals(rightOp))
429                        return false;
430
431                return true;
432        }

434        // This local is declared using a shortcut declaration hence needs to be
435        // somehow removed from the declared locals list
436        // actually all that is needed is that this localnot be printed there! it
437        // might be a good
438        // idea to keep the local stored there since other analyses use that as the
439        // list of all
440        // declared locals in the method
441        public void markLocal(Value shortcutLocal) {
442                if (!(shortcutLocal instanceof Local))
443                        throw new DecompilationException(
444                                        "Found non local. report to developer.");
445
446                methodNode.addToDontPrintLocalsList((Local) shortcutLocal);
447        }
