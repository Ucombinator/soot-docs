030/*
031  Nomair A. Naeem 21-FEB-2005
032
033  In the depthFirstAdaptor children of a ASTNode
034  are gotten in three ways
035  a, ASTStatementSequenceNode uses one way see caseASTStatementSequenceNode in DepthFirstAdapter
036  b, ASTTryNode uses another way see caseASTTryNode in DepthFirstAdapter
037  c, All other nodes use normalRetrieving method to retrieve the children
038
039  TO MAKE CODE EFFECIENT BLOCK THE ANALYSIS TO GOING INTO STATEMENTS
040  this is done by overriding the caseASTStatementSequenceNode
041
042  Current tasks of the cleaner
043         Invoke UselessLabeledBlockRemover
044         Invoke EmptyElseRemover
045         Apply OrAggregatorThree
046*/
047

062      Note the ASTNode in this case can be any of the following:
063      ASTMethodNode
064      ASTSwitchNode

074    public void normalRetrieving(ASTNode node){
075        if(node instanceof ASTSwitchNode){
076                   dealWithSwitchNode((ASTSwitchNode)node);
077            return;
078        }
079
080        //from the Node get the subBodes
081        Iterator<Object> sbit = node.get_SubBodies().iterator();
082
083        //onlyASTIfElseNode has 2 subBodies but we need to deal with that
084        int subBodyNumber=0; 
085        while (sbit.hasNext()) {
086            Object subBody = sbit.next();
087            Iterator it = ((List) subBody).iterator();
088
089            int nodeNumber=0;
090            //go over the ASTNodes in this subBody and apply
091            while (it.hasNext()){
092                ASTNode temp = (ASTNode) it.next();
093                if(temp instanceof ASTLabeledBlockNode){
094                    //check if the label is null
095                    ASTLabeledBlockNode labelBlock = (ASTLabeledBlockNode)temp;
096                    SETNodeLabel label = labelBlock.get_Label();
097                    if(label.toString()==null){
098                        //uselessLabeledBlock Found REMOVE IT
099                        UselessLabeledBlockRemover.removeLabeledBlock(node,labelBlock,subBodyNumber,nodeNumber);
100                        if(G.v().ASTTransformations_modified){
101                            return;
102                        }                       
103                    }
104                }
105                else if(temp instanceof ASTIfElseNode){
106                    //check if there is an empty else body
107                    List<Object> elseBody = ((ASTIfElseNode)temp).getElseBody();
108                    if(elseBody.size()==0){
109                        EmptyElseRemover.removeElseBody(node,(ASTIfElseNode)temp,subBodyNumber,nodeNumber);
110                    }
111                }
112                else if(temp instanceof ASTIfNode){
113                    //check if the next node in the subBody is also an ASTIfNode in which case invoke OrAggregatorThree
114                    if(it.hasNext()){//means we can get the nodeNumber+1
115                        ASTNode nextNode = (ASTNode)((List)subBody).get(nodeNumber+1);
116                        if(nextNode instanceof ASTIfNode){
117                            //found an If followed by another if might match Patter 3. 
118                            OrAggregatorThree.checkAndTransform(node,(ASTIfNode)temp,(ASTIfNode)nextNode,nodeNumber,subBodyNumber);
119                            if(G.v().ASTTransformations_modified){
120                                //if we modified something we want to stop since the tree is stale
121                                //System.out.println("here");
122                                return;
123                            }
124                            
125                        }
126                    }
127                }
128                temp.apply(this);
129                nodeNumber++;
130            }
131            subBodyNumber++;
132        }//end of going over subBodies
133        }

135    public void caseASTTryNode(ASTTryNode node){
136        inASTTryNode(node);
137
138        //get try body 
139        List<Object> tryBody = node.get_TryBody();
140        Iterator<Object> it = tryBody.iterator();
141
142        int nodeNumber=0;
143        //go over the ASTNodes and apply
144        while (it.hasNext()){
145            ASTNode temp = (ASTNode) it.next();
146            if(temp instanceof ASTLabeledBlockNode){
147                //check if the label is null
148                ASTLabeledBlockNode labelBlock = (ASTLabeledBlockNode)temp;
149                SETNodeLabel label = labelBlock.get_Label();
150                if(label.toString()==null){
151                    //uselessLabeledBlock Found REMOVE IT
152                    
153                    List<Object> newBody=UselessLabeledBlockRemover.createNewSubBody(tryBody,nodeNumber,labelBlock);
154                    if(newBody!=null){
155                        //something did not go wrong
156                        node.replaceTryBody(newBody);
157                        G.v().ASTTransformations_modified = true;
158                        //System.out.println("REMOVED LABEL from within trybody");
159                    }
160                }
161            }
162            else if(temp instanceof ASTIfElseNode){
163                //check if there is an empty else body
164                List<Object> elseBody = ((ASTIfElseNode)temp).getElseBody();
165                if(elseBody.size()==0){
166                    //System.out.println("Empty else body found"+temp);
167                    List<Object> newBody=EmptyElseRemover.createNewNodeBody(tryBody,nodeNumber,(ASTIfElseNode)temp);
168                    if(newBody!=null){
169                        //something did not go wrong
170                        node.replaceTryBody(newBody);
171                        G.v().ASTTransformations_modified = true;
172                        //System.out.println("REMOVED ELSEBODY from within trybody");
173                        return;
174                    }
175                }
176            }
177            else if(temp instanceof ASTIfNode){
178                //check if the next node in the subBody is also an ASTIfNode in which case invoke OrAggregatorThree
179                if(it.hasNext()){//means we can get the nodeNumber+1
180                    ASTNode nextNode = (ASTNode)tryBody.get(nodeNumber+1);
181                    if(nextNode instanceof ASTIfNode){
182                        //found an If followed by another if might match Patter 3. 
183                        List<Object> newBody=OrAggregatorThree.createNewNodeBody(tryBody,nodeNumber,(ASTIfNode)temp,(ASTIfNode)nextNode);
184                        if(newBody!=null){
185                            //something did not go wrong and pattern was matched
186                            node.replaceTryBody(newBody);
187                            G.v().ASTTransformations_modified = true;
188                            //we modified something we want to stop since the tree is stale
189                            //System.out.println("here");
190                            return;
191                            //System.out.println("OR AGGREGATOR THREE");
192                        }
193                    }
194                }
195            }
196            temp.apply(this);
197            nodeNumber++;
198        }
199
200
201
202
203        Map<Object, Object> exceptionMap = node.get_ExceptionMap();
204        Map<Object, Object> paramMap = node.get_ParamMap();
205        //get catch list and apply on the following
206        // a, type of exception caught
207        // b, local of exception
208        // c, catchBody
209        List<Object> catchList = node.get_CatchList();
210        Iterator<Object> itBody=null;
211        it = catchList.iterator();
212        while (it.hasNext()) {
213            ASTTryNode.container catchBody = (ASTTryNode.container)it.next();
214            
215            SootClass sootClass = ((SootClass)exceptionMap.get(catchBody));
216            Type type = sootClass.getType();
217            
218            //apply on type of exception
219            caseType(type);
220
221            //apply on local of exception
222            Local local = (Local)paramMap.get(catchBody);
223            decideCaseExprOrRef(local);
224
225            //apply on catchBody
226            List<Object> body = (List<Object>)catchBody.o;
227            itBody = body.iterator();
228
229            nodeNumber=0;
230            //go over the ASTNodes and apply
231            while (itBody.hasNext()){
232                ASTNode temp = (ASTNode) itBody.next();
233                if(temp instanceof ASTLabeledBlockNode){
234                    //check if the label is null
235                    ASTLabeledBlockNode labelBlock = (ASTLabeledBlockNode)temp;
236                    SETNodeLabel label = labelBlock.get_Label();
237                    if(label.toString()==null){
238                        //uselessLabeledBlock Found REMOVE IT
239                        
240                        List<Object> newBody=UselessLabeledBlockRemover.createNewSubBody(body,nodeNumber,labelBlock);
241                        if(newBody!=null){
242                            //something did not go wrong
243                            catchBody.replaceBody(newBody);
244                            G.v().ASTTransformations_modified = true;
245                            //System.out.println("REMOVED LABEL from within catchlist");
246                        }
247                        
248                    }
249                }
250                else if(temp instanceof ASTIfElseNode){
251                    //check if there is an empty else body
252                    List<Object> elseBody = ((ASTIfElseNode)temp).getElseBody();
253                    if(elseBody.size()==0){
254                        //System.out.println("Empty else body found"+temp);
255                        List<Object> newBody=EmptyElseRemover.createNewNodeBody(body,nodeNumber,(ASTIfElseNode)temp);
256                        if(newBody!=null){
257                            //something did not go wrong
258                            catchBody.replaceBody(newBody);
259                            G.v().ASTTransformations_modified = true;
260                            //System.out.println("REMOVED ELSEBODY FROm within catchlist");
261                            return;
262                        }
263                    }
264                }
265                else if(temp instanceof ASTIfNode){
266                    //check if the next node in the subBody is also an ASTIfNode in which case invoke OrAggregatorThree
267                    if(itBody.hasNext()){//means we can get the nodeNumber+1
268                        ASTNode nextNode = (ASTNode)body.get(nodeNumber+1);
269                        if(nextNode instanceof ASTIfNode){
270                            //found an If followed by another if might match Patter 3. 
271                            List<Object> newBody=OrAggregatorThree.createNewNodeBody(body,nodeNumber,(ASTIfNode)temp,(ASTIfNode)nextNode);
272                            if(newBody!=null){
273                                //something did not go wrong and pattern was matched
274                                catchBody.replaceBody(newBody);
275                                G.v().ASTTransformations_modified = true;
276                                //System.out.println("OR AGGREGATOR THREE");
277                                return;
278                            }
279                        }
280                    }
281                }
282                temp.apply(this);
283                nodeNumber++;
284            }
285        }
286        
287        outASTTryNode(node);
288    }

291    private void dealWithSwitchNode(ASTSwitchNode node){
292        //do a depthfirst on elements of the switchNode
293
294        List<Object> indexList = node.getIndexList();
295        Map<Object, List<Object>> index2BodyList = node.getIndex2BodyList();
296
297        Iterator<Object> it = indexList.iterator();
298        while (it.hasNext()) {//going through all the cases of the switch statement
299            Object currentIndex = it.next();
300            List<Object> body = index2BodyList.get( currentIndex);
301            
302            if (body != null){
303                //this body is a list of ASTNodes 
304
305                Iterator<Object> itBody = body.iterator();
306                int nodeNumber=0;
307                //go over the ASTNodes and apply
308                while (itBody.hasNext()){
309                    ASTNode temp = (ASTNode) itBody.next();
310                    if(temp instanceof ASTLabeledBlockNode){
311                        //check if the label is null
312                        ASTLabeledBlockNode labelBlock = (ASTLabeledBlockNode)temp;
313                        SETNodeLabel label = labelBlock.get_Label();
314                        if(label.toString()==null){
315                            //uselessLabeledBlock Found REMOVE IT
316                    
317                            List<Object> newBody=UselessLabeledBlockRemover.createNewSubBody(body,nodeNumber,labelBlock);
318                            if(newBody!=null){
319                                //something did not go wrong
320                                
321                                //put this body in the Map
322                                index2BodyList.put(currentIndex,newBody);
323                                //replace in actual switchNode
324                                node.replaceIndex2BodyList(index2BodyList);
325                                G.v().ASTTransformations_modified = true;
326                                //System.out.println("REMOVED LABEL From Within Switch");
327                            }
328                        }
329                    }
330                    else if(temp instanceof ASTIfElseNode){
331                        //check if there is an empty else body
332                        List<Object> elseBody = ((ASTIfElseNode)temp).getElseBody();
333                        if(elseBody.size()==0){
334                            //System.out.println("Empty else body found"+temp);
335                            List<Object> newBody=EmptyElseRemover.createNewNodeBody(body,nodeNumber,(ASTIfElseNode)temp);
336                            if(newBody!=null){
337                                //something did not go wrong
338                                
339                               //put this body in the Map
340                                index2BodyList.put(currentIndex,newBody);
341                                //replace in actual switchNode
342                                node.replaceIndex2BodyList(index2BodyList);
343                                G.v().ASTTransformations_modified = true;
344                                //System.out.println("REMOVED ELSEBODY FROM WITHIN SWITCH");
345                                return;
346                            }
347                        }
348                    }
349                    else if(temp instanceof ASTIfNode){
350                        //check if the next node in the subBody is also an ASTIfNode in which case invoke OrAggregatorThree
351                        if(itBody.hasNext()){//means we can get the nodeNumber+1
352                            ASTNode nextNode = (ASTNode)body.get(nodeNumber+1);
353                            if(nextNode instanceof ASTIfNode){
354                                //found an If followed by another if might match Patter 3. 
355                                List<Object> newBody=OrAggregatorThree.createNewNodeBody(body,nodeNumber,(ASTIfNode)temp,(ASTIfNode)nextNode);
356                                if(newBody!=null){
357                                    //something did not go wrong and pattern was matched
358
359                                    //put this body in the Map
360                                    index2BodyList.put(currentIndex,newBody);
361                                    //replace in actual switchNode
362                                    node.replaceIndex2BodyList(index2BodyList);
363
364                                    G.v().ASTTransformations_modified = true;
365                                    //System.out.println("OR AGGREGATOR THREE");
366                                    return;
367                                }
368                            }
369                        }
370                    }
371                    temp.apply(this);
372                    nodeNumber++;
373                }
374            }
375        }
376    }
