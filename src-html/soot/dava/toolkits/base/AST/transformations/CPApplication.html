030/*
031 * The traversal utilizes the results of the CP (constant propagation analysis) to substitute uses
032 * of locals where ever possible. Note we also have information about the constant fields in the program but
033 * we are not going to de-inline those fields because it is thought that refereing to the field as a TYPE-DEF 
034 * gives more info than the actual value
035 * 
036 * 
037 Need to be very clear when a local can be used
038 It can be used in the following places:
039 a, a conditional in if, ifelse, 
040 a1 while , do while, for condition  (after set of these will be used since those are the vars true at the start of the loop
041 i.e. when the loop has not executed and also true at the end of the loop
042
043 b, in the for init or update
044 c, in a switch choice 
045 d, in a syncrhnoized block //wont do dont think we need to since this is a local
046 and synching is always done on objects and we are not tracking objects 
047 d, in a statement
048 *
049 */
050
051public class CPApplication extends DepthFirstAdapter {
052        CP cp = null;
053        String className = null;

056                        HashMap<String, Object> constantValueFields,

072        public void inASTSwitchNode(ASTSwitchNode node) {
073                Object obj = cp.getBeforeSet(node);
074                if (obj == null)
075                        return;
076                if (!(obj instanceof CPFlowSet))
077                        return;
078
079                // before set is a non null CPFlowSet
080                CPFlowSet beforeSet = (CPFlowSet) obj;
081
082                Value key = node.get_Key();
083                if (key instanceof Local) {
084                        Local useLocal = (Local) key;
085                        // System.out.println("switch key is a local: "+useLocal);
086                        Object value = beforeSet.contains(className, useLocal.toString());
087                        if (value != null) {
088                                // System.out.println("switch key Local "+useLocal+"is present in before set with value"+value);
089                                // create constant value for the value and replace this local
090                                // use with the constant value use
091                                Value newValue = CPHelper.createConstant(value);
092                                if (newValue != null) {
093                                        // System.out.println("Substituted the switch key local use with the constant value"+newValue);
094                                        node.set_Key(newValue);
095                                } else {
096                                        // System.out.println("FAILED TO Substitute the local use with the constant value");
097                                }
098                        }
099                } else if (key instanceof FieldRef) {
100                        FieldRef useField = (FieldRef) key;
101                        // System.out.println("switch key is a FieldRef which is: "+useField);
102                        SootField usedSootField = useField.getField();
103                        Object value = beforeSet.contains(usedSootField.getDeclaringClass()
104                                        .getName(), usedSootField.getName().toString());
105                        if (value != null) {
106                                // System.out.println("FieldRef "+usedSootField+"is present in before set with value"+value);
107                                // create constant value for the value and replace this local
108                                // use with the constant value use
109                                Value newValue = CPHelper.createConstant(value);
110                                if (newValue != null) {
111                                        // System.out.println("Substituted the constant field ref use with the constant value"+newValue);
112                                        node.set_Key(newValue);
113                                } else {
114                                        // System.out.println("FAILED TO Substitute the constant field ref use with the constant value");
115                                }
116                        }
117
118                }
119        }

121        public void inASTForLoopNode(ASTForLoopNode node) {
122                /*
123                 * For the init part we should actually use the before set for each init
124                 * stmt
125                 */
126                for (AugmentedStmt as : node.getInit()) {
127                        Stmt s = as.get_Stmt();
128                        List useBoxes = s.getUseBoxes();
129
130                        Object obj = cp.getBeforeSet(s);
131                        if (obj == null)
132                                continue;
133                        if (!(obj instanceof CPFlowSet))
134                                continue;
135
136                        // before set is a non null CPFlowSet
137                        CPFlowSet beforeSet = (CPFlowSet) obj;
138
139                        // System.out.println("Init Statement: "+s);
140                        // System.out.println("Before set is: "+beforeSet.toString());
141
142                        /*
143                         * get all use boxes see if their value is determined from the
144                         * before set if yes replace them
145                         */
146                        substituteUses(useBoxes, beforeSet);
147                }
148
149                // get after set for the condition and update
150                Object obj = cp.getAfterSet(node);
151
152                if (obj == null)
153                        return;
154                if (!(obj instanceof CPFlowSet))
155                        return;
156
157                // after set is a non null CPFlowSet
158                CPFlowSet afterSet = (CPFlowSet) obj;
159
160                // conditon
161                ASTCondition cond = node.get_Condition();
162
163                // System.out.println("For Loop with condition: "+cond);
164                // System.out.println("After set is: "+afterSet.toString());
165
166                changedCondition(cond, afterSet);
167
168                // update
169                for (AugmentedStmt as: node.getUpdate()) {
170                        Stmt s = as.get_Stmt();
171
172                        List useBoxes = s.getUseBoxes();
173
174                        // System.out.println("For update Statement: "+s);
175                        // System.out.println("After set is: "+afterSet.toString());
176
177                        /*
178                         * get all use boxes see if their value is determined from the
179                         * before set if yes replace them
180                         */
181                        substituteUses(useBoxes, afterSet);
182                }
183
184        }

265        /*
266         * Given a unary/binary or aggregated condition this method is used to find
267         * all the useBoxes or locals or fieldref in the case of unary conditions
268         * and then the set is checked for appropriate substitutions
269         */
270        public ASTCondition changedCondition(ASTCondition cond, CPFlowSet set) {
271                if (cond instanceof ASTAggregatedCondition) {
272                        ASTCondition left = changedCondition(
273                                        ((ASTAggregatedCondition) cond).getLeftOp(), set);
274                        ASTCondition right = changedCondition(
275                                        ((ASTAggregatedCondition) cond).getRightOp(), set);
276                        ((ASTAggregatedCondition) cond).setLeftOp(left);
277                        ((ASTAggregatedCondition) cond).setRightOp(right);
278                        // System.out.println("New condition is: "+cond);
279                        return cond;
280                } else if (cond instanceof ASTUnaryCondition) {
281                        Value val = ((ASTUnaryCondition) cond).getValue();
282                        if (val instanceof Local) {
283                                Object value = set
284                                                .contains(className, ((Local) val).toString());
285                                if (value != null) {
286                                        // System.out.println("if Condition Local "+((Local)val)+"is present in before set with value"+value);
287                                        // create constant value for the value and replace this
288                                        // local use with the constant value use
289                                        Value newValue = CPHelper.createConstant(value);
290                                        if (newValue != null) {
291                                                // System.out.println("Substituted the local use with the constant value"+newValue);
292                                                ((ASTUnaryCondition) cond).setValue(newValue);
293                                        } else {
294                                                // System.out.println("FAILED TO Substitute the local use with the constant value");
295                                        }
296                                }
297                        } else if (val instanceof FieldRef) {
298                                FieldRef useField = (FieldRef) val;
299                                SootField usedSootField = useField.getField();
300                                Object value = set.contains(usedSootField.getDeclaringClass()
301                                                .getName(), usedSootField.getName().toString());
302                                if (value != null) {
303                                        // System.out.println("if condition FieldRef "+usedSootField+"is present in before set with value"+value);
304                                        // create constant value for the value and replace this
305                                        // field use with the constant value use
306                                        Value newValue = CPHelper.createConstant(value);
307                                        if (newValue != null) {
308                                                // System.out.println("Substituted the constant field ref use with the constant value"+newValue);
309                                                ((ASTUnaryCondition) cond).setValue(newValue);
310                                        } else {
311                                                // System.out.println("FAILED TO Substitute the constant field ref use with the constant value");
312                                        }
313                                }
314                        } else {
315                                substituteUses(val.getUseBoxes(), set);
316                        }
317                        // System.out.println("New condition is: "+cond);
318                        return cond;
319                } else if (cond instanceof ASTBinaryCondition) {
320                        // get uses from binaryCondition
321                        Value val = ((ASTBinaryCondition) cond).getConditionExpr();
322                        substituteUses(val.getUseBoxes(), set);
323
324                        // System.out.println("New condition is: "+cond);
325                        return cond;
326                } else {
327                        throw new RuntimeException(
328                                        "Method getUseList in ASTUsesAndDefs encountered unknown condition type");
329                }
330        }

332        public void inASTStatementSequenceNode(ASTStatementSequenceNode node) {
333                for (AugmentedStmt as : node.getStatements()) {
334                        Stmt s = as.get_Stmt();
335
336                        List useBoxes = s.getUseBoxes();
337
338                        Object obj = cp.getBeforeSet(s);
339
340                        if (obj == null)
341                                continue;
342                        if (!(obj instanceof CPFlowSet))
343                                continue;
344
345                        // before set is a non null CPFlowSet
346                        CPFlowSet beforeSet = (CPFlowSet) obj;
347
348                        // System.out.println("Statement: "+s);
349                        // System.out.println("Before set is: "+beforeSet.toString());
350
351                        /*
352                         * get all use boxes see if their value is determined from the
353                         * before set if yes replace them
354                         */
355                        substituteUses(useBoxes, beforeSet);
356
357                }
358        }

360        public void substituteUses(List useBoxes, CPFlowSet beforeSet) {
361                Iterator useIt = useBoxes.iterator();
362                while (useIt.hasNext()) {
363                        Object useObj = useIt.next();
364                        Value use = ((ValueBox) useObj).getValue();
365                        if (use instanceof Local) {
366                                Local useLocal = (Local) use;
367                                // System.out.println("local is: "+useLocal);
368                                Object value = beforeSet.contains(className,
369                                                useLocal.toString());
370                                if (value != null) {
371                                        // System.out.println("Local "+useLocal+"is present in before set with value"+value);
372                                        // create constant value for the value and replace this
373                                        // local use with the constant value use
374                                        Value newValue = CPHelper.createConstant(value);
375                                        if (newValue != null) {
376                                                // System.out.println("Substituted the local use with the constant value"+newValue);
377                                                ((ValueBox) useObj).setValue(newValue);
378                                        } else {
379                                                // System.out.println("FAILED TO Substitute the local use with the constant value");
380                                        }
381                                }
382                        } else if (use instanceof FieldRef) {
383                                FieldRef useField = (FieldRef) use;
384                                // System.out.println("FieldRef is: "+useField);
385                                SootField usedSootField = useField.getField();
386                                Object value = beforeSet.contains(usedSootField
387                                                .getDeclaringClass().getName(), usedSootField.getName()
388                                                .toString());
389                                if (value != null) {
390                                        // System.out.println("FieldRef "+usedSootField+"is present in before set with value"+value);
391                                        // create constant value for the value and replace this
392                                        // local use with the constant value use
393                                        Value newValue = CPHelper.createConstant(value);
394                                        if (newValue != null) {
395                                                // System.out.println("Substituted the constant field ref use with the constant value"+newValue);
396                                                ((ValueBox) useObj).setValue(newValue);
397                                        } else {
398                                                // System.out.println("FAILED TO Substitute the constant field ref use with the constant value");
399                                        }
400                                }
401
402                        }
403                }
404        }
