049 * This class makes sure there is an initialization of all final variables
050 * (static or non static). If we cant guarantee initialization (may be
051 * initialized on multiple paths but not all) then we remove the final keyword
052 */

062        public FinalFieldDefinition(ASTMethodNode node) {
063                davaBody = node.getDavaBody();
064                sootMethod = davaBody.getMethod();
065                sootClass = sootMethod.getDeclaringClass();
066
067                String subSignature = sootMethod.getName();
068                if (!(subSignature.compareTo("<clinit>") == 0 || subSignature
069                                .compareTo("<init>") == 0)) {
070                        // dont care about these since we want only static block and
071                        // constructors
072                        // System.out.println("\n\nName"+sootMethod.getName()+"
073                        // SubSignature:"+sootMethod.getSubSignature());
074                        return;
075                }
076
077                // create a list of interesting vars
078                ArrayList<SootField> interesting = findFinalFields();
079                if (interesting.size() == 0) {
080                        // no final fields of interest
081                        return;
082                }
083
084                cancelFinalModifier = new ArrayList<SootField>();
085                analyzeMethod(node, interesting);
086
087                Iterator<SootField> it = cancelFinalModifier.iterator();
088                while (it.hasNext()) {
089                        SootField field = it.next();
090                        field.setModifiers((soot.Modifier.FINAL ^ 0xFFFF)
091                                        & field.getModifiers());
092                }
093        }

095        /*
096         * this method finds all the final fields in this class and assigns them to
097         * the finalFields list
098         * 
099         * Note this stores a list of SootFields!!!
100         * 
101         * Fields which are initialized in their declaration should not be added
102         */
103        public ArrayList<SootField> findFinalFields() {
104
105                // first thing is to get a list of all final fields in the class
106                ArrayList<SootField> interestingFinalFields = new ArrayList<SootField>();
107
108                Iterator fieldIt = sootClass.getFields().iterator();
109                while (fieldIt.hasNext()) {
110                        SootField tempField = (SootField) fieldIt.next();
111                        if (tempField.isFinal()) {
112
113                                // if its static final and method is static add
114                                if (tempField.isStatic()
115                                                && sootMethod.getName().compareTo("<clinit>") == 0) {
116                                        interestingFinalFields.add(tempField);
117                                }
118
119                                // if its non static and final and method is constructor add
120                                if ((!tempField.isStatic())
121                                                && sootMethod.getName().compareTo("<init>") == 0) {
122                                        interestingFinalFields.add(tempField);
123                                }
124                        }
125                }
126                return interestingFinalFields;
127        }

129        public void analyzeMethod(ASTMethodNode node, List<SootField> varsOfInterest) {
130                MustMayInitialize must = new MustMayInitialize(node,
131                                MustMayInitialize.MUST);
132
133                Iterator<SootField> it = varsOfInterest.iterator();
134                while (it.hasNext()) {
135                        SootField interest = it.next();
136
137                        // check for constant value tags
138                        Type fieldType = interest.getType();
139                        if (fieldType instanceof DoubleType
140                                        && interest.hasTag("DoubleConstantValueTag")) {
141                                continue;
142                        } else if (fieldType instanceof FloatType
143                                        && interest.hasTag("FloatConstantValueTag")) {
144                                continue;
145                        } else if (fieldType instanceof LongType
146                                        && interest.hasTag("LongConstantValueTag")) {
147                                continue;
148                        } else if (fieldType instanceof CharType
149                                        && interest.hasTag("IntegerConstantValueTag")) {
150                                continue;
151                        } else if (fieldType instanceof BooleanType
152                                        && interest.hasTag("IntegerConstantValueTag")) {
153                                continue;
154                        } else if ((fieldType instanceof IntType
155                                        || fieldType instanceof ByteType || fieldType instanceof ShortType)
156                                        && interest.hasTag("IntegerConstantValueTag")) {
157                                continue;
158                        } else if (interest.hasTag("StringConstantValueTag")) {
159                                continue;
160                        }
161
162                        if (must.isMustInitialized(interest)) {
163                                // was initialized on all paths couldnt ask for more
164                                continue;
165                        }
166
167                        // System.out.println("SootField: "+interest+" not initialized.
168                        // checking may analysis");
169                        MustMayInitialize may = new MustMayInitialize(node,
170                                        MustMayInitialize.MAY);
171                        if (may.isMayInitialized(interest)) {
172                                // System.out.println("It is initialized on some path just not
173                                // all paths\n");
174                                List defs = must.getDefs(interest);
175                                if (defs == null)
176                                        throw new RuntimeException("Sootfield: " + interest
177                                                        + " is mayInitialized but the defs is null");
178
179                                handleAssignOnSomePaths(node, interest, defs);
180                        } else {
181                                // not initialized on any path., assign default
182                                // System.out.println("Final field is not initialized on any
183                                // path--------ASSIGN DEFAULT VALUE");
184                                assignDefault(node, interest);
185                        }
186                }
187        }

189        /*
190         * One gets to this method only if there was NO definition of a static final
191         * field in the static body At the same time no TAG with a constant value
192         * matched, so we know the static final was not initialized at declaration
193         * time If this happens: though it shouldnt unless u come from non-java
194         * compilers...insert default value initialization into the static
195         * method...right at the end to make things easy
196         */
197        public void assignDefault(ASTMethodNode node, SootField f) {
198
199                // create initialization stmt
200                AugmentedStmt defaultStmt = createDefaultStmt(f);
201
202                if (defaultStmt == null)
203                        return;
204
205                List<Object> subBodies = node.get_SubBodies();
206                if (subBodies.size() != 1)
207                        throw new RuntimeException(
208                                        "SubBodies size of method node not equal to 1");
209
210                List<Object> body = (List<Object>) subBodies.get(0);
211
212                // check if the bodys last node is an ASTStatementSequenceNode where we
213                // might be able to add
214
215                boolean done = false;
216                if (body.size() != 0) {
217                        ASTNode lastNode = (ASTNode) body.get(body.size() - 1);
218                        if (lastNode instanceof ASTStatementSequenceNode) {
219                                List<AugmentedStmt> stmts = ((ASTStatementSequenceNode) lastNode)
220                                                .getStatements();
221                                if (stmts.size() != 0) {
222                                        Stmt s = stmts.get(0).get_Stmt();
223                                        if (!(s instanceof DVariableDeclarationStmt)) {
224                                                // can add statement here
225                                                stmts.add(defaultStmt);
226
227                                                ASTStatementSequenceNode newNode = new ASTStatementSequenceNode(
228                                                                stmts);
229                                                // replace this node with the original node
230
231                                                body.remove(body.size() - 1);
232                                                body.add(newNode);
233
234                                                node.replaceBody(body);
235                                                done = true;
236                                        }
237                                }
238                        }
239                }
240                if (!done) {
241                        List<AugmentedStmt> newBody = new ArrayList<AugmentedStmt>();
242                        newBody.add(defaultStmt);
243
244                        ASTStatementSequenceNode newNode = new ASTStatementSequenceNode(
245                                        newBody);
246                        body.add(newNode);
247
248                        node.replaceBody(body);
249                }
250
251        }

253        public AugmentedStmt createDefaultStmt(Object field) {
254
255                Value ref = null;
256                Type fieldType = null;
257                if (field instanceof SootField) {
258                        // have to make a static field ref
259                        SootFieldRef tempFieldRef = ((SootField) field).makeRef();
260
261                        fieldType = ((SootField) field).getType();
262                        if (((SootField) field).isStatic())
263                                ref = new DStaticFieldRef(tempFieldRef, true);
264                        else
265                                ref = new DInstanceFieldRef(new JimpleLocal("this", fieldType),
266                                                tempFieldRef, new HashSet<Object>());
267
268                } else if (field instanceof Local) {
269                        ref = (Local) field;
270                        fieldType = ((Local) field).getType();
271                }
272
273                GAssignStmt assignStmt = null;
274
275                if (fieldType instanceof RefType) {
276                        assignStmt = new GAssignStmt(ref, NullConstant.v());
277                } else if (fieldType instanceof DoubleType) {
278                        assignStmt = new GAssignStmt(ref, DoubleConstant.v(0));
279                } else if (fieldType instanceof FloatType) {
280                        assignStmt = new GAssignStmt(ref, FloatConstant.v(0));
281                } else if (fieldType instanceof LongType) {
282                        assignStmt = new GAssignStmt(ref, LongConstant.v(0));
283                } else if (fieldType instanceof IntType
284                                || fieldType instanceof ByteType
285                                || fieldType instanceof ShortType
286                                || fieldType instanceof CharType
287                                || fieldType instanceof BooleanType) {
288
289                        assignStmt = new GAssignStmt(ref, DIntConstant.v(0, fieldType));
290                }
291
292                if (assignStmt != null) {
293                        // System.out.println("AssignStmt is"+assignStmt);
294                        AugmentedStmt as = new AugmentedStmt(assignStmt);
295                        return as;
296                } else
297                        return null;
298
299        }

310        public void handleAssignOnSomePaths(ASTMethodNode node, SootField field,
311                        List defs) {
312        
313                if (defs.size() != 1) {
314                        // give up by removing "final" if there are more than one defs
315                        cancelFinalModifier.add(field);
316                } else {
317                        // if there is only one definition 
318
319                        // see if there is no use of def
320                        AllVariableUses varUses = new AllVariableUses(node);
321                        node.apply(varUses);
322
323                        List allUses = varUses.getUsesForField(field);
324
325                        if (allUses != null && allUses.size() != 0) {
326                                /*
327                                 * if the number of uses is not 0 then we dont want to get into
328                                 * trying to delay initialization just before assignment.
329                                 * Easier to remove "final"
330                                 */
331                                cancelFinalModifier.add(field);
332                        } 
333                        else {
334                                /*
335                                 * we have a final field with 1 def and 0 uses but is not initialized on all paths
336                                 * we can try to delay initialization using an indirect approach
337                                 *                         STMT0        TYPE DavaTemp_fieldName;            
338                                 *                         STMT1        DavaTemp_fieldname = DEFAULT
339                                 * try{                                  try{
340                                 *     field = ...         STMT2            DavaTemp_fieldname = ... 
341                                 *    }                                         X
342                                 *    catch(...){                        }catch(..){
343                                 *       ....                                   ....
344                                 *    }                                  }
345                                 *                        STMT3          field = Dava_tempVar
346                                 * 
347                                 * Notice the following code will try to place the field assignment
348                                 * as close to the original assignment as possible.
349                                 * 
350                                 * TODO: However there might still be issues with delaying this assignment
351                                 * e.g. what if the place marked by X (more specifically between
352                                 * the original def and the new def includes a method invocation
353                                 * which access the delayed field.
354                                 * 
355                                 * Original Comment February 2nd, 2006: Laurie mentioned that apart from direct
356                                 * uses we also have to be conservative about method calls since
357                                 * we are dealing with fields here What if some method was
358                                 * invoked and it tried to use a field whose initialization we
359                                 * are about to delay. This can be done by implementing a small
360                                 * analysis. (See end of this class file.       
361                                 *              
362                                 *  TODO: SHOULD BE CHECKED FOR CODE BETWEEN THE OLD DEF AND THE NEW ASSIGNMENT
363                                 *        Currently checks from some point till end of method
364                                 *
365                                 * MethodCallFinder myMethodCallFinder = new MethodCallFinder( (GAssignStmt) defs.get(0));
366                                 * node.apply(myMethodCallFinder);
367                                 * if (myMethodCallFinder.anyMethodCalls()) {
368                                 *              // there was some method call after the definition stmt so
369                                 *      // we cant continue
370                                 *               // remove the final modifier and leave
371                                 *                //System.out.println("Method invoked somewhere after definition");
372                                 *                 cancelFinalModifier.add(field);
373                                 *                 return;
374                                 *                 }
375                                 */
376                                
377                                
378                                
379                                // Creating STMT0
380                                Type localType = field.getType();
381                                Local newLocal = new JimpleLocal("DavaTemp_" + field.getName(),localType);
382
383                                DVariableDeclarationStmt varStmt = new DVariableDeclarationStmt(localType,davaBody);
384                                
385                                varStmt.addLocal(newLocal);
386                                AugmentedStmt as = new AugmentedStmt(varStmt);
387
388                                // System.out.println("Var Decl stmt"+as);
389
390                                // STORE IT IN Methods Declaration Node
391                                ASTStatementSequenceNode declNode = node.getDeclarations();
392                                List<AugmentedStmt> stmts = declNode.getStatements();
393                                stmts.add(as);
394                        
395                                declNode = new ASTStatementSequenceNode(stmts);
396
397                                List<Object> subBodies = node.get_SubBodies();
398                                if (subBodies.size() != 1)
399                                        throw new DecompilationException("ASTMethodNode does not have one subBody");
400
401                                List<Object> body = (List<Object>) subBodies.get(0);
402
403                                body.remove(0);
404                                body.add(0, declNode);
405
406                                node.replaceBody(body);
407
408                                node.setDeclarations(declNode);
409
410                                
411                                // STMT1 initialization
412                                AugmentedStmt initialization = createDefaultStmt(newLocal);
413                                /*
414                                 * The first node in a method is the declarations
415                                 * we know there is a second node because originaly the
416                                 * field was initialized on some path
417                                 */
418                                if (body.size() < 2)
419                                        throw new RuntimeException("Size of body is less than 1");
420
421                                /* 
422                                 * If the second node is a stmt seq we put STMT1 there
423                                 * otherwise we create a new stmt seq node
424                                 */ 
425
426                                ASTNode nodeSecond = (ASTNode) body.get(1);
427                                if (nodeSecond instanceof ASTStatementSequenceNode) {
428                                        // the second node is a stmt seq node just add the stmt here
429                                        List<AugmentedStmt> stmts1 = ((ASTStatementSequenceNode) nodeSecond).getStatements();
430                                        stmts1.add(initialization);
431                                        nodeSecond = new ASTStatementSequenceNode(stmts1);
432                                        // System.out.println("Init added in exisiting node");
433                                        body.remove(1);
434                                } else {
435                                        //System.out.println("had to add new node");
436                                        List<AugmentedStmt> tempList = new ArrayList<AugmentedStmt>();
437                                        tempList.add(initialization);
438                                        nodeSecond = new ASTStatementSequenceNode(tempList);
439                                }
440                                body.add(1, nodeSecond);
441                                node.replaceBody(body);
442
443                                
444                                
445                                //STMT2
446                                //done by simply replacing the leftop in the original stmt
447                                ((GAssignStmt) defs.get(0)).setLeftOp(newLocal);
448                                
449                                
450                                //STMT3
451                                
452                                // have to make a field ref
453                                SootFieldRef tempFieldRef = (field).makeRef();
454
455                                Value ref;
456                                if (field.isStatic())
457                                        ref = new DStaticFieldRef(tempFieldRef, true);
458                                else {
459                                        ref = new DInstanceFieldRef(new JimpleLocal("this", field
460                                                        .getType()), tempFieldRef, new HashSet<Object>());
461                                        // throw new RuntimeException("STOPPED");
462                                }
463
464                                GAssignStmt assignStmt = new GAssignStmt(ref, newLocal);
465                                AugmentedStmt assignStmt1 = new AugmentedStmt(assignStmt);
466
467
468
469                                /*
470                                 * 14th February 2006
471                                 * Should add this statement to the first place in the code where
472                                 * we will have a mustInitialize satisfied
473                                 */
474
475                                //the def is at (GAssignStmt) defs.get(0)
476                                //its parent is ASTStatementSequence and its parent is now needed
477                                soot.dava.toolkits.base.AST.traversals.ASTParentNodeFinder parentFinder = 
478                                        new soot.dava.toolkits.base.AST.traversals.ASTParentNodeFinder();
479                                node.apply(parentFinder);
480                                
481                                
482                                Object parent = parentFinder.getParentOf(defs.get(0));
483                                if(!(parent instanceof ASTStatementSequenceNode)){
484                                        throw new DecompilationException("Parent of stmt was not a stmt seq node");
485                                }
486                                
487                                Object grandParent = parentFinder.getParentOf(parent);
488                                if(grandParent == null){
489                                        throw new DecompilationException("Parent of stmt seq node was null");
490                                }
491                                
492                                //so we have the parent stmt seq node and the grandparent node
493                                //so it is the grandparent which is causing the error in MUSTINitialize
494                                //we should move our assign right after the grandParent is done
495                                
496                                
497                                MustMayInitialize must = new MustMayInitialize(node,MustMayInitialize.MUST);
498                                while(!must.isMustInitialized(field)) {
499                                        
500                                        //System.out.println("not must initialized");
501                                        Object parentOfGrandParent = parentFinder.getParentOf(grandParent);
502                                        if( !(grandParent instanceof ASTMethodNode) && parentOfGrandParent == null){
503                                                throw new DecompilationException("Parent of non method node was null");
504                                        }
505                                        boolean notResolved=false;
506                                        // look for grandParent in parentOfGrandParent
507                                        ASTNode ancestor = (ASTNode)parentOfGrandParent;
508                                        List<Object> ancestorBodies = ancestor.get_SubBodies();
509                                        Iterator<Object> it = ancestorBodies.iterator();
510                                        while(it.hasNext()){
511                                                List<ASTStatementSequenceNode> ancestorSubBody = null;
512                                        
513                                                if (ancestor instanceof ASTTryNode)
514                                                        ancestorSubBody = (List<ASTStatementSequenceNode>) ((ASTTryNode.container) it.next()).o;
515                                                else
516                                                        ancestorSubBody = (List<ASTStatementSequenceNode>) it.next();
517                                         
518                                                if(ancestorSubBody.indexOf(grandParent) > -1) {
519                                                        //grandParent is present in this body
520                                                        int index = ancestorSubBody.indexOf(grandParent); 
521                                                        //check the next index
522                                                
523                                                        if(index+1 < ancestorSubBody.size() && ancestorSubBody.get(index+1) instanceof ASTStatementSequenceNode ){
524                                                                //there is an stmt seq node node after the grandParent
525                                                                ASTStatementSequenceNode someNode = ancestorSubBody.get(index+1);
526                                                         
527                                                                //add the assign stmt here
528                                                         
529                                                                List<AugmentedStmt> stmtsLast = (someNode).getStatements();
530                                                                List<AugmentedStmt> newStmts = new ArrayList<AugmentedStmt>();
531                                                                newStmts.add(assignStmt1);
532                                                                newStmts.addAll(stmtsLast);
533                                                                someNode.setStatements(newStmts);
534                                                                //System.out.println("here1");
535                                                                //check if problem is solved else remove the assign and change parents
536                                                                must = new MustMayInitialize(node,MustMayInitialize.MUST);
537                                                                if(!must.isMustInitialized(field)){
538                                                                        //problem not solved remove the stmt just added
539                                                                        someNode.setStatements(stmtsLast);
540                                                                        notResolved=true;
541                                                                }
542                                                        }
543                                                        else{
544                                                                //create a new stmt seq node and add it here
545                                                                        List<AugmentedStmt> tempList = new ArrayList<AugmentedStmt>();
546                                                                        tempList.add(assignStmt1);
547                                                                        ASTStatementSequenceNode lastNode = new ASTStatementSequenceNode(tempList);
548                                                                        ancestorSubBody.add(index+1,lastNode);
549                                                                        //node.replaceBody(body);
550                                                                        //System.out.println("here2");
551                                                                        //check if problem is solved else remove the assign and change parents
552                                                                        must = new MustMayInitialize(node,MustMayInitialize.MUST);
553                                                                        if(!must.isMustInitialized(field)){
554                                                                                //problem not solved remove the stmt just added
555                                                                                ancestorSubBody.remove(index+1);
556                                                                                notResolved=true;
557                                                                        }                                                                       
558                                                        }
559                                                        break;//break the loop going through subBodies
560                                                } //if ancestor was found
561                                                                
562                                         }      //next subBody
563                                        if(notResolved){
564                                                //meaning we still dont have must initialization 
565                                                //we should put assign in one level above than current
566                                                grandParent = parentFinder.getParentOf(grandParent);
567                                                //System.out.println("Going one level up");                                             
568                                        }                                                                       
569                                }//while ! ismustinitialized
570                        }
571                }
572        }

582class MethodCallFinder extends DepthFirstAdapter {
583
584        GAssignStmt def;
585
586        boolean foundIt = false;
587
588        boolean anyMethodCalls = false;

590        public MethodCallFinder(GAssignStmt def) {
591                this.def = def;
592        }
593
594        public MethodCallFinder(boolean verbose, GAssignStmt def) {
595                super(verbose);
596                this.def = def;
597        }
598
599        public void outDefinitionStmt(DefinitionStmt s) {
600                if (s instanceof GAssignStmt) {
601                        if (((GAssignStmt) s).equals(def)) {
602                                foundIt = true;
603                                //System.out.println("Found it" + s);
604                        }
605                }
606        }
607
608        public void inInvokeExpr(InvokeExpr ie) {
609                //System.out.println("In invoke Expr");
610                if (foundIt) {
611                        //System.out.println("oops invoking something after definition");
612                        anyMethodCalls = true;
613                }
614        }
615
616        /*
617         * Method will return false if there were no method calls made after the
618         * definition stmt
619         */
620        public boolean anyMethodCalls() {
621                return anyMethodCalls;
622                //return false;
623        }
