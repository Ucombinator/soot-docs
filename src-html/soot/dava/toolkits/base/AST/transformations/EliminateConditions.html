048/*
049 * if (true)   ---> remove conditional copy ifbody to parent
050 * 
051 * if(false) eliminate in all entirety
052 * 
053 * if(true)
054 *    bla1
055 * else 
056 *    bla2        remove conditional copy bla1 to parent
057 *    
058 * if(false)
059 *    bla1
060 *  else
061 *    bla2      remoce conditional copy bla2 to parent
062 *    
063 *       
064 * while(false)  eliminate in entirety... notice this is not an Uncondition loop but a ASTWhileNode
065 * 
066 * do{ .... } while(false)  eliminate loop copy body to parent
067 * 
068 * for(int i =0;false;i++)   remove for .  copy init stmts to parent
069 */
070public class EliminateConditions extends DepthFirstAdapter {

091        public void normalRetrieving(ASTNode node) {
092                modified = false;
093                if (node instanceof ASTSwitchNode) {
094                        do {
095                                modified = false;
096                                dealWithSwitchNode((ASTSwitchNode) node);
097                        } while (modified);
098                        return;
099                }
100                // from the Node get the subBodes
101                Iterator<Object> sbit = node.get_SubBodies().iterator();
102                while (sbit.hasNext()) {
103                        List subBody = (List) sbit.next();
104                        Iterator it = subBody.iterator();
105                        ASTNode temp = null;
106                        Boolean returned = null;
107                        while (it.hasNext()) {
108                                temp = (ASTNode) it.next();
109
110                                // only check condition if this is a control flow node
111                                if (temp instanceof ASTControlFlowNode) {
112                                        bodyContainingNode = null;
113                                        returned = eliminate(temp);
114                                        if (returned != null && canChange(returned, temp)) {
115                                                break;
116                                        } else {
117                                                if (DEBUG)
118                                                        System.out.println("returned is null"
119                                                                        + temp.getClass());
120                                                bodyContainingNode = null;
121                                        }
122                                }
123                                temp.apply(this);
124                        }// end while going through nodes in subBody
125
126                        boolean changed = change(returned, temp);
127                        if (changed)
128                                modified = true;
129                }// end of going over subBodies
130
131                if (modified) {
132                        // repeat the whole thing
133                        normalRetrieving(node);
134                }
135        }

137        public Boolean eliminate(ASTNode node) {
138                ASTCondition cond = null;
139                if (node instanceof ASTControlFlowNode)
140                        cond = ((ASTControlFlowNode) node).get_Condition();
141                else
142                        return null;
143
144                if (cond == null || !(cond instanceof ASTUnaryCondition))
145                        return null;
146
147                ASTUnaryCondition unary = (ASTUnaryCondition) cond;
148                Value unaryValue = unary.getValue();
149
150                boolean notted = false;
151                if (unaryValue instanceof DNotExpr) {
152                        notted = true;
153                        unaryValue = ((DNotExpr) unaryValue).getOp();
154                }
155
156                Boolean isBoolean = isBooleanConstant(unaryValue);
157                if (isBoolean == null) {
158                        // not a constant
159                        return null;
160                }
161
162                boolean trueOrFalse = isBoolean.booleanValue();
163                if (notted) {
164                        // since it is notted we reverse the booleans
165                        trueOrFalse = !trueOrFalse;
166                }
167
168                AST.apply(finder);
169
170                Object temp = finder.getParentOf(node);
171                if (temp == null)
172                        return null;
173
174                ASTNode parent = (ASTNode) temp;
175                List<Object> subBodies = parent.get_SubBodies();
176                Iterator<Object> it = subBodies.iterator();
177
178                int index = -1;
179                while (it.hasNext()) {
180                        bodyContainingNode = (List<Object>) it.next();
181                        index = bodyContainingNode.indexOf(node);
182                        if (index < 0) {
183                                bodyContainingNode = null;
184                        } else {
185                                // bound the body containing Node
186                                return new Boolean(trueOrFalse);
187                        }
188                }
189                return null;
190
191        }

193        /*
194         * Method returns null if the Value is not a constant or not a boolean
195         * constant return true if the constant is true return false if the constant
196         * is false
197         */
198        public Boolean isBooleanConstant(Value internal) {
199
200                if (!(internal instanceof DIntConstant))
201                        return null;
202
203                if (DEBUG)
204                        System.out.println("Found Constant");
205
206                DIntConstant intConst = (DIntConstant) internal;
207
208                if (!(intConst.type instanceof BooleanType))
209                        return null;
210
211                // either true or false
212                if (DEBUG)
213                        System.out.println("Found Boolean Constant");
214
215                if (intConst.value == 1) {
216                        return new Boolean(true);
217                } else if (intConst.value == 0) {
218                        return new Boolean(false);
219                } else
220                        throw new RuntimeException(
221                                        "BooleanType found with value different than 0 or 1");
222        }

224        public Boolean eliminateForTry(ASTNode node) {
225                ASTCondition cond = null;
226                if (node instanceof ASTControlFlowNode)
227                        cond = ((ASTControlFlowNode) node).get_Condition();
228                else
229                        return null;
230
231                if (cond == null || !(cond instanceof ASTUnaryCondition))
232                        return null;
233
234                ASTUnaryCondition unary = (ASTUnaryCondition) cond;
235                Value unaryValue = unary.getValue();
236
237                boolean notted = false;
238                if (unaryValue instanceof DNotExpr) {
239                        notted = true;
240                        unaryValue = ((DNotExpr) unaryValue).getOp();
241                }
242
243                Boolean isBoolean = isBooleanConstant(unaryValue);
244                if (isBoolean == null) {
245                        // not a constant
246                        return null;
247                }
248
249                boolean trueOrFalse = isBoolean.booleanValue();
250                if (notted) {
251                        // since it is notted we reverse the booleans
252                        trueOrFalse = !trueOrFalse;
253                }
254
255                AST.apply(finder);
256
257                Object temp = finder.getParentOf(node);
258                if (temp == null)
259                        return null;
260
261                if (!(temp instanceof ASTTryNode))
262                        throw new RuntimeException(
263                                        "eliminateTry called when parent was not a try node");
264
265                ASTTryNode parent = (ASTTryNode) temp;
266
267                List<Object> tryBody = parent.get_TryBody();
268
269                int index = tryBody.indexOf(node);
270                if (index >= 0) {
271                        // bound the body containing Node
272                        bodyContainingNode = tryBody;
273                        return new Boolean(trueOrFalse);
274                }
275
276                List<Object> catchList = parent.get_CatchList();
277                Iterator<Object> it = catchList.iterator();
278                while (it.hasNext()) {
279                        ASTTryNode.container catchBody = (ASTTryNode.container) it.next();
280
281                        List<Object> body = (List<Object>) catchBody.o;
282                        index = body.indexOf(node);
283                        if (index >= 0) {
284                                // bound the body containing Node
285                                bodyContainingNode = body;
286                                return new Boolean(trueOrFalse);
287                        }
288
289                }
290                return null;
291        }

293        public void caseASTTryNode(ASTTryNode node) {
294                modified = false;
295                inASTTryNode(node);
296                // get try body iterator
297                Iterator<Object> it = node.get_TryBody().iterator();
298
299                Boolean returned = null;
300                ASTNode temp = null;
301                while (it.hasNext()) {
302                        temp = (ASTNode) it.next();
303                        // only check condition if this is a control flow node
304                        if (temp instanceof ASTControlFlowNode) {
305                                bodyContainingNode = null;
306                                returned = eliminateForTry(temp);
307                                if (returned != null && canChange(returned, temp))
308                                        break;
309                                else
310                                        bodyContainingNode = null;
311                        }
312                        temp.apply(this);
313                }// end while
314
315                boolean changed = change(returned, temp);
316                if (changed)
317                        modified = true;
318
319                // get catch list and apply on the following
320                // a, type of exception caught ......... NO NEED
321                // b, local of exception ............... NO NEED
322                // c, catchBody
323                List<Object> catchList = node.get_CatchList();
324                Iterator itBody = null;
325                it = catchList.iterator();
326                while (it.hasNext()) {
327                        ASTTryNode.container catchBody = (ASTTryNode.container) it.next();
328                        List body = (List) catchBody.o;
329                        itBody = body.iterator();
330
331                        returned = null;
332                        temp = null;
333                        // go over the ASTNodes and apply
334                        while (itBody.hasNext()) {
335                                temp = (ASTNode) itBody.next();
336                                // System.out.println("Next node is "+temp);
337                                // only check condition if this is a control flow node
338                                if (temp instanceof ASTControlFlowNode) {
339                                        bodyContainingNode = null;
340                                        returned = eliminateForTry(temp);
341                                        if (returned != null && canChange(returned, temp))
342                                                break;
343                                        else
344                                                bodyContainingNode = null;
345                                }
346                                temp.apply(this);
347                        }
348                        changed = change(returned, temp);
349                        if (changed)
350                                modified = true;
351                }
352                outASTTryNode(node);
353                if (modified) {
354                        // repeat the whole thing
355                        caseASTTryNode(node);
356                }
357        }

360                return true;

363        public boolean change(Boolean returned, ASTNode temp) {
364                if (bodyContainingNode != null && returned != null && temp != null) {
365
366                        int index = bodyContainingNode.indexOf(temp);
367                        if (DEBUG)
368                                System.out.println("in change");
369                        if (temp instanceof ASTIfNode) {
370                                bodyContainingNode.remove(temp);
371
372                                if (returned.booleanValue()) {
373                                        // if statement and value was true put the body of if into
374                                        // the code
375                                        // if its a labeled stmt we need a labeled block instead
376                                        // notice that its okkay to put a labeled block since other
377                                        // transformations might remove it
378                                        String label = ((ASTLabeledNode) temp).get_Label()
379                                                        .toString();
380                                        if (label != null) {
381                                                ASTLabeledBlockNode labeledNode = new ASTLabeledBlockNode(
382                                                                ((ASTLabeledNode) temp).get_Label(),
383                                                                (List<Object>) temp.get_SubBodies().get(0));
384                                                bodyContainingNode.add(index, labeledNode);
385                                        } else {
386                                                bodyContainingNode.addAll(index, (List) temp
387                                                                .get_SubBodies().get(0));
388                                        }
389                                }
390                                if (DEBUG)
391                                        System.out.println("Removed if" + temp);
392                                return true;
393                        } else if (temp instanceof ASTIfElseNode) {
394                                bodyContainingNode.remove(temp);
395                                if (returned.booleanValue()) {
396                                        // true so the if branch's body has to be added
397                                        // if its a labeled stmt we need a labeled block instead
398                                        // notice that its okkay to put a labeled block since other
399                                        // transformations might remove it
400                                        String label = ((ASTLabeledNode) temp).get_Label()
401                                                        .toString();
402                                        if (label != null) {
403                                                ASTLabeledBlockNode labeledNode = new ASTLabeledBlockNode(
404                                                                ((ASTLabeledNode) temp).get_Label(),
405                                                                (List<Object>) temp.get_SubBodies().get(0));
406                                                bodyContainingNode.add(index, labeledNode);
407                                        } else {
408                                                bodyContainingNode.addAll(index, (List) temp
409                                                                .get_SubBodies().get(0));
410                                        }
411                                } else {
412                                        // if its a labeled stmt we need a labeled block instead
413                                        // notice that its okkay to put a labeled block since other
414                                        // transformations might remove it
415                                        String label = ((ASTLabeledNode) temp).get_Label()
416                                                        .toString();
417                                        if (label != null) {
418                                                ASTLabeledBlockNode labeledNode = new ASTLabeledBlockNode(
419                                                                ((ASTLabeledNode) temp).get_Label(),
420                                                                (List<Object>) temp.get_SubBodies().get(1));
421                                                bodyContainingNode.add(index, labeledNode);
422                                        } else {
423                                                bodyContainingNode.addAll(index, (List) temp
424                                                                .get_SubBodies().get(1));
425                                        }
426                                }
427                                return true;
428                        } else if (temp instanceof ASTWhileNode
429                                        && returned.booleanValue() == false) {
430                                // notice we only remove if ASTWhileNode has false condition
431                                bodyContainingNode.remove(temp);
432                                return true;
433                        } else if (temp instanceof ASTDoWhileNode
434                                        && returned.booleanValue() == false) {
435                                // System.out.println("in try dowhile false");
436                                // remove the loop copy the body out since it gets executed once
437                                bodyContainingNode.remove(temp);
438                                bodyContainingNode.addAll(index, (List) temp.get_SubBodies()
439                                                .get(0));
440                                return true;
441                        } else if (temp instanceof ASTForLoopNode
442                                        && returned.booleanValue() == false) {
443                                bodyContainingNode.remove(temp);
444                                ASTStatementSequenceNode newNode = new ASTStatementSequenceNode(
445                                                ((ASTForLoopNode) temp).getInit());
446                                bodyContainingNode.add(index, newNode);
447                                return true;
448                        }
449                }
450                return false;
451        }

453        public void dealWithSwitchNode(ASTSwitchNode node) {
454                List<Object> indexList = node.getIndexList();
455                Map<Object, List<Object>> index2BodyList = node.getIndex2BodyList();
456
457                Iterator<Object> it = indexList.iterator();
458                while (it.hasNext()) {
459                        // going through all the cases of the switch statement
460                        Object currentIndex = it.next();
461                        List body = index2BodyList.get(currentIndex);
462
463                        if (body != null) {
464                                // this body is a list of ASTNodes
465
466                                Iterator itBody = body.iterator();
467                                Boolean returned = null;
468                                ASTNode temp = null;
469                                while (itBody.hasNext()) {
470                                        temp = (ASTNode) itBody.next();
471                                        if (temp instanceof ASTControlFlowNode) {
472                                                bodyContainingNode = null;
473                                                returned = eliminate(temp);
474                                                if (returned != null && canChange(returned, temp))
475                                                        break;
476                                                else
477                                                        bodyContainingNode = null;
478                                        }
479                                        temp.apply(this);
480                                }
481                                boolean changed = change(returned, temp);
482                                if (changed)
483                                        modified = true;
484                        }// end while changed
485                }
486        }
