043/*
044 THE ALGORITHM USES THE RESULTS OF REACHINGDEFS STRUCTURAL FLOW ANALYSIS
045
046 DEFINITION uD Chain:
047 For a use of variable x, the uD Chain gives ALL POSSIBLE definitions of x that can reach the use x
048
049 DEFINITION dU Chain:
050 For a definition d, the dU Chain gives all places where this definition is used
051
052
053 Need to be very clear when a local can be used
054 It can be used in the following places:
055 a, a conditional in if, ifelse, while , do while, for condition
056 b, in the for init or update
057 c, in a switch choice
058 d, in a syncrhnoized block 
059 d, in a statement
060
061 */
062public class ASTUsesAndDefs extends DepthFirstAdapter {
063        public static boolean DEBUG = false;
064        HashMap<Object, List<DefinitionStmt>> uD; // mapping a use to all possible
065                                                                                                // definitions
066        HashMap<Object, List> dU; // mapping a def to all possible uses
067        ReachingDefs reaching; // using structural analysis information
068
069        public ASTUsesAndDefs(ASTNode AST) {
070                uD = new HashMap<Object, List<DefinitionStmt>>();
071                dU = new HashMap<Object, List>();
072                reaching = new ReachingDefs(AST);
073        }
074
075        public ASTUsesAndDefs(boolean verbose, ASTNode AST) {
076                super(verbose);
077                uD = new HashMap<Object, List<DefinitionStmt>>();
078                dU = new HashMap<Object, List>();
079                reaching = new ReachingDefs(AST);
080        }

082        /*
083         * Method is used to strip away boxes from the actual values only those are
084         * returned which are locals
085         */
086        private List<Value> getUsesFromBoxes(List useBoxes) {
087                ArrayList<Value> toReturn = new ArrayList<Value>();
088                Iterator it = useBoxes.iterator();
089                while (it.hasNext()) {
090                        Value val = ((ValueBox) it.next()).getValue();
091                        if (val instanceof Local)
092                                toReturn.add(val);
093                }
094                // System.out.println("VALUES:"+toReturn);
095                return toReturn;
096        }
097
098        public void checkStatementUses(Stmt s, Object useNodeOrStatement) {
099                List useBoxes = s.getUseBoxes();
100                // System.out.println("Uses in this statement:"+useBoxes);
101                List<Value> uses = getUsesFromBoxes(useBoxes);
102                // System.out.println("Local Uses in this statement:"+uses);
103
104                Iterator<Value> it = uses.iterator();
105                while (it.hasNext()) {
106                        Local local = (Local) it.next();
107                        createUDDUChain(local, useNodeOrStatement);
108                }// end of going through all locals uses in statement
109
110                /*
111                 * see if this is a def stmt in which case add an empty entry into the
112                 * dU chain
113                 * 
114                 * The wisdowm behind this is that later on when this definition is used
115                 * we will use this arraylist to store the uses of this definition
116                 */
117                if (s instanceof DefinitionStmt) {
118                        // check if dU doesnt already have something for this
119                        if (dU.get(s) == null) {
120                                dU.put(s, new ArrayList());
121                        }
122                }
123        }
124
125        /*
126         * The method gets the reaching defs of local used Then all the possible
127         * defs are added into the uD chain of the node The use is added to all the
128         * defs reaching this node
129         */
130        public void createUDDUChain(Local local, Object useNodeOrStatement) {
131                // System.out.println("Local is:"+local);
132                // System.out.println("useNodeOrStatement is"+useNodeOrStatement);
133
134                List<DefinitionStmt> reachingDefs = reaching.getReachingDefs(local,
135                                useNodeOrStatement);
136                if (DEBUG)
137                        System.out.println("Reaching def for:" + local + " are:"
138                                        + reachingDefs);
139
140                // add the reaching defs into the use def chain
141                Object tempObj = uD.get(useNodeOrStatement);
142                if (tempObj != null) {
143                        List<DefinitionStmt> tempList = (List<DefinitionStmt>) tempObj;
144                        tempList.addAll(reachingDefs);
145                        uD.put(useNodeOrStatement, tempList);
146                } else
147                        uD.put(useNodeOrStatement, reachingDefs);
148
149                // add the use into the def use chain
150                Iterator<DefinitionStmt> defIt = reachingDefs.iterator();
151                while (defIt.hasNext()) {
152                        // for each reaching def
153                        Object defStmt = defIt.next();
154
155                        // get the dU Chain
156                        Object useObj = dU.get(defStmt);
157                        List<Object> uses = null;
158                        if (useObj == null)
159                                uses = new ArrayList<Object>();
160                        else
161                                uses = (List<Object>) useObj;
162
163                        // add the new local use to this list (we add the node since thats
164                        // where the local is used
165                        uses.add(useNodeOrStatement);
166                        // System.out.println("Adding definition:"+defStmt+"with uses:"+uses);
167                        dU.put(defStmt, uses);
168                }
169        }

171        /*
172         * Given a unary/binary or aggregated condition this method is used to find
173         * the locals used in the condition
174         */
175        public List<Value> getUseList(ASTCondition cond) {
176                ArrayList<Value> useList = new ArrayList<Value>();
177                if (cond instanceof ASTAggregatedCondition) {
178                        useList.addAll(getUseList(((ASTAggregatedCondition) cond)
179                                        .getLeftOp()));
180                        useList.addAll(getUseList(((ASTAggregatedCondition) cond)
181                                        .getRightOp()));
182                        return useList;
183                } else if (cond instanceof ASTUnaryCondition) {
184                        // get uses from unary condition
185                        List<Value> uses = new ArrayList<Value>();
186
187                        Value val = ((ASTUnaryCondition) cond).getValue();
188                        if (val instanceof Local) {
189                                if (DEBUG)
190                                        System.out
191                                                        .println("adding local from unary condition as a use"
192                                                                        + val);
193                                uses.add(val);
194                        } else {
195                                List useBoxes = val.getUseBoxes();
196                                uses = getUsesFromBoxes(useBoxes);
197                        }
198                        return uses;
199                } else if (cond instanceof ASTBinaryCondition) {
200                        // get uses from binaryCondition
201                        Value val = ((ASTBinaryCondition) cond).getConditionExpr();
202                        List useBoxes = val.getUseBoxes();
203                        return getUsesFromBoxes(useBoxes);
204                } else {
205                        throw new RuntimeException(
206                                        "Method getUseList in ASTUsesAndDefs encountered unknown condition type");
207                }
208        }

210        /*
211         * This method gets a list of all uses of locals in the condition Then it
212         * invokes the createUDDUChain for each local
213         */
214        public void checkConditionalUses(ASTCondition cond, ASTNode node) {
215                List<Value> useList = getUseList(cond);
216
217                // System.out.println("FOR NODE with condition:"+cond+"USE list is:"+useList);
218
219                // FOR EACH USE
220                Iterator<Value> it = useList.iterator();
221                while (it.hasNext()) {
222                        Local local = (Local) it.next();
223                        // System.out.println("creating uddu for "+local);
224                        createUDDUChain(local, node);
225                }// end of going through all locals uses in condition
226        }

228        /*
229         * The key in a switch stmt can be a local or a value which can contain
230         * Locals
231         * 
232         * Hence the some what indirect approach
233         */
234        public void inASTSwitchNode(ASTSwitchNode node) {
235                Value val = node.get_Key();
236                List<Value> uses = new ArrayList<Value>();
237                if (val instanceof Local) {
238                        uses.add(val);
239                } else {
240                        List useBoxes = val.getUseBoxes();
241                        uses = getUsesFromBoxes(useBoxes);
242                }
243
244                Iterator<Value> it = uses.iterator();
245                // System.out.println("SWITCH uses start:");
246                while (it.hasNext()) {
247                        Local local = (Local) it.next();
248                        // System.out.println(local);
249                        createUDDUChain(local, node);
250                }// end of going through all locals uses in switch key
251                        // System.out.println("SWITCH uses end:");
252        }

254        public void inASTSynchronizedBlockNode(ASTSynchronizedBlockNode node) {
255                Local local = node.getLocal();
256                createUDDUChain(local, node);
257        }

291        /*
292         * The init of a for loop can use a local The condition of a for node can
293         * use a local The update in a for loop can use a local
294         */
295        public void inASTForLoopNode(ASTForLoopNode node) {
296
297                // checking uses in init
298                for (AugmentedStmt as : node.getInit()) {
299                        Stmt s = as.get_Stmt();
300                        checkStatementUses(s, node);
301                }
302
303                // checking uses in condition
304                ASTCondition cond = node.get_Condition();
305                checkConditionalUses(cond, node);
306
307                // checking uses in update
308                for (AugmentedStmt as : node.getUpdate()) {
309                        Stmt s = as.get_Stmt();
310                        checkStatementUses(s, node);
311                }
312        }
313
314        public void inASTStatementSequenceNode(ASTStatementSequenceNode node) {
315                for (AugmentedStmt as : node.getStatements()) {
316                        Stmt s = as.get_Stmt();
317                        // in the case of stmtts in a stmtt sequence each stmt is considered
318                        // an entity
319                        // compared to the case where these stmts occur within other
320                        // constructs
321                        // where the node is the entity
322                        checkStatementUses(s, s);
323                }
324        }

353        public void print() {
354                System.out
355                                .println("\n\n\nPRINTING uD dU CHAINS ______________________________");
356                Iterator<Object> it = dU.keySet().iterator();
357                while (it.hasNext()) {
358                        DefinitionStmt s = (DefinitionStmt) it.next();
359                        System.out.println("*****The def  " + s + " has following uses:");
360                        Object obj = dU.get(s);
361                        if (obj != null) {
362                                ArrayList list = (ArrayList) obj;
363                                Iterator tempIt = list.iterator();
364                                while (tempIt.hasNext()) {
365                                        Object tempUse = tempIt.next();
366                                        System.out.println("-----------Use  " + tempUse);
367                                        System.out.println("----------------Defs of this use:   "
368                                                        + uD.get(tempUse));
369                                }
370                        }
371                }
372                System.out
373                                .println("END --------PRINTING uD dU CHAINS ______________________________");
374        }
