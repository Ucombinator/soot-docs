051/*
052 * TODO: shouldnt this be a transformation and hence under the transformation package???
053 This analysis uses the results from
054 1 ReachingCopies 
055 2 uD and dU chains 
056 to eliminate extra copies
057
058
059 ALGORITHM:
060 When you encounter a copy stmt (a=b) find all uses of local a (using dU chain)
061
062 if For ALL uses the ReachingCopies set contains copy stmt (a=b)
063 Remove Copy Stmt
064 Replace use of a with use of b
065
066 Note:
067 copy stmts can be encountered in:
068 a, ASTStatementSequenceNode
069 b, for loop init      -------> dont want to remove this
070 c, for loop update   ---------> dont want to remove this
071
072 */
073

108        private void setup() {
109                // create the uD and dU chains
110                if (DEBUG)
111                        System.out.println("computing usesAnd Defs");
112                useDefs = new ASTUsesAndDefs(AST);
113                AST.apply(useDefs);
114
115                if (DEBUG)
116                        System.out.println("computing usesAnd Defs....done");
117
118                // apply the reaching copies Structural flow Analysis
119                reachingCopies = new ReachingCopies(AST);
120
121                parentOf = new ASTParentNodeFinder();
122                AST.apply(parentOf);
123        }

143        public void inASTStatementSequenceNode(ASTStatementSequenceNode node) {
144                for (AugmentedStmt as : node.getStatements()) {
145                        Stmt s = as.get_Stmt();
146                        if (isCopyStmt(s)) {
147                                handleCopyStmt((DefinitionStmt) s);
148                        }
149                }
150        }

152        /*
153         * Method returns true if s is a copy stmt i.e. a = b where a and b are both
154         * locals
155         */
156        public boolean isCopyStmt(Stmt s) {
157                if (!(s instanceof DefinitionStmt)) {
158                        // only definition stmts can be copy stmts
159                        return false;
160                }
161
162                // x = expr;
163                // check if expr is a local in which case this is a copy
164                Value leftOp = ((DefinitionStmt) s).getLeftOp();
165                Value rightOp = ((DefinitionStmt) s).getRightOp();
166
167                if (leftOp instanceof Local && rightOp instanceof Local) {
168                        // this is a copy statement
169                        return true;
170                }
171                return false;
172        }

174        /*
175         * Given a copy stmt (a=b) find all uses of local a (using dU chain)
176         * 
177         * if For ALL uses the ReachingCopies set contains copy stmt (a=b) Remove
178         * Copy Stmt Replace use of a with use of b
179         */
180        public void handleCopyStmt(DefinitionStmt copyStmt) {
181                // System.out.println("COPY STMT FOUND-----------------------------------"+copyStmt);
182
183                // get defined local...safe to cast since this is copyStmt
184                Local definedLocal = (Local) copyStmt.getLeftOp();
185
186                // get all uses of this local from the dU chain
187                Object temp = useDefs.getDUChain(copyStmt);
188
189                ArrayList uses = new ArrayList();
190                if (temp != null) {
191                        uses = (ArrayList) temp;
192                }
193
194                // the uses list contains all stmts / nodes which use the definedLocal
195
196                // check if uses is non-empty
197                if (uses.size() != 0) {
198                        if (DEBUG) {
199                                System.out.println(">>>>The defined local:" + definedLocal
200                                                + " is used in the following");
201                                System.out.println("\n numof uses:" + uses.size() + uses
202                                                + ">>>>>>>>>>>>>>>\n\n");
203                        }
204
205                        // continuing with copy propagation algorithm
206
207                        // create localPair for copy stmt in question...same to cast as its
208                        // a copyStmt
209                        Local leftLocal = (Local) copyStmt.getLeftOp();
210                        Local rightLocal = (Local) copyStmt.getRightOp();
211
212                        ReachingCopies.LocalPair localPair = reachingCopies.new LocalPair(
213                                        leftLocal, rightLocal);
214
215                        // check for all the non zero uses
216                        Iterator useIt = uses.iterator();
217                        while (useIt.hasNext()) {
218                                // check that the reaching copies of each use has the copy stmt
219                                // a use is either a statement or a node(condition, synch,
220                                // switch , for etc)
221                                Object tempUse = useIt.next();
222
223                                DavaFlowSet reaching = reachingCopies
224                                                .getReachingCopies(tempUse);
225
226                                if (!reaching.contains(localPair)) {
227                                        // this copy stmt does not reach this use
228                                        // no copy elimination can be done
229                                        return;
230                                }
231                        }
232
233                        // if we get here that means that the copy stmt reached each use
234
235                        // replace each use of a with b
236                        useIt = uses.iterator();
237                        while (useIt.hasNext()) {
238                                Object tempUse = useIt.next();
239                                if (DEBUG) {
240                                        System.out.println("copy stmt reached this use" + tempUse);
241                                }
242                                replace(leftLocal, rightLocal, tempUse);
243                        }
244
245                        // remove copy stmt a=b
246                        removeStmt(copyStmt);
247
248                        if (someCopyStmtModified) {
249                                // get all the analyses re-set
250                                setup();
251                                someCopyStmtModified = false;
252                        }
253                } else {
254                        // the copy stmt is usesless since the definedLocal is not being
255                        // used anywhere after definition
256                        // System.out.println("The defined local:"+definedLocal+" is not used anywhere");
257                        removeStmt(copyStmt);
258                }
259        }

261        public void removeStmt(Stmt stmt) {
262                Object tempParent = parentOf.getParentOf(stmt);
263                if (tempParent == null) {
264                        // System.out.println("NO PARENT FOUND CANT DO ANYTHING");
265                        return;
266                }
267
268                // parents are always ASTNodes, hence safe to cast
269                ASTNode parent = (ASTNode) tempParent;
270
271                // REMOVING STMT
272                if (!(parent instanceof ASTStatementSequenceNode)) {
273                        // parent of a statement should always be a ASTStatementSequenceNode
274                        throw new RuntimeException(
275                                        "Found a stmt whose parent is not an ASTStatementSequenceNode");
276                }
277                ASTStatementSequenceNode parentNode = (ASTStatementSequenceNode) parent;
278
279                ArrayList<AugmentedStmt> newSequence = new ArrayList<AugmentedStmt>();
280
281                for (AugmentedStmt as : parentNode.getStatements()) {
282                        Stmt s = as.get_Stmt();
283                        if (s.toString().compareTo(stmt.toString()) != 0) {
284                                // this is not the stmt to be removed
285                                newSequence.add(as);
286                        }
287                }
288                // System.out.println("STMT REMOVED---------------->"+stmt);
289                parentNode.setStatements(newSequence);
290
291                ASTMODIFIED = true;
292                return;
293        }

295        /*
296         * Method goes depth first into the condition tree and finds any use of
297         * local "from" If it finds the use it replaces it with the use of local
298         * "to"
299         */
300        public void modifyUses(Local from, Local to, ASTCondition cond) {
301                if (cond instanceof ASTAggregatedCondition) {
302                        // ArrayList useList = new ArrayList();
303                        modifyUses(from, to, ((ASTAggregatedCondition) cond).getLeftOp());
304                        modifyUses(from, to, ((ASTAggregatedCondition) cond).getRightOp());
305
306                } else if (cond instanceof ASTUnaryCondition) {
307                        // get uses from unary condition
308                        Value val = ((ASTUnaryCondition) cond).getValue();
309                        if (val instanceof Local) {
310                                Local local = (Local) val;
311                                if (local.getName().compareTo(from.getName()) == 0) {
312                                        // replace the name with the one in "to"
313                                        ((ASTUnaryCondition) cond).setValue(to);
314                                        ASTMODIFIED = true;
315                                }
316                        } else {
317                                Iterator it = val.getUseBoxes().iterator();
318                                while (it.hasNext()) {
319                                        ValueBox valBox = (ValueBox) it.next();
320                                        Value tempVal = valBox.getValue();
321                                        if (tempVal instanceof Local) {
322                                                Local local = (Local) tempVal;
323                                                if (local.getName().compareTo(from.getName()) == 0) {
324                                                        // replace the name with the one in "to"
325                                                        valBox.setValue(to);
326                                                        ASTMODIFIED = true;
327                                                }
328                                        }
329                                }
330                        }
331                }// end unary condition
332                else if (cond instanceof ASTBinaryCondition) {
333                        // get uses from binaryCondition
334                        Value val = ((ASTBinaryCondition) cond).getConditionExpr();
335                        Iterator it = val.getUseBoxes().iterator();
336                        while (it.hasNext()) {
337                                ValueBox valBox = (ValueBox) it.next();
338                                Value tempVal = valBox.getValue();
339                                if (tempVal instanceof Local) {
340                                        Local local = (Local) tempVal;
341                                        if (local.getName().compareTo(from.getName()) == 0) {
342                                                // replace the name with the one in "to"
343                                                valBox.setValue(to);
344                                                ASTMODIFIED = true;
345                                        }
346                                }
347                        }
348                } else {
349                        throw new RuntimeException(
350                                        "Method getUseList in CopyPropagation encountered unknown condition type");
351                }
352        }

354        public void modifyUseBoxes(Local from, Local to, List useBoxes) {
355
356                Iterator it = useBoxes.iterator();
357                while (it.hasNext()) {
358                        ValueBox valBox = (ValueBox) it.next();
359                        Value tempVal = valBox.getValue();
360                        if (tempVal instanceof Local) {
361                                Local local = (Local) tempVal;
362                                if (local.getName().compareTo(from.getName()) == 0) {
363                                        // replace the name with the one in "to"
364                                        valBox.setValue(to);
365                                        ASTMODIFIED = true;
366                                }
367                        }
368                }
369        }

371        /*
372         * Invoked by handleCopyStmt to replace the use of local <from> to the use
373         * of local <to> in <use>
374         * 
375         * Notice <use> can be a stmt or an ASTNode
376         */
377
378        public void replace(Local from, Local to, Object use) {
379                if (use instanceof Stmt) {
380                        Stmt s = (Stmt) use;
381                        if (isCopyStmt(s)) {
382                                someCopyStmtModified = true;
383                        }
384                        List useBoxes = s.getUseBoxes();
385                        if (DEBUG)
386                                System.out.println("Printing uses for stmt" + useBoxes);
387
388                        // TODO
389                        modifyUseBoxes(from, to, useBoxes);
390
391                } else if (use instanceof ASTNode) {
392                        if (use instanceof ASTSwitchNode) {
393                                ASTSwitchNode temp = (ASTSwitchNode) use;
394                                Value val = temp.get_Key();
395                                if (val instanceof Local) {
396                                        if (((Local) val).getName().compareTo(from.getName()) == 0) {
397                                                // replace the name with the one in "to"
398                                                ASTMODIFIED = true;
399                                                temp.set_Key(to);
400                                        }
401                                } else {
402                                        List useBoxes = val.getUseBoxes();
403                                        // TODO
404                                        modifyUseBoxes(from, to, useBoxes);
405                                }
406                        } else if (use instanceof ASTSynchronizedBlockNode) {
407                                ASTSynchronizedBlockNode temp = (ASTSynchronizedBlockNode) use;
408                                Local local = temp.getLocal();
409                                if (local.getName().compareTo(from.getName()) == 0) {
410                                        // replace the name with the one in "to"
411                                        temp.setLocal(to);
412                                        ASTMODIFIED = true;
413                                }
414                        } else if (use instanceof ASTIfNode) {
415                                if (DEBUG)
416                                        System.out.println("Use is an instanceof if node");
417
418                                ASTIfNode temp = (ASTIfNode) use;
419                                ASTCondition cond = temp.get_Condition();
420                                modifyUses(from, to, cond);
421                        } else if (use instanceof ASTIfElseNode) {
422                                ASTIfElseNode temp = (ASTIfElseNode) use;
423                                ASTCondition cond = temp.get_Condition();
424                                modifyUses(from, to, cond);
425                        } else if (use instanceof ASTWhileNode) {
426                                ASTWhileNode temp = (ASTWhileNode) use;
427                                ASTCondition cond = temp.get_Condition();
428                                modifyUses(from, to, cond);
429                        } else if (use instanceof ASTDoWhileNode) {
430                                ASTDoWhileNode temp = (ASTDoWhileNode) use;
431                                ASTCondition cond = temp.get_Condition();
432                                modifyUses(from, to, cond);
433                        } else if (use instanceof ASTForLoopNode) {
434                                ASTForLoopNode temp = (ASTForLoopNode) use;
435
436                                // init
437                                for (AugmentedStmt as : temp.getInit()) {
438                                        Stmt s = as.get_Stmt();
439                                        replace(from, to, s);
440                                }
441
442                                // update
443                                for (AugmentedStmt as : temp.getUpdate()) {
444                                        Stmt s = as.get_Stmt();
445                                        replace(from, to, s);
446                                }
447
448                                // condition
449                                ASTCondition cond = temp.get_Condition();
450                                modifyUses(from, to, cond);
451                        } else
452                                throw new RuntimeException(
453                                                "Encountered an unknown ASTNode in copyPropagation method replace");
454                } else
455                        throw new RuntimeException(
456                                        "Encountered an unknown use in copyPropagation method replace");
457
458        }
