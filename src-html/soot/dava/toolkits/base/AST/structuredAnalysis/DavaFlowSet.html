147        @Override
148        public void add(T e) {
149                /* Expand only if necessary! and removes one if too:) */
150
151                // Add element
152                if (!contains(e)) {
153                        // Expand array if necessary
154                        if (numElements == maxElements)
155                                doubleCapacity();
156                        elements[numElements++] = e;
157                }

160        private void doubleCapacity() {
161                int newSize = maxElements * 2;
162
163                @SuppressWarnings("unchecked")
164                T[] newElements = (T[]) new Object[newSize];
165
166                System.arraycopy(elements, 0, newElements, 0, numElements);
167                elements = newElements;
168                maxElements = newSize;
169        }

184        /**
185         * Notice that the union method only merges the elements of the flow set
186         * DavaFlowSet also contains information regarding abrupt control flow This
187         * should also be merged using the copyInternalDataFrom method
188         */
189        public void union(FlowSet<T> otherFlow, FlowSet<T> destFlow) {
190                if (sameType(otherFlow) && sameType(destFlow)) {
191                        DavaFlowSet<T> other = (DavaFlowSet<T>) otherFlow;
192                        DavaFlowSet<T> dest = (DavaFlowSet<T>) destFlow;
193
194                        // For the special case that dest == other
195                        if (dest == other) {
196                                for (int i = 0; i < this.numElements; i++)
197                                        dest.add(this.elements[i]);
198                        }
199
200                        // Else, force that dest starts with contents of this
201                        else {
202                                if (this != dest)
203                                        copy(dest);
204
205                                for (int i = 0; i < other.numElements; i++)
206                                        dest.add(other.elements[i]);
207                        }
208                } else
209                        super.union(otherFlow, destFlow);
210        }

212        /**
213         * Notice that the intersection method only merges the elements of the flow
214         * set DavaFlowSet also contains information regarding abrupt control flow
215         * This should also be merged using the copyInternalDataFrom method
216         */
217        public void intersection(FlowSet<T> otherFlow, FlowSet<T> destFlow) {
218                // System.out.println("DAVA FLOWSET INTERSECTION INVOKED!!!");
219                if (sameType(otherFlow) && sameType(destFlow)) {
220                        DavaFlowSet<T> other = (DavaFlowSet<T>) otherFlow;
221                        DavaFlowSet<T> dest = (DavaFlowSet<T>) destFlow;
222                        DavaFlowSet<T> workingSet;
223
224                        if (dest == other || dest == this)
225                                workingSet = new DavaFlowSet<T>();
226                        else {
227                                workingSet = dest;
228                                workingSet.clear();
229                        }
230
231                        for (int i = 0; i < this.numElements; i++) {
232                                if (other.contains(this.elements[i]))
233                                        workingSet.add(this.elements[i]);
234                        }
235
236                        if (workingSet != dest)
237                                workingSet.copy(dest);
238                } else
239                        super.intersection(otherFlow, destFlow);
240        }

243                if (sameType(otherFlow) && sameType(destFlow)) {
244                        DavaFlowSet<T> other = (DavaFlowSet<T>) otherFlow;
245                        DavaFlowSet<T> dest = (DavaFlowSet<T>) destFlow;
246                        DavaFlowSet<T> workingSet;
247
248                        if (dest == other || dest == this)
249                                workingSet = new DavaFlowSet<T>();
250                        else {
251                                workingSet = dest;
252                                workingSet.clear();
253                        }
254
255                        for (int i = 0; i < this.numElements; i++) {
256                                if (!other.contains(this.elements[i]))
257                                        workingSet.add(this.elements[i]);
258                        }
259
260                        if (workingSet != dest)
261                                workingSet.copy(dest);
262                } else
263                        super.difference(otherFlow, destFlow);
264        }

274        /**
275         * Notice that the equals method only checks the equality of the elements of
276         * the flow set DavaFlowSet also contains information regarding abrupt
277         * control flow This should also be checked by invoking the
278         * internalDataMatchesTo method
279         */
280        public boolean equals(Object otherFlow) {
281                if (sameType(otherFlow)) {
282                        @SuppressWarnings("unchecked")
283                        DavaFlowSet<T> other = (DavaFlowSet<T>) otherFlow;
284
285                        if (other.numElements != this.numElements)
286                                return false;
287
288                        int size = this.numElements;
289
290                        // Make sure that thisFlow is contained in otherFlow
291                        for (int i = 0; i < size; i++)
292                                if (!other.contains(this.elements[i]))
293                                        return false;
294
295                        /*
296                         * both arrays have the same size, no element appears twice in one
297                         * array, all elements of ThisFlow are in otherFlow -> they are
298                         * equal! we don't need to test again! // Make sure that otherFlow
299                         * is contained in ThisFlow for(int i = 0; i < size; i++)
300                         * if(!this.contains(other.elements[i])) return false;
301                         */
302
303                        return true;
304                } else
305                        return super.equals(otherFlow);
306        }

308        public void copy(FlowSet<T> destFlow) {
309                if (this == destFlow)
310                        return;
311                if (sameType(destFlow)) {
312                        DavaFlowSet<T> dest = (DavaFlowSet<T>) destFlow;
313
314                        while (dest.maxElements < this.maxElements)
315                                dest.doubleCapacity();
316
317                        dest.numElements = this.numElements;
318
319                        System.arraycopy(this.elements, 0, dest.elements, 0, this.numElements);
320                } else
321                        super.copy(destFlow);
322        }

324        /**
325         * A private method used to add an element into a List if it is NOT a
326         * duplicate
327         */
328        private List<DavaFlowSet<T>> addIfNotDuplicate(List<DavaFlowSet<T>> into,
329                        DavaFlowSet<T> addThis) {
330                // if set is not already present in the labelsBreakList then add it
331                Iterator<DavaFlowSet<T>> it = into.iterator();
332                boolean found = false;
333                while (it.hasNext()) {
334                        DavaFlowSet<T> temp = it.next();
335                        if (temp.equals(addThis) && temp.internalDataMatchesTo(addThis)) {
336                                found = true;
337                                break;
338                        }
339                }
340                if (!found)
341                        into.add(addThis);
342                return into;
343        }

345        /**
346         * When an explicit break statement is encountered this method should be
347         * called to store the current davaflowset
348         */
349        public void addToBreakList(String labelBroken, DavaFlowSet<T> set) {
350                List<DavaFlowSet<T>> labelsBreakList = breakList.get(labelBroken);
351                if (labelsBreakList == null) {
352                        labelsBreakList = new ArrayList<DavaFlowSet<T>>();
353                        labelsBreakList.add(set);
354                        breakList.put(labelBroken, labelsBreakList);
355                        // System.out.println("ADDED"+labelBroken+" with"+set.toString());
356                } else {
357                        // add set into this list if its not a duplicate and update the
358                        // hashMap
359                        breakList.put(labelBroken, addIfNotDuplicate(labelsBreakList, set));
360                }
361        }

363        /**
364         * When an explicit continue statement is encountered this method should be
365         * called to store the current davaflowset
366         */
367        public void addToContinueList(String labelContinued, DavaFlowSet<T> set) {
368                List<DavaFlowSet<T>> labelsContinueList = continueList.get(labelContinued);
369                if (labelsContinueList == null) {
370                        labelsContinueList = new ArrayList<DavaFlowSet<T>>();
371                        labelsContinueList.add(set);
372                        continueList.put(labelContinued, labelsContinueList);
373                } else {
374                        continueList.put(labelContinued, addIfNotDuplicate(labelsContinueList, set));
375                }
376        }

378        /**
379         * Checks whether the input stmt is an implicit break/continue A abrupt stmt
380         * is implicit if the SETLabelNode is null or the label.toString results in
381         * null
382         */
383        private boolean checkImplicit(DAbruptStmt ab) {
384                SETNodeLabel label = ab.getLabel();
385                if (label == null)
386                        return true;
387                if (label.toString() == null)
388                        return true;
389                return false;
390        }

392        /**
393         * The next two methods take an abruptStmt as input along with a flowSet. It
394         * should be only invoked for abrupt stmts which do not have explicit labels
395         * 
396         * The node being targeted by this implicit stmt should be found Then the
397         * flow set should be added to the list within the appropriate hashmap
398         */
399        public void addToImplicitBreaks(DAbruptStmt ab, DavaFlowSet<T> set) {
400                if (!checkImplicit(ab))
401                        throw new RuntimeException("Tried to add explicit break statement in the implicit list in");
402
403                if (!ab.is_Break())
404                        throw new RuntimeException(
405                                        "Tried to add continue statement in the break list in DavaFlowSet.addToImplicitBreaks");
406
407                // okkay so its an implicit break
408                // get the targetted node, use the ClosestAbruptTargetFinder
409                ASTNode node = ClosestAbruptTargetFinder.v().getTarget(ab);
410
411                // get the list of flow sets already stored for this node
412                List<DavaFlowSet<T>> listSets = implicitBreaks.get(node);
413                if (listSets == null)
414                        listSets = new ArrayList<DavaFlowSet<T>>();
415                
416                // if set is not already present in listSets add it and update hashMap
417                implicitBreaks.put(node, addIfNotDuplicate(listSets, set));
418        }

420        public void addToImplicitContinues(DAbruptStmt ab, DavaFlowSet<T> set) {
421                if (!checkImplicit(ab))
422                        throw new RuntimeException("Tried to add explicit continue statement in the implicit list ");
423
424                if (!ab.is_Continue())
425                        throw new RuntimeException("Tried to add break statement in the continue list");

427                // okkay so its an implicit continue
428                // get the targetted node, use the ClosestAbruptTargetFinder

435                
436                // if set is not already present in listSets add it and update hashMap
437                implicitContinues.put(node, addIfNotDuplicate(listSets, set));
438        }

470        /**
471         * An internal method used to copy non-duplicate entries from the temp list
472         * into the currentList
473         */
474        private List<DavaFlowSet<T>> copyDavaFlowSetList(List<DavaFlowSet<T>> currentList,
475                        List<DavaFlowSet<T>> temp) {
476                Iterator<DavaFlowSet<T>> tempIt = temp.iterator();
477                while (tempIt.hasNext()) {
478                        DavaFlowSet<T> check = tempIt.next();
479                        Iterator<DavaFlowSet<T>> currentListIt = currentList.iterator();
480                        boolean found = false;
481                        while (currentListIt.hasNext()) {
482                                // see if currentList has check
483                                DavaFlowSet<T> currentSet = currentListIt.next();
484                                if (check.equals(currentSet) && check.internalDataMatchesTo(currentSet)) {
485                                        found = true;
486                                        break;
487                                }
488                        }
489                        if (!found) {
490                                currentList.add(check);
491                        }
492                }
493                return currentList;
494        }

496        public void copyInternalDataFrom(DavaFlowSet<T> fromThis) {
497                if (!sameType(fromThis))
498                        return;
499
500                // copy elements of breaklist
501                {
502                        Map<Serializable, List<DavaFlowSet<T>>> fromThisBreakList = fromThis.getBreakList();
503
504                        Iterator<Serializable> keys = fromThisBreakList.keySet().iterator();
505                        while (keys.hasNext()) {
506                                String labelBroken = (String) keys.next();
507                                List<DavaFlowSet<T>> temp = fromThisBreakList.get(labelBroken);
508                                List<DavaFlowSet<T>> currentList = breakList.get(labelBroken);
509
510                                if (currentList == null) {
511                                        breakList.put(labelBroken, temp);
512                                } else {
513                                        List<DavaFlowSet<T>> complete = copyDavaFlowSetList(currentList, temp);
514                                        breakList.put(labelBroken, complete);
515                                }
516                        }
517                }
518
519                // copy elements of continuelist
520                {
521                        HashMap<Serializable, List<DavaFlowSet<T>>> fromThisContinueList = fromThis.getContinueList();
522
523                        Iterator<Serializable> keys = fromThisContinueList.keySet().iterator();
524                        while (keys.hasNext()) {
525                                String labelContinued = (String) keys.next();
526                                List<DavaFlowSet<T>> temp = fromThisContinueList.get(labelContinued);
527                                List<DavaFlowSet<T>> currentList = continueList.get(labelContinued);
528                                if (currentList == null) {
529                                        continueList.put(labelContinued, temp);
530                                } else {
531                                        List<DavaFlowSet<T>> complete = copyDavaFlowSetList(currentList, temp);
532                                        continueList.put(labelContinued, complete);
533                                }
534                        }
535                }
536
537                // copy elements of implicitBreaks
538                // this hashMap contains a mapping of ASTNodes to DavaFlowSets due to
539                // impicit breaks
540                {
541                        HashMap<Serializable, List<DavaFlowSet<T>>> copyThis = fromThis.getImplicitBreaks();
542                        Iterator<Serializable> it = copyThis.keySet().iterator();
543                        while (it.hasNext()) { // going through all nodes in the other
544                                                                        // objects implicitBreaks hashMap
545                                // each is a node
546                                ASTNode node = (ASTNode) it.next();
547                                // get list of dava flow sets targetting this node implicitly
548                                List<DavaFlowSet<T>> fromDavaFlowSets = copyThis.get(node);
549                                // Have copy non duplicates in this to the implicitbreak hashMap
550                                // the current dava flow set has
551
552                                List<DavaFlowSet<T>> toDavaFlowSets = implicitBreaks.get(node);
553                                if (toDavaFlowSets == null) {
554                                        // there was no dava flow set currently targetting this node
555                                        // implicitly
556                                        // put the fromDavaFlowSets into the hashMap
557                                        implicitBreaks.put(node, fromDavaFlowSets);
558                                } else {
559                                        List<DavaFlowSet<T>> complete = copyDavaFlowSetList(toDavaFlowSets, fromDavaFlowSets);
560                                        implicitBreaks.put(node, complete);
561                                }
562                        }
563                }
564
565                // copy elements of implicitContinues
566                // this hashMap contains a mapping of ASTNodes to DavaFlowSets due to
567                // impicit continues
568                {
569                        HashMap<Serializable, List<DavaFlowSet<T>>> copyThis = fromThis.getImplicitContinues();
570                        Iterator<Serializable> it = copyThis.keySet().iterator();
571                        while (it.hasNext()) { // going through all nodes in the other
572                                                                        // objects implicitcontinues hashMap
573                                // each is a node
574                                ASTNode node = (ASTNode) it.next();
575                                // get list of dava flow sets targetting this node implicitly
576                                List<DavaFlowSet<T>> fromDavaFlowSets = copyThis.get(node);
577                                // Have copy non duplicates in this to the implicitContinue
578                                // hashMap the current dava flow set has
579
580                                List<DavaFlowSet<T>> toDavaFlowSets = implicitContinues.get(node);
581                                if (toDavaFlowSets == null) {
582                                        // there was no dava flow set currently targetting this node
583                                        // implicitly
584                                        // put the fromDavaFlowSets into the hashMap
585                                        implicitContinues.put(node, fromDavaFlowSets);
586                                } else {
587                                        List<DavaFlowSet<T>> complete = copyDavaFlowSetList(toDavaFlowSets, fromDavaFlowSets);
588                                        implicitContinues.put(node, complete);
589                                }
590                        }
591
592                }
593
594        }

596        private <X> boolean compareLists(List<X> listOne, List<X> listTwo) {
597                if (listOne == null && listTwo == null)
598                        return true;
599                if (listOne == null || listTwo == null)
600                        return false;
601                
602                // compare elements of the list
603                if (listOne.size() != listTwo.size()) {
604                        // size has to be same for lists to match
605                        return false;
606                }
607                Iterator<X> listOneIt = listOne.iterator();
608                boolean found = false;
609                while (listOneIt.hasNext()) {
610                        // going through the first list
611                        Object listOneObj = listOneIt.next();
612
613                        Iterator<X> listTwoIt = listTwo.iterator();
614                        while (listTwoIt.hasNext()) {
615                                // find the object in the second list
616                                Object listTwoObj = listTwoIt.next();
617                                if (listOneObj.equals(listTwoObj)) {
618                                        // if object is found stop search
619                                        found = true;
620                                        break;
621                                }
622                        }
623                        if (!found) {
624                                // if didnt find object return false
625                                return false;
626                        }
627                        found = false;
628                }
629                return true;
630        }

632        public boolean internalDataMatchesTo(Object otherObj) {
633                if (!(otherObj instanceof DavaFlowSet))
634                        return false;
635
636                @SuppressWarnings("unchecked")
637                DavaFlowSet<T> other = (DavaFlowSet<T>) otherObj;
638
639                // check if same break list
640                HashMap<Serializable, List<DavaFlowSet<T>>> otherMap = other.getBreakList();
641                if (!compareHashMaps(breakList, otherMap))
642                        return false;
643
644                // check if same continue list
645                otherMap = other.getContinueList();
646                if (!compareHashMaps(continueList, otherMap))
647                        return false;
648
649                // check implicitBreaks match
650                otherMap = other.getImplicitBreaks();
651                if (!compareHashMaps(implicitBreaks, otherMap))
652                        return false;
653
654                // check implicitContinues match
655                otherMap = other.getImplicitContinues();
656                if (!compareHashMaps(implicitContinues, otherMap))
657                        return false;
658
659                return true;
660        }

662        private boolean compareHashMaps(HashMap<Serializable, List<DavaFlowSet<T>>> thisMap,
663                        HashMap<Serializable, List<DavaFlowSet<T>>> otherMap) {
664                List<String> otherKeyList = new ArrayList<String>();
665
666                Iterator<Serializable> keys = otherMap.keySet().iterator();
667                while (keys.hasNext()) {
668                        String otherKey = (String) keys.next();
669                        otherKeyList.add(otherKey);
670
671                        List<DavaFlowSet<T>> listOther = otherMap.get(otherKey);
672                        List<DavaFlowSet<T>> listThis = thisMap.get(otherKey);
673
674                        // compare the two lists
675                        if (!compareLists(listOther, listThis)) {
676                                // if lists dont match internalData doesnt match
677                                return false;
678                        }
679                }
680                // have gone through otherMap
681
682                // going through thisMap
683                keys = thisMap.keySet().iterator();
684                while (keys.hasNext()) {
685                        String key = (String) keys.next();
686
687                        Iterator<String> keyListIt = otherKeyList.iterator();
688                        boolean alreadyDone = false;
689
690                        while (keyListIt.hasNext()) {
691                                String doneKey = keyListIt.next();
692                                if (key.equals(doneKey)) {
693                                        alreadyDone = true;
694                                        break;
695                                }
696                        }
697                        if (!alreadyDone) {
698                                /*
699                                 * we have come across a label which was not done by the first
700                                 * hashmap meaning it was NOT in the first hashMap
701                                 */
702                                return false;
703                        }
704                }
705                return true;
706        }
