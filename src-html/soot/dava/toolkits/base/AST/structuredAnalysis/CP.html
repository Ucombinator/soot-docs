038public class CP extends StructuredAnalysis {
039        /*
040         * Constant Propagation:
041         * 
042         * Step 1: Sets of CPTuple (className, CPVariable, value) where: CPVariable
043         * contains a local or SootField
044         * 
045         * Step 2: A local or SootField has a constant value at a program point p if
046         * on all program paths from the start of the method to point p the local or
047         * Sootfield has only been assigned this constant value.
048         * 
049         * Step 3: Forward Analysis
050         * 
051         * Step 4: Intersection (See intersection method in CPFlowSet)
052         * 
053         * Step 5: See method processStatement
054         * 
055         * Step 6: out(start) = all locals set to bottom(they shouldnt be present in
056         * the initialSet all formals set to Top, all constant fields set to
057         * constant values
058         * 
059         * newInitialFlow: all locals and formals set to Top, all constant fields
060         * set to constant values remember new InitialFlow is ONLY used for input to
061         * catchBodies
062         * 
063         * Any local or field which is not in the flow set at any time is
064         * necessarily bottom
065         * 
066         * 
067         * knowing how a condition evaluates can give us useful insight to the
068         * values of variables Handle this for a == b where both a and b are
069         * primtype and special case handle it for A where A is a unary boolean
070         * condition
071         * 
072         * See following over-riden methods:
073         * 
074         * public Object processASTIfElseNode(ASTIfElseNode node,Object input)
075         * public Object processASTIfNode(ASTIfNode node,Object input) public Object
076         * processASTSwitchNode(ASTSwitchNode node,Object input)
077         */
078

097        public CP(ASTMethodNode analyze, HashMap<String, Object> constantFields,
098                        HashMap<String, SootField> classNameFieldNameToSootFieldMapping) {
099                super();
100                /*
101                 * DEBUG = true; DEBUG_IF = true; DEBUG_WHILE = true; DEBUG_STATEMENTS =
102                 * true;
103                 */
104                this.methodNode = analyze;
105                localClassName = analyze.getDavaBody().getMethod().getDeclaringClass().getName();
106
107                // Create a list of VariableValueTuples for all the constantFields
108                createConstantFieldsList(constantFields, classNameFieldNameToSootFieldMapping);
109
110                // Create the complete list of vars to go into constant propagation
111                createInitialInput();
112
113                // input to constant propagation should not be an empty flow set it is
114                // the set of all constant fields, locals assigned 0 and formals
115                // assigned T
116                CPFlowSet initialSet = new CPFlowSet();
117                Iterator<CPTuple> it = initialInput.iterator();
118                while (it.hasNext())
119                        initialSet.add(it.next());
120
121                // System.out.println("Initial set"+initialSet.toString());
122                CPFlowSet result = (CPFlowSet) process(analyze, initialSet);
123                // System.out.println("Last result :"+result.toString());
124        }

126        /*
127         * constant fields added with KNOWN CONSTANT VALUE formals added with TOP
128         * locals added with 0 other fields IGNORED
129         */
130        public void createInitialInput() {
131                initialInput = new ArrayList<CPTuple>();
132
133                // adding constant fields
134                initialInput.addAll(constantFieldTuples);
135
136                // String className =
137                // analyze.getDavaBody().getMethod().getDeclaringClass().getName();
138
139                // adding formals
140                formals = new ArrayList<CPTuple>();
141                // System.out.println("Adding following formals: with TOP");
142                Collection col = methodNode.getDavaBody().get_ParamMap().values();
143                Iterator it = col.iterator();
144                while (it.hasNext()) {
145                        Object temp = it.next();
146                        if (temp instanceof Local) {
147                                Local tempLocal = (Local) temp;
148                                if (!(tempLocal.getType() instanceof PrimType))
149                                        continue;
150
151                                CPVariable newVar = new CPVariable(tempLocal);
152
153                                // new tuple set to top since this is a formal and we dont know
154                                // what value we will get into it
155                                CPTuple newTuple = new CPTuple(localClassName, newVar, true);
156                                initialInput.add(newTuple);
157                                formals.add(newTuple);
158                                // System.out.print("\t"+tempLocal.getName());
159                        }
160                }
161                // System.out.println();
162
163                // adding locals
164                List decLocals = methodNode.getDeclaredLocals();
165                it = decLocals.iterator();
166                locals = new ArrayList<CPTuple>();
167                // System.out.println("Adding following locals with default values:");
168                while (it.hasNext()) {
169                        Object temp = it.next();
170                        if (temp instanceof Local) {
171                                Local tempLocal = (Local) temp;
172                                Type localType = tempLocal.getType();
173
174                                if (!(localType instanceof PrimType))
175                                        continue;
176
177                                CPVariable newVar = new CPVariable(tempLocal);
178
179                                // store the default value into this object
180                                Object value;
181
182                                // locals value is set to the default value that it can have
183                                // depending on its type
184                                if (localType instanceof BooleanType)
185                                        value = new Boolean(false);
186                                else if (localType instanceof ByteType)
187                                        value = new Integer(0);
188                                else if (localType instanceof CharType)
189                                        value = new Integer(0);
190                                else if (localType instanceof DoubleType)
191                                        value = new Double(0);
192                                else if (localType instanceof FloatType)
193                                        value = new Float(0);
194                                else if (localType instanceof IntType)
195                                        value = new Integer(0);
196                                else if (localType instanceof LongType)
197                                        value = new Long(0);
198                                else if (localType instanceof ShortType)
199                                        value = new Integer(0);
200                                else
201                                        throw new DavaFlowAnalysisException("Unknown PrimType");
202
203                                CPTuple newTuple = new CPTuple(localClassName, newVar, value);
204
205                                /*
206                                 * Commenting the next line since we dont want initial Input to
207                                 * have any locals in it all locals are considered bottom
208                                 * initially
209                                 */
210                                // initialInput.add(newTuple);
211
212                                locals.add(newTuple);
213                                // System.out.print("\t"+tempLocal.getName());
214                        }// was a local
215                }
216                // System.out.println();
217        }

219        /*
220         * Uses the results of the ConstantValueFinder to create a list of
221         * constantField CPTuple
222         */
223        private void createConstantFieldsList(HashMap<String, Object> constantFields,
224                        HashMap<String, SootField> classNameFieldNameToSootFieldMapping) {
225                constantFieldTuples = new ArrayList<CPTuple>();
226
227                Iterator<String> it = constantFields.keySet().iterator();
228                // System.out.println("Adding constant fields to initial set: ");
229                while (it.hasNext()) {
230                        String combined = it.next();
231
232                        int temp = combined.indexOf(ConstantFieldValueFinder.combiner, 0);
233                        if (temp > 0) {
234                                String className = combined.substring(0, temp);
235
236                                // String fieldName = combined.substring(temp+
237                                // ConstantFieldValueFinder.combiner.length());
238                                SootField field = classNameFieldNameToSootFieldMapping.get(combined);
239                                // String fieldName = field.getName();
240
241                                if (!(field.getType() instanceof PrimType)) {
242                                        // we only care about PrimTypes
243                                        continue;
244                                }
245
246                                // object type is double float long boolean or integer
247                                Object value = constantFields.get(combined);
248
249                                CPVariable var = new CPVariable(field);
250
251                                CPTuple newTuples = new CPTuple(className, var, value);
252                                constantFieldTuples.add(newTuples);
253                                // System.out.print("Class: "+className +
254                                // " Field: "+fieldName+" Value: "+value+"   ");
255                        } else {
256                                throw new DavaFlowAnalysisException("Second argument of VariableValuePair not a variable");
257                        }
258
259                }
260                // System.out.println("");
261        }

278        /*
279         * newInitialFlow is invoked for the input set of the catchBodies
280         * 
281         * formals initialized to top since we dont know what has happened so far in
282         * the body locals initialized to top since we dont know what has happened
283         * so far in the method body constant fields present with their constant
284         * value since that never changes
285         */
286        @Override
287        public DavaFlowSet newInitialFlow() {
288                CPFlowSet flowSet = new CPFlowSet();
289
290                // formals and locals should be both initialized to top since we dont
291                // know what has happened so far in the body
292                ArrayList<CPTuple> localsAndFormals = new ArrayList<CPTuple>();
293                localsAndFormals.addAll(formals);
294                localsAndFormals.addAll(locals);
295
296                Iterator<CPTuple> it = localsAndFormals.iterator();
297                while (it.hasNext()) {
298                        CPTuple tempTuple = (CPTuple) it.next().clone();
299
300                        // just making sure all are set to Top
301                        if (!tempTuple.isTop())
302                                tempTuple.setTop();
303
304                        flowSet.add(tempTuple);
305                }
306
307                // constant fields should be present with their constant value since
308                // that never changes
309                it = constantFieldTuples.iterator();
310                while (it.hasNext()) {
311                        flowSet.add(it.next());
312                }
313
314                return flowSet;
315        }

369        /*
370         * x = expr;
371         * 
372         * expr is a constant
373         * 
374         * epr is a local (Note we are not going to worry about fieldRefs)
375         * 
376         * expr is a Unary op with neg followed by a local or field
377         * 
378         * expr is actually exp1 op expr2 ..... ( x = x+1, y = i * 3, x = 3 *0
379         * 
380         * expr contains an invokeExpr
381         */
382        @Override
383        public DavaFlowSet processStatement(Stmt s, DavaFlowSet input) {
384                if (!(input instanceof CPFlowSet))
385                        throw new RuntimeException("processStatement is not implemented for other flowSet types");
386
387                CPFlowSet inSet = (CPFlowSet) input;
388                if (inSet == NOPATH)
389                        return inSet;
390
391                if (!(s instanceof DefinitionStmt))
392                        return inSet;
393
394                DefinitionStmt defStmt = (DefinitionStmt) s;
395                // x = expr;
396                // confirm that the left side is a local with a primitive type
397                Value left = defStmt.getLeftOp();
398                if (!(left instanceof Local && ((Local) left).getType() instanceof PrimType))
399                        return inSet;
400
401                // left is a primitive primitive local
402                CPFlowSet toReturn = (CPFlowSet) cloneFlowSet(inSet);
403
404                /*
405                 * KILL ANY PREVIOUS VALUE OF this local as this is an assignment
406                 * Remember the returned value can be null if the element was not found
407                 * or it was TOP
408                 */
409                Object killedValue = killButGetValueForUse((Local) left, toReturn);
410
411                Value right = defStmt.getRightOp();
412
413                Object value = CPHelper.isAConstantValue(right);
414                if (value != null) {
415                        // EXPR IS A CONSTANT
416                        if (left.getType() instanceof BooleanType) {
417                                Integer tempValue = (Integer) value;
418                                if (tempValue.intValue() == 0)
419                                        value = new Boolean(false);
420                                else
421                                        value = new Boolean(true);
422                        }
423                        addOrUpdate(toReturn, (Local) left, value);
424                } else {
425                        // EXPR IS NOT A CONSTANT
426                        handleMathematical(toReturn, (Local) left, right, killedValue);
427                }
428                return toReturn;
429        }

431        /*
432         * The returned value is the current constant associated with left.
433         * Integer/Long/Double/Float/Boolean The returned value can also be null if
434         * the element was not found or it was TOP
435         */
436        public Object killButGetValueForUse(Local left, CPFlowSet toReturn) {
437
438                for (CPTuple tempTuple : toReturn) {
439                        if (!(tempTuple.getSootClassName().equals(localClassName)))
440                                continue;
441
442                        // className is the same check if the variable is the same or not
443                        if (tempTuple.containsLocal()) { // remmeber the sets contain
444                                                                                                // constant fields also
445                                Local tempLocal = tempTuple.getVariable().getLocal();
446                                if (left.getName().equals(tempLocal.getName())) {
447                                        // KILL THIS SUCKA!!!
448                                        Object killedValue = tempTuple.getValue();
449                                        tempTuple.setTop();
450                                        return killedValue;
451                                }
452                        }
453                } // going through all elements of the flow set
454
455                // if this element was no where enter it with top
456                CPVariable newVar = new CPVariable(left);
457
458                // create the CPTuple
459                // System.out.println("trying to kill something which was not present so added with TOP");
460                CPTuple newTuple = new CPTuple(localClassName, newVar, false);
461                toReturn.add(newTuple);
462                return null;
463        }

469                CPVariable newVar = new CPVariable(left);

471                CPTuple newTuple = new CPTuple(localClassName, newVar, val);
472                toReturn.addIfNotPresent(newTuple);
473                // System.out.println("DefinitionStmt checked right expr for constants"+toReturn.toString());
474        }

476        /*
477         * x = b where b is in the before set of the statement as a constant then we
478         * can simply say x = that constant also
479         * 
480         * TODO: DONT WANT TO DO IT:::: If right expr is a unary expression see if
481         * the stuff inside is a Local
482         * 
483         * x = exp1 op exp2 (check if both exp1 and exp2 are int constants
484         * 
485         * killedValuse is either the constant value which left had before this
486         * assignment stmt or null if left was Top or not in the set
487         * 
488         * handle the special case when the inputset could not find a value because
489         * its the killed value //eg. x = x+1 since we top x first we will never get
490         * a match IMPORTANT
491         */
492        private void handleMathematical(CPFlowSet toReturn, Local left, Value right, Object killedValue) {
493
494                // if right expr is a local or field
495                Object value = isANotTopConstantInInputSet(toReturn, right);
496                if (value != null) {
497                        // right was a local or field with a value other than top
498                        // dont send value SEND A CLONE OF VALUE.....IMPORTANT!!!!
499                        Object toSend = CPHelper.wrapperClassCloner(value);
500
501                        if (toSend != null) {
502                                addOrUpdate(toReturn, left, toSend);
503                        }
504
505                        // return if value was != null as this means that the left was a
506                        // primitive local assigned some value from the right
507                        return;
508                }
509
510                // if we get here we know that right is not a local or field whose value
511                // we could find in the set
512                if (right instanceof BinopExpr) {
513                        Value op1 = ((BinopExpr) right).getOp1();
514                        Value op2 = ((BinopExpr) right).getOp2();
515
516                        Object op1Val = CPHelper.isAConstantValue(op1);
517                        Object op2Val = CPHelper.isAConstantValue(op2);
518
519                        if (op1Val == null)
520                                op1Val = isANotTopConstantInInputSet(toReturn, op1);
521
522                        if (op2Val == null)
523                                op2Val = isANotTopConstantInInputSet(toReturn, op2);
524
525                        if (op1 == left) {
526                                // System.out.println("\n\n>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>OP1 is the same as LHS");
527                                op1Val = killedValue;
528                        }
529
530                        if (op2 == left) {
531                                // System.out.println("\n\n>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>>OP2 is the same as LHS");
532                                op2Val = killedValue;
533                        }
534
535                        if (op1Val != null && op2Val != null) {
536                                // System.out.println("found constant values for both operands of binary expression");
537                                if (left.getType() instanceof IntType && op1Val instanceof Integer && op2Val instanceof Integer) {
538                                        // only caring about operations on two integers and result
539                                        // is an integer
540
541                                        int op1IntValue = ((Integer) op1Val).intValue();
542                                        int op2IntValue = ((Integer) op2Val).intValue();
543
544                                        String tempStr = ((BinopExpr) right).getSymbol();
545                                        if (tempStr.length() > 1) {
546                                                char symbol = tempStr.charAt(1);
547                                                // System.out.println("found symbol "+symbol+" for the operands of binary expression");
548                                                int newValue = 0;
549                                                boolean set = false;
550                                                switch (symbol) {
551                                                case '+':
552                                                        // System.out.println("Adding");
553                                                        newValue = op1IntValue + op2IntValue;
554                                                        set = true;
555                                                        break;
556                                                case '-':
557                                                        // System.out.println("Subtracting");
558                                                        newValue = op1IntValue - op2IntValue;
559                                                        set = true;
560
561                                                        break;
562
563                                                case '*':
564                                                        // System.out.println("Multiplying");
565                                                        newValue = op1IntValue * op2IntValue;
566                                                        set = true;
567
568                                                        break;
569                                                }
570
571                                                if (set) {
572                                                        // we have our new value
573                                                        Integer newValueObject = new Integer(newValue);
574                                                        addOrUpdate(toReturn, left, newValueObject);
575                                                        return;
576                                                }
577                                        }
578                                }
579                        } else {
580                                // System.out.println("atleast one value is not constant so cant simplify expression");
581                        }
582                }
583                // System.out.println("DefinitionStmt checked right expr for mathematical stuff"+toReturn.toString());
584        }

586        /*
587         * Check whether it is a local or field which has a constant value (could be
588         * top) in the current inSet
589         * 
590         * The method returns null if its not found or its TOP Otherwise it will
591         * return the constant value
592         */
593        private Object isANotTopConstantInInputSet(CPFlowSet set, Value toCheck) {
594                if (toCheck instanceof Local || toCheck instanceof FieldRef) {
595                        String toCheckClassName = null;
596                        if (toCheck instanceof Local)
597                                toCheckClassName = localClassName;
598                        else
599                                toCheckClassName = ((FieldRef) toCheck).getField().getDeclaringClass().getName();
600
601                        for (CPTuple tempTuple : set) {
602                                // check that the classNames are the same
603                                if (!(tempTuple.getSootClassName().equals(toCheckClassName))) {
604                                        // classNames are not the same no point in continuing with
605                                        // checks
606                                        continue;
607                                }
608
609                                boolean tupleFound = false;
610                                if (tempTuple.containsLocal() && toCheck instanceof Local) {
611                                        // check they are the same Local
612                                        Local tempLocal = tempTuple.getVariable().getLocal();
613                                        if (tempLocal.getName().equals(((Local) toCheck).getName())) {
614                                                // the set does have a constant value for this local
615                                                tupleFound = true;
616                                        }
617                                } else if (tempTuple.containsField() && toCheck instanceof FieldRef) {
618                                        SootField toCheckField = ((FieldRef) toCheck).getField();
619                                        SootField tempField = tempTuple.getVariable().getSootField();
620
621                                        if (tempField.getName().equals(toCheckField.getName())) {
622                                                // the set contains a constant value for this field
623                                                tupleFound = true;
624                                        }
625                                }
626
627                                if (tupleFound) {
628                                        if (tempTuple.isTop())
629                                                return null;
630                                        else
631                                                return tempTuple.getValue();
632                                }
633                        }
634
635                }
636                return null;
637        }

639        /*
640         * over-riding the StructuredFlow Analysis implementation because we want to
641         * be able to gather information about the truth value in the condition
642         */
643        @Override
644        public DavaFlowSet processASTIfNode(ASTIfNode node, DavaFlowSet input) {
645                if (DEBUG_IF)
646                        System.out.println("Processing if node using over-ridden process if method" + input.toString());
647                ;
648
649                input = processCondition(node.get_Condition(), input);
650
651                if (!(input instanceof CPFlowSet)) {
652                        throw new DavaFlowAnalysisException("not a flow set");
653                }
654
655                CPFlowSet inputToBody = ((CPFlowSet) input).clone();
656
657                CPTuple tuple = checkForValueHints(node.get_Condition(), inputToBody, false);
658
659                if (tuple != null) {
660                        // if not null, is a belief going into the if branch simply add it
661                        // into the input set
662                        // System.out.println(">>>>>Adding tuple because of condition"+tuple.toString());
663                        inputToBody.addIfNotPresentButDontUpdate(tuple);
664                }
665
666                DavaFlowSet output1 = processSingleSubBodyNode(node, inputToBody);
667
668                if (DEBUG_IF)
669                        System.out.println("\n\nINPUTS TO MERGE ARE input (original):" + input.toString()
670                                        + "processingBody output:" + output1.toString() + "\n\n\n");
671
672                // merge with input which tells if the cond did not evaluate to true
673                DavaFlowSet output2 = merge(input, output1);
674
675                // handle break
676                String label = getLabel(node);
677
678                DavaFlowSet temp = handleBreak(label, output2, node);
679
680                if (DEBUG_IF)
681                        System.out.println("Exiting if node" + temp.toString());
682                ;
683
684                return temp;
685        }

687        @Override
688        public DavaFlowSet processASTIfElseNode(ASTIfElseNode node, DavaFlowSet input) {
689                if (DEBUG_IF)
690                        System.out.println("Processing IF-ELSE node using over-ridden process if method" + input.toString());
691                ;
692
693                if (!(input instanceof CPFlowSet)) {
694                        throw new DavaFlowAnalysisException("not a flow set");
695                }
696
697                // get the subBodies
698                List<Object> subBodies = node.get_SubBodies();
699                if (subBodies.size() != 2) {
700                        throw new RuntimeException("processASTIfElseNode called with a node without two subBodies");
701                }
702                // we know there is only two subBodies
703                List subBodyOne = (List) subBodies.get(0);
704                List subBodyTwo = (List) subBodies.get(1);
705
706                // process Condition
707                input = processCondition(node.get_Condition(), input);
708
709                // the current input flowset is sent to both branches
710                DavaFlowSet clonedInput = cloneFlowSet(input);
711
712                CPTuple tuple = checkForValueHints(node.get_Condition(), (CPFlowSet) clonedInput, false);
713                if (tuple != null) {
714                        // if not null, is a belief going into the if branch simply add it
715                        // into the input set
716                        // System.out.println(">>>>>Adding tuple because of condition into if branch"+tuple.toString());
717                        ((CPFlowSet) clonedInput).addIfNotPresentButDontUpdate(tuple);
718                }
719
720                DavaFlowSet output1 = process(subBodyOne, clonedInput);
721                clonedInput = cloneFlowSet(input);
722                CPTuple tuple1 = checkForValueHints(node.get_Condition(), (CPFlowSet) clonedInput, true);
723
724                if (tuple1 != null) {
725                        // if not null, is a belief going into the else branch simply add it
726                        // into the input set
727                        // System.out.println(">>>>>Adding tuple because of condition  into else branch"+tuple1.toString());
728                        ((CPFlowSet) clonedInput).addIfNotPresentButDontUpdate(tuple1);
729                }
730                DavaFlowSet output2 = process(subBodyTwo, clonedInput);
731
732                if (DEBUG_IF) {
733
734                        System.out.println("\n\n  IF-ELSE   INPUTS TO MERGE ARE input (if):" + output1.toString() + " else:"
735                                        + output2.toString() + "\n\n\n");
736                }
737                DavaFlowSet temp = merge(output1, output2);
738
739                // notice we handle breaks only once since these are breaks to the same
740                // label or same node
741                String label = getLabel(node);
742                output1 = handleBreak(label, temp, node);
743                if (DEBUG_IF) {
744                        System.out.println("Exiting ifelse node" + output1.toString());
745                        ;
746                }
747
748                return output1;
749
750        }

752        /*
753         * The isElseBranch flag is true if the caller is the else branch of the
754         * ifelse statement. In that case we might be able to send something for the
755         * else branch
756         */
757        public CPTuple checkForValueHints(ASTCondition cond, CPFlowSet input, boolean isElseBranch) {
758                if (cond instanceof ASTUnaryCondition) {
759                        // check for lone boolean if(notDone)
760                        ASTUnaryCondition unary = (ASTUnaryCondition) cond;
761                        Value unaryValue = unary.getValue();
762
763                        boolean NOTTED = false;
764                        // Get the real value if this is a notted expression
765                        if (unaryValue instanceof DNotExpr) {
766                                unaryValue = ((DNotExpr) unaryValue).getOp();
767                                NOTTED = true;
768                        }
769
770                        if (!(unaryValue instanceof Local)) {
771                                // since we only track locals we cant possibly add info to the
772                                // inset
773                                return null;
774                        }
775
776                        // the unary value is a local add the value to the inset which woul
777                        // dbe present in the if branch
778                        CPVariable variable = new CPVariable((Local) unaryValue);
779
780                        // since NOTTED true means the if branch has variable with value
781                        // false and vice verse
782                        if (!isElseBranch) {
783                                // we are in the if branch hence notted true would mean the
784                                // variable is actually false here
785                                Boolean boolVal = new Boolean(!NOTTED);
786                                return new CPTuple(localClassName, variable, boolVal);
787                        } else {
788                                // in the else branch NOTTED true means the variable is true
789                                Boolean boolVal = new Boolean(NOTTED);
790                                return new CPTuple(localClassName, variable, boolVal);
791                        }
792                } else if (cond instanceof ASTBinaryCondition) {
793                        ASTBinaryCondition binary = (ASTBinaryCondition) cond;
794                        ConditionExpr expr = binary.getConditionExpr();
795
796                        Boolean equal = null;
797
798                        String symbol = expr.getSymbol();
799                        if (symbol.indexOf("==") > -1) {
800                                // System.out.println("!!!!!!!!!1 FOUND == in binary comparison operaiton");
801                                equal = new Boolean(true);
802                        } else if (symbol.indexOf("!=") > -1) {
803                                equal = new Boolean(false);
804                                // System.out.println("!!!!!!!!!!!!!! FOUND != in binary comparison operaiton");
805                        } else {
806                                // a symbol we are not interested in
807                                // System.out.println("symbol is"+symbol);
808                                return null;
809                        }
810
811                        // we have a comparison the truth value of equal tells whether we
812                        // are doing == or !=
813                        Value a = expr.getOp1();
814                        Value b = expr.getOp2();
815
816                        // see if its possible to deduce a hint from these values
817                        CPTuple tuple = createCPTupleIfPossible(a, b, input);
818
819                        // if the tuple is not created
820                        if (tuple == null)
821                                return null;
822
823                        // we have to make sure is that the == and != are taken into account
824                        if (equal.booleanValue()) {
825                                // using equality comparison a == b this then means in the if
826                                // branch a is in fact equal to b
827                                if (!isElseBranch)
828                                        return tuple;
829                                else
830                                        return null;
831                        } else {
832                                if (isElseBranch)
833                                        return tuple;
834                                else
835                                        return null;
836                        }
837
838                }
839                return null;
840        }

842        /*
843         * Should create the final tuple to add
844         * 
845         * a == b
846         * 
847         * case 1 a is constant b is constant dont give a damm case 2 a is constant
848         * b is a local useful case 3 a is a local b is a constant useful case 4 a
849         * is a local or sootfield b is a local or sootfield useful if one of them
850         * is in the inset
851         */
852        public CPTuple createCPTupleIfPossible(Value a, Value b, CPFlowSet input) {
853                Object aVal = CPHelper.isAConstantValue(a);
854                Object bVal = CPHelper.isAConstantValue(b);
855
856                if (aVal != null && bVal != null) {
857                        // both are constants dont want to do anything..case 1
858                        return null;
859                }
860
861                CPVariable cpVar = null;
862                Object constantToUse = null;
863
864                if (aVal == null && bVal == null) {
865                        // both are not constants but one of their values could be known in
866                        // the inset ... case 4
867                        // System.out.println("a:"+a+" is not a constant b:"+b+" is not");
868
869                        // check the input set to see if either a or b have known beliefs.
870                        // its useful if one and only one of the two has a known belief
871                        Object av1 = isANotTopConstantInInputSet(input, a);
872                        Object av2 = isANotTopConstantInInputSet(input, b);
873                        if (av1 == null && av2 == null) {
874                                // either top or not present hence useless
875                                return null;
876                        } else if (av1 == null && av2 != null) {
877                                // no value of a found but value of b was found <classname, a,b>
878                                // System.out.println("From INSET: a:"+a+" is not a constant b "+b+" is"
879                                // );
880                                if (!(a instanceof Local && ((Local) a).getType() instanceof PrimType)) {
881                                        // we only hanlde primitive locals
882                                        return null;
883                                }
884                                cpVar = new CPVariable((Local) a);
885                                constantToUse = av2;
886                        } else if (av1 != null && av2 == null) {
887                                // no value of b found but value of a was found <classname, b,a>
888                                // System.out.println("From INSET: a:"+a+" is a constant b "+b+" is not"
889                                // );
890                                if (!(b instanceof Local && ((Local) b).getType() instanceof PrimType)) {
891                                        // we only hanlde primitive locals
892                                        return null;
893                                }
894                                cpVar = new CPVariable((Local) b);
895                                constantToUse = av1;
896                        }
897                } else if (aVal != null && bVal == null) {
898                        // CASE 2: a is a constant and b is not so we have a chance of
899                        // entering a tuple <className,b,a> maybe
900                        // System.out.println("a:"+a+" is a constant b:"+b+" is not");
901
902                        if (!(b instanceof Local && ((Local) b).getType() instanceof PrimType)) {
903                                // we only hanlde primitive locals
904                                return null;
905                        }
906
907                        // able to create cpVar
908                        cpVar = new CPVariable((Local) b);
909                        constantToUse = aVal;
910                } else if (aVal == null && bVal != null) {
911                        // CASE 3: a is not a constant but b is a constant so we have a
912                        // chance of entering a tuple <className,a,b>
913                        // System.out.println("a:"+a+" is not a constant b:"+b+" is ");
914
915                        if (!(a instanceof Local && ((Local) a).getType() instanceof PrimType)) {
916                                // we only hanlde primitive locals
917                                return null;
918                        }
919
920                        // able to create cpVar
921                        cpVar = new CPVariable((Local) a);
922                        constantToUse = bVal;
923
924                }
925
926                // if cpVar is not null and constantToUse is not null thats good
927                if (cpVar != null && constantToUse != null) {
928                        // create a CPTuple which contains the belief for cpVar with
929                        // constantToUse we will for sure have going into the if branch
930                        // we know cpVar is always a local since we create it only for
931                        // locals
932
933                        // need to see if constant is supposed to be a boolean
934                        // (isAConstantValue returns an Integer for a boolean)
935                        if (cpVar.getLocal().getType() instanceof BooleanType) {
936                                if (!(constantToUse instanceof Integer)) {
937                                        // booleans are represented by Integeres in the
938                                        // isConstantValue method what happened here???
939                                        return null;
940                                }
941
942                                Integer tempValue = (Integer) constantToUse;
943                                if (tempValue.intValue() == 0)
944                                        constantToUse = new Boolean(false);
945                                else
946                                        constantToUse = new Boolean(true);
947                        }
948
949                        // ready to create the CPTuple
950
951                        return new CPTuple(localClassName, cpVar, constantToUse);
952                }
953                return null;
954        }
