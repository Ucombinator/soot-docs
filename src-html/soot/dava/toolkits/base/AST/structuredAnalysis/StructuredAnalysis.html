177        /**
178         * This implementation breaks down the aggregated condition to the terminal
179         * conditions which all have type ASTUnaryBinaryCondition. Once these are
180         * obtained the abstract method processUnaryBinaryCondition is invoked. For
181         * aggregated conditions the merging is done in a depth first order of the
182         * condition tree.
183         */
184        public DavaFlowSet<E> processCondition(ASTCondition cond, DavaFlowSet<E> input) {
185                if (cond instanceof ASTUnaryBinaryCondition) {
186                        return processUnaryBinaryCondition((ASTUnaryBinaryCondition) cond, input);
187                } else if (cond instanceof ASTAggregatedCondition) {
188                        ASTCondition left = ((ASTAggregatedCondition) cond).getLeftOp();
189                        DavaFlowSet<E> output1 = processCondition(left, input);
190
191                        ASTCondition right = ((ASTAggregatedCondition) cond).getRightOp();
192                        DavaFlowSet<E> output2 = processCondition(right, output1);
193
194                        return merge(output1, output2);
195                } else {
196                        throw new RuntimeException("Unknown ASTCondition found in structred flow analysis");
197                }
198        }

200        /*
201         * The parameter body contains the body to be analysed It can be an ASTNode,
202         * a Stmt, an augmentedStmt or a list of ASTNodes The input is any data that
203         * is gathered plus any info needed for making decisions during the analysis
204         */
205        public DavaFlowSet<E> process(Object body, DavaFlowSet<E> input) {
206                if (body instanceof ASTNode) {
207                        beforeSets.put(body, input);
208                        DavaFlowSet<E> temp = processASTNode((ASTNode) body, input);
209                        afterSets.put(body, temp);
210                        return temp;
211                } else if (body instanceof Stmt) {
212                        beforeSets.put(body, input);
213                        DavaFlowSet<E> result = processAbruptStatements((Stmt) body, input);
214                        afterSets.put(body, result);
215                        return result;
216                } else if (body instanceof AugmentedStmt) {
217                        AugmentedStmt as = (AugmentedStmt) body;
218                        Stmt s = as.get_Stmt();
219
220                        beforeSets.put(s, input);
221                        DavaFlowSet<E> result = processAbruptStatements(s, input);
222                        afterSets.put(s, result);
223                        return result;
224
225                } else if (body instanceof List) {
226                        // this should always be a list of ASTNodes
227                        Iterator it = ((List) body).iterator();
228                        DavaFlowSet<E> result = input;
229                        while (it.hasNext()) {
230                                Object temp = it.next();
231                                if (!(temp instanceof ASTNode))
232                                        throw new RuntimeException("Body sent to be processed by "
233                                                        + "StructuredAnalysis contains a list which does not have ASTNodes");
234                                else {
235                                        /*
236                                         * As we are simply going through a list of ASTNodes The
237                                         * output of the previous becomes the input of the next
238                                         */
239                                        beforeSets.put(temp, result);
240                                        result = processASTNode((ASTNode) temp, result);
241                                        afterSets.put(temp, result);
242                                }
243                        }// end of going through list
244
245                        // at this point the result var contains the result of processing
246                        // the List
247                        return result;
248                } else {
249                        throw new RuntimeException("Body sent to be processed by " + "StructuredAnalysis is not a valid body");
250                }
251        }

253        /*
254         * This method internally invoked by the process method decides which
255         * ASTNode specialized method to call
256         */
257        public DavaFlowSet<E> processASTNode(ASTNode node, DavaFlowSet<E> input) {
258                if (node instanceof ASTDoWhileNode) {
259                        return processASTDoWhileNode((ASTDoWhileNode) node, input);
260                } else if (node instanceof ASTForLoopNode) {
261                        return processASTForLoopNode((ASTForLoopNode) node, input);
262                } else if (node instanceof ASTIfElseNode) {
263                        return processASTIfElseNode((ASTIfElseNode) node, input);
264                } else if (node instanceof ASTIfNode) {
265                        return processASTIfNode((ASTIfNode) node, input);
266                } else if (node instanceof ASTLabeledBlockNode) {
267                        return processASTLabeledBlockNode((ASTLabeledBlockNode) node, input);
268                } else if (node instanceof ASTMethodNode) {
269                        return processASTMethodNode((ASTMethodNode) node, input);
270                } else if (node instanceof ASTStatementSequenceNode) {
271                        return processASTStatementSequenceNode((ASTStatementSequenceNode) node, input);
272                } else if (node instanceof ASTSwitchNode) {
273                        return processASTSwitchNode((ASTSwitchNode) node, input);
274                } else if (node instanceof ASTSynchronizedBlockNode) {
275                        return processASTSynchronizedBlockNode((ASTSynchronizedBlockNode) node, input);
276                } else if (node instanceof ASTTryNode) {
277                        return processASTTryNode((ASTTryNode) node, input);
278                } else if (node instanceof ASTWhileNode) {
279                        return processASTWhileNode((ASTWhileNode) node, input);
280                } else if (node instanceof ASTUnconditionalLoopNode) {
281                        return processASTUnconditionalLoopNode((ASTUnconditionalLoopNode) node, input);
282                } else {
283                        throw new RuntimeException("processASTNode called using unknown node type");
284                }
285        }

288         * This method is called from the specialized ASTNodes. The purpose was to
289         * deal with different ASTNodes with similar structure in one go. The method
290         * will deal with retrieve the body of the ASTNode which are known to have
291         * only one subBody
292         */
293        public final DavaFlowSet<E> processSingleSubBodyNode(ASTNode node, DavaFlowSet<E> input) {
294                // get the subBodies
295                List<Object> subBodies = node.get_SubBodies();
296                if (subBodies.size() != 1) {
297                        throw new RuntimeException("processSingleSubBodyNode called with a node without one subBody");
298                }
299                // we know there is only one
300                List subBody = (List) subBodies.get(0);
301                return process(subBody, input);
302        }

317        /**
318         * Whenever a statement has to be processed the first step is to invoke this
319         * method. This is to remove the tedious work of adding code to deal with
320         * abrupt control flow from the programmer of the analysis. The method
321         * invokes the processStatement method for all other statements
322         * 
323         * A programmer can decide to override this method if they want to do
324         * something specific
325         */
326        public DavaFlowSet<E> processAbruptStatements(Stmt s, DavaFlowSet<E> input) {
327                if (s instanceof ReturnStmt || s instanceof RetStmt || s instanceof ReturnVoidStmt) {
328                        // dont need to remember this path
329                        return NOPATH;
330                } else if (s instanceof DAbruptStmt) {
331                        DAbruptStmt abStmt = (DAbruptStmt) s;
332
333                        // see if its a break or continue
334                        if (!(abStmt.is_Continue() || abStmt.is_Break())) {
335                                // DAbruptStmt is of only two kinds
336                                throw new RuntimeException("Found a DAbruptStmt which is neither break nor continue!!");
337                        }
338
339                        DavaFlowSet<E> temp = NOPATH;
340                        SETNodeLabel nodeLabel = abStmt.getLabel();
341                        // System.out.println("here");
342                        if (nodeLabel != null && nodeLabel.toString() != null) {
343                                // explicit abrupt stmt
344                                if (abStmt.is_Continue())
345                                        temp.addToContinueList(nodeLabel.toString(), input);
346                                else if (abStmt.is_Break())
347                                        temp.addToBreakList(nodeLabel.toString(), input);
348                                else
349                                        throw new RuntimeException("Found abruptstmt which is neither break nor continue");
350                        } else {
351                                // found implicit break/continue
352                                if (abStmt.is_Continue())
353                                        temp.addToImplicitContinues(abStmt, input);
354                                else if (abStmt.is_Break())
355                                        temp.addToImplicitBreaks(abStmt, input);
356                                else
357                                        throw new RuntimeException("Found abruptstmt which is neither break nor continue");
358                        }
359                        return temp;
360                } else {
361                        /**************************************************************/
362                        /****** ALL OTHER STATEMENTS HANDLED BY PROGRAMMER **************/
363                        /**************************************************************/
364                        return processStatement(s, input);
365                }
366        }

437        public DavaFlowSet<E> processASTIfElseNode(ASTIfElseNode node, DavaFlowSet<E> input) {
438                // get the subBodies
439                List<Object> subBodies = node.get_SubBodies();
440                if (subBodies.size() != 2) {
441                        throw new RuntimeException("processASTIfElseNode called with a node without two subBodies");
442                }
443                // we know there is only two subBodies
444                List subBodyOne = (List) subBodies.get(0);
445                List subBodyTwo = (List) subBodies.get(1);
446
447                // process Condition
448                input = processCondition(node.get_Condition(), input);
449                // the current input flowset is sent to both branches
450                DavaFlowSet<E> clonedInput = cloneFlowSet(input);
451                DavaFlowSet<E> output1 = process(subBodyOne, clonedInput);
452
453                clonedInput = cloneFlowSet(input);
454                DavaFlowSet<E> output2 = process(subBodyTwo, clonedInput);
455
456                DavaFlowSet<E> temp = merge(output1, output2);
457                // notice we handle breaks only once since these are breaks to the same
458                // label or same node
459                String label = getLabel(node);
460                output1 = handleBreak(label, temp, node);
461
462                return output1;
463        }

465        public DavaFlowSet<E> processASTWhileNode(ASTWhileNode node, DavaFlowSet<E> input) {
466                DavaFlowSet<E> lastin = null;
467                DavaFlowSet<E> initialInput = cloneFlowSet(input);
468
469                String label = getLabel(node);
470                DavaFlowSet<E> output = null;
471
472                input = processCondition(node.get_Condition(), input);
473                if (DEBUG_WHILE)
474                        System.out.println("Going int while (condition processed): " + input.toString());
475
476                do {
477                        lastin = cloneFlowSet(input);
478                        output = processSingleSubBodyNode(node, input);
479
480                        // handle continue
481                        output = handleContinue(label, output, node);
482
483                        // merge with the initial input
484                        input = merge(initialInput, output);
485                        input = processCondition(node.get_Condition(), input);
486                } while (isDifferent(lastin, input));
487
488                // input contains the result of the fixed point
489                DavaFlowSet<E> temp = handleBreak(label, input, node);
490                if (DEBUG_WHILE)
491                        System.out.println("Going out of while: " + temp.toString());
492                return temp;
493        }

517                DavaFlowSet<E> temp = handleBreak(label, output, node);
518                if (DEBUG_WHILE)

555        public DavaFlowSet<E> processASTForLoopNode(ASTForLoopNode node, DavaFlowSet<E> input) {
556                for (AugmentedStmt as : node.getInit()) {
557                        Stmt s = as.get_Stmt();
558                        input = process(s, input);
559                }
560
561                // finished processing the init part of the for loop
562                DavaFlowSet<E> initialInput = cloneFlowSet(input);
563
564                input = processCondition(node.get_Condition(), input);
565                DavaFlowSet<E> lastin = null;
566                String label = getLabel(node);
567                DavaFlowSet<E> output2 = null;
568                do {
569                        lastin = cloneFlowSet(input);
570
571                        // process body
572                        DavaFlowSet<E> output1 = processSingleSubBodyNode(node, input);
573
574                        // handle continues (Notice this is done before update!!!)
575                        output1 = handleContinue(label, output1, node);
576
577                        // notice that we dont merge with the initial output1 from
578                        // processing singleSubBody
579                        // the handlecontinue function takes care of it
580
581                        // handle update
582                        output2 = cloneFlowSet(output1);// if there is nothing in update
583
584                        for (AugmentedStmt as : node.getUpdate()) {
585                                Stmt s = as.get_Stmt();
586                                /*
587                                 * Since we are just going over a list of statements the output
588                                 * of each statement is the input of the next
589                                 */
590                                output2 = process(s, output2);
591                        }
592
593                        // output2 is the final result
594
595                        // merge this with the input
596                        input = merge(initialInput, output2);
597                        input = processCondition(node.get_Condition(), input);
598                } while (isDifferent(lastin, input));
599
600                // handle break
601                return handleBreak(label, input, node);
602        }

604        /*
605         * Notice ASTSwitch is horribly conservative....eg. if all cases break
606         * properly it will still merge with defaultOut which will be a NOPATH and
607         * bound to have empty or full sets
608         */
609        public DavaFlowSet<E> processASTSwitchNode(ASTSwitchNode node, DavaFlowSet<E> input) {
610                if (DEBUG)
611                        System.out.println("Going into switch: " + input.toString());
612
613                List<Object> indexList = node.getIndexList();
614                Map<Object, List<Object>> index2BodyList = node.getIndex2BodyList();
615
616                Iterator<Object> it = indexList.iterator();
617
618                input = processSwitchKey(node.get_Key(), input);
619                DavaFlowSet<E> initialIn = cloneFlowSet(input);
620
621                DavaFlowSet<E> out = null;
622                DavaFlowSet<E> defaultOut = null;
623
624                List<DavaFlowSet<E>> toMergeBreaks = new ArrayList<DavaFlowSet<E>>();
625
626                while (it.hasNext()) {// going through all the cases of the switch
627                                                                // statement
628                        Object currentIndex = it.next();
629                        List body = index2BodyList.get(currentIndex);
630
631                        // BUG FIX if body is null (fall through we shouldnt invoke process
632                        // Reported by Steffen Pingel 14th Jan 2006 on the soot mailing list
633                        if (body != null) {
634                                out = process(body, input);
635
636                                // System.out.println("Breaklist for this out is"+out.getBreakList());
637                                toMergeBreaks.add(cloneFlowSet(out));
638
639                                if (currentIndex instanceof String) {
640                                        // this is the default
641                                        defaultOut = out;
642                                }
643
644                                // the input to the next can be a fall through or directly input
645                                input = merge(out, initialIn);
646                        }// body was non null
647                }
648
649                // have to handle the case when no case matches. The input is the output
650                DavaFlowSet<E> output = null;
651                if (out != null) {// just to make sure that there were some cases
652                                                        // present
653
654                        /*
655                         * January 30th 2006, FOUND BUG The initialIn should only be merge
656                         * with the out if there is no default in the list of switch cases
657                         * If there is a default then there is no way that the initialIn is
658                         * the actual result. Then its either the default or one of the
659                         * outs!!!
660                         */
661                        if (defaultOut != null) {
662                                // there was a default
663                                // System.out.println("DEFAULTSET");
664                                // System.out.println("defaultOut is"+defaultOut);
665                                // System.out.println("out is"+out);
666
667                                output = merge(defaultOut, out);
668                        } else {
669                                // there was no default so no case might match
670                                output = merge(initialIn, out);
671                        }
672
673                } else
674                        output = initialIn;
675
676                // handle break
677                String label = getLabel(node);
678
679                // have to handleBreaks for all the different cases
680                List<DavaFlowSet<E>> outList = new ArrayList<DavaFlowSet<E>>();
681
682                // handling breakLists of each of the toMergeBreaks
683                for (DavaFlowSet<E> mset : toMergeBreaks)
684                        outList.add(handleBreak(label, mset, node));
685
686                // merge all outList elements. since these are the outputs with breaks
687                // handled
688                DavaFlowSet<E> finalOut = output;
689                for (DavaFlowSet<E> outIt : outList)
690                        finalOut = merge(finalOut, outIt);
691
692                if (DEBUG)
693                        System.out.println("Going out of switch: " + finalOut.toString());
694
695                return finalOut;
696        }

698        public DavaFlowSet<E> processASTTryNode(ASTTryNode node, DavaFlowSet<E> input) {
699                if (DEBUG_TRY)
700                        System.out.println("TRY START is:" + input);
701
702                // System.out.println("SET beginning of tryBody is:"+input);
703                List<Object> tryBody = node.get_TryBody();
704                DavaFlowSet<E> tryBodyOutput = process(tryBody, input);
705                // System.out.println("SET end of tryBody is:"+tryBodyOutput);
706
707                /*
708                 * By default take either top or bottom as the input to the catch
709                 * statements Which goes in depends on the type of analysis.
710                 */
711                DavaFlowSet<E> inputCatch = newInitialFlow();
712                if (DEBUG_TRY)
713                        System.out.println("TRY initialFLOW is:" + inputCatch);
714
715                List<Object> catchList = node.get_CatchList();
716                Iterator<Object> it = catchList.iterator();
717                List<DavaFlowSet<E>> catchOutput = new ArrayList<DavaFlowSet<E>>();
718
719                while (it.hasNext()) {
720                        ASTTryNode.container catchBody = (ASTTryNode.container) it.next();
721
722                        List<E> body = (List<E>) catchBody.o;
723                        // list of ASTNodes
724
725                        // result because of going through the catchBody
726                        DavaFlowSet<E> tempResult = process(body, cloneFlowSet(inputCatch));
727                        // System.out.println("TempResult going through body"+tempResult);
728                        catchOutput.add(tempResult);
729                }
730
731                // handle breaks
732                String label = getLabel(node);
733
734                /*
735                 * 30th Jan 2005, Found bug in handling out breaks what was being done
736                 * was that handleBreak was invoked using just
737                 * handleBreak(label,tryBodyoutput,node) Now what it does is that it
738                 * looks for the breakList stored in the tryBodyOutput node What might
739                 * happen is that there might be some breaks in the catchOutput which
740                 * would have gotten stored in the breakList of the respective
741                 * catchoutput
742                 * 
743                 * The correct way to handle this is create a list of handledBreak
744                 * objects (in the outList) And then to merge them
745                 */
746                List<DavaFlowSet<E>> outList = new ArrayList<DavaFlowSet<E>>();
747
748                // handle breaks out of tryBodyOutput
749                outList.add(handleBreak(label, tryBodyOutput, node));
750                // System.out.println("After handling break from tryBodyOutput"+outList.get(0));
751
752                // handling breakLists of each of the catchOutputs
753                for (DavaFlowSet<E> co : catchOutput) {
754                        DavaFlowSet<E> temp = handleBreak(label, co, node);
755
756                        if (DEBUG_TRY)
757                                System.out.println("TRY handling breaks is:" + temp);
758
759                        outList.add(temp);
760                }
761
762                // merge all outList elements. since these are the outputs with breaks
763                // handled
764                DavaFlowSet<E> out = tryBodyOutput;
765                for (DavaFlowSet<E> co : outList)
766                        out = merge(out, co);
767                
768                if (DEBUG_TRY)
769                        System.out.println("TRY after merge outList is:" + out);
770
771                // System.out.println("After handling break"+out);
772
773                for (DavaFlowSet<E> co : catchOutput)
774                        out = merge(out, co);
775                
776                if (DEBUG_TRY)
777                        System.out.println("TRY END RESULT is:" + out);
778
779                return out;
780        }

782        /*
783         * MERGETYPE var has to be set 0, means no type set 1, means union 2, means
784         * intersection
785         */
786        public DavaFlowSet<E> merge(DavaFlowSet<E> in1, DavaFlowSet<E> in2) {
787                if (MERGETYPE == 0)
788                        throw new RuntimeException("Use the setMergeType method to set the type of merge used in the analysis");
789                
790                DavaFlowSet<E> out;
791                if (in1 == NOPATH && in2 != NOPATH) {
792                        out = in2.clone();
793                        out.copyInternalDataFrom(in1);
794                        return out;
795                } else if (in1 != NOPATH && in2 == NOPATH) {
796                        out = in1.clone();
797                        out.copyInternalDataFrom(in2);
798                        return out;
799                } else if (in1 == NOPATH && in2 == NOPATH) {
800                        out = in1.clone();
801                        out.copyInternalDataFrom(in2);
802                        return out; // meaning return NOPATH
803                } else {// both are not NOPATH
804                        out = emptyFlowSet();
805                        if (MERGETYPE == 1)// union
806                                in1.union(in2, out);
807                        else if (MERGETYPE == 2)// intersection
808                                in1.intersection(in2, out);
809                        else
810                                throw new RuntimeException("Merge type value" + MERGETYPE + " not recognized");
811                        out.copyInternalDataFrom(in1);
812                        out.copyInternalDataFrom(in2);
813                        return out;
814                }
815        }

817        public DavaFlowSet<E> mergeExplicitAndImplicit(String label, DavaFlowSet<E> output,
818                        List<DavaFlowSet<E>> explicitSet, List<DavaFlowSet<E>> implicitSet) {
819                DavaFlowSet<E> toReturn = output.clone();
820
821                if (label != null) {
822                        // use the explicit list
823                        /*
824                         * If there is no list associated with this label or the list is
825                         * empty there no explicit merging to be done
826                         */
827                        if (explicitSet != null && explicitSet.size() != 0) {
828                                // explicitSet is a list of DavaFlowSets
829                                Iterator<DavaFlowSet<E>> it = explicitSet.iterator();
830
831                                // we know there is atleast one element
832                                toReturn = merge(output, it.next());
833
834                                while (it.hasNext()) {
835                                        // merge this with toReturn
836                                        toReturn = merge(toReturn, it.next());
837                                }
838                        }// a non empty explicitSet was found
839                }// label not null could have explicit sets
840
841                // toReturn contains result of dealing with explicit stmts
842
843                // dealing with implicit set now
844                if (implicitSet != null) {
845                        // implicitSet is a list of DavaFlowSets
846                        Iterator<DavaFlowSet<E>> it = implicitSet.iterator();
847                        while (it.hasNext()) {
848                                // merge this with toReturn
849                                toReturn = merge(toReturn, it.next());
850                        }
851                }
852                return toReturn;
853        }

855        /**
856         * Need to handleBreak stmts There can be explicit breaks (in which case
857         * label is non null) There can always be implicit breaks ASTNode is non
858         * null
859         */
860        public DavaFlowSet<E> handleBreak(String label, DavaFlowSet<E> out, ASTNode node) {
861                // get the explicit list with this label from the breakList
862                List<DavaFlowSet<E>> explicitSet = out.getBreakSet(label);
863                // System.out.println("\n\nExplicit set is"+explicitSet);
864                // getting the implicit list now
865                if (node == null)
866                        throw new RuntimeException("ASTNode sent to handleBreak was null");
867
868                List<DavaFlowSet<E>> implicitSet = out.getImplicitlyBrokenSets(node);
869                // System.out.println("\n\nImplicit set is"+implicitSet);
870
871                // invoke mergeExplicitAndImplicit
872                return mergeExplicitAndImplicit(label, out, explicitSet, implicitSet);
873        }

876         * Need to handleContinue stmts There can be explicit continues (in which
877         * case label is non null) There can always be implicit continues ASTNode is
878         * non null
879         */
880        public DavaFlowSet<E> handleContinue(String label, DavaFlowSet<E> out, ASTNode node) {
881                // get the explicit list with this label from the continueList
882                List<DavaFlowSet<E>> explicitSet = out.getContinueSet(label);
883
884                // getting the implicit list now
885                if (node == null)
886                        throw new RuntimeException("ASTNode sent to handleContinue was null");
887
888                List<DavaFlowSet<E>> implicitSet = out.getImplicitlyContinuedSets(node);
889
890                // invoke mergeExplicitAndImplicit
891                return mergeExplicitAndImplicit(label, out, explicitSet, implicitSet);
892        }

894        /**
895         * Invoked from within the UnconditionalWhile processing method Need to
896         * handle both explicit and implicit breaks
897         */
898        private DavaFlowSet<E> getMergedBreakList(String label, DavaFlowSet<E> output, ASTNode node) {
899                List<DavaFlowSet<E>> breakSet = output.getBreakSet(label);
900                DavaFlowSet<E> toReturn = null;
901
902                if (breakSet == null) {
903                        // there is no list associated with this label hence no merging to
904                        // be done
905                        // since this is a call from unconditional this means there should
906                        // have been an implicit break
907                        toReturn = NOPATH;
908                } else if (breakSet.size() == 0) {
909                        // list is empty for this label hence no merging to be done
910                        // since this is a call from unconditional this means there should
911                        // have been an implicit break
912                        toReturn = NOPATH;
913                } else {
914                        // breakSet is a list of DavaFlowSets
915                        Iterator<DavaFlowSet<E>> it = breakSet.iterator();
916
917                        // we know there is atleast one element
918                        // making sure we dont send NOPATH
919                        toReturn = it.next();
920
921                        while (it.hasNext()) {
922                                // merge this with toReturn
923                                toReturn = merge(toReturn, it.next());
924                        }
925                }// a non empty breakSet was found
926
927                // dealing with implicit set now
928
929                List<DavaFlowSet<E>> implicitSet = output.getImplicitlyBrokenSets(node);
930                if (implicitSet != null) {
931                        // implicitSet is a list of DavaFlowSets
932                        Iterator<DavaFlowSet<E>> it = implicitSet.iterator();
933
934                        // making sure that we dont send NOPATH
935                        if (implicitSet.size() > 0)
936                                toReturn = it.next();
937
938                        while (it.hasNext()) {
939                                // merge this with toReturn
940                                toReturn = merge(toReturn, it.next());
941                        }
942                }
943                return toReturn;
944        }
