032        public CPFlowSet(CPFlowSet other){

034        numElements = other.numElements;

053        continueList = (HashMap<Serializable, List<DavaFlowSet<CPTuple>>>)other.continueList.clone();
054        implicitBreaks = (HashMap<Serializable, List<DavaFlowSet<CPTuple>>>)other.implicitBreaks.clone();

061        /*
062         * helper method to be invoked by CPApplication when trying to do transformation
063         * 
064         * returns a non null if the local or field is contained and has a constant value
065         * returns a null if the local or field is eithe rnot present or is trop
066         */
067        public Object contains(String className, String localOrField){
068                for(int i = 0; i < this.numElements; i++) {
069                        CPTuple current = getElementAt(i);
070                        if(! (current.getSootClassName().equals(className)))
071                                continue;
072                        
073                        if(current.containsField() ){
074                                if(! current.getVariable().getSootField().getName().equals(localOrField) )
075                                        continue;
076                                else
077                                        return current.getValue();
078                        }
079                        else if (current.containsLocal()){
080                                if(! current.getVariable().getLocal().getName().equals(localOrField))
081                                        continue;
082                                else 
083                                        return current.getValue();
084                        }
085                }
086                return null;
087        }

091        /*
092         * This is more of an update method than an add method.
093         * 
094         * Go through all the elements in the flowSet
095         * See if we can find an element (CPTuple with the same className and same CPVariable)
096         * if we dont find one: add this to the flowset
097         * if we find one: update the Value with the value of the newTuple
098         */
099        public void addIfNotPresent(CPTuple newTuple){
100                //System.out.println("addIfnotPresent invoked");
101                //going through all the elements in the set
102                for(int i = 0; i < this.numElements; i++) {
103                        CPTuple current = (CPTuple)elements[i];
104                        if(! (current.getSootClassName().equals(newTuple.getSootClassName())) ){
105                                //different classNAmes
106                                continue;
107                        }
108                        
109                        //same class names
110                        
111                        CPVariable curVar = current.getVariable();
112                        CPVariable newTupleVar = newTuple.getVariable();
113                        
114                        if(! (curVar.equals(newTupleVar) ) ){
115                                //different variable
116                                continue;
117                        }
118                        
119                        //same class and same variable
120                        //UPDATE this elements VALUE
121                        //System.out.println("got here"+newTuple.getValue());
122                        
123                        current.setValue(newTuple.getValue());
124                        //since the tuple was present no need to ADD since we updated
125                        return;
126                }
127                
128                //if we get to this part we know that we need to add
129                //System.out.println("no got here");
130                this.add(newTuple);
131        }

152         * Specialized method for handling conditionals
153         * this is used to add a belief derived from the condition
154         * of an if or ifelse

156         * The belief is only added if the current belief

162        public void addIfNotPresentButDontUpdate(CPTuple newTuple){
163                //System.out.println("addIfnotPresent invoked");
164                //going through all the elements in the set
165                for(int i = 0; i < this.numElements; i++) {
166                        CPTuple current = (CPTuple)elements[i];
167                        if(! (current.getSootClassName().equals(newTuple.getSootClassName())) ){
168                                //different classNAmes
169                                continue;
170                        }
171                        
172                        //same class names
173                        
174                        CPVariable curVar = current.getVariable();
175                        CPVariable newTupleVar = newTuple.getVariable();
176                        
177                        if(! (curVar.equals(newTupleVar) ) ){
178                                //different variable
179                                continue;
180                        }
181
182                        /*
183                         * We only assign value if there is one not already
184                         * present
185                         */
186                        if( current.isTop())
187                                current.setValue(newTuple.getValue());
188                        
189                        return;
190                }
191                
192                //if we get to this part we know that we need to add
193                /*
194                 * DO NOT ADD IF NOT FOUND SINCE THAT MEANS IT IS BOTTOM and we dont want the following to occur
195                 * 
196                 *    if( bla ){
197                 *    
198                 *                              if a var is bottom before the loop and we add something because of bla
199                 *    }                         then the merge rules will cause the after set of if to have the something 
200                 *    Body A                    since bottom merged with something is that something
201                 *                      THIS WOULD BE INCORRECT
202                 */
203                //this.add(newTuple);
204        }

208        /*
209         * The intersection method is called by the object whose set is to be intersected with otherFlow
210         * and the result to be stored in destFlow
211         * 
212         *  So we will be intersecting elements of "this" and otherFlow
213         *  
214         *  Definition of Intersection: 
215         *  If an element e belongs to Set A then Set A ^ B contains e
216         *  if B contains an element such that the constantpropagationFlowSet equals method returns true 
217         *  (this will happen if they have the same variable and the same value which better not be TOP)
218         *  
219         *   If the element e is not added to set A ^ B then element e should be changed to hold TOP as the value
220         *   is unknown and that value should be added to the set A ^ B
221         *   
222         * 
223         */
224        public void intersection(FlowSet otherFlow, FlowSet destFlow){
225                //System.out.println("In specialized intersection for CopyPropagation");
226                if ( ! (otherFlow instanceof CPFlowSet  && destFlow instanceof CPFlowSet ) ) {
227                  super.intersection(otherFlow, destFlow);
228                  return;
229                }
230
231                CPFlowSet other = (CPFlowSet) otherFlow;
232                CPFlowSet dest = (CPFlowSet) destFlow;
233                CPFlowSet workingSet;
234                        
235                if(dest == other || dest == this)
236                        workingSet = new CPFlowSet();
237                else {  
238                        workingSet = dest;
239                        workingSet.clear();
240                }
241
242                /*
243                 * HERE IS THE MERGE TABLE
244                 * 
245                 *        
246                 *            THIS       OTHER          RESULT
247                 *          
248                 *      1    BOTTOM       BOTTOM        WHO CARES                              
249                 *      2    BOTTOM         C              C       (Laurie Convinced me)  
250                 *      3    BOTTOM        TOP            TOP                                                   
251                 *      4     C           BOTTOM           C        (Laurie Convinced me)
252                 *      5     C1             C2            C if C1 == C2 else TOP 
253                 *      6     C            TOP            TOP 
254                 *      7    TOP          BOTTOM          TOP 
255                 *      8    TOP            C             TOP 
256                 *      9    TOP           TOP            TOP
257                 */
258                
259                for(int i = 0; i < this.numElements; i++) {
260                        CPTuple thisTuple = this.getElementAt(i);
261                        String className = thisTuple.getSootClassName();
262                        CPVariable thisVar = thisTuple.getVariable();
263                        
264                        CPTuple matchFound=null;
265                        /*
266                         * Find a matching tuple if one exists
267                         */
268                        CPTuple otherTuple = null;
269                        for(int j=0;j<other.numElements;j++){
270                                otherTuple = other.getElementAt(j);
271                                        
272                                /*
273                                 * wont use the CPTuple equal method since that ignores tops
274                                 * we want to implement the intersection rules given in the merge table
275                                 */
276                                        
277                                //check that the two tuples have the same class
278                                String tempClass = otherTuple.getSootClassName();
279                                if( ! tempClass.equals(className))
280                                        continue;
281                                                                                
282                                //Check that the variable contained is the same name and type (local or sootfield)
283                                if( ! otherTuple.getVariable().equals(thisVar))
284                                        continue;
285                                                
286                                //match of sootClass and Variable have to use merge table on matchFound and elements[i]
287                                matchFound = otherTuple;
288                                break;                                          
289                        } //end looking for a match in other set
290                                
291                        if(matchFound != null){
292                                //cases 5 6 8 and 9 are the ones in which both sets have a value for this variable
293                                        
294                                if(thisTuple.isTop()){
295                                        //cases 8 and 9
296                                        workingSet.add(thisTuple.clone());
297                                }
298                                else if(matchFound.isTop()){
299                                        //case 6
300                                        workingSet.add(matchFound.clone());
301                                }
302                                else if( !matchFound.isTop() && !thisTuple.isTop() ){
303                                        //this has to be case 5 since there is no other case possible
304                                                
305                                        Object matchedValue = matchFound.getValue();
306                                        Object thisValue = thisTuple.getValue();
307                                                
308                                        //using the equals method of Boolean/Float/Integer etc
309                                        if(matchedValue.equals(thisValue)){
310                                                workingSet.add(thisTuple.clone());
311                                        }
312                                        else{
313                                                //if we get here either the types dont match or the values didnt match just add top
314                                                workingSet.add( new CPTuple(className,thisVar,true));
315                                        }
316                                }
317                                else{
318                                        throw new DecompilationException("Ran out of cases in CPVariable values...report bug to developer");                                                    
319                                }
320                        }//if match was found
321                        else{
322                                //could not find a match for element[i] in other hence its bottom in other
323                                //CASE 4 and 7 (cant be case 1 since element[i]s presence means its not bottom
324                                        
325                                //add CLONE OF element[i] to working set unchanged
326                                /*
327                                 * TODO: why should a field be turned to TOP... a field is only a constant value field and hence 
328                                 * should never be changed!!!!!
329                                 * 
330                                 * BUG FOUND DUE TO CHROMOSOME benchmark if(Debug.flag) was not being detected as it was being
331                                 * set to top
332                                 * 
333                                 * April 3rd 2006
334                                 */
335                                /*              if(thisTuple.containsField()){
336                                 //add top
337                                  workingSet.add(new CPTuple(thisTuple.getSootClassName(),thisTuple.getVariable(),true));
338                                  }
339                                  else if(thisTuple.containsLocal()){
340                                  */
341                                
342                                workingSet.add(thisTuple.clone());                                      
343                                
344                                
345                                /*}
346                                else
347                                        throw new DecompilationException("CPVariable is not local and not field");
348                                */
349
350                        }
351                                
352                                
353                }//end going through all elements of this flowset
354                
355        
356                /*
357                 * havent covered cases 2 and 3 in which case this has bottom (a.k.a variable is not present)
358                 * and the other has the elements
359                 */
360
361                for(int i=0;i<other.numElements;i++){
362                        CPTuple otherTuple = other.getElementAt(i);
363                        String otherClassName = otherTuple.getSootClassName();
364                        CPVariable otherVar = otherTuple.getVariable();
365                        //System.out.print("\t Other:"+otherVar.toString());
366                        boolean inBoth=false;
367                        for(int j=0;j<this.numElements;j++){
368                                CPTuple thisTuple = this.getElementAt(j);
369                                String thisClassName = thisTuple.getSootClassName();
370                                CPVariable thisVar = thisTuple.getVariable();
371                                
372                                if(! otherClassName.equals(thisClassName)){
373                                        continue;
374                                }
375                                
376                                if(! thisVar.equals(otherVar)){
377                                        continue;
378                                }
379                                
380
381                                //if we get here we know both sets have this variable so this is not case 2 or 3
382                                //System.out.println(">>>> FOUND"+thisVar.toString());
383                                inBoth=true;
384                                break;
385                        }
386                        
387                        if(!inBoth){
388                                //System.out.println("....NOT FOUND ....SET IS:"+this.toString());
389                                //not in both so this is case 2 or 3
390                                /*
391                                 * clone and add if its local
392                                 * 
393                                 * if field then add top
394                                 */
395                                
396                                
397                                /*
398                                 * TODO why should a field be converted to TOP when all the fiels are only constant value fields
399                                 *//*                           if(otherTuple.containsField()){
400                                 //add top
401                                  workingSet.add(new CPTuple(otherTuple.getSootClassName(),otherTuple.getVariable(),true));
402                                  }
403                                  else if(otherTuple.containsLocal()){
404                                  */
405                                
406                                workingSet.add(otherTuple.clone());                                             
407                                /*              }
408                                 else
409                                 throw new DecompilationException("CPVariable is not local and not field");
410                                 */
411                        }                                       
412                }//end going through other elements     
413        }
