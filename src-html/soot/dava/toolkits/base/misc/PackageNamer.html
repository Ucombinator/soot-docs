038    public boolean use_ShortName( String fixedPackageName, String fixedShortClassName){
039        if (fixed == false)
040                return false;
041
042        if (fixedPackageName.equals( Dava.v().get_CurrentPackage()))
043                return true;
044
045        IterableSet packageContext = Dava.v().get_CurrentPackageContext();
046        if (packageContext == null)
047                return true;
048
049        packageContext = patch_PackageContext( packageContext);
050
051        int count = 0;
052        StringTokenizer st = new StringTokenizer( classPath, pathSep);
053        while (st.hasMoreTokens()) {
054                String classpathDir = st.nextToken();
055                
056                Iterator packIt = packageContext.iterator();
057                while (packIt.hasNext()) 
058                        if (package_ContainsClass( classpathDir, (String) packIt.next(), fixedShortClassName))
059                                if (++count > 1)
060                                        return false;
061        }
062
063        return true;
064    }

066    public String get_FixedClassName( String originalFullClassName)
067    {
068        if (fixed == false)
069            return originalFullClassName;
070
071        Iterator<NameHolder> it = appRoots.iterator();
072        while (it.hasNext()) {
073            NameHolder h = it.next();
074            if (h.contains_OriginalName( new StringTokenizer( originalFullClassName, "."), true))
075                return h.get_FixedName( new StringTokenizer( originalFullClassName, "."), true);
076        }
077        
078        return originalFullClassName.substring( originalFullClassName.lastIndexOf( ".") + 1);
079    }

081    public String get_FixedPackageName( String originalPackageName)
082    {
083        if (fixed == false)
084            return originalPackageName;
085
086        if (originalPackageName.equals( ""))
087            return "";
088
089        Iterator<NameHolder> it = appRoots.iterator();
090        while (it.hasNext()) {
091            NameHolder h = it.next();
092            if (h.contains_OriginalName( new StringTokenizer( originalPackageName, "."), false))
093                return h.get_FixedName( new StringTokenizer( originalPackageName, "."), false);
094        }
095
096        return originalPackageName;
097    }

100    private class NameHolder
101    {
102        private final String originalName;
103                private String packageName, className;
104        private final ArrayList<NameHolder> children;
105        private NameHolder parent;
106        private boolean isClass;

109        public NameHolder( String name, NameHolder parent, boolean isClass)
110        {
111                originalName = name;
112                className = name;
113                packageName = name;
114                
115                this.parent = parent;
116                this.isClass = isClass;
117
118                children = new ArrayList<NameHolder>();
119        }

122                        return parent;
123                }

126                        isClass = true;
127                }

130                        if (children.isEmpty())
131                                return true;
132                        else
133                                return isClass;
134                }

136                public boolean is_Package() {
137                        return (children.isEmpty() == false);

141                        return packageName;

145                        return className;

149                        this.packageName = packageName;

157                        return originalName;
158                }

165                        if (parent == null)

172        {
173            if (parent == null)
174                return packageName;

176            return parent.get_FixedPackageName() + "." + packageName;

179        public String get_FixedName( StringTokenizer st, boolean forClass)
180        {
181            if (st.nextToken().equals( originalName) == false)
182                throw new RuntimeException( "Unable to resolve naming.");
183
184            return retrieve_FixedName( st, forClass);
185        }
186
187        private String retrieve_FixedName( StringTokenizer st, boolean forClass)
188        {
189            if (st.hasMoreTokens() == false) {
190                if (forClass)
191                    return className;
192                else
193                    return packageName;
194            }
195
196            String subName = st.nextToken();
197            Iterator<NameHolder> cit = children.iterator();
198            while (cit.hasNext()) {
199                NameHolder h = cit.next();
200
201                if (h.get_OriginalName().equals( subName)) {
202                        if (forClass)
203                                return h.retrieve_FixedName( st, forClass);
204                        else 
205                                return packageName + "." + h.retrieve_FixedName( st, forClass);
206                }
207            }
208            throw new RuntimeException( "Unable to resolve naming.");
209        }
210
211        public String get_OriginalPackageName( StringTokenizer st)
212        {
213            if (st.hasMoreTokens() == false)
214                return get_OriginalName();
215
216            String subName = st.nextToken();
217
218            Iterator<NameHolder> cit = children.iterator();
219            while (cit.hasNext()) {
220                NameHolder h = cit.next();
221
222                if (h.get_PackageName().equals( subName)) {
223                    String originalSubPackageName = h.get_OriginalPackageName( st);
224
225                    if (originalSubPackageName == null)
226                        return null;
227                    else
228                        return get_OriginalName() + "." + originalSubPackageName;
229                }
230            }
231
232            return null;
233        }
234
235        public boolean contains_OriginalName( StringTokenizer st, boolean forClass)
236        {
237            if (get_OriginalName().equals( st.nextToken()) == false)
238                return false;
239
240            return finds_OriginalName( st, forClass);
241        }
242
243        private boolean finds_OriginalName( StringTokenizer st, boolean forClass)
244        {
245            if (st.hasMoreTokens() == false)
246                return (((forClass) && (is_Class())) || ((!forClass) && (is_Package())));
247
248            String subName = st.nextToken();
249            Iterator<NameHolder> cit = children.iterator();
250            while (cit.hasNext()) {
251                NameHolder h = cit.next();
252
253                if (h.get_OriginalName().equals( subName))
254                    return h.finds_OriginalName( st, forClass);
255            }
256            
257            return false;
258        }

260        public void fix_ClassNames( String curPackName)
261        {
262            if ((is_Class()) && (keywords.contains( className))) {
263                String tClassName = className;
264                
265                if (Character.isLowerCase( className.charAt( 0))) {
266                    tClassName = tClassName.substring( 0, 1).toUpperCase() + tClassName.substring( 1);
267                    className = tClassName;
268                }
269                
270                for (int i=0; keywords.contains( className); i++)
271                    className = tClassName + "_c" + i;
272            }
273
274            Iterator<NameHolder> it = children.iterator();
275            while (it.hasNext())
276                it.next().fix_ClassNames( curPackName + "." + packageName);
277        }
278
279        public void fix_PackageNames()
280        {
281            if ((is_Package()) && (verify_PackageName() == false)) {
282                String tPackageName = packageName;
283
284                if (Character.isUpperCase( packageName.charAt( 0))) {
285                    tPackageName = tPackageName.substring( 0, 1).toLowerCase() + tPackageName.substring( 1);
286                    packageName = tPackageName;
287                }
288                
289                for (int i=0; verify_PackageName() == false; i++)
290                    packageName = tPackageName + "_p" + i;
291            }
292
293            Iterator<NameHolder> it = children.iterator();
294            while (it.hasNext())
295                it.next().fix_PackageNames();
296        }

299        public boolean verify_PackageName()
300        {
301            return ((keywords.contains( packageName) == false) && 
302                    (siblingClashes( packageName) == false) &&
303                    ((is_Class() == false) || (className.equals( packageName) == false)));
304        }
305
306        public boolean siblingClashes( String name)
307        {
308            Iterator<NameHolder> it = null;
309
310            if (parent == null) {
311
312                if (appRoots.contains( this))
313                    it = appRoots.iterator();
314                else
315                    throw new RuntimeException( "Unable to find package siblings.");
316            }
317            else 
318                it = parent.get_Children().iterator();
319
320            while (it.hasNext()) {
321                NameHolder sibling = it.next();
322
323                if (sibling == this)
324                    continue;
325
326                if (((sibling.is_Package()) && (sibling.get_PackageName().equals( name))) ||
327                    ((sibling.is_Class()) && (sibling.get_ClassName().equals( name))))
328                    
329                    return true;
330            }
331
332            return false;
333        }

335        public void dump( String indentation)
336        {
337            G.v().out.print( indentation + "\"" + originalName + "\", \"" + packageName + "\", \"" + className + "\" (");
338            if (is_Class())
339                G.v().out.print("c");
340            if (is_Package())
341                G.v().out.print("p");
342            G.v().out.println( ")");
343
344            Iterator<NameHolder> it = children.iterator();
345            while (it.hasNext())
346                it.next().dump( indentation + "  ");
347                }

350    private boolean fixed = false;
351    private final ArrayList<NameHolder> appRoots = new ArrayList<NameHolder>();
352    private final ArrayList<NameHolder> otherRoots = new ArrayList<NameHolder>();
353    private final HashSet<String> keywords = new HashSet<String>();
354    private char fileSep;
355    private String classPath, pathSep;
356    
357    public void fixNames()
358    {
359        if (fixed)
360            return;
361
362        String[] keywordArray =
363        {
364            "abstract",     "default",      "if",            "private",     "this",         "boolean",
365            "do",           "implements",           "protected",            "throw",        "break",
366            "double",       "import",       "public",       "throws",       "byte",         "else",
367            "instanceof",           "return",       "transient",            "case",         "extends",
368            "int",          "short",        "try",          "catch",        "final",        "interface",
369            "static",       "void",             "char",     "finally",      "long",         "strictfp",
370            "volatile",     "class",        "float",        "native",       "super",        "while",
371            "const",        "for",          "new",          "switch",       "continue",     "goto",
372            "package",      "synchronized",         "true",         "false",        "null"
373        };
374
375        for (String element : keywordArray)
376                keywords.add( element);
377
378        Iterator classIt = Scene.v().getLibraryClasses().iterator();
379        while (classIt.hasNext())
380            add_ClassName( ((SootClass) classIt.next()).getName(), otherRoots);
381
382        classIt = Scene.v().getApplicationClasses().iterator();
383        while (classIt.hasNext())
384            add_ClassName( ((SootClass) classIt.next()).getName(), appRoots);
385
386        Iterator<NameHolder> arit = appRoots.iterator();
387        while (arit.hasNext())
388            arit.next().fix_ClassNames( "");
389
390        arit = appRoots.iterator();
391        while (arit.hasNext())
392            arit.next().fix_PackageNames();
393        
394        fileSep = System.getProperty( "file.separator").charAt(0);
395        pathSep = System.getProperty( "path.separator");
396        classPath = System.getProperty( "java.class.path");
397        
398        fixed = true;
399    }

401    private void add_ClassName( String className, ArrayList<NameHolder> roots)
402    {
403        ArrayList<NameHolder> children = roots;
404        NameHolder curNode = null;
405        
406        StringTokenizer st = new StringTokenizer( className, ".");
407        while (st.hasMoreTokens()) {
408            String curName = st.nextToken();
409
410            NameHolder child = null;
411            boolean found = false;
412            Iterator<NameHolder> lit = children.iterator();
413            
414            while (lit.hasNext()) {
415                child = lit.next();
416                
417                if (child.get_OriginalName().equals( curName)) {
418
419                    if (st.hasMoreTokens() == false)
420                        child.set_ClassAttr();
421
422                    found = true;
423                    break;
424                }
425            }
426                
427            if (!found) {
428                child = new NameHolder( curName, curNode, st.hasMoreTokens() == false);
429                children.add( child);
430            }
431            
432            curNode = child;
433            children = child.get_Children();
434        }
435    }

437    public boolean package_ContainsClass( String classpathDir, String packageName, String className)
438    {
439        File p = new File( classpathDir);
440
441        if (p.exists() == false)
442            return false;
443
444        packageName = packageName.replace( '.', fileSep);
445        if ((packageName.length() > 0) && (packageName.charAt( packageName.length() - 1) != fileSep))
446            packageName += fileSep;
447
448        String name = packageName + className + ".class";
449
450        if (p.isDirectory()) {
451            if ((classpathDir.length() > 0) && (classpathDir.charAt( classpathDir.length() - 1) != fileSep))
452                classpathDir += fileSep;
453            
454            return (new File( classpathDir + name)).exists();
455        }
456
457        else {
458            JarFile jf = null;
459
460            try {
461                jf = new JarFile( p);
462            }
463            catch (IOException ioe) {
464                return false;
465            }
466
467            return (jf.getJarEntry( name) != null);
468        }
469    }

471    IterableSet patch_PackageContext( IterableSet currentContext) 
472    {
473        IterableSet newContext = new IterableSet();
474
475        Iterator it = currentContext.iterator();
476        while (it.hasNext()) {
477            String 
478                currentPackage = (String) it.next(),
479                newPackage = null;
480
481            StringTokenizer st = new StringTokenizer( currentPackage, ".");
482            
483            if (st.hasMoreTokens() == false) {
484                newContext.add( currentPackage);
485                continue;
486            }
487
488            String firstToken = st.nextToken();
489            Iterator<NameHolder> arit = appRoots.iterator();
490            while (arit.hasNext()) {
491                NameHolder h = arit.next();
492
493                if (h.get_PackageName().equals( firstToken)) {
494                    newPackage = h.get_OriginalPackageName( st);
495                    break;
496                }
497            }
498            if (newPackage != null)
499                newContext.add( newPackage);
500            else
501                newContext.add( currentPackage);
502        }
503
504        return newContext;
505    }
