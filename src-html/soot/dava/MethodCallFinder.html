088        /*
089         * Notice that since this class is only invoked for clinit methods this
090         * invoke statement is some invocation that occured within the clinit method
091         */
092        public void inInvokeStmt(InvokeStmt s) {
093                InvokeExpr invokeExpr = s.getInvokeExpr();
094                SootMethod maybeInline = invokeExpr.getMethod();
095
096                // check whether we want to inline
097                ASTMethodNode toInlineASTMethod = cleaner.inline(maybeInline);
098                if (toInlineASTMethod == null) {
099                        // not to inline
100                        return;
101                } else {// yes we want to inline
102                                // we know that the method to be inlined has no declarations.
103                        List<Object> subBodies = toInlineASTMethod.get_SubBodies();
104                        if (subBodies.size() != 1) {
105                                throw new RuntimeException(
106                                                "Found ASTMEthod node with more than one subBodies");
107                        }
108                        List body = (List) subBodies.get(0);
109
110                        ASTParentNodeFinder finder = new ASTParentNodeFinder();
111                        underAnalysis.apply(finder);
112
113                        List<ASTStatementSequenceNode> newChangedBodyPart = createChangedBodyPart(
114                                        s, body, finder);
115
116                        boolean replaced = replaceSubBody(s, newChangedBodyPart, finder);
117
118                        if (replaced) {
119                                // so the invoke stmt has been replaced with the body of the
120                                // method invoked
121
122                                /*
123                                 * if the inlined method contained an assignment to a static
124                                 * field we want to replace that with a throw stmt
125                                 */
126                                StaticDefinitionFinder defFinder = new StaticDefinitionFinder(
127                                                maybeInline);
128                                toInlineASTMethod.apply(defFinder);
129
130                                if (defFinder.anyFinalFieldDefined()) {
131                                        // create throw stmt to be added to inlined method
132
133                                        // create a SootMethodRef
134                                        SootClass runtime = Scene.v().loadClassAndSupport(
135                                                        "java.lang.RuntimeException");
136                                        if (runtime.declaresMethod("void <init>(java.lang.String)")) {
137                                                SootMethod sootMethod = runtime
138                                                                .getMethod("void <init>(java.lang.String)");
139                                                SootMethodRef methodRef = sootMethod.makeRef();
140                                                RefType myRefType = RefType.v(runtime);
141                                                StringConstant tempString = StringConstant
142                                                                .v("This method used to have a definition of a final variable. "
143                                                                                + "Dava inlined the definition into the static initializer");
144                                                List list = new ArrayList();
145                                                list.add(tempString);
146
147                                                GNewInvokeExpr newInvokeExpr = new GNewInvokeExpr(
148                                                                myRefType, methodRef, list);
149
150                                                GThrowStmt throwStmt = new GThrowStmt(newInvokeExpr);
151
152                                                AugmentedStmt augStmt = new AugmentedStmt(throwStmt);
153                                                List<AugmentedStmt> sequence = new ArrayList<AugmentedStmt>();
154                                                sequence.add(augStmt);
155                                                ASTStatementSequenceNode seqNode = new ASTStatementSequenceNode(
156                                                                sequence);
157                                                List<Object> subBody = new ArrayList<Object>();
158                                                subBody.add(seqNode);
159
160                                                toInlineASTMethod.replaceBody(subBody);
161                                        }
162                                }
163                        }
164
165                }
166        }

168        public List<Object> getSubBodyFromSingleSubBodyNode(ASTNode node) {
169                List<Object> subBodies = node.get_SubBodies();
170                if (subBodies.size() != 1)
171                        throw new RuntimeException(
172                                        "Found a single subBody node with more than 1 subBodies");
173
174                return (List<Object>) subBodies.get(0);
175        }

177        public List<Object> createNewSubBody(List<Object> orignalBody,
178                        List<ASTStatementSequenceNode> partNewBody, Object stmtSeqNode) {
179
180                List<Object> newBody = new ArrayList<Object>();
181
182                Iterator<Object> it = orignalBody.iterator();
183                while (it.hasNext()) {
184                        Object temp = it.next();
185                        if (temp != stmtSeqNode)
186                                newBody.add(temp);
187                        else {
188                                // breaks out of the loop as soon as stmtSeqNode is reached
189                                break;
190                        }
191                }
192                // dont add this stmt sequence node instead add the modified stmt seq
193                // nodes and the to be inline method
194                newBody.addAll(partNewBody);
195
196                // add remaining stuff drom the orignalBody
197                while (it.hasNext()) {
198                        newBody.add(it.next());
199                }
200
201                return newBody;
202        }

204        public boolean replaceSubBody(InvokeStmt s,
205                        List<ASTStatementSequenceNode> newChangedBodyPart,
206                        ASTParentNodeFinder finder) {
207
208                // get the stmt seq node of invoke stmt
209                Object stmtSeqNode = finder.getParentOf(s);
210
211                // find the parent node of the stmt seq node
212                Object ParentOfStmtSeq = finder.getParentOf(stmtSeqNode);
213
214                if (ParentOfStmtSeq == null) {
215                        throw new RuntimeException(
216                                        "MethodCall FInder: parent of stmt seq node not found");
217                }
218
219                ASTNode node = (ASTNode) ParentOfStmtSeq;
220
221                // the decision what to replace and how to replace depends on the type
222                // of ASTNode
223
224                if (node instanceof ASTMethodNode) {
225                        // get the subBody to replace
226                        List<Object> subBodyToReplace = getSubBodyFromSingleSubBodyNode(node);
227                        List<Object> newBody = createNewSubBody(subBodyToReplace,
228                                        newChangedBodyPart, stmtSeqNode);
229                        ((ASTMethodNode) node).replaceBody(newBody);
230                        return true;
231                } else if (node instanceof ASTSynchronizedBlockNode) {
232                        // get the subBody to replace
233                        List<Object> subBodyToReplace = getSubBodyFromSingleSubBodyNode(node);
234                        List<Object> newBody = createNewSubBody(subBodyToReplace,
235                                        newChangedBodyPart, stmtSeqNode);
236                        ((ASTSynchronizedBlockNode) node).replaceBody(newBody);
237                        return true;
238                } else if (node instanceof ASTLabeledBlockNode) {
239                        // get the subBody to replace
240                        List<Object> subBodyToReplace = getSubBodyFromSingleSubBodyNode(node);
241                        List<Object> newBody = createNewSubBody(subBodyToReplace,
242                                        newChangedBodyPart, stmtSeqNode);
243                        ((ASTLabeledBlockNode) node).replaceBody(newBody);
244                        return true;
245                } else if (node instanceof ASTUnconditionalLoopNode) {
246                        // get the subBody to replace
247                        List<Object> subBodyToReplace = getSubBodyFromSingleSubBodyNode(node);
248                        List<Object> newBody = createNewSubBody(subBodyToReplace,
249                                        newChangedBodyPart, stmtSeqNode);
250                        ((ASTUnconditionalLoopNode) node).replaceBody(newBody);
251                        return true;
252                } else if (node instanceof ASTIfNode) {
253                        // get the subBody to replace
254                        List<Object> subBodyToReplace = getSubBodyFromSingleSubBodyNode(node);
255                        List<Object> newBody = createNewSubBody(subBodyToReplace,
256                                        newChangedBodyPart, stmtSeqNode);
257                        ((ASTIfNode) node).replaceBody(newBody);
258                        return true;
259                } else if (node instanceof ASTWhileNode) {
260                        // get the subBody to replace
261                        List<Object> subBodyToReplace = getSubBodyFromSingleSubBodyNode(node);
262                        List<Object> newBody = createNewSubBody(subBodyToReplace,
263                                        newChangedBodyPart, stmtSeqNode);
264                        ((ASTWhileNode) node).replaceBody(newBody);
265                        return true;
266                } else if (node instanceof ASTDoWhileNode) {
267                        // get the subBody to replace
268                        List<Object> subBodyToReplace = getSubBodyFromSingleSubBodyNode(node);
269                        List<Object> newBody = createNewSubBody(subBodyToReplace,
270                                        newChangedBodyPart, stmtSeqNode);
271                        ((ASTDoWhileNode) node).replaceBody(newBody);
272                        return true;
273                } else if (node instanceof ASTForLoopNode) {
274                        // get the subBody to replace
275                        List<Object> subBodyToReplace = getSubBodyFromSingleSubBodyNode(node);
276                        List<Object> newBody = createNewSubBody(subBodyToReplace,
277                                        newChangedBodyPart, stmtSeqNode);
278                        ((ASTForLoopNode) node).replaceBody(newBody);
279                        return true;
280                } else if (node instanceof ASTIfElseNode) {
281                        List<Object> subBodies = node.get_SubBodies();
282                        if (subBodies.size() != 2)
283                                throw new RuntimeException(
284                                                "Found an ifelse ASTNode which does not have two bodies");
285                        List<Object> ifBody = (List<Object>) subBodies.get(0);
286                        List<Object> elseBody = (List<Object>) subBodies.get(1);
287
288                        // find out which of these bodies has the stmt seq node with the
289                        // invoke stmt
290                        int subBodyNumber = -1;
291                        Iterator<Object> it = ifBody.iterator();
292                        while (it.hasNext()) {
293                                Object temp = it.next();
294                                if (temp == stmtSeqNode) {
295                                        subBodyNumber = 0;
296                                        break;
297                                }
298                        }
299                        if (subBodyNumber != 0) {
300                                it = elseBody.iterator();
301                                while (it.hasNext()) {
302                                        Object temp = it.next();
303                                        if (temp == stmtSeqNode) {
304                                                subBodyNumber = 1;
305                                                break;
306                                        }
307                                }
308                        }
309
310                        List<Object> subBodyToReplace = null;
311                        if (subBodyNumber == 0)
312                                subBodyToReplace = ifBody;
313                        else if (subBodyNumber == 1)
314                                subBodyToReplace = elseBody;
315                        else
316                                throw new RuntimeException(
317                                                "Could not find the related ASTNode in the method");
318
319                        List<Object> newBody = createNewSubBody(subBodyToReplace,
320                                        newChangedBodyPart, stmtSeqNode);
321
322                        if (subBodyNumber == 0) {
323                                ((ASTIfElseNode) node).replaceBody(newBody, elseBody);
324                                return true;
325                        } else if (subBodyNumber == 1) {
326                                ((ASTIfElseNode) node).replaceBody(ifBody, newBody);
327                                return true;
328                        }
329                } else if (node instanceof ASTTryNode) {
330
331                        // NOTE THAT method INLINING Is currently only done in the tryBody
332                        // and not the catchBody
333                        // THe only reason for this being that mostly method calls are made
334                        // in the try and not the catch
335
336                        // get try body
337                        List<Object> tryBody = ((ASTTryNode) node).get_TryBody();
338                        Iterator<Object> it = tryBody.iterator();
339
340                        // find whether stmtSeqNode is in the tryBody
341                        boolean inTryBody = false;
342                        while (it.hasNext()) {
343                                ASTNode temp = (ASTNode) it.next();
344                                if (temp == stmtSeqNode) {
345                                        inTryBody = true;
346                                        break;
347                                }
348                        }
349                        if (!inTryBody) {
350                                // return without making any changes
351                                return false;
352                        }
353
354                        List<Object> newBody = createNewSubBody(tryBody,
355                                        newChangedBodyPart, stmtSeqNode);
356                        ((ASTTryNode) node).replaceTryBody(newBody);
357                        return true;
358                } else if (node instanceof ASTSwitchNode) {
359
360                        List<Object> indexList = ((ASTSwitchNode) node).getIndexList();
361                        Map<Object, List<Object>> index2BodyList = ((ASTSwitchNode) node)
362                                        .getIndex2BodyList();
363
364                        Iterator<Object> it = indexList.iterator();
365                        while (it.hasNext()) {// going through all the cases of the switch
366                                                                        // statement
367                                Object currentIndex = it.next();
368                                List<Object> body = index2BodyList.get(currentIndex);
369
370                                if (body != null) {
371                                        // this body is a list of ASTNodes
372
373                                        // see if it contains stmtSeqNode
374                                        boolean found = false;
375                                        Iterator<Object> itBody = body.iterator();
376                                        while (itBody.hasNext()) {
377                                                ASTNode temp = (ASTNode) itBody.next();
378                                                if (temp == stmtSeqNode) {
379                                                        found = true;
380                                                        break;
381                                                }
382                                        }
383                                        if (found) {
384                                                // this is the body which has the stmt seq node
385                                                List<Object> newBody = createNewSubBody(body,
386                                                                newChangedBodyPart, stmtSeqNode);
387
388                                                // put this body in the Map
389                                                index2BodyList.put(currentIndex, newBody);
390                                                // replace in actual switchNode
391                                                ((ASTSwitchNode) node)
392                                                                .replaceIndex2BodyList(index2BodyList);
393                                                return true;
394                                        }
395                                }// if body not null
396                        }// going through all cases
397                }
398                return false;
399        }

401        /*
402         * Given an invoke stmt this method finds the parent of this stmt which
403         * should always be a StatementSequenceNode Then the sequence is broken into
404         * three parts. The first part contains stmts till above the invoke stmt.
405         * The second part contains the body argument which is the body of the
406         * inlined method and the third part are the stmts below the invoke stmt
407         */
408
409        public List<ASTStatementSequenceNode> createChangedBodyPart(InvokeStmt s,
410                        List body, ASTParentNodeFinder finder) {
411                // get parent node of invoke stmt
412                Object parent = finder.getParentOf(s);
413                if (parent == null) {
414                        throw new RuntimeException(
415                                        "MethodCall FInder: parent of invoke stmt not found");
416                }
417
418                ASTNode parentNode = (ASTNode) parent;
419                if (!(parentNode instanceof ASTStatementSequenceNode)) {
420                        throw new RuntimeException(
421                                        "MethodCall FInder: parent node not a stmt seq node");
422                }
423
424                ASTStatementSequenceNode orignal = (ASTStatementSequenceNode) parentNode;
425
426                // copying the stmts till above the inoke stmt into one stmt sequence
427                // node
428                List<AugmentedStmt> newInitialNode = new ArrayList<AugmentedStmt>();
429                Iterator<AugmentedStmt> it = orignal.getStatements().iterator();
430                while (it.hasNext()) {
431                        AugmentedStmt as = it.next();
432                        Stmt tempStmt = as.get_Stmt();
433                        if (tempStmt != s) {
434                                newInitialNode.add(as);
435                        } else {
436                                // the first time we get to a stmt which points to the invoke
437                                // stmt we break
438                                break;
439                        }
440                }
441
442                // copy remaining stmts into the AFTER stmt sequence node
443                List<AugmentedStmt> newSecondNode = new ArrayList<AugmentedStmt>();
444                while (it.hasNext()) {
445                        newSecondNode.add(it.next());
446                }
447
448                List<ASTStatementSequenceNode> toReturn = new ArrayList<ASTStatementSequenceNode>();
449
450                if (newInitialNode.size() != 0)
451                        toReturn.add(new ASTStatementSequenceNode(newInitialNode));
452
453                // add inline methods body
454                toReturn.addAll(body);
455
456                if (newSecondNode.size() != 0)
457                        toReturn.add(new ASTStatementSequenceNode(newSecondNode));
458
459                return toReturn;
460        }
