067    private void printStatementsInBody(Body body, java.io.PrintWriter out) {
068      
069      if (Options.v().verbose())
070        System.out.println("Printing "+body.getMethod().getName());
071      
072        Chain<Unit> units = ((DavaBody) body).getUnits();
073
074        if (units.size() != 1) {
075            throw new RuntimeException("DavaBody AST doesn't have single root.");
076        }
077
078        UnitPrinter up = new DavaUnitPrinter((DavaBody)body);
079        ((ASTNode) units.getFirst()).toString(up);
080        out.print( up.toString() );
081    }

083    public void printTo(SootClass cl, PrintWriter out) {
084        
085        //IterableSet packagesUsed = new IterableSet();
086        IterableSet importList = new IterableSet();
087        {
088            String curPackage = cl.getJavaPackageName();
089
090            if (!curPackage.equals("")) {
091                out.println("package " + curPackage + ";");
092                out.println();
093            }
094
095            if (cl.hasSuperclass()) {
096                SootClass superClass = cl.getSuperclass();
097                importList.add(superClass.toString());
098                //packagesUsed.add(superClass.getJavaPackageName());
099            }
100
101            Iterator<SootClass> interfaceIt = cl.getInterfaces().iterator();
102            while (interfaceIt.hasNext()) {
103                String interfacePackage = ((SootClass) interfaceIt.next()).toString();
104                
105                if(!importList.contains(interfacePackage))
106                        importList.add(interfacePackage);
107                        
108                //if (!packagesUsed.contains(interfacePackage))
109                  //  packagesUsed.add(interfacePackage);
110            }
111
112            Iterator<SootMethod> methodIt = cl.methodIterator();
113            while (methodIt.hasNext()) {
114                SootMethod dm = (SootMethod) methodIt.next();
115
116                if (dm.hasActiveBody()){
117                        //packagesUsed = packagesUsed.union(((DavaBody) dm.getActiveBody()).get_PackagesUsed());
118                    importList = importList.union(((DavaBody) dm.getActiveBody()).getImportList());
119                }
120
121                Iterator<SootClass> eit = dm.getExceptions().iterator();
122                while (eit.hasNext()) {
123                    String thrownPackage =eit.next().toString();
124                    if(!importList.contains(thrownPackage))
125                        importList.add(thrownPackage);
126                    
127                    //if (!packagesUsed.contains(thrownPackage))
128                      //  packagesUsed.add(thrownPackage);
129                }
130
131                Iterator<Type> pit = dm.getParameterTypes().iterator();
132                while (pit.hasNext()) {
133                    Type t = (Type) pit.next();
134
135                    if (t instanceof RefType) {
136                        String paramPackage = ((RefType) t).getSootClass().toString();
137                        
138                        if (!importList.contains(paramPackage))
139                            importList.add(paramPackage);
140                        
141                        //if (packagesUsed.contains(paramPackage) == false)
142                          //  packagesUsed.add(paramPackage);
143                    }
144                }
145
146                Type t = dm.getReturnType();
147                if (t instanceof RefType) {
148                    String returnPackage = ((RefType) t).getSootClass().toString();
149                    
150                    if (!importList.contains(returnPackage))
151                        importList.add(returnPackage);
152
153                
154                    //if (packagesUsed.contains(returnPackage) == false)
155                      //  packagesUsed.add(returnPackage);
156                }
157            }
158
159            Iterator<SootField> fieldIt = cl.getFields().iterator();
160            while (fieldIt.hasNext()) {
161                SootField f = (SootField) fieldIt.next();
162
163                if (f.isPhantom())
164                    continue;
165
166                Type t = f.getType();
167
168                if (t instanceof RefType) {
169                    String fieldPackage = ((RefType) t).getSootClass().toString();
170                    
171                    if (!importList.contains(fieldPackage))
172                        importList.add(fieldPackage);
173                }
174            }
175
176
177            Iterator<String> pit = importList.iterator();
178            List<String> toImport = new ArrayList<String>();
179            while (pit.hasNext()){
180                /*
181                 * dont import any file which has currentPackage.className
182                 * dont import any file which starts with java.lang
183                 */
184                String temp = (String)pit.next();
185                //System.out.println("temp is "+temp);
186                if(temp.indexOf("java.lang")>-1 ){
187                        //problem is that we need to import sub packages java.lang.ref 
188                        //for instance if the type is java.lang.ref.WeakReference
189                        String tempClassName = RemoveFullyQualifiedName.getClassName(temp);
190                        if(temp.equals("java.lang."+tempClassName)){
191                                //System.out.println("temp was not printed as it belongs to java.lang");
192                                continue;
193                        }
194                }
195
196                if(curPackage.length()>0 && temp.indexOf(curPackage)>-1){
197                        //System.out.println("here  "+temp);
198                        continue;
199                }
200                
201                if(cl.toString().equals(temp))
202                        continue;
203                               
204                
205                //System.out.println("printing"+);
206                toImport.add(temp);
207
208
209            }
210
211            /*
212             * Check that we are not importing two classes with the same last name
213             * If yes then remove explicit import and import the whole package
214             * else output explicit import statement
215             */
216            Iterator it = toImport.iterator();
217            while(it.hasNext()){
218                String temp = (String)it.next();
219                if(RemoveFullyQualifiedName.containsMultiple(toImport.iterator(),temp,null)){
220                        //there are atleast two imports with this className
221                        //import package add *
222                        if(temp.lastIndexOf('.')>-1){
223                                temp = temp.substring(0,temp.lastIndexOf('.'));
224                                out.println("import " + temp + ".*;");
225                        }
226                        else
227                                throw new DecompilationException("Cant find the DOT . for fullyqualified name");
228                }
229                else{
230                        if(temp.lastIndexOf('.')==-1){
231                                //dot not found this is a class belonging to this package so dont add
232                        }
233                        else
234                                out.println("import " + temp + ";");
235                }
236            }
237            boolean addNewLine=false;
238            addNewLine=true;
239            
240           // out.println("import " + temp + ";");
241            
242            
243            if(addNewLine)
244                out.println();
245            
246            /*if (!packagesUsed.isEmpty())
247                out.println();
248
249            packagesUsed.add("java.lang");
250            packagesUsed.add(curPackage);
251            */
252            Dava.v().set_CurrentPackageContext(importList);
253            //Dava.v().set_CurrentPackageContext(packagesUsed);
254            Dava.v().set_CurrentPackage(curPackage);
255        }
256
257        // Print class name + modifiers
258        {
259            String classPrefix = "";
260
261            classPrefix =
262                classPrefix + " " + Modifier.toString(cl.getModifiers());
263            classPrefix = classPrefix.trim();
264
265            if (!cl.isInterface()) {
266                classPrefix = classPrefix + " class";
267                classPrefix = classPrefix.trim();
268            }
269
270            out.print(classPrefix + " " + cl.getShortJavaStyleName());
271        }
272
273        // Print extension
274        if (cl.hasSuperclass()
275            && !(cl.getSuperclass().getName().equals("java.lang.Object"))){
276                        
277                String superClassName = cl.getSuperclass().getName();
278                
279                //Nomair Naeem 8th Feb 2006
280                //also check if the super class name is not a fully qualified
281                //name. in which case if the package is imported no need for
282                //the long name
283
284                superClassName = RemoveFullyQualifiedName.getReducedName(importList,superClassName,cl.getType());
285            out.print(" extends " + superClassName + "");
286        }
287
288
289        // Print interfaces
290        {
291            Iterator<SootClass> interfaceIt = cl.getInterfaces().iterator();
292
293            if (interfaceIt.hasNext()) {
294                if( cl.isInterface() ) out.print(" extends ");
295                else out.print(" implements ");
296
297                out.print("" + (interfaceIt.next()).getName() + "");
298
299                while (interfaceIt.hasNext())
300                    out.print(
301                        ", " + (interfaceIt.next()).getName() + "");
302            }
303        }
304
305        out.println();
306        out.println("{");
307
308        // Print fields
309        {
310                        Iterator<SootField> fieldIt = cl.getFields().iterator();
311                        if (fieldIt.hasNext()) {
312                                while (fieldIt.hasNext()) {
313                                        SootField f = fieldIt.next();
314
315                                        if (f.isPhantom())
316                                                continue;
317
318
319                                        String declaration = null;
320                                        
321                                        Type fieldType = f.getType();
322                                        
323                                
324                                String qualifiers = Modifier.toString(f.getModifiers()) + " ";
325                                
326                                qualifiers += RemoveFullyQualifiedName.getReducedName(importList,fieldType.toString(),fieldType); 
327                                
328                                qualifiers = qualifiers.trim();
329
330                                if(qualifiers.equals(""))
331                                    declaration =  Scene.v().quotedNameOf(f.getName());
332                                else
333                                    declaration = qualifiers + " " + Scene.v().quotedNameOf(f.getName()) + "";
334
335
336                                if (f.isFinal() && f.isStatic()) {
337                                                                                
338                                                if (fieldType instanceof DoubleType && f.hasTag("DoubleConstantValueTag")) {
339                                                        
340                                                        double val = ((DoubleConstantValueTag) f.getTag("DoubleConstantValueTag")).getDoubleValue();
341                                                        out.println("    " + declaration + " = "+ val + ";");
342                                                        
343                                                } else if (fieldType instanceof FloatType && f.hasTag("FloatConstantValueTag")) {
344                                                        
345                                                        float val = ((FloatConstantValueTag) f.getTag("FloatConstantValueTag")).getFloatValue();
346                                                        out.println("    " + declaration + " = "+ val + "f;");
347                                                        
348                                                } else if (fieldType instanceof LongType && f.hasTag("LongConstantValueTag")) {
349
350                                                        long val = ((LongConstantValueTag) f.getTag("LongConstantValueTag")).getLongValue();
351                                                        out.println("    " + declaration + " = "+ val + "l;");
352                                                        
353                                                } else if (fieldType instanceof CharType && f.hasTag("IntegerConstantValueTag")) {
354
355                                                        int val = ((IntegerConstantValueTag) f.getTag("IntegerConstantValueTag")).getIntValue();
356                                                        out.println("    " + declaration + " = '" + ((char) val) + "';");
357
358                                                } else if (fieldType instanceof BooleanType && f.hasTag("IntegerConstantValueTag")) {
359                                                        
360                                                        int val = ((IntegerConstantValueTag) f.getTag("IntegerConstantValueTag")).getIntValue();
361
362                                                        if (val == 0)
363                                                                out.println("    " + declaration+ " = false;");
364                                                        else
365                                                                out.println("    " + declaration+ " = true;");
366                                                        
367                                                } else if ((fieldType instanceof IntType
368                                                                || fieldType instanceof ByteType || 
369                                                                fieldType instanceof ShortType)
370                                                                && f.hasTag("IntegerConstantValueTag")) {
371                                                        
372                                                        int val = ((IntegerConstantValueTag) f.getTag("IntegerConstantValueTag")).getIntValue();
373                                                        out.println("    " + declaration + " = "+ val + ";");
374                                                        
375                                                } else if (f.hasTag("StringConstantValueTag")) {
376
377                                                        String val = ((StringConstantValueTag) f.getTag("StringConstantValueTag")).getStringValue();
378                                                        out.println("    " + declaration + " = \""+ val + "\";");
379                                                        
380                                                } else {
381                                                        // System.out.println("Couldnt find type of
382                                                        // field"+f.getDeclaration());
383                                                        out.println("    " + declaration + ";");
384                                                }
385                                        } // field is static final
386                                        else {
387                                                out.println("    " + declaration + ";");
388                                        }
389                                }
390                        }
391                }
392
393        // Print methods
394        {
395            Iterator<SootMethod> methodIt = cl.methodIterator();
396
397            if (methodIt.hasNext()) {
398                if (cl.getMethodCount() != 0)
399                    out.println();
400
401                while (methodIt.hasNext()) {
402                    SootMethod method = (SootMethod) methodIt.next();
403
404                    if (method.isPhantom())
405                        continue;
406
407                    if (!Modifier.isAbstract(method.getModifiers())
408                        && !Modifier.isNative(method.getModifiers())) {
409                        if (!method.hasActiveBody())
410                            throw new RuntimeException(
411                                "method "
412                                    + method.getName()
413                                    + " has no active body!");
414                        else
415                            printTo(method.getActiveBody(), out);
416
417                        if (methodIt.hasNext())
418                            out.println();
419                    } else {
420                        //if method is abstract then print the declaration
421                        out.print("    ");
422                        out.print(method.getDavaDeclaration());
423                        out.println(";");
424
425                        if (methodIt.hasNext())
426                            out.println();
427                    }
428                }
429            }
430        }
431
432
433
434
435        /*
436         * January 23rd, 2006
437         * In trying to handle the suepr class problem we need to introduce an inner class
438         * Instead of creating a data structure for it we are right now just going to print it in the form
439         * of a string
440         *
441         * It would be interesting to later have an internal inner class structure so that we could 
442         * decompile inner classes into inner classes
443         */
444
445        if(G.v().SootClassNeedsDavaSuperHandlerClass.contains(cl)){
446            out.println("\n    private static class DavaSuperHandler{");
447            out.println("         java.util.Vector myVector = new java.util.Vector();");
448
449            out.println("\n         public Object get(int pos){");
450            out.println("            return myVector.elementAt(pos);");
451            out.println("         }");
452
453            out.println("\n         public void store(Object obj){");
454            out.println("            myVector.add(obj);");
455            out.println("         }");
456            out.println("    }");
457        }
458
459
460        out.println("}");
461    }

468    /**
469     *   Prints out the method corresponding to b Body, (declaration and body),
470     *   in the textual format corresponding to the IR used to encode b body.
471     *
472     *   @param out a PrintWriter instance to print to.
473     */
474    private void printTo(Body b, PrintWriter out) {
475        b.validate();
476
477        String decl = b.getMethod().getDavaDeclaration();
478
479        {
480            out.println("    " + decl);
481            for( Iterator<Tag> tIt = b.getMethod().getTags().iterator(); tIt.hasNext(); ) {
482                final Tag t = tIt.next();
483                if (Options.v().print_tags_in_output()){
484                    out.println(t);
485                }
486            }
487            out.println("    {");
488
489
490            /*
491              The locals are now printed out from within the toString method of ASTMethodNode
492              Nomair A Naeem 10-MARCH-2005
493            */
494            //printLocalsInBody(b, out);
495        }
496
497        printStatementsInBody(b, out);
498
499        out.println("    }");
500
501    }
