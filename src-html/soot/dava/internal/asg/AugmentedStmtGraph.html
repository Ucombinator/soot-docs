052        public AugmentedStmtGraph(AugmentedStmtGraph other) {
053                this();
054
055                HashMap<AugmentedStmt, AugmentedStmt> old2new = new HashMap<AugmentedStmt, AugmentedStmt>();
056
057                for (AugmentedStmt oas : other.aug_list) {
058                        Stmt s = oas.get_Stmt();
059
060                        AugmentedStmt nas = new AugmentedStmt(s);
061                        aug_list.add(nas);
062                        stmt_list.add(s);
063                        binding.put(s, nas);
064
065                        old2new.put(oas, nas);
066                }
067
068                for (AugmentedStmt oas : other.aug_list) {
069                        AugmentedStmt nas = (AugmentedStmt) old2new.get(oas);
070
071                        for (AugmentedStmt aug : oas.bpreds)
072                                nas.bpreds.add(old2new.get(aug));
073                        if (nas.bpreds.isEmpty())
074                                bheads.add(nas);
075
076                        for (AugmentedStmt aug : oas.cpreds)
077                                nas.cpreds.add(old2new.get(aug));
078                        if (nas.cpreds.isEmpty())
079                                cheads.add(nas);
080
081                        for (AugmentedStmt aug : oas.bsuccs)
082                                nas.bsuccs.add(old2new.get(aug));
083                        if (nas.bsuccs.isEmpty())
084                                btails.add(nas);
085
086                        for (AugmentedStmt aug : oas.csuccs)
087                                nas.csuccs.add(old2new.get(aug));
088                        if (nas.csuccs.isEmpty())
089                                ctails.add(nas);
090                }
091
092                find_Dominators();
093        }

095        public AugmentedStmtGraph(BriefUnitGraph bug, TrapUnitGraph cug) {
096                this();
097
098                Dava.v().log(
099                                "AugmentedStmtGraph::AugmentedStmtGraph() - cug.size() = "
100                                                + cug.size());
101
102                // make the augmented statements
103                for (Unit u : cug) {
104                        Stmt s = (Stmt) u;
105                        add_StmtBinding(s, new AugmentedStmt(s));
106                }
107
108                // make the list of augmented statements in pseudo topological order!
109                List<Unit> cugList = (new PseudoTopologicalOrderer<Unit>()).newList(cug,
110                                false);
111                for (Unit u : cugList) {
112                        Stmt s = (Stmt) u;
113                        aug_list.add(get_AugStmt(s));
114                        stmt_list.add(s);
115                }
116
117                // now that we've got all the augmented statements, mirror the statement
118                // graph
119                for (AugmentedStmt as : aug_list) {
120                        mirror_PredsSuccs(as, bug);
121                        mirror_PredsSuccs(as, cug);
122                }
123
124                find_Dominators();
125        }

139        public void add_AugmentedStmt(AugmentedStmt as) {
140                Stmt s = as.get_Stmt();
141
142                aug_list.add(as);
143                stmt_list.add(s);
144
145                add_StmtBinding(s, as);
146
147                if (as.bpreds.isEmpty())
148                        bheads.add(as);
149
150                if (as.cpreds.isEmpty())
151                        cheads.add(as);
152
153                if (as.bsuccs.isEmpty())
154                        btails.add(as);
155
156                if (as.csuccs.isEmpty())
157                        ctails.add(as);
158
159                check_List(as.bpreds, btails);
160                check_List(as.bsuccs, bheads);
161                check_List(as.cpreds, ctails);
162                check_List(as.csuccs, cheads);
163        }

177        private <T> void check_List(List<T> psList, List<T> htList) {
178                for (T t : psList) {
179                        if (htList.contains(t))

181                }

184        public void calculate_Reachability(AugmentedStmt source,
185                        Set<AugmentedStmt> blockers, AugmentedStmt dominator) {
186                if (blockers == null)
187                        throw new RuntimeException(
188                                        "Tried to call AugmentedStmtGraph:calculate_Reachability() with null blockers.");
189
190                if (source == null)
191                        return;
192
193                LinkedList<AugmentedStmt> worklist = new LinkedList<AugmentedStmt>();
194                HashSet<AugmentedStmt> touchSet = new HashSet<AugmentedStmt>();
195
196                worklist.addLast(source);
197                touchSet.add(source);
198
199                while (worklist.isEmpty() == false) {
200                        AugmentedStmt as = worklist.removeFirst();
201
202                        for (AugmentedStmt sas : as.csuccs) {
203                                if ((touchSet.contains(sas))
204                                                || (sas.get_Dominators().contains(dominator) == false))
205                                        continue;
206
207                                touchSet.add(sas);
208
209                                IterableSet<AugmentedStmt> reachers = sas.get_Reachers();
210
211                                if (reachers.contains(source) == false)
212                                        reachers.add(source);
213
214                                if (blockers.contains(sas) == false)
215                                        worklist.addLast(sas);
216                        }
217                }
218        }

220        public void calculate_Reachability(Collection<AugmentedStmt> sources,
221                        Set<AugmentedStmt> blockers, AugmentedStmt dominator) {
222                Iterator<AugmentedStmt> srcIt = sources.iterator();
223                while (srcIt.hasNext())
224                        calculate_Reachability(srcIt.next(), blockers,
225                                        dominator);
226        }

330        public boolean remove_AugmentedStmt(AugmentedStmt toRemove) {
331                if (aug_list.contains(toRemove) == false)
332                        return false;
333
334                for (AugmentedStmt pas : toRemove.bpreds) {
335                        if (pas.bsuccs.contains(toRemove))
336                                pas.bsuccs.remove(toRemove);
337                }
338
339                for (AugmentedStmt pas : toRemove.cpreds) {
340                        if (pas.csuccs.contains(toRemove))
341                                pas.csuccs.remove(toRemove);
342                }
343
344                for (AugmentedStmt sas : toRemove.bsuccs) {
345                        if (sas.bpreds.contains(toRemove))
346                                sas.bpreds.remove(toRemove);
347                }
348
349                for (AugmentedStmt sas : toRemove.csuccs) {
350                        if (sas.cpreds.contains(toRemove))
351                                sas.cpreds.remove(toRemove);
352                }
353
354                aug_list.remove(toRemove);
355                stmt_list.remove(toRemove.get_Stmt());
356
357                bheads.remove(toRemove);
358                btails.remove(toRemove);
359                cheads.remove(toRemove);
360                ctails.remove(toRemove);
361
362                binding.remove(toRemove.get_Stmt());
363
364                return true;
365
366                // NOTE: we do *NOT* touch the underlying unit graphs.
367        }

369        public String toString() {
370                StringBuffer b = new StringBuffer();
371                String cr = "\n";
372
373                b.append("AugmentedStmtGraph (size: " + size() + " stmts)" + cr);
374
375                for (AugmentedStmt as : aug_list) {
376                        b.append("| .---" + cr + "| | AugmentedStmt " + as.toString() + cr
377                                        + "| |" + cr + "| |  preds:");
378
379                        for (AugmentedStmt pas : as.cpreds) {
380                                b.append(" " + pas.toString());
381                        }
382
383                        b.append(cr + "| |" + cr + "| |  succs:");
384                        for (AugmentedStmt sas : as.csuccs) {
385                                b.append(" " + sas.toString());
386                        }
387
388                        b.append(cr + "| |" + cr + "| |  doms:");
389                        for (AugmentedStmt das : as.get_Dominators()) {
390                                b.append(" " + das.toString());
391                        }
392
393                        b.append(cr + "| `---" + cr);
394                }
395
396                b.append("-" + cr);
397                return b.toString();
398        }

400        private void mirror_PredsSuccs(AugmentedStmt as, UnitGraph ug) {
401                Stmt s = as.get_Stmt();
402
403                LinkedList<AugmentedStmt> preds = new LinkedList<AugmentedStmt>(), succs = new LinkedList<AugmentedStmt>();
404
405                // mirror the predecessors
406                for (Unit u : ug.getPredsOf(s)) {
407                        AugmentedStmt po = get_AugStmt((Stmt) u);
408                        if (preds.contains(po) == false)
409                                preds.add(po);
410                }
411
412                // mirror the successors
413                for (Unit u : ug.getSuccsOf(s)) {
414                        AugmentedStmt so = get_AugStmt((Stmt) u);
415                        if (succs.contains(so) == false)
416                                succs.add(so);
417                }
418
419                // attach the mirrors properly to the AugmentedStmt
420                if (ug instanceof BriefUnitGraph) {
421                        as.bpreds = preds;
422                        as.bsuccs = succs;
423
424                        if (preds.size() == 0)
425                                bheads.add(as);
426                        if (succs.size() == 0)
427                                btails.add(as);
428                } else if (ug instanceof TrapUnitGraph) {
429                        as.cpreds = preds;
430                        as.csuccs = succs;
431
432                        if (preds.size() == 0)
433                                cheads.add(as);
434                        if (succs.size() == 0)
435                                ctails.add(as);
436                } else
437                        throw new RuntimeException("Unknown UnitGraph type: "
438                                        + ug.getClass());
439        }

441        public IterableSet<AugmentedStmt> clone_Body(IterableSet<AugmentedStmt> oldBody) {
442                HashMap<AugmentedStmt, AugmentedStmt> old2new = new HashMap<AugmentedStmt, AugmentedStmt>(),
443                                new2old = new HashMap<AugmentedStmt, AugmentedStmt>();
444
445                IterableSet<AugmentedStmt> newBody = new IterableSet<AugmentedStmt>();
446
447                for (AugmentedStmt as : oldBody) {
448                        AugmentedStmt clone = (AugmentedStmt) as.clone();
449
450                        original2clone.put(as, clone);
451
452                        old2new.put(as, clone);
453                        new2old.put(clone, as);
454                        newBody.add(clone);
455                }
456
457                for (AugmentedStmt newAs : newBody) {
458                        AugmentedStmt oldAs = (AugmentedStmt) new2old.get(newAs);
459
460                        mirror_PredsSuccs(oldAs, oldAs.bpreds, newAs.bpreds, old2new);
461                        mirror_PredsSuccs(oldAs, oldAs.cpreds, newAs.cpreds, old2new);
462                        mirror_PredsSuccs(oldAs, oldAs.bsuccs, newAs.bsuccs, old2new);
463                        mirror_PredsSuccs(oldAs, oldAs.csuccs, newAs.csuccs, old2new);
464                }
465
466                for (AugmentedStmt au : newBody)
467                        add_AugmentedStmt(au);
468
469                HashMap<Stmt, Stmt> so2n = new HashMap<Stmt, Stmt>();
470                
471                for (AugmentedStmt as : oldBody) {
472                        Stmt os = as.get_Stmt();
473                        Stmt ns = old2new.get(as).get_Stmt();
474
475                        so2n.put(os, ns);
476                }
477                
478                for (AugmentedStmt nas : newBody) {
479                        AugmentedStmt oas = (AugmentedStmt) new2old.get(nas);
480
481                        Stmt ns = nas.get_Stmt(), os = oas.get_Stmt();
482
483                        if (os instanceof IfStmt) {
484                                Unit target = ((IfStmt) os).getTarget(), newTgt = null;
485
486                                if ((newTgt = so2n.get(target)) != null)
487                                        ((IfStmt) ns).setTarget(newTgt);
488                                else
489                                        ((IfStmt) ns).setTarget(target);
490                        }
491
492                        else if (os instanceof TableSwitchStmt) {
493                                TableSwitchStmt otss = (TableSwitchStmt) os, ntss = (TableSwitchStmt) ns;
494
495                                Unit target = otss.getDefaultTarget(), newTgt = null;
496
497                                if ((newTgt = so2n.get(target)) != null)
498                                        ntss.setDefaultTarget(newTgt);
499                                else
500                                        ntss.setDefaultTarget(target);
501
502                                LinkedList<Unit> new_target_list = new LinkedList<Unit>();
503
504                                int target_count = otss.getHighIndex() - otss.getLowIndex() + 1;
505                                for (int i = 0; i < target_count; i++) {
506                                        target = otss.getTarget(i);
507                                        newTgt = null;
508
509                                        if ((newTgt = so2n.get(target)) != null)
510                                                new_target_list.add(newTgt);
511                                        else
512                                                new_target_list.add(target);
513                                }
514                                ntss.setTargets(new_target_list);
515                        }
516
517                        else if (os instanceof LookupSwitchStmt) {
518                                LookupSwitchStmt olss = (LookupSwitchStmt) os, nlss = (LookupSwitchStmt) ns;
519
520                                Unit target = olss.getDefaultTarget(), newTgt = null;
521
522                                if ((newTgt = so2n.get(target)) != null)
523                                        nlss.setDefaultTarget(newTgt);
524                                else
525                                        nlss.setDefaultTarget(target);
526
527                                Unit[] new_target_list = new Unit[olss.getTargetCount()];
528
529                                for (int i = 0; i < new_target_list.length; i++) {
530                                        target = olss.getTarget(i);
531                                        newTgt = null;
532
533                                        if ((newTgt = so2n.get(target)) != null)
534                                                new_target_list[i] = newTgt;
535                                        else
536                                                new_target_list[i] = target;
537                                }
538                                nlss.setTargets(new_target_list);
539                                nlss.setLookupValues(olss.getLookupValues());
540                        }
541                }
542
543                return newBody;
544        }

546        private void mirror_PredsSuccs(AugmentedStmt originalAs,
547                        List<AugmentedStmt> oldList, List<AugmentedStmt> newList,
548                        Map<AugmentedStmt, AugmentedStmt> old2new) {
549                for (AugmentedStmt oldAs : oldList) {
550                        AugmentedStmt newAs = (AugmentedStmt) old2new.get(oldAs);
551
552                        if (newAs != null)
553                                newList.add(newAs);
554
555                        else {
556                                newList.add(oldAs);
557
558                                AugmentedStmt clonedAs = (AugmentedStmt) old2new
559                                                .get(originalAs);
560
561                                if (oldList == originalAs.bpreds)
562                                        oldAs.bsuccs.add(clonedAs);
563                                else if (oldList == originalAs.cpreds)
564                                        oldAs.csuccs.add(clonedAs);
565                                else if (oldList == originalAs.bsuccs)
566                                        oldAs.bpreds.add(clonedAs);
567                                else if (oldList == originalAs.csuccs)
568                                        oldAs.cpreds.add(clonedAs);
569                                else
570                                        throw new RuntimeException(
571                                                        "Error mirroring preds and succs in Try block splitting.");
572                        }
573                }
574        }

576        public void find_Dominators() {
577                // set up the dominator sets for all the nodes in the graph
578                for (AugmentedStmt as : aug_list) {
579                        // Dominators:
580                        // safe starting approximation for S(0) ... empty set
581                        // unsafe starting approximation for S(i) .. full set
582                        if (as.cpreds.size() != 0) {
583
584                                if (as.get_Dominators().isEmpty() == false)
585                                        as.get_Dominators().clear();
586
587                                as.get_Dominators().addAll(aug_list);
588                        } else
589                                as.get_Dominators().clear();
590                }
591
592                // build the worklist
593                IterableSet<AugmentedStmt> worklist = new IterableSet<AugmentedStmt>();
594                worklist.addAll(aug_list);
595
596                // keep going until the worklist is empty
597                while (!worklist.isEmpty()) {
598                        AugmentedStmt as = worklist.getFirst();
599                        worklist.removeFirst();
600
601                        IterableSet<AugmentedStmt> pred_intersection = new IterableSet<AugmentedStmt>();
602                        boolean first_pred = true;
603
604                        // run through all the predecessors and get their dominance
605                        // intersection
606                        for (AugmentedStmt pas : as.cpreds) {
607                                // for the first predecessor just take all his dominators
608                                if (first_pred) {
609                                        pred_intersection.addAll(pas.get_Dominators());
610                                        if (pred_intersection.contains(pas) == false)
611                                                pred_intersection.add(pas);
612                                        first_pred = false;
613                                }
614
615                                // for the subsequent predecessors remove the ones they do not
616                                // have from the intersection
617                                else {
618                                        Iterator<AugmentedStmt> piit = pred_intersection
619                                                        .snapshotIterator();
620                                        while (piit.hasNext()) {
621                                                AugmentedStmt pid = piit.next();
622
623                                                if ((pas.get_Dominators().contains(pid) == false)
624                                                                && (pas != pid))
625                                                        pred_intersection.remove(pid);
626                                        }
627                                }
628                        }
629
630                        // update dominance if we have a change
631                        if (as.get_Dominators().equals(pred_intersection) == false) {
632                                for (AugmentedStmt o : as.csuccs) {
633                                        if (worklist.contains(o) == false)
634                                                worklist.add(o);
635                                }
636
637                                as.get_Dominators().clear();
638                                as.get_Dominators().addAll(pred_intersection);
639                        }
640                }
641        }
