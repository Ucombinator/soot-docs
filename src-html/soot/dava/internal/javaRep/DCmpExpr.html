022import soot.*;
023import soot.util.*;

025import soot.jimple.*;

028/*
029 * TODO
030 */
031public class DCmpExpr extends AbstractGrimpIntBinopExpr implements CmpExpr
032{
033    public DCmpExpr(Value op1, Value op2) { super(op1, op2); }
034    public final String getSymbol() { return " - "; }
035    public final int getPrecedence() { return 700; }
036    public void apply(Switch sw) { ((ExprSwitch) sw).caseCmpExpr(this); }
037
038    public Object clone() 
039    {
040        return new DCmpExpr(Grimp.cloneIfNecessary(getOp1()), Grimp.cloneIfNecessary(getOp2()));
041    }
042    
043    public Type getType(){
044        if (getOp1().getType().equals(getOp2().getType())){
045                return getOp1().getType();
046        }
047                
048        return IntType.v();
049    }
050}
