020package soot.dava.internal.AST;
021
022import soot.*;
023import soot.dava.*;
024import soot.dava.toolkits.base.AST.analysis.*;
025
026public class ASTOrCondition extends ASTAggregatedCondition{
027    public ASTOrCondition(ASTCondition left, ASTCondition right){
028        super(left,right);
029    }
030    
031    public void apply(Analysis a){
032        a.caseASTOrCondition(this);
033    }
034
035    public String toString(){
036        if(left instanceof ASTUnaryBinaryCondition){
037            if(right instanceof ASTUnaryBinaryCondition){
038                if(not)
039                    return "!(" + left.toString() + " || "+ right.toString()+")";
040                else
041                    return left.toString() + " || "+ right.toString();
042            }
043            else{ //right is ASTAggregatedCondition
044                if(not)
045                    return "!("+left.toString() + " || ("+right.toString() +" ))";
046                else
047                    return left.toString() + " || ("+right.toString() +" )";
048            }
049        }
050        else{ //left is ASTAggregatedCondition
051            if(right instanceof ASTUnaryBinaryCondition){
052                if(not)
053                    return "!(( "+ left.toString() + ") || "+ right.toString()+")";
054                else
055                    return "( "+ left.toString() + ") || "+ right.toString();
056            }
057            else{ //right is ASTAggregatedCondition also
058                if(not)
059                    return "!(( "+left.toString() + ") || ("+right.toString() +" ))";
060                else
061                    return "( "+left.toString() + ") || ("+right.toString() +" )";
062            }
063        }
064    }
065
066    public void toString(UnitPrinter up){
067        if(up instanceof DavaUnitPrinter){
068            
069            if(not){
070                //print !
071                ((DavaUnitPrinter)up).addNot();
072                //print LeftParen
073                ((DavaUnitPrinter)up).addLeftParen();
074            }
075            if(left instanceof ASTUnaryBinaryCondition){
076                if(right instanceof ASTUnaryBinaryCondition){
077                    
078                    left.toString(up);
079                    
080                    ((DavaUnitPrinter)up).addAggregatedOr();
081                    
082                    right.toString(up);
083                }
084                else{ //right is ASTAggregatedCondition
085                    
086                    left.toString(up); 
087                    
088                    ((DavaUnitPrinter)up).addAggregatedOr(); 
089                    
090                    ((DavaUnitPrinter)up).addLeftParen();
091                    right.toString(up); 
092                    ((DavaUnitPrinter)up).addRightParen();
093                }
094            }
095            else{ //left is ASTAggregatedCondition
096                if(right instanceof ASTUnaryBinaryCondition){
097                    
098                    ((DavaUnitPrinter)up).addLeftParen();
099                    left.toString(up); 
100                    ((DavaUnitPrinter)up).addRightParen();
101                    
102                    ((DavaUnitPrinter)up).addAggregatedOr(); 
103                    
104                    right.toString(up); 
105                }
106                else{ //right is ASTAggregatedCondition also
107                    
108                    ((DavaUnitPrinter)up).addLeftParen();
109                    left.toString(up); 
110                    ((DavaUnitPrinter)up).addRightParen();
111                    
112                    ((DavaUnitPrinter)up).addAggregatedOr(); 
113                    
114                    ((DavaUnitPrinter)up).addLeftParen();
115                    right.toString(up); 
116                    ((DavaUnitPrinter)up).addRightParen();
117                }
118            }
119            if(not)
120                ((DavaUnitPrinter)up).addRightParen();
121        }
122        else
123            throw new RuntimeException();
124    }
125    
126}
