021package soot.dava.internal.AST;
022
023import soot.*;
024import java.util.*;
025import soot.jimple.*;
026import soot.dava.internal.SET.*;
027import soot.dava.toolkits.base.AST.analysis.*;
028
029public class ASTIfNode extends ASTControlFlowNode
030{
031    private List<Object> body;
032
033    public ASTIfNode( SETNodeLabel label, ConditionExpr condition, List<Object> body)
034    {
035        super( label, condition);
036        this.body = body;
037
038        subBodies.add( body);
039    }
040
041
042    /*
043      Nomair A. Naeem 17-FEB-05
044      Needed because of change of grammar of condition being stored as a ASTCondition rather 
045      than the ConditionExpr which was the case before
046    */
047    public ASTIfNode( SETNodeLabel label, ASTCondition condition, List<Object> body)
048    {
049        super( label, condition);
050        this.body = body;
051
052        subBodies.add( body);
053    }
054
055
056    /*
057      Nomair A. Naeem 21-FEB-05
058      Used by OrAggregatorTwo
059    */
060    public List<Object> getIfBody(){
061        return body;
062    }
063
064    public Object clone()
065    {
066        return new ASTIfNode( get_Label(), get_Condition(), body);
067    }
068
069    /*
070      Nomair A. Naeem 19-FEB-2005
071      Added to support aggregation of conditions
072    */
073    public void replace(SETNodeLabel label,ASTCondition condition, List<Object> body){
074        this.body=body;
075        subBodies= new ArrayList<Object>();
076        subBodies.add(body);
077        set_Condition(condition);
078        set_Label(label);
079    }
080
081
082    /*
083      Nomair A Naeem 20-FEB-2005
084      Added for UselessLabeledBlockRemover
085    */
086    public void replaceBody(List<Object> body){
087        this.body=body;
088        subBodies=new ArrayList<Object>();
089        subBodies.add(body);
090    }
091
092
093    public void toString(UnitPrinter up) 
094    {
095        label_toString(up);
096
097        up.literal( "if" );
098        up.literal( " " );
099        up.literal( "(" );
100        condition.toString( up );
101        up.literal( ")");
102        up.newline();
103        
104        up.literal( "{" );
105        up.newline();
106
107        up.incIndent();
108        body_toString( up, body );
109        up.decIndent();
110
111        up.literal( "}" );
112        up.newline();
113    }
114    public String toString()
115    {
116        StringBuffer b = new StringBuffer();
117        
118        b.append( label_toString());
119
120        b.append( "if (");
121        b.append( get_Condition().toString());
122        b.append( ")");
123        b.append( NEWLINE);
124        
125        b.append( "{");
126        b.append( NEWLINE);
127
128        b.append( body_toString(body));
129
130        b.append( "}");
131        b.append( NEWLINE);
132
133        return b.toString();
134    }
135
136
137
138    /*
139      Nomair A. Naeem, 7-FEB-05
140      Part of Visitor Design Implementation for AST
141      See: soot.dava.toolkits.base.AST.analysis For details
142    */
143    public void apply(Analysis a){
144        a.caseASTIfNode(this);
145    }
146}
