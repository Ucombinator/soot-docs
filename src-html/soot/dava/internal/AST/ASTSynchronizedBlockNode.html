021package soot.dava.internal.AST;
022
023import soot.*;
024import soot.jimple.*;
025import java.util.*;
026import soot.dava.internal.SET.*;
027import soot.dava.toolkits.base.AST.analysis.*;
028
029public class ASTSynchronizedBlockNode extends ASTLabeledNode
030{
031    private List<Object> body;
032    private ValueBox localBox;
033
034    public ASTSynchronizedBlockNode( SETNodeLabel label, List<Object> body, Value local)
035    {
036        super( label);
037        this.body = body;
038        this.localBox = Jimple.v().newLocalBox( local );
039
040        subBodies.add( body);
041    }
042
043    /*
044      Nomair A Naeem 21-FEB-2005
045      Used by UselessLabeledBlockRemove to update a body
046    */
047    public void replaceBody(List<Object> body){
048        this.body=body;
049        subBodies=new ArrayList<Object>();
050        subBodies.add(body);
051    }
052
053    public int size()
054    {
055        return body.size();
056    }
057    
058    public Local getLocal() {
059        return (Local) localBox.getValue();
060    }
061
062
063    public void setLocal(Local local){
064        this.localBox = Jimple.v().newLocalBox( local );
065    }
066
067    public Object clone()
068    {
069        return new ASTSynchronizedBlockNode( get_Label(), body, getLocal());
070    }
071
072    public void toString( UnitPrinter up)
073    {
074        label_toString(up);
075
076        /*        up.literal( "synchronized" );
077                  up.literal( " " );
078                  up.literal( "(" );
079        */
080        up.literal( "synchronized (");
081        localBox.toString(up);
082        up.literal( ")");
083        up.newline();
084
085        up.literal( "{" );
086        up.newline();
087 
088        up.incIndent();
089        body_toString( up, body );
090        up.decIndent();
091
092        up.literal( "}" );
093        up.newline();
094    }
095
096    public String toString()
097    {
098        StringBuffer b = new StringBuffer();
099
100        b.append( label_toString(  ));
101
102        b.append( "synchronized (");
103        b.append( getLocal());
104        b.append( ")");
105        b.append( NEWLINE);
106
107        b.append( "{");
108        b.append( NEWLINE);
109 
110        b.append( body_toString( body));
111
112        b.append( "}");
113        b.append( NEWLINE);
114
115        return b.toString();
116    }
117
118
119
120
121    /*
122      Nomair A. Naeem, 7-FEB-05
123      Part of Visitor Design Implementation for AST
124      See: soot.dava.toolkits.base.AST.analysis For details
125    */
126    public void apply(Analysis a){
127        a.caseASTSynchronizedBlockNode(this);
128    }
129}
