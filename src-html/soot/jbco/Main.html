055    Vector<String> transformsToAdd = new Vector<String>();
056    boolean remove[] = new boolean[argv.length];
057    for (int i = 0; i < argv.length; i++) {
058      String arg = argv[i];
059      if (arg.equals("-jbco:help")) {
060        System.out.println("The Java Bytecode Obfuscator (JBCO)\n\nGeneral Options:");

084        metrics = true;
085        remove[i] = true;

087      } // temp dumby arg to printout name of benchmark for output files
088      else if (arg.startsWith("-jbco:name:")) {
089        remove[i] = true;
090        rcount++;
091      } // dumby dumby dumby
092      else if (arg.startsWith("-t:")) {
093        arg = arg.substring(3);
094        
095        int tweight = 9;
096        char cweight = arg.charAt(0);
097        if (cweight >= '0' && cweight <= '9') {
098          try {
099            tweight = Integer.parseInt(arg.substring(0,1));

104          arg = arg.substring(arg.indexOf(':') + 1);

127        Object o = null;
128        arg = arg.substring(4);
129
130        int tweight = 0;
131        char cweight = arg.charAt(0);
132        if (cweight >= '0' && cweight <= '9') {
133          try {
134            tweight = Integer.parseInt(arg.substring(0,1));
135          } catch (NumberFormatException nfe) {
136            G.v().out.println("Improperly formatted transformation weight: "+argv[i]);

139          if (arg.indexOf(':') < 0) {

146        int index = arg.indexOf(':');
147        if (index < 0) {
148          G.v().out.println("Illegally Formatted Option: "+argv[i]);

287  private static String[] checkWhole(String argv[], boolean add) {
288    for (int i = 0; i < argv.length; i++) {
289      if (argv[i].equals("-w")) { 
290        if (add) {
291          return argv;
292        } else {
293          String tmp[] = new String[argv.length - 1];
294          if (i == 0)
295            System.arraycopy(argv, 1, tmp, 0, tmp.length);
296          else {
297            System.arraycopy(argv, 0, tmp, 0, i);
298            if (i<(argv.length - 1))
299              System.arraycopy(argv, i+1, tmp, i, tmp.length - i);
300          }
301          return tmp;
302        }
303      }
304    } 
305    
306    if (add) {
307      String tmp[] = new String[argv.length + 1];
308      tmp[0] = "-w";
309      System.arraycopy(argv, 0, tmp, 1, argv.length);
310      return tmp;
311    }
312    
313    return argv;
314  }

321  private static IJbcoTransform getTransform(String name) {
322    if (name.startsWith("bb.jbco_rrps"))
323      return new RemoveRedundantPushStores();
324    if (name.startsWith("bb.printout"))
325      return new BAFPrintout(name, true);
326    if (name.equals("bb.jbco_j2bl"))
327      return new Jimple2BafLocalBuilder();
328    if (name.equals("jtp.jbco_gia"))
329      return new GotoInstrumenter();
330    if (name.equals("jtp.jbco_cae2bo"))
331      return new ArithmeticTransformer();
332    if (name.equals("wjtp.jbco_cc"))
333      return new CollectConstants();
334    if (name.equals("bb.jbco_ecvf"))
335      return new UpdateConstantsToFields();
336    if (name.equals("bb.jbco_rds"))
337      return new FindDuplicateSequences();
338    if (name.equals("bb.jbco_plvb"))
339      return new LocalsToBitField();
340    if (name.equals("bb.jbco_rlaii"))
341      return new MoveLoadsAboveIfs();
342    if (name.equals("bb.jbco_ptss"))
343      return new WrapSwitchesInTrys();
344    if (name.equals("bb.jbco_iii"))
345      return new IndirectIfJumpsToCaughtGotos();
346    if (name.equals("bb.jbco_ctbcb"))
347      return new TryCatchCombiner();
348    if (name.equals("bb.jbco_cb2ji"))
349      return new AddJSRs();
350    if (name.equals("bb.jbco_riitcb"))
351      return new IfNullToTryCatch();
352    if (name.equals("wjtp.jbco_blbc"))
353      return new LibraryMethodWrappersBuilder();
354    if (name.equals("wjtp.jbco_bapibm"))
355      return new BuildIntermediateAppClasses();
356    if (name.equals("wjtp.jbco_cr"))
357      return new ClassRenamer();
358    if (name.equals("bb.jbco_ful"))
359      return new FixUndefinedLocals();
360    if (name.equals("wjtp.jbco_fr"))
361      return new FieldRenamer();
362    if (name.equals("wjtp.jbco_mr"))
363      return new MethodRenamer();
364    if (name.equals("jtp.jbco_adss"))
365      return new AddSwitches();
366    if (name.equals("jtp.jbco_jl"))
367      return new CollectJimpleLocals();
368    if (name.equals("bb.jbco_dcc"))
369      return new ConstructorConfuser();
370    if (name.equals("bb.jbco_counter"))
371      return new BAFCounter();
372    
373    return null;
374  }

376  private static int getWeight(String phasename) {
377    int weight = 9;
378    Integer intg = transformsToWeights.get(phasename);
379    if (intg != null)
380      weight = intg.intValue();
381    return weight;
382  }

384  public static int getWeight(String phasename, String method) {
385    Map<Object,Integer> htmp = transformsToMethodsToWeights.get(phasename);
386    
387    int result = 10;
388    
389    if (htmp != null)
390    {
391      Iterator<Object> keys = htmp.keySet().iterator(); 
392      
393      while (keys.hasNext()) {
394        Integer intg = null;
395        Object o = keys.next();
396        if (o instanceof java.util.regex.Pattern) {
397          if (((java.util.regex.Pattern)o).matcher(method).matches())
398            intg = (Integer)htmp.get(o);
399        } else if (o instanceof String && method.equals(o)) {
400          intg = (Integer)htmp.get(o);
401        }
402        
403        if (intg != null && intg.intValue() < result)
404          result = intg.intValue();
405      }
406    }
407    if (result > 9  || result < 0)
408      result = getWeight(phasename);
409    
410    if (jbcoVerbose)
411      G.v().out.println("["+phasename+"] Processing "+method+" with weight: "+result);
412    return result;
413  }
