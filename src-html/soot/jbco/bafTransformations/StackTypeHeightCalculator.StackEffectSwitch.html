479  public static Map<Unit,Stack<Type>> calculateStackHeights(Body b, boolean jimpleLocals) {
480    if (!(b instanceof BafBody)) 
481      throw new java.lang.RuntimeException("Expecting Baf Body");
482    //System.out.println("\n"+b.getMethod().getName());
483    
484    Map<Unit,Stack<Type>> results = new HashMap<Unit,Stack<Type>>();
485    bug = new BriefUnitGraph(b);
486    List<Unit> heads = bug.getHeads();
487    for (int i = 0; i < heads.size(); i++)
488    {
489      Unit h = heads.get(i);
490      RefType handlerExc = isHandlerUnit(b.getTraps(),h);
491      Stack<Type> stack = (Stack<Type>)results.get(h);
492      if (stack != null) {
493        if (stack.size() != (handlerExc!=null ? 1 : 0))
494          throw new java.lang.RuntimeException("Problem with stack height - head expects ZERO or one if handler");
495        continue;
496      }
497        
498      List<Unit> worklist = new ArrayList<Unit>();
499      stack = new Stack<Type>();
500      if (handlerExc!=null)
501        stack.push(handlerExc);
502      results.put(h,stack);
503      worklist.add(h);
504      while (!worklist.isEmpty()) {
505        Inst inst = (Inst)worklist.remove(0);
506        
507        inst.apply(sw);
508
509        stack = updateStack(sw,(Stack<Type>)results.get(inst));
510        Iterator<Unit> lit = bug.getSuccsOf(inst).iterator();
511        while (lit.hasNext()) {
512          Unit next = lit.next();
513          Stack<Type> nxtStck = results.get(next);
514          if (nxtStck != null) {
515            if (nxtStck.size() != stack.size()) {
516              printStack(b.getUnits(),results,false);
517              throw new java.lang.RuntimeException("Problem with stack height at: "+next + "\n\rHas Stack "+nxtStck+" but is expecting "+stack);
518            }
519            continue;
520          }
521          
522          results.put(next,stack);
523          worklist.add(next);
524        }
525      }
526    }
527    
528    return results;
529  }

532    u.apply(sw);
533    return updateStack(sw,st);

536  public static Stack<Type> updateStack(StackEffectSwitch sw, Stack<Type> st) {
537    @SuppressWarnings("unchecked")
538        Stack<Type> clone = (Stack<Type>)st.clone();
539    
540    if (sw.remove_types != null) {
541            if (sw.remove_types.length > clone.size()) {
542          String exc = "Expecting values on stack: ";
543          for (Type element : sw.remove_types) {
544            String type = element.toString();
545            if (type.trim().length() == 0) type = element instanceof RefLikeType ? "L" : "U";
546            
547            exc += type + "  ";
548          }
549          exc += "\n\tbut only found: "; 
550          for (int i = 0; i < clone.size(); i++) {
551            String type = clone.get(i).toString();
552            if (type.trim().length() == 0) type = clone.get(i) instanceof RefLikeType ? "L" : "U";
553          
554            exc += type + "  ";
555          }
556          
557          if (sw.shouldThrow)
558            throw new RuntimeException(exc);
559          else
560            G.v().out.println(exc);
561        }
562            for (int i = sw.remove_types.length - 1; i >= 0; i--) {
563              try {
564            Type t = clone.pop();
565         
566            if (!checkTypes(t,sw.remove_types[i])) {
567              //System.out.println("Incompatible types: " + t + "  :  "+sw.remove_types[i]);
568            }
569          } catch (Exception exc) {
570            return null;
571          }
572            }
573    }
574    
575    if (sw.add_types != null)
576                for (Type element : sw.add_types)
577                        clone.push(element);
578    
579    return clone;
580  }

582  private static boolean checkTypes(Type t1, Type t2) {
583    if (t1 == t2) 
584      return true;
585    
586    if (t1 instanceof RefLikeType && t2 instanceof RefLikeType)
587      return true;
588    
589    if (t1 instanceof IntegerType && t2 instanceof IntegerType) 
590      return true;
591    
592    if (t1 instanceof LongType && t2 instanceof LongType)
593      return true;
594    
595    if (t1 instanceof DoubleType && t2 instanceof DoubleType)
596      return true;
597    
598    if (t1 instanceof FloatType && t2 instanceof FloatType)
599      return true;
600    
601    return false;
602  }

604  public static void printStack(PatchingChain<Unit> units, Map<Unit,Stack<Type>> stacks, boolean before) {
605    
606    int count = 0;
607    sw.shouldThrow = false;
608    Map<Unit,Integer> indexes = new HashMap<Unit,Integer>();
609    Iterator<Unit> it = units.snapshotIterator();
610    while (it.hasNext())
611      indexes.put(it.next(),new Integer(count++));
612    it = units.snapshotIterator();
613    while (it.hasNext()) {
614      String s = "";
615      Unit unit = it.next();
616      if (unit instanceof TargetArgInst) {
617        Object t = ((TargetArgInst)unit).getTarget();
618        s = indexes.get(t).toString();
619      } else if (unit instanceof TableSwitchInst) {
620        TableSwitchInst tswi = (TableSwitchInst)unit;
621        s+= "\r\tdefault: " + tswi.getDefaultTarget() + "  "+indexes.get(tswi.getDefaultTarget());
622        int index = 0;
623        for (int x = tswi.getLowIndex(); x <= tswi.getHighIndex(); x++)
624          s+= "\r\t "+x+": " + tswi.getTarget(index) + "  "+indexes.get(tswi.getTarget(index++));
625      }
626      try {
627        s = indexes.get(unit) + " " + unit + "  " + s + "   [";
628      } catch (Exception e) {
629        G.v().out.println("Error in StackTypeHeightCalculator trying to find index of unit");
630      }
631      Stack<Type> stack = stacks.get(unit);
632      if (stack != null) {
633        if (!before) {
634          ((Unit)unit).apply(sw);
635          stack = updateStack(sw,stack);
636          if (stack == null) {
637            soot.jbco.util.Debugger.printUnits(units, " StackTypeHeightCalc failed");
638            sw.shouldThrow = true;
639            return;
640          }
641        }
642        for (int i = 0; i < stack.size(); i++)
643          s += printType(stack.get(i));
644      } else s+="***missing***";
645      System.out.println(s+"]");  
646    }
647    sw.shouldThrow = true;
648  }

650  private static String printType(Type t) {
651    if (t instanceof IntegerType) {
652      return "I";
653    } else if (t instanceof FloatType) { 
654      return "F";
655    } else if (t instanceof DoubleType) {
656      return "D";
657    } else if (t instanceof LongType) {
658      return "J";
659    } else if (t instanceof RefLikeType) {
660      
661      //if (t instanceof RefType && ((RefType)t).getSootClass() != null)
662      //  return "L(" + ((RefType)t).getSootClass().getName()+")";
663      //else
664        return "L"+t.toString();
665    } else {
666      return "U("+t.getClass().toString()+")";
667    }
668  }

670  private static RefType isHandlerUnit(Chain<Trap> traps, Unit h) {
671    Iterator<Trap> it = traps.iterator();
672    while (it.hasNext()) {
673      Trap t = (Trap)it.next();
674      if (t.getHandlerUnit() == h)
675        return t.getException().getType();
676    }
677    return null;
678  }

681    Stack<Type> stack = calculateStackHeights(b).get(u);
682    u.apply(sw);
683    return updateStack(sw,stack);
684  }
