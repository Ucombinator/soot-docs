020package soot.javaToJimple;
021
022import soot.*;
023
024public class AnonClassInitMethodSource extends soot.javaToJimple.PolyglotMethodSource {
025
026    private boolean hasOuterRef;
027    public void hasOuterRef(boolean b){
028        hasOuterRef = b;
029    }
030    public boolean hasOuterRef(){
031        return hasOuterRef;
032    }
033    
034    private boolean hasQualifier;
035    public void hasQualifier(boolean b){
036        hasQualifier = b;
037    }
038    public boolean hasQualifier(){
039        return hasQualifier;
040    }
041    
042    
043    private boolean inStaticMethod;
044
045    public void inStaticMethod(boolean b){
046        inStaticMethod = b;
047    }
048    public boolean inStaticMethod(){
049        return inStaticMethod;
050    }
051    
052
053    private boolean isSubType = false;
054    public void isSubType(boolean b){
055        isSubType = b;
056    }
057    public boolean isSubType(){
058        return isSubType;
059    }
060    
061    private soot.Type superOuterType = null;
062    private soot.Type thisOuterType = null;
063
064    public void superOuterType(soot.Type t){
065        superOuterType = t;
066    }
067    public soot.Type superOuterType(){
068        return superOuterType;
069    }
070
071    public void thisOuterType(soot.Type t){
072        thisOuterType = t;
073    }
074    public soot.Type thisOuterType(){
075        return thisOuterType;
076    }
077
078    private polyglot.types.ClassType polyglotType;
079    public void polyglotType(polyglot.types.ClassType type){
080        polyglotType = type;
081    }
082    public polyglot.types.ClassType polyglotType() {
083        return polyglotType;
084    }
085    
086    private polyglot.types.ClassType anonType;
087    public void anonType(polyglot.types.ClassType type){
088        anonType = type;
089    }
090    public polyglot.types.ClassType anonType() {
091        return anonType;
092    }
093    
094    public soot.Body getBody(soot.SootMethod sootMethod, String phaseName){
095        AnonInitBodyBuilder aibb = new AnonInitBodyBuilder();
096        soot.jimple.JimpleBody body = aibb.createBody(sootMethod);
097        
098        PackManager.v().getPack("jj").apply(body);
099    
100        return body;
101    }
102    
103    private soot.Type outerClassType;
104
105    public soot.Type outerClassType(){
106        return outerClassType;
107    }
108    
109    public void outerClassType(soot.Type type){
110        outerClassType = type;
111    }
112}
