025public class InnerClassInfoFinder extends polyglot.visit.NodeVisitor {
026
027    private final ArrayList<Node> localClassDeclList;
028    private final ArrayList<Node> anonBodyList;
029    private final ArrayList<Node> memberList;
030    //private ArrayList declaredInstList;
031    //private ArrayList usedInstList;
032    
033    public ArrayList<Node> memberList(){
034        return memberList;
035    }
036
037    /*public ArrayList declaredInstList(){
038        return declaredInstList;
039    }
040

042        return usedInstList;
043    }*/
044
045    public ArrayList<Node> localClassDeclList(){
046        return localClassDeclList;
047    }
048
049    public ArrayList<Node> anonBodyList(){
050        return anonBodyList;
051    }
052
053    public InnerClassInfoFinder(){
054        //declFound = null;
055        localClassDeclList = new ArrayList<Node>();
056        anonBodyList = new ArrayList<Node>();
057        memberList = new ArrayList<Node>();
058        //declaredInstList = new ArrayList();
059        //usedInstList = new ArrayList();
060    }
061
062    public polyglot.visit.NodeVisitor enter(polyglot.ast.Node parent, polyglot.ast.Node n) {
063    
064        if (n instanceof polyglot.ast.LocalClassDecl) {
065            localClassDeclList.add(n);
066        }
067        if (n instanceof polyglot.ast.New) {
068            if (((polyglot.ast.New)n).anonType() != null){
069                anonBodyList.add(n);
070            }
071            /*polyglot.types.ProcedureInstance pi = ((polyglot.ast.New)n).constructorInstance();
072            if (pi.isPrivate()){
073                usedInstList.add(new polyglot.util.IdentityKey(pi));
074            }*/
075        }
076        
077        if (n instanceof polyglot.ast.ProcedureDecl) {
078            memberList.add(n);
079            /*polyglot.types.ProcedureInstance pi = ((polyglot.ast.ProcedureDecl)n).procedureInstance();
080            if (pi.flags().isPrivate()){
081                declaredInstList.add(new polyglot.util.IdentityKey(pi));
082            }*/
083        }
084        if (n instanceof polyglot.ast.FieldDecl) {
085            memberList.add(n);
086            /*polyglot.types.FieldInstance fi = ((polyglot.ast.FieldDecl)n).fieldInstance();
087            if (fi.flags().isPrivate()){
088                declaredInstList.add(new polyglot.util.IdentityKey(fi));
089            }*/
090        }
091        if (n instanceof polyglot.ast.Initializer) {
092            memberList.add(n);
093        }
094
095        /*if (n instanceof polyglot.ast.Field) {
096            polyglot.types.FieldInstance fi = ((polyglot.ast.Field)n).fieldInstance();
097            if (fi.isPrivate()){
098                usedInstList.add(new polyglot.util.IdentityKey(fi));
099            }
100        }
101        if  (n instanceof polyglot.ast.Call){
102            polyglot.types.ProcedureInst pi = ((polyglot.ast.Call)n).methodInstance();
103            if (pi.isPrivate()){
104                usedInstList.add(new polyglot.util.IdentityKey(pi));
105            }
106        }
107        if (n instanceof polyglot.ast.ConstructorCall){
108            polyglot.types.ProcedureInstance pi = ((polyglot.ast.ConstructorCall)n).constructorInstance();
109            if (pi.isPrivate()){
110                usedInstList.add(new polyglot.util.IdentityKey(pi));
111            }
112        }*/
113        
114        
115        return enter(n);
116    }
117}
