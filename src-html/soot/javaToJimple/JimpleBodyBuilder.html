081    public soot.jimple.JimpleBody createJimpleBody(polyglot.ast.Block block, List formals, soot.SootMethod sootMethod){
082     
083        createBody(sootMethod);
084
085        lg = new LocalGenerator(body);
086        // create this formal except for static methods
087        if (!soot.Modifier.isStatic(sootMethod.getModifiers())) {
088
089            soot.RefType type = sootMethod.getDeclaringClass().getType();
090            specialThisLocal = soot.jimple.Jimple.v().newLocal("this", type);
091            body.getLocals().add(specialThisLocal);
092                                                
093            soot.jimple.ThisRef thisRef = soot.jimple.Jimple.v().newThisRef(type);
094            
095            soot.jimple.Stmt thisStmt = soot.jimple.Jimple.v().newIdentityStmt(specialThisLocal, thisRef);
096            body.getUnits().add(thisStmt);
097            
098            // this is causing problems - no this in java code -> no tags
099            //Util.addLineTag(thisStmt, block);
100        }
101        
102        int formalsCounter = 0;
103        
104        //create outer class this param ref for inner classes except for static inner classes - this is not needed
105        int outerIndex = sootMethod.getDeclaringClass().getName().lastIndexOf("$");
106            
107        if ((outerIndex != -1) && (sootMethod.getName().equals("<init>"))) {
108                SootField this0Field = sootMethod.getDeclaringClass().getFieldByNameUnsafe("this$0");
109                if (this0Field != null) {
110                    // we know its an inner non static class can get outer class
111                    // from field ref of the this$0 field
112                    soot.SootClass outerClass = ((soot.RefType)this0Field.getType()).getSootClass();
113                    soot.Local outerLocal = lg.generateLocal(outerClass.getType());
114                    
115                    soot.jimple.ParameterRef paramRef = soot.jimple.Jimple.v().newParameterRef(outerClass.getType(), formalsCounter);
116                    paramRefCount++;
117                    soot.jimple.Stmt stmt = soot.jimple.Jimple.v().newIdentityStmt(outerLocal, paramRef);
118                    stmt.addTag(new soot.tagkit.EnclosingTag());
119                    body.getUnits().add(stmt);
120                    
121                    ((soot.javaToJimple.PolyglotMethodSource)sootMethod.getSource()).setOuterClassThisInit(outerLocal);
122                    outerClassParamLocal = outerLocal;
123                    formalsCounter++;
124                }
125        }
126
127        // handle formals
128        if (formals != null) {
129            ArrayList<String> formalNames = new ArrayList<String>();
130            Iterator formalsIt = formals.iterator();
131            while (formalsIt.hasNext()) { 
132                polyglot.ast.Formal formal = (polyglot.ast.Formal)formalsIt.next();
133                createFormal(formal, formalsCounter);
134                formalNames.add(formal.name());
135                formalsCounter++;
136            }
137            body.getMethod().addTag(new soot.tagkit.ParamNamesTag(formalNames));
138        }
139        
140        // handle final local params
141        ArrayList<SootField> finalsList = ((PolyglotMethodSource)body.getMethod().getSource()).getFinalsList();
142        if (finalsList != null){
143            Iterator<SootField> finalsIt = finalsList.iterator();
144            while (finalsIt.hasNext()){
145                soot.SootField sf = finalsIt.next();
146                soot.jimple.ParameterRef paramRef = soot.jimple.Jimple.v().newParameterRef(sf.getType(), formalsCounter);
147                paramRefCount++;
148                soot.jimple.Stmt stmt = soot.jimple.Jimple.v().newIdentityStmt(lg.generateLocal(sf.getType()), paramRef);
149                body.getUnits().add(stmt);
150                formalsCounter++;
151            }
152        }
153        
154        createBlock(block);
155
156        
157        // if method is <clinit> handle static field inits
158        if (sootMethod.getName().equals("<clinit>")){
159            
160            handleAssert(sootMethod);
161            handleStaticFieldInits(sootMethod);
162            handleStaticInitializerBlocks(sootMethod);
163        }
164       
165        // determine if body has a return stmt
166        boolean hasReturn = false;
167                if (block != null) {
168            Iterator it = block.statements().iterator();
169                    while (it.hasNext()){
170                            Object next = it.next();
171                            if (next instanceof polyglot.ast.Return){
172                    hasReturn = true; 
173                }
174            }
175        }
176        
177        soot.Type retType = body.getMethod().getReturnType();
178        // only do this if noexplicit return
179            if ((!hasReturn) && (retType instanceof soot.VoidType)) {
180            soot.jimple.Stmt retStmt = soot.jimple.Jimple.v().newReturnVoidStmt();
181            body.getUnits().add(retStmt);
182        }
183
184        // add exceptions from exceptionTable
185        if (exceptionTable != null) {
186            Iterator<Trap> trapsIt = exceptionTable.iterator();
187            while (trapsIt.hasNext()){
188                body.getTraps().add(trapsIt.next());
189            }
190        }
191        return body;
192    
193    }

211    protected void handleFieldInits(ArrayList<FieldDecl> fieldInits){
212        Iterator<FieldDecl> fieldInitsIt = fieldInits.iterator();
213        while (fieldInitsIt.hasNext()) {
214            polyglot.ast.FieldDecl field = fieldInitsIt.next();
215            String fieldName = field.name();
216            polyglot.ast.Expr initExpr = field.init();
217            soot.SootClass currentClass = body.getMethod().getDeclaringClass();
218            soot.SootFieldRef sootField = soot.Scene.v().makeFieldRef(currentClass, fieldName, Util.getSootType(field.type().type()), field.flags().isStatic());
219                
220            soot.Local base = specialThisLocal;
221                   
222            soot.jimple.FieldRef fieldRef = soot.jimple.Jimple.v().newInstanceFieldRef(base, sootField);
223                
224            soot.Value sootExpr;
225            if (initExpr instanceof polyglot.ast.ArrayInit) {
226                sootExpr = getArrayInitLocal((polyglot.ast.ArrayInit)initExpr, field.type().type());
227            }
228            else {
229                //System.out.println("field init expr: "+initExpr);
230                sootExpr = base().createAggressiveExpr(initExpr, false, false);
231                //System.out.println("soot expr: "+sootExpr);
232            }
233            if (sootExpr instanceof soot.jimple.ConditionExpr) {
234                sootExpr = handleCondBinExpr((soot.jimple.ConditionExpr)sootExpr); 
235            }
236                
237            soot.jimple.AssignStmt assign;
238            if (sootExpr instanceof soot.Local){
239                assign = soot.jimple.Jimple.v().newAssignStmt(fieldRef, (soot.Local)sootExpr);
240            }
241            else if (sootExpr instanceof soot.jimple.Constant){
242                assign = soot.jimple.Jimple.v().newAssignStmt(fieldRef, (soot.jimple.Constant)sootExpr);
243            }
244            else {
245                throw new RuntimeException("fields must assign to local or constant only");
246            }
247            body.getUnits().add(assign);
248            Util.addLnPosTags(assign, initExpr.position());
249            Util.addLnPosTags(assign.getRightOpBox(), initExpr.position());
250        }
251    }

273            
274        ArrayList<FieldDecl> staticFieldInits = ((soot.javaToJimple.PolyglotMethodSource)sootMethod.getSource()).getStaticFieldInits();
275        if (staticFieldInits != null) {
276            Iterator<FieldDecl> staticFieldInitsIt = staticFieldInits.iterator();
277            while (staticFieldInitsIt.hasNext()) {
278                polyglot.ast.FieldDecl field = staticFieldInitsIt.next();
279                String fieldName = field.name();
280                polyglot.ast.Expr initExpr = field.init();

282                soot.SootFieldRef sootField = soot.Scene.v().makeFieldRef(currentClass, fieldName, Util.getSootType(field.type().type()), field.flags().isStatic());

284                
285                //System.out.println("initExpr: "+initExpr);
286                soot.Value sootExpr;
287                if (initExpr instanceof polyglot.ast.ArrayInit) {
288                    sootExpr = getArrayInitLocal((polyglot.ast.ArrayInit)initExpr, field.type().type());
289                }
290                else {
291                    //System.out.println("field init expr: "+initExpr);
292                    sootExpr = base().createAggressiveExpr(initExpr, false, false);
293                    //System.out.println("soot expr: "+sootExpr);
294                    
295                    if (sootExpr instanceof soot.jimple.ConditionExpr) {
296                        sootExpr = handleCondBinExpr((soot.jimple.ConditionExpr)sootExpr);
297                    }
298                }
299
300                soot.jimple.Stmt assign = soot.jimple.Jimple.v().newAssignStmt(fieldRef, sootExpr);
301
302                body.getUnits().add(assign);
303                Util.addLnPosTags(assign, initExpr.position());
304                
305            }
306        }

354     * Block creation
355     */
356    private void createBlock(polyglot.ast.Block block){
357        
358        if (block == null) return;
359        
360                // handle stmts
361                Iterator it = block.statements().iterator();
362                while (it.hasNext()){
363                        Object next = it.next();
364            if (next instanceof polyglot.ast.Stmt){
365                                createStmt((polyglot.ast.Stmt)next);
366                        }
367                        else {
368                                throw new RuntimeException("Unexpected - Unhandled Node");
369                        }
370                }
371    }

411    /**
412     * Literal Creation
413     */
414    private soot.Value createLiteral(polyglot.ast.Lit lit) {
415                if (lit instanceof polyglot.ast.IntLit) {
416                        polyglot.ast.IntLit intLit = (polyglot.ast.IntLit)lit;
417                        long litValue = intLit.value();
418                        if (intLit.kind() == polyglot.ast.IntLit.INT) {
419                return soot.jimple.IntConstant.v((int)litValue);
420                        }
421                        else {
422                //System.out.println(litValue);
423                                return soot.jimple.LongConstant.v(litValue);
424                        }
425                }
426                else if (lit instanceof polyglot.ast.StringLit) {
427                        String litValue = ((polyglot.ast.StringLit)lit).value();
428                        return soot.jimple.StringConstant.v(litValue);          
429                }
430                else if (lit instanceof polyglot.ast.NullLit) {
431                        return soot.jimple.NullConstant.v();            
432                }
433                else if (lit instanceof polyglot.ast.FloatLit) {
434                        polyglot.ast.FloatLit floatLit = (polyglot.ast.FloatLit)lit;
435                        double litValue = floatLit.value();
436                        if (floatLit.kind() == polyglot.ast.FloatLit.DOUBLE) {
437                                return soot.jimple.DoubleConstant.v(floatLit.value());          
438                        }
439                        else {
440                                return soot.jimple.FloatConstant.v((float)(floatLit.value()));          
441                        }
442                }
443                else if (lit instanceof polyglot.ast.CharLit) {
444                        char litValue = ((polyglot.ast.CharLit)lit).value();
445            return soot.jimple.IntConstant.v(litValue);
446                }
447                else if (lit instanceof polyglot.ast.BooleanLit) {
448                        boolean litValue = ((polyglot.ast.BooleanLit)lit).value();
449            if (litValue) return soot.jimple.IntConstant.v(1);
450            else return soot.jimple.IntConstant.v(0);
451                }
452        else if (lit instanceof polyglot.ast.ClassLit){
453            return getSpecialClassLitLocal((polyglot.ast.ClassLit)lit);
454        }
455        else {
456            throw new RuntimeException("Unknown Literal - Unhandled: "+lit.getClass());
457        }
458    }

490    /**
491     * Local Retreival
492     */
493    private soot.Local getLocal(polyglot.types.LocalInstance li) {
494        
495        if (localsMap.containsKey(new polyglot.util.IdentityKey(li))){
496            soot.Local sootLocal = localsMap.get(new polyglot.util.IdentityKey(li));
497            return sootLocal;
498        }
499        else if (body.getMethod().getDeclaringClass().declaresField("val$"+li.name(), Util.getSootType(li.type()))){
500            soot.Local fieldLocal = generateLocal(li.type());
501            soot.SootFieldRef field = soot.Scene.v().makeFieldRef(body.getMethod().getDeclaringClass(), "val$"+li.name(), Util.getSootType(li.type()), false);
502            soot.jimple.FieldRef fieldRef = soot.jimple.Jimple.v().newInstanceFieldRef(specialThisLocal, field);
503            soot.jimple.AssignStmt assign = soot.jimple.Jimple.v().newAssignStmt(fieldLocal, fieldRef);
504            body.getUnits().add(assign);
505            return fieldLocal;
506        }
507        /*else {
508            throw new RuntimeException("Trying unsuccessfully to get local: "+li.name());
509        }*/
510        else {
511            //else create access meth in outer for val$fieldname
512            // get the this$0 field to find the type of an outer class - has
513            // to have one because local/anon inner can't declare static
514            // memebers so for deepnesting not in static context for these
515            // cases
516
517            soot.SootClass currentClass = body.getMethod().getDeclaringClass();
518            boolean fieldFound = false;
519            
520            while (!fieldFound){
521                if (!currentClass.declaresFieldByName("this$0")){
522                    throw new RuntimeException("Trying to get field val$"+li.name()+" from some outer class but can't access the outer class of: "+currentClass.getName()+"!"+" current class contains fields: "+currentClass.getFields());
523                }
524                soot.SootClass outerClass = ((soot.RefType)currentClass.getFieldByName("this$0").getType()).getSootClass();
525                // look for field of type li.type and name val$li.name in outer 
526                // class 
527                if (outerClass.declaresField("val$"+li.name(), Util.getSootType(li.type()))){
528                    fieldFound = true;
529                }
530                currentClass = outerClass;
531                // repeat until found in some outer class
532            }
533            // create and add accessor to that outer class (indic as current)
534            soot.SootMethod methToInvoke = makeLiFieldAccessMethod(currentClass, li);
535            
536            // invoke and return
537            // generate a local that corresponds to the invoke of that meth
538            ArrayList methParams = new ArrayList();
539            methParams.add(getThis(currentClass.getType()));
540        
541            soot.Local res = Util.getPrivateAccessFieldInvoke(methToInvoke.makeRef(), methParams, body, lg);
542            return res;
543        }
544    }

546    private soot.SootMethod makeLiFieldAccessMethod(soot.SootClass classToInvoke, polyglot.types.LocalInstance li){
547        String name = "access$"+soot.javaToJimple.InitialResolver.v().getNextPrivateAccessCounter()+"00";
548        ArrayList paramTypes = new ArrayList();
549        paramTypes.add(classToInvoke.getType());
550        
551        soot.SootMethod meth = new soot.SootMethod(name, paramTypes, Util.getSootType(li.type()), soot.Modifier.STATIC);
552
553        classToInvoke.addMethod(meth);
554        PrivateFieldAccMethodSource src = new PrivateFieldAccMethodSource(
555                Util.getSootType(li.type()),
556                "val$"+li.name(),
557                false,
558                classToInvoke
559        );
560        meth.setActiveBody(src.getBody(meth, null));
561        meth.addTag(new soot.tagkit.SyntheticTag());
562        return meth;
563    }

565    /**
566     * Stmt creation
567     */
568    protected void createStmt(polyglot.ast.Stmt stmt) {
569        //System.out.println("stmt: "+stmt.getClass());
570        if (stmt instanceof polyglot.ast.Eval) {
571                        base().createAggressiveExpr(((polyglot.ast.Eval)stmt).expr(), false, false);  
572        }
573        else if (stmt instanceof polyglot.ast.If) {
574           createIf2((polyglot.ast.If)stmt);
575        }
576                else if (stmt instanceof polyglot.ast.LocalDecl) {
577                        createLocalDecl((polyglot.ast.LocalDecl)stmt);
578                }
579                else if (stmt instanceof polyglot.ast.Block) {
580                        createBlock((polyglot.ast.Block)stmt);
581                }
582                else if (stmt instanceof polyglot.ast.While) {
583                    createWhile2((polyglot.ast.While)stmt);
584                }
585                else if (stmt instanceof polyglot.ast.Do) {
586                        createDo2((polyglot.ast.Do)stmt);
587                }
588                else if (stmt instanceof polyglot.ast.For) {
589                    createForLoop2((polyglot.ast.For)stmt);
590                }
591                else if (stmt instanceof polyglot.ast.Switch) {
592                        createSwitch((polyglot.ast.Switch)stmt);
593                }
594                else if (stmt instanceof polyglot.ast.Return) {
595                        createReturn((polyglot.ast.Return)stmt);
596                }
597                else if (stmt instanceof polyglot.ast.Branch) {
598                        createBranch((polyglot.ast.Branch)stmt);
599                }
600                else if (stmt instanceof polyglot.ast.ConstructorCall) {
601                        createConstructorCall((polyglot.ast.ConstructorCall)stmt);
602                }
603                else if (stmt instanceof polyglot.ast.Empty) {
604                    // do nothing empty stmt
605        }

739    private void createBranchingExpr(polyglot.ast.Expr expr, soot.jimple.Stmt tgt, boolean boto){
740        if (expr instanceof polyglot.ast.Binary && ((polyglot.ast.Binary)expr).operator() == polyglot.ast.Binary.COND_AND){
741            polyglot.ast.Binary cond_and = (polyglot.ast.Binary)expr;
742            if (boto){
743                soot.jimple.Stmt t1 = soot.jimple.Jimple.v().newNopStmt();
744                createBranchingExpr(cond_and.left(), t1, false);
745                createBranchingExpr(cond_and.right(), tgt, true);
746                body.getUnits().add(t1);
747            }
748            else {
749                createBranchingExpr(cond_and.left(), tgt, false);
750                createBranchingExpr(cond_and.right(), tgt, false);
751            }
752        }
753        else if (expr instanceof polyglot.ast.Binary && ((polyglot.ast.Binary)expr).operator() == polyglot.ast.Binary.COND_OR){

755            if (boto){

758            }

761                createBranchingExpr(cond_or.left(), t1, true);
762                createBranchingExpr(cond_or.right(), tgt, false);
763                body.getUnits().add(t1);
764            }
765            
766        }
767        else if (expr instanceof polyglot.ast.Unary && ((polyglot.ast.Unary)expr).operator() == polyglot.ast.Unary.NOT){
768            polyglot.ast.Unary not = (polyglot.ast.Unary)expr;
769            createBranchingExpr(not.expr(), tgt, !boto);
770        }
771        else {
772            soot.Value sootCond = base().createAggressiveExpr(expr, false, false);
773

797            }

799            //do nothing at all

1383     * always get the same result as javac 
1384     * lookup: non-table
1385     * table: sequential (no gaps)
1386     */
1387    private boolean isLookupSwitch(polyglot.ast.Switch switchStmt){
1388
1389        int lowest = 0;
1390        int highest = 0;
1391        int counter = 0;
1392        Iterator it = switchStmt.elements().iterator();
1393        while (it.hasNext()){
1394            Object next = it.next();
1395            if (next instanceof polyglot.ast.Case) {
1396                polyglot.ast.Case caseStmt = (polyglot.ast.Case)next;
1397                if (caseStmt.isDefault()) continue;
1398                int caseValue = (int)caseStmt.value();
1399                if (caseValue <= lowest || counter == 0 ) {
1400                    lowest = caseValue;
1401                }
1402                if (caseValue >= highest || counter == 0) {
1403                    highest = caseValue;
1404                }
1405                counter++;
1406            }
1407        }
1408
1409        if ((counter-1) == (highest - lowest)) return false;
1410        return true;

1764        
1765        /* add to the exceptionList all the ranges dictated by the returns found
1766         * (the problem is that return statements are not within the traps

1768         */
1769        List<Stmt>  before = beforeReturn.get(beforeReturn.size()-1); //last element
1770        List<Stmt>  after = afterReturn.get(afterReturn.size()-1); //last element

1776                {
1777                  addToExceptionList(after.get(i-1), before.get(i), catchAllBeforeNoop, soot.Scene.v().getSootClass("java.lang.Throwable"));

1783          addToExceptionList(startNoop, endNoop, catchAllBeforeNoop, soot.Scene.v().getSootClass("java.lang.Throwable"));
1784        }
1785        beforeReturn.remove(before);
1786        afterReturn.remove(after);
1787
1788        addToExceptionList(catchBeforeNoop, catchAfterNoop, catchAllBeforeNoop, soot.Scene.v().getSootClass("java.lang.Throwable"));
1789                
1790    }
