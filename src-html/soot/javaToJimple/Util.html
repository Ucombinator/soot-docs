041    public static String getParamNameForClassLit(polyglot.types.Type type){
042        String name = "";
043        if (type.isArray()){
044            int dims = ((polyglot.types.ArrayType)type).dims();
045            polyglot.types.Type arrType = ((polyglot.types.ArrayType)type).base();
046            while (arrType instanceof polyglot.types.ArrayType) {
047              arrType = ((polyglot.types.ArrayType)arrType).base();
048            }
049            String fieldName = "";
050            if (arrType.isBoolean()){
051                fieldName = "Z";
052            }
053            else if (arrType.isByte()){
054                fieldName = "B";
055            }
056            else if (arrType.isChar()){
057                fieldName = "C";
058            }
059            else if (arrType.isDouble()){
060                fieldName = "D";
061            }
062            else if (arrType.isFloat()){
063                fieldName = "F";
064            }
065            else if (arrType.isInt()){
066                fieldName = "I";
067            }
068            else if (arrType.isLong()){
069                fieldName = "J";
070            }
071            else if (arrType.isShort()){
072                fieldName = "S";
073            }
074            else {
075                String typeSt = getSootType(arrType).toString();
076                fieldName = "L"+typeSt;
077            }
078            
079            for (int i = 0; i < dims; i++){
080                name += "[";
081            }
082            name += fieldName;
083            if (!arrType.isPrimitive()){
084                name += ";";
085            }
086        }
087        else {
088            name = getSootType(type).toString();
089        }
090        return name;
091    }

093    public static String getFieldNameForClassLit(polyglot.types.Type type){
094        String fieldName = "";
095        if (type.isArray()){
096            int dims = ((polyglot.types.ArrayType)type).dims();
097            polyglot.types.Type arrType = ((polyglot.types.ArrayType)type).base();
098            while (arrType instanceof polyglot.types.ArrayType) {
099              arrType = ((polyglot.types.ArrayType)arrType).base();
100            }
101            fieldName = "array$";
102            for (int i = 0; i < (dims - 1); i++){
103                fieldName += "$";
104            }
105            if (arrType.isBoolean()){
106                fieldName += "Z";
107            }
108            else if (arrType.isByte()){
109                fieldName += "B";
110            }
111            else if (arrType.isChar()){
112                fieldName += "C";
113            }
114            else if (arrType.isDouble()){
115                fieldName += "D";
116            }
117            else if (arrType.isFloat()){
118                fieldName += "F";
119            }
120            else if (arrType.isInt()){
121                fieldName += "I";
122            }
123            else if (arrType.isLong()){
124                fieldName += "J";
125            }
126            else if (arrType.isShort()){
127                fieldName += "S";
128            }
129            else {
130                String typeSt = getSootType(arrType).toString();
131                typeSt = typeSt.replaceAll(".", "$");
132
133                fieldName = fieldName+"L"+typeSt;
134            }
135       }
136       else {
137            fieldName = "class$";
138            String typeSt = getSootType(type).toString();
139            typeSt = typeSt.replaceAll(".", "$");
140            fieldName = fieldName+typeSt;
141       }
142       
143       return fieldName;
144    }

146    public static String getSourceFileOfClass(soot.SootClass sootClass){
147        String name = sootClass.getName();
148        int index = name.indexOf("$");
149        
150        // inner classes are found in the very outer class
151        if (index != -1){
152            name = name.substring(0, index);
153        }
154        return name;
155    }

235    public static soot.Local getThis(soot.Type sootType, soot.Body body, HashMap getThisMap, LocalGenerator lg){
236
237        if (InitialResolver.v().hierarchy() == null){
238            InitialResolver.v().hierarchy(new soot.FastHierarchy());
239        }
240        
241        soot.FastHierarchy fh = InitialResolver.v().hierarchy();
242       
243        //System.out.println("getting this for type: "+sootType);
244        // if this for type already created return it from map
245        //if (getThisMap.containsKey(sootType)){
246        //    return (soot.Local)getThisMap.get(sootType);
247        //}
248        soot.Local specialThisLocal = body.getThisLocal();
249        // if need this just return it
250        if (specialThisLocal.getType().equals(sootType)) {
251           
252            getThisMap.put(sootType, specialThisLocal);
253            return specialThisLocal;
254        }
255       
256        // check to see if this method has a local of the correct type (it will
257        // if its an initializer - then ust use it)
258        // here we need an exact type I think
259        if (bodyHasLocal(body, sootType)){
260            soot.Local l = getLocalOfType(body, sootType);
261            getThisMap.put(sootType, l);
262            return l;
263        }
264        
265        // otherwise get this$0 for one level up
266        soot.SootClass classToInvoke = ((soot.RefType)specialThisLocal.getType()).getSootClass();
267        soot.SootField outerThisField = classToInvoke.getFieldByName("this$0");
268        soot.Local t1 = lg.generateLocal(outerThisField.getType());
269        
270        soot.jimple.FieldRef fieldRef = soot.jimple.Jimple.v().newInstanceFieldRef(specialThisLocal, outerThisField.makeRef());
271        soot.jimple.AssignStmt fieldAssignStmt = soot.jimple.Jimple.v().newAssignStmt(t1, fieldRef);
272        body.getUnits().add(fieldAssignStmt);
273         
274        if (fh.canStoreType(t1.getType(), sootType)){
275            getThisMap.put(sootType, t1);
276            return t1;            
277        }
278        
279        // check to see if this method has a local of the correct type (it will
280        // if its an initializer - then ust use it)
281        // here we need an exact type I think

283            soot.Local l = getLocalOfType(body, sootType);
284            getThisMap.put(sootType, l);
285            return l;
286        }*/
287        
288        // otherwise make a new access method
289        soot.Local t2 = t1;
290
291        return getThisGivenOuter(sootType, getThisMap, body, lg, t2);
292    }

294    private static soot.Local getLocalOfType(soot.Body body, soot.Type type) {
295        soot.FastHierarchy fh = InitialResolver.v().hierarchy();
296        Iterator stmtsIt = body.getUnits().iterator();
297        soot.Local correctLocal = null;
298        while (stmtsIt.hasNext()){
299            soot.jimple.Stmt s = (soot.jimple.Stmt)stmtsIt.next();
300            if (s instanceof soot.jimple.IdentityStmt && (s.hasTag("EnclosingTag") || s.hasTag("QualifyingTag"))){
301                Iterator it = s.getDefBoxes().iterator();
302                while (it.hasNext()){
303                    soot.ValueBox vb = (soot.ValueBox)it.next();
304                    if ((vb.getValue() instanceof soot.Local) && (fh.canStoreType(type, vb.getValue().getType()))){//(vb.getValue().getType().equals(type))){
305                        correctLocal = (soot.Local)vb.getValue();
306                    }
307                }
308            }
309        }
310        return correctLocal;
311    }

341    public static soot.Local getThisGivenOuter(soot.Type sootType, HashMap getThisMap, soot.Body body, LocalGenerator lg, soot.Local t2){
342       
343        if (InitialResolver.v().hierarchy() == null){
344            InitialResolver.v().hierarchy(new soot.FastHierarchy());
345        }
346        
347        soot.FastHierarchy fh = InitialResolver.v().hierarchy();
348        
349        while (!fh.canStoreType(t2.getType(),sootType)){
350            soot.SootClass classToInvoke = ((soot.RefType)t2.getType()).getSootClass();
351            // make an access method and add it to that class for accessing 
352            // its private this$0 field
353            soot.SootMethod methToInvoke = makeOuterThisAccessMethod(classToInvoke);
354            
355            // generate a local that corresponds to the invoke of that meth
356            soot.Local t3 = lg.generateLocal(methToInvoke.getReturnType());
357            ArrayList methParams = new ArrayList();
358            methParams.add(t2);
359            soot.Local res = getPrivateAccessFieldInvoke(methToInvoke.makeRef(), methParams, body, lg);
360            soot.jimple.AssignStmt assign = soot.jimple.Jimple.v().newAssignStmt(t3, res);
361            body.getUnits().add(assign);
362            t2 = t3;
363        }
364            
365        getThisMap.put(sootType, t2);
366
367        return t2;        
368    }

371    private static soot.SootMethod makeOuterThisAccessMethod(soot.SootClass classToInvoke){
372        String name = "access$"+soot.javaToJimple.InitialResolver.v().getNextPrivateAccessCounter()+"00";
373        ArrayList paramTypes = new ArrayList();
374        paramTypes.add(classToInvoke.getType());
375        
376        soot.SootMethod meth = new soot.SootMethod(name, paramTypes, classToInvoke.getFieldByName("this$0").getType(), soot.Modifier.STATIC);
377
378        classToInvoke.addMethod(meth);
379        PrivateFieldAccMethodSource src = new PrivateFieldAccMethodSource(
380            classToInvoke.getFieldByName("this$0").getType(),
381            "this$0",
382            classToInvoke.getFieldByName("this$0").isStatic(),
383            classToInvoke
384            );
385        meth.setActiveBody(src.getBody(meth, null));
386        meth.addTag(new soot.tagkit.SyntheticTag());
387        return meth;
388    }

390    public static soot.Local getPrivateAccessFieldInvoke(soot.SootMethodRef toInvoke, ArrayList params, soot.Body body, LocalGenerator lg){
391        soot.jimple.InvokeExpr invoke = soot.jimple.Jimple.v().newStaticInvokeExpr(toInvoke, params);
392
393        soot.Local retLocal = lg.generateLocal(toInvoke.returnType());
394
395        soot.jimple.AssignStmt stmt = soot.jimple.Jimple.v().newAssignStmt(retLocal, invoke);
396        body.getUnits().add(stmt);
397
398        return retLocal;
399    }

407    /**
408     * Type handling
409     */
410    public static soot.Type getSootType(polyglot.types.Type type) {
411
412        if (type == null){
413            throw new RuntimeException("Trying to get soot type for null polyglot type");
414        }
415                soot.Type sootType = null;
416        
417                if (type.isInt()){
418                        sootType = soot.IntType.v();
419                }
420                else if (type.isArray()){
421
422            polyglot.types.Type polyglotBase = ((polyglot.types.ArrayType)type).base();
423            while (polyglotBase instanceof polyglot.types.ArrayType) {
424              polyglotBase = ((polyglot.types.ArrayType)polyglotBase).base();
425            }
426                        soot.Type baseType = getSootType(polyglotBase);
427                        int dims = ((polyglot.types.ArrayType)type).dims();
428
429            // do something here if baseType is still an array
430                        sootType = soot.ArrayType.v(baseType, dims);
431                }
432                else if (type.isBoolean()){
433                        sootType = soot.BooleanType.v();
434                }
435                else if (type.isByte()){
436                        sootType = soot.ByteType.v();
437                }
438                else if (type.isChar()){
439                        sootType = soot.CharType.v();
440                }
441                else if (type.isDouble()){
442                        sootType = soot.DoubleType.v();
443                }
444                else if (type.isFloat()){
445                        sootType = soot.FloatType.v();
446                }
447                else if (type.isLong()){
448                        sootType = soot.LongType.v();
449                }
450                else if (type.isShort()){
451                        sootType = soot.ShortType.v();
452                }
453                else if (type.isNull()){
454                        sootType = soot.NullType.v();
455                }
456                else if (type.isVoid()){
457                        sootType = soot.VoidType.v();
458                }
459                else if (type.isClass()){
460            polyglot.types.ClassType classType = (polyglot.types.ClassType)type;
461            String className;
462            if (classType.isNested()) {
463                if (classType.isAnonymous() && (soot.javaToJimple.InitialResolver.v().getAnonTypeMap() != null) && soot.javaToJimple.InitialResolver.v().getAnonTypeMap().containsKey(new polyglot.util.IdentityKey(classType))){
464                    className = soot.javaToJimple.InitialResolver.v().getAnonTypeMap().get(new polyglot.util.IdentityKey(classType));   
465                }
466                else if (classType.isLocal() && (soot.javaToJimple.InitialResolver.v().getLocalTypeMap() != null) && soot.javaToJimple.InitialResolver.v().getLocalTypeMap().containsKey(new polyglot.util.IdentityKey(classType))) {
467                    className = soot.javaToJimple.InitialResolver.v().getLocalTypeMap().get(new polyglot.util.IdentityKey(classType));    
468                }
469                else {
470                    String pkgName = "";
471                    if (classType.package_() != null){
472                        pkgName = classType.package_().fullName();
473                    }
474                    className = classType.name();
475                    
476                    if (classType.outer().isAnonymous() || classType.outer().isLocal()){
477                        className = getSootType(classType.outer()).toString()+"$"+className;
478                    }
479                    else {
480                        while (classType.outer() != null){
481                            className = classType.outer().name()+"$"+className;
482                            classType = classType.outer();
483                        }
484
485                        if (!pkgName.equals("")){
486                            className = pkgName+"."+className;
487                        }
488                    }
489                }
490            }
491            else {
492                            className = classType.fullName();
493            }
494            
495                        sootType = soot.RefType.v(className);
496                }
497                else{
498                        throw new RuntimeException("Unknown Type");
499                }
500                return sootType;
501    }

504    /**
505     * Modifier Creation
506     */
507        public static int getModifier(polyglot.types.Flags flags) {
508
509                int modifier = 0;
510                
511                if (flags.isPublic()){
512                        modifier = modifier | soot.Modifier.PUBLIC;     
513                }
514                if (flags.isPrivate()){
515                        modifier = modifier | soot.Modifier.PRIVATE;
516                }
517                if (flags.isProtected()){
518                        modifier = modifier | soot.Modifier.PROTECTED;
519                }
520                if (flags.isFinal()){
521                        modifier = modifier | soot.Modifier.FINAL;
522                }
523                if (flags.isStatic()){
524                        modifier = modifier | soot.Modifier.STATIC;
525                }
526                if (flags.isNative()){
527                        modifier = modifier | soot.Modifier.NATIVE;
528                }
529                if (flags.isAbstract()){
530                        modifier = modifier | soot.Modifier.ABSTRACT;
531                }
532                if (flags.isVolatile()){
533                        modifier = modifier | soot.Modifier.VOLATILE;
534                }
535                if (flags.isTransient()){
536                        modifier = modifier | soot.Modifier.TRANSIENT;
537                }
538                if (flags.isSynchronized()){
539                        modifier = modifier | soot.Modifier.SYNCHRONIZED;
540                }
541                if (flags.isInterface()){
542                        modifier = modifier | soot.Modifier.INTERFACE;
543                }
544        if (flags.isStrictFP()) {
545            modifier = modifier | soot.Modifier.STRICTFP;
546        }
547                return modifier;
548        }
