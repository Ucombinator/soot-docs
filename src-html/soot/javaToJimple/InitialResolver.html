109    /**
110     * Invokes polyglot and gets the AST for the source given in fullPath
111     */
112    public void formAst(String fullPath, List<String> locations, String className){
113    
114        JavaToJimple jtj = new JavaToJimple();
115        polyglot.frontend.ExtensionInfo extInfo = jtj.initExtInfo(fullPath, locations);
116        // only have one compiler - for memory issues
117        if (compiler == null) {
118            compiler = new polyglot.frontend.Compiler(extInfo);
119        }
120        // build ast
121        astNode = jtj.compile(compiler, fullPath, extInfo);
122 
123        resolveAST();
124
125
126    }

144    private void makeASTMap() {
145        ClassDeclFinder finder = new ClassDeclFinder();
146        astNode.visit(finder);
147        Iterator<ClassDecl> it = finder.declsFound().iterator();
148        while (it.hasNext()){
149            polyglot.ast.ClassDecl decl = it.next();
150            polyglot.types.ClassType type = decl.type();
151            if (type.flags().isInterface()){
152                if (interfacesList ==  null){
153                    interfacesList = new ArrayList<String>();
154                }
155                interfacesList.add(Util.getSootType(type).toString());
156            }
157            addNameToAST(Util.getSootType(type).toString());
158        }
159    }

172    public void resolveAST(){
173        buildInnerClassInfo();
174        if (astNode instanceof polyglot.ast.SourceFile) {
175            createClassToSourceMap((polyglot.ast.SourceFile)astNode);
176        }
177    }

179    // resolves all types and deals with .class literals and asserts
180    public Dependencies resolveFromJavaFile(soot.SootClass sc) {
181        Dependencies dependencies = new Dependencies();
182        //conservatively load all to signatures
183        ClassResolver cr = new ClassResolver(sc, dependencies.typesToSignature);
184        
185        // create class to source map first 
186        // create source file
187        if (astNode instanceof polyglot.ast.SourceFile) {
188            cr.createSource((polyglot.ast.SourceFile)astNode);
189        }
190        
191        cr.addSourceFileTag(sc);
192        
193        makeASTMap();
194        
195        
196        return dependencies;
197    }

200    private void createClassToSourceMap(polyglot.ast.SourceFile src){
201       
202        String srcName = src.source().path();
203        String srcFileName = null;
204        if (src.package_() != null){
205            String slashedPkg = src.package_().package_().fullName().replaceAll(".", System.getProperty("file.separator"));
206            srcFileName = srcName.substring(srcName.lastIndexOf(slashedPkg));
207        }
208        else {
209            srcFileName = srcName.substring(srcName.lastIndexOf(System.getProperty("file.separator"))+1);
210        }
211
212        ArrayList list = new ArrayList();
213        Iterator it = src.decls().iterator();
214        while (it.hasNext()){
215            polyglot.ast.ClassDecl nextDecl = (polyglot.ast.ClassDecl)it.next();
216            addToClassToSourceMap(Util.getSootType(nextDecl.type()).toString(), srcFileName); 
217        }
218
219    }

221    private void createLocalAndAnonClassNames(ArrayList<Node> anonBodyList, ArrayList<Node> localClassDeclList){
222        Iterator<Node> anonBodyIt = anonBodyList.iterator();
223        while (anonBodyIt.hasNext()){
224            createAnonClassName((polyglot.ast.New)anonBodyIt.next());
225        }
226        Iterator<Node> localClassDeclIt = localClassDeclList.iterator();
227        while (localClassDeclIt.hasNext()){
228            createLocalClassName((polyglot.ast.LocalClassDecl)localClassDeclIt.next());
229        }
230    }

237    private void createAnonClassName(polyglot.ast.New nextNew){
238        // maybe this anon has already been resolved
239        if (anonClassMap == null){
240            anonClassMap = new BiMap();
241        }
242        if (anonTypeMap == null){
243            anonTypeMap = new HashMap<IdentityKey, String>();
244        }
245        if (!anonClassMap.containsKey(nextNew)){
246            int nextAvailNum = 1;
247            polyglot.types.ClassType outerToMatch = nextNew.anonType().outer();
248            while (outerToMatch.isNested()){
249                outerToMatch = outerToMatch.outer();
250            }
251
252            if (!anonTypeMap.isEmpty()){
253                Iterator<IdentityKey> matchIt = anonTypeMap.keySet().iterator();
254                while (matchIt.hasNext()){
255                    polyglot.types.ClassType pType = (polyglot.types.ClassType)matchIt.next().object();
256                    polyglot.types.ClassType outerMatch = pType.outer();
257                    while (outerMatch.isNested()){
258                        outerMatch = outerMatch.outer();
259                    }
260                    if (outerMatch.equals(outerToMatch)){
261                        int numFound = getAnonClassNum(anonTypeMap.get(new polyglot.util.IdentityKey(pType)));
262                        if (numFound >= nextAvailNum){
263                            nextAvailNum = numFound+1;
264                        }
265                    }
266                }
267            }
268            
269            String realName = outerToMatch.fullName()+"$"+nextAvailNum;
270            anonClassMap.put(nextNew, realName);
271            anonTypeMap.put(new polyglot.util.IdentityKey(nextNew.anonType()), realName);
272            addNameToAST(realName);
273            
274        }
275    }

277    private void createLocalClassName(polyglot.ast.LocalClassDecl lcd){
278        // maybe this localdecl has already been resolved 
279        if (localClassMap == null){
280            localClassMap = new BiMap();
281        }
282        if (localTypeMap == null){
283            localTypeMap = new HashMap<IdentityKey, String>();
284        }
285        
286        if (!localClassMap.containsKey(lcd)){
287            int nextAvailNum = 1;
288            polyglot.types.ClassType outerToMatch = lcd.decl().type().outer();
289            while (outerToMatch.isNested()){
290                outerToMatch = outerToMatch.outer();
291            }
292
293            if (!localTypeMap.isEmpty()){
294                Iterator<IdentityKey> matchIt = localTypeMap.keySet().iterator();
295                while (matchIt.hasNext()){
296                    polyglot.types.ClassType pType = (polyglot.types.ClassType)matchIt.next().object();
297                    polyglot.types.ClassType outerMatch = pType.outer();
298                    while (outerMatch.isNested()){
299                        outerMatch = outerMatch.outer();
300                    }
301                    if (outerMatch.equals(outerToMatch)){
302                        int numFound = getLocalClassNum(localTypeMap.get(new polyglot.util.IdentityKey(pType)), lcd.decl().name());
303                        if (numFound >= nextAvailNum){
304                            nextAvailNum = numFound+1;
305                        }
306                    }
307                }
308            }
309
310            String realName = outerToMatch.fullName()+"$"+nextAvailNum+lcd.decl().name();
311            localClassMap.put(lcd, realName);
312            localTypeMap.put(new polyglot.util.IdentityKey(lcd.decl().type()), realName);
313            addNameToAST(realName);
314        }
315    }

317    private static final int NO_MATCH = 0;
318    
319    private int getLocalClassNum(String realName, String simpleName){
320        // a local inner class is named outer$NsimpleName where outer 
321        // is the very outer most class
322        int dIndex = realName.indexOf("$");
323        int nIndex = realName.indexOf(simpleName, dIndex);
324        if (nIndex == -1) return NO_MATCH;
325        if (dIndex == -1) {
326            throw new RuntimeException("Matching an incorrectly named local inner class: "+realName);
327        }
328        String numString = realName.substring(dIndex+1, nIndex);
329        for (int i = 0; i < numString.length(); i++){
330            if (!Character.isDigit(numString.charAt(i))) return NO_MATCH;
331        }
332        return (new Integer(numString)).intValue();
333    }
334    
335    private int getAnonClassNum(String realName){
336        // a anon inner class is named outer$N where outer 
337        // is the very outer most class
338        int dIndex = realName.indexOf("$");
339        if (dIndex == -1) {
340            throw new RuntimeException("Matching an incorrectly named anon inner class: "+realName);
341        }
342        return (new Integer(realName.substring(dIndex+1))).intValue();
343    }

359    public boolean hasClassInnerTag(soot.SootClass sc, String innerName){
360        Iterator it = sc.getTags().iterator();
361        while (it.hasNext()){
362            soot.tagkit.Tag t = (soot.tagkit.Tag)it.next();
363            if (t instanceof soot.tagkit.InnerClassTag) {
364                soot.tagkit.InnerClassTag tag = (soot.tagkit.InnerClassTag)t;
365                if (tag.getInnerClass().equals(innerName)) return true;
366            }
367        }
368        return false;
369    }

378    private void buildFinalLocalMap(ArrayList<Node> memberList){
379        Iterator<Node> it = memberList.iterator();
380        while (it.hasNext()){
381            handleFinalLocals((polyglot.ast.ClassMember)it.next());
382        }

385    private void handleFinalLocals(polyglot.ast.ClassMember member){
386        MethodFinalsChecker mfc = new MethodFinalsChecker();
387        member.visit(mfc);
388        //System.out.println("member: "+member);
389        //System.out.println("mcf final locals avail: "+mfc.finalLocals());
390        //System.out.println("mcf locals used: "+mfc.typeToLocalsUsed());
391        //System.out.println("mfc inners: "+mfc.inners());
392        if (cCallList == null){
393            cCallList = new ArrayList<Node>();
394        }
395        cCallList.addAll(mfc.ccallList());
396        //System.out.println("cCallList: "+cCallList);
397        AnonLocalClassInfo alci = new AnonLocalClassInfo();
398        if (member instanceof polyglot.ast.ProcedureDecl){
399            polyglot.ast.ProcedureDecl procedure = (polyglot.ast.ProcedureDecl)member;
400            // not sure if this will break deep nesting
401            alci.finalLocalsAvail(mfc.finalLocals());
402            if (procedure.flags().isStatic()){
403                alci.inStaticMethod(true);
404            }
405        }
406        else if (member instanceof polyglot.ast.FieldDecl){
407            alci.finalLocalsAvail(new ArrayList<IdentityKey>());
408            if (((polyglot.ast.FieldDecl)member).flags().isStatic()){
409                alci.inStaticMethod(true);
410            }
411        }
412        else if (member instanceof polyglot.ast.Initializer){
413            // for now don't make final locals avail in init blocks
414            // need to test this
415            alci.finalLocalsAvail(mfc.finalLocals());
416            if (((polyglot.ast.Initializer)member).flags().isStatic()){
417                alci.inStaticMethod(true);
418            }
419        }
420        if (finalLocalInfo == null){
421            finalLocalInfo = new HashMap<IdentityKey, AnonLocalClassInfo>();
422        }
423        Iterator<IdentityKey> it = mfc.inners().iterator();
424        while (it.hasNext()){
425            
426            polyglot.types.ClassType cType = (polyglot.types.ClassType)it.next().object();
427            // do the comparison about locals avail and locals used here
428            HashMap<IdentityKey, ArrayList<IdentityKey>> typeToLocalUsed = mfc.typeToLocalsUsed();
429            ArrayList<IdentityKey> localsUsed = new ArrayList<IdentityKey>();
430            if (typeToLocalUsed.containsKey(new polyglot.util.IdentityKey(cType))){
431                ArrayList localsNeeded = typeToLocalUsed.get(new polyglot.util.IdentityKey(cType));
432                Iterator usesIt = localsNeeded.iterator();
433                while (usesIt.hasNext()){
434                    polyglot.types.LocalInstance li = (polyglot.types.LocalInstance)((polyglot.util.IdentityKey)usesIt.next()).object();
435                    if (alci.finalLocalsAvail().contains(new polyglot.util.IdentityKey(li))){
436                        localsUsed.add(new polyglot.util.IdentityKey(li));
437                    }
438                }
439            }
440                
441            
442            AnonLocalClassInfo info = new AnonLocalClassInfo();
443            info.inStaticMethod(alci.inStaticMethod());
444            info.finalLocalsAvail(localsUsed);
445            if (!finalLocalInfo.containsKey(new polyglot.util.IdentityKey(cType))){
446                finalLocalInfo.put(new polyglot.util.IdentityKey(cType), info);
447            }
448        }
449    }

451    public boolean isAnonInCCall(polyglot.types.ClassType anonType){
452        //System.out.println("checking type: "+anonType);
453        Iterator<Node> it = cCallList.iterator();
454        while (it.hasNext()){
455            polyglot.ast.ConstructorCall cCall = (polyglot.ast.ConstructorCall)it.next();
456            //System.out.println("cCall params: "+cCall.arguments());
457            Iterator argsIt = cCall.arguments().iterator();
458            while (argsIt.hasNext()){
459                Object next = argsIt.next();
460                if (next instanceof polyglot.ast.New && ((polyglot.ast.New)next).anonType() != null){
461                    //System.out.println("comparing: "+((polyglot.ast.New)next).anonType());
462                    if (((polyglot.ast.New)next).anonType().equals(anonType)) return true;
463                }
464            }
465        }
466        return false;
467    }
