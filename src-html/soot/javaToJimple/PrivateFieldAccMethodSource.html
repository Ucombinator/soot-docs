020package soot.javaToJimple;
021
022import java.util.*;
023public class PrivateFieldAccMethodSource implements soot.MethodSource {
024
025    private final soot.Type fieldType;
026    private final String fieldName;
027    private final boolean isStatic;
028    private final soot.SootClass classToInvoke;
029    
030    public PrivateFieldAccMethodSource(soot.Type fieldType, String fieldName, boolean isStatic, soot.SootClass classToInvoke ) {
031        this.fieldType = fieldType;
032        this.fieldName = fieldName;
033        this.isStatic = isStatic;
034        this.classToInvoke = classToInvoke;
035    }
036    
037 
038    public soot.Body getBody(soot.SootMethod sootMethod, String phaseName){
039        
040        soot.Body body = soot.jimple.Jimple.v().newBody(sootMethod);
041        LocalGenerator lg = new LocalGenerator(body);
042        
043        soot.Local fieldBase = null;
044        // create parameters
045        Iterator paramIt = sootMethod.getParameterTypes().iterator();
046        while (paramIt.hasNext()) {
047            soot.Type sootType = (soot.Type)paramIt.next();
048            soot.Local paramLocal = lg.generateLocal(sootType);
049            
050            soot.jimple.ParameterRef paramRef = soot.jimple.Jimple.v().newParameterRef(sootType, 0);
051            soot.jimple.Stmt stmt = soot.jimple.Jimple.v().newIdentityStmt(paramLocal, paramRef);
052            body.getUnits().add(stmt);
053            fieldBase = paramLocal;
054        }
055        
056        // create field type local
057        soot.Local fieldLocal = lg.generateLocal(fieldType);
058        // assign local to fieldRef
059        soot.SootFieldRef field = soot.Scene.v().makeFieldRef( classToInvoke, fieldName, fieldType, isStatic);
060
061        soot.jimple.FieldRef fieldRef = null;
062        if (isStatic) {
063            fieldRef = soot.jimple.Jimple.v().newStaticFieldRef(field);
064        }
065        else {
066            fieldRef = soot.jimple.Jimple.v().newInstanceFieldRef(fieldBase, field);
067        }
068        soot.jimple.AssignStmt assign = soot.jimple.Jimple.v().newAssignStmt(fieldLocal, fieldRef);
069        body.getUnits().add(assign);
070
071        //return local
072        soot.jimple.Stmt retStmt = soot.jimple.Jimple.v().newReturnStmt(fieldLocal);
073        body.getUnits().add(retStmt);
074        
075        return body;
076     
077    }
078}
