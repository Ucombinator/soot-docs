354    private final Map<String, Pack> packNameToPack = new HashMap<String, Pack>();
355    private final List<Pack> packList = new LinkedList<Pack>();
356
357    private void addPack( Pack p ) {
358        if( packNameToPack.containsKey( p.getPhaseName() ) )
359            throw new RuntimeException( "Duplicate pack "+p.getPhaseName() );
360        packNameToPack.put( p.getPhaseName(), p );
361        packList.add( p );
362    }

377    public HasPhaseOptions getPhase(String phaseName) {
378        int index = phaseName.indexOf( "." );
379        if( index < 0 ) return getPack( phaseName );
380        String packName = phaseName.substring(0,index);
381        if( !hasPack( packName ) ) return null;
382        return getPack( packName ).get( phaseName );
383    }

385    public Transform getTransform(String phaseName) {
386        return (Transform) getPhase( phaseName );
387    }

402        private void runPacksForOneClassAtATime() {
403                if (Options.v().src_prec() == Options.src_prec_class && Options.v().keep_line_number()){
404            LineNumberAdder lineNumAdder = LineNumberAdder.v();
405            lineNumAdder.internalTransform("", null);
406        }
407
408                setupJAR();
409        for( String path: (Collection<String>)Options.v().process_dir()) {
410            // hack1: resolve to signatures only
411            for (String cl : SourceLocator.v().getClassesUnder(path)) {
412                SootClass clazz = Scene.v().forceResolve(cl, SootClass.SIGNATURES);
413                clazz.setApplicationClass();
414            }
415            // hack2: for each class one after another:
416            //     a) resolve to bodies
417            //     b) run packs
418            //     c) write class
419            //     d) remove bodies
420            for (String cl : SourceLocator.v().getClassesUnder(path)) {
421                SootClass clazz = null;
422                ClassSource source = SourceLocator.v().getClassSource(cl);
423                try {
424                        if (source == null)
425                                throw new RuntimeException("Could not locate class source");
426                        clazz = Scene.v().getSootClass(cl);
427                        clazz.setResolvingLevel(SootClass.BODIES);
428                        source.resolve(clazz);
429                }
430                finally {
431                        if (source != null)
432                                source.close();
433                }
434                
435                // Create tags from all values we only have in code assingments now
436                for (SootClass sc : Scene.v().getApplicationClasses()) {
437                    if( Options.v().validate() )
438                        sc.validate();
439                        if (!sc.isPhantom)
440                                ConstantInitializerToTagTransformer.v().transformClass(sc, true);
441                }
442                
443                                runBodyPacks(clazz);
444                                //generate output
445                                writeClass(clazz);
446
447                                if (!Options.v().no_writeout_body_releasing())
448                                        releaseBodies(clazz);
449            }
450
451//            for (String cl : SourceLocator.v().getClassesUnder(path)) {
452//                SootClass clazz = Scene.v().forceResolve(cl, SootClass.BODIES);
453//                              releaseBodies(clazz);
454//                              Scene.v().removeClass(clazz);
455//            }
456        }
457        tearDownJAR();
458
459        handleInnerClasses();
460    }

462        private void runPacksNormally() {
463                if (Options.v().src_prec() == Options.src_prec_class && Options.v().keep_line_number()){
464            LineNumberAdder lineNumAdder = LineNumberAdder.v();
465            lineNumAdder.internalTransform("", null);
466        }
467                
468        if (Options.v().whole_program() || Options.v().whole_shimple()) {
469            runWholeProgramPacks();
470        }
471        retrieveAllBodies();
472        
473        // Create tags from all values we only have in code assignments now
474        for (SootClass sc : Scene.v().getApplicationClasses()) {
475            if( Options.v().validate() )
476                sc.validate();
477                if (!sc.isPhantom)
478                        ConstantInitializerToTagTransformer.v().transformClass(sc, true);
479        }
480        
481        // if running coffi cfg metrics, print out results and exit
482        if (soot.jbco.Main.metrics) {
483          coffiMetrics();
484          System.exit(0);
485        }
486
487        preProcessDAVA();
488        if (Options.v().interactive_mode()){
489            if (InteractionHandler.v().getInteractionListener() == null){
490                G.v().out.println("Cannot run in interactive mode. No listeners available. Continuing in regular mode.");
491                Options.v().set_interactive_mode(false);
492            }
493            else {
494                G.v().out.println("Running in interactive mode.");
495            }
496        }
497        runBodyPacks();
498        handleInnerClasses();
499        }

501    public void coffiMetrics() {
502      int tV = 0, tE = 0, hM = 0;
503      double aM = 0;
504      HashMap<SootMethod, int[]> hashVem = soot.coffi.CFG.methodsToVEM;
505      Iterator<SootMethod> it = hashVem.keySet().iterator();
506      while (it.hasNext()) {
507        int vem[] = hashVem.get(it.next());
508        tV+= vem[0];
509        tE+= vem[1];
510        aM+= vem[2];
511        if (vem[2]>hM) hM = vem[2];
512      }
513      if (hashVem.size()>0)
514        aM/=hashVem.size();
515
516      G.v().out.println("Vertices, Edges, Avg Degree, Highest Deg:    "+tV+"  "+tE+"  "+aM+"  "+hM);
517    }

529    public void writeOutput() {
530        setupJAR();
531        if(Options.v().verbose())
532            PhaseDumper.v().dumpBefore("output");
533        if( Options.v().output_format() == Options.output_format_dava ) {
534            postProcessDAVA();
535        }
536        else if (Options.v().output_format() == Options.output_format_dex
537                        || Options.v().output_format() == Options.output_format_force_dex) {
538                dexPrinter = new DexPrinter();
539                writeOutput(reachableClasses());
540                dexPrinter.print();
541                dexPrinter = null;
542        } else {
543            writeOutput( reachableClasses() );
544            tearDownJAR();
545        }
546        postProcessXML( reachableClasses() );
547
548                if (!Options.v().no_writeout_body_releasing())
549                        releaseBodies( reachableClasses() );
550        if(Options.v().verbose())
551            PhaseDumper.v().dumpAfter("output");
552    }

557        if (Options.v().output_jar()) {

564        } else {
565            jarFile = null;

592                if(!isSourceJavac){

594                         * It turns out that the exception attributes of a method i.e. those exceptions that
595                         * a method can throw are only checked by the Java compiler and not the JVM
596                         *
597                         * Javac does place this information into the attributes but other compilers dont
598                         * hence if the source is not javac then we have to do this fancy analysis

603                         * See ThrowFinder for more details
604                         */

606                                System.out.println("Source is not Javac hence invoking ThrowFinder");
607

609                }
610                else{
611                        if(DEBUG)
612                                System.out.println("Source is javac hence we dont need to invoke ThrowFinder");
613                }

615            PackageNamer.v().fixNames();
616
617            G.v().out.println();
618        }

621    private void runBodyPacks( final Iterator<SootClass> classes ) {
622        int threadNum = Runtime.getRuntime().availableProcessors();
623        CountingThreadPoolExecutor executor =  new CountingThreadPoolExecutor(threadNum,
624                        threadNum, 30, TimeUnit.SECONDS,
625                        new LinkedBlockingQueue<Runnable>());
626        
627        while( classes.hasNext() ) {
628                final SootClass c = classes.next();
629                executor.execute(new Runnable() {
630                                
631                                @Override
632                                public void run() {
633                                        runBodyPacks(c);
634                                }
635                                
636                });
637        }
638        
639        // Wait till all packs have been executed
640        try {
641                executor.awaitCompletion();
642                        executor.shutdown();
643                } catch (InterruptedException e) {
644                        // Something went horribly wrong
645                        throw new RuntimeException("Could not wait for pack threads to "
646                                        + "finish: " + e.getMessage(), e);
647                }
648        
649        // If something went wrong, we tell the world
650        if (executor.getException() != null)
651                throw (RuntimeException) executor.getException();
652    }

664                        && jarFile == null ? Runtime.getRuntime().availableProcessors() : 1;
665        CountingThreadPoolExecutor executor =  new CountingThreadPoolExecutor(threadNum,

667                        new LinkedBlockingQueue<Runnable>());
668        
669        while( classes.hasNext() ) {
670                final SootClass c = classes.next();

672                                
673                                @Override
674                                public void run() {
675                                        writeClass( c );

681        // Wait till all classes have been written
682        try {
683                executor.awaitCompletion();
684                        executor.shutdown();

686                        // Something went horribly wrong
687                        throw new RuntimeException("Could not wait for writer threads to "
688                                        + "finish: " + e.getMessage(), e);

692        if (executor.getException() != null)
693                throw (RuntimeException) executor.getException();

738                        RemoveEmptyBodyDefaultConstructor.checkAndRemoveDefault(s);
739
740
741
742                /*
743                 * Nomair A. Naeem 1st March 2006
744                 * Check if we want to apply transformations

746                 */
747
748                //debug("analyzeAST","Advanced Analyses ALL DISABLED");
749
750                G.v().out.println("Analyzing " + fileName + "... ");
751
752                /*
753                 * Nomair A. Naeem 29th Jan 2006
754                 * Added hook into going through each decompiled method again
755                 * Need it for all the implemented AST analyses
756                 */
757                for (SootMethod m : s.getMethods()) {
758                        /*
759                         * 3rd April 2006
760                         * Fixing RuntimeException caused when you
761                         * retrieve an active body when one is not present
762                         *
763                         */
764                        if(m.hasActiveBody()){
765                                DavaBody body = (DavaBody)m.getActiveBody();
766                                //System.out.println("body"+body.toString());

771                                body.applyBugFixes();
772                        }
773                        }
774                        else
775                                continue;
776                }
777
778        } //going through all classes
779
780        /*
781         * Nomair A. Naeem March 6th, 2006
782         *
783         * SHOULD BE INVOKED ONLY ONCE!!!
784         * If interprocedural analyses are turned off they are checked within this
785         * method.
786         *
787         * HAVE TO invoke this analysis since this invokes the renamer!!
788         */

790                InterProceduralAnalyses.applyInterProceduralAnalyses();

798
799
800         /*
801          * Generate decompiled code
802          */
803        String pathForBuild=null;

829                    streamOut = new GZIPOutputStream(streamOut);

831                    new PrintWriter(new OutputStreamWriter(streamOut));

833                throw new CompilationDeathException("Cannot output file " + fileName,e);

882    @SuppressWarnings("fallthrough")
883    private void runBodyPacks(SootClass c) {
884        final int format = Options.v().output_format();
885        if (format == Options.output_format_dava) {
886            G.v().out.print("Decompiling ");
887
888             //January 13th, 2006  SootMethodAddedByDava is set to false for SuperFirstStmtHandler
889            G.v().SootMethodAddedByDava=false;
890        } else {
891            G.v().out.print("Transforming ");
892        }
893        G.v().out.println(c.getName() + "... ");
894
895        boolean produceBaf = false, produceGrimp = false, produceDava = false,
896            produceJimple = true, produceShimple = false;
897
898        switch (format) {
899            case Options.output_format_none :
900            case Options.output_format_xml :
901            case Options.output_format_jimple :
902            case Options.output_format_jimp :
903            case Options.output_format_template :
904            case Options.output_format_dex :
905            case Options.output_format_force_dex :
906                break;
907            case Options.output_format_shimp:
908            case Options.output_format_shimple:
909                produceShimple = true;
910                // FLIP produceJimple
911                produceJimple = false;
912                break;
913            case Options.output_format_dava :
914                produceDava = true;
915                // FALL THROUGH
916            case Options.output_format_grimp :
917            case Options.output_format_grimple :
918                produceGrimp = true;
919                break;
920            case Options.output_format_baf :
921            case Options.output_format_b :
922                produceBaf = true;
923                break;
924            case Options.output_format_jasmin :
925            case Options.output_format_class :
926            case Options.output_format_asm :
927                produceGrimp = Options.v().via_grimp();
928                produceBaf = !produceGrimp;
929                break;
930            default :
931                throw new RuntimeException();
932        }
933
934        soot.xml.TagCollector tc = new soot.xml.TagCollector();
935
936        boolean wholeShimple = Options.v().whole_shimple();
937        if( Options.v().via_shimple() ) produceShimple = true;
938
939        //here we create a copy of the methods so that transformers are able
940        //to add method bodies during the following iteration;
941        //such adding of methods happens in rare occasions: for instance when
942        //resolving a method reference to a non-existing method, then this
943        //method is created as a phantom method when phantom-refs are enabled
944        LinkedList<SootMethod> methodsCopy = new LinkedList<SootMethod>(c.getMethods());
945        for (SootMethod m : methodsCopy) {
946            if(DEBUG){
947                if(m.getExceptions().size()!=0)
948                        System.out.println("PackManager printing out jimple body exceptions for method "+m.toString()+" " + m.getExceptions().toString());
949            }
950
951            if (!m.isConcrete()) continue;
952
953            if (produceShimple || wholeShimple) {
954                ShimpleBody sBody = null;
955
956                // whole shimple or not?
957                {
958                    Body body = m.retrieveActiveBody();
959
960                    if(body instanceof ShimpleBody){
961                        sBody = (ShimpleBody) body;
962                        if(!sBody.isSSA())
963                            sBody.rebuild();
964                    }
965                    else{
966                        sBody = Shimple.v().newBody(body);
967                    }
968                }
969
970                m.setActiveBody(sBody);
971                PackManager.v().getPack("stp").apply(sBody);
972                PackManager.v().getPack("sop").apply(sBody);
973
974                if( produceJimple || (wholeShimple && !produceShimple) )
975                    m.setActiveBody(sBody.toJimpleBody());
976            }
977
978            if (produceJimple) {
979                Body body = m.retrieveActiveBody();
980                //Change
981                CopyPropagator.v().transform(body);
982                ConditionalBranchFolder.v().transform(body);
983                UnreachableCodeEliminator.v().transform(body);
984                DeadAssignmentEliminator.v().transform(body);
985                UnusedLocalEliminator.v().transform(body);
986                PackManager.v().getPack("jtp").apply(body);
987                if( Options.v().validate() ) {
988                    body.validate();
989                }
990                PackManager.v().getPack("jop").apply(body);
991                PackManager.v().getPack("jap").apply(body);
992                if (Options.v().xml_attributes() && Options.v().output_format() != Options.output_format_jimple) {
993                    //System.out.println("collecting body tags");
994                    tc.collectBodyTags(body);
995                }
996            }
997
998            //PackManager.v().getPack("cfg").apply(m.retrieveActiveBody());
999
1000            if (produceGrimp) {
1001                m.setActiveBody(Grimp.v().newBody(m.getActiveBody(), "gb"));
1002                PackManager.v().getPack("gop").apply(m.getActiveBody());
1003            } else if (produceBaf) {
1004                        m.setActiveBody(convertJimpleBodyToBaf(m));
1005            }
1006        }
1007
1008        if (Options.v().xml_attributes() && Options.v().output_format() != Options.output_format_jimple) {
1009            processXMLForClass(c, tc);
1010            //System.out.println("processed xml for class");
1011        }
1012
1013        if (produceDava) {
1014            for (SootMethod m : c.getMethods()) {
1015                if (!m.isConcrete())
1016                        continue;
1017                //all the work done in decompilation is done in DavaBody which is invoked from within newBody
1018                m.setActiveBody(Dava.v().newBody(m.getActiveBody()));
1019            }
1020
1021            /*
1022             * January 13th, 2006
1023             * SuperFirstStmtHandler might have set SootMethodAddedByDava if it needs to create a new
1024             * method.
1025             */
1026            //could use G to add new method...................
1027            if(G.v().SootMethodAddedByDava){
1028                //System.out.println("PACKMANAGER SAYS:----------------Have to add the new method(s)");
1029                ArrayList<SootMethod> sootMethodsAdded = G.v().SootMethodsAdded;
1030                Iterator<SootMethod> it = sootMethodsAdded.iterator();
1031                while(it.hasNext()){
1032                        c.addMethod((SootMethod)it.next());
1033                }
1034                G.v().SootMethodsAdded = new ArrayList<SootMethod>();
1035                G.v().SootMethodAddedByDava=false;
1036            }
1037
1038        }//end if produceDava
1039    }

1057    private void writeClass(SootClass c) {
1058        // Create code assignments for those values we only have in code assignments
1059        if (Options.v().output_format() == Options.output_format_jimple)
1060                if (!c.isPhantom)
1061                        ConstantValueToInitializerTransformer.v().transformClass(c);
1062        
1063        final int format = Options.v().output_format();
1064        if( format == Options.output_format_none ) return;
1065        if( format == Options.output_format_dava ) return;
1066        if (format == Options.output_format_dex
1067                        || format == Options.output_format_force_dex) {
1068                // just add the class to the dex printer, writing is done after adding all classes
1069                dexPrinter.add(c);
1070                return;
1071        }
1072
1073        OutputStream streamOut = null;
1074        PrintWriter writerOut = null;
1075
1076        String fileName = SourceLocator.v().getFileNameFor(c, format);
1077        if( Options.v().gzip() ) fileName = fileName+".gz";
1078
1079        try {
1080            if( jarFile != null ) {
1081                // Fix path delimiters according to ZIP specification
1082                fileName = fileName.replace("\\", "/");
1083                JarEntry entry = new JarEntry(fileName);
1084                entry.setMethod(ZipEntry.DEFLATED);
1085                jarFile.putNextEntry(entry);
1086                streamOut = jarFile;
1087            } else {
1088                new File(fileName).getParentFile().mkdirs();
1089                streamOut = new FileOutputStream(fileName);
1090            }
1091            if( Options.v().gzip() ) {
1092                streamOut = new GZIPOutputStream(streamOut);
1093            }
1094            if(format == Options.output_format_class) {
1095                if(!Options.v().asm_backend()){
1096                        streamOut = new JasminOutputStream(streamOut);
1097                }
1098            }
1099            writerOut = new PrintWriter(new OutputStreamWriter(streamOut));
1100            G.v().out.println( "Writing to "+fileName );
1101        } catch (IOException e) {
1102            throw new CompilationDeathException("Cannot output file " + fileName,e);
1103        }
1104
1105        if (Options.v().xml_attributes()) {
1106            Printer.v().setOption(Printer.ADD_JIMPLE_LN);
1107        }
1108        
1109        int java_version = Options.v().java_version();
1110        
1111        switch (format) {
1112            case Options.output_format_class :
1113                if(Options.v().asm_backend()){
1114                        new BafASMBackend(c, java_version).generateClassFile(streamOut);
1115                        break;
1116                }
1117            case Options.output_format_jasmin :
1118                if (c.containsBafBody())
1119                    new soot.baf.JasminClass(c).print(writerOut);
1120                else
1121                    new soot.jimple.JasminClass(c).print(writerOut);
1122                break;
1123            case Options.output_format_jimp :
1124            case Options.output_format_shimp :
1125            case Options.output_format_b :
1126            case Options.output_format_grimp :
1127                Printer.v().setOption(Printer.USE_ABBREVIATIONS);
1128                Printer.v().printTo(c, writerOut);
1129                break;
1130            case Options.output_format_baf :
1131            case Options.output_format_jimple :
1132            case Options.output_format_shimple :
1133            case Options.output_format_grimple :
1134                writerOut =
1135                    new PrintWriter(
1136                        new EscapedWriter(new OutputStreamWriter(streamOut)));
1137                Printer.v().printTo(c, writerOut);
1138                break;
1139            case Options.output_format_xml :
1140                writerOut =
1141                    new PrintWriter(
1142                        new EscapedWriter(new OutputStreamWriter(streamOut)));
1143                XMLPrinter.v().printJimpleStyleTo(c, writerOut);
1144                break;
1145            case Options.output_format_template :
1146                writerOut =
1147                    new PrintWriter(
1148                        new OutputStreamWriter(streamOut));
1149                TemplatePrinter.v().printTo(c, writerOut);
1150                break;
1151            case Options.output_format_asm :
1152                new BafASMBackend(c, java_version).generateTextualRepresentation(writerOut);
1153                break;
1154            default :
1155                throw new RuntimeException();
1156        }
1157
1158        try {
1159            writerOut.flush();
1160            if( jarFile == null ) {
1161                    streamOut.close();
1162                    writerOut.close();
1163            }
1164            else
1165                jarFile.closeEntry();
1166        } catch (IOException e) {
1167            throw new CompilationDeathException("Cannot close output file " + fileName);
1168        }
1169    }

1210    private void retrieveAllBodies() {
1211        // The old coffi front-end is not thread-safe
1212        int threadNum = Options.v().coffi() ? 1 : Runtime.getRuntime().availableProcessors();
1213        CountingThreadPoolExecutor executor =  new CountingThreadPoolExecutor(threadNum,
1214                        threadNum, 30, TimeUnit.SECONDS,
1215                        new LinkedBlockingQueue<Runnable>());
1216        
1217        Iterator<SootClass> clIt = reachableClasses();
1218        while( clIt.hasNext() ) {
1219            SootClass cl = (SootClass) clIt.next();
1220            //note: the following is a snapshot iterator;
1221            //this is necessary because it can happen that phantom methods
1222            //are added during resolution
1223            Iterator<SootMethod> methodIt = cl.getMethods().iterator();
1224            while (methodIt.hasNext()) {
1225                final SootMethod m = methodIt.next();
1226                if( m.isConcrete() ) {
1227                        executor.execute(new Runnable() {
1228                                                
1229                                                @Override
1230                                                public void run() {
1231                                    m.retrieveActiveBody();
1232                                                }
1233                                                
1234                                        });
1235                }
1236            }
1237        }
1238        
1239        // Wait till all method bodies have been loaded
1240        try {
1241                executor.awaitCompletion();
1242                        executor.shutdown();
1243                } catch (InterruptedException e) {
1244                        // Something went horribly wrong
1245                        throw new RuntimeException("Could not wait for loader threads to "
1246                                        + "finish: " + e.getMessage(), e);
1247                }
1248        
1249        // If something went wrong, we tell the world
1250        if (executor.getException() != null)
1251                throw (RuntimeException) executor.getException();
1252    }
